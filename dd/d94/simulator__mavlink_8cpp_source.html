80 void Simulator::pack_actuator_message(mavlink_hil_actuator_controls_t &msg, unsigned index)
81 {
82  msg.time_usec = hrt_absolute_time();
83 
84  bool armed = (_vehicle_status.arming_state == vehicle_status_s::ARMING_STATE_ARMED);
85 
86  const float pwm_center = (PWM_DEFAULT_MAX + PWM_DEFAULT_MIN) / 2;
87 
88  /* scale outputs depending on system type */
89  if (_system_type == MAV_TYPE_QUADROTOR ||
90  _system_type == MAV_TYPE_HEXAROTOR ||
91  _system_type == MAV_TYPE_OCTOROTOR ||
92  _system_type == MAV_TYPE_VTOL_DUOROTOR ||
93  _system_type == MAV_TYPE_VTOL_QUADROTOR ||
94  _system_type == MAV_TYPE_VTOL_TILTROTOR ||
95  _system_type == MAV_TYPE_VTOL_RESERVED2) {
96 
97  /* multirotors: set number of rotor outputs depending on type */
98 
99  unsigned n;
100 
101  switch (_system_type) {
102  case MAV_TYPE_VTOL_DUOROTOR:
103  n = 2;
104  break;
105 
106  case MAV_TYPE_QUADROTOR:
107  case MAV_TYPE_VTOL_QUADROTOR:
108  case MAV_TYPE_VTOL_TILTROTOR:
109  n = 4;
110  break;
111 
112  case MAV_TYPE_VTOL_RESERVED2:
113  // this is the standard VTOL / quad plane with 5 propellers
114  n = 5;
115  break;
116 
117  case MAV_TYPE_HEXAROTOR:
118  n = 6;
119  break;
120 
121  default:
122  n = 8;
123  break;
124  }
125 
126  for (unsigned i = 0; i < 16; i++) {
127  if (_actuators[index].output[i] > PWM_DEFAULT_MIN / 2) {
128  if (i < n) {
129  /* scale PWM out PWM_DEFAULT_MIN..PWM_DEFAULT_MAX us to 0..1 for rotors */
130  msg.controls[i] = (_actuators[index].output[i] - PWM_DEFAULT_MIN) / (PWM_DEFAULT_MAX - PWM_DEFAULT_MIN);
131 
132  } else {
133  /* scale PWM out PWM_DEFAULT_MIN..PWM_DEFAULT_MAX us to -1..1 for other channels */
134  msg.controls[i] = (_actuators[index].output[i] - pwm_center) / ((PWM_DEFAULT_MAX - PWM_DEFAULT_MIN) / 2);
135  }
136 
137  } else {
138  /* send 0 when disarmed and for disabled channels */
139  msg.controls[i] = 0.0f;
140  }
141  }
142 
143  } else {
144  /* fixed wing: scale throttle to 0..1 and other channels to -1..1 */
145 
146  for (unsigned i = 0; i < 16; i++) {
147  if (_actuators[index].output[i] > PWM_DEFAULT_MIN / 2) {
148  if (i != 4) {
149  /* scale PWM out PWM_DEFAULT_MIN..PWM_DEFAULT_MAX us to -1..1 for normal channels */
150  msg.controls[i] = (_actuators[index].output[i] - pwm_center) / ((PWM_DEFAULT_MAX - PWM_DEFAULT_MIN) / 2);
151 
152  } else {
153  /* scale PWM out PWM_DEFAULT_MIN..PWM_DEFAULT_MAX us to 0..1 for throttle */
154  msg.controls[i] = (_actuators[index].output[i] - PWM_DEFAULT_MIN) / (PWM_DEFAULT_MAX - PWM_DEFAULT_MIN);
155  }
156 
157  } else {
158  /* set 0 for disabled channels */
159  msg.controls[i] = 0.0f;
160  }
161  }
162  }
163 
164  msg.mode = mode_flag_custom;
165  msg.mode |= (armed) ? mode_flag_armed : 0;
166  msg.flags = 0;
167 }

280 void Simulator::handle_message(mavlink_message_t *msg, bool publish)
281 {
282  switch (msg->msgid) {
283  case MAVLINK_MSG_ID_HIL_SENSOR: {
284  mavlink_hil_sensor_t imu;
285  mavlink_msg_hil_sensor_decode(msg, &imu);
286 
287  bool compensation_enabled = (imu.time_usec > 0);
288 
289  // set temperature to a decent value
290  imu.temperature = 32.0f;
291 
292  struct timespec ts;
293  // clock_gettime(CLOCK_MONOTONIC, &ts);
294  // uint64_t host_time = ts_to_abstime(&ts);
295 
296  hrt_abstime curr_sitl_time = hrt_absolute_time();
297  hrt_abstime curr_sim_time = imu.time_usec;
298 
299  if (compensation_enabled && _initialized
300  && _last_sim_timestamp > 0 && _last_sitl_timestamp > 0
301  && _last_sitl_timestamp < curr_sitl_time
302  && _last_sim_timestamp < curr_sim_time) {
303 
304  px4_clock_gettime(CLOCK_MONOTONIC, &ts);
305  uint64_t timestamp = ts_to_abstime(&ts);
306 
307  perf_set_elapsed(_perf_sim_delay, timestamp - curr_sim_time);
308  perf_count(_perf_sim_interval);
309 
310  int64_t dt_sitl = curr_sitl_time - _last_sitl_timestamp;
311  int64_t dt_sim = curr_sim_time - _last_sim_timestamp;
312 
313  double curr_factor = ((double)dt_sim / (double)dt_sitl);
314 
315  if (curr_factor < 5.0) {
316  _realtime_factor = _realtime_factor * 0.99 + 0.01 * curr_factor;
317  }
318 
319  // calculate how much the system needs to be delayed
320  int64_t sysdelay = dt_sitl - dt_sim;
321 
322  unsigned min_delay = 200;
323 
324  if (dt_sitl < 1e5
325  && dt_sim < 1e5
326  && sysdelay > min_delay + 100) {
327 
328  // the correct delay is exactly the scale between
329  // the last two intervals
330  px4_sim_start_delay();
331  hrt_start_delay();
332 
333  unsigned exact_delay = sysdelay / _realtime_factor;
334  unsigned usleep_delay = (sysdelay - min_delay) / _realtime_factor;
335 
336  // extend by the realtime factor to avoid drift
337  usleep(usleep_delay);
338  hrt_stop_delay_delta(exact_delay);
339  px4_sim_stop_delay();
340  }
341  }
342 
343  hrt_abstime now = hrt_absolute_time();
344 
345  _last_sitl_timestamp = curr_sitl_time;
346  _last_sim_timestamp = curr_sim_time;
347 
348  // correct timestamp
349  imu.time_usec = now;
350 
351  if (publish) {
352  publish_sensor_topics(&imu);
353  }
354 
355  update_sensors(&imu);
356 
357  // battery simulation (limit update to 100Hz)
358  if (hrt_elapsed_time(&_battery_status.timestamp) >= 10000) {
359 
360  const float discharge_interval_us = _battery_drain_interval_s.get() * 1000 * 1000;
361 
362  bool armed = (_vehicle_status.arming_state == vehicle_status_s::ARMING_STATE_ARMED);
363 
364  if (!armed || batt_sim_start == 0 || batt_sim_start > now) {
365  batt_sim_start = now;
366  }
367 
368  float ibatt = -1.0f; // no current sensor in simulation
369  const float minimum_percentage = 0.5f; // change this value if you want to simulate low battery reaction
370 
371  /* Simulate the voltage of a linearly draining battery but stop at the minimum percentage */
372  float battery_percentage = (now - batt_sim_start) / discharge_interval_us;
373  battery_percentage = math::min(battery_percentage, minimum_percentage);
374  float vbatt = math::gradual(battery_percentage, 0.f, 1.f, _battery.full_cell_voltage(), _battery.empty_cell_voltage());
375  vbatt *= _battery.cell_count();
376 
377  const float throttle = 0.0f; // simulate no throttle compensation to make the estimate predictable
378  _battery.updateBatteryStatus(now, vbatt, ibatt, true, true, 0, throttle, armed, &_battery_status);
379 
380  // publish the battery voltage
381  int batt_multi;
382  orb_publish_auto(ORB_ID(battery_status), &_battery_pub, &_battery_status, &batt_multi, ORB_PRIO_HIGH);
383  }
384  }
385  break;
386 
387  case MAVLINK_MSG_ID_HIL_OPTICAL_FLOW:
388  mavlink_hil_optical_flow_t flow;
389  mavlink_msg_hil_optical_flow_decode(msg, &flow);
390  publish_flow_topic(&flow);
391  break;
392 
393  case MAVLINK_MSG_ID_VISION_POSITION_ESTIMATE:
394  mavlink_vision_position_estimate_t ev;
395  mavlink_msg_vision_position_estimate_decode(msg, &ev);
396  publish_ev_topic(&ev);
397  break;
398 
399  case MAVLINK_MSG_ID_DISTANCE_SENSOR:
400  mavlink_distance_sensor_t dist;
401  mavlink_msg_distance_sensor_decode(msg, &dist);
402  publish_distance_topic(&dist);
403  break;
404 
405  case MAVLINK_MSG_ID_HIL_GPS:
406  mavlink_hil_gps_t gps_sim;
407  mavlink_msg_hil_gps_decode(msg, &gps_sim);
408 
409  if (publish) {
410  //PX4_WARN("FIXME: Need to publish GPS topic. Not done yet.");
411  }
412 
413  update_gps(&gps_sim);
414  break;
415 
416  case MAVLINK_MSG_ID_RC_CHANNELS:
417  mavlink_rc_channels_t rc_channels;
418  mavlink_msg_rc_channels_decode(msg, &rc_channels);
419  fill_rc_input_msg(&_rc_input, &rc_channels);
420 
421  // publish message
422  if (publish) {
423  int rc_multi;
424  orb_publish_auto(ORB_ID(input_rc), &_rc_channels_pub, &_rc_input, &rc_multi, ORB_PRIO_HIGH);
425  }
426 
427  break;
428 
429  case MAVLINK_MSG_ID_LANDING_TARGET:
430  mavlink_landing_target_t landing_target_mavlink;
431  mavlink_msg_landing_target_decode(msg, &landing_target_mavlink);
432 
433  struct irlock_report_s report;
434  memset(&report, 0, sizeof(report));
435 
436  report.timestamp = hrt_absolute_time();
437  report.signature = landing_target_mavlink.target_num;
438  report.pos_x = landing_target_mavlink.angle_x;
439  report.pos_y = landing_target_mavlink.angle_y;
440  report.size_x = landing_target_mavlink.size_x;
441  report.size_y = landing_target_mavlink.size_y;
442 
443  int irlock_multi;
444  orb_publish_auto(ORB_ID(irlock_report), &_irlock_report_pub, &report, &irlock_multi, ORB_PRIO_HIGH);
445 
446  break;
447 
448  case MAVLINK_MSG_ID_HIL_STATE_QUATERNION:
449  mavlink_hil_state_quaternion_t hil_state;
450  mavlink_msg_hil_state_quaternion_decode(msg, &hil_state);
451 
452  uint64_t timestamp = hrt_absolute_time();
453 
454  /* attitude */
455  struct vehicle_attitude_s hil_attitude = {};
456  {
457  hil_attitude.timestamp = timestamp;
458 
459  matrix::Quatf q(hil_state.attitude_quaternion);
460  q.copyTo(hil_attitude.q);
461 
462  hil_attitude.rollspeed = hil_state.rollspeed;
463  hil_attitude.pitchspeed = hil_state.pitchspeed;
464  hil_attitude.yawspeed = hil_state.yawspeed;
465 
466  // always publish ground truth attitude message
467  int hilstate_multi;
468  orb_publish_auto(ORB_ID(vehicle_attitude_groundtruth), &_attitude_pub, &hil_attitude, &hilstate_multi, ORB_PRIO_HIGH);
469  }
470 
471  /* global position */
472  struct vehicle_global_position_s hil_gpos = {};
473  {
474  hil_gpos.timestamp = timestamp;
475 
476  hil_gpos.lat = hil_state.lat / 1E7;//1E7
477  hil_gpos.lon = hil_state.lon / 1E7;//1E7
478  hil_gpos.alt = hil_state.alt / 1E3;//1E3
479 
480  hil_gpos.vel_n = hil_state.vx / 100.0f;
481  hil_gpos.vel_e = hil_state.vy / 100.0f;
482  hil_gpos.vel_d = hil_state.vz / 100.0f;
483 
484  // always publish ground truth attitude message
485  int hil_gpos_multi;
486  orb_publish_auto(ORB_ID(vehicle_global_position_groundtruth), &_gpos_pub, &hil_gpos, &hil_gpos_multi,
487  ORB_PRIO_HIGH);
488  }
489 
490  /* local position */
491  struct vehicle_local_position_s hil_lpos = {};
492  {
493  hil_lpos.timestamp = timestamp;
494 
495  double lat = hil_state.lat * 1e-7;
496  double lon = hil_state.lon * 1e-7;
497 
498  if (!_hil_local_proj_inited) {
499  _hil_local_proj_inited = true;
500  map_projection_init(&_hil_local_proj_ref, lat, lon);
501  _hil_ref_timestamp = timestamp;
502  _hil_ref_lat = lat;
503  _hil_ref_lon = lon;
504  _hil_ref_alt = hil_state.alt / 1000.0f;
505  }
506 
507  float x;
508  float y;
509  map_projection_project(&_hil_local_proj_ref, lat, lon, &x, &y);
510  hil_lpos.timestamp = timestamp;
511  hil_lpos.xy_valid = true;
512  hil_lpos.z_valid = true;
513  hil_lpos.v_xy_valid = true;
514  hil_lpos.v_z_valid = true;
515  hil_lpos.x = x;
516  hil_lpos.y = y;
517  hil_lpos.z = _hil_ref_alt - hil_state.alt / 1000.0f;
518  hil_lpos.vx = hil_state.vx / 100.0f;
519  hil_lpos.vy = hil_state.vy / 100.0f;
520  hil_lpos.vz = hil_state.vz / 100.0f;
521  matrix::Eulerf euler = matrix::Quatf(hil_attitude.q);
522  hil_lpos.yaw = euler.psi();
523  hil_lpos.xy_global = true;
524  hil_lpos.z_global = true;
525  hil_lpos.ref_lat = _hil_ref_lat;
526  hil_lpos.ref_lon = _hil_ref_lon;
527  hil_lpos.ref_alt = _hil_ref_alt;
528  hil_lpos.ref_timestamp = _hil_ref_timestamp;
529  hil_lpos.vxy_max = std::numeric_limits<float>::infinity();
530  hil_lpos.vz_max = std::numeric_limits<float>::infinity();
531  hil_lpos.hagl_min = std::numeric_limits<float>::infinity();
532  hil_lpos.hagl_max = std::numeric_limits<float>::infinity();
533 
534  // always publish ground truth attitude message

537  ORB_PRIO_HIGH);
538  }
539 
540  break;
541  }
