98 void
99 isr_debug(uint8_t level, const char *fmt, ...)
100 {
101  if (level > r_page_setup[PX4IO_P_SETUP_SET_DEBUG]) {
102  return;
103  }
104 
105  va_list ap;
106  va_start(ap, fmt);
107  vsnprintf(msg[msg_next_in], sizeof(msg[0]), fmt, ap);
108  va_end(ap);
109  msg_next_in = (msg_next_in + 1) % NUM_MSG;
110  msg_counter++;
111 }

116 static void
117 show_debug_messages(void)
118 {
119  if (msg_counter != last_msg_counter) {
120  uint32_t n = msg_counter - last_msg_counter;
121 
122  if (n > NUM_MSG) { n = NUM_MSG; }
123 
124  last_msg_counter = msg_counter;
125 
126  while (n--) {
127  debug("%s", msg[msg_next_out]);
128  msg_next_out = (msg_next_out + 1) % NUM_MSG;

136 static void
137 update_mem_usage(void)
138 {
139  if (/* IO armed */ (r_status_flags & PX4IO_P_STATUS_FLAGS_SAFETY_OFF)
140  /* and FMU is armed */ && (r_setup_arming & PX4IO_P_SETUP_ARMING_FMU_ARMED)) {
141  return;
142  }
143 
144  static uint64_t last_mem_time = 0;
145  uint64_t now = hrt_absolute_time();

147  if (now - last_mem_time > (500 * 1000)) {
148  struct mallinfo minfo = mallinfo();

161 static void
162 ring_blink(void)
163 {
164 #ifdef GPIO_LED4
165 
166  if (/* IO armed */ (r_status_flags & PX4IO_P_STATUS_FLAGS_SAFETY_OFF)
167  /* and FMU is armed */ && (r_setup_arming & PX4IO_P_SETUP_ARMING_FMU_ARMED)) {
168  LED_RING(1);
169  return;
170  }
171 
172  // XXX this led code does have
173  // intentionally a few magic numbers.
174  const unsigned max_brightness = 118;
175 
176  static unsigned counter = 0;
177  static unsigned brightness = max_brightness;
178  static unsigned brightness_counter = 0;
179  static unsigned on_counter = 0;
180 
181  if (brightness_counter < max_brightness) {
182 
183  bool on = ((on_counter * 100) / brightness_counter + 1) <= ((brightness * 100) / max_brightness + 1);
184 
185  // XXX once led is PWM driven,
186  // remove the ! in the line below
187  // to return to the proper breathe
188  // animation / pattern (currently inverted)
189  LED_RING(!on);
190  brightness_counter++;
191 
192  if (on) {
193  on_counter++;
194  }
195 
196  } else {
197 
198  if (counter >= 62) {
199  counter = 0;
200  }
201 
202  int n;
203 
204  if (counter < 32) {
205  n = counter;
206 
207  } else {
208  n = 62 - counter;
209  }
210 
211  brightness = (n * n) / 8;
212  brightness_counter = 0;
213  on_counter = 0;
214  counter++;
215  }
216 
217 #endif
218 }

240 static void
241 calculate_fw_crc(void)
242 {
243 #define APP_SIZE_MAX 0xf000
244 #define APP_LOAD_ADDRESS 0x08001000
245  // compute CRC of the current firmware
246  uint32_t sum = 0;
247 
248  for (unsigned p = 0; p < APP_SIZE_MAX; p += 4) {
249  uint32_t bytes = *(uint32_t *)(p + APP_LOAD_ADDRESS);
250  sum = crc32part((uint8_t *)&bytes, sizeof(bytes), sum);
251  }
252 
253  r_page_setup[PX4IO_P_SETUP_CRC] = sum & 0xFFFF;
254  r_page_setup[PX4IO_P_SETUP_CRC + 1] = sum >> 16;
255 }

257 int
258 user_start(int argc, char *argv[])
259 {
260  /* configure the first 8 PWM outputs (i.e. all of them) */
261  up_pwm_servo_init(0xff);
262 
263 #if defined(CONFIG_HAVE_CXX) && defined(CONFIG_HAVE_CXXINITIALIZE)
264 
265  /* run C++ ctors before we go any further */
266 
267  up_cxxinitialize();
268 
269 # if defined(CONFIG_EXAMPLES_NSH_CXXINITIALIZE)
270 # error CONFIG_EXAMPLES_NSH_CXXINITIALIZE Must not be defined! Use CONFIG_HAVE_CXX and CONFIG_HAVE_CXXINITIALIZE.
271 # endif
272 
273 #else
274 # error platform is dependent on c++ both CONFIG_HAVE_CXX and CONFIG_HAVE_CXXINITIALIZE must be defined.
275 #endif
276 
277  /* reset all to zero */
278  memset(&system_state, 0, sizeof(system_state));
279 
280  /* configure the high-resolution time/callout interface */
281  hrt_init();
282 
283  /* calculate our fw CRC so FMU can decide if we need to update */
284  calculate_fw_crc();
285 
286  /*
287  * Poll at 1ms intervals for received bytes that have not triggered
288  * a DMA event.
289  */
290 #ifdef CONFIG_ARCH_DMA
291  hrt_call_every(&serial_dma_call, 1000, 1000, (hrt_callout)stm32_serial_dma_poll, NULL);
292 #endif
293 
294  /* print some startup info */
295  syslog(LOG_INFO, "\nPX4IO: starting\n");
296 
297  /* default all the LEDs to off while we start */
298  LED_AMBER(false);
299  LED_BLUE(false);
300  LED_SAFETY(false);
301 #ifdef GPIO_LED4
302  LED_RING(false);
303 #endif
304 
305  /* turn on servo power (if supported) */
306 #ifdef POWER_SERVO
307  POWER_SERVO(true);
308 #endif
309 
310  /* turn off S.Bus out (if supported) */
311 #ifdef ENABLE_SBUS_OUT
312  ENABLE_SBUS_OUT(false);
313 #endif
314 
315  /* start the safety switch handler */
316  safety_init();
317 
318  /* initialise the control inputs */
319  controls_init();
320 
321  /* set up the ADC */
322  adc_init();
323 
324  /* start the FMU interface */
325  interface_init();
326 
327  /* add a performance counter for mixing */
328  perf_counter_t mixer_perf = perf_alloc(PC_ELAPSED, "mix");
329 
330  /* add a performance counter for controls */
331  perf_counter_t controls_perf = perf_alloc(PC_ELAPSED, "controls");
332 
333  /* and one for measuring the loop rate */
334  perf_counter_t loop_perf = perf_alloc(PC_INTERVAL, "loop");
335 
336  struct mallinfo minfo = mallinfo();
337  r_page_status[PX4IO_P_STATUS_FREEMEM] = minfo.mxordblk;
338  syslog(LOG_INFO, "MEM: free %u, largest %u\n", minfo.mxordblk, minfo.fordblks);
339 
340  /* initialize PWM limit lib */
341  pwm_limit_init(&pwm_limit);
342 
343  /*
344  * P O L I C E L I G H T S
345  *
346  * Not enough memory, lock down.
347  *
348  * We might need to allocate mixers later, and this will
349  * ensure that a developer doing a change will notice
350  * that he just burned the remaining RAM with static
351  * allocations. We don't want him to be able to
352  * get past that point. This needs to be clearly
353  * documented in the dev guide.
354  *
355  */
356  if (minfo.mxordblk < 600) {
357 
358  syslog(LOG_ERR, "ERR: not enough MEM");
359  bool phase = false;
360 
361  while (true) {
362 
363  if (phase) {
364  LED_AMBER(true);
365  LED_BLUE(false);
366 
367  } else {
368  LED_AMBER(false);
369  LED_BLUE(true);
370  }
371 
372  up_udelay(250000);
373 
374  phase = !phase;
375  }
376  }
377 
378  /* Start the failsafe led init */
379  failsafe_led_init();
380 
381  /*
382  * Run everything in a tight loop.
383  */
384 
385  uint64_t last_debug_time = 0;
386  uint64_t last_heartbeat_time = 0;
387  uint64_t last_loop_time = 0;
388 
389  for (;;) {
390  dt = (hrt_absolute_time() - last_loop_time) / 1000000.0f;
391  last_loop_time = hrt_absolute_time();
392 
393  if (dt < 0.0001f) {
394  dt = 0.0001f;
395 
396  } else if (dt > 0.02f) {
397  dt = 0.02f;
398  }
399 
400  /* track the rate at which the loop is running */
401  perf_count(loop_perf);
402 
403  /* kick the mixer */
404  perf_begin(mixer_perf);
405  mixer_tick();
406  perf_end(mixer_perf);
407 
408  /* kick the control inputs */
409  perf_begin(controls_perf);
410  controls_tick();
411  perf_end(controls_perf);
412 
413  /* some boards such as Pixhawk 2.1 made
414  the unfortunate choice to combine the blue led channel with
415  the IMU heater. We need a software hack to fix the hardware hack
416  by allowing to disable the LED / heater.
417  */
418  if (r_page_setup[PX4IO_P_SETUP_THERMAL] == PX4IO_THERMAL_IGNORE) {
419  /*
420  blink blue LED at 4Hz in normal operation. When in
421  override blink 4x faster so the user can clearly see
422  that override is happening. This helps when
423  pre-flight testing the override system
424  */
425  uint32_t heartbeat_period_us = 250 * 1000UL;
426 
427  if (r_status_flags & PX4IO_P_STATUS_FLAGS_OVERRIDE) {
428  heartbeat_period_us /= 4;
429  }
430 
431  if ((hrt_absolute_time() - last_heartbeat_time) > heartbeat_period_us) {
432  last_heartbeat_time = hrt_absolute_time();
433  heartbeat_blink();
434  }
435 
436  } else if (r_page_setup[PX4IO_P_SETUP_THERMAL] < PX4IO_THERMAL_FULL) {
437  /* switch resistive heater off */
438  LED_BLUE(false);
439 
440  } else {
441  /* switch resistive heater hard on */
442  LED_BLUE(true);
443  }
444 
445  update_mem_usage();
446 
447  ring_blink();
448 
449  check_reboot();
450 
451  /* check for debug activity (default: none) */
452  show_debug_messages();
453 
454  /* post debug state at ~1Hz - this is via an auxiliary serial port
455  * DEFAULTS TO OFF!
456  */
457  if (hrt_absolute_time() - last_debug_time > (1000 * 1000)) {
458 
459  isr_debug(1, "d:%u s=0x%x a=0x%x f=0x%x m=%u",
460  (unsigned)r_page_setup[PX4IO_P_SETUP_SET_DEBUG],
461  (unsigned)r_status_flags,
462  (unsigned)r_setup_arming,
463  (unsigned)r_setup_features,
464  (unsigned)mallinfo().mxordblk);
465  last_debug_time = hrt_absolute_time();
466  }
467  }
468 }
