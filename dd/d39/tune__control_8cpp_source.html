67  "tune_control play [-t <default tunes>] [-f <frequency>] [-d <duration>] [-s <strength>] [-m < melody>]\n"

79 static void publish_tune_control(tune_control_s &tune_control)
80 {
81  tune_control.timestamp = hrt_absolute_time();
82 
83  if (tune_control_pub == nullptr) {
84  tune_control_pub = orb_advertise(ORB_ID(tune_control), &tune_control);
85 
86  } else {
87  orb_publish(ORB_ID(tune_control), tune_control_pub, &tune_control);
88  }
89 }

91 int
92 tune_control_main(int argc, char *argv[])
93 {
94  Tunes tunes;
95  bool string_input = false;
96  const char *tune_string = nullptr;
97  int myoptind = 1;
98  int ch;
99  const char *myoptarg = nullptr;
100  unsigned int value;
101  tune_control_s tune_control = {};
102  tune_control.tune_id = 0;
103  tune_control.strength = tune_control_s::STRENGTH_NORMAL;
104 
105  while ((ch = px4_getopt(argc, argv, "f:d:t:m:s:", &myoptind, &myoptarg)) != EOF) {
106  switch (ch) {
107  case 'f':
108  value = (uint16_t)(strtol(myoptarg, nullptr, 0));
109 
110  if (value > 0 && value < 22000) {
111  tune_control.frequency = value;
112 
113  } else {
114  usage();
115  return 1;
116  }
117 
118  break;
119 
120  case 'd':
121  tune_control.duration = (uint32_t)(strtol(myoptarg, nullptr, 0));
122  break;
123 
124  case 't':
125  value = (uint8_t)(strtol(myoptarg, nullptr, 0));
126 
127  if (value > 0 && value < tunes.get_default_tunes_size()) {
128  tune_control.tune_id = value;
129 
130  } else {
131  usage();
132  return 1;
133  }
134 
135  break;
136 
137  case 'm':
138  string_input = true;
139  tune_string = myoptarg;
140 
141  // check if string is a valid melody string
142  if (tune_string[0] != 'M') {
143  usage();
144  return 1;
145  }
146 
147  break;
148 
149  case 's':
150  value = (uint8_t)(strtol(myoptarg, nullptr, 0));
151 
152  if (value > 0 && value < 100) {
153  tune_control.strength = value;
154 
155  } else {
156  tune_control.strength = tune_control_s::STRENGTH_NORMAL;
157  }
158 
159  break;
160 
161  default:
162  usage();
163  return -1;
164  break;
165  }
166  }
167 
168  if (myoptind >= argc) {
169  usage();
170  return 1;
171  }
172 
173  unsigned frequency, duration, silence;
174  uint8_t strength;
175  int exit_counter = 0;
176 
177  if (!strcmp(argv[myoptind], "play")) {
178  if (string_input) {
179  PX4_INFO("Start playback...");
180  tunes.set_string(tune_string, tune_control.strength);
181 
182  while (tunes.get_next_tune(frequency, duration, silence, strength) > 0) {
183  tune_control.tune_id = 0;
184  tune_control.frequency = (uint16_t)frequency;
185  tune_control.duration = (uint32_t)duration;
186  tune_control.silence = (uint32_t)silence;
187  tune_control.strength = (uint8_t)strength;
188  publish_tune_control(tune_control);
189  usleep(duration + silence);
190  exit_counter++;
191 
192  // exit if the loop is doing too many iterations
193  if (exit_counter > MAX_NOTE_ITERATION) {
194  break;
195  }
196  }
197 
198  PX4_INFO("Playback finished.");
199 
200  } else { // tune id instead of string has been provided
201  if (tune_control.tune_id == 0) {
202  tune_control.tune_id = 1;
203  }
204 
205  PX4_INFO("Publishing standard tune %d", tune_control.tune_id);
206  publish_tune_control(tune_control);
207  }
208 
209  } else if (!strcmp(argv[myoptind], "libtest")) {
210  int ret = tunes.set_control(tune_control);
211 
212  if (ret == -EINVAL) {
213  PX4_WARN("Tune ID not recognized.");
214  }
215 
216  while (tunes.get_next_tune(frequency, duration, silence, strength) > 0) {
217  PX4_INFO("frequency: %d, duration %d, silence %d, strength%d",
218  frequency, duration, silence, strength);
219  usleep(500000);
220  exit_counter++;
221 
222  // exit if the loop is doing too many iterations
223  if (exit_counter > MAX_NOTE_ITERATION) {
224  break;
225  }
226  }
227 
228  } else if (!strcmp(argv[myoptind], "stop")) {
229  PX4_INFO("Stopping playback...");
230  tune_control.tune_id = 0;
231  tune_control.frequency = 0;
232  tune_control.duration = 0;
233  tune_control.silence = 0;
234  tune_control.tune_override = true;
235  publish_tune_control(tune_control);
236 
237  } else {
238  usage();
239  return 1;
240  }
241 
242  return 0;
243 }
