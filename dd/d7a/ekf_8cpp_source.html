100 bool Ekf::update()
101 {
102  bool updated = false;
103 
104  if (!_filter_initialised) {
105  _filter_initialised = initialiseFilter();
106 
107  if (!_filter_initialised) {
108  return false;
109  }
110  }
111 
112  // Only run the filter if IMU data in the buffer has been updated
113  if (_imu_updated) {
114 
115  // perform state and covariance prediction for the main filter
116  predictState();
117  predictCovariance();
118 
119  // control fusion of observation data
120  controlFusionModes();
121 
122  // run a separate filter for terrain estimation
123  runTerrainEstimator();
124 
125  updated = true;
126  }
127 
128  // the output observer always runs
129  // Use full rate IMU data at the current time horizon
130  calculateOutputStates();
131 
132  return updated;
133 }

135 bool Ekf::initialiseFilter()
136 {
137  // Keep accumulating measurements until we have a minimum of 10 samples for the required sensors
138 
139  // Sum the IMU delta angle measurements
140  const imuSample &imu_init = _imu_buffer.get_newest();
141  _delVel_sum += imu_init.delta_vel;
142 
143  // Sum the magnetometer measurements
144  if (_mag_buffer.pop_first_older_than(_imu_sample_delayed.time_us, &_mag_sample_delayed)) {
145  if ((_mag_counter == 0) && (_mag_sample_delayed.time_us != 0)) {
146  // initialise the counter when we start getting data from the buffer
147  _mag_counter = 1;
148 
149  } else if ((_mag_counter != 0) && (_mag_sample_delayed.time_us != 0)) {
150  // increment the sample count and apply a LPF to the measurement
151  _mag_counter ++;
152 
153  // don't start using data until we can be certain all bad initial data has been flushed
154  if (_mag_counter == (uint8_t)(_obs_buffer_length + 1)) {
155  // initialise filter states
156  _mag_filt_state = _mag_sample_delayed.mag;
157 
158  } else if (_mag_counter > (uint8_t)(_obs_buffer_length + 1)) {
159  // noise filter the data
160  _mag_filt_state = _mag_filt_state * 0.9f + _mag_sample_delayed.mag * 0.1f;
161  }
162  }
163  }
164 
165  // Count the number of external vision measurements received
166  if (_ext_vision_buffer.pop_first_older_than(_imu_sample_delayed.time_us, &_ev_sample_delayed)) {
167  if ((_ev_counter == 0) && (_ev_sample_delayed.time_us != 0)) {
168  // initialise the counter
169  _ev_counter = 1;
170 
171  // set the height fusion mode to use external vision data when we start getting valid data from the buffer
172  if (_primary_hgt_source == VDIST_SENSOR_EV) {
173  _control_status.flags.baro_hgt = false;
174  _control_status.flags.gps_hgt = false;
175  _control_status.flags.rng_hgt = false;
176  _control_status.flags.ev_hgt = true;
177  }
178 
179  } else if ((_ev_counter != 0) && (_ev_sample_delayed.time_us != 0)) {
180  // increment the sample count
181  _ev_counter ++;
182  }
183  }
184 
185  // set the default height source from the adjustable parameter
186  if (_hgt_counter == 0) {
187  _primary_hgt_source = _params.vdist_sensor_type;
188  }
189 
190  // accumulate enough height measurements to be confident in the quality of the data
191  // we use baro height initially and switch to GPS/range/EV finder later when it passes checks.
192  if (_baro_buffer.pop_first_older_than(_imu_sample_delayed.time_us, &_baro_sample_delayed)) {
193  if ((_hgt_counter == 0) && (_baro_sample_delayed.time_us != 0)) {
194  // initialise the counter and height fusion method when we start getting data from the buffer
195  setControlBaroHeight();
196  _hgt_counter = 1;
197 
198  } else if ((_hgt_counter != 0) && (_baro_sample_delayed.time_us != 0)) {
199  // increment the sample count and apply a LPF to the measurement
200  _hgt_counter ++;
201 
202  // don't start using data until we can be certain all bad initial data has been flushed
203  if (_hgt_counter == (uint8_t)(_obs_buffer_length + 1)) {
204  // initialise filter states
205  _baro_hgt_offset = _baro_sample_delayed.hgt;
206 
207  } else if (_hgt_counter > (uint8_t)(_obs_buffer_length + 1)) {
208  // noise filter the data
209  _baro_hgt_offset = 0.9f * _baro_hgt_offset + 0.1f * _baro_sample_delayed.hgt;
210  }
211  }
212  }
213 
214  // check to see if we have enough measurements and return false if not
215  bool hgt_count_fail = _hgt_counter <= 2u * _obs_buffer_length;
216  bool mag_count_fail = _mag_counter <= 2u * _obs_buffer_length;
217 
218  if (hgt_count_fail || mag_count_fail) {
219  return false;
220 
221  } else {
222  // reset variables that are shared with post alignment GPS checks
223  _gps_drift_velD = 0.0f;
224  _gps_alt_ref = 0.0f;
225 
226  // Zero all of the states
227  _state.vel.setZero();
228  _state.pos.setZero();
229  _state.gyro_bias.setZero();
230  _state.accel_bias.setZero();
231  _state.mag_I.setZero();
232  _state.mag_B.setZero();
233  _state.wind_vel.setZero();
234 
235  // get initial roll and pitch estimate from delta velocity vector, assuming vehicle is static
236  float pitch = 0.0f;
237  float roll = 0.0f;
238 
239  if (_delVel_sum.norm() > 0.001f) {
240  _delVel_sum.normalize();
241  pitch = asinf(_delVel_sum(0));
242  roll = atan2f(-_delVel_sum(1), -_delVel_sum(2));
243 
244  } else {
245  return false;
246  }
247 
248  // calculate initial tilt alignment
249  Eulerf euler_init(roll, pitch, 0.0f);
250  _state.quat_nominal = Quatf(euler_init);
251  _output_new.quat_nominal = _state.quat_nominal;
252 
253  // update transformation matrix from body to world frame
254  _R_to_earth = quat_to_invrotmat(_state.quat_nominal);
255 
256  // calculate the averaged magnetometer reading
257  Vector3f mag_init = _mag_filt_state;
258 
259  // calculate the initial magnetic field and yaw alignment
260  _control_status.flags.yaw_align = resetMagHeading(mag_init);
261 
262  if (_control_status.flags.rng_hgt) {
263  // if we are using the range finder as the primary source, then calculate the baro height at origin so we can use baro as a backup
264  // so it can be used as a backup ad set the initial height using the range finder
265  const baroSample &baro_newest = _baro_buffer.get_newest();
266  _baro_hgt_offset = baro_newest.hgt;
267  _state.pos(2) = -math::max(_rng_filt_state * _R_rng_to_earth_2_2, _params.rng_gnd_clearance);
268  ECL_INFO("EKF using range finder height - commencing alignment");
269 
270  } else if (_control_status.flags.ev_hgt) {
271  // if we are using external vision data for height, then the vertical position state needs to be reset
272  // because the initialisation position is not the zero datum
273  resetHeight();
274 
275  }
276 
277  // initialise the state covariance matrix
278  initialiseCovariance();
279 
280  // try to initialise the terrain estimator
281  _terrain_initialised = initHagl();
282 
283  // reset the essential fusion timeout counters
284  _time_last_hgt_fuse = _time_last_imu;
285  _time_last_pos_fuse = _time_last_imu;
286  _time_last_delpos_fuse = _time_last_imu;
287  _time_last_vel_fuse = _time_last_imu;
288  _time_last_hagl_fuse = _time_last_imu;
289  _time_last_of_fuse = _time_last_imu;
290 
291  // reset the output predictor state history to match the EKF initial values
292  alignOutputFilter();
293 
294  return true;
295  }

298 void Ekf::predictState()
299 {
300  if (!_earth_rate_initialised) {
301  if (_NED_origin_initialised) {
302  calcEarthRateNED(_earth_rate_NED, (float)_pos_ref.lat_rad);
303  _earth_rate_initialised = true;
304  }
305  }
306 
307  // apply imu bias corrections
308  Vector3f corrected_delta_ang = _imu_sample_delayed.delta_ang - _state.gyro_bias;
309  Vector3f corrected_delta_vel = _imu_sample_delayed.delta_vel - _state.accel_bias;
310 
311  // correct delta angles for earth rotation rate
312  corrected_delta_ang -= -_R_to_earth.transpose() * _earth_rate_NED * _imu_sample_delayed.delta_ang_dt;
313 
314  // convert the delta angle to a delta quaternion
315  Quatf dq;
316  dq.from_axis_angle(corrected_delta_ang);
317 
318  // rotate the previous quaternion by the delta quaternion using a quaternion multiplication
319  _state.quat_nominal = _state.quat_nominal * dq;
320 
321  // quaternions must be normalised whenever they are modified
322  _state.quat_nominal.normalize();
323 
324  // save the previous value of velocity so we can use trapzoidal integration
325  Vector3f vel_last = _state.vel;
326 
327  // update transformation matrix from body to world frame
328  _R_to_earth = quat_to_invrotmat(_state.quat_nominal);
329 
330  // Calculate an earth frame delta velocity
331  Vector3f corrected_delta_vel_ef = _R_to_earth * corrected_delta_vel;
332 
333  // calculate a filtered horizontal acceleration with a 1 sec time constant
334  // this are used for manoeuvre detection elsewhere
335  float alpha = 1.0f - _imu_sample_delayed.delta_vel_dt;
336  _accel_lpf_NE(0) = _accel_lpf_NE(0) * alpha + corrected_delta_vel_ef(0);
337  _accel_lpf_NE(1) = _accel_lpf_NE(1) * alpha + corrected_delta_vel_ef(1);
338 
339  // calculate the increment in velocity using the current orientation
340  _state.vel += corrected_delta_vel_ef;
341 
342  // compensate for acceleration due to gravity
343  _state.vel(2) += CONSTANTS_ONE_G * _imu_sample_delayed.delta_vel_dt;
344 
345  // predict position states via trapezoidal integration of velocity
346  _state.pos += (vel_last + _state.vel) * _imu_sample_delayed.delta_vel_dt * 0.5f;
347 
348  constrainStates();
349 
350  // calculate an average filter update time
351  float input = 0.5f * (_imu_sample_delayed.delta_vel_dt + _imu_sample_delayed.delta_ang_dt);
352 
353  // filter and limit input between -50% and +100% of nominal value
354  input = math::constrain(input, 0.5f * FILTER_UPDATE_PERIOD_S, 2.0f * FILTER_UPDATE_PERIOD_S);
355  _dt_ekf_avg = 0.99f * _dt_ekf_avg + 0.01f * input;
356 }

358 bool Ekf::collect_imu(const imuSample &imu)
359 {
360  // accumulate and downsample IMU data across a period FILTER_UPDATE_PERIOD_MS long
361 
362  // copy imu data to local variables
363  _imu_sample_new = imu;
364 
365  // accumulate the time deltas
366  _imu_down_sampled.delta_ang_dt += imu.delta_ang_dt;
367  _imu_down_sampled.delta_vel_dt += imu.delta_vel_dt;
368 
369  // use a quaternion to accumulate delta angle data
370  // this quaternion represents the rotation from the start to end of the accumulation period
371  Quatf delta_q;
372  delta_q.rotate(imu.delta_ang);
373  _q_down_sampled = _q_down_sampled * delta_q;
374  _q_down_sampled.normalize();
375 
376  // rotate the accumulated delta velocity data forward each time so it is always in the updated rotation frame
377  Dcmf delta_R(delta_q.inversed());
378  _imu_down_sampled.delta_vel = delta_R * _imu_down_sampled.delta_vel;
379 
380  // accumulate the most recent delta velocity data at the updated rotation frame
381  // assume effective sample time is halfway between the previous and current rotation frame
382  _imu_down_sampled.delta_vel += (imu.delta_vel + delta_R * imu.delta_vel) * 0.5f;
383 
384  // if the target time delta between filter prediction steps has been exceeded
385  // write the accumulated IMU data to the ring buffer
386  const float target_dt = FILTER_UPDATE_PERIOD_S;
387 
388  if (_imu_down_sampled.delta_ang_dt >= target_dt - _imu_collection_time_adj) {
389 
390  // accumulate the amount of time to advance the IMU collection time so that we meet the
391  // average EKF update rate requirement
392  _imu_collection_time_adj += 0.01f * (_imu_down_sampled.delta_ang_dt - target_dt);
393  _imu_collection_time_adj = math::constrain(_imu_collection_time_adj, -0.5f * target_dt, 0.5f * target_dt);
394 
395  imuSample imu_sample_new;
396  imu_sample_new.delta_ang = _q_down_sampled.to_axis_angle();
397  imu_sample_new.delta_vel = _imu_down_sampled.delta_vel;
398  imu_sample_new.delta_ang_dt = _imu_down_sampled.delta_ang_dt;
399  imu_sample_new.delta_vel_dt = _imu_down_sampled.delta_vel_dt;
400  imu_sample_new.time_us = imu.time_us;
401 
402  _imu_buffer.push(imu_sample_new);
403 
404  // get the oldest data from the buffer
405  _imu_sample_delayed = _imu_buffer.get_oldest();
406 
407  // calculate the minimum interval between observations required to guarantee no loss of data
408  // this will occur if data is overwritten before its time stamp falls behind the fusion time horizon
409  _min_obs_interval_us = (imu_sample_new.time_us - _imu_sample_delayed.time_us) / (_obs_buffer_length - 1);
410 
411  // reset
412  _imu_down_sampled.delta_ang.setZero();
413  _imu_down_sampled.delta_vel.setZero();
414  _imu_down_sampled.delta_ang_dt = 0.0f;
415  _imu_down_sampled.delta_vel_dt = 0.0f;
416  _q_down_sampled(0) = 1.0f;
417  _q_down_sampled(1) = _q_down_sampled(2) = _q_down_sampled(3) = 0.0f;
418 
419  _imu_updated = true;
420 
421  } else {
422  _imu_updated = false;
423  }
424 
425  return _imu_updated;
426 }

428 /*
429  * Implement a strapdown INS algorithm using the latest IMU data at the current time horizon.
430  * Buffer the INS states and calculate the difference with the EKF states at the delayed fusion time horizon.
431  * Calculate delta angle, delta velocity and velocity corrections from the differences and apply them at the
432  * current time horizon so that the INS states track the EKF states at the delayed fusion time horizon.
433  * The inspiration for using a complementary filter to correct for time delays in the EKF
434  * is based on the work by A Khosravian:
435  * “Recursive Attitude Estimation in the Presence of Multi-rate and Multi-delay Vector Measurements”
436  * A Khosravian, J Trumpf, R Mahony, T Hamel, Australian National University
437 */
438 void Ekf::calculateOutputStates()
439 {
440  // Use full rate IMU data at the current time horizon
441  const imuSample &imu = _imu_sample_new;
442 
443  // correct delta angles for bias offsets
444  const float dt_scale_correction = _dt_imu_avg / _dt_ekf_avg;
445 
446  // Apply corrections to the delta angle required to track the quaternion states at the EKF fusion time horizon
447  const Vector3f delta_angle{imu.delta_ang - _state.gyro_bias * dt_scale_correction + _delta_angle_corr};
448 
449  // calculate a yaw change about the earth frame vertical
450  const float spin_del_ang_D = _R_to_earth_now(2, 0) * delta_angle(0) +
451  _R_to_earth_now(2, 1) * delta_angle(1) +
452  _R_to_earth_now(2, 2) * delta_angle(2);
453  _yaw_delta_ef += spin_del_ang_D;
454 
455  // Calculate filtered yaw rate to be used by the magnetometer fusion type selection logic
456  // Note fixed coefficients are used to save operations. The exact time constant is not important.
457  _yaw_rate_lpf_ef = 0.95f * _yaw_rate_lpf_ef + 0.05f * spin_del_ang_D / imu.delta_ang_dt;
458 
459  // convert the delta angle to an equivalent delta quaternions
460  Quatf dq;
461  dq.from_axis_angle(delta_angle);
462 
463  // rotate the previous INS quaternion by the delta quaternions
464  _output_new.time_us = imu.time_us;
465  _output_new.quat_nominal = _output_new.quat_nominal * dq;
466 
467  // the quaternions must always be normalised afer modification
468  _output_new.quat_nominal.normalize();
469 
470  // calculate the rotation matrix from body to earth frame
471  _R_to_earth_now = quat_to_invrotmat(_output_new.quat_nominal);
472 
473  // correct delta velocity for bias offsets
474  const Vector3f delta_vel{imu.delta_vel - _state.accel_bias * dt_scale_correction};
475 
476  // rotate the delta velocity to earth frame
477  Vector3f delta_vel_NED{_R_to_earth_now * delta_vel};
478 
479  // correct for measured acceleration due to gravity
480  delta_vel_NED(2) += CONSTANTS_ONE_G * imu.delta_vel_dt;
481 
482  // calculate the earth frame velocity derivatives
483  if (imu.delta_vel_dt > 1e-4f) {
484  _vel_deriv_ned = delta_vel_NED * (1.0f / imu.delta_vel_dt);
485  }
486 
487  // save the previous velocity so we can use trapezoidal integration
488  const Vector3f vel_last{_output_new.vel};
489 
490  // increment the INS velocity states by the measurement plus corrections
491  // do the same for vertical state used by alternative correction algorithm
492  _output_new.vel += delta_vel_NED;
493  _output_vert_new.vel_d += delta_vel_NED(2);
494 
495  // use trapezoidal integration to calculate the INS position states
496  // do the same for vertical state used by alternative correction algorithm
497  const Vector3f delta_pos_NED = (_output_new.vel + vel_last) * (imu.delta_vel_dt * 0.5f);
498  _output_new.pos += delta_pos_NED;
499  _output_vert_new.vel_d_integ += delta_pos_NED(2);
500 
501  // accumulate the time for each update
502  _output_vert_new.dt += imu.delta_vel_dt;
503 
504  // correct velocity for IMU offset
505  if (imu.delta_ang_dt > 1e-4f) {
506  // calculate the average angular rate across the last IMU update
507  const Vector3f ang_rate = imu.delta_ang * (1.0f / imu.delta_ang_dt);
508 
509  // calculate the velocity of the IMU relative to the body origin
510  const Vector3f vel_imu_rel_body = cross_product(ang_rate, _params.imu_pos_body);
511 
512  // rotate the relative velocity into earth frame
513  _vel_imu_rel_body_ned = _R_to_earth_now * vel_imu_rel_body;
514  }
515 
516  // store the INS states in a ring buffer with the same length and time coordinates as the IMU data buffer
517  if (_imu_updated) {
518  _output_buffer.push(_output_new);
519  _output_vert_buffer.push(_output_vert_new);
520 
521  // get the oldest INS state data from the ring buffer
522  // this data will be at the EKF fusion time horizon
523  _output_sample_delayed = _output_buffer.get_oldest();
524  _output_vert_delayed = _output_vert_buffer.get_oldest();
525 
526  // calculate the quaternion delta between the INS and EKF quaternions at the EKF fusion time horizon
527  Quatf quat_inv = _state.quat_nominal.inversed();
528  Quatf q_error = quat_inv * _output_sample_delayed.quat_nominal;
529  q_error.normalize();
530 
531  // convert the quaternion delta to a delta angle
532  float scalar;
533 
534  if (q_error(0) >= 0.0f) {
535  scalar = -2.0f;
536 
537  } else {
538  scalar = 2.0f;
539  }
540 
541  const Vector3f delta_ang_error{scalar * q_error(1), scalar * q_error(2), scalar * q_error(3)};
542 
543  // calculate a gain that provides tight tracking of the estimator attitude states and
544  // adjust for changes in time delay to maintain consistent damping ratio of ~0.7
545  const float time_delay = fmaxf((imu.time_us - _imu_sample_delayed.time_us) * 1e-6f, _dt_imu_avg);
546  const float att_gain = 0.5f * _dt_imu_avg / time_delay;
547 
548  // calculate a corrrection to the delta angle
549  // that will cause the INS to track the EKF quaternions
550  _delta_angle_corr = delta_ang_error * att_gain;
551 
552  // calculate velocity and position tracking errors
553  const Vector3f vel_err{_state.vel - _output_sample_delayed.vel};
554  const Vector3f pos_err{_state.pos - _output_sample_delayed.pos};
555 
556  // collect magnitude tracking error for diagnostics
557  _output_tracking_error[0] = delta_ang_error.norm();
558  _output_tracking_error[1] = vel_err.norm();
559  _output_tracking_error[2] = pos_err.norm();
560 
561  /*
562  * Loop through the output filter state history and apply the corrections to the velocity and position states.
563  * This method is too expensive to use for the attitude states due to the quaternion operations required
564  * but because it eliminates the time delay in the 'correction loop' it allows higher tracking gains
565  * to be used and reduces tracking error relative to EKF states.
566  */
567 
568  // Complementary filter gains
569  const float vel_gain = _dt_ekf_avg / math::constrain(_params.vel_Tau, _dt_ekf_avg, 10.0f);
570  const float pos_gain = _dt_ekf_avg / math::constrain(_params.pos_Tau, _dt_ekf_avg, 10.0f);
571  {
572  /*
573  * Calculate a correction to be applied to vel_d that casues vel_d_integ to track the EKF
574  * down position state at the fusion time horizon using an alternative algorithm to what
575  * is used for the vel and pos state tracking. The algorithm applies a correction to the vel_d
576  * state history and propagates vel_d_integ forward in time using the corrected vel_d history.
577  * This provides an alternative vertical velocity output that is closer to the first derivative
578  * of the position but does degrade tracking relative to the EKF state.
579  */
580 
581  // calculate down velocity and position tracking errors
582  const float vel_d_err = (_state.vel(2) - _output_vert_delayed.vel_d);
583  const float pos_d_err = (_state.pos(2) - _output_vert_delayed.vel_d_integ);
584 
585  // calculate a velocity correction that will be applied to the output state history
586  // using a PD feedback tuned to a 5% overshoot
587  const float vel_d_correction = pos_d_err * pos_gain + vel_d_err * pos_gain * 1.1f;
588 
589  /*
590  * Calculate corrections to be applied to vel and pos output state history.
591  * The vel and pos state history are corrected individually so they track the EKF states at
592  * the fusion time horizon. This option provides the most accurate tracking of EKF states.
593  */
594 
595  // loop through the vertical output filter state history starting at the oldest and apply the corrections to the
596  // vel_d states and propagate vel_d_integ forward using the corrected vel_d
597  uint8_t index = _output_vert_buffer.get_oldest_index();
598 
599  const uint8_t size = _output_vert_buffer.get_length();
600 
601  for (uint8_t counter = 0; counter < (size - 1); counter++) {
602  const uint8_t index_next = (index + 1) % size;
603  outputVert &current_state = _output_vert_buffer[index];
604  outputVert &next_state = _output_vert_buffer[index_next];
605 
606  // correct the velocity
607  if (counter == 0) {
608  current_state.vel_d += vel_d_correction;
609  }
610 
611  next_state.vel_d += vel_d_correction;
612 
613  // position is propagated forward using the corrected velocity and a trapezoidal integrator
614  next_state.vel_d_integ = current_state.vel_d_integ + (current_state.vel_d + next_state.vel_d) * 0.5f * next_state.dt;
615 
616  // advance the index
617  index = (index + 1) % size;
618  }
619 
620  // update output state to corrected values
621  _output_vert_new = _output_vert_buffer.get_newest();
622 
623  // reset time delta to zero for the next accumulation of full rate IMU data
624  _output_vert_new.dt = 0.0f;
625  }
626 
627  {
628  /*
629  * Calculate corrections to be applied to vel and pos output state history.
630  * The vel and pos state history are corrected individually so they track the EKF states at
631  * the fusion time horizon. This option provides the most accurate tracking of EKF states.
632  */
633 
634  // calculate a velocity correction that will be applied to the output state history
635  _vel_err_integ += vel_err;
636  const Vector3f vel_correction = vel_err * vel_gain + _vel_err_integ * sq(vel_gain) * 0.1f;
637 
638  // calculate a position correction that will be applied to the output state history
639  _pos_err_integ += pos_err;
640  const Vector3f pos_correction = pos_err * pos_gain + _pos_err_integ * sq(pos_gain) * 0.1f;
641 
642  // loop through the output filter state history and apply the corrections to the velocity and position states
643  for (uint8_t index = 0; index < _output_buffer.get_length(); index++) {
644  // a constant velocity correction is applied
645  _output_buffer[index].vel += vel_correction;
646 
647  // a constant position correction is applied
648  _output_buffer[index].pos += pos_correction;
649  }
650 
651  // update output state to corrected values
652  _output_new = _output_buffer.get_newest();
653  }
654  }
655 }
656 
657 /*
658  * Predict the previous quaternion output state forward using the latest IMU delta angle data.
659 */
660 Quatf Ekf::calculate_quaternion() const
661 {
662  // Correct delta angle data for bias errors using bias state estimates from the EKF and also apply
663  // corrections required to track the EKF quaternion states
664  const Vector3f delta_angle{_imu_sample_new.delta_ang - _state.gyro_bias * (_dt_imu_avg / _dt_ekf_avg) + _delta_angle_corr};
665 
666  // increment the quaternions using the corrected delta angle vector
667  // the quaternions must always be normalised after modification
668  return Quatf{_output_new.quat_nominal * AxisAnglef{delta_angle}}.unit();
669 }
