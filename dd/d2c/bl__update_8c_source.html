80 int
81 bl_update_main(int argc, char *argv[])
82 {
83 #if !defined (CONFIG_STM32_STM32F4XXX)
84  PX4_ERR("Not supported on this HW");
85  return 1;
86 }
87 #else
88 
89  if (argc != 2)
90  {
91  print_usage("missing firmware filename or command");
92  return 1;
93  }
94 
95  if (!strcmp(argv[1], "setopt"))
96  {
97  return setopt();
98  }
99 
100  int fd = open(argv[1], O_RDONLY);
101 
102  if (fd < 0)
103  {
104  PX4_ERR("open %s failed", argv[1]);
105  return 1;
106  }
107 
108  struct stat s;
109 
110  if (stat(argv[1], &s) != 0)
111  {
112  PX4_ERR("stat %s failed", argv[1]);
113  close(fd);
114  return 1;
115  }
116 
117  /* sanity-check file size */
118  if (s.st_size > BL_FILE_SIZE_LIMIT)
119  {
120  PX4_ERR("%s: file too large (limit: %u, actual: %d)", argv[1], BL_FILE_SIZE_LIMIT, s.st_size);
121  close(fd);
122  return 1;
123  }
124 
125  uint8_t *buf = malloc(s.st_size);
126 
127  if (buf == NULL)
128  {
129  PX4_ERR("failed to allocate %u bytes for firmware buffer", s.st_size);
130  close(fd);
131  return 1;
132  }
133 
134  if (read(fd, buf, s.st_size) != s.st_size)
135  {
136  PX4_ERR("firmware read error");
137  close(fd);
138  free(buf);
139  return 1;
140  }
141 
142  close(fd);
143 
144  uint32_t *hdr = (uint32_t *)buf;
145 
146  if ((hdr[0] < 0x20000000) || /* stack not below RAM */
147  (hdr[0] > (0x20000000 + (128 * 1024))) || /* stack not above RAM */
148  (hdr[1] < PX4_FLASH_BASE) || /* entrypoint not below flash */
149  ((hdr[1] - PX4_FLASH_BASE) > BL_FILE_SIZE_LIMIT)) /* entrypoint not outside bootloader */
150  {
151  free(buf);
152  PX4_ERR("not a bootloader image");
153  return 1;
154  }
155 
156  PX4_INFO("image validated, erasing bootloader...");
157  usleep(10000);
158 
159  /* prevent other tasks from running while we do this */
160  sched_lock();
161 
162  const size_t page = 0;
163  uint8_t *base = (uint8_t *) PX4_FLASH_BASE;
164 
165  ssize_t size = up_progmem_erasepage(page);
166 
167  if (size != BL_FILE_SIZE_LIMIT)
168  {
169  PX4_ERR("erase error at 0x%08x", &base[size]);
170  }
171 
172  PX4_INFO("flashing...");
173 
174  /* now program the bootloader - speed is not critical so use x8 mode */
175 
176  size = up_progmem_write((size_t) base, buf, s.st_size);
177 
178  if (size != s.st_size)
179  {
180  PX4_ERR("program error at 0x%0x8", &base[size]);
181  goto flash_end;
182  }
183 
184  /* re-lock the flash control register */
185 
186  stm32_flash_lock();
187 
188  PX4_INFO("verifying...");
189 
190  /* now run a verify pass */
191  for (int i = 0; i < s.st_size; i++)
192  {
193  if (base[i] != buf[i]) {
194  PX4_WARN("verify failed at %u - retry update, DO NOT reboot", i);
195  goto flash_end;
196  }
197  }
198 
199  PX4_INFO("bootloader update complete");
200 
201 flash_end:
202  /* unlock the scheduler */
203  sched_unlock();
204 
205  free(buf);
206  exit(0);
207 }

209 static int
210 setopt(void)
211 {
212  volatile uint32_t *optcr = (volatile uint32_t *)0x40023c14;
213 
214  const uint16_t opt_mask = (3 << 2); /* BOR_LEV bitmask */
215  const uint16_t opt_bits = (0 << 2); /* BOR = 0, setting for 2.7-3.6V operation */
216 
217  if ((*optcr & opt_mask) == opt_bits) {
218  PX4_INFO("option bits are already set as required");
219  return 0;
220  }
221 
222  /* unlock the control register */
223  volatile uint32_t *optkeyr = (volatile uint32_t *)0x40023c08;
224  *optkeyr = 0x08192a3bU;
225  *optkeyr = 0x4c5d6e7fU;
226 
227  if (*optcr & 1) {
228  PX4_ERR("option control register unlock failed");
229  return 1;
230  }
231 
232  /* program the new option value */
233  *optcr = (*optcr & ~opt_mask) | opt_bits | (1 << 1);
234 
235  usleep(1000);
236 
237  if ((*optcr & opt_mask) == opt_bits) {
238  PX4_INFO("option bits set");
239  return 0;
240  }
241 
242  PX4_ERR("option bits setting failed; readback 0x%04x", *optcr);
243  return 1;
244 }
