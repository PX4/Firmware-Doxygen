89 bool FlightTaskAuto::_evaluateTriplets()
90 {
91  // TODO: fix the issues mentioned below
92  // We add here some conditions that are only required because:
93  // 1. navigator continuously sends triplet during mission due to yaw setpoint. This
94  // should be removed in the navigator and only updates if the current setpoint actually has changed.
95  //
96  // 2. navigator should be responsible to send always three valid setpoints. If there is only one setpoint,
97  // then previous will be set to current vehicle position and next will be set equal to setpoint.
98  //
99  // 3. navigator originally only supports gps guided maneuvers. However, it now also supports some flow-specific features
100  // such as land and takeoff. The navigator should use for auto takeoff/land with flow the position in xy at the moment the
101  // takeoff/land was initiated. Until then we do this kind of logic here.
102 
103  // Check if triplet is valid. There must be at least a valid altitude.
104 
105  if (!_sub_triplet_setpoint->get().current.valid || !PX4_ISFINITE(_sub_triplet_setpoint->get().current.alt)) {
106  // Best we can do is to just set all waypoints to current state and return false.
107  _prev_prev_wp = _triplet_prev_wp = _triplet_target = _triplet_next_wp = _position;
108  _type = WaypointType::position;
109  return false;
110  }
111 
112  _type = (WaypointType)_sub_triplet_setpoint->get().current.type;
113 
114  // Always update cruise speed since that can change without waypoint changes.
115  _mc_cruise_speed = _sub_triplet_setpoint->get().current.cruising_speed;
116 
117  if (!PX4_ISFINITE(_mc_cruise_speed) || (_mc_cruise_speed < 0.0f) || (_mc_cruise_speed > _constraints.speed_xy)) {
118  // Use default limit.
119  _mc_cruise_speed = _constraints.speed_xy;
120  }
121 
122  // Temporary target variable where we save the local reprojection of the latest navigator current triplet.
123  matrix::Vector3f tmp_target;
124 
125  if (!PX4_ISFINITE(_sub_triplet_setpoint->get().current.lat)
126  || !PX4_ISFINITE(_sub_triplet_setpoint->get().current.lon)) {
127  // No position provided in xy. Lock position
128  if (!PX4_ISFINITE(_lock_position_xy(0))) {
129  tmp_target(0) = _lock_position_xy(0) = _position(0);
130  tmp_target(1) = _lock_position_xy(1) = _position(1);
131 
132  } else {
133  tmp_target(0) = _lock_position_xy(0);
134  tmp_target(1) = _lock_position_xy(1);
135  _lock_position_xy *= NAN;
136  }
137 
138  } else {
139  // Convert from global to local frame.
140  map_projection_project(&_reference_position,
141  _sub_triplet_setpoint->get().current.lat, _sub_triplet_setpoint->get().current.lon, &tmp_target(0), &tmp_target(1));
142  }
143 
144  tmp_target(2) = -(_sub_triplet_setpoint->get().current.alt - _reference_altitude);
145 
146  // Check if anything has changed. We do that by comparing the temporary target
147  // to the internal _triplet_target.
148  // TODO This is a hack and it would be much better if the navigator only sends out a waypoints once they have changed.
149 
150  bool triplet_update = true;
151 
152  if (!(fabsf(_triplet_target(0) - tmp_target(0)) > 0.001f || fabsf(_triplet_target(1) - tmp_target(1)) > 0.001f
153  || fabsf(_triplet_target(2) - tmp_target(2)) > 0.001f)) {
154  // Nothing has changed: just keep old waypoints.
155  triplet_update = false;
156 
157  } else {
158  _triplet_target = tmp_target;
159 
160  if (!PX4_ISFINITE(_triplet_target(0)) || !PX4_ISFINITE(_triplet_target(1))) {
161  // Horizontal target is not finite.
162  _triplet_target(0) = _position(0);
163  _triplet_target(1) = _position(1);
164  }
165 
166  if (!PX4_ISFINITE(_triplet_target(2))) {
167  _triplet_target(2) = _position(2);
168  }
169 
170  // If _triplet_target has updated, update also _triplet_prev_wp and _triplet_next_wp.
171  _prev_prev_wp = _triplet_prev_wp;
172 
173  if (_isFinite(_sub_triplet_setpoint->get().previous) && _sub_triplet_setpoint->get().previous.valid) {
174  map_projection_project(&_reference_position, _sub_triplet_setpoint->get().previous.lat,
175  _sub_triplet_setpoint->get().previous.lon, &_triplet_prev_wp(0), &_triplet_prev_wp(1));
176  _triplet_prev_wp(2) = -(_sub_triplet_setpoint->get().previous.alt - _reference_altitude);
177 
178  } else {
179  _triplet_prev_wp = _position;
180  }
181 
182  if (_type == WaypointType::loiter) {
183  _triplet_next_wp = _triplet_target;
184 
185  } else if (_isFinite(_sub_triplet_setpoint->get().next) && _sub_triplet_setpoint->get().next.valid) {
186  map_projection_project(&_reference_position, _sub_triplet_setpoint->get().next.lat,
187  _sub_triplet_setpoint->get().next.lon, &_triplet_next_wp(0), &_triplet_next_wp(1));
188  _triplet_next_wp(2) = -(_sub_triplet_setpoint->get().next.alt - _reference_altitude);
189 
190  } else {
191  _triplet_next_wp = _triplet_target;
192  }
193  }
194 
195  // set heading
196  if (_type == WaypointType::follow_target && _sub_triplet_setpoint->get().current.yawspeed_valid) {
197  _yawspeed_setpoint = _sub_triplet_setpoint->get().current.yawspeed;
198  _yaw_setpoint = NAN;
199 
200  } else {
201  if (_sub_triplet_setpoint->get().current.yaw_valid) {
202  _yaw_setpoint = _sub_triplet_setpoint->get().current.yaw;
203 
204  } else {
205  _set_heading_from_mode();
206  }
207 
208  _yawspeed_setpoint = NAN;
209  }
210 
211  // Calculate the current vehicle state and check if it has updated.
212  State previous_state = _current_state;
213  _current_state = _getCurrentState();
214 
215  if (triplet_update || (_current_state != previous_state)) {
216  _updateInternalWaypoints();
217  _updateAvoidanceWaypoints();
218  }
219 
220  return true;
221 }

223 void FlightTaskAuto::_set_heading_from_mode()
224 {
225 
226  matrix::Vector2f v; // Vector that points towards desired location
227 
228  switch (MPC_YAW_MODE.get()) {
229 
230  case 0: { // Heading points towards the current waypoint.
231  v = Vector2f(&_target(0)) - Vector2f(&_position(0));
232  break;
233  }
234 
235  case 1: { // Heading points towards home.
236  if (_sub_home_position->get().valid_hpos) {
237  v = Vector2f(_sub_home_position->get().x, _sub_home_position->get().y) - Vector2f(&_position(0));
238  }
239 
240  break;
241  }
242 
243  case 2: { // Heading point away from home.
244  if (_sub_home_position->get().valid_hpos) {
245  v = Vector2f(&_position(0)) - Vector2f(_sub_home_position->get().x, _sub_home_position->get().y);
246  }
247 
248  break;
249  }
250 
251  case 3: { // Along trajectory.
252  // The heading depends on the kind of setpoint generation. This needs to be implemented
253  // in the subclasses where the velocity setpoints are generated.
254  v *= NAN;
255  }
256  }
257 
258  if (PX4_ISFINITE(v.length())) {
259  // We only adjust yaw if vehicle is outside of acceptance radius.
260  // This prevents excessive yawing.
261  if (v.length() > NAV_ACC_RAD.get()) {
262  _compute_heading_from_2D_vector(_yaw_setpoint, v);
263  }
264 
265  } else {
266  _yaw_setpoint = NAN;
267  }

352 State FlightTaskAuto::_getCurrentState()
353 {
354  // Calculate the vehicle current state based on the Navigator triplets and the current position.
355  Vector2f u_prev_to_target = Vector2f(&(_triplet_target - _triplet_prev_wp)(0)).unit_or_zero();
356  Vector2f pos_to_target = Vector2f(&(_triplet_target - _position)(0));
357  Vector2f prev_to_pos = Vector2f(&(_position - _triplet_prev_wp)(0));
358  // Calculate the closest point to the vehicle position on the line prev_wp - target
359  _closest_pt = Vector2f(&_triplet_prev_wp(0)) + u_prev_to_target * (prev_to_pos * u_prev_to_target);
360 
361  State return_state = State::none;
362 
363  if (u_prev_to_target * pos_to_target < 0.0f) {
364  // Target is behind.
365  return_state = State::target_behind;
366 
367  } else if (u_prev_to_target * prev_to_pos < 0.0f && prev_to_pos.length() > _mc_cruise_speed) {
368  // Current position is more than cruise speed in front of previous setpoint.
369  return_state = State::previous_infront;
370 
371  } else if (Vector2f(Vector2f(&_position(0)) - _closest_pt).length() > _mc_cruise_speed) {
372  // Vehicle is more than cruise speed off track.
373  return_state = State::offtrack;
374 
375  }
376 
377  return return_state;
378 }

380 void FlightTaskAuto::_updateInternalWaypoints()
381 {
382  // The internal Waypoints might differ from _triplet_prev_wp, _triplet_target and _triplet_next_wp.
383  // The cases where it differs:
384  // 1. The vehicle already passed the target -> go straight to target
385  // 2. The vehicle is more than cruise speed in front of previous waypoint -> go straight to previous waypoint
386  // 3. The vehicle is more than cruise speed from track -> go straight to closest point on track
387  //
388  // If a new target is available, then the speed at the target is computed from the angle previous-target-next.
389 
390  switch (_current_state) {
391 
392  case State::target_behind: {
393  _target = _triplet_target;
394  _prev_wp = _position;
395  _next_wp = _triplet_next_wp;
396  //_current_state = State::target_behind;
397 
398  float angle = 2.0f;
399  _speed_at_target = 0.0f;
400 
401  // angle = cos(x) + 1.0
402  // angle goes from 0 to 2 with 0 = large angle, 2 = small angle: 0 = PI ; 2 = PI*0
403 
404  if (Vector2f(&(_target - _next_wp)(0)).length() > 0.001f &&
405  (Vector2f(&(_target - _prev_wp)(0)).length() > NAV_ACC_RAD.get())) {
406 
407  angle = Vector2f(&(_target - _prev_wp)(0)).unit_or_zero()
408  * Vector2f(&(_target - _next_wp)(0)).unit_or_zero()
409  + 1.0f;
410  _speed_at_target = _getVelocityFromAngle(angle);
411  }
412  }
413  break;
414 
415  case State::previous_infront: {
416  _next_wp = _triplet_target;
417  _target = _triplet_prev_wp;
418  _prev_wp = _position;
419 
420  float angle = 2.0f;
421  _speed_at_target = 0.0f;
422 
423  // angle = cos(x) + 1.0
424  // angle goes from 0 to 2 with 0 = large angle, 2 = small angle: 0 = PI ; 2 = PI*0
425  if (Vector2f(&(_target - _next_wp)(0)).length() > 0.001f &&
426  (Vector2f(&(_target - _prev_wp)(0)).length() > NAV_ACC_RAD.get())) {
427 
428  angle = Vector2f(&(_target - _prev_wp)(0)).unit_or_zero()
429  * Vector2f(&(_target - _next_wp)(0)).unit_or_zero()
430  + 1.0f;
431  _speed_at_target = _getVelocityFromAngle(angle);
432  }
433  }
434  break;
435 
436  case State::offtrack: {
437  _next_wp = _triplet_target;
438  _target = matrix::Vector3f(_closest_pt(0), _closest_pt(1), _triplet_target(2));
439  _prev_wp = _position;
440 
441  float angle = 2.0f;
442  _speed_at_target = 0.0f;
443 
444  // angle = cos(x) + 1.0
445  // angle goes from 0 to 2 with 0 = large angle, 2 = small angle: 0 = PI ; 2 = PI*0
446  if (Vector2f(&(_target - _next_wp)(0)).length() > 0.001f &&
447  (Vector2f(&(_target - _prev_wp)(0)).length() > NAV_ACC_RAD.get())) {
448 
449  angle = Vector2f(&(_target - _prev_wp)(0)).unit_or_zero()
450  * Vector2f(&(_target - _next_wp)(0)).unit_or_zero()
451  + 1.0f;
452  _speed_at_target = _getVelocityFromAngle(angle);
453  }
454  }
455  break;
456 
457  case State::none: {
458  _target = _triplet_target;
459  _prev_wp = _triplet_prev_wp;
460  _next_wp = _triplet_next_wp;
461 
462  float angle = 2.0f;
463  _speed_at_target = 0.0f;
464 
465  // angle = cos(x) + 1.0
466  // angle goes from 0 to 2 with 0 = large angle, 2 = small angle: 0 = PI ; 2 = PI*0
467  if (Vector2f(&(_target - _next_wp)(0)).length() > 0.001f &&
468  (Vector2f(&(_target - _prev_wp)(0)).length() > NAV_ACC_RAD.get())) {
469 
470  angle =
471  Vector2f(&(_target - _prev_wp)(0)).unit_or_zero()

473  + 1.0f;
474  _speed_at_target = _getVelocityFromAngle(angle);

476 
477  break;
478  }
479 
480  default:
481  break;
482  }

485 bool FlightTaskAuto::_compute_heading_from_2D_vector(float &heading, matrix::Vector2f v)
486 {
487  if (PX4_ISFINITE(v.length()) && v.length() > SIGMA_NORM) {
488  v.normalize();
489  // To find yaw: take dot product of x = (1,0) and v
490  // and multiply by the sign given of cross product of x and v.
491  // Dot product: (x(0)*v(0)+(x(1)*v(1)) = v(0)
492  // Cross product: x(0)*v(1) - v(0)*x(1) = v(1)
493  heading = math::sign(v(1)) * wrap_pi(acosf(v(0)));
494  return true;
495  }
496 
497  // heading unknown and therefore do not change heading
498  return false;
499 }

502 float FlightTaskAuto::_getVelocityFromAngle(const float angle)
503 {
504  // minimum cruise speed when passing waypoint
505  float min_cruise_speed = 0.0f;
506 
507  // make sure that cruise speed is larger than minimum
508  if ((_mc_cruise_speed - min_cruise_speed) < SIGMA_NORM) {
509  return _mc_cruise_speed;
510  }
511 
512  // Middle cruise speed is a number between maximum cruising speed and minimum cruising speed and corresponds to speed at angle of 90degrees.
513  // It needs to be always larger than minimum cruise speed.
514  float middle_cruise_speed = MPC_CRUISE_90.get();
515 
516  if ((middle_cruise_speed - min_cruise_speed) < SIGMA_NORM) {
517  middle_cruise_speed = min_cruise_speed + SIGMA_NORM;
518  }
519 
520  if ((_mc_cruise_speed - middle_cruise_speed) < SIGMA_NORM) {
521  middle_cruise_speed = (_mc_cruise_speed + min_cruise_speed) * 0.5f;
522  }
523 
524  // If middle cruise speed is exactly in the middle, then compute speed linearly.
525  bool use_linear_approach = false;
526 
527  if (((_mc_cruise_speed + min_cruise_speed) * 0.5f) - middle_cruise_speed < SIGMA_NORM) {
528  use_linear_approach = true;
529  }
530 
531  // compute speed sp at target
532  float speed_close;
533 
534  if (use_linear_approach) {
535 
536  // velocity close to target adjusted to angle:
537  // vel_close = m*x+q
538  float slope = -(_mc_cruise_speed - min_cruise_speed) / 2.0f;
539  speed_close = slope * angle + _mc_cruise_speed;
540 
541  } else {
542 
543  // Speed close to target adjusted to angle x.
544  // speed_close = a *b ^x + c; where at angle x = 0 -> speed_close = cruise; angle x = 1 -> speed_close = middle_cruise_speed (this means that at 90degrees
545  // the velocity at target is middle_cruise_speed);
546  // angle x = 2 -> speed_close = min_cruising_speed
547 
548  // from maximum cruise speed, minimum cruise speed and middle cruise speed compute constants a, b and c
549  float a = -((middle_cruise_speed - _mc_cruise_speed) * (middle_cruise_speed - _mc_cruise_speed))
550  / (2.0f * middle_cruise_speed - _mc_cruise_speed - min_cruise_speed);
551  float c = _mc_cruise_speed - a;
552  float b = (middle_cruise_speed - c) / a;
553  speed_close = a * powf(b, angle) + c;
554  }
555 
556  // speed_close needs to be in between max and min
557  return math::constrain(speed_close, min_cruise_speed, _mc_cruise_speed);
558 }
