83 __EXPORT void board_spi_reset(int ms)
84 {
85  /* Goal not to back feed the chips on the bus via IO lines */
86 
87  /* First float the A_MISO line tied to SA0 to ensure SPI Auto selection*/
88 
89  kinetis_pinconfig(PX4_MK_GPIO(PIN_SPI1_SIN, GPIO_OPENDRAIN));
90  kinetis_gpiowrite(PIN_SPI1_SIN, 1);
91 
92  /* Next Change CS to inputs with pull downs */
93 
94  for (unsigned int cs = 0; cs < arraySize(spi1selects_gpio); cs++) {
95  if (spi1selects_gpio[cs] != 0) {
96  kinetis_pinconfig(PX4_MK_GPIO(spi1selects_gpio[cs], GPIO_PULLDOWN));
97  }
98  }
99 
100  /* Turn all the int inputs to inputs with pull down */
101 
102  kinetis_pinconfig(PX4_MK_GPIO(GPIO_EXTI_GYRO_INT1, GPIO_PULLDOWN));
103  kinetis_pinconfig(PX4_MK_GPIO(GPIO_EXTI_GYRO_INT2, GPIO_PULLDOWN));
104  kinetis_pinconfig(PX4_MK_GPIO(GPIO_EXTI_ACCEL_MAG_INT1, GPIO_PULLDOWN));
105  kinetis_pinconfig(PX4_MK_GPIO(GPIO_EXTI_ACCEL_MAG_INT2, GPIO_PULLDOWN));
106  kinetis_pinconfig(PX4_MK_GPIO(GPIO_EXTI_BARO_INT1, GPIO_PULLDOWN));
107  kinetis_pinconfig(PX4_MK_GPIO(GPIO_EXTI_BARO_INT2, GPIO_PULLDOWN));
108 
109  /* Drive the Reset Pins LOW
110  * For the Gyro FXAS21002C this is RESET
111  * for the Accel FXOS8700CQ this is not RESET */
112 
113  kinetis_gpiowrite(GPIO_GM_nRST, false);
114  kinetis_gpiowrite(GPIO_A_RST, false);
115 
116  /* Power Down The Sensors */
117 
118  VDD_3V3_SENSORS_EN(false);
119  up_mdelay(ms);
120 
121  /* Power Up The Sensors */
122  VDD_3V3_SENSORS_EN(true);
123  up_mdelay(2);
124 
125  /* Restore all the CS to ouputs inactive */
126 
127  for (unsigned int cs = 0; cs < arraySize(spi1selects_gpio); cs++) {
128  if (spi1selects_gpio[cs] != 0) {
129  kinetis_pinconfig(spi1selects_gpio[cs]);
130  }
131  }
132 
133  /* Restore all the int inputs to inputs */
134 
135  kinetis_pinconfig(GPIO_EXTI_GYRO_INT1);
136  kinetis_pinconfig(GPIO_EXTI_GYRO_INT2);
137  kinetis_pinconfig(GPIO_EXTI_ACCEL_MAG_INT1);
138  kinetis_pinconfig(GPIO_EXTI_ACCEL_MAG_INT2);
139  kinetis_pinconfig(GPIO_EXTI_BARO_INT1);
140  kinetis_pinconfig(GPIO_EXTI_BARO_INT2);
141 
142  /* Set Rests Active */
143 
144  /* Accel Assert Reset to reset the FXOS8700CQ */
145  /* The Gyro was reset above */
146 
147  kinetis_gpiowrite(GPIO_A_RST, true);
148  up_mdelay(ms);
149 
150  /* Accel & Gyro release Reset */
151 
152  kinetis_gpiowrite(GPIO_A_RST, false);
153  kinetis_gpiowrite(GPIO_GM_nRST, true);
154 
155  /* Allow the Accel time to see PIN_SPI1_SIN as a float */
156 
157  up_mdelay(2);
158 
159  kinetis_pinconfig(PIN_SPI1_SIN);
160 }

204 __EXPORT int nxphlite_spi_bus_initialize(void)
205 {
206  /* Configure SPI-based devices */
207 
208  spi_sensors = px4_spibus_initialize(PX4_SPI_BUS_SENSORS);
209 
210  if (!spi_sensors) {
211  message("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_SENSORS);
212  return -ENODEV;
213  }
214 
215  /* Default PX4_SPI_BUS_SENSORS to 1MHz and de-assert the known chip selects.
216  */
217 
218  SPI_SETFREQUENCY(spi_sensors, 1 * 1000 * 1000);
219  SPI_SETBITS(spi_sensors, 8);
220  SPI_SETMODE(spi_sensors, SPIDEV_MODE0);
221 
222  for (int cs = PX4_SENSOR_BUS_FIRST_CS; cs <= PX4_SENSOR_BUS_LAST_CS; cs++) {
223  SPI_SELECT(spi_sensors, cs, false);
224  }
225 
226  /* Get the SPI port for the Memory */
227 
228  spi_memory = px4_spibus_initialize(PX4_SPI_BUS_MEMORY);
229 
230  if (!spi_memory) {
231  message("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_MEMORY);
232  return -ENODEV;
233  }
234 
235  /* Default PX4_SPI_BUS_MEMORY to 12MHz and de-assert the known chip selects.
236  */
237 
238  SPI_SETFREQUENCY(spi_memory, 12 * 1000 * 1000);
239  SPI_SETBITS(spi_memory, 8);
240  SPI_SETMODE(spi_memory, SPIDEV_MODE3);
241 
242  for (int cs = PX4_MEMORY_BUS_FIRST_CS; cs <= PX4_MEMORY_BUS_LAST_CS; cs++) {
243  SPI_SELECT(spi_memory, cs, false);
244  }
245 
246  /* Configure EXTERNAL SPI-based devices */
247 
248  spi_ext = px4_spibus_initialize(PX4_SPI_BUS_EXTERNAL);
249 
250  if (!spi_ext) {
251  message("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_EXTERNAL);
252  return -ENODEV;
253  }
254 
255  /* Default PX4_SPI_BUS_SENSORS to 1MHz and de-assert the known chip selects.
256  */
257 
258  SPI_SETFREQUENCY(spi_ext, 8 * 1000 * 1000);
259  SPI_SETBITS(spi_ext, 8);
260  SPI_SETMODE(spi_ext, SPIDEV_MODE3);
261 
262  for (int cs = PX4_EXTERNAL_BUS_FIRST_CS; cs <= PX4_EXTERNAL_BUS_LAST_CS; cs++) {
263  SPI_SELECT(spi_ext, cs, false);
264  }
265 
266 
267  return OK;
268 
269 }

271 /************************************************************************************
272  * Name: kinetis_spi[n]select, kinetis_spi[n]status, and kinetis_spi[n]cmddata
273  *
274  * Description:
275  * These external functions must be provided by board-specific logic. They are
276  * implementations of the select, status, and cmddata methods of the SPI interface
277  * defined by struct spi_ops_s (see include/nuttx/spi/spi.h). All other methods
278  * including kinetis_spibus_initialize()) are provided by common Kinetis logic.
279  * To use this common SPI logic on your board:
280  *
281  * 1. Provide logic in kinetis_boardinitialize() to configure SPI chip select
282  * pins.
283  * 2. Provide kinetis_spi[n]select() and kinetis_spi[n]status() functions
284  * in your board-specific logic. These functions will perform chip selection
285  * and status operations using GPIOs in the way your board is configured.
286  * 2. If CONFIG_SPI_CMDDATA is defined in the NuttX configuration, provide
287  * kinetis_spi[n]cmddata() functions in your board-specific logic. These
288  * functions will perform cmd/data selection operations using GPIOs in the way
289  * your board is configured.
290  * 3. Add a call to kinetis_spibus_initialize() in your low level application
291  * initialization logic
292  * 4. The handle returned by kinetis_spibus_initialize() may then be used to bind the
293  * SPI driver to higher level logic (e.g., calling
294  * mmcsd_spislotinitialize(), for example, will bind the SPI driver to
295  * the SPI MMC/SD driver).
296  *
297  ************************************************************************************/

299 void kinetis_spi0select(FAR struct spi_dev_s *dev, uint32_t devid, bool selected)
300 {
301  spiinfo("devid: %d CS: %s\n", (int)devid, selected ? "assert" : "de-assert");
302 
303  /* SPI select is active low, so write !selected to select the device */
304 
305  uint32_t sel = devid;
306 
307  if (devid == SPIDEV_FLASH(0)) {
308  sel = PX4_SPIDEV_MEMORY;
309  }
310 
311  ASSERT(PX4_SPI_BUS_ID(sel) == PX4_SPI_BUS_MEMORY);
312 
313  /* Making sure the other peripherals are not selected */
314 
315  for (unsigned int cs = 0; arraySize(spi0selects_gpio) > 1 && cs < arraySize(spi0selects_gpio); cs++) {
316  if (spi0selects_gpio[cs] != 0) {
317  kinetis_gpiowrite(spi0selects_gpio[cs], 1);
318  }
319  }
320 
321  uint32_t gpio = spi0selects_gpio[PX4_SPI_DEV_ID(sel)];
322 
323  if (gpio) {
324  kinetis_gpiowrite(gpio, !selected);
325  }
326 }
