143  * The motor_ramp app only briefly exists to start
144  * the background job. The stack size assigned in the
145  * Makefile does only apply to this management task.
146  *
147  * The actual stack size should be set in the call
148  * to task_create().

150 int motor_ramp_main(int argc, char *argv[])
151 {
152  if (argc < 4) {
153  usage("missing parameters");
154  return 1;
155  }
156 
157  if (_thread_running) {
158  PX4_WARN("motor_ramp already running\n");
159  /* this is not an error */
160  return 0;
161  }
162 
163  if (!strcmp(argv[1], "ramp")) {
164  _mode = RAMP;
165 
166  } else if (!strcmp(argv[1], "sine")) {
167  _mode = SINE;
168 
169  } else if (!strcmp(argv[1], "square")) {
170  _mode = SQUARE;
171 
172  } else {
173  usage("selected mode not valid");
174  return 1;
175  }
176 
177  _mode_c = argv[1];
178 
179  _min_pwm = atoi(argv[2]);
180 
181  if (!min_pwm_valid(_min_pwm)) {
182  usage("min PWM not in range");
183  return 1;
184  }
185 
186  _ramp_time = atof(argv[3]);
187 
188  if (argc > 4) {
189  _max_pwm = atoi(argv[4]);
190 
191  if (!max_pwm_valid(_max_pwm)) {
192  usage("max PWM not in range");
193  return 1;
194  }
195 
196  } else {
197  _max_pwm = 2000;
198  }
199 
200  if (!(_ramp_time > 0)) {
201  usage("ramp time must be greater than 0");
202  return 1;
203  }
204 
205  _thread_should_exit = false;
206  _motor_ramp_task = px4_task_spawn_cmd("motor_ramp",
207  SCHED_DEFAULT,
208  SCHED_PRIORITY_DEFAULT + 40,
209  2000,
210  motor_ramp_thread_main,
211  (argv) ? (char *const *)&argv[2] : (char *const *)nullptr);
212  return 0;
213 }

226 {
227  int ret;
228 
229  struct pwm_output_values pwm_values;

266 int prepare(int fd, unsigned long *max_channels)
267 {
268  /* make sure no other source is publishing control values now */
269  struct actuator_controls_s actuators;
270  int act_sub = orb_subscribe(ORB_ID_VEHICLE_ATTITUDE_CONTROLS);
271 
272  /* clear changed flag */
273  orb_copy(ORB_ID_VEHICLE_ATTITUDE_CONTROLS, act_sub, &actuators);
274 
275  /* wait 50 ms */
276  usleep(50000);
277 
278  /* now expect nothing changed on that topic */
279  bool orb_updated;
280  orb_check(act_sub, &orb_updated);
281 
282  if (orb_updated) {
283  PX4_ERR("ABORTING! Attitude control still active. Please ensure to shut down all controllers:\n"
284  "\tmc_att_control stop\n"
285  "\tfw_att_control stop\n");
286  return 1;
287  }
288 
289  /* get number of channels available on the device */
290  if (px4_ioctl(fd, PWM_SERVO_GET_COUNT, (unsigned long)max_channels) != OK) {
291  PX4_ERR("PWM_SERVO_GET_COUNT");
292  return 1;
293  }
294 
295  /* tell IO/FMU that its ok to disable its safety with the switch */
296  if (px4_ioctl(fd, PWM_SERVO_SET_ARM_OK, 0) != OK) {
297  PX4_ERR("PWM_SERVO_SET_ARM_OK");
298  return 1;
299  }
300 
301  /* tell IO/FMU that the system is armed (it will output values if safety is off) */
302  if (px4_ioctl(fd, PWM_SERVO_ARM, 0) != OK) {
303  PX4_ERR("PWM_SERVO_ARM");
304  return 1;
305  }
306 
307  /* tell IO to switch off safety without using the safety switch */
308  if (px4_ioctl(fd, PWM_SERVO_SET_FORCE_SAFETY_OFF, 0) != OK) {
309  PX4_ERR("PWM_SERVO_SET_FORCE_SAFETY_OFF");
310  return 1;
311  }
312 
313  return 0;
314 }

316 int motor_ramp_thread_main(int argc, char *argv[])
317 {
318  _thread_running = true;
319 
320  char *dev = PWM_OUTPUT0_DEVICE_PATH;
321  unsigned long max_channels = 0;
322 
323  int fd = px4_open(dev, 0);
324 
325  if (fd < 0) {
326  PX4_ERR("can't open %s", dev);
327  }
328 
329  if (prepare(fd, &max_channels) != OK) {
330  _thread_should_exit = true;
331  }
332 
333  set_out(fd, max_channels, 0.0f);
334 
335  float dt = 0.001f; // prevent division with 0
336  float timer = 0.0f;
337  hrt_abstime start = 0;
338  hrt_abstime last_run = 0;
339 
340  enum RampState ramp_state = RAMP_INIT;
341  float output = 0.0f;
342 
343  while (!_thread_should_exit) {
344 
345  if (last_run > 0) {
346  dt = hrt_elapsed_time(&last_run) * 1e-6;
347 
348  } else {
349  start = hrt_absolute_time();
350  }
351 
352  last_run = hrt_absolute_time();
353  timer = hrt_elapsed_time(&start) * 1e-6;
354 
355  switch (ramp_state) {
356  case RAMP_INIT: {
357  PX4_INFO("setting pwm min: %d", _min_pwm);
358  set_min_pwm(fd, max_channels, _min_pwm);
359  ramp_state = RAMP_MIN;
360  break;
361  }
362 
363  case RAMP_MIN: {
364  if (timer > 3.0f) {
365  PX4_INFO("starting %s: %.2f sec", _mode_c, (double)_ramp_time);
366  start = hrt_absolute_time();
367  ramp_state = RAMP_RAMP;
368  }
369 
370  set_out(fd, max_channels, output);
371  break;
372  }
373 
374  case RAMP_RAMP: {
375  if (_mode == RAMP) {
376  output += 1000.0f * dt / (_max_pwm - _min_pwm) / _ramp_time;
377 
378  } else if (_mode == SINE) {
379  // sine outpout with period T = _ramp_time and magnitude between [0,1]
380  // phase shift makes sure that it starts at zero when timer is zero
381  output = 0.5f * (1.0f + sinf(M_TWOPI_F * timer / _ramp_time - M_PI_2_F));
382 
383  } else if (_mode == SQUARE) {
384  output = fmodf(timer, _ramp_time) > (_ramp_time * 0.5f) ? 1.0f : 0.0f;
385  }
386 
387  if ((output > 1.0f && _mode == RAMP) || (timer > 3.0f * _ramp_time)) {
388  // for ramp mode we set output to 1, for others we just leave it as is
389  output = _mode != RAMP ? output : 1.0f;
390  start = hrt_absolute_time();
391  ramp_state = RAMP_WAIT;
392  PX4_INFO("%s finished, waiting", _mode_c);
393  }
394 
395  set_out(fd, max_channels, output);
396  break;
397  }
398 
399  case RAMP_WAIT: {
400  if (timer > 0.5f) {
401  _thread_should_exit = true;
402  PX4_INFO("stopping");
403  break;
404  }
405 
406  set_out(fd, max_channels, output);
407  break;
408  }
409  }
410 
411  // rate limit
412  usleep(2000);
413  }
414 
415  if (fd >= 0) {
416  /* disarm */
417  ioctl(fd, PWM_SERVO_DISARM, 0);
418 
419  px4_close(fd);
420  }
421 
422  _thread_running = false;
423 
424  return 0;
425 }
