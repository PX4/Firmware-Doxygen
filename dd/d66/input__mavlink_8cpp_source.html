83 int InputMavlinkROI::update_impl(unsigned int timeout_ms, ControlData **control_data, bool already_active)
84 {
85  // already_active is unused, we don't care what happened previously.
86 
87  // Default to no change, set if we receive anything.
88  *control_data = nullptr;
89 
90  const int num_poll = 2;
91  px4_pollfd_struct_t polls[num_poll];
92  polls[0].fd = _vehicle_roi_sub;
93  polls[0].events = POLLIN;
94  polls[1].fd = _position_setpoint_triplet_sub;
95  polls[1].events = POLLIN;
96 
97  int ret = px4_poll(polls, num_poll, timeout_ms);
98 
99  if (ret < 0) {
100  return -errno;
101  }
102 
103  if (ret == 0) {
104  // Timeout, _control_data is already null
105 
106  } else {
107  if (polls[0].revents & POLLIN) {
108  vehicle_roi_s vehicle_roi;
109  orb_copy(ORB_ID(vehicle_roi), _vehicle_roi_sub, &vehicle_roi);
110 
111  _control_data.gimbal_shutter_retract = false;
112 
113  if (vehicle_roi.mode == vehicle_roi_s::ROI_NONE) {
114 
115  _control_data.type = ControlData::Type::Neutral;
116  *control_data = &_control_data;
117 
118  } else if (vehicle_roi.mode == vehicle_roi_s::ROI_WPNEXT) {
119  _control_data.type = ControlData::Type::LonLat;
120  _read_control_data_from_position_setpoint_sub();
121  _control_data.type_data.lonlat.pitch_fixed_angle = -10.f;
122 
123  _control_data.type_data.lonlat.roll_angle = vehicle_roi.roll_offset;
124  _control_data.type_data.lonlat.pitch_angle_offset = vehicle_roi.pitch_offset;
125  _control_data.type_data.lonlat.yaw_angle_offset = vehicle_roi.yaw_offset;
126 
127  *control_data = &_control_data;
128 
129  } else if (vehicle_roi.mode == vehicle_roi_s::ROI_LOCATION) {
130  control_data_set_lon_lat(vehicle_roi.lon, vehicle_roi.lat, vehicle_roi.alt);
131 
132  *control_data = &_control_data;
133 
134  } else if (vehicle_roi.mode == vehicle_roi_s::ROI_TARGET) {
135  //TODO is this even suported?
136  }
137 
138  _cur_roi_mode = vehicle_roi.mode;
139 
140  //set all other control data fields to defaults
141  for (int i = 0; i < 3; ++i) {
142  _control_data.stabilize_axis[i] = false;
143  }
144  }
145 
146  // check whether the position setpoint got updated
147  if (polls[1].revents & POLLIN) {
148  if (_cur_roi_mode == vehicle_roi_s::ROI_WPNEXT) {
149  _read_control_data_from_position_setpoint_sub();
150  *control_data = &_control_data;
151 
152  } else { // must do an orb_copy() in *every* case
153  position_setpoint_triplet_s position_setpoint_triplet;
154  orb_copy(ORB_ID(position_setpoint_triplet), _position_setpoint_triplet_sub, &position_setpoint_triplet);
155  }
156  }
157  }
158 
159  return 0;
160 }

200 int InputMavlinkCmdMount::initialize()
201 {
202  if ((_vehicle_command_sub = orb_subscribe(ORB_ID(vehicle_command))) < 0) {
203  return -errno;
204  }
205 
206  // rate-limit inputs to 100Hz. If we don't do this and the output is configured to mavlink mode,
207  // it will publish vehicle_command's as well, causing the input poll() in here to return
208  // immediately, which in turn will cause an output update and thus a busy loop.
209  orb_set_interval(_vehicle_command_sub, 10);
210 
211  return 0;
212 }

215 int InputMavlinkCmdMount::update_impl(unsigned int timeout_ms, ControlData **control_data, bool already_active)
216 {
217  // Default to notify that there was no change.
218  *control_data = nullptr;
219 
220  const int num_poll = 1;
221  px4_pollfd_struct_t polls[num_poll];
222  polls[0].fd = _vehicle_command_sub;
223  polls[0].events = POLLIN;
224 
225  int poll_timeout = (int)timeout_ms;
226 
227  bool exit_loop = false;
228 
229  while (!exit_loop && poll_timeout >= 0) {
230  hrt_abstime poll_start = hrt_absolute_time();
231 
232  int ret = px4_poll(polls, num_poll, poll_timeout);
233 
234  if (ret < 0) {
235  return -errno;
236  }
237 
238  poll_timeout -= (hrt_absolute_time() - poll_start) / 1000;
239 
240  // if we get a command that we need to handle, we exit the loop, otherwise we poll until we reach the timeout
241  exit_loop = true;
242 
243  if (ret == 0) {
244  // Timeout control_data already null.
245 
246  } else {
247  if (polls[0].revents & POLLIN) {
248  vehicle_command_s vehicle_command;
249  orb_copy(ORB_ID(vehicle_command), _vehicle_command_sub, &vehicle_command);
250 
251  // Process only if the command is for us or for anyone (component id 0).
252  const bool sysid_correct = (vehicle_command.target_system == _mav_sys_id);
253  const bool compid_correct = ((vehicle_command.target_component == _mav_comp_id) ||
254  (vehicle_command.target_component == 0));
255 
256  if (!sysid_correct || !compid_correct) {
257  exit_loop = false;
258  continue;
259  }
260 
261  for (int i = 0; i < 3; ++i) {
262  _control_data.stabilize_axis[i] = _stabilize[i];
263  }
264 
265  _control_data.gimbal_shutter_retract = false;
266 
267  if (vehicle_command.command == vehicle_command_s::VEHICLE_CMD_DO_MOUNT_CONTROL) {
268 
269  switch ((int)vehicle_command.param7) {
270  case vehicle_command_s::VEHICLE_MOUNT_MODE_RETRACT:
271  _control_data.gimbal_shutter_retract = true;
272 
273  /* FALLTHROUGH */
274 
275  case vehicle_command_s::VEHICLE_MOUNT_MODE_NEUTRAL:
276  _control_data.type = ControlData::Type::Neutral;
277 
278  *control_data = &_control_data;
279  break;
280 
281  case vehicle_command_s::VEHICLE_MOUNT_MODE_MAVLINK_TARGETING:
282  _control_data.type = ControlData::Type::Angle;
283  _control_data.type_data.angle.is_speed[0] = false;
284  _control_data.type_data.angle.is_speed[1] = false;
285  _control_data.type_data.angle.is_speed[2] = false;
286  // vmount spec has roll on channel 0, MAVLink spec has pitch on channel 0
287  _control_data.type_data.angle.angles[0] = vehicle_command.param2 * M_DEG_TO_RAD_F;
288  // vmount spec has pitch on channel 1, MAVLink spec has roll on channel 1
289  _control_data.type_data.angle.angles[1] = vehicle_command.param1 * M_DEG_TO_RAD_F;
290  // both specs have yaw on channel 2
291  _control_data.type_data.angle.angles[2] = vehicle_command.param3 * M_DEG_TO_RAD_F;
292 
293  // We expect angle of [-pi..+pi]. If the input range is [0..2pi] we can fix that.
294  if (_control_data.type_data.angle.angles[2] > M_PI_F) {
295  _control_data.type_data.angle.angles[2] -= 2 * M_PI_F;
296  }
297 
298  *control_data = &_control_data;
299  break;
300 
301  case vehicle_command_s::VEHICLE_MOUNT_MODE_RC_TARGETING:
302  break;
303 
304  case vehicle_command_s::VEHICLE_MOUNT_MODE_GPS_POINT:
305  control_data_set_lon_lat((double)vehicle_command.param2, (double)vehicle_command.param1, vehicle_command.param3);
306 
307  *control_data = &_control_data;
308  break;
309  }
310 
311  _ack_vehicle_command(&vehicle_command);
312 
313  } else if (vehicle_command.command == vehicle_command_s::VEHICLE_CMD_DO_MOUNT_CONFIGURE) {
314  _stabilize[0] = (uint8_t) vehicle_command.param2 == 1;
315  _stabilize[1] = (uint8_t) vehicle_command.param3 == 1;
316  _stabilize[2] = (uint8_t) vehicle_command.param4 == 1;
317  _control_data.type = ControlData::Type::Neutral; //always switch to neutral position
318 
319  *control_data = &_control_data;
320  _ack_vehicle_command(&vehicle_command);
321 
322  } else {
323  exit_loop = false;
324  }
325  }
326 
327  }
328  }
329 
330  return 0;
331 }

333 void InputMavlinkCmdMount::_ack_vehicle_command(vehicle_command_s *cmd)
334 {
335  vehicle_command_ack_s vehicle_command_ack = {};
336 
337  vehicle_command_ack.timestamp = hrt_absolute_time();
338  vehicle_command_ack.command = cmd->command;
339  vehicle_command_ack.result = vehicle_command_s::VEHICLE_CMD_RESULT_ACCEPTED;
340  vehicle_command_ack.target_system = cmd->source_system;
341  vehicle_command_ack.target_component = cmd->source_component;
342 
343  if (_vehicle_command_ack_pub == nullptr) {
344  _vehicle_command_ack_pub = orb_advertise_queue(ORB_ID(vehicle_command_ack), &vehicle_command_ack,
345  vehicle_command_ack_s::ORB_QUEUE_LENGTH);
346 
347  } else {
348  orb_publish(ORB_ID(vehicle_command_ack), _vehicle_command_ack_pub, &vehicle_command_ack);
