109 __EXPORT int stm32_spi_bus_initialize(void)
110 {
111  /* Configure SPI-based devices */
112 
113 
114  /* Get the SPI port for the Sensors */
115  spi_sensor1 = stm32_spibus_initialize(PX4_SPI_BUS_SENSOR1);
116 
117  if (!spi_sensor1) {
118  PX4_ERR("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_SENSOR1);
119  return -ENODEV;
120  }
121 
122  /* Default PX4_SPI_BUS_SENSORS to 1MHz and de-assert the known chip selects. */
123  SPI_SETFREQUENCY(spi_sensor1, 1000000);
124  SPI_SETBITS(spi_sensor1, 16);
125  SPI_SETMODE(spi_sensor1, SPIDEV_MODE3);
126 
127  for (int cs = PX4_SENSOR1_BUS_FIRST_CS; cs <= PX4_SENSOR1_BUS_LAST_CS; cs++) {
128  SPI_SELECT(spi_sensor1, cs, false);
129  }
130 
131 
132  /* Get the SPI port for the Memory */
133  spi_sensor2 = stm32_spibus_initialize(PX4_SPI_BUS_SENSOR2);
134 
135  if (!spi_sensor2) {
136  PX4_ERR("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_SENSOR2);
137  return -ENODEV;
138  }
139 
140  /* Default PX4_SPI_BUS_SENSOR2 to 1MHz and de-assert the known chip selects. */
141  SPI_SETFREQUENCY(spi_sensor2, 1000000);
142  SPI_SETBITS(spi_sensor2, 8);
143  SPI_SETMODE(spi_sensor2, SPIDEV_MODE3);
144 
145  for (int cs = PX4_SENSOR2_BUS_FIRST_CS; cs <= PX4_SENSOR2_BUS_LAST_CS; cs++) {
146  SPI_SELECT(spi_sensor2, cs, false);
147  }
148 
149 
150  /* Get the SPI port for the BARO */
151  spi_sensor4 = stm32_spibus_initialize(PX4_SPI_BUS_SENSOR4);
152 
153  if (!spi_sensor4) {
154  PX4_ERR("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_SENSOR4);
155  return -ENODEV;
156  }
157 
158  SPI_SETFREQUENCY(spi_sensor4, 1 * 1000 * 1000);
159  SPI_SETBITS(spi_sensor4, 8);
160  SPI_SETMODE(spi_sensor4, SPIDEV_MODE3);
161 
162  for (int cs = PX4_SENSOR4_BUS_FIRST_CS; cs <= PX4_SENSOR4_BUS_LAST_CS; cs++) {
163  SPI_SELECT(spi_sensor4, cs, false);
164  }
165 
166 
167  /* Get the SPI port for the PX4_SPI_EXTERNAL1 */
168  spi_sensor5 = stm32_spibus_initialize(PX4_SPI_BUS_SENSOR5);
169 
170  if (!spi_sensor5) {
171  PX4_ERR("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_SENSOR5);
172  return -ENODEV;
173  }
174 
175  SPI_SETFREQUENCY(spi_sensor5, 1 * 1000 * 1000);
176  SPI_SETBITS(spi_sensor5, 8);
177  SPI_SETMODE(spi_sensor5, SPIDEV_MODE3);
178 
179  for (int cs = PX4_SENSOR5_BUS_FIRST_CS; cs <= PX4_SENSOR5_BUS_LAST_CS; cs++) {
180  SPI_SELECT(spi_sensor5, cs, false);
181  }
182 
183  return OK;

194 __EXPORT void stm32_spi1select(FAR struct spi_dev_s *dev, uint32_t devid, bool selected)
195 {
196  /* SPI select is active low, so write !selected to select the device */
197  int sel = (int) devid;
198  ASSERT(PX4_SPI_BUS_ID(sel) == PX4_SPI_BUS_SENSOR1);
199 
200  /* Making sure the other peripherals are not selected */
201  for (size_t cs = 0; arraySize(spi1selects_gpio) > 1 && cs < arraySize(spi1selects_gpio); cs++) {
202  if (spi1selects_gpio[cs] != 0) {
203  stm32_gpiowrite(spi1selects_gpio[cs], 1);
204  }
205  }
206 
207  uint32_t gpio = spi1selects_gpio[PX4_SPI_DEV_ID(sel)];
208 
209  if (gpio) {
210  stm32_gpiowrite(gpio, !selected);
211  }
212 }

325 __EXPORT void board_spi_reset(int ms)
326 {
327  /* disable SPI bus */
328 
329  // SPI1
330  for (size_t cs = 0; arraySize(spi1selects_gpio) > 1 && cs < arraySize(spi1selects_gpio); cs++) {
331  if (spi1selects_gpio[cs] != 0) {
332  stm32_configgpio(_PIN_OFF(spi1selects_gpio[cs]));
333  }
334  }
335 
336  stm32_configgpio(GPIO_SPI1_SCK_OFF);
337  stm32_configgpio(GPIO_SPI1_MISO_OFF);
338  stm32_configgpio(GPIO_SPI1_MOSI_OFF);
339 
340  // SPI2
341  for (size_t cs = 0; arraySize(spi2selects_gpio) > 1 && cs < arraySize(spi2selects_gpio); cs++) {
342  if (spi2selects_gpio[cs] != 0) {
343  stm32_configgpio(_PIN_OFF(spi2selects_gpio[cs]));
344  }
345  }
346 
347  stm32_configgpio(GPIO_SPI2_SCK_OFF);
348  stm32_configgpio(GPIO_SPI2_MISO_OFF);
349  stm32_configgpio(GPIO_SPI2_MOSI_OFF);
350 
351  // SPI4
352  for (size_t cs = 0; arraySize(spi4selects_gpio) > 1 && cs < arraySize(spi4selects_gpio); cs++) {
353  if (spi4selects_gpio[cs] != 0) {
354  stm32_configgpio(_PIN_OFF(spi4selects_gpio[cs]));
355  }
356  }
357 
358  stm32_configgpio(GPIO_SPI4_SCK_OFF);

372 
373

375  usleep(ms * 1000);
376  PX4_INFO("reset done, %d ms", ms);

389  }
390  }
