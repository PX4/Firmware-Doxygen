974 {
975  switch (cmd) {
976 
977  case SENSORIOCRESET:
978  return reset();
979 
980  case SENSORIOCSPOLLRATE: {
981  switch (arg) {
982 
983  /* switching to manual polling */
984  case SENSOR_POLLRATE_MANUAL:
985  stop();
986  _call_interval = 0;
987  return OK;
988 
989  /* external signalling not supported */
990  case SENSOR_POLLRATE_EXTERNAL:
991 
992  /* zero would be bad */
993  case 0:
994  return -EINVAL;
995 
996  /* set default/max polling rate */
997  case SENSOR_POLLRATE_MAX:
998  return ioctl(filp, SENSORIOCSPOLLRATE, 1000);
999 
1000  case SENSOR_POLLRATE_DEFAULT:
1001  return ioctl(filp, SENSORIOCSPOLLRATE, ADIS16448_ACCEL_DEFAULT_RATE);
1002 
1003  /* adjust to a legal polling interval in Hz */
1004  default: {
1005  /* do we need to start internal polling? */
1006  bool want_start = (_call_interval == 0);
1007 
1008  /* convert hz to hrt interval via microseconds */
1009  unsigned ticks = 1000000 / arg;
1010 
1011  /* check against maximum sane rate */
1012  if (ticks < 1000) {
1013  return -EINVAL;
1014  }
1015 
1016  // adjust filters
1017  float cutoff_freq_hz = _accel_filter_x.get_cutoff_freq();
1018  float sample_rate = 1.0e6f / ticks;
1019  _accel_filter_x.set_cutoff_frequency(sample_rate, cutoff_freq_hz);
1020  _accel_filter_y.set_cutoff_frequency(sample_rate, cutoff_freq_hz);
1021  _accel_filter_z.set_cutoff_frequency(sample_rate, cutoff_freq_hz);
1022

1026  _gyro_filter_y.set_cutoff_frequency(sample_rate, cutoff_freq_hz_gyro);
1027  _gyro_filter_z.set_cutoff_frequency(sample_rate, cutoff_freq_hz_gyro);
1028 
1029 
1030  float cutoff_freq_hz_mag = _mag_filter_x.get_cutoff_freq();
1031  _mag_filter_x.set_cutoff_frequency(sample_rate, cutoff_freq_hz_mag);
1032  _mag_filter_y.set_cutoff_frequency(sample_rate, cutoff_freq_hz_mag);
1033  _mag_filter_z.set_cutoff_frequency(sample_rate, cutoff_freq_hz_mag);
1034 
1035  /* update interval for next measurement */
1036  /* XXX this is a bit shady, but no other way to adjust... */
1037  _call.period = _call_interval = ticks;
1038 
1039  /* if we need to start the poll state machine, do it */
1040  if (want_start) {
1041  start();
1042  }
1043 
1044  return OK;
1045  }
1046  }
1047  }
1048 
1049  case SENSORIOCGPOLLRATE:
1050  if (_call_interval == 0) {
1051  return SENSOR_POLLRATE_MANUAL;
1052  }
1053 
1054  return 1000000 / _call_interval;
1055 
1056  case SENSORIOCSQUEUEDEPTH: {
1057  /* lower bound is mandatory, upper bound is a sanity check */
1058  if ((arg < 1) || (arg > 100)) {
1059  return -EINVAL;
1060  }
1061 
1062  irqstate_t flags = px4_enter_critical_section();
1063 
1064  if (!_accel_reports->resize(arg)) {
1065  px4_leave_critical_section(flags);
1066  return -ENOMEM;
1067  }
1068 
1069  px4_leave_critical_section(flags);
1070 
1071  return OK;
1072  }
1073 
1074  case ACCELIOCGSAMPLERATE:
1075  return _sample_rate;
1076 
1077  case ACCELIOCSSAMPLERATE:
1078  _set_sample_rate(arg);
1079  return OK;
1080 
1081  case ACCELIOCSSCALE: {
1082  /* copy scale, but only if off by a few percent */
1083  struct accel_calibration_s *s = (struct accel_calibration_s *) arg;
1084  float sum = s->x_scale + s->y_scale + s->z_scale;
1085 
1086  if (sum > 2.0f && sum < 4.0f) {
1087  memcpy(&_accel_scale, s, sizeof(_accel_scale));
1088  return OK;
1089 
1090  } else {
1091  return -EINVAL;
1092  }
1093  }
1094 
1095  case ACCELIOCGSCALE:
1096  /* copy scale out */
1097  memcpy((struct accel_calibration_s *) arg, &_accel_scale, sizeof(_accel_scale));
1098  return OK;
1099 
1100  case ACCELIOCSRANGE:
1101  return -EINVAL;
1102 
1103  case ACCELIOCGRANGE:
1104  return (unsigned long)((_accel_range_m_s2) / CONSTANTS_ONE_G + 0.5f);
1105 
1106  case ACCELIOCTYPE:
1107  return (ADIS16448_Product);
1108 
1109  default:
1110  /* give it to the superclass */
1111  return SPI::ioctl(filp, cmd, arg);

1328

1331  int16_t gyro_y;
1332  int16_t gyro_z;
1333  int16_t accel_x;
1334  int16_t accel_y;
1335  int16_t accel_z;
1336  int16_t mag_x;
1337  int16_t mag_y;
1338  int16_t mag_z;
1339  int16_t baro;
1340  int16_t temp;
1341  } report;
1342 
1343  /* start measuring */
1344  perf_begin(_sample_perf);
1345 
1346  /*
1347  * Fetch the full set of measurements from the ADIS16448 in one pass (burst read).
1348  */
1349 
1350  adis_report.cmd = ((ADIS16448_GLOB_CMD | DIR_READ) << 8) & 0xff00;
1351 
1352  if (OK != transferhword((uint16_t *)&adis_report, ((uint16_t *)&adis_report), sizeof(adis_report) / sizeof(uint16_t))) {
1353  return -EIO;
1354  }
1355 
1356  report.gyro_x = (int16_t) adis_report.gyro_x;
1357  report.gyro_y = (int16_t) adis_report.gyro_y;
1358  report.gyro_z = (int16_t) adis_report.gyro_z;
1359  report.accel_x = (int16_t) adis_report.accel_x;
1360  report.accel_y = (int16_t) adis_report.accel_y;
1361  report.accel_z = (int16_t) adis_report.accel_z;
1362  report.mag_x = (int16_t) adis_report.mag_x;
1363  report.mag_y = (int16_t) adis_report.mag_y;
1364  report.mag_z = (int16_t) adis_report.mag_z;
1365  report.baro = (int16_t) adis_report.baro;
1366  report.temp = convert12BitToINT16(adis_report.temp);
1367 
1368  if (report.gyro_x == 0 && report.gyro_y == 0 && report.gyro_z == 0 &&
1369  report.accel_x == 0 && report.accel_y == 0 && report.accel_z == 0 &&
1370  report.mag_x == 0 && report.mag_y == 0 && report.mag_z == 0 &&
1371  report.baro == 0 && report.temp == 0) {
1372  perf_count(_bad_transfers);
1373  perf_end(_sample_perf);
1374  return -EIO;
1375  }
1376 
1377  /*
1378  * Report buffers.
1379  */
1380  accel_report arb;
1381  gyro_report grb;
1382  mag_report mrb;
1383 
1384  grb.timestamp = arb.timestamp = mrb.timestamp = hrt_absolute_time();
1385  grb.error_count = arb.error_count = mrb.error_count = perf_event_count(_bad_transfers);
1386 
1387  /* Gyro report: */
1388  grb.x_raw = report.gyro_x;
1389  grb.y_raw = report.gyro_y;
1390  grb.z_raw = report.gyro_z;
1391

1393  float yraw_f = report.gyro_y;
1394  float zraw_f = report.gyro_z;
1395 
1396  // apply user specified rotation
1397  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
1398 
1399  float x_gyro_in_new = ((xraw_f * _gyro_range_scale) * M_PI_F / 180.0f - _gyro_scale.x_offset) * _gyro_scale.x_scale;
1400  float y_gyro_in_new = ((yraw_f * _gyro_range_scale) * M_PI_F / 180.0f - _gyro_scale.y_offset) * _gyro_scale.y_scale;
1401  float z_gyro_in_new = ((zraw_f * _gyro_range_scale) * M_PI_F / 180.0f - _gyro_scale.z_offset) * _gyro_scale.z_scale;
1402 
1403  if (FW_FILTER) {
1404  grb.x = _gyro_filter_x.apply(x_gyro_in_new);
1405  grb.y = _gyro_filter_y.apply(y_gyro_in_new);
1406  grb.z = _gyro_filter_z.apply(z_gyro_in_new);
1407 
1408  } else {
1409  grb.x = x_gyro_in_new;
1410  grb.y = y_gyro_in_new;
1411  grb.z = z_gyro_in_new;
1412  }
1413 
1414  grb.scaling = _gyro_range_scale * M_PI_F / 180.0f;
1415 
1416  /* Accel report: */
1417  arb.x_raw = report.accel_x;
1418  arb.y_raw = report.accel_y;
1419  arb.z_raw = report.accel_z;
1420 
1421  xraw_f = report.accel_x;
1422  yraw_f = report.accel_y;
1423  zraw_f = report.accel_z;
1424 
1425  // apply user specified rotation
1426  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
1427

1429  float y_in_new = ((yraw_f * _accel_range_scale) - _accel_scale.y_offset) * _accel_scale.y_scale;
1430  float z_in_new = ((zraw_f * _accel_range_scale) - _accel_scale.z_offset) * _accel_scale.z_scale;
1431 
1432  if (FW_FILTER) {
1433  arb.x = _accel_filter_x.apply(x_in_new);
1434  arb.y = _accel_filter_y.apply(y_in_new);
1435  arb.z = _accel_filter_z.apply(z_in_new);
1436 
1437  } else {
1438  arb.x = x_in_new;
1439  arb.y = y_in_new;
1440  arb.z = z_in_new;
1441  }
1442 
1443  arb.scaling = _accel_range_scale;
1444

1446  mrb.x_raw = report.mag_x;
1447  mrb.y_raw = report.mag_y;
1448  mrb.z_raw = report.mag_z;
1449 
1450  xraw_f = report.mag_x;
1451  yraw_f = report.mag_y;
1452  zraw_f = report.mag_z;
1453 
1454  // apply user specified rotation
1455  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
1456

1458  float y_mag_new = ((yraw_f * _mag_range_scale) - _mag_scale.y_offset) * _mag_scale.y_scale;
1459  float z_mag_new = ((zraw_f * _mag_range_scale) - _mag_scale.z_offset) * _mag_scale.z_scale;
1460 
1461  if (FW_FILTER) {
1462  mrb.x = _mag_filter_x.apply(x_mag_new) / 1000.0f;
1463  mrb.y = _mag_filter_y.apply(y_mag_new) / 1000.0f;
1464  mrb.z = _mag_filter_z.apply(z_mag_new) / 1000.0f;
1465 
1466  } else {
1467  mrb.x = x_mag_new / 1000.0f;
1468  mrb.y = y_mag_new / 1000.0f;
1469  mrb.z = z_mag_new / 1000.0f;
1470  }
1471 
1472  mrb.scaling = _mag_range_scale / 1000.0f;
1473 
1474  /* Temperature report: */
1475  grb.temperature = (report.temp * 0.07386f) + 31.0f;
1476  arb.temperature = grb.temperature;
1477 
1478  matrix::Vector3f aval(x_in_new, y_in_new, z_in_new);
1479  matrix::Vector3f aval_integrated;
1480 
1481  bool accel_notify = _accel_int.put(arb.timestamp, aval, aval_integrated, arb.integral_dt);
1482  arb.x_integral = aval_integrated(0);
1483  arb.y_integral = aval_integrated(1);
1484  arb.z_integral = aval_integrated(2);
1485 
1486  matrix::Vector3f gval(x_gyro_in_new, y_gyro_in_new, z_gyro_in_new);
1487  matrix::Vector3f gval_integrated;
1488 
1489  bool gyro_notify = _gyro_int.put(grb.timestamp, gval, gval_integrated, grb.integral_dt);
1490  grb.x_integral = gval_integrated(0);
1491  grb.y_integral = gval_integrated(1);
1492  grb.z_integral = gval_integrated(2);
1493 
1494  /* return device ID */
1495  arb.device_id = _device_id.devid;
1496  grb.device_id = _gyro->_device_id.devid;
1497  mrb.device_id = _mag->_device_id.devid;
1498 
1499  _gyro_reports ->force(&grb);
1500  _accel_reports->force(&arb);
1501  _mag_reports ->force(&mrb);
1502 
1503  /* notify anyone waiting for data */
1504  if (accel_notify) {
1505  poll_notify(POLLIN);
1506  }
1507 
1508  if (gyro_notify) {
1509  _gyro->parent_poll_notify();
1510  }
1511 
1512  _mag->parent_poll_notify();
1513 
1514  if (accel_notify && !(_pub_blocked)) {
1515  /* publish it */
1516  orb_publish(ORB_ID(sensor_accel), _accel_topic, &arb);
1517  }
1518 
1519  if (gyro_notify && !(_pub_blocked)) {
1520  /* publish it */
1521  orb_publish(ORB_ID(sensor_gyro), _gyro->_gyro_topic, &grb);
1522  }
1523 
1524  if (!(_pub_blocked) && ((adis_report.status >> 7) & 0x1)) { /* Mag data validity bit (bit 8 DIAG_STAT) */
1525  /* publish it */
1526  orb_publish(ORB_ID(sensor_mag), _mag->_mag_topic, &mrb);
1527  }
1528 
1529  /* stop measuring */
1530  perf_end(_sample_perf);
1531  return OK;
