158 /************************************************************************************
159  * Name: board_on_reset
160  *
161  * Description:
162  * Optionally provided function called on entry to board_system_reset
163  * It should perform any house keeping prior to the rest.
164  *
165  * status - 1 if resetting to boot loader
166  * 0 if just resetting
167  *
168  ************************************************************************************/
169 __EXPORT void board_on_reset(int status)
170 {
171  UNUSED(status);
172  /* configure the GPIO pins to outputs and keep them low */
173 
174  stm32_configgpio(GPIO_GPIO0_OUTPUT);
175  stm32_configgpio(GPIO_GPIO1_OUTPUT);
176  stm32_configgpio(GPIO_GPIO2_OUTPUT);
177  stm32_configgpio(GPIO_GPIO3_OUTPUT);
178  stm32_configgpio(GPIO_GPIO4_OUTPUT);
179  stm32_configgpio(GPIO_GPIO5_OUTPUT);
180 
181  /* On resets invoked from system (not boot) insure we establish a low
182  * output state (discharge the pins) on PWM pins before they become inputs.
183  *
184  * We also delay the onset of the that 3.1 Ms pulse as boot. This has
185  * triggered some ESC to spin. By adding this delay here the reset
186  * is pushed out > 400 ms. So the ESC PWM input can not mistake
187  * the 3.1 Ms pulse as a valid PWM command.
188  *
189  * fixme:Establish in upstream NuttX an CONFIG_IO_INIT_STATE to
190  * the initialize the IO lines in the clock config.
191  *
192  */
193 
194  if (status >= 0) {
195  up_mdelay(400);
196  }
197 }
