74 /* The MPU6000 can only handle high SPI bus speeds of 20Mhz on the sensor and
75 * interrupt status registers. All other registers have a maximum 1MHz
76 * SPI speed
77 *
78 * The ICM parts are not rated as high.
79 *
80 * The Actual Value will be rounded down by the spi driver.
81 * 168 Mhz CPU 180 Mhz CPU
82 * Selected ------------actual---------------
83 * 20 Mhz 10.5 Mhz 11.250 Mhz
84 * 10 Mhz 5.250 Mhz 5.625 Mhz
85 * 8 Mhz 5.250 Mhz 5.625 Mhz
86 * 1 Mhz 0.703125 Mhz 0.65625 Mhz
87 *
88 */

90 #define MPU6000_HIGH_SPI_BUS_SPEED 20*1000*1000

270 int
271 MPU6000_SPI::write(unsigned reg_speed, void *data, unsigned count)
272 {
273  uint8_t cmd[MPU_MAX_WRITE_BUFFER_SIZE];
274 
275  if (sizeof(cmd) < (count + 1)) {
276  return -EIO;
277  }
278 
279  /* Set the desired speed and isolate the register */
280 
281  set_bus_frequency(reg_speed);
282 
283  cmd[0] = reg_speed | DIR_WRITE;
284  cmd[1] = *(uint8_t *)data;
285 
286  return transfer(&cmd[0], &cmd[0], count + 1);
287 }

289 int
290 MPU6000_SPI::read(unsigned reg_speed, void *data, unsigned count)
291 {
292  /* We want to avoid copying the data of MPUReport: So if the caller
293  * supplies a buffer not MPUReport in size, it is assume to be a reg or reg 16 read
294  * and we need to provied the buffer large enough for the callers data
295  * and our command.
296  */
297  uint8_t cmd[3] = {0, 0, 0};
298 
299  uint8_t *pbuff = count < sizeof(MPUReport) ? cmd : (uint8_t *) data ;
300 
301 
302  if (count < sizeof(MPUReport)) {
303 
304  /* add command */
305 
306  count++;
307  }
308 
309  set_bus_frequency(reg_speed);
310 
311  /* Set command */
312 
313  pbuff[0] = reg_speed | DIR_READ ;
314 
315  /* Transfer the command and get the data */
316 
317  int ret = transfer(pbuff, pbuff, count);
318 
319  if (ret == OK && pbuff == &cmd[0]) {
320 
321  /* Adjust the count back */
322 
323  count--;
324 
325  /* Return the data */
326 
327  memcpy(data, &cmd[1], count);
328 
329  }
330 
331  return ret == OK ? count : ret;
332 }
