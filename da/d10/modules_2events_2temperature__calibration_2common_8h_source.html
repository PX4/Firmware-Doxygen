118 template <int Dim, int PolyfitOrder>
119 class TemperatureCalibrationCommon : public TemperatureCalibrationBase
120 {
121 public:
122  TemperatureCalibrationCommon(float min_temperature_rise, float min_start_temperature, float max_start_temperature)
123  : TemperatureCalibrationBase(min_temperature_rise, min_start_temperature, max_start_temperature) {}
124 
125  virtual ~TemperatureCalibrationCommon() = default;
126 
127  /**
128  * @see TemperatureCalibrationBase::update()
129  */
130  int update()
131  {
132  int num_not_complete = 0;
133 
134  if (_num_sensor_instances == 0) {
135  return -TC_ERROR_COMMUNICATION;
136  }
137 
138  for (unsigned uorb_index = 0; uorb_index < _num_sensor_instances; uorb_index++) {
139  int status = update_sensor_instance(_data[uorb_index], _sensor_subs[uorb_index]);
140 
141  if (status < 0) {
142  return status;
143  }
144 
145  num_not_complete += status;
146  }
147 
148  if (num_not_complete > 0) {
149  // calculate progress
150  float min_diff = _min_temperature_rise;
151 
152  for (unsigned uorb_index = 0; uorb_index < _num_sensor_instances; uorb_index++) {
153  float cur_diff = _data[uorb_index].high_temp - _data[uorb_index].low_temp;
154 
155  if (cur_diff < min_diff) {
156  min_diff = cur_diff;
157  }
158  }
159 
160  return math::min(100, (int)(min_diff / _min_temperature_rise * 100.f));
161  }
162 
163  return 110;
164  }
