105 int VideoDevice::open_device()
106 {
107  struct stat st;
108 
109  // Check if device is usable
110  int ret = stat(_dev_name, &st);
111 
112  if (ret < 0) {
113  PX4_ERR("Cannot identify %s: %d", _dev_name, errno);
114  return -EIO;
115  }
116 
117  if (!S_ISCHR(st.st_mode)) {
118  PX4_ERR("%s is no device", _dev_name);
119  return -EIO;
120  }
121 
122  // Open V4L2 device nonblocking
123  _fd = open(_dev_name, O_RDWR | O_NONBLOCK);
124 
125  if (_fd < 0) {
126  PX4_ERR("Unable to open %s", _dev_name);
127  return -EIO;
128  }
129 
130  return OK;
131 }

145 int VideoDevice::init_device()
146 {
147  struct v4l2_capability cap;
148  memset(&cap, 0, sizeof(cap));
149 
150  int ret = ioctl(_fd, VIDIOC_QUERYCAP, &cap);
151 
152  if (ret < 0) {
153  if (EINVAL == errno) {
154  PX4_ERR("No V4L2 device: %s", _dev_name);
155  return -EINVAL;
156 
157  } else {
158  PX4_ERR("VIDIOC_QUERYCAP failed: %s", _dev_name);
159  return -1;
160  }
161  }
162 
163  if (!(cap.capabilities & V4L2_CAP_VIDEO_CAPTURE)) {
164  PX4_ERR("Device is no video capture device: %s", _dev_name);
165  return -EIO;
166  }
167 
168  ret = init_crop();
169 
170  if (ret < 0) {
171  PX4_ERR("Error when setting image crop");
172  return -1;
173  }
174 
175  return init_buffers();
176 }

252 int VideoDevice::init_buffers()
253 {
254  struct v4l2_requestbuffers req;
255 
256  memset(&req, 0, sizeof(req));
257 
258  req.count = _n_buffers;
259  req.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
260  req.memory = V4L2_MEMORY_MMAP;
261 
262  int ret = ioctl(_fd, VIDIOC_REQBUFS, &req);
263 
264  if (ret < 0) {
265  PX4_ERR("Unable to request buffers: %s", _dev_name);
266  return -1;
267  }
268 
269  _buffers = (struct buffer *) malloc(_n_buffers * sizeof(_buffers[0]));
270 
271  if (_buffers == nullptr) {
272  PX4_ERR("Unable to allocate buffers");
273  return -1;
274  }
275 
276  for (unsigned int i = 0; i < _n_buffers; ++i) {
277  struct v4l2_buffer buf;
278  memset(&buf, 0, sizeof(buf));
279 
280  buf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
281  buf.memory = V4L2_MEMORY_MMAP;
282  buf.index = i;
283 
284  ret = ioctl(_fd, VIDIOC_QUERYBUF, &buf);
285 
286  if (ret < 0) {
287  PX4_ERR("Error QUERYBUF");
288  return -1;
289  }
290 
291  _buffers[i].length = buf.length;
292  _buffers[i].start = mmap(nullptr, buf.length, PROT_READ | PROT_WRITE,
293  MAP_SHARED, _fd, buf.m.offset);
294 
295  if (_buffers[i].start == MAP_FAILED) {
296  PX4_ERR("Out of memory");
297  return -1;
298  }
299  }
300 
301  return OK;
302 }

353 
354  buf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
355  buf.memory = V4L2_MEMORY_MMAP;
356 
357  int ret = ioctl(_fd, VIDIOC_DQBUF, &buf);

359  if (ret < 0) {

367

369  PX4_ERR("Buffer deque error");

372  }
373 
374  frame.data = _buffers[buf.index].start;
375  frame.seq = buf.sequence;
376  frame.timestamp = buf.timestamp.tv_sec * 1000000 + buf.timestamp.tv_usec;
377  frame.bytes = buf.bytesused;
