315 void
316 MavlinkMissionManager::send_mission_item(uint8_t sysid, uint8_t compid, uint16_t seq)
317 {
318  mission_item_s mission_item = {};
319  bool read_success = false;
320 
321  switch (_mission_type) {
322 
323  case MAV_MISSION_TYPE_MISSION: {
324  read_success = dm_read(_dataman_id, seq, &mission_item, sizeof(mission_item_s)) == sizeof(mission_item_s);
325  }
326  break;
327 
328  case MAV_MISSION_TYPE_FENCE: { // Read a geofence point
329  mission_fence_point_s mission_fence_point;
330  read_success = dm_read(DM_KEY_FENCE_POINTS, seq + 1, &mission_fence_point, sizeof(mission_fence_point_s)) ==
331  sizeof(mission_fence_point_s);
332 
333  mission_item.nav_cmd = mission_fence_point.nav_cmd;
334  mission_item.frame = mission_fence_point.frame;
335  mission_item.lat = mission_fence_point.lat;
336  mission_item.lon = mission_fence_point.lon;
337  mission_item.altitude = mission_fence_point.alt;
338 
339  if (mission_fence_point.nav_cmd == MAV_CMD_NAV_FENCE_POLYGON_VERTEX_INCLUSION ||
340  mission_fence_point.nav_cmd == MAV_CMD_NAV_FENCE_POLYGON_VERTEX_EXCLUSION) {
341  mission_item.vertex_count = mission_fence_point.vertex_count;
342 
343  } else {
344  mission_item.circle_radius = mission_fence_point.circle_radius;
345  }
346  }
347  break;
348 
349  case MAV_MISSION_TYPE_RALLY: { // Read a safe point / rally point
350  mission_save_point_s mission_save_point;
351  read_success = dm_read(DM_KEY_SAFE_POINTS, seq + 1, &mission_save_point, sizeof(mission_save_point_s)) ==
352  sizeof(mission_save_point_s);
353 
354  mission_item.nav_cmd = MAV_CMD_NAV_RALLY_POINT;
355  mission_item.frame = mission_save_point.frame;
356  mission_item.lat = mission_save_point.lat;
357  mission_item.lon = mission_save_point.lon;
358  mission_item.altitude = mission_save_point.alt;
359  }
360  break;
361 
362  default:
363  _mavlink->send_statustext_critical("Received unknown mission type, abort.");
364  break;
365  }
366 
367  if (read_success) {
368  _time_last_sent = hrt_absolute_time();
369 
370  if (_int_mode) {
371  mavlink_mission_item_int_t wp;
372  format_mavlink_mission_item(&mission_item, reinterpret_cast<mavlink_mission_item_t *>(&wp));
373 
374  wp.target_system = sysid;
375  wp.target_component = compid;
376  wp.seq = seq;
377  wp.current = (_current_seq == seq) ? 1 : 0;
378 
379  mavlink_msg_mission_item_int_send_struct(_mavlink->get_channel(), &wp);
380 
381  PX4_DEBUG("WPM: Send MISSION_ITEM_INT seq %u to ID %u", wp.seq, wp.target_system);
382 
383  } else {
384  mavlink_mission_item_t wp;
385  format_mavlink_mission_item(&mission_item, &wp);
386 
387  wp.target_system = sysid;
388  wp.target_component = compid;
389  wp.seq = seq;
390  wp.current = (_current_seq == seq) ? 1 : 0;
391 
392  mavlink_msg_mission_item_send_struct(_mavlink->get_channel(), &wp);
393 
394  PX4_DEBUG("WPM: Send MISSION_ITEM seq %u to ID %u", wp.seq, wp.target_system);
395  }
396 
397  } else {
398  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ERROR);
399 
400  if (_filesystem_errcount++ < FILESYSTEM_ERRCOUNT_NOTIFY_LIMIT) {
401  _mavlink->send_statustext_critical("Mission storage: Unable to read from microSD");
402  }
403 
404  PX4_DEBUG("WPM: Send MISSION_ITEM ERROR: could not read seq %u from dataman ID %i", seq, _dataman_id);
405  }

480 void
481 MavlinkMissionManager::send(const hrt_abstime now)
482 {
483  // do not send anything over high latency communication
484  if (_mavlink->get_mode() == Mavlink::MAVLINK_MODE_IRIDIUM) {
485  return;
486  }
487 
488  bool updated = false;
489  orb_check(_mission_result_sub, &updated);
490 
491  if (updated) {
492  mission_result_s mission_result;
493  orb_copy(ORB_ID(mission_result), _mission_result_sub, &mission_result);
494 
495  if (_current_seq != mission_result.seq_current) {
496  _current_seq = mission_result.seq_current;
497 
498  PX4_DEBUG("WPM: got mission result, new current_seq: %u", _current_seq);
499  }
500 
501  if (_last_reached != mission_result.seq_reached) {
502  _last_reached = mission_result.seq_reached;
503  _reached_sent_count = 0;
504 
505  if (_last_reached >= 0) {
506  send_mission_item_reached((uint16_t)mission_result.seq_reached);
507  }
508 
509  PX4_DEBUG("WPM: got mission result, new seq_reached: %d", _last_reached);
510  }
511 
512  send_mission_current(_current_seq);
513 
514  if (mission_result.item_do_jump_changed) {
515  /* send a mission item again if the remaining DO_JUMPs has changed */
516  send_mission_item(_transfer_partner_sysid, _transfer_partner_compid,
517  (uint16_t)mission_result.item_changed_index);
518  }
519 
520  } else {
521  if (_slow_rate_limiter.check(now)) {
522  send_mission_current(_current_seq);
523 
524  // send the reached message another 10 times
525  if (_last_reached >= 0 && (_reached_sent_count < 10)) {
526  send_mission_item_reached((uint16_t)_last_reached);
527  _reached_sent_count++;
528  }
529  }
530  }
531 
532  /* check for timed-out operations */
533  if (_state == MAVLINK_WPM_STATE_GETLIST && (_time_last_sent > 0)
534  && hrt_elapsed_time(&_time_last_sent) > MAVLINK_MISSION_RETRY_TIMEOUT_DEFAULT) {
535 
536  // try to request item again after timeout
537  send_mission_request(_transfer_partner_sysid, _transfer_partner_compid, _transfer_seq);
538 
539  } else if (_state != MAVLINK_WPM_STATE_IDLE && (_time_last_recv > 0)
540  && hrt_elapsed_time(&_time_last_recv) > MAVLINK_MISSION_PROTOCOL_TIMEOUT_DEFAULT) {
541 
542  _mavlink->send_statustext_critical("Operation timeout");
543 
544  PX4_DEBUG("WPM: Last operation (state=%u) timed out, changing state to MAVLINK_WPM_STATE_IDLE", _state);
545 
546  switch_to_idle_state();
547 
548  // since we are giving up, reset this state also, so another request can be started.
549  _transfer_in_progress = false;
550 
551  } else if (_state == MAVLINK_WPM_STATE_IDLE) {
552  // reset flags
553  _time_last_sent = 0;
554  _time_last_recv = 0;
555  }
556 }

607 {

609  mavlink_msg_mission_ack_decode(msg, &wpa);
610 
611  if (CHECK_SYSID_COMPID_MISSION(wpa)) {
612  if ((msg->sysid == _transfer_partner_sysid && msg->compid == _transfer_partner_compid)) {
613  if (_state == MAVLINK_WPM_STATE_SENDLIST && _mission_type == wpa.mission_type) {
614  _time_last_recv = hrt_absolute_time();
615 
616  if (_transfer_seq == current_item_count()) {
617  PX4_DEBUG("WPM: MISSION_ACK OK all items sent, switch to state IDLE");
618 
619  } else {
620  _mavlink->send_statustext_critical("WPM: ERR: not all items sent -> IDLE");
621 
622  PX4_DEBUG("WPM: MISSION_ACK ERROR: not all items sent, switch to state IDLE anyway");
623  }
624 
625  switch_to_idle_state();
626 
627  } else if (_state == MAVLINK_WPM_STATE_GETLIST) {
628 
629  // INT or float mode is not supported
630  if (wpa.type == MAV_MISSION_UNSUPPORTED) {

767  * are basically the same, so we can ignore it. */
768  mavlink_mission_request_t wpr;

770

774 
775  if (_mission_type != wpr.mission_type) {

778  }

780  _time_last_recv = hrt_absolute_time();
781

791  } else {

793  PX4_DEBUG("WPM: MISSION_ITEM_REQUEST(_INT) ERROR: seq %u from ID %u unexpected, must be %i or %i", wpr.seq, msg->sysid,
794  _transfer_seq - 1, _transfer_seq);
795 
796  } else if (_transfer_seq <= 0) {
797  PX4_DEBUG("WPM: MISSION_ITEM_REQUEST(_INT) ERROR: seq %u from ID %u unexpected, must be %i", wpr.seq, msg->sysid,
798  _transfer_seq);
799 
800  } else {
801  PX4_DEBUG("WPM: MISSION_ITEM_REQUEST(_INT) ERROR: seq %u from ID %u unexpected, must be %i", wpr.seq, msg->sysid,
802  _transfer_seq - 1);
803  }
804 
805  switch_to_idle_state();
806 
807  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ERROR);
808  _mavlink->send_statustext_critical("WPM: REJ. CMD: Req. WP was unexpected");

813  if (wpr.seq < current_item_count()) {

815 
816  } else {

818  current_item_count() - 1);
819 
820  switch_to_idle_state();
821 
822  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ERROR);
823  _mavlink->send_statustext_critical("WPM: REJ. CMD: Req. WP was unexpected");
824  }
825 
826  } else if (_state == MAVLINK_WPM_STATE_IDLE) {
827  PX4_DEBUG("WPM: MISSION_ITEM_REQUEST(_INT) ERROR: no transfer");
828 
829  // Silently ignore this as some OSDs have buggy mission protocol implementations
830  //_mavlink->send_statustext_critical("IGN MISSION_ITEM_REQUEST(_INT): No active transfer");
831 
832  } else {
833  PX4_DEBUG("WPM: MISSION_ITEM_REQUEST(_INT) ERROR: busy (state %d).", _state);

847 void
848 MavlinkMissionManager::handle_mission_count(const mavlink_message_t *msg)
849 {
850  mavlink_mission_count_t wpc;
851  mavlink_msg_mission_count_decode(msg, &wpc);
852 
853  if (CHECK_SYSID_COMPID_MISSION(wpc)) {
854  if (_state == MAVLINK_WPM_STATE_IDLE) {
855  _time_last_recv = hrt_absolute_time();
856 
857  if (_transfer_in_progress) {
858  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ERROR);
859  return;
860  }
861 
862  _transfer_in_progress = true;
863  _mission_type = (MAV_MISSION_TYPE)wpc.mission_type;
864 
865  if (wpc.count > current_max_item_count()) {
866  PX4_DEBUG("WPM: MISSION_COUNT ERROR: too many waypoints (%d), supported: %d", wpc.count, current_max_item_count());
867 
868  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_NO_SPACE);
869  _transfer_in_progress = false;
870  return;
871  }
872 
873  if (wpc.count == 0) {
874  PX4_DEBUG("WPM: MISSION_COUNT 0, clearing waypoints list and staying in state MAVLINK_WPM_STATE_IDLE");
875 
876  switch (_mission_type) {
877  case MAV_MISSION_TYPE_MISSION:
878 
879  /* alternate dataman ID anyway to let navigator know about changes */
880 
881  if (_dataman_id == DM_KEY_WAYPOINTS_OFFBOARD_0) {
882  update_active_mission(DM_KEY_WAYPOINTS_OFFBOARD_1, 0, 0);
883 
884  } else {
885  update_active_mission(DM_KEY_WAYPOINTS_OFFBOARD_0, 0, 0);
886  }
887 
888  break;
889 
890  case MAV_MISSION_TYPE_FENCE:
891  update_geofence_count(0);
892  break;
893 
894  case MAV_MISSION_TYPE_RALLY:
895  update_safepoint_count(0);
896  break;
897 
898  default:
899  PX4_ERR("mission type %u not handled", _mission_type);
900  break;
901  }
902 
903  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ACCEPTED);
904  _transfer_in_progress = false;
905  return;
906  }
907 
908  PX4_DEBUG("WPM: MISSION_COUNT %u from ID %u, changing state to MAVLINK_WPM_STATE_GETLIST", wpc.count, msg->sysid);
909 
910  _state = MAVLINK_WPM_STATE_GETLIST;
911  _transfer_seq = 0;
912  _transfer_partner_sysid = msg->sysid;
913  _transfer_partner_compid = msg->compid;
914  _transfer_count = wpc.count;
915  _transfer_dataman_id = (_dataman_id == DM_KEY_WAYPOINTS_OFFBOARD_0 ? DM_KEY_WAYPOINTS_OFFBOARD_1 :
916  DM_KEY_WAYPOINTS_OFFBOARD_0); // use inactive storage for transmission
917  _transfer_current_seq = -1;
918 
919  if (_mission_type == MAV_MISSION_TYPE_FENCE) {
920  // We're about to write new geofence items, so take the lock. It will be released when
921  // switching back to idle
922  PX4_DEBUG("locking fence dataman items");
923 
924  int ret = dm_lock(DM_KEY_FENCE_POINTS);
925 
926  if (ret == 0) {
927  _geofence_locked = true;
928 
929  } else {
930  PX4_ERR("locking failed (%i)", errno);
931  }
932  }
933 
934  } else if (_state == MAVLINK_WPM_STATE_GETLIST) {
935  _time_last_recv = hrt_absolute_time();
936 
937  if (_transfer_seq == 0) {
938  /* looks like our MISSION_REQUEST was lost, try again */
939  PX4_DEBUG("WPM: MISSION_COUNT %u from ID %u (again)", wpc.count, msg->sysid);
940 
941  } else {
942  PX4_DEBUG("WPM: MISSION_COUNT ERROR: busy, already receiving seq %u", _transfer_seq);
943 
944  _mavlink->send_statustext_critical("WPM: REJ. CMD: Busy");
945 
946  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ERROR);
947  return;
948  }
949 
950  } else {
951  PX4_DEBUG("WPM: MISSION_COUNT ERROR: busy, state %i", _state);
952 
953  _mavlink->send_statustext_critical("WPM: IGN MISSION_COUNT: Busy");
954  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ERROR);
955  return;
956  }
957 
958  send_mission_request(_transfer_partner_sysid, _transfer_partner_compid, _transfer_seq);
959  }

1000 void
1001 MavlinkMissionManager::handle_mission_item_both(const mavlink_message_t *msg)
1002 {
1003 
1004  // The mavlink_message could also contain a mavlink_mission_item_int_t. We ignore that here
1005  // and take care of it later in parse_mavlink_mission_item depending on _int_mode.
1006 
1007  mavlink_mission_item_t wp;
1008  mavlink_msg_mission_item_decode(msg, &wp);
1009 
1010  if (CHECK_SYSID_COMPID_MISSION(wp)) {
1011 
1012  if (wp.mission_type != _mission_type) {
1013  PX4_WARN("WPM: Unexpected mission type (%u %u)", (int)wp.mission_type, (int)_mission_type);
1014  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ERROR);
1015  return;
1016  }
1017 
1018  if (_state == MAVLINK_WPM_STATE_GETLIST) {
1019  _time_last_recv = hrt_absolute_time();
1020 
1021  if (wp.seq != _transfer_seq) {
1022  PX4_DEBUG("WPM: MISSION_ITEM ERROR: seq %u was not the expected %u", wp.seq, _transfer_seq);
1023 
1024  /* request next item again */
1025  send_mission_request(_transfer_partner_sysid, _transfer_partner_compid, _transfer_seq);
1026  return;
1027  }
1028 
1029  } else if (_state == MAVLINK_WPM_STATE_IDLE) {
1030  if (_transfer_seq == wp.seq + 1) {
1031  // Assume this is a duplicate, where we already successfully got all mission items,
1032  // but the GCS did not receive the last ack and sent the same item again
1033  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ACCEPTED);
1034 
1035  } else {
1036  PX4_DEBUG("WPM: MISSION_ITEM ERROR: no transfer");
1037 
1038  _mavlink->send_statustext_critical("IGN MISSION_ITEM: No transfer");
1039  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ERROR);
1040  }
1041 
1042  return;
1043 
1044  } else {
1045  PX4_DEBUG("WPM: MISSION_ITEM ERROR: busy, state %i", _state);
1046 
1047  _mavlink->send_statustext_critical("IGN MISSION_ITEM: Busy");
1048  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ERROR);
1049  return;
1050  }
1051 
1052  struct mission_item_s mission_item = {};
1053 
1054  int ret = parse_mavlink_mission_item(&wp, &mission_item);
1055 
1056  if (ret != PX4_OK) {
1057  PX4_DEBUG("WPM: MISSION_ITEM ERROR: seq %u invalid item", wp.seq);
1058 
1059  _mavlink->send_statustext_critical("IGN MISSION_ITEM: Busy");
1060 
1061  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, ret);
1062  switch_to_idle_state();
1063  _transfer_in_progress = false;
1064  return;
1065  }
1066 
1067  bool write_failed = false;
1068  bool check_failed = false;
1069 
1070  switch (_mission_type) {
1071 
1072  case MAV_MISSION_TYPE_MISSION: {
1073  // check that we don't get a wrong item (hardening against wrong client implementations, the list here
1074  // does not need to be complete)
1075  if (mission_item.nav_cmd == MAV_CMD_NAV_FENCE_POLYGON_VERTEX_INCLUSION ||
1076  mission_item.nav_cmd == MAV_CMD_NAV_FENCE_POLYGON_VERTEX_EXCLUSION ||
1077  mission_item.nav_cmd == MAV_CMD_NAV_FENCE_CIRCLE_INCLUSION ||
1078  mission_item.nav_cmd == MAV_CMD_NAV_FENCE_CIRCLE_EXCLUSION ||
1079  mission_item.nav_cmd == MAV_CMD_NAV_RALLY_POINT) {
1080  check_failed = true;
1081 
1082  } else {
1083  dm_item_t dm_item = _transfer_dataman_id;
1084 
1085  write_failed = dm_write(dm_item, wp.seq, DM_PERSIST_POWER_ON_RESET, &mission_item,
1086  sizeof(struct mission_item_s)) != sizeof(struct mission_item_s);
1087 
1088  if (!write_failed) {
1089  /* waypoint marked as current */
1090  if (wp.current) {
1091  _transfer_current_seq = wp.seq;
1092  }
1093  }
1094  }
1095  }
1096  break;
1097 
1098  case MAV_MISSION_TYPE_FENCE: { // Write a geofence point
1099  mission_fence_point_s mission_fence_point;
1100  mission_fence_point.nav_cmd = mission_item.nav_cmd;
1101  mission_fence_point.lat = mission_item.lat;
1102  mission_fence_point.lon = mission_item.lon;
1103  mission_fence_point.alt = mission_item.altitude;
1104 
1105  if (mission_item.nav_cmd == MAV_CMD_NAV_FENCE_POLYGON_VERTEX_INCLUSION ||
1106  mission_item.nav_cmd == MAV_CMD_NAV_FENCE_POLYGON_VERTEX_EXCLUSION) {
1107  mission_fence_point.vertex_count = mission_item.vertex_count;
1108 
1109  if (mission_item.vertex_count < 3) { // feasibility check
1110  PX4_ERR("Fence: too few vertices");
1111  check_failed = true;
1112  update_geofence_count(0);
1113  }
1114 
1115  } else {
1116  mission_fence_point.circle_radius = mission_item.circle_radius;
1117  }
1118 
1119  mission_fence_point.frame = mission_item.frame;
1120 
1121  if (!check_failed) {
1122  write_failed = dm_write(DM_KEY_FENCE_POINTS, wp.seq + 1, DM_PERSIST_POWER_ON_RESET, &mission_fence_point,
1123  sizeof(mission_fence_point_s)) != sizeof(mission_fence_point_s);
1124  }
1125 
1126  }
1127  break;
1128 
1129  case MAV_MISSION_TYPE_RALLY: { // Write a safe point / rally point
1130  mission_save_point_s mission_save_point;
1131  mission_save_point.lat = mission_item.lat;
1132  mission_save_point.lon = mission_item.lon;
1133  mission_save_point.alt = mission_item.altitude;
1134  mission_save_point.frame = mission_item.frame;
1135  write_failed = dm_write(DM_KEY_SAFE_POINTS, wp.seq + 1, DM_PERSIST_POWER_ON_RESET, &mission_save_point,
1136  sizeof(mission_save_point_s)) != sizeof(mission_save_point_s);
1137  }
1138  break;
1139 
1140  default:
1141  _mavlink->send_statustext_critical("Received unknown mission type, abort.");
1142  break;
1143  }
1144 
1145  if (write_failed || check_failed) {
1146  PX4_DEBUG("WPM: MISSION_ITEM ERROR: error writing seq %u to dataman ID %i", wp.seq, _transfer_dataman_id);
1147 
1148  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ERROR);
1149 
1150  if (write_failed) {
1151  _mavlink->send_statustext_critical("Unable to write on micro SD");
1152  }
1153 
1154  switch_to_idle_state();
1155  _transfer_in_progress = false;
1156  return;
1157  }
1158 
1159  /* waypoint marked as current */
1160  if (wp.current) {
1161  _transfer_current_seq = wp.seq;
1162  }
1163 
1164  PX4_DEBUG("WPM: MISSION_ITEM seq %u received", wp.seq);
1165 
1166  _transfer_seq = wp.seq + 1;
1167 
1168  if (_transfer_seq == _transfer_count) {
1169  /* got all new mission items successfully */
1170  PX4_DEBUG("WPM: MISSION_ITEM got all %u items, current_seq=%u, changing state to MAVLINK_WPM_STATE_IDLE",
1171  _transfer_count, _transfer_current_seq);
1172 
1173  ret = 0;
1174 
1175  switch (_mission_type) {
1176  case MAV_MISSION_TYPE_MISSION:
1177  ret = update_active_mission(_transfer_dataman_id, _transfer_count, _transfer_current_seq);
1178  break;
1179 
1180  case MAV_MISSION_TYPE_FENCE:
1181  ret = update_geofence_count(_transfer_count);
1182  break;
1183 
1184  case MAV_MISSION_TYPE_RALLY:
1185  ret = update_safepoint_count(_transfer_count);
1186  break;
1187 
1188  default:
1189  PX4_ERR("mission type %u not handled", _mission_type);
1190  break;
1191  }
1192 
1193  // Note: the switch to idle needs to happen after update_geofence_count is called, for proper unlocking order
1194  switch_to_idle_state();
1195 
1196 
1197  if (ret == PX4_OK) {
1198  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ACCEPTED);
1199 
1200  } else {
1201  send_mission_ack(_transfer_partner_sysid, _transfer_partner_compid, MAV_MISSION_ERROR);
1202  }
1203 
1204  _transfer_in_progress = false;
1205 
1206  } else {
1207  /* request next item */
1208  send_mission_request(_transfer_partner_sysid, _transfer_partner_compid, _transfer_seq);
1209  }
1210  }
