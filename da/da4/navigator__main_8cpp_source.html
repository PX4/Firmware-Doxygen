179 Navigator::run()
180 {
181  bool have_geofence_position_data = false;
182 
183  /* Try to load the geofence:
184  * if /fs/microsd/etc/geofence.txt load from this file */
185  struct stat buffer;
186 
187  if (stat(GEOFENCE_FILENAME, &buffer) == 0) {
188  PX4_INFO("Loading geofence from %s", GEOFENCE_FILENAME);
189  _geofence.loadFromFile(GEOFENCE_FILENAME);
190  }
191 
192  /* do subscriptions */
193  _global_pos_sub = orb_subscribe(ORB_ID(vehicle_global_position));
194  _local_pos_sub = orb_subscribe(ORB_ID(vehicle_local_position));
195  _gps_pos_sub = orb_subscribe(ORB_ID(vehicle_gps_position));
196  _fw_pos_ctrl_status_sub = orb_subscribe(ORB_ID(fw_pos_ctrl_status));
197  _vstatus_sub = orb_subscribe(ORB_ID(vehicle_status));
198  _land_detected_sub = orb_subscribe(ORB_ID(vehicle_land_detected));
199  _home_pos_sub = orb_subscribe(ORB_ID(home_position));
200  _offboard_mission_sub = orb_subscribe(ORB_ID(mission));
201  _param_update_sub = orb_subscribe(ORB_ID(parameter_update));
202  _vehicle_command_sub = orb_subscribe(ORB_ID(vehicle_command));
203  _traffic_sub = orb_subscribe(ORB_ID(transponder_report));
204 
205  /* copy all topics first time */
206  vehicle_status_update();
207  vehicle_land_detected_update();
208  global_position_update();
209  local_position_update();
210  gps_position_update();
211  home_position_update(true);
212  fw_pos_ctrl_status_update(true);
213  params_update();
214 
215  /* wakeup source(s) */
216  px4_pollfd_struct_t fds[1] = {};
217 
218  /* Setup of loop */
219  fds[0].fd = _local_pos_sub;
220  fds[0].events = POLLIN;
221 
222  /* rate-limit position subscription to 20 Hz / 50 ms */
223  orb_set_interval(_local_pos_sub, 50);
224 
225  hrt_abstime last_geofence_check = 0;
226 
227  while (!should_exit()) {
228 
229  /* wait for up to 1000ms for data */
230  int pret = px4_poll(&fds[0], (sizeof(fds) / sizeof(fds[0])), 1000);
231 
232  if (pret == 0) {
233  /* Let the loop run anyway, don't do `continue` here. */
234 
235  } else if (pret < 0) {
236  /* this is undesirable but not much we can do - might want to flag unhappy status */
237  PX4_ERR("poll error %d, %d", pret, errno);
238  usleep(10000);
239  continue;
240 
241  } else {
242  if (fds[0].revents & POLLIN) {
243  /* success, local pos is available */
244  local_position_update();
245  }
246  }
247 
248  perf_begin(_loop_perf);
249 
250  bool updated;
251 
252  /* gps updated */
253  orb_check(_gps_pos_sub, &updated);
254 
255  if (updated) {
256  gps_position_update();
257 
258  if (_geofence.getSource() == Geofence::GF_SOURCE_GPS) {
259  have_geofence_position_data = true;
260  }
261  }
262 
263  /* global position updated */
264  orb_check(_global_pos_sub, &updated);
265 
266  if (updated) {
267  global_position_update();
268 
269  if (_geofence.getSource() == Geofence::GF_SOURCE_GLOBALPOS) {
270  have_geofence_position_data = true;
271  }
272  }
273 
274  /* parameters updated */
275  orb_check(_param_update_sub, &updated);
276 
277  if (updated) {
278  params_update();
279  }
280 
281  /* vehicle status updated */
282  orb_check(_vstatus_sub, &updated);
283 
284  if (updated) {
285  vehicle_status_update();
286  }
287 
288  /* vehicle land detected updated */
289  orb_check(_land_detected_sub, &updated);
290 
291  if (updated) {
292  vehicle_land_detected_update();
293  }
294 
295  /* navigation capabilities updated */
296  orb_check(_fw_pos_ctrl_status_sub, &updated);
297 
298  if (updated) {
299  fw_pos_ctrl_status_update();
300  }
301 
302  /* home position updated */
303  orb_check(_home_pos_sub, &updated);
304 
305  if (updated) {
306  home_position_update();
307  }
308 
309  /* vehicle_command updated */
310  orb_check(_vehicle_command_sub, &updated);
311 
312  if (updated) {
313  vehicle_command_s cmd;
314  orb_copy(ORB_ID(vehicle_command), _vehicle_command_sub, &cmd);
315 
316  if (cmd.command == vehicle_command_s::VEHICLE_CMD_DO_GO_AROUND) {
317 
318  // DO_GO_AROUND is currently handled by the position controller (unacknowledged)
319  // TODO: move DO_GO_AROUND handling to navigator
320  publish_vehicle_command_ack(cmd, vehicle_command_s::VEHICLE_CMD_RESULT_ACCEPTED);
321 
322  } else if (cmd.command == vehicle_command_s::VEHICLE_CMD_DO_REPOSITION) {
323 
324  position_setpoint_triplet_s *rep = get_reposition_triplet();
325  position_setpoint_triplet_s *curr = get_position_setpoint_triplet();
326 
327  // store current position as previous position and goal as next
328  rep->previous.yaw = get_global_position()->yaw;
329  rep->previous.lat = get_global_position()->lat;
330  rep->previous.lon = get_global_position()->lon;
331  rep->previous.alt = get_global_position()->alt;
332 
333  rep->current.loiter_radius = get_loiter_radius();
334  rep->current.loiter_direction = 1;
335  rep->current.type = position_setpoint_s::SETPOINT_TYPE_LOITER;
336  rep->current.cruising_speed = get_cruising_speed();
337  rep->current.cruising_throttle = get_cruising_throttle();
338 
339  // Go on and check which changes had been requested
340  if (PX4_ISFINITE(cmd.param4)) {
341  rep->current.yaw = cmd.param4;
342 
343  } else {
344  rep->current.yaw = NAN;
345  }
346 
347  if (PX4_ISFINITE(cmd.param5) && PX4_ISFINITE(cmd.param6)) {
348 
349  // Position change with optional altitude change
350  rep->current.lat = (cmd.param5 < 1000) ? cmd.param5 : cmd.param5 / (double)1e7;
351  rep->current.lon = (cmd.param6 < 1000) ? cmd.param6 : cmd.param6 / (double)1e7;
352 
353  if (PX4_ISFINITE(cmd.param7)) {
354  rep->current.alt = cmd.param7;
355 
356  } else {
357  rep->current.alt = get_global_position()->alt;
358  }
359 
360  } else if (PX4_ISFINITE(cmd.param7) && curr->current.valid
361  && PX4_ISFINITE(curr->current.lat)
362  && PX4_ISFINITE(curr->current.lon)) {
363 
364  // Altitude without position change
365  rep->current.lat = curr->current.lat;
366  rep->current.lon = curr->current.lon;
367  rep->current.alt = cmd.param7;
368 
369  } else {
370  // All three set to NaN - hold in current position
371  rep->current.lat = get_global_position()->lat;
372  rep->current.lon = get_global_position()->lon;
373  rep->current.alt = get_global_position()->alt;
374  }
375 
376  rep->previous.valid = true;
377  rep->current.valid = true;
378  rep->next.valid = false;
379 
380  // CMD_DO_REPOSITION is acknowledged by commander
381 
382  } else if (cmd.command == vehicle_command_s::VEHICLE_CMD_NAV_TAKEOFF) {
383  position_setpoint_triplet_s *rep = get_takeoff_triplet();
384 
385  // store current position as previous position and goal as next
386  rep->previous.yaw = get_global_position()->yaw;
387  rep->previous.lat = get_global_position()->lat;
388  rep->previous.lon = get_global_position()->lon;
389  rep->previous.alt = get_global_position()->alt;
390 
391  rep->current.loiter_radius = get_loiter_radius();
392  rep->current.loiter_direction = 1;
393  rep->current.type = position_setpoint_s::SETPOINT_TYPE_TAKEOFF;
394 
395  if (home_position_valid()) {
396  rep->current.yaw = cmd.param4;
397  rep->previous.valid = true;
398 
399  } else {
400  rep->current.yaw = get_local_position()->yaw;
401  rep->previous.valid = false;
402  }
403 
404  if (PX4_ISFINITE(cmd.param5) && PX4_ISFINITE(cmd.param6)) {
405  rep->current.lat = (cmd.param5 < 1000) ? cmd.param5 : cmd.param5 / (double)1e7;
406  rep->current.lon = (cmd.param6 < 1000) ? cmd.param6 : cmd.param6 / (double)1e7;
407 
408  } else {
409  // If one of them is non-finite, reset both
410  rep->current.lat = (double)NAN;
411  rep->current.lon = (double)NAN;
412  }
413 
414  rep->current.alt = cmd.param7;
415 
416  rep->current.valid = true;
417  rep->next.valid = false;
418 
419  // CMD_NAV_TAKEOFF is acknowledged by commander
420 
421  } else if (cmd.command == vehicle_command_s::VEHICLE_CMD_DO_LAND_START) {
422 
423  /* find NAV_CMD_DO_LAND_START in the mission and
424  * use MAV_CMD_MISSION_START to start the mission there
425  */
426  if (_mission.land_start()) {
427  vehicle_command_s vcmd = {};
428  vcmd.command = vehicle_command_s::VEHICLE_CMD_MISSION_START;
429  vcmd.param1 = _mission.get_land_start_index();
430  publish_vehicle_cmd(&vcmd);
431 
432  } else {
433  PX4_WARN("planned mission landing not available");
434  }
435 
436  publish_vehicle_command_ack(cmd, vehicle_command_s::VEHICLE_CMD_RESULT_ACCEPTED);
437 
438  } else if (cmd.command == vehicle_command_s::VEHICLE_CMD_MISSION_START) {
439  if (_mission_result.valid && PX4_ISFINITE(cmd.param1) && (cmd.param1 >= 0)) {
440  if (!_mission.set_current_offboard_mission_index(cmd.param1)) {
441  PX4_WARN("CMD_MISSION_START failed");
442  }
443  }
444 
445  // CMD_MISSION_START is acknowledged by commander
446 
447  } else if (cmd.command == vehicle_command_s::VEHICLE_CMD_DO_CHANGE_SPEED) {
448  if (cmd.param2 > FLT_EPSILON) {
449  // XXX not differentiating ground and airspeed yet
450  set_cruising_speed(cmd.param2);
451 
452  } else {
453  set_cruising_speed();
454 
455  /* if no speed target was given try to set throttle */
456  if (cmd.param3 > FLT_EPSILON) {
457  set_cruising_throttle(cmd.param3 / 100);
458 
459  } else {
460  set_cruising_throttle();
461  }
462  }
463 
464  // TODO: handle responses for supported DO_CHANGE_SPEED options?
465  publish_vehicle_command_ack(cmd, vehicle_command_s::VEHICLE_CMD_RESULT_ACCEPTED);
466 
467  } else if (cmd.command == vehicle_command_s::VEHICLE_CMD_DO_SET_ROI
468  || cmd.command == vehicle_command_s::VEHICLE_CMD_NAV_ROI
469  || cmd.command == vehicle_command_s::VEHICLE_CMD_DO_SET_ROI_LOCATION
470  || cmd.command == vehicle_command_s::VEHICLE_CMD_DO_SET_ROI_WPNEXT_OFFSET
471  || cmd.command == vehicle_command_s::VEHICLE_CMD_DO_SET_ROI_NONE) {
472  _vroi = {};
473 
474  switch (cmd.command) {
475  case vehicle_command_s::VEHICLE_CMD_DO_SET_ROI:
476  case vehicle_command_s::VEHICLE_CMD_NAV_ROI:
477  _vroi.mode = cmd.param1;
478  break;
479 
480  case vehicle_command_s::VEHICLE_CMD_DO_SET_ROI_LOCATION:
481  _vroi.mode = vehicle_command_s::VEHICLE_ROI_LOCATION;
482  _vroi.lat = cmd.param5;
483  _vroi.lon = cmd.param6;
484  _vroi.alt = cmd.param7;
485  break;
486 
487  case vehicle_command_s::VEHICLE_CMD_DO_SET_ROI_WPNEXT_OFFSET:
488  _vroi.mode = vehicle_command_s::VEHICLE_ROI_WPNEXT;
489  _vroi.pitch_offset = (float)cmd.param5 * M_DEG_TO_RAD_F;
490  _vroi.roll_offset = (float)cmd.param6 * M_DEG_TO_RAD_F;
491  _vroi.yaw_offset = (float)cmd.param7 * M_DEG_TO_RAD_F;
492  break;
493 
494  case vehicle_command_s::VEHICLE_CMD_DO_SET_ROI_NONE:
495  _vroi.mode = vehicle_command_s::VEHICLE_ROI_NONE;
496  break;
497 
498  default:
499  _vroi.mode = vehicle_command_s::VEHICLE_ROI_NONE;
500  break;
501  }
502 
503  _vroi.timestamp = hrt_absolute_time();
504 
505  if (_vehicle_roi_pub != nullptr) {
506  orb_publish(ORB_ID(vehicle_roi), _vehicle_roi_pub, &_vroi);
507 
508  } else {
509  _vehicle_roi_pub = orb_advertise(ORB_ID(vehicle_roi), &_vroi);
510  }
511 
512  publish_vehicle_command_ack(cmd, vehicle_command_s::VEHICLE_CMD_RESULT_ACCEPTED);
513  }
514  }
515 
516  /* Check for traffic */
517  check_traffic();
518 
519  /* Check geofence violation */
520  if (have_geofence_position_data &&
521  (_geofence.getGeofenceAction() != geofence_result_s::GF_ACTION_NONE) &&
522  (hrt_elapsed_time(&last_geofence_check) > GEOFENCE_CHECK_INTERVAL)) {
523 
524  bool inside = _geofence.check(_global_pos, _gps_pos, _home_pos,
525  home_position_valid());
526  last_geofence_check = hrt_absolute_time();
527  have_geofence_position_data = false;
528 
529  _geofence_result.timestamp = hrt_absolute_time();
530  _geofence_result.geofence_action = _geofence.getGeofenceAction();
531  _geofence_result.home_required = _geofence.isHomeRequired();
532 
533  if (!inside) {
534  /* inform other apps via the mission result */
535  _geofence_result.geofence_violated = true;
536 
537  /* Issue a warning about the geofence violation once */
538  if (!_geofence_violation_warning_sent) {
539  mavlink_log_critical(&_mavlink_log_pub, "Geofence violation");
540  _geofence_violation_warning_sent = true;
541  }
542 
543  } else {
544  /* inform other apps via the mission result */
545  _geofence_result.geofence_violated = false;
546 
547  /* Reset the _geofence_violation_warning_sent field */
548  _geofence_violation_warning_sent = false;
549  }
550 
551  publish_geofence_result();
552  }
553 
554  /* Do stuff according to navigation state set by commander */
555  NavigatorMode *navigation_mode_new{nullptr};
556 
557  switch (_vstatus.nav_state) {
558  case vehicle_status_s::NAVIGATION_STATE_AUTO_MISSION:
559  _pos_sp_triplet_published_invalid_once = false;
560 
561  _mission.set_execution_mode(mission_result_s::MISSION_EXECUTION_MODE_NORMAL);
562  navigation_mode_new = &_mission;
563 
564  break;
565 
566  case vehicle_status_s::NAVIGATION_STATE_AUTO_LOITER:
567  _pos_sp_triplet_published_invalid_once = false;
568  navigation_mode_new = &_loiter;
569  break;
570 
571  case vehicle_status_s::NAVIGATION_STATE_AUTO_RCRECOVER:
572  _pos_sp_triplet_published_invalid_once = false;
573  navigation_mode_new = &_rcLoss;
574  break;
575 
576  case vehicle_status_s::NAVIGATION_STATE_AUTO_RTL: {
577  _pos_sp_triplet_published_invalid_once = false;
578 
579  const bool rtl_activated = _previous_nav_state != vehicle_status_s::NAVIGATION_STATE_AUTO_RTL;
580 
581  switch (rtl_type()) {
582  case RTL::RTL_LAND:
583  if (rtl_activated) {
584  mavlink_and_console_log_info(get_mavlink_log_pub(), "RTL LAND activated");
585  }
586 
587  // if RTL is set to use a mission landing and mission has a planned landing, then use MISSION to fly there directly
588  if (on_mission_landing() && !get_land_detected()->landed) {
589  _mission.set_execution_mode(mission_result_s::MISSION_EXECUTION_MODE_FAST_FORWARD);
590  navigation_mode_new = &_mission;
591 
592  } else {
593  navigation_mode_new = &_rtl;
594  }
595 
596  break;
597 
598  case RTL::RTL_MISSION:
599  if (_mission.get_land_start_available() && !get_land_detected()->landed) {
600  // the mission contains a landing spot
601  _mission.set_execution_mode(mission_result_s::MISSION_EXECUTION_MODE_FAST_FORWARD);
602 
603  if (_navigation_mode != &_mission) {
604  if (_navigation_mode == nullptr) {
605  // switching from an manual mode, go to landing if not already landing
606  if (!on_mission_landing()) {
607  start_mission_landing();
608  }
609 
610  } else {
611  // switching from an auto mode, continue the mission from the closest item
612  _mission.set_closest_item_as_current();
613  }
614  }
615 
616  if (rtl_activated) {
617  mavlink_and_console_log_info(get_mavlink_log_pub(), "RTL Mission activated, continue mission");
618  }
619 
620  navigation_mode_new = &_mission;
621 
622  } else {
623  // fly the mission in reverse if switching from a non-manual mode
624  _mission.set_execution_mode(mission_result_s::MISSION_EXECUTION_MODE_REVERSE);
625 
626  if ((_navigation_mode != nullptr && (_navigation_mode != &_rtl || _mission.get_mission_changed())) &&
627  (! _mission.get_mission_finished()) &&
628  (!get_land_detected()->landed)) {
629  // determine the closest mission item if switching from a non-mission mode, and we are either not already
630  // mission mode or the mission waypoints changed.
631  // The seconds condition is required so that when no mission was uploaded and one is available the closest
632  // mission item is determined and also that if the user changes the active mission index while rtl is active
633  // always that waypoint is tracked first.
634  if ((_navigation_mode != &_mission) && (rtl_activated || _mission.get_mission_waypoints_changed())) {
635  _mission.set_closest_item_as_current();
636  }
637 
638  if (rtl_activated) {
639  mavlink_and_console_log_info(get_mavlink_log_pub(), "RTL Mission activated, fly mission in reverse");
640  }
641 
642  navigation_mode_new = &_mission;
643 
644  } else {
645  if (rtl_activated) {
646  mavlink_and_console_log_info(get_mavlink_log_pub(), "RTL Mission activated, fly to home");
647  }
648 
649  navigation_mode_new = &_rtl;
650  }
651  }
652 
653  break;
654 
655  default:
656  if (rtl_activated) {
657  mavlink_and_console_log_info(get_mavlink_log_pub(), "RTL HOME activated");
658  }
659 
660  navigation_mode_new = &_rtl;
661  break;
662  }
663 
664  break;
665  }
666 
667  case vehicle_status_s::NAVIGATION_STATE_AUTO_TAKEOFF:
668  _pos_sp_triplet_published_invalid_once = false;
669  navigation_mode_new = &_takeoff;
670  break;
671 
672  case vehicle_status_s::NAVIGATION_STATE_AUTO_LAND:
673  _pos_sp_triplet_published_invalid_once = false;
674  navigation_mode_new = &_land;
675  break;
676 
677  case vehicle_status_s::NAVIGATION_STATE_AUTO_PRECLAND:
678  _pos_sp_triplet_published_invalid_once = false;
679  navigation_mode_new = &_precland;
680  _precland.set_mode(PrecLandMode::Required);
681  break;
682 
683  case vehicle_status_s::NAVIGATION_STATE_DESCEND:
684  _pos_sp_triplet_published_invalid_once = false;
685  navigation_mode_new = &_land;
686  break;
687 
688  case vehicle_status_s::NAVIGATION_STATE_AUTO_RTGS:
689  _pos_sp_triplet_published_invalid_once = false;
690  navigation_mode_new = &_dataLinkLoss;
691  break;
692 
693  case vehicle_status_s::NAVIGATION_STATE_AUTO_LANDENGFAIL:
694  _pos_sp_triplet_published_invalid_once = false;
695  navigation_mode_new = &_engineFailure;
696  break;
697 
698  case vehicle_status_s::NAVIGATION_STATE_AUTO_LANDGPSFAIL:
699  _pos_sp_triplet_published_invalid_once = false;
700  navigation_mode_new = &_gpsFailure;
701  break;
702 
703  case vehicle_status_s::NAVIGATION_STATE_AUTO_FOLLOW_TARGET:
704  _pos_sp_triplet_published_invalid_once = false;
705  navigation_mode_new = &_follow_target;
706  break;
707 
708  case vehicle_status_s::NAVIGATION_STATE_MANUAL:
709  case vehicle_status_s::NAVIGATION_STATE_ACRO:
710  case vehicle_status_s::NAVIGATION_STATE_ALTCTL:
711  case vehicle_status_s::NAVIGATION_STATE_POSCTL:
712  case vehicle_status_s::NAVIGATION_STATE_TERMINATION:
713  case vehicle_status_s::NAVIGATION_STATE_OFFBOARD:
714  case vehicle_status_s::NAVIGATION_STATE_STAB:
715  default:
716  _pos_sp_triplet_published_invalid_once = false;
717  navigation_mode_new = nullptr;
718  _can_loiter_at_sp = false;
719  break;
720  }
721 
722  // update the vehicle status
723  _previous_nav_state = _vstatus.nav_state;
724 
725  /* we have a new navigation mode: reset triplet */
726  if (_navigation_mode != navigation_mode_new) {
727  // We don't reset the triplet if we just did an auto-takeoff and are now
728  // going to loiter. Otherwise, we lose the takeoff altitude and end up lower
729  // than where we wanted to go.
730  //
731  // FIXME: a better solution would be to add reset where they are needed and remove
732  // this general reset here.
733  if (!(_navigation_mode == &_takeoff &&
734  navigation_mode_new == &_loiter)) {
735  reset_triplets();
736  }
737  }
738 
739  _navigation_mode = navigation_mode_new;
740 
741  /* iterate through navigation modes and set active/inactive for each */
742  for (unsigned int i = 0; i < NAVIGATOR_MODE_ARRAY_SIZE; i++) {
743  _navigation_mode_array[i]->run(_navigation_mode == _navigation_mode_array[i]);
744  }
745 
746  /* if we landed and have not received takeoff setpoint then stay in idle */
747  if (_land_detected.landed &&
748  !((_vstatus.nav_state == vehicle_status_s::NAVIGATION_STATE_AUTO_TAKEOFF)
749  || (_vstatus.nav_state == vehicle_status_s::NAVIGATION_STATE_AUTO_MISSION))) {

761  reset_triplets();

992 void Navigator::check_traffic()
993 {
994  double lat = get_global_position()->lat;
995  double lon = get_global_position()->lon;
996  float alt = get_global_position()->alt;
997 
998  // TODO for non-multirotors predicting the future
999  // position as accurately as possible will become relevant
1000  // float vel_n = get_global_position()->vel_n;
1001  // float vel_e = get_global_position()->vel_e;
1002  // float vel_d = get_global_position()->vel_d;
1003 
1004  bool changed;
1005  orb_check(_traffic_sub, &changed);
1006 
1007  float horizontal_separation = 500;
1008  float vertical_separation = 500;
1009 
1010  while (changed) {
1011 
1012  transponder_report_s tr;
1013  orb_copy(ORB_ID(transponder_report), _traffic_sub, &tr);
1014 
1015  uint16_t required_flags = transponder_report_s::PX4_ADSB_FLAGS_VALID_COORDS |
1016  transponder_report_s::PX4_ADSB_FLAGS_VALID_HEADING |
1017  transponder_report_s::PX4_ADSB_FLAGS_VALID_VELOCITY | transponder_report_s::PX4_ADSB_FLAGS_VALID_ALTITUDE;
1018 
1019  if ((tr.flags & required_flags) != required_flags) {
1020  orb_check(_traffic_sub, &changed);
1021  continue;
1022  }
1023 
1024  float d_hor, d_vert;
1025  get_distance_to_point_global_wgs84(lat, lon, alt,
1026  tr.lat, tr.lon, tr.altitude, &d_hor, &d_vert);
1027 
1028 
1029  // predict final altitude (positive is up) in prediction time frame
1030  float end_alt = tr.altitude + (d_vert / tr.hor_velocity) * tr.ver_velocity;
1031 
1032  // Predict until the vehicle would have passed this system at its current speed
1033  float prediction_distance = d_hor + 1000.0f;
1034 
1035  // If the altitude is not getting close to us, do not calculate
1036  // the horizontal separation.
1037  // Since commercial flights do most of the time keep flight levels
1038  // check for the current and for the predicted flight level.
1039  // we also make the implicit assumption that this system is on the lowest
1040  // flight level close to ground in the
1041  // (end_alt - horizontal_separation < alt) condition. If this system should
1042  // ever be used in normal airspace this implementation would anyway be
1043  // inappropriate as it should be replaced with a TCAS compliant solution.
1044 
1045  if ((fabsf(alt - tr.altitude) < vertical_separation) || ((end_alt - horizontal_separation) < alt)) {
1046 
1047  double end_lat, end_lon;
1048  waypoint_from_heading_and_distance(tr.lat, tr.lon, tr.heading, prediction_distance, &end_lat, &end_lon);
1049 
1050  struct crosstrack_error_s cr;
1051 
1052  if (!get_distance_to_line(&cr, lat, lon, tr.lat, tr.lon, end_lat, end_lon)) {
1053 
1054  if (!cr.past_end && (fabsf(cr.distance) < horizontal_separation)) {
1055 
1056  // direction of traffic in human-readable 0..360 degree in earth frame
1057  int traffic_direction = math::degrees(tr.heading) + 180;
1058 
1059  switch (_param_traffic_avoidance_mode.get()) {
1060 
1061  case 0: {
1062  /* ignore */
1063  PX4_WARN("TRAFFIC %s, hdg: %d", tr.flags & transponder_report_s::PX4_ADSB_FLAGS_VALID_CALLSIGN ? tr.callsign :
1064  "unknown",
1065  traffic_direction);
1066  break;
1067  }
1068 
1069  case 1: {
1070  mavlink_log_critical(&_mavlink_log_pub, "WARNING TRAFFIC %s at heading %d, land immediately",
1071  tr.flags & transponder_report_s::PX4_ADSB_FLAGS_VALID_CALLSIGN ? tr.callsign : "unknown",
1072  traffic_direction);
1073  break;
1074  }
1075 
1076  case 2: {
1077  mavlink_log_critical(&_mavlink_log_pub, "AVOIDING TRAFFIC %s heading %d, returning home",
1078  tr.flags & transponder_report_s::PX4_ADSB_FLAGS_VALID_CALLSIGN ? tr.callsign : "unknown",
1079  traffic_direction);
1080 
1081  // set the return altitude to minimum
1082  _rtl.set_return_alt_min(true);
1083 
1084  // ask the commander to execute an RTL
1085  vehicle_command_s vcmd = {};
1086  vcmd.command = vehicle_command_s::VEHICLE_CMD_NAV_RETURN_TO_LAUNCH;
1087  publish_vehicle_cmd(&vcmd);
1088  break;
1089  }
1090  }
1091  }
1092  }
1093  }
1094 
1095  orb_check(_traffic_sub, &changed);
1096  }
1097 }
