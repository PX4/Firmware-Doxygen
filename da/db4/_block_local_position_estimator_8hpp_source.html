59 // dynamics:
60 //
61 // x(+) = A * x(-) + B * u(+)
62 // y_i = C_i*x
63 //
64 // kalman filter
65 //
66 // E[xx'] = P
67 // E[uu'] = W
68 // E[y_iy_i'] = R_i
69 //
70 // prediction
71 // x(+|-) = A*x(-|-) + B*u(+)
72 // P(+|-) = A*P(-|-)*A' + B*W*B'
73 //
74 // correction
75 // x(+|+) = x(+|-) + K_i * (y_i - H_i * x(+|-) )
76 //
77 //
78 // input:
79 // ax, ay, az (acceleration NED)
80 //
81 // states:
82 // px, py, pz , ( position NED, m)
83 // vx, vy, vz ( vel NED, m/s),
84 // bx, by, bz ( accel bias, m/s^2)
85 // tz (terrain altitude, ASL, m)
86 //
87 // measurements:
88 //
89 // sonar: pz (measured d*cos(phi)*cos(theta))
90 //
91 // baro: pz
92 //
93 // flow: vx, vy (flow is in body x, y frame)
94 //
95 // gps: px, py, pz, vx, vy, vz (flow is in body x, y frame)
96 //
97 // lidar: pz (actual measured d*cos(phi)*cos(theta))
98 //
99 // vision: px, py, pz, vx, vy, vz
100 //
101 // mocap: px, py, pz
102 //
103 // land (detects when landed)): pz (always measures agl = 0)
104 //
