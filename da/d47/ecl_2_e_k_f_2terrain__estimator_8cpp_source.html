46 bool Ekf::initHagl()
47 {
48  // get most recent range measurement from buffer
49  const rangeSample &latest_measurement = _range_buffer.get_newest();
50 
51  if ((_time_last_imu - latest_measurement.time_us) < (uint64_t)2e5 && _R_rng_to_earth_2_2 > _params.range_cos_max_tilt) {
52  // if we have a fresh measurement, use it to initialise the terrain estimator
53  _terrain_vpos = _state.pos(2) + latest_measurement.rng * _R_rng_to_earth_2_2;
54  // initialise state variance to variance of measurement
55  _terrain_var = sq(_params.range_noise);
56  // success
57  return true;
58 
59  } else if (!_control_status.flags.in_air) {
60  // if on ground we assume a ground clearance
61  _terrain_vpos = _state.pos(2) + _params.rng_gnd_clearance;
62  // Use the ground clearance value as our uncertainty
63  _terrain_var = sq(_params.rng_gnd_clearance);
64  // ths is a guess
65  return false;
66 
67  } else {
68  // no information - cannot initialise
69  return false;
70  }
71 }

73 void Ekf::runTerrainEstimator()
74 {
75  // Perform a continuity check on range finder data
76  checkRangeDataContinuity();
77 
78  // Perform initialisation check
79  if (!_terrain_initialised) {
80  _terrain_initialised = initHagl();
81 
82  } else {
83 
84  // predict the state variance growth where the state is the vertical position of the terrain underneath the vehicle
85 
86  // process noise due to errors in vehicle height estimate
87  _terrain_var += sq(_imu_sample_delayed.delta_vel_dt * _params.terrain_p_noise);
88 
89  // process noise due to terrain gradient
90  _terrain_var += sq(_imu_sample_delayed.delta_vel_dt * _params.terrain_gradient) * (sq(_state.vel(0)) + sq(_state.vel(
91  1)));
92 
93  // limit the variance to prevent it becoming badly conditioned
94  _terrain_var = math::constrain(_terrain_var, 0.0f, 1e4f);
95 
96  // Fuse range finder data if available
97  if (_range_data_ready && !_rng_hgt_faulty) {
98  fuseHagl();
99 
100  // update range sensor angle parameters in case they have changed
101  // we do this here to avoid doing those calculations at a high rate
102  _sin_tilt_rng = sinf(_params.rng_sens_pitch);
103  _cos_tilt_rng = cosf(_params.rng_sens_pitch);
104 
105  }
106 
107  //constrain _terrain_vpos to be a minimum of _params.rng_gnd_clearance larger than _state.pos(2)
108  if (_terrain_vpos - _state.pos(2) < _params.rng_gnd_clearance) {
109  _terrain_vpos = _params.rng_gnd_clearance + _state.pos(2);
110  }
111  }
112 
113  // Update terrain validity
114  update_terrain_valid();
115 }

117 void Ekf::fuseHagl()
118 {
119  // If the vehicle is excessively tilted, do not try to fuse range finder observations
120  if (_R_rng_to_earth_2_2 > _params.range_cos_max_tilt) {
121  // get a height above ground measurement from the range finder assuming a flat earth
122  float meas_hagl = _range_sample_delayed.rng * _R_rng_to_earth_2_2;
123 
124  // predict the hagl from the vehicle position and terrain height
125  float pred_hagl = _terrain_vpos - _state.pos(2);
126 
127  // calculate the innovation
128  _hagl_innov = pred_hagl - meas_hagl;
129 
130  // calculate the observation variance adding the variance of the vehicles own height uncertainty
131  float obs_variance = fmaxf(P[9][9], 0.0f) + sq(_params.range_noise) + sq(_params.range_noise_scaler * _range_sample_delayed.rng);
132 
133  // calculate the innovation variance - limiting it to prevent a badly conditioned fusion
134  _hagl_innov_var = fmaxf(_terrain_var + obs_variance, obs_variance);
135 
136  // perform an innovation consistency check and only fuse data if it passes
137  float gate_size = fmaxf(_params.range_innov_gate, 1.0f);
138  _terr_test_ratio = sq(_hagl_innov) / (sq(gate_size) * _hagl_innov_var);
139 
140  if (_terr_test_ratio <= 1.0f) {
141  // calculate the Kalman gain
142  float gain = _terrain_var / _hagl_innov_var;
143  // correct the state
144  _terrain_vpos -= gain * _hagl_innov;
145  // correct the variance
146  _terrain_var = fmaxf(_terrain_var * (1.0f - gain), 0.0f);
147  // record last successful fusion event
148  _time_last_hagl_fuse = _time_last_imu;
149  _innov_check_fail_status.flags.reject_hagl = false;
150  } else {
151  // If we have been rejecting range data for too long, reset to measurement
152  if (_time_last_imu - _time_last_hagl_fuse > (uint64_t)10E6) {
153  _terrain_vpos = _state.pos(2) + meas_hagl;
154  _terrain_var = obs_variance;
155  } else {
156  _innov_check_fail_status.flags.reject_hagl = true;
157  }
158  }
159  } else {
160  _innov_check_fail_status.flags.reject_hagl = true;
161  return;
162  }
163 }

200 // check that the range finder data is continuous
201 void Ekf::checkRangeDataContinuity()
202 {
203  // update range data continuous flag (1Hz ie 2000 ms)
204  /* Timing in micro seconds */
205 
206  /* Apply a 2.0 sec low pass filter to the time delta from the last range finder updates */
207  float alpha = 0.5f * _dt_update;
208  _dt_last_range_update_filt_us = _dt_last_range_update_filt_us * (1.0f - alpha) + alpha *
209  (_time_last_imu - _time_last_range);
210 
211  _dt_last_range_update_filt_us = fminf(_dt_last_range_update_filt_us, 4e6f);
212 
213  if (_dt_last_range_update_filt_us < 2e6f) {
214  _range_data_continuous = true;
215 
216  } else {
217  _range_data_continuous = false;
218  }
