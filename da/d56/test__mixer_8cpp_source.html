206  bool ret = load_mixer(buf, 0);

208  if (!ret) {

251 bool MixerTest::load_mixer(const char *filename, const char *buf, unsigned loaded, unsigned expected_count,
252  const unsigned chunk_size,
253  bool verbose)
254 {
255 
256 
257  /* load the mixer in chunks, like
258  * in the case of a remote load,
259  * e.g. on PX4IO.
260  */
261 
262  /* load at once test */
263  unsigned xx = loaded;
264  mixer_group.reset();
265  mixer_group.load_from_buf(&buf[0], xx);
266 
267  if (expected_count > 0) {
268  ut_compare("check number of mixers loaded", mixer_group.count(), expected_count);
269  }
270 
271  unsigned empty_load = 2;
272  char empty_buf[2];
273  empty_buf[0] = ' ';
274  empty_buf[1] = '\0';
275  mixer_group.reset();
276  mixer_group.load_from_buf(&empty_buf[0], empty_load);
277 
278  if (verbose) {
279  PX4_INFO("empty buffer load: loaded %u mixers, used: %u", mixer_group.count(), empty_load);
280  }
281 
282  ut_compare("empty buffer load", empty_load, 0);
283 
284  /* reset, load in chunks */
285  mixer_group.reset();
286  char mixer_text[PX4IO_MAX_MIXER_LENGTH]; /* large enough for one mixer */
287 
288  unsigned mixer_text_length = 0;
289  unsigned transmitted = 0;
290  unsigned resid = 0;
291 
292  while (transmitted < loaded) {
293 
294  unsigned text_length = (loaded - transmitted > chunk_size) ? chunk_size : loaded - transmitted;
295 
296  /* check for overflow - this would be really fatal */
297  if ((mixer_text_length + text_length + 1) > sizeof(mixer_text)) {
298  PX4_ERR("Mixer text length overflow for file: %s. Is PX4IO_MAX_MIXER_LENGTH too small? (curr len: %d)", filename,
299  PX4IO_MAX_MIXER_LENGTH);
300  return false;
301  }
302 
303  /* append mixer text and nul-terminate */
304  memcpy(&mixer_text[mixer_text_length], &buf[transmitted], text_length);
305  mixer_text_length += text_length;
306  mixer_text[mixer_text_length] = '\0';
307  //fprintf(stderr, "buflen %u, text:\n\"%s\"\n", mixer_text_length, &mixer_text[0]);
308 
309  /* process the text buffer, adding new mixers as their descriptions can be parsed */
310  resid = mixer_text_length;
311  mixer_group.load_from_buf(&mixer_text[0], resid);
312 
313  /* if anything was parsed */
314  if (resid != mixer_text_length) {
315  //PX4_INFO("loaded %d mixers, used %u\n", mixer_group.count(), mixer_text_length - resid);
316 
317  /* copy any leftover text to the base of the buffer for re-use */
318  if (resid > 0) {
319  memmove(&mixer_text[0], &mixer_text[mixer_text_length - resid], resid);
320  /* enforce null termination */
321  mixer_text[resid] = '\0';
322  }
323 
324  mixer_text_length = resid;
325  }
326 
327  transmitted += text_length;
328 
329  if (verbose) {
330  PX4_INFO("transmitted: %d, loaded: %d", transmitted, loaded);
331  }
332  }
333 
334  if (verbose) {
335  PX4_INFO("chunked load: loaded %u mixers", mixer_group.count());
336  }
337 
338  if (expected_count > 0 && mixer_group.count() != expected_count) {
339  PX4_ERR("Load of mixer failed, last chunk: %s, transmitted: %u, text length: %u, resid: %u", mixer_text, transmitted,
340  mixer_text_length, resid);
341  ut_compare("check number of mixers loaded (chunk)", mixer_group.count(), expected_count);
342  }
343 
344  return true;
345 }

597  }
