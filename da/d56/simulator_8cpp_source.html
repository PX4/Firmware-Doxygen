154 
155  if (argv[2][1] == 's') {
156  _instance->initializeSensorData();
157 #ifndef __PX4_QURT

163  // Update sensor data

166  } else {
167  _instance->initializeSensorData();
168  _instance->_initialized = true;
169  }
170 
171  } else {
172  PX4_WARN("Simulator creation failed");
173  ret = 1;
174  }
175 
176  return ret;

193  int simulator_main(int argc, char *argv[])
194  {
195  int ret = 0;
196 
197  if (argc > 2 && strcmp(argv[1], "start") == 0) {
198  if (strcmp(argv[2], "-s") == 0 ||
199  strcmp(argv[2], "-p") == 0 ||
200  strcmp(argv[2], "-t") == 0) {
201 
202  if (g_sim_task >= 0) {
203  warnx("Simulator already started");
204  return 0;
205  }
206 
207  // enable lockstep support
208  px4_enable_sim_lockstep();
209 
210  g_sim_task = px4_task_spawn_cmd("simulator",
211  SCHED_DEFAULT,
212  SCHED_PRIORITY_MAX,
213  1500,
214  Simulator::start,
215  argv);
216 
217  // now wait for the command to complete
218  while (!px4_exit_requested()) {
219  if (Simulator::getInstance() && Simulator::getInstance()->isInitialized()) {
220  break;
221 
222  } else {
223  usleep(100000);
224  }
225  }
226 
227  } else {
228  usage();
229  ret = -EINVAL;
230  }
231 
232  } else if (argc == 2 && strcmp(argv[1], "stop") == 0) {
233  if (g_sim_task < 0) {
234  PX4_WARN("Simulator not running");
235 
236  } else {
237  px4_task_delete(g_sim_task);
238  g_sim_task = -1;
239  }
240 
241  } else {
242  usage();
243  ret = -EINVAL;
244  }
245 
246  return ret;
247  }
