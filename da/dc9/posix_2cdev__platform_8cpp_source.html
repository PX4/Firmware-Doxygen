158  {
159  PX4_DEBUG("px4_open");

161  int ret = 0;
162  int i;
163  mode_t mode;
164 
165  if (!dev && (flags & PX4_F_WRONLY) != 0 &&
166  strncmp(path, "/obj/", 5) != 0 &&
167  strncmp(path, "/dev/", 5) != 0) {
168  va_list p;
169  va_start(p, flags);
170  mode = va_arg(p, int);
171  va_end(p);
172 
173  // Create the file
174  PX4_DEBUG("Creating virtual file %s", path);
175  dev = device::VFile::createFile(path, mode);
176  }
177 
178  if (dev) {
179 
180  pthread_mutex_lock(&filemutex);
181 
182  for (i = 0; i < PX4_MAX_FD; ++i) {
183  if (filemap[i].vdev == nullptr) {
184  filemap[i] = device::file_t(flags, dev);
185  break;
186  }
187  }
188 
189  pthread_mutex_unlock(&filemutex);
190 
191  if (i < PX4_MAX_FD) {
192  ret = dev->open(&filemap[i]);
193 
194  } else {
195 
196  const unsigned NAMELEN = 32;
197  char thread_name[NAMELEN] = {};
198 
199 #ifndef __PX4_QURT
200  int nret = pthread_getname_np(pthread_self(), thread_name, NAMELEN);
201 
202  if (nret || thread_name[0] == 0) {
203  PX4_WARN("failed getting thread name");
204  }
205 
206  PX4_BACKTRACE();
207 #endif
208 
209  PX4_WARN("%s: exceeded maximum number of file descriptors, accessing %s",
210  thread_name, path);
211  ret = -ENOENT;
212  }
213 
214  } else {
215  ret = -EINVAL;
216  }
217 
218  if (ret < 0) {
219  return -1;
220  }
221 
222  PX4_DEBUG("px4_open fd = %d", i);

318  int px4_poll(px4_pollfd_struct_t *fds, nfds_t nfds, int timeout)
319  {
320  if (nfds == 0) {
321  PX4_WARN("px4_poll with no fds");
322  return -1;
323  }
324 
325  px4_sem_t sem;
326  int count = 0;
327  int ret = -1;
328  unsigned int i;
329 
330  const unsigned NAMELEN = 32;
331  char thread_name[NAMELEN] = {};
332 
333 #ifndef __PX4_QURT
334  int nret = pthread_getname_np(pthread_self(), thread_name, NAMELEN);
335 
336  if (nret || thread_name[0] == 0) {
337  PX4_WARN("failed getting thread name");
338  }
339 
340 #endif
341 
342  while (sim_delay) {
343  usleep(100);
344  }
345 
346  PX4_DEBUG("Called px4_poll timeout = %d", timeout);
347 
348  px4_sem_init(&sem, 0, 0);
349 
350  // sem use case is a signal
351  px4_sem_setprotocol(&sem, SEM_PRIO_NONE);
352 
353  // Go through all fds and check them for a pollable state
354  bool fd_pollable = false;
355 
356  for (i = 0; i < nfds; ++i) {
357  fds[i].sem = &sem;
358  fds[i].revents = 0;
359  fds[i].priv = nullptr;
360 
361  device::CDev *dev = get_vdev(fds[i].fd);
362 
363  // If fd is valid
364  if (dev) {
365  PX4_DEBUG("%s: px4_poll: CDev->poll(setup) %d", thread_name, fds[i].fd);
366  ret = dev->poll(&filemap[fds[i].fd], &fds[i], true);
367 
368  if (ret < 0) {
369  PX4_WARN("%s: px4_poll() error: %s",
370  thread_name, strerror(errno));
371  break;
372  }
373 
374  if (ret >= 0) {
375  fd_pollable = true;
376  }
377  }
378  }
379 
380  // If any FD can be polled, lock the semaphore and
381  // check for new data
382  if (fd_pollable) {
383  if (timeout > 0) {
384 
385  // Get the current time
386  struct timespec ts;
387  // FIXME: check if QURT should probably be using CLOCK_MONOTONIC
388  px4_clock_gettime(CLOCK_REALTIME, &ts);
389 
390  // Calculate an absolute time in the future
391  const unsigned billion = (1000 * 1000 * 1000);
392  unsigned tdiff = timeout;
393  uint64_t nsecs = ts.tv_nsec + (tdiff * 1000 * 1000);
394  ts.tv_sec += nsecs / billion;
395  nsecs -= (nsecs / billion) * billion;
396  ts.tv_nsec = nsecs;
397 
398  // Execute a blocking wait for that time in the future
399  errno = 0;
400  ret = px4_sem_timedwait(&sem, &ts);
401 #ifndef __PX4_DARWIN
402  ret = errno;
403 #endif
404 
405  // Ensure ret is negative on failure
406  if (ret > 0) {
407  ret = -ret;
408  }
409 
410  if (ret && ret != -ETIMEDOUT) {
411  PX4_WARN("%s: px4_poll() sem error", thread_name);
412  }
413 
414  } else if (timeout < 0) {
415  px4_sem_wait(&sem);
416  }
417 
418  // We have waited now (or not, depending on timeout),
419  // go through all fds and count how many have data
420  for (i = 0; i < nfds; ++i) {
421 
422  device::CDev *dev = get_vdev(fds[i].fd);
423 
424  // If fd is valid
425  if (dev) {
426  PX4_DEBUG("%s: px4_poll: CDev->poll(teardown) %d", thread_name, fds[i].fd);
427  ret = dev->poll(&filemap[fds[i].fd], &fds[i], false);
428 
429  if (ret < 0) {
430  PX4_WARN("%s: px4_poll() 2nd poll fail", thread_name);
431  break;
432  }
433 
434  if (fds[i].revents) {
435  count += 1;
436  }
437  }
438  }
439  }
440 
441  px4_sem_destroy(&sem);
442 
443  // Return the positive count if present,
444  // return the negative error number if failed
445  return (count) ? count : ret;
446  }
