162 int roboclaw_thread_main(int argc, char *argv[])
163 {
164  printf("[roboclaw] starting\n");
165 
166  // skip parent process args
167  argc -= 2;
168  argv += 2;
169 
170  if (argc < 3) {
171  printf("usage: roboclaw start device address\n");
172  return -1;
173  }
174 
175  const char *deviceName = argv[1];
176  uint8_t address = strtoul(argv[2], nullptr, 0);
177  uint16_t pulsesPerRev = strtoul(argv[3], nullptr, 0);
178 
179  printf("device:\t%s\taddress:\t%d\tpulses per rev:\t%ld\n",
180  deviceName, address, pulsesPerRev);
181 
182  // start
183  RoboClaw roboclaw(deviceName, address, pulsesPerRev);
184 
185  thread_running = true;
186 
187  // loop
188  while (!thread_should_exit) {
189  roboclaw.update();
190  }
191 
192  // exit
193  printf("[roboclaw] exiting.\n");
194  thread_running = false;
195  return 0;
196 }
