124 int MD25::readData()
125 {
126  uint8_t sendBuf[1];
127  sendBuf[0] = REG_SPEED1_RW;
128  uint8_t recvBuf[17];
129  int ret = transfer(sendBuf, sizeof(sendBuf),
130  recvBuf, sizeof(recvBuf));
131 
132  if (ret == OK) {
133  _version = recvBuf[REG_SW_VERSION_R];
134  _motor1Speed = _uint8ToNorm(recvBuf[REG_SPEED1_RW]);
135  _motor2Speed = _uint8ToNorm(recvBuf[REG_SPEED2_RW]);
136  _revolutions1 = -int32_t((recvBuf[REG_ENC1A_R] << 24) +
137  (recvBuf[REG_ENC1B_R] << 16) +
138  (recvBuf[REG_ENC1C_R] << 8) +
139  recvBuf[REG_ENC1D_R]) / 360.0;
140  _revolutions2 = -int32_t((recvBuf[REG_ENC2A_R] << 24) +
141  (recvBuf[REG_ENC2B_R] << 16) +
142  (recvBuf[REG_ENC2C_R] << 8) +
143  recvBuf[REG_ENC2D_R]) / 360.0;
144  _batteryVoltage = recvBuf[REG_BATTERY_VOLTS_R] / 10.0;
145  _motor1Current = recvBuf[REG_MOTOR1_CURRENT_R] / 10.0;
146  _motor2Current = recvBuf[REG_MOTOR2_CURRENT_R] / 10.0;
147  _motorAccel = recvBuf[REG_ACCEL_RATE_RW];
148  _mode = e_mode(recvBuf[REG_MODE_RW]);
149  _command = e_cmd(recvBuf[REG_COMMAND_RW]);
150  }
151 
152  return ret;
153 }

326 void MD25::update()
327 {
328  // wait for an actuator publication,
329  // check for exit condition every second
330  // note "::poll" is required to distinguish global
331  // poll from member function for driver
332  if (::poll(&_controlPoll, 1, 1000) < 0) { return; } // poll error
333 
334  // if new data, send to motors
335  if (_actuators.updated()) {
336  _actuators.update();
337  setMotor1Speed(_actuators.control[CH_SPEED_LEFT]);
338  setMotor2Speed(_actuators.control[CH_SPEED_RIGHT]);
339  }

342 int MD25::probe()
343 {
344  uint8_t goodAddress = 0;
345  bool found = false;
346  int ret = OK;
347 
348  // try initial address first, if good, then done
349  if (readData() == OK) { return ret; }
350 
351  // try all other addresses
352  uint8_t testAddress = 0;
353 
354  //printf("searching for MD25 address\n");
355  while (true) {
356  set_device_address(testAddress);
357  ret = readData();
358 
359  if (ret == OK && !found) {
360  //printf("device found at address: 0x%X\n", testAddress);
361  if (!found) {
362  found = true;
363  goodAddress = testAddress;
364  }
365  }
366 
367  if (testAddress > 254) {
368  break;
369  }
370 
371  testAddress++;
372  }
373 
374  if (found) {
375  set_device_address(goodAddress);
376  return OK;
377 
378  } else {
379  set_device_address(0);
380  return ret;
381  }
382 }
