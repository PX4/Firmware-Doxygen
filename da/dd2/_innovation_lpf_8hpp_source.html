48  /**
49  * Update the filter with a new value and returns the filtered state
50  * The new value is constained by the limit set in setSpikeLimit
51  * @param val new input
52  * @param alpha normalized weight of the new input
53  * @param spike_limit the amplitude of the saturation at the input of the filter
54  * @return filtered output
55  */
56  float update(float val, float alpha, float spike_limit)
57  {
58  float val_constrained = math::constrain(val, -spike_limit, spike_limit);
59  float beta = 1.f - alpha;
60 
61  _x = beta * _x + alpha * val_constrained;
62 
63  return _x;
64  }
65 
66  /**
67  * Helper function to compute alpha from dt and the inverse of tau
68  * @param dt sampling time in seconds
69  * @param tau_inv inverse of the time constant of the filter
70  * @return alpha, the normalized weight of a new measurement
71  */
72  static float computeAlphaFromDtAndTauInv(float dt, float tau_inv)
73  {
74  return math::constrain(dt * tau_inv, 0.f, 1.f);
75  }
76 
77 private:
