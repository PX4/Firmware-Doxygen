51 void board_get_uuid(uuid_byte_t uuid_bytes)
52 {
53  uint32_t *chip_uuid = (uint32_t *) DUMMY_SIM_UIDH;
54  uint32_t *uuid_words = (uint32_t *) uuid_bytes;

66 int board_get_uuid32_formated(char *format_buffer, int size,
67  const char *format,
68  const char *seperator)
69 {
70  uuid_uint32_t uuid;
71  board_get_uuid32(uuid);
72 
73  int offset = 0;
74  int sep_size = seperator ? strlen(seperator) : 0;
75 
76  for (unsigned int i = 0; i < PX4_CPU_UUID_WORD32_LENGTH; i++) {
77  offset += snprintf(&format_buffer[offset], size - ((i * 2 * sizeof(uint32_t)) + 1), format, uuid[i]);
78 
79  if (sep_size && i < PX4_CPU_UUID_WORD32_LENGTH - 1) {
80  strcat(&format_buffer[offset], seperator);
81  offset += sep_size;
82  }
83  }
84 
85  return 0;
86 }
