54 int do_trim_calibration(orb_advert_t *mavlink_log_pub)
55 {
56  int sub_man = orb_subscribe(ORB_ID(manual_control_setpoint));
57  usleep(400000);
58  struct manual_control_setpoint_s sp;
59  bool changed;
60  orb_check(sub_man, &changed);
61 
62  if (!changed) {
63  mavlink_log_critical(mavlink_log_pub, "no inputs, aborting");
64  return PX4_ERROR;
65  }
66 
67  orb_copy(ORB_ID(manual_control_setpoint), sub_man, &sp);
68 
69  /* load trim values which are active */
70  float roll_trim_active;
71  param_get(param_find("TRIM_ROLL"), &roll_trim_active);
72  float pitch_trim_active;
73  param_get(param_find("TRIM_PITCH"), &pitch_trim_active);
74  float yaw_trim_active;
75  param_get(param_find("TRIM_YAW"), &yaw_trim_active);
76 
77  /* get manual control scale values */
78  float roll_scale;
79  param_get(param_find("FW_MAN_R_SC"), &roll_scale);
80  float pitch_scale;
81  param_get(param_find("FW_MAN_P_SC"), &pitch_scale);
82  float yaw_scale;
83  param_get(param_find("FW_MAN_Y_SC"), &yaw_scale);
84 
85  /* set parameters: the new trim values are the combination of active trim values
86  and the values coming from the remote control of the user
87  */
88  float p = sp.y * roll_scale + roll_trim_active;
89  int p1r = param_set(param_find("TRIM_ROLL"), &p);
90  /*
91  we explicitly swap sign here because the trim is added to the actuator controls
92  which are moving in an inverse sense to manual pitch inputs
93  */
94  p = -sp.x * pitch_scale + pitch_trim_active;
95  int p2r = param_set(param_find("TRIM_PITCH"), &p);
96  p = sp.r * yaw_scale + yaw_trim_active;
97  int p3r = param_set(param_find("TRIM_YAW"), &p);
98 
99  if (p1r != 0 || p2r != 0 || p3r != 0) {
100  mavlink_log_critical(mavlink_log_pub, "TRIM: PARAM SET FAIL");
101  px4_close(sub_man);
102  return PX4_ERROR;
103  }
104 
105  mavlink_log_info(mavlink_log_pub, "trim cal done");
106  px4_close(sub_man);
107  return PX4_OK;
108 }
