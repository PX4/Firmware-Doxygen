79 void LandDetector::_cycle()
80 {
81  perf_begin(_cycle_perf);
82 
83  if (_object == nullptr) { // not initialized yet
84  // Advertise the first land detected uORB.
85  _landDetected.timestamp = hrt_absolute_time();
86  _landDetected.freefall = false;
87  _landDetected.landed = true;
88  _landDetected.ground_contact = false;
89  _landDetected.maybe_landed = false;
90 
91  _p_total_flight_time_high = param_find("LND_FLIGHT_T_HI");
92  _p_total_flight_time_low = param_find("LND_FLIGHT_T_LO");
93 
94  // Initialize uORB topics.
95  _armingSub = orb_subscribe(ORB_ID(actuator_armed));
96  _initialize_topics();
97 
98  _check_params(true);
99 
100  _object = this;
101  }
102 
103  _check_params(false);
104  _orb_update(ORB_ID(actuator_armed), _armingSub, &_arming);
105  _update_topics();
106  _update_state();
107 
108  const bool landDetected = (_state == LandDetectionState::LANDED);
109  const bool freefallDetected = (_state == LandDetectionState::FREEFALL);
110  const bool maybe_landedDetected = (_state == LandDetectionState::MAYBE_LANDED);
111  const bool ground_contactDetected = (_state == LandDetectionState::GROUND_CONTACT);
112  const float alt_max = _get_max_altitude() > 0.0f ? _get_max_altitude() : INFINITY;
113 
114  const hrt_abstime now = hrt_absolute_time();
115 
116  // publish at 1 Hz, very first time, or when the result has changed
117  if ((hrt_elapsed_time(&_landDetected.timestamp) >= 1_s) ||
118  (_landDetectedPub == nullptr) ||
119  (_landDetected.landed != landDetected) ||
120  (_landDetected.freefall != freefallDetected) ||
121  (_landDetected.maybe_landed != maybe_landedDetected) ||
122  (_landDetected.ground_contact != ground_contactDetected) ||
123  (fabsf(_landDetected.alt_max - alt_max) > FLT_EPSILON)) {
124 
125  if (!landDetected && _landDetected.landed) {
126  // We did take off
127  _takeoff_time = now;
128  }
129 
130  _landDetected.timestamp = hrt_absolute_time();
131  _landDetected.landed = landDetected;
132  _landDetected.freefall = freefallDetected;
133  _landDetected.maybe_landed = maybe_landedDetected;
134  _landDetected.ground_contact = ground_contactDetected;
135  _landDetected.alt_max = alt_max;
136 
137  int instance;
138  orb_publish_auto(ORB_ID(vehicle_land_detected), &_landDetectedPub, &_landDetected,
139  &instance, ORB_PRIO_DEFAULT);
140  }
141 
142  // set the flight time when disarming (not necessarily when landed, because all param changes should
143  // happen on the same event and it's better to set/save params while not in armed state)
144  if (_takeoff_time != 0 && !_arming.armed && _previous_arming_state) {
145  _total_flight_time += now - _takeoff_time;
146  _takeoff_time = 0;
147  uint32_t flight_time = (_total_flight_time >> 32) & 0xffffffff;
148  param_set_no_notification(_p_total_flight_time_high, &flight_time);
149  flight_time = _total_flight_time & 0xffffffff;
150  param_set_no_notification(_p_total_flight_time_low, &flight_time);
151  }
152 
153  _previous_arming_state = _arming.armed;
154 
155  perf_end(_cycle_perf);
156 
157  if (!should_exit()) {
158 
159  // Schedule next cycle.
160  work_queue(HPWORK, &_work, (worker_t)&LandDetector::_cycle_trampoline, this,
161  USEC2TICK(1_s / LAND_DETECTOR_UPDATE_RATE_HZ));
162 
163  } else {
164  exit_and_cleanup();
165  }
166 }
167 void LandDetector::_check_params(const bool force)
168 {
169  bool updated;
170  parameter_update_s paramUpdate;
171 
172  orb_check(_parameterSub, &updated);
173 
174  if (updated) {
175  orb_copy(ORB_ID(parameter_update), _parameterSub, &paramUpdate);
176  }
177 
178  if (updated || force) {
179  _update_params();
180  uint32_t flight_time;
181  param_get(_p_total_flight_time_high, (int32_t *)&flight_time);
182  _total_flight_time = ((uint64_t)flight_time) << 32;
183  param_get(_p_total_flight_time_low, (int32_t *)&flight_time);
184  _total_flight_time |= flight_time;
185  }
186 }

188 void LandDetector::_update_state()
189 {
190  /* when we are landed we also have ground contact for sure but only one output state can be true at a particular time
191  * with higher priority for landed */
192  _freefall_hysteresis.set_state_and_update(_get_freefall_state());
193  _landed_hysteresis.set_state_and_update(_get_landed_state());
194  _maybe_landed_hysteresis.set_state_and_update(_get_maybe_landed_state());
195  _ground_contact_hysteresis.set_state_and_update(_get_ground_contact_state());
196 
197  if (_freefall_hysteresis.get_state()) {
198  _state = LandDetectionState::FREEFALL;
199 
200  } else if (_landed_hysteresis.get_state()) {
201  _state = LandDetectionState::LANDED;
202 
203  } else if (_maybe_landed_hysteresis.get_state()) {
204  _state = LandDetectionState::MAYBE_LANDED;
205 
206  } else if (_ground_contact_hysteresis.get_state()) {
207  _state = LandDetectionState::GROUND_CONTACT;
208 
209  } else {
210  _state = LandDetectionState::FLYING;
211  }
212 }

214 bool LandDetector::_orb_update(const struct orb_metadata *meta, int handle, void *buffer)
215 {
216  bool newData = false;
217 
218  // check if there is new data to grab
219  if (orb_check(handle, &newData) != OK) {
220  return false;
221  }
222 
223  if (!newData) {
224  return false;
225  }
226 
227  if (orb_copy(meta, handle, buffer) != OK) {
228  return false;
229  }
230 
231  return true;
232 }
