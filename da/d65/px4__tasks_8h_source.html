107 // Fast drivers - they need to run as quickly as possible to minimize control
108 // latency.
109 #define SCHED_PRIORITY_FAST_DRIVER (SCHED_PRIORITY_MAX - 0)
110 
111 // Attitude controllers typically are in a blocking wait on driver data
112 // they should be the first to run on an update, using the current sensor
113 // data and the *previous* attitude reference from the position controller
114 // which typically runs at a slower rate
115 #define SCHED_PRIORITY_ATTITUDE_CONTROL (SCHED_PRIORITY_MAX - 4)
116 
117 // Actuator outputs should run before right after the attitude controller
118 // updated
119 #define SCHED_PRIORITY_ACTUATOR_OUTPUTS (SCHED_PRIORITY_MAX - 4)
120 
121 // Position controllers typically are in a blocking wait on estimator data
122 // so when new sensor data is available they will run last. Keeping them
123 // on a high priority ensures that they are the first process to be run
124 // when the estimator updates.
125 #define SCHED_PRIORITY_POSITION_CONTROL (SCHED_PRIORITY_MAX - 5)
126 
127 // Estimators should run after the attitude controller but before anything
128 // else in the system. They wait on sensor data which is either coming
129 // from the sensor hub or from a driver. Keeping this class at a higher
130 // priority ensures that the estimator runs first if it can, but will
131 // wait for the sensor hub if its data is coming from it.
132 #define SCHED_PRIORITY_ESTIMATOR (SCHED_PRIORITY_MAX - 5)
133 
134 // The sensor hub conditions sensor data. It is not the fastest component
135 // in the controller chain, but provides easy-to-use data to the more
136 // complex downstream consumers
137 #define SCHED_PRIORITY_SENSOR_HUB (SCHED_PRIORITY_MAX - 6)
138 
139 // The log capture (which stores log data into RAM) should run faster
140 // than other components, but should not run before the control pipeline
141 #define SCHED_PRIORITY_LOG_CAPTURE (SCHED_PRIORITY_MAX - 10)
142 
143 // Slow drivers should run at a rate where they do not impact the overall
144 // system execution
145 #define SCHED_PRIORITY_SLOW_DRIVER (SCHED_PRIORITY_MAX - 35)
146 
147 // The navigation system needs to execute regularly but has no realtime needs
