149 void MultirotorMixer::mix()
150 {
151  float roll = math::constrain(inputs.control[0], -1.0f, 1.0f);
152  float pitch = math::constrain(inputs.control[1], -1.0f, 1.0f);
153  float yaw = math::constrain(inputs.control[2], -1.0f, 1.0f);
154  float thrust = math::constrain(inputs.control[3], 0.0f, 1.0f);
155  float min_out = 0.0f;
156  float max_out = 0.0f;
157 
158  /* perform initial mix pass yielding unbounded outputs, ignore yaw */
159  for (unsigned i = 0; i < _rotor_count; i++) {
160  float out = roll * _rotors[i].roll_scale
161  + pitch * _rotors[i].pitch_scale + thrust;
162 
163  /* limit yaw if it causes outputs clipping */
164  if (out >= 0.0f && out < -yaw * _rotors[i].yaw_scale) {
165  yaw = -out / _rotors[i].yaw_scale;
166  }
167 
168  /* calculate min and max output values */
169  if (out < min_out) {
170  min_out = out;
171  }
172 
173  if (out > max_out) {
174  max_out = out;
175  }
176 
177  outputs.control[i] = out;
178  }
179 
180  /* scale down roll/pitch controls if some outputs are negative, don't add yaw, keep total thrust */
181  if (min_out < 0.0f) {
182  float scale_in = thrust / (thrust - min_out);
183 
184  /* mix again with adjusted controls */
185  for (unsigned i = 0; i < _rotor_count; i++) {
186  outputs.control[i] = scale_in
187  * (roll * _rotors[i].roll_scale
188  + pitch * _rotors[i].pitch_scale) + thrust;
189  }
190 
191  } else {
192  /* roll/pitch mixed without limiting, add yaw control */
193  for (unsigned i = 0; i < _rotor_count; i++) {
194  outputs.control[i] += yaw * _rotors[i].yaw_scale;
195  }
196  }
197 
198  /* scale down all outputs if some outputs are too large, reduce total thrust */
199  float scale_out;
200 
201  if (max_out > 1.0f) {
202  scale_out = 1.0f / max_out;
203 
204  } else {
205  scale_out = 1.0f;
206  }
207 
208  /* scale outputs to range _idle_speed..1, and do final limiting */
209  for (unsigned i = 0; i < _rotor_count; i++) {
210  outputs.control[i] = math::constrain(outputs.control[i], 0.0f, 1.0f);
211  }
212 }

214 void MultirotorMixer::actuatorControlsCallback(const px4::actuator_controls_0 &msg)
215 {
216  // read message
217  for (int i = 0; i < msg.NUM_ACTUATOR_CONTROLS; i++) {
218  inputs.control[i] = msg.control[i];
219  }
220 
221  // switch mixer if necessary
222  std::string mixer_name;
223  _n.getParamCached("mixer", mixer_name);
224 
225  if (mixer_name == "x") {
226  _rotors = _config_index[0];
227 
228  } else if (mixer_name == "+") {
229  _rotors = _config_index[1];
230 
231  } else if (mixer_name == "e") {
232  _rotors = _config_index[2];
233 
234  } else if (mixer_name == "w") {
235  _rotors = _config_index[3];
236 
237  } else if (mixer_name == "i") {
238  _rotors = _config_index[4];
239  }
240 
241  // mix
242  mix();
243 
244  // publish message
245  mav_msgs::CommandMotorSpeed rotor_vel_msg;
246  double scaling;
247  double offset;
248  _n.getParamCached("motor_scaling_radps", scaling);
249  _n.getParamCached("motor_offset_radps", offset);
250 
251  if (_armed) {
252  for (int i = 0; i < _rotor_count; i++) {
253  rotor_vel_msg.motor_speed.push_back(outputs.control[i] * scaling + offset);
254  }
255 
256  } else {
257  for (int i = 0; i < _rotor_count; i++) {
258  rotor_vel_msg.motor_speed.push_back(0.0);
259  }
260  }
261 
262  _pub.publish(rotor_vel_msg);
263 }
