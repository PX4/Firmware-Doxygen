124 /************************************************************************************
125  * Name: board_on_reset
126  *
127  * Description:
128  * Optionally provided function called on entry to board_system_reset
129  * It should perform any house keeping prior to the rest.
130  *
131  * status - 1 if resetting to boot loader
132  * 0 if just resetting
133  *
134  ************************************************************************************/
135 __EXPORT void board_on_reset(int status)
136 {
137  // Configure the GPIO pins to outputs and keep them low.
138  stm32_configgpio(GPIO_GPIO0_OUTPUT);
139  stm32_configgpio(GPIO_GPIO1_OUTPUT);
140  stm32_configgpio(GPIO_GPIO2_OUTPUT);
141  stm32_configgpio(GPIO_GPIO3_OUTPUT);
142  stm32_configgpio(GPIO_GPIO4_OUTPUT);
143  stm32_configgpio(GPIO_GPIO5_OUTPUT);
144 
145  /**
146  * On resets invoked from system (not boot) insure we establish a low
147  * output state (discharge the pins) on PWM pins before they become inputs.
148  */
149 
150  if (status >= 0) {
151  up_mdelay(400);
152  }
153 }

161 __EXPORT void board_peripheral_reset(int ms)
162 {
163  /* set the peripheral rails off */
164  stm32_configgpio(GPIO_VDD_5V_PERIPH_EN);
165  stm32_gpiowrite(GPIO_VDD_5V_PERIPH_EN, 1);
166 
167  /* wait for the peripheral rail to reach GND */
168  usleep(ms * 1000);
169  warnx("reset done, %d ms", ms);
170 
171  /* re-enable power */
172 
173  /* switch the peripheral rail back on */
174  stm32_gpiowrite(GPIO_VDD_5V_PERIPH_EN, 0);
175 }

228 /****************************************************************************
229  * Name: board_app_initialize
230  *
231  * Description:
232  * Perform application specific initialization. This function is never
233  * called directly from application code, but only indirectly via the
234  * (non-standard) boardctl() interface using the command BOARDIOC_INIT.
235  *
236  * Input Parameters:
237  * arg - The boardctl() argument is passed to the board_app_initialize()
238  * implementation without modification. The argument has no
239  * meaning to NuttX; the meaning of the argument is a contract
240  * between the board-specific initalization logic and the the
241  * matching application logic. The value cold be such things as a
242  * mode enumeration value, a set of DIP switch switch settings, a
243  * pointer to configuration data read from a file or serial FLASH,
244  * or whatever you would like to do with it. Every implementation
245  * should accept zero/NULL as a default configuration.
246  *
247  * Returned Value:
248  * Zero (OK) is returned on success; a negated errno value is returned on
249  * any failure to indicate the nature of the failure.
250  *
251  ****************************************************************************/

321 
322  /* Panic Logging in Battery Backed Up Files */
323 
324  /*
325  * In an ideal world, if a fault happens in flight the
326  * system save it to BBSRAM will then reboot. Upon
327  * rebooting, the system will log the fault to disk, recover
328  * the flight state and continue to fly. But if there is
329  * a fault on the bench or in the air that prohibit the recovery
330  * or committing the log to disk, the things are too broken to
331  * fly. So the question is:
332  *
333  * Did we have a hard fault and not make it far enough
334  * through the boot sequence to commit the fault data to
335  * the SD card?
336  */
337 
338  /* Do we have an uncommitted hard fault in BBSRAM?
339  * - this will be reset after a successful commit to SD
340  */
341  int hadCrash = hardfault_check_status("boot");
342 
343  if (hadCrash == OK) {
344 
345  message("[boot] There is a hard fault logged. Hold down the SPACE BAR," \
346  " while booting to halt the system!\n");
347

349  * commit to SD
350  */
351 
352  int reboots = hardfault_increment_reboot("boot", false);
353 
354  /* Also end the misery for a user that holds for a key down on the console */
355 
356  int bytesWaiting;
357  ioctl(fileno(stdin), FIONREAD, (unsigned long)((uintptr_t) &bytesWaiting));
358 
359  if (reboots > 2 || bytesWaiting != 0) {
360 
361  /* Since we can not commit the fault dump to disk. Display it
362  * to the console.
363  */
364 
365  hardfault_write("boot", fileno(stdout), HARDFAULT_DISPLAY_FORMAT, false);
366 
367  message("[boot] There were %d reboots with Hard fault that were not committed to disk - System halted %s\n",
368  reboots,
369  (bytesWaiting == 0 ? "" : " Due to Key Press\n"));
370 
371 
372  /* For those of you with a debugger set a break point on up_assert and
373  * then set dbgContinue = 1 and go.
374  */
375 
376  /* Clear any key press that got us here */
377 
378  static volatile bool dbgContinue = false;
379  int c = '>';
380 
381  while (!dbgContinue) {
382 
383  switch (c) {
384

386

409  break;
410

414  break;
415 
416  default:
417  break;
418  } // Inner Switch
419 
420  message("\nEnter B - Continue booting\n" \
421  "Enter C - Clear the fault log\n" \
422  "Enter D - Dump fault log\n\n?>");
423  fflush(stdout);
424 
425  if (!dbgContinue) {
426  c = getchar();
427  }
428 
429  break;
430 
431  } // outer switch
432  } // for
433 
434  } // inner if
435  } // outer if
436  } // hardfault_check_status
437 
438 #endif // CONFIG_STM32_SAVE_CRASHDUMP
439 #endif // CONFIG_STM32_BBSRAM
