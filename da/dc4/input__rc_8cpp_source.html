77 int InputRC::update_impl(unsigned int timeout_ms, ControlData **control_data, bool already_active)
78 {
79  // Default to no change signalled by NULL.
80  *control_data = nullptr;
81 
82  px4_pollfd_struct_t polls[1];
83  polls[0].fd = _manual_control_setpoint_sub;
84  polls[0].events = POLLIN;
85 
86  int ret = px4_poll(polls, 1, timeout_ms);
87 
88  if (ret < 0) {
89  return -errno;
90  }
91 
92  if (ret == 0) {
93  // Timeout, leave NULL
94  } else {
95  if (polls[0].revents & POLLIN) {
96  // Only if there was a change, we update the control data, otherwise leave it NULL.
97  if (_read_control_data_from_subscription(_control_data, already_active)) {
98  *control_data = &_control_data;
99  }
100  }
101  }
102 
103  return 0;
104 }

106 bool InputRC::_read_control_data_from_subscription(ControlData &control_data, bool already_active)
107 {
108  manual_control_setpoint_s manual_control_setpoint;
109  orb_copy(ORB_ID(manual_control_setpoint), _manual_control_setpoint_sub, &manual_control_setpoint);
110  control_data.type = ControlData::Type::Angle;
111 
112  float new_aux_values[3];
113 
114  for (int i = 0; i < 3; ++i) {
115  new_aux_values[i] = _get_aux_value(manual_control_setpoint, i);
116  }
117 
118  // If we were already active previously, we just update normally. Otherwise, there needs to be
119  // a major stick movement to re-activate manual (or it's running for the very first time).
120  bool major_movement = false;
121 
122  // Detect a big stick movement
123  for (int i = 0; i < 3; ++i) {
124  if (fabsf(_last_set_aux_values[i] - new_aux_values[i]) > 0.25f) {
125  major_movement = true;
126  }
127  }
128 
129  if (already_active || major_movement || _first_time) {
130 
131  _first_time = false;
132 
133  for (int i = 0; i < 3; ++i) {
134  control_data.type_data.angle.is_speed[i] = false;
135  control_data.type_data.angle.angles[i] = new_aux_values[i] * M_PI_F;
136  control_data.stabilize_axis[i] = _do_stabilization;
137 
138  _last_set_aux_values[i] = new_aux_values[i];
139  }
140 
141  control_data.gimbal_shutter_retract = false;
142  return true;
143 
144  } else {
145  return false;
146  }
147 }
