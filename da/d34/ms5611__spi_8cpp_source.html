176 int
177 MS5611_SPI::read(unsigned offset, void *data, unsigned count)
178 {
179  union _cvt {
180  uint8_t b[4];
181  uint32_t w;
182  } *cvt = (_cvt *)data;
183  uint8_t buf[4] = { 0 | DIR_WRITE, 0, 0, 0 };
184 
185  /* read the most recent measurement */
186  int ret = _transfer(&buf[0], &buf[0], sizeof(buf));
187 
188  if (ret == OK) {
189  /* fetch the raw value */
190  cvt->b[0] = buf[3];
191  cvt->b[1] = buf[2];
192  cvt->b[2] = buf[1];
193  cvt->b[3] = 0;
194 
195  ret = count;
196  }
197 
198  return ret;
199 }

244 int
245 MS5611_SPI::_read_prom()
246 {
247  /*
248  * Wait for PROM contents to be in the device (2.8 ms) in the case we are
249  * called immediately after reset.
250  */
251  usleep(3000);
252 
253  /* read and convert PROM words */
254  bool all_zero = true;
255 
256  for (int i = 0; i < 8; i++) {
257  uint8_t cmd = (ADDR_PROM_SETUP + (i * 2));
258  _prom.c[i] = _reg16(cmd);
259 
260  if (_prom.c[i] != 0) {
261  all_zero = false;
262  }
263 
264  //DEVICE_DEBUG("prom[%u]=0x%x", (unsigned)i, (unsigned)_prom.c[i]);
265  }
266 
267  /* calculate CRC and return success/failure accordingly */
268  int ret = ms5611::crc4(&_prom.c[0]) ? OK : -EIO;
269 
270  if (ret != OK) {
271  DEVICE_DEBUG("crc failed");
272  }
273 
274  if (all_zero) {
275  DEVICE_DEBUG("prom all zero");
276  ret = -EIO;
277  }
278 
279  return ret;
280 }

283 MS5611_SPI::_reg16(unsigned reg)
284 {
