70 {

82 
83  return res;

86 void UavcanEscController::update_outputs(float *outputs, unsigned num_outputs)
87 {
88  if ((outputs == nullptr) ||
89  (num_outputs > uavcan::equipment::esc::RawCommand::FieldTypes::cmd::MaxSize) ||
90  (num_outputs > esc_status_s::CONNECTED_ESC_MAX)) {
91  perf_count(_perfcnt_invalid_input);
92  return;
93  }
94 
95  /*
96  * Rate limiting - we don't want to congest the bus
97  */
98  const auto timestamp = _node.getMonotonicTime();
99 
100  if ((timestamp - _prev_cmd_pub).toUSec() < (1000000 / MAX_RATE_HZ)) {
101  return;
102  }
103 
104  _prev_cmd_pub = timestamp;
105 
106  /*
107  * Fill the command message
108  * If unarmed, we publish an empty message anyway
109  */
110  uavcan::equipment::esc::RawCommand msg;
111 
112  actuator_outputs_s actuator_outputs = {};
113  actuator_outputs.noutputs = num_outputs;
114  actuator_outputs.timestamp = hrt_absolute_time();
115 
116  static const int cmd_max = uavcan::equipment::esc::RawCommand::FieldTypes::cmd::RawValueType::max();
117  const float cmd_min = _run_at_idle_throttle_when_armed ? 1.0F : 0.0F;
118 
119  for (unsigned i = 0; i < num_outputs; i++) {
120  if (_armed_mask & MOTOR_BIT(i)) {
121  float scaled = (outputs[i] + 1.0F) * 0.5F * cmd_max;
122 
123  // trim negative values back to minimum
124  if (scaled < cmd_min) {
125  scaled = cmd_min;
126  perf_count(_perfcnt_scaling_error);
127  }
128 
129  if (scaled > cmd_max) {
130  scaled = cmd_max;
131  perf_count(_perfcnt_scaling_error);
132  }
133 
134  msg.cmd.push_back(static_cast<int>(scaled));
135 
136  _esc_status.esc[i].esc_setpoint_raw = abs(static_cast<int>(scaled));
137  actuator_outputs.output[i] = scaled;
138 
139  } else {
140  msg.cmd.push_back(static_cast<unsigned>(0));
141  actuator_outputs.output[i] = 0.0f;
142  }
143  }
144 
145  /*
146  * Remove channels that are always zero.
147  * The objective of this optimization is to avoid broadcasting multi-frame transfers when a single frame
148  * transfer would be enough. This is a valid optimization as the UAVCAN specification implies that all
149  * non-specified ESC setpoints should be considered zero.
150  * The positive outcome is a (marginally) lower bus traffic and lower CPU load.
151  *
152  * From the standpoint of the PX4 architecture, however, this is a hack. It should be investigated why
153  * the mixer returns more outputs than are actually used.
154  */
155  for (int index = int(msg.cmd.size()) - 1; index >= _max_number_of_nonzero_outputs; index--) {
156  if (msg.cmd[index] != 0) {
157  _max_number_of_nonzero_outputs = index + 1;
158  break;
159  }
160  }
161 
162  msg.cmd.resize(_max_number_of_nonzero_outputs);
163 
164  /*
165  * Publish the command message to the bus
166  * Note that for a quadrotor it takes one CAN frame
167  */
168  (void)_uavcan_pub_raw_cmd.broadcast(msg);
169 
170  // Publish actuator outputs
171  if (_actuator_outputs_pub != nullptr) {
172  orb_publish(ORB_ID(actuator_outputs), _actuator_outputs_pub, &actuator_outputs);
173 
174  } else {
175  int instance;
176  _actuator_outputs_pub = orb_advertise_multi(ORB_ID(actuator_outputs), &actuator_outputs,
177  &instance, ORB_PRIO_DEFAULT);
178  }
179 
180 }
