78 int up_pwm_servo_init(uint32_t channel_mask)
79 {
80  /* Init channels */
81  uint32_t current = io_timer_get_mode_channels(IOTimerChanMode_PWMOut);
82 
83  // First free the current set of PWMs
84 
85  for (unsigned channel = 0; current != 0 && channel < MAX_TIMER_IO_CHANNELS; channel++) {
86  if (current & (1 << channel)) {
87  io_timer_free_channel(channel);
88  current &= ~(1 << channel);
89  }
90  }
91 
92  // Now allocate the new set
93 
94  for (unsigned channel = 0; channel_mask != 0 && channel < MAX_TIMER_IO_CHANNELS; channel++) {
95  if (channel_mask & (1 << channel)) {
96 
97  // First free any that were not PWM mode before
98 
99  if (-EBUSY == io_timer_is_channel_free(channel)) {
100  io_timer_free_channel(channel);
101  }
102 
103  io_timer_channel_init(channel, IOTimerChanMode_PWMOut, NULL, NULL);
104  channel_mask &= ~(1 << channel);
105  }
106  }
107 
108  return OK;
109 }

117 int up_pwm_servo_set_rate_group_update(unsigned group, unsigned rate)
118 {
119  /* limit update rate to 1..10000Hz; somewhat arbitrary but safe */
120  if (rate < 1) {
121  return -ERANGE;
122  }
123 
124  if (rate > 10000) {
125  return -ERANGE;
126  }
127 
128  if ((group >= MAX_IO_TIMERS) || (io_timers[group].base == 0)) {
129  return ERROR;
130  }
131 
132  io_timer_set_rate(group, rate);
133 
134  return OK;
135 }
