172  static leddar_one *inst = nullptr;
173 
174  while ((ch = px4_getopt(argc, argv, "d:r", &myoptind, &myoptarg)) != EOF) {
175  switch (ch) {
176  case 'd':
177  serial_port = myoptarg;
178  break;
179 
180  case 'r':
181  rotation = (uint8_t)atoi(myoptarg);
182  break;
183 
184  default:
185  help();
186  return PX4_ERROR;
187  }
188  }
189 
190  if (myoptind >= argc) {
191  help();
192  return PX4_ERROR;
193  }
194 
195  const char *verb = argv[myoptind];
196 
197  if (!strcmp(verb, "start")) {
198  inst = new leddar_one(DEVICE_PATH, serial_port, rotation);
199 
200  if (!inst) {
201  PX4_ERR("No memory to allocate " NAME);
202  return PX4_ERROR;
203  }
204 
205  if (inst->init() != PX4_OK) {
206  delete inst;
207  return PX4_ERROR;
208  }
209 
210  inst->start();
211 
212  } else if (!strcmp(verb, "stop")) {
213  delete inst;
214  inst = nullptr;
215 
216  } else if (!strcmp(verb, "test")) {

278 int leddar_one::_fd_open()
279 {
280  if (!_serial_port) {
281  return -1;
282  }
283 
284  _fd = ::open(_serial_port, O_RDWR | O_NOCTTY | O_NONBLOCK);
285 
286  if (_fd < 0) {
287  return -1;
288  }
289 
290  struct termios config;
291 
292  int r = tcgetattr(_fd, &config);
293 
294  if (r) {
295  PX4_ERR("Unable to get termios from %s.", _serial_port);
296  goto error;
297  }
298 
299  /* clear: data bit size, two stop bits, parity, hardware flow control */
300  config.c_cflag &= ~(CSIZE | CSTOPB | PARENB | CRTSCTS);
301  /* set: 8 data bits, enable receiver, ignore modem status lines */
302  config.c_cflag |= (CS8 | CREAD | CLOCAL);
303  /* turn off output processing */
304  config.c_oflag = 0;
305  /* clear: echo, echo new line, canonical input and extended input */
306  config.c_lflag &= (ECHO | ECHONL | ICANON | IEXTEN);
307 
308  r = cfsetispeed(&config, B115200);
309  r |= cfsetospeed(&config, B115200);
310 
311  if (r) {
312  PX4_ERR("Unable to set baudrate");
313  goto error;
314  }
315 
316  r = tcsetattr(_fd, TCSANOW, &config);
317 
318  if (r) {
319  PX4_ERR("Unable to set termios to %s", _serial_port);
320  goto error;
321  }
322 
323  return PX4_OK;
324 
325 error:
326  ::close(_fd);
327  _fd = -1;
328  return PX4_ERROR;
329 }

436 static uint16_t _calc_crc16(const uint8_t *buffer, uint8_t len)
437 {
438  uint16_t crc = 0xFFFF;
439 
440  for (uint8_t i = 0; i < len; i++) {
441  crc ^= buffer[i];
442 
443  for (uint8_t j = 0; j < 8; j++) {
444  if (crc & 1) {
445  crc = (crc >> 1) ^ 0xA001;
446 
447  } else {
448  crc >>= 1;
449  }
450  }
451  }
452 
453  return crc;
454 }

460 int leddar_one::_collect()
461 {
462  struct reading_msg *msg;
463 
464  int r = ::read(_fd, _buffer + _buffer_len, sizeof(_buffer) - _buffer_len);
465 
466  if (r < 1) {
467  return 0;
468  }
469 
470  _buffer_len += r;
471 
472  if (_buffer_len < sizeof(struct reading_msg)) {
473  return 0;
474  }
475 
476  msg = (struct reading_msg *)_buffer;
477 
478  if (msg->slave_addr != MODBUS_SLAVE_ADDRESS || msg->function != MODBUS_READING_FUNCTION) {
479  return -1;
480  }
481 
482  const uint16_t crc16_calc = _calc_crc16(_buffer, _buffer_len - 2);
483 
484  if (crc16_calc != msg->crc) {
485  return -1;
486  }
487 
488  /* NOTE: little-endian support only */
489  _publish(msg->first_dist_high_byte << 8 | msg->first_dist_low_byte);
490  return 1;
491 }

493 int leddar_one::_cycle()
494 {
495  int ret = 0;
496  const hrt_abstime now = hrt_absolute_time();
497 
498  switch (_state) {
499  case state_waiting_reading:
500  if (now > _timeout_usec) {
501  if (_request()) {
502  perf_begin(_sample_perf);
503  _buffer_len = 0;
504  _state = state_reading_requested;
505  _timeout_usec = now + COLLECT_USEC_TIMEOUT;
506  }
507  }
508 
509  break;
510 
511  case state_reading_requested:
512  ret = _collect();
513 
514  if (ret == 1) {
515  perf_end(_sample_perf);
516  _state = state_waiting_reading;
517  _timeout_usec = now + READING_USEC_PERIOD;
518 
519  } else {
520  if (ret == 0 && now < _timeout_usec) {
521  /* still waiting for reading */
522  break;
523  }
524 
525  if (ret == -1) {
526  perf_count(_comm_error);
527 
528  } else {
529  perf_count(_collect_timeout_perf);
530  }
531 
532  perf_cancel(_sample_perf);
533  _state = state_waiting_reading;
534  _timeout_usec = 0;
535  return _cycle();
536  }
537  }
538 
539  return ret;
540 }
