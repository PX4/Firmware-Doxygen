50 enum class GPSCallbackType {
51  /**
52  * Read data from device. This is a blocking operation with a timeout.
53  * data1: points to a buffer to be written to. The first sizeof(int) bytes contain the
54  * timeout in ms when calling the method.
55  * data2: buffer length in bytes. Less bytes than this can be read.
56  * return: num read bytes, 0 on timeout (the method can actually also return 0 before
57  * the timeout happens).
58  */
59  readDeviceData = 0,
60 
61  /**
62  * Write data to device
63  * data1: data to be written
64  * data2: number of bytes to write
65  * return: num written bytes
66  */
67  writeDeviceData,
68 
69  /**
70  * set Baudrate
71  * data1: ignored
72  * data2: baudrate
73  * return: 0 on success
74  */
75  setBaudrate,
76 
77  /**
78  * Got an RTCM message from the device.
79  * data1: pointer to the message
80  * data2: message length
81  * return: ignored
82  */
83  gotRTCMMessage,
84 
85  /**
86  * message about current survey-in status
87  * data1: points to a SurveyInStatus struct
88  * data2: ignored
89  * return: ignored
90  */
91  surveyInStatus,

93  /**
94  * can be used to set the current clock accurately

96  * data2: ignored

99  setClock,
