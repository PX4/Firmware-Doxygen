59 bool Ekf::collect_gps(uint64_t time_usec, struct gps_message *gps)
60 {
61  // Run GPS checks always
62  bool gps_checks_pass = gps_is_good(gps);
63  if (!_NED_origin_initialised && gps_checks_pass) {
64  // If we have good GPS data set the origin's WGS-84 position to the last gps fix
65  double lat = gps->lat / 1.0e7;
66  double lon = gps->lon / 1.0e7;
67  map_projection_init_timestamped(&_pos_ref, lat, lon, _time_last_imu);
68 
69  // if we are already doing aiding, corect for the change in posiiton since the EKF started navigating
70  if (_control_status.flags.opt_flow || _control_status.flags.gps || _control_status.flags.ev_pos) {
71  double est_lat, est_lon;
72  map_projection_reproject(&_pos_ref, -_state.pos(0), -_state.pos(1), &est_lat, &est_lon);
73  map_projection_init_timestamped(&_pos_ref, est_lat, est_lon, _time_last_imu);
74  }
75 
76  // Take the current GPS height and subtract the filter height above origin to estimate the GPS height of the origin
77  _gps_alt_ref = 1e-3f * (float)gps->alt + _state.pos(2);
78  _NED_origin_initialised = true;
79  _last_gps_origin_time_us = _time_last_imu;
80  // set the magnetic declination returned by the geo library using the current GPS position
81  _mag_declination_gps = math::radians(get_mag_declination(lat, lon));
82  // save the horizontal and vertical position uncertainty of the origin
83  _gps_origin_eph = gps->eph;
84  _gps_origin_epv = gps->epv;
85 
86  // if the user has selected GPS as the primary height source, switch across to using it
87  if (_primary_hgt_source == VDIST_SENSOR_GPS) {
88  ECL_INFO("EKF GPS checks passed (WGS-84 origin set, using GPS height)");
89  _control_status.flags.baro_hgt = false;
90  _control_status.flags.gps_hgt = true;
91  _control_status.flags.rng_hgt = false;
92  // zero the sensor offset
93  _hgt_sensor_offset = 0.0f;
94  } else {
95  ECL_INFO("EKF GPS checks passed (WGS-84 origin set)");
96  }
97  }
98 
99  // start collecting GPS if there is a 3D fix and the NED origin has been set
100  return _NED_origin_initialised && (gps->fix_type >= 3);
101 }

103 /*
104  * Return true if the GPS solution quality is adequate to set an origin for the EKF
105  * and start GPS aiding.
106  * All activated checks must pass for 10 seconds.
107  * Checks are activated using the EKF2_GPS_CHECK bitmask parameter
108  * Checks are adjusted using the EKF2_REQ_* parameters
109 */
110 bool Ekf::gps_is_good(struct gps_message *gps)
111 {
112  // Check the fix type
113  _gps_check_fail_status.flags.fix = (gps->fix_type < 3);
114 
115  // Check the number of satellites
116  _gps_check_fail_status.flags.nsats = (gps->nsats < _params.req_nsats);
117 
118  // Check the geometric dilution of precision
119  _gps_check_fail_status.flags.gdop = (gps->gdop > _params.req_gdop);
120 
121  // Check the reported horizontal and vertical position accuracy
122  _gps_check_fail_status.flags.hacc = (gps->eph > _params.req_hacc);
123  _gps_check_fail_status.flags.vacc = (gps->epv > _params.req_vacc);
124 
125  // Check the reported speed accuracy
126  _gps_check_fail_status.flags.sacc = (gps->sacc > _params.req_sacc);
127 
128  // check if GPS quality is degraded
129  _gps_error_norm = fmaxf((gps->eph / _params.req_hacc) , (gps->epv / _params.req_vacc));
130  _gps_error_norm = fmaxf(_gps_error_norm , (gps->sacc / _params.req_sacc));
131 
132  // Calculate time lapsed since last update, limit to prevent numerical errors and calculate a lowpass filter coefficient
133  const float filt_time_const = 10.0f;
134  float dt = fminf(fmaxf(float(_time_last_imu - _gps_pos_prev.timestamp) * 1e-6f, 0.001f), filt_time_const);
135  float filter_coef = dt / filt_time_const;
136 
137  // The following checks are only valid when the vehicle is at rest
138  double lat = gps->lat * 1.0e-7;
139  double lon = gps->lon * 1.0e-7;
140  if (!_control_status.flags.in_air && _vehicle_at_rest) {
141  // Calculate position movement since last measurement
142  float delta_posN = 0.0f;
143  float delta_PosE = 0.0f;
144 
145  // calculate position movement since last GPS fix
146  if (_gps_pos_prev.timestamp > 0) {
147  map_projection_project(&_gps_pos_prev, lat, lon, &delta_posN, &delta_PosE);
148 
149  } else {
150  // no previous position has been set
151  map_projection_init_timestamped(&_gps_pos_prev, lat, lon, _time_last_imu);
152  _gps_alt_prev = 1e-3f * (float)gps->alt;
153 
154  }
155 
156  // Calculate the horizontal drift velocity components and limit to 10x the threshold
157  float vel_limit = 10.0f * _params.req_hdrift;
158  float velN = fminf(fmaxf(delta_posN / dt, -vel_limit), vel_limit);
159  float velE = fminf(fmaxf(delta_PosE / dt, -vel_limit), vel_limit);
160 
161  // Apply a low pass filter
162  _gpsDriftVelN = velN * filter_coef + _gpsDriftVelN * (1.0f - filter_coef);
163  _gpsDriftVelE = velE * filter_coef + _gpsDriftVelE * (1.0f - filter_coef);
164 
165  // Calculate the horizontal drift speed and fail if too high
166  float drift_speed = sqrtf(_gpsDriftVelN * _gpsDriftVelN + _gpsDriftVelE * _gpsDriftVelE);
167  _gps_check_fail_status.flags.hdrift = (drift_speed > _params.req_hdrift);
168 
169  // Calculate the vertical drift velocity and limit to 10x the threshold
170  float vz_drift_limit = 10.0f * _params.req_vdrift;
171  float gps_alt_m = 1e-3f * (float)gps->alt;
172  float velD = math::constrain(((_gps_alt_prev - gps_alt_m) / dt), -vz_drift_limit, vz_drift_limit);
173 
174  // Apply a low pass filter to the vertical velocity
175  _gps_drift_velD = velD * filter_coef + _gps_drift_velD * (1.0f - filter_coef);
176 
177  // Fail if the vertical drift speed is too high
178  _gps_check_fail_status.flags.vdrift = (fabsf(_gps_drift_velD) > _params.req_vdrift);
179 
180  // Check the magnitude of the filtered horizontal GPS velocity
181  float vxy_drift_limit = 10.0f * _params.req_hdrift;
182  float gps_velN = fminf(fmaxf(gps->vel_ned[0], -vxy_drift_limit), vxy_drift_limit);
183  float gps_velE = fminf(fmaxf(gps->vel_ned[1], -vxy_drift_limit), vxy_drift_limit);
184  _gps_velN_filt = gps_velN * filter_coef + _gps_velN_filt * (1.0f - filter_coef);
185  _gps_velE_filt = gps_velE * filter_coef + _gps_velE_filt * (1.0f - filter_coef);
186  float horiz_speed = sqrtf(_gps_velN_filt * _gps_velN_filt + _gps_velE_filt * _gps_velE_filt);
187  _gps_check_fail_status.flags.hspeed = (horiz_speed > _params.req_hdrift);
188 
189  } else if (_control_status.flags.in_air) {
190  // These checks are always declared as passed when flying
191  // If on ground and moving, the last result before movemenent commenced is kept
192  _gps_check_fail_status.flags.hdrift = false;
193  _gps_check_fail_status.flags.vdrift = false;
194  _gps_check_fail_status.flags.hspeed = false;
195 
196  }
197 
198  // save GPS fix for next time
199  map_projection_init_timestamped(&_gps_pos_prev, lat, lon, _time_last_imu);
200  _gps_alt_prev = 1e-3f * (float)gps->alt;
201 
202  // Check the filtered difference between GPS and EKF vertical velocity
203  float vz_diff_limit = 10.0f * _params.req_vdrift;
204  float vertVel = fminf(fmaxf((gps->vel_ned[2] - _state.vel(2)), -vz_diff_limit), vz_diff_limit);
205  _gps_velD_diff_filt = vertVel * filter_coef + _gps_velD_diff_filt * (1.0f - filter_coef);
206  _gps_check_fail_status.flags.vspeed = (fabsf(_gps_velD_diff_filt) > _params.req_vdrift);
207 
208  // assume failed first time through
209  if (_last_gps_fail_us == 0) {
210  _last_gps_fail_us = _time_last_imu;
211  }
212 
213  // if any user selected checks have failed, record the fail time
214  if (
215  _gps_check_fail_status.flags.fix ||
216  (_gps_check_fail_status.flags.nsats && (_params.gps_check_mask & MASK_GPS_NSATS)) ||
217  (_gps_check_fail_status.flags.gdop && (_params.gps_check_mask & MASK_GPS_GDOP)) ||
218  (_gps_check_fail_status.flags.hacc && (_params.gps_check_mask & MASK_GPS_HACC)) ||
219  (_gps_check_fail_status.flags.vacc && (_params.gps_check_mask & MASK_GPS_VACC)) ||
220  (_gps_check_fail_status.flags.sacc && (_params.gps_check_mask & MASK_GPS_SACC)) ||
221  (_gps_check_fail_status.flags.hdrift && (_params.gps_check_mask & MASK_GPS_HDRIFT)) ||
222  (_gps_check_fail_status.flags.vdrift && (_params.gps_check_mask & MASK_GPS_VDRIFT)) ||
223  (_gps_check_fail_status.flags.hspeed && (_params.gps_check_mask & MASK_GPS_HSPD)) ||
224  (_gps_check_fail_status.flags.vspeed && (_params.gps_check_mask & MASK_GPS_VSPD))
225  ) {
226  _last_gps_fail_us = _time_last_imu;
227  } else {
228  _last_gps_pass_us = _time_last_imu;
229  }
230 
231  // continuous period without fail of 10 seconds required to return a healthy status
232  return (_time_last_imu - _last_gps_fail_us > (uint64_t)1e7);
233 }
