80 GroundRoverAttitudeControl::~GroundRoverAttitudeControl()
81 {
82  if (_control_task != -1) {
83 
84  /* task wakes up every 100ms or so at the longest */
85  _task_should_exit = true;
86 
87  /* wait for a second for the task to quit at our request */
88  unsigned i = 0;
89 
90  do {
91  /* wait 20ms */
92  usleep(20000);
93 
94  /* if we have given up, kill it */
95  if (++i > 50) {
96  px4_task_delete(_control_task);
97  break;
98  }
99  } while (_control_task != -1);
100  }
101 
102  perf_free(_loop_perf);
103  perf_free(_nonfinite_input_perf);
104  perf_free(_nonfinite_output_perf);
105 
106  att_gnd_control::g_control = nullptr;
107 }

184 void
185 GroundRoverAttitudeControl::task_main()
186 {
187  _att_sp_sub = orb_subscribe(ORB_ID(vehicle_attitude_setpoint));
188  _att_sub = orb_subscribe(ORB_ID(vehicle_attitude));
189  _vcontrol_mode_sub = orb_subscribe(ORB_ID(vehicle_control_mode));
190  _params_sub = orb_subscribe(ORB_ID(parameter_update));
191  _manual_sub = orb_subscribe(ORB_ID(manual_control_setpoint));
192  _battery_status_sub = orb_subscribe(ORB_ID(battery_status));
193 
194  parameters_update();
195 
196  /* get an initial update for all sensor and status data */
197  vehicle_attitude_setpoint_poll();
198  vehicle_control_mode_poll();
199  manual_control_setpoint_poll();
200  battery_status_poll();
201 
202  /* wakeup source */
203  px4_pollfd_struct_t fds[2];
204 
205  /* Setup of loop */
206  fds[0].fd = _params_sub;
207  fds[0].events = POLLIN;
208  fds[1].fd = _att_sub;
209  fds[1].events = POLLIN;
210 
211  _task_running = true;
212 
213  while (!_task_should_exit) {
214  static int loop_counter = 0;
215 
216  /* wait for up to 500ms for data */
217  int pret = px4_poll(&fds[0], (sizeof(fds) / sizeof(fds[0])), 100);
218 
219  /* timed out - periodic check for _task_should_exit, etc. */
220  if (pret == 0) {
221  continue;
222  }
223 
224  /* this is undesirable but not much we can do - might want to flag unhappy status */
225  if (pret < 0) {
226  warn("poll error %d, %d", pret, errno);
227  continue;
228  }
229 
230  perf_begin(_loop_perf);
231 
232  /* only update parameters if they changed */
233  if (fds[0].revents & POLLIN) {
234  /* read from param to clear updated flag */
235  struct parameter_update_s update;
236  orb_copy(ORB_ID(parameter_update), _params_sub, &update);
237 
238  /* update parameters from storage */
239  parameters_update();
240  }
241 
242  /* only run controller if attitude changed */
243  if (fds[1].revents & POLLIN) {
244  static uint64_t last_run = 0;
245  float deltaT = (hrt_absolute_time() - last_run) / 1000000.0f;
246  last_run = hrt_absolute_time();
247 
248  /* guard against too large deltaT's */
249  if (deltaT > 1.0f ||
250  fabsf(deltaT) < 0.00001f ||
251  !PX4_ISFINITE(deltaT)) {
252  deltaT = 0.01f;
253  }
254 
255  /* load local copies */
256  orb_copy(ORB_ID(vehicle_attitude), _att_sub, &_att);
257 
258  vehicle_attitude_setpoint_poll();
259  vehicle_control_mode_poll();
260  manual_control_setpoint_poll();
261  battery_status_poll();
262 
263  /* decide if in stabilized or full manual control */
264  if (_vcontrol_mode.flag_control_rates_enabled) {
265  /* Run attitude controllers */
266  if (_vcontrol_mode.flag_control_attitude_enabled) {
267 
268  Eulerf euler_angles(matrix::Quatf(_att.q));
269 
270  /* Calculate the control output for the steering as yaw */
271  float yaw_u = pid_calculate(&_steering_ctrl, _att_sp.yaw_body, euler_angles.psi(), _att.yawspeed, deltaT);
272 
273  float angle_diff = 0.0f;
274 
275  if (_att_sp.yaw_body * euler_angles.psi() < 0.0f) {
276  if (_att_sp.yaw_body < 0.0f) {
277  angle_diff = euler_angles.psi() - _att_sp.yaw_body ;
278 
279  } else {
280  angle_diff = _att_sp.yaw_body - euler_angles.psi();
281  }
282 
283  // a switch might have happened
284  if ((double)angle_diff > M_PI) {
285  yaw_u = -yaw_u;
286  }
287 
288  }
289 
290  math::constrain(yaw_u, -1.0f, 1.0f);
291 
292  if (PX4_ISFINITE(yaw_u)) {
293  _actuators.control[actuator_controls_s::INDEX_YAW] = yaw_u + _parameters.trim_yaw;
294 
295  } else {
296  _actuators.control[actuator_controls_s::INDEX_YAW] = _parameters.trim_yaw;
297 
298  perf_count(_nonfinite_output_perf);
299 
300  if (_debug && loop_counter % 10 == 0) {
301  warnx("yaw_u %.4f", (double)yaw_u);
302  }
303  }
304 
305  /* throttle passed through if it is finite and if no engine failure was detected */
306  _actuators.control[actuator_controls_s::INDEX_THROTTLE] = _att_sp.thrust;
307 
308  /* scale effort by battery status */
309  if (_parameters.bat_scale_en && _battery_status.scale > 0.0f &&
310  _actuators.control[actuator_controls_s::INDEX_THROTTLE] > 0.1f) {
311 
312  _actuators.control[actuator_controls_s::INDEX_THROTTLE] *= _battery_status.scale;
313  }
314  }
315 
316  } else {
317  /* manual/direct control */
318  _actuators.control[actuator_controls_s::INDEX_ROLL] = _manual.y;
319  _actuators.control[actuator_controls_s::INDEX_PITCH] = -_manual.x;
320  _actuators.control[actuator_controls_s::INDEX_YAW] = _manual.r * _parameters.man_yaw_scale + _parameters.trim_yaw;
321  _actuators.control[actuator_controls_s::INDEX_THROTTLE] = _manual.z;
322  }
323 
324  /* lazily publish the setpoint only once available */
325  _actuators.timestamp = hrt_absolute_time();
326  _actuators.timestamp_sample = _att.timestamp;
327 
328  /* Only publish if any of the proper modes are enabled */
329  if (_vcontrol_mode.flag_control_attitude_enabled ||
330  _vcontrol_mode.flag_control_manual_enabled) {
331 
332  /* publish the actuator controls */
333  if (_actuators_0_pub != nullptr) {
334  orb_publish(ORB_ID_VEHICLE_ATTITUDE_CONTROLS, _actuators_0_pub, &_actuators);
335 
336  } else {
337  _actuators_0_pub = orb_advertise(ORB_ID_VEHICLE_ATTITUDE_CONTROLS, &_actuators);
338  }
339  }
340  }
341 
342  loop_counter++;
343  perf_end(_loop_perf);
344  }
345 
346  warnx("exiting.\n");
347 
348  _control_task = -1;
349  _task_running = false;
350 }

373 int gnd_att_control_main(int argc, char *argv[])
374 {
375  if (argc < 2) {
376  warnx("usage: gnd_att_control {start|stop|status}");
377  return 1;
378  }
379 
380  if (!strcmp(argv[1], "start")) {
381 
382  if (att_gnd_control::g_control != nullptr) {
383  warnx("already running");
384  return 1;
385  }
386 
387  att_gnd_control::g_control = new GroundRoverAttitudeControl;
388 
389  if (att_gnd_control::g_control == nullptr) {
390  warnx("alloc failed");
391  return 1;
392  }
393 
394  if (PX4_OK != att_gnd_control::g_control->start()) {
395  delete att_gnd_control::g_control;
396  att_gnd_control::g_control = nullptr;
397  warn("start failed");
398  return 1;
399  }
400 
401  /* check if the waiting is necessary at all */
402  if (att_gnd_control::g_control == nullptr || !att_gnd_control::g_control->task_running()) {
403 
404  /* avoid memory fragmentation by not exiting start handler until the task has fully started */
405  while (att_gnd_control::g_control == nullptr || !att_gnd_control::g_control->task_running()) {
406  usleep(50000);
407  printf(".");
408  fflush(stdout);
409  }
410 
411  printf("\n");
412  }
413 
414  return 0;
415  }
416 
417  if (!strcmp(argv[1], "stop")) {
418  if (att_gnd_control::g_control == nullptr) {
419  warnx("not running");
420  return 1;
421  }
422 
423  delete att_gnd_control::g_control;
424  att_gnd_control::g_control = nullptr;
425  return 0;
426  }
427 
428  if (!strcmp(argv[1], "status")) {
429  if (att_gnd_control::g_control) {
430  warnx("running");
431  return 0;
432 
433  } else {
434  warnx("not running");
435  return 1;
436  }
437  }
438 
439  warnx("unrecognized command");
440  return 1;
441 }
