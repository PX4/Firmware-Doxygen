301 void
302 OREOLED::cycle()
303 {
304  /* check time since startup */
305  uint64_t now = hrt_absolute_time();
306  bool startup_timeout = (now - _start_time > OREOLED_TIMEOUT_USEC);
307 
308  /* prepare the response buffer */
309  uint8_t reply[OREOLED_CMD_READ_LENGTH_MAX];
310 
311  /* during startup period keep searching for unhealthy LEDs */
312  if (!startup_timeout && _num_healthy < OREOLED_NUM_LEDS) {
313  /* prepare command to turn off LED */
314  /* add two bytes of pre-amble to for higher signal to noise ratio */
315  uint8_t msg[] = {0xAA, 0x55, OREOLED_PATTERN_OFF, 0x00};
316 
317  /* attempt to contact each unhealthy LED */
318  for (uint8_t i = 0; i < OREOLED_NUM_LEDS; i++) {
319  if (!_healthy[i]) {
320  perf_begin(_probe_perf);
321 
322  /* set I2C address */
323  set_device_address(OREOLED_BASE_I2C_ADDR + i);
324 
325  /* Calculate XOR CRC and append to the i2c write data */
326  msg[sizeof(msg) - 1] = OREOLED_BASE_I2C_ADDR + i;
327 
328  for (uint8_t j = 0; j < sizeof(msg) - 1; j++) {
329  msg[sizeof(msg) - 1] ^= msg[j];
330  }
331 
332  /* send I2C command */
333  if (transfer(msg, sizeof(msg), reply, 3) == OK) {
334  if (reply[1] == OREOLED_BASE_I2C_ADDR + i &&
335  reply[2] == msg[sizeof(msg) - 1]) {
336  DEVICE_LOG("oreoled %u ok - in bootloader", (unsigned)i);
337  _healthy[i] = true;
338  _num_healthy++;
339 
340  /* If slaves are in application record that so we can reset if we need to bootload */
341  /* This additional check is required for LED firmwares below v1.3 and can be
342  deprecated once all LEDs in the wild have firmware >= v1.3 */
343  if (bootloader_ping(i) == OK) {
344  _in_boot[i] = true;
345  _num_inboot++;
346  }
347 
348  /* Check for a reply with a checksum offset of 1,
349  which indicates a response from firmwares >= 1.3 */
350 
351  } else if (reply[1] == OREOLED_BASE_I2C_ADDR + i &&
352  reply[2] == msg[sizeof(msg) - 1] + 1) {
353  DEVICE_LOG("oreoled %u ok - in application", (unsigned)i);
354  _healthy[i] = true;
355  _num_healthy++;
356 
357  } else {
358  DEVICE_LOG("oreo reply errors: %u", (unsigned)_reply_errors);
359  perf_count(_reply_errors);
360  }
361 
362  } else {
363  perf_count(_comms_errors);
364  }
365 
366  perf_end(_probe_perf);
367  }
368  }
369 
370  /* schedule another attempt in 0.1 sec */
371  work_queue(HPWORK, &_work, (worker_t)&OREOLED::cycle_trampoline, this,
372  USEC2TICK(OREOLED_STARTUP_INTERVAL_US));
373  return;
374 
375  } else if (_alwaysupdate) {
376  /* reset each healthy LED */
377  for (uint8_t i = 0; i < OREOLED_NUM_LEDS; i++) {
378  if (_healthy[i] && !_in_boot[i]) {
379  /* reset the LED if it's not in the bootloader */
380  /* (this happens during a pixhawk OTA update, since the LEDs stay powered) */
381  bootloader_app_reset(i);
382  }
383  }
384 
385  /* attempt to update each healthy LED */
386  for (uint8_t i = 0; i < OREOLED_NUM_LEDS; i++) {
387  if (_healthy[i] && _in_boot[i]) {
388  /* flash the new firmware */
389  bootloader_flash(i);
390  }
391  }
392 
393  /* boot each healthy LED */
394  for (uint8_t i = 0; i < OREOLED_NUM_LEDS; i++) {
395  if (_healthy[i] && _in_boot[i]) {
396  /* boot the application */
397  bootloader_boot(i);
398  }
399  }
400 
401  /* coerce LEDs with startup issues to be healthy again */
402  bootloader_coerce_healthy();
403 
404  /* mandatory updating has finished */
405  _alwaysupdate = false;
406 
407  /* schedule a fresh cycle call when the measurement is done */
408  work_queue(HPWORK, &_work, (worker_t)&OREOLED::cycle_trampoline, this,
409  USEC2TICK(OREOLED_UPDATE_INTERVAL_US));
410  return;
411 
412  } else if (_autoupdate) {
413  /* check booted oreoleds to see if the app can report it's checksum (release versions >= v1.2) */
414  for (uint8_t i = 0; i < OREOLED_NUM_LEDS; i++) {
415  if (_healthy[i] && !_in_boot[i]) {
416  /* put any out of date oreoleds into bootloader mode */
417  /* being in bootloader mode signals to be code below that the will likey need updating */
418  if (bootloader_inapp_checksum(i) != bootloader_fw_checksum()) {
419  bootloader_app_reset(i);
420  }
421  }
422  }
423 
424  /* reset all healthy oreoleds if the number of outdated oreoled's is > 0 */
425  /* this is done for consistency, so if only one oreoled is updating, all LEDs show the same behaviour */
426  /* otherwise a single oreoled could appear broken or failed. */
427  if (_num_inboot > 0) {
428  for (uint8_t i = 0; i < OREOLED_NUM_LEDS; i++) {
429  if (_healthy[i] && !_in_boot[i]) {
430  /* reset the LED if it's not in the bootloader */
431  /* (this happens during a pixhawk OTA update, since the LEDs stay powered) */
432  bootloader_app_reset(i);
433  }
434  }
435 
436  /* update each outdated and healthy LED in bootloader mode */
437  for (uint8_t i = 0; i < OREOLED_NUM_LEDS; i++) {
438  if (_healthy[i] && _in_boot[i]) {
439  /* only flash LEDs with an old version of the applictioon */
440  if (bootloader_app_checksum(i) != bootloader_fw_checksum()) {
441  /* flash the new firmware */
442  bootloader_flash(i);
443  }
444  }
445  }
446 
447  /* boot each healthy LED */
448  for (uint8_t i = 0; i < OREOLED_NUM_LEDS; i++) {
449  if (_healthy[i] && _in_boot[i]) {
450  /* boot the application */
451  bootloader_boot(i);
452  }
453  }
454 
455  /* coerce LEDs with startup issues to be healthy again */
456  bootloader_coerce_healthy();
457  }
458 
459  /* auto updating has finished */
460  _autoupdate = false;
461 
462  /* schedule a fresh cycle call when the measurement is done */
463  work_queue(HPWORK, &_work, (worker_t)&OREOLED::cycle_trampoline, this,
464  USEC2TICK(OREOLED_UPDATE_INTERVAL_US));
465  return;
466 
467  } else if (_num_inboot > 0) {
468  /* boot any LEDs which are in still in bootloader mode */
469  for (uint8_t i = 0; i < OREOLED_NUM_LEDS; i++) {
470  if (_in_boot[i]) {
471  bootloader_boot(i);
472  }
473  }
474 
475  /* coerce LEDs with startup issues to be healthy again */
476  bootloader_coerce_healthy();
477 
478  /* ensure we don't get stuck in a loop */
479  _num_inboot = 0;
480 
481  /* schedule a fresh cycle call when the measurement is done */
482  work_queue(HPWORK, &_work, (worker_t)&OREOLED::cycle_trampoline, this,
483  USEC2TICK(OREOLED_UPDATE_INTERVAL_US));
484  return;
485 
486  } else if (!_is_ready) {
487  /* indicate a ready state since startup has finished */
488  _is_ready = true;
489  }
490 
491  /* get next command from queue */
492  oreoled_cmd_t next_cmd;
493 
494  while (_cmd_queue->get(&next_cmd, sizeof(oreoled_cmd_t))) {
495  /* send valid messages to healthy LEDs */
496  if ((next_cmd.led_num < OREOLED_NUM_LEDS) && _healthy[next_cmd.led_num]
497  && (next_cmd.num_bytes <= OREOLED_CMD_LENGTH_MAX)) {
498  /* start performance timer */
499  perf_begin(_call_perf);
500 
501  /* set I2C address */
502  set_device_address(OREOLED_BASE_I2C_ADDR + next_cmd.led_num);
503 
504  /* Calculate XOR CRC and append to the i2c write data */
505  uint8_t next_cmd_xor = OREOLED_BASE_I2C_ADDR + next_cmd.led_num;
506 
507  for (uint8_t i = 0; i < next_cmd.num_bytes; i++) {
508  next_cmd_xor ^= next_cmd.buff[i];
509  }
510 
511  next_cmd.buff[next_cmd.num_bytes++] = next_cmd_xor;
512 
513  /* send I2C command with a retry limit */
514  for (uint8_t retry = OEROLED_COMMAND_RETRIES; retry > 0; retry--) {
515  if (transfer(next_cmd.buff, next_cmd.num_bytes, reply, 3) == OK) {
516  if (reply[1] == (OREOLED_BASE_I2C_ADDR + next_cmd.led_num) && reply[2] == next_cmd_xor) {
517  /* slave returned a valid response */
518  break;
519 
520  } else {
521  perf_count(_reply_errors);
522  }
523 
524  } else {
525  perf_count(_comms_errors);
526  }
527  }
528 
529  perf_end(_call_perf);
530  }
531  }
532 
533  /* send general call every 4 seconds, if we aren't bootloading*/
534  if (!_is_bootloading && ((now - _last_gencall) > OREOLED_GENERALCALL_US)) {
535  perf_begin(_gcall_perf);
536  send_general_call();
537  perf_end(_gcall_perf);
538  }
539 
540  /* schedule a fresh cycle call when the command is sent */
541  work_queue(HPWORK, &_work, (worker_t)&OREOLED::cycle_trampoline, this,
542  USEC2TICK(OREOLED_UPDATE_INTERVAL_US));
543 }

546 OREOLED::bootloader_app_reset(int led_num)
547 {
548  _is_bootloading = true;
549  oreoled_cmd_t boot_cmd;
550  boot_cmd.led_num = led_num;
551 
552  int ret = -1;
553 
554  /* Set the current address */
555  set_device_address(OREOLED_BASE_I2C_ADDR + boot_cmd.led_num);
556 
557  /* send a reset */
558  boot_cmd.buff[0] = OREOLED_PATTERN_PARAMUPDATE;
559  boot_cmd.buff[1] = OREOLED_PARAM_RESET;
560  boot_cmd.buff[2] = OEROLED_RESET_NONCE;
561  boot_cmd.buff[3] = OREOLED_BASE_I2C_ADDR + boot_cmd.led_num;
562  boot_cmd.num_bytes = 4;
563 
564  for (uint8_t j = 0; j < boot_cmd.num_bytes - 1; j++) {
565  boot_cmd.buff[boot_cmd.num_bytes - 1] ^= boot_cmd.buff[j];
566  }
567 
568  uint8_t reply[OREOLED_CMD_READ_LENGTH_MAX];
569 
570  /* send I2C command with a retry limit */
571  for (uint8_t retry = OEROLED_COMMAND_RETRIES; retry > 0; retry--) {
572  if (transfer(boot_cmd.buff, boot_cmd.num_bytes, reply, 3) == OK) {
573  if (reply[1] == (OREOLED_BASE_I2C_ADDR + boot_cmd.led_num) &&
574  reply[2] == boot_cmd.buff[boot_cmd.num_bytes - 1]) {
575  /* slave returned a valid response */
576  ret = OK;
577  /* set this LED as being in boot mode now */
578  _in_boot[led_num] = true;
579  _num_inboot++;
580  break;
581  }
582  }
583  }
584 
585  /* Allow time for the LED to reboot */
586  usleep(OREOLED_BOOT_FLASH_WAITMS * 1000 * 10);
587  usleep(OREOLED_BOOT_FLASH_WAITMS * 1000 * 10);
588  usleep(OREOLED_BOOT_FLASH_WAITMS * 1000 * 10);
589  usleep(OREOLED_BOOT_FLASH_WAITMS * 1000 * 10);
590 
591  _is_bootloading = false;
592  return ret;
593 }

983 OREOLED::bootloader_flash(int led_num)
984 {
985  _is_bootloading = true;
986  oreoled_cmd_t boot_cmd;
987  boot_cmd.led_num = led_num;
988 
989  /* Open the bootloader file */
990  int fd = ::open(OREOLED_FW_FILE, O_RDONLY);
991 
992  /* check for error opening the file */
993  if (fd < 0) {
994  return -1;
995  }
996 
997  struct stat s;
998 
999  /* attempt to stat the file */
1000  if (stat(OREOLED_FW_FILE, &s) != 0) {
1001  ::close(fd);
1002  return -1;
1003  }
1004 
1005  uint16_t fw_length = s.st_size - OREOLED_FW_FILE_HEADER_LENGTH;
1006 
1007  /* sanity-check file size */
1008  if (fw_length > OREOLED_FW_FILE_SIZE_LIMIT) {
1009  ::close(fd);
1010  return -1;
1011  }
1012 
1013  uint8_t *buf = new uint8_t[s.st_size];
1014 
1015  /* check that the buffer has been allocated */
1016  if (buf == NULL) {
1017  ::close(fd);
1018  return -1;
1019  }
1020 
1021  /* check that the firmware can be read into the buffer */
1022  if (::read(fd, buf, s.st_size) != s.st_size) {
1023  ::close(fd);
1024  delete[] buf;
1025  return -1;
1026  }
1027 
1028  ::close(fd);
1029 
1030  /* Grab the version bytes from the binary */
1031  uint8_t version_major = buf[0];
1032  uint8_t version_minor = buf[1];
1033 
1034  /* calculate flash pages (rounded up to nearest integer) */
1035  uint8_t flash_pages = ((fw_length + 64 - 1) / 64);
1036 
1037  /* Set the current address */
1038  set_device_address(OREOLED_BASE_I2C_ADDR + boot_cmd.led_num);
1039 
1040  uint8_t reply[OREOLED_CMD_READ_LENGTH_MAX];
1041 
1042  /* Loop through flash pages */
1043  for (uint8_t page_idx = 0; page_idx < flash_pages; page_idx++) {
1044 
1045  /* Send the first half of the 64 byte flash page */
1046  memset(boot_cmd.buff, 0, sizeof(boot_cmd.buff));
1047  boot_cmd.buff[0] = OREOLED_BOOT_CMD_WRITE_FLASH_A;
1048  boot_cmd.buff[1] = page_idx;
1049  memcpy(boot_cmd.buff + 2, buf + (page_idx * 64) + OREOLED_FW_FILE_HEADER_LENGTH, 32);
1050  boot_cmd.buff[32 + 2] = OREOLED_BASE_I2C_ADDR + boot_cmd.led_num;
1051  boot_cmd.num_bytes = 32 + 3;
1052 
1053  for (uint8_t k = 0; k < boot_cmd.num_bytes - 1; k++) {
1054  boot_cmd.buff[boot_cmd.num_bytes - 1] ^= boot_cmd.buff[k];
1055  }
1056 
1057  for (uint8_t retry = OEROLED_BOOT_COMMAND_RETRIES; retry > 0; retry--) {
1058  /* Send the I2C Write+Read */
1059  memset(reply, 0, sizeof(reply));
1060  transfer(boot_cmd.buff, boot_cmd.num_bytes, reply, 4);
1061 
1062  /* Check the response */
1063  if (reply[1] == OREOLED_BASE_I2C_ADDR + boot_cmd.led_num &&
1064  reply[2] == OREOLED_BOOT_CMD_WRITE_FLASH_A &&
1065  reply[3] == boot_cmd.buff[boot_cmd.num_bytes - 1]) {
1066  warnx("bl flash %ia OK for LED %i", page_idx, boot_cmd.led_num);
1067  break;
1068 
1069  } else {
1070  warnx("bl flash %ia FAIL for LED %i", page_idx, boot_cmd.led_num);
1071  warnx("bl flash %ia response ADDR: 0x%x", page_idx, reply[1]);
1072  warnx("bl flash %ia response CMD: 0x%x", page_idx, reply[2]);
1073  warnx("bl flash %ia response XOR: 0x%x", page_idx, reply[3]);
1074 
1075  if (retry > 1) {
1076  warnx("bl flash %ia retrying LED %i", page_idx, boot_cmd.led_num);
1077 
1078  } else {
1079  warnx("bl flash %ia failed on LED %i", page_idx, boot_cmd.led_num);
1080  delete[] buf;
1081  return -1;
1082  }
1083  }
1084  }
1085 
1086  /* Send the second half of the 64 byte flash page */
1087  memset(boot_cmd.buff, 0, sizeof(boot_cmd.buff));
1088  boot_cmd.buff[0] = OREOLED_BOOT_CMD_WRITE_FLASH_B;
1089  memcpy(boot_cmd.buff + 1, buf + (page_idx * 64) + 32 + OREOLED_FW_FILE_HEADER_LENGTH, 32);
1090  boot_cmd.buff[32 + 1] = OREOLED_BASE_I2C_ADDR + boot_cmd.led_num;
1091  boot_cmd.num_bytes = 32 + 2;
1092 
1093  for (uint8_t k = 0; k < boot_cmd.num_bytes - 1; k++) {
1094  boot_cmd.buff[boot_cmd.num_bytes - 1] ^= boot_cmd.buff[k];
1095  }
1096 
1097  for (uint8_t retry = OEROLED_BOOT_COMMAND_RETRIES; retry > 0; retry--) {
1098  /* Send the I2C Write+Read */
1099  memset(reply, 0, sizeof(reply));
1100  transfer(boot_cmd.buff, boot_cmd.num_bytes, reply, 4);
1101 
1102  /* Check the response */
1103  if (reply[1] == OREOLED_BASE_I2C_ADDR + boot_cmd.led_num &&
1104  reply[2] == OREOLED_BOOT_CMD_WRITE_FLASH_B &&
1105  reply[3] == boot_cmd.buff[boot_cmd.num_bytes - 1]) {
1106  warnx("bl flash %ib OK for LED %i", page_idx, boot_cmd.led_num);
1107  break;
1108 
1109  } else {
1110  warnx("bl flash %ib FAIL for LED %i", page_idx, boot_cmd.led_num);
1111  warnx("bl flash %ib response ADDR: 0x%x", page_idx, reply[1]);
1112  warnx("bl flash %ib response CMD: 0x%x", page_idx, reply[2]);
1113  warnx("bl flash %ib response XOR: 0x%x", page_idx, reply[3]);
1114 
1115  if (retry > 1) {
1116  warnx("bl flash %ib retrying LED %i", page_idx, boot_cmd.led_num);
1117 
1118  } else {
1119  errx(1, "bl flash %ib failed on LED %i", page_idx, boot_cmd.led_num);
1120  delete[] buf;
1121  return -1;
1122  }
1123  }
1124  }
1125 
1126  /* Sleep to allow flash to write */
1127  /* Wait extra long on the first write, to allow time for EEPROM updates */
1128  if (page_idx == 0) {
1129  usleep(OREOLED_BOOT_FLASH_WAITMS * 1000 * 10);
1130 
1131  } else {
1132  usleep(OREOLED_BOOT_FLASH_WAITMS * 1000);
1133  }
1134  }
1135 
1136  uint16_t app_checksum = bootloader_fw_checksum();
1137 
1138  /* Flash writes must have succeeded so finalise the flash */
1139  boot_cmd.buff[0] = OREOLED_BOOT_CMD_FINALISE_FLASH;
1140  boot_cmd.buff[1] = version_major;
1141  boot_cmd.buff[2] = version_minor;
1142  boot_cmd.buff[3] = (uint8_t)(fw_length >> 8);
1143  boot_cmd.buff[4] = (uint8_t)(fw_length & 0xFF);
1144  boot_cmd.buff[5] = (uint8_t)(app_checksum >> 8);
1145  boot_cmd.buff[6] = (uint8_t)(app_checksum & 0xFF);
1146  boot_cmd.buff[7] = OREOLED_BASE_I2C_ADDR + boot_cmd.led_num;
1147  boot_cmd.num_bytes = 8;
1148 
1149  for (uint8_t k = 0; k < boot_cmd.num_bytes - 1; k++) {
1150  boot_cmd.buff[boot_cmd.num_bytes - 1] ^= boot_cmd.buff[k];
1151  }
1152 
1153  /* Try to finalise for twice the amount of normal retries */
1154  for (uint8_t retry = OEROLED_BOOT_COMMAND_RETRIES * 2; retry > 0; retry--) {
1155  /* Send the I2C Write */
1156  memset(reply, 0, sizeof(reply));
1157  transfer(boot_cmd.buff, boot_cmd.num_bytes, reply, 4);
1158 
1159  /* Check the response */
1160  if (reply[1] == OREOLED_BASE_I2C_ADDR + boot_cmd.led_num &&
1161  reply[2] == OREOLED_BOOT_CMD_FINALISE_FLASH &&
1162  reply[3] == boot_cmd.buff[boot_cmd.num_bytes - 1]) {
1163  warnx("bl finalise OK from LED %i", boot_cmd.led_num);
1164  break;
1165 
1166  } else {
1167  warnx("bl finalise response ADDR: 0x%x", reply[1]);
1168  warnx("bl finalise response CMD: 0x%x", reply[2]);
1169  warnx("bl finalise response XOR: 0x%x", reply[3]);
1170 
1171  if (retry > 1) {
1172  warnx("bl finalise retrying LED %i", boot_cmd.led_num);
1173 
1174  } else {
1175  warnx("bl finalise failed on LED %i", boot_cmd.led_num);
1176  delete[] buf;
1177  return -1;
1178  }
1179  }
1180  }
1181 
1182  /* allow time for flash to finalise */
1183  usleep(OREOLED_BOOT_FLASH_WAITMS * 1000 * 10);
1184  usleep(OREOLED_BOOT_FLASH_WAITMS * 1000 * 10);
1185 
1186  /* clean up file buffer */
1187  delete[] buf;
1188 
1189  _is_bootloading = false;
1190  return 1;

1319  warnx("fw checksum = %i", app_checksum);
