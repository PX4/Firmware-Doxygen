49 void Ekf::initialiseCovariance()
50 {
51  for (unsigned i = 0; i < _k_num_states; i++) {
52  for (unsigned j = 0; j < _k_num_states; j++) {
53  P[i][j] = 0.0f;
54  }
55  }
56 
57  // calculate average prediction time step in sec
58  float dt = FILTER_UPDATE_PERIOD_S;
59 
60  // define the initial angle uncertainty as variances for a rotation vector
61  Vector3f rot_vec_var;
62  rot_vec_var(2) = rot_vec_var(1) = rot_vec_var(0) = sq(_params.initial_tilt_err);
63 
64  // update the quaternion state covariances
65  initialiseQuatCovariances(rot_vec_var);
66 
67  // velocity
68  P[4][4] = sq(fmaxf(_params.gps_vel_noise, 0.01f));
69  P[5][5] = P[4][4];
70  P[6][6] = sq(1.5f) * P[4][4];
71 
72  // position
73  P[7][7] = sq(fmaxf(_params.gps_pos_noise, 0.01f));
74  P[8][8] = P[7][7];
75 
76  if (_control_status.flags.rng_hgt) {
77  P[9][9] = sq(fmaxf(_params.range_noise, 0.01f));
78 
79  } else if (_control_status.flags.gps_hgt) {
80  float lower_limit = fmaxf(_params.gps_pos_noise, 0.01f);
81  float upper_limit = fmaxf(_params.pos_noaid_noise, lower_limit);
82  P[9][9] = sq(1.5f * math::constrain(_gps_sample_delayed.vacc, lower_limit, upper_limit));
83 
84  } else {
85  P[9][9] = sq(fmaxf(_params.baro_noise, 0.01f));
86  }
87 
88  // gyro bias
89  P[10][10] = sq(_params.switch_on_gyro_bias * dt);
90  P[11][11] = P[10][10];
91  P[12][12] = P[10][10];
92 
93  // accel bias
94  _prev_dvel_bias_var(0) = P[13][13] = sq(_params.switch_on_accel_bias * dt);
95  _prev_dvel_bias_var(1) = P[14][14] = P[13][13];
96  _prev_dvel_bias_var(2) = P[15][15] = P[13][13];
97 
98  // record IMU bias state covariance reset time - used to prevent resets being performed too often
99  _last_imu_bias_cov_reset_us = _imu_sample_delayed.time_us;
100 
101  // variances for optional states
102 
103  // earth frame and body frame magnetic field
104  // set to observation variance
105  for (uint8_t index = 16; index <= 21; index ++) {
106  P[index][index] = sq(_params.mag_noise);
107  }
108 
109  // wind
110  P[22][22] = sq(_params.initial_wind_uncertainty);
111  P[23][23] = sq(_params.initial_wind_uncertainty);
112 
113 }

133  float q1 = _state.quat_nominal(1);
134  float q2 = _state.quat_nominal(2);
135  float q3 = _state.quat_nominal(3);
136 
137  float dax = _imu_sample_delayed.delta_ang(0);
138  float day = _imu_sample_delayed.delta_ang(1);
139  float daz = _imu_sample_delayed.delta_ang(2);
140 
141  float dvx = _imu_sample_delayed.delta_vel(0);
142  float dvy = _imu_sample_delayed.delta_vel(1);
143  float dvz = _imu_sample_delayed.delta_vel(2);
144 
145  float dax_b = _state.gyro_bias(0);
146  float day_b = _state.gyro_bias(1);
147  float daz_b = _state.gyro_bias(2);
148 
149  float dvx_b = _state.accel_bias(0);
150  float dvy_b = _state.accel_bias(1);
151  float dvz_b = _state.accel_bias(2);
152 
153  float dt = math::constrain(_imu_sample_delayed.delta_ang_dt, 0.5f * FILTER_UPDATE_PERIOD_S, 2.0f * FILTER_UPDATE_PERIOD_S);
154  float dt_inv = 1.0f / dt;
155 
156  // compute noise variance for stationary processes
157  float process_noise[_k_num_states] = {};
158 
159  // convert rate of change of rate gyro bias (rad/s**2) as specified by the parameter to an expected change in delta angle (rad) since the last update
160  float d_ang_bias_sig = dt * dt * math::constrain(_params.gyro_bias_p_noise, 0.0f, 1.0f);
161 
162  // convert rate of change of accelerometer bias (m/s**3) as specified by the parameter to an expected change in delta velocity (m/s) since the last update
163  float d_vel_bias_sig = dt * dt * math::constrain(_params.accel_bias_p_noise, 0.0f, 1.0f);
164 
165  // inhibit learning of imu acccel bias if the manoeuvre levels are too high to protect against the effect of sensor nonlinearities or bad accel data is detected
166  float alpha = math::constrain((dt / _params.acc_bias_learn_tc), 0.0f, 1.0f);
167  float beta = 1.0f - alpha;
168  _ang_rate_mag_filt = fmaxf(dt_inv * _imu_sample_delayed.delta_ang.norm(), beta * _ang_rate_mag_filt);
169  _accel_mag_filt = fmaxf(dt_inv * _imu_sample_delayed.delta_vel.norm(), beta * _accel_mag_filt);
170  _accel_vec_filt(0) = alpha * dt_inv * _imu_sample_delayed.delta_vel(0) + beta * _accel_vec_filt(0);
171  _accel_vec_filt(1) = alpha * dt_inv * _imu_sample_delayed.delta_vel(1) + beta * _accel_vec_filt(1);
172  _accel_vec_filt(2) = alpha * dt_inv * _imu_sample_delayed.delta_vel(2) + beta * _accel_vec_filt(2);
173 
174  if (_ang_rate_mag_filt > _params.acc_bias_learn_gyr_lim
175  || _accel_mag_filt > _params.acc_bias_learn_acc_lim
176  || _bad_vert_accel_detected) {
177 
178  // store the bias state variances to be reinstated later
179  if (!_accel_bias_inhibit) {
180  _prev_dvel_bias_var(0) = P[13][13];
181  _prev_dvel_bias_var(1) = P[14][14];
182  _prev_dvel_bias_var(2) = P[15][15];
183  }
184 
185  _accel_bias_inhibit = true;
186 
187  } else {
188  if (_accel_bias_inhibit) {
189  // reinstate the bias state variances
190  P[13][13] = _prev_dvel_bias_var(0);
191  P[14][14] = _prev_dvel_bias_var(1);
192  P[15][15] = _prev_dvel_bias_var(2);
193 
194  } else {
195  // store the bias state variances to be reinstated later
196  _prev_dvel_bias_var(0) = P[13][13];
197  _prev_dvel_bias_var(1) = P[14][14];
198  _prev_dvel_bias_var(2) = P[15][15];
199  }
200 
201  _accel_bias_inhibit = false;
202  }
203 
204  // Don't continue to grow the earth field variances if they are becoming too large or we are not doing 3-axis fusion as this can make the covariance matrix badly conditioned
205  float mag_I_sig;
206 
207  if (_control_status.flags.mag_3D && (P[16][16] + P[17][17] + P[18][18]) < 0.1f) {
208  mag_I_sig = dt * math::constrain(_params.mage_p_noise, 0.0f, 1.0f);
209 
210  } else {
211  mag_I_sig = 0.0f;
212  }
213 
214  // Don't continue to grow the body field variances if they is becoming too large or we are not doing 3-axis fusion as this can make the covariance matrix badly conditioned
215  float mag_B_sig;
216 
217  if (_control_status.flags.mag_3D && (P[19][19] + P[20][20] + P[21][21]) < 0.1f) {
218  mag_B_sig = dt * math::constrain(_params.magb_p_noise, 0.0f, 1.0f);
219 
220  } else {
221  mag_B_sig = 0.0f;
222  }
223 
224  float wind_vel_sig;
225 
226  // Don't continue to grow wind velocity state variances if they are becoming too large or we are not using wind velocity states as this can make the covariance matrix badly conditioned
227  if (_control_status.flags.wind && (P[22][22] + P[23][23]) < sq(_params.initial_wind_uncertainty)) {
228  wind_vel_sig = dt * math::constrain(_params.wind_vel_p_noise, 0.0f, 1.0f);
229 
230  } else {
231  wind_vel_sig = 0.0f;
232  }
233 
234  // Construct the process noise variance diagonal for those states with a stationary process model
235  // These are kinematic states and their error growth is controlled separately by the IMU noise variances
236  for (unsigned i = 0; i <= 9; i++) {
237  process_noise[i] = 0.0f;
238  }
239 
240  // delta angle bias states
241  process_noise[12] = process_noise[11] = process_noise[10] = sq(d_ang_bias_sig);
242  // delta_velocity bias states
243  process_noise[15] = process_noise[14] = process_noise[13] = sq(d_vel_bias_sig);
244  // earth frame magnetic field states
245  process_noise[18] = process_noise[17] = process_noise[16] = sq(mag_I_sig);
246  // body frame magnetic field states
247  process_noise[21] = process_noise[20] = process_noise[19] = sq(mag_B_sig);
248  // wind velocity states
249  process_noise[23] = process_noise[22] = sq(wind_vel_sig);
250 
251  // assign IMU noise variances
252  // inputs to the system are 3 delta angles and 3 delta velocities
253  float daxVar, dayVar, dazVar;
254  float dvxVar, dvyVar, dvzVar;
255  float gyro_noise = math::constrain(_params.gyro_noise, 0.0f, 1.0f);
256  daxVar = dayVar = dazVar = sq(dt * gyro_noise);
257  float accel_noise = math::constrain(_params.accel_noise, 0.0f, 1.0f);
258 
259  if (_bad_vert_accel_detected) {
260  // Increase accelerometer process noise if bad accel data is detected. Measurement errors due to
261  // vibration induced clipping commonly reach an equivalent 0.5g offset.
262  accel_noise = BADACC_BIAS_PNOISE;
263  }
264 
265  dvxVar = dvyVar = dvzVar = sq(dt * accel_noise);
266 
267  // predict the covariance
268 
269  // intermediate calculations
270  float SF[21];
271  SF[0] = dvz - dvz_b;
272  SF[1] = dvy - dvy_b;
273  SF[2] = dvx - dvx_b;
274  SF[3] = 2*q1*SF[2] + 2*q2*SF[1] + 2*q3*SF[0];
275  SF[4] = 2*q0*SF[1] - 2*q1*SF[0] + 2*q3*SF[2];
276  SF[5] = 2*q0*SF[2] + 2*q2*SF[0] - 2*q3*SF[1];
277  SF[6] = day/2 - day_b/2;
278  SF[7] = daz/2 - daz_b/2;
279  SF[8] = dax/2 - dax_b/2;
280  SF[9] = dax_b/2 - dax/2;
281  SF[10] = daz_b/2 - daz/2;
282  SF[11] = day_b/2 - day/2;
283  SF[12] = 2*q1*SF[1];
284  SF[13] = 2*q0*SF[0];
285  SF[14] = q1/2;
286  SF[15] = q2/2;
287  SF[16] = q3/2;
288  SF[17] = sq(q3);
289  SF[18] = sq(q2);
290  SF[19] = sq(q1);
291  SF[20] = sq(q0);
292 
293  float SG[8];
294  SG[0] = q0/2;
295  SG[1] = sq(q3);
296  SG[2] = sq(q2);
297  SG[3] = sq(q1);
298  SG[4] = sq(q0);
299  SG[5] = 2*q2*q3;
300  SG[6] = 2*q1*q3;
301  SG[7] = 2*q1*q2;
302 
303  float SQ[11];
304  SQ[0] = dvzVar*(SG[5] - 2*q0*q1)*(SG[1] - SG[2] - SG[3] + SG[4]) - dvyVar*(SG[5] + 2*q0*q1)*(SG[1] - SG[2] + SG[3] - SG[4]) + dvxVar*(SG[6] - 2*q0*q2)*(SG[7] + 2*q0*q3);
305  SQ[1] = dvzVar*(SG[6] + 2*q0*q2)*(SG[1] - SG[2] - SG[3] + SG[4]) - dvxVar*(SG[6] - 2*q0*q2)*(SG[1] + SG[2] - SG[3] - SG[4]) + dvyVar*(SG[5] + 2*q0*q1)*(SG[7] - 2*q0*q3);
306  SQ[2] = dvzVar*(SG[5] - 2*q0*q1)*(SG[6] + 2*q0*q2) - dvyVar*(SG[7] - 2*q0*q3)*(SG[1] - SG[2] + SG[3] - SG[4]) - dvxVar*(SG[7] + 2*q0*q3)*(SG[1] + SG[2] - SG[3] - SG[4]);
307  SQ[3] = (dayVar*q1*SG[0])/2 - (dazVar*q1*SG[0])/2 - (daxVar*q2*q3)/4;
308  SQ[4] = (dazVar*q2*SG[0])/2 - (daxVar*q2*SG[0])/2 - (dayVar*q1*q3)/4;
309  SQ[5] = (daxVar*q3*SG[0])/2 - (dayVar*q3*SG[0])/2 - (dazVar*q1*q2)/4;
310  SQ[6] = (daxVar*q1*q2)/4 - (dazVar*q3*SG[0])/2 - (dayVar*q1*q2)/4;
311  SQ[7] = (dazVar*q1*q3)/4 - (daxVar*q1*q3)/4 - (dayVar*q2*SG[0])/2;
312  SQ[8] = (dayVar*q2*q3)/4 - (daxVar*q1*SG[0])/2 - (dazVar*q2*q3)/4;
313  SQ[9] = sq(SG[0]);
314  SQ[10] = sq(q1);
315 
316  float SPP[11] = {};
317  SPP[0] = SF[12] + SF[13] - 2*q2*SF[2];
318  SPP[1] = SF[17] - SF[18] - SF[19] + SF[20];
319  SPP[2] = SF[17] - SF[18] + SF[19] - SF[20];
320  SPP[3] = SF[17] + SF[18] - SF[19] - SF[20];
321  SPP[4] = 2*q0*q2 - 2*q1*q3;
322  SPP[5] = 2*q0*q1 - 2*q2*q3;
323  SPP[6] = 2*q0*q3 - 2*q1*q2;
324  SPP[7] = 2*q0*q1 + 2*q2*q3;
325  SPP[8] = 2*q0*q3 + 2*q1*q2;
326  SPP[9] = 2*q0*q2 + 2*q1*q3;
327  SPP[10] = SF[16];

725  // quaternion states

896  // set the wind covariance terms to zero
897  zeroRows(P, 22, 23);
898  zeroCols(P, 22, 23);

901  // Use airspeed and zer sideslip assumption to set initial covariance values for wind states
902 
903  // calculate the wind speed and bearing
904  float spd = sqrtf(sq(_state.wind_vel(0)) + sq(_state.wind_vel(1)));
905  float yaw = atan2f(_state.wind_vel(1), _state.wind_vel(0));
906 
907  // calculate the uncertainty in wind speed and direction using the uncertainty in airspeed and sideslip angle
908  // used to calculate the initial wind speed
909  float R_spd = sq(math::constrain(_params.eas_noise, 0.5f, 5.0f) * math::constrain(_airspeed_sample_delayed.eas2tas, 0.9f, 10.0f));
910  float R_yaw = sq(math::radians(10.0f));
911 
912  // calculate the variance and covariance terms for the wind states
913  float cos_yaw = cosf(yaw);
914  float sin_yaw = sinf(yaw);
915  float cos_yaw_2 = sq(cos_yaw);
916  float sin_yaw_2 = sq(sin_yaw);
917  float sin_cos_yaw = sin_yaw * cos_yaw;
918  float spd_2 = sq(spd);
919  P[22][22] = R_yaw * spd_2 * sin_yaw_2 + R_spd * cos_yaw_2;
920  P[22][23] = - R_yaw * sin_cos_yaw * spd_2 + R_spd * sin_cos_yaw;
921  P[23][22] = P[22][23];
922  P[23][23] = R_yaw * spd_2 * cos_yaw_2 + R_spd * sin_yaw_2;

925  P[22][22] += P[4][4];
926  P[23][23] += P[5][5];
927 
928  } else {
929  // without airspeed, start with a small initial uncertainty to improve the initial estimate
930  P[22][22] = sq(1.0f);
931  P[23][23] = sq(1.0f);
