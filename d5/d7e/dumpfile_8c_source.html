79 
80  /* get file size */
81  fseek(f, 0L, SEEK_END);

85  printf("File size: %d bytes\n", size);
86 
87  /* configure stdout */
88  int out = fileno(stdout);
89 
90  struct termios tc;
91  struct termios tc_old;
92  tcgetattr(out, &tc);
93 
94  /* save old terminal attributes to restore it later on exit */
95  memcpy(&tc_old, &tc, sizeof(tc));
96 
97  /* don't add CR on each LF*/
98  tc.c_oflag &= ~ONLCR;
99 
100  if (tcsetattr(out, TCSANOW, &tc) < 0) {
101  PX4_ERR("failed setting stdout attributes");
102  return 1;
103  }
104 
105  char buf[512];
106  int nread;
107 
108  /* dump file */
109  while ((nread = fread(buf, 1, sizeof(buf), f)) > 0) {
110  if (write(out, buf, nread) <= 0) {
111  PX4_ERR("write failed");
112  break;
113  }
114  }
115 
116  fsync(out);
117  fclose(f);
118 
119  /* restore old terminal attributes */
120  if (tcsetattr(out, TCSANOW, &tc_old) < 0) {
121  PX4_ERR("failed to restore stdout attributes");
122  return 1;
123  }
