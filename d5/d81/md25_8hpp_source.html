60  /**
61  * modes
62  *
63  * NOTE: this driver assumes we are always
64  * in mode 0!
65  *
66  * seprate speed mode:
67  * motor speed1 = speed1
68  * motor speed2 = speed2
69  * turn speed mode:
70  * motor speed1 = speed1 + speed2
71  * motor speed2 = speed2 - speed2
72  * unsigned:
73  * full rev (0), stop(128), full fwd (255)
74  * signed:
75  * full rev (-127), stop(0), full fwd (128)
76  *
77  * modes numbers:
78  * 0 : unsigned separate (default)
79  * 1 : signed separate
80  * 2 : unsigned turn
81  * 3 : signed turn
82  */
83  enum e_mode {
84  MODE_UNSIGNED_SPEED = 0,
85  MODE_SIGNED_SPEED,
86  MODE_UNSIGNED_SPEED_TURN,
87  MODE_SIGNED_SPEED_TURN,
88  };
