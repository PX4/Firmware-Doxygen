115 static unsigned
116 get_lookup_table_index(float *val, float min, float max)
117 {
118  /* for the rare case of hitting the bounds exactly
119  * the rounding logic wouldn't fit, so enforce it.
120  */
121 
122  /* limit to table bounds - required for maxima even when table spans full globe range */
123  /* limit to (table bounds - 1) because bilinear interpolation requires checking (index + 1) */
124  *val = constrain(*val, min, max - SAMPLING_RES);
125 
126  return static_cast<unsigned>((-(min) + *val) / SAMPLING_RES);
127 }

129 static float
130 get_table_data(float lat, float lon, const int8_t table[13][37])
131 {
132  /*
133  * If the values exceed valid ranges, return zero as default
134  * as we have no way of knowing what the closest real value
135  * would be.
136  */
137  if (lat < -90.0f || lat > 90.0f ||
138  lon < -180.0f || lon > 180.0f) {
139  return 0.0f;
140  }
141 
142  /* round down to nearest sampling resolution */
143  float min_lat = floorf(lat / SAMPLING_RES) * SAMPLING_RES;
144  float min_lon = floorf(lon / SAMPLING_RES) * SAMPLING_RES;
145 
146  /* find index of nearest low sampling point */
147  unsigned min_lat_index = get_lookup_table_index(&min_lat, SAMPLING_MIN_LAT, SAMPLING_MAX_LAT);
148  unsigned min_lon_index = get_lookup_table_index(&min_lon, SAMPLING_MIN_LON, SAMPLING_MAX_LON);
149 
150  const float data_sw = table[min_lat_index][min_lon_index];
151  const float data_se = table[min_lat_index][min_lon_index + 1];
152  const float data_ne = table[min_lat_index + 1][min_lon_index + 1];
153  const float data_nw = table[min_lat_index + 1][min_lon_index];
154 
155  /* perform bilinear interpolation on the four grid corners */
156  const float lat_scale = constrain((lat - min_lat) / SAMPLING_RES, 0.0f, 1.0f);
157  const float lon_scale = constrain((lon - min_lon) / SAMPLING_RES, 0.0f, 1.0f);
158 
159  const float data_min = lon_scale * (data_se - data_sw) + data_sw;
160  const float data_max = lon_scale * (data_ne - data_nw) + data_nw;
161 
162  return lat_scale * (data_max - data_min) + data_min;
163 }
