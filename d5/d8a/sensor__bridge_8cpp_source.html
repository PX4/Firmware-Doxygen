69 void UavcanCDevSensorBridgeBase::publish(const int node_id, const void *report)
70 {
71  assert(report != nullptr);
72 
73  Channel *channel = nullptr;
74 
75  // Checking if such channel already exists
76  for (unsigned i = 0; i < _max_channels; i++) {
77  if (_channels[i].node_id == node_id) {
78  channel = _channels + i;
79  break;
80  }
81  }
82 
83  // No such channel - try to create one
84  if (channel == nullptr) {
85  if (_out_of_channels) {
86  return; // Give up immediately - saves some CPU time
87  }
88 
89  DEVICE_LOG("adding channel %d...", node_id);
90 
91  // Search for the first free channel
92  for (unsigned i = 0; i < _max_channels; i++) {
93  if (_channels[i].node_id < 0) {
94  channel = _channels + i;
95  break;
96  }
97  }
98 
99  // No free channels left
100  if (channel == nullptr) {
101  _out_of_channels = true;
102  DEVICE_LOG("out of channels");
103  return;
104  }
105 
106  // update device id as we now know our device node_id
107  _device_id.devid_s.address = static_cast<uint8_t>(node_id);
108 
109  // Ask the CDev helper which class instance we can take
110  const int class_instance = register_class_devname(_class_devname);
111 
112  if (class_instance < 0 || class_instance >= int(_max_channels)) {
113  _out_of_channels = true;
114  DEVICE_LOG("out of class instances");
115  (void)unregister_class_devname(_class_devname, class_instance);
116  return;
117  }
118 
119  // Publish to the appropriate topic, abort on failure
120  channel->node_id = node_id;
121  channel->class_instance = class_instance;
122 
123  channel->orb_advert = orb_advertise_multi(_orb_topic, report, &channel->orb_instance, ORB_PRIO_VERY_HIGH);
124 
125  if (channel->orb_advert == nullptr) {
126  DEVICE_LOG("ADVERTISE FAILED");
127  (void)unregister_class_devname(_class_devname, class_instance);
128  *channel = Channel();
129  return;
130  }
131 
132  DEVICE_LOG("channel %d class instance %d ok", channel->node_id, channel->class_instance);
133  }
134 
135  assert(channel != nullptr);
136 
137  (void)orb_publish(_orb_topic, channel->orb_advert, report);
138 }
