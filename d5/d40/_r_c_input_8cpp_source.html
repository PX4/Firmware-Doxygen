109 int
110 RCInput::task_spawn(int argc, char *argv[])
111 {
112  bool run_as_task = false;
113  bool error_flag = false;
114 
115  int myoptind = 1;
116  int ch;
117  const char *myoptarg = nullptr;
118 
119  while ((ch = px4_getopt(argc, argv, "t", &myoptind, &myoptarg)) != EOF) {
120  switch (ch) {
121  case 't':
122  run_as_task = true;
123  break;
124 
125  case '?':
126  error_flag = true;
127  break;
128 
129  default:
130  PX4_WARN("unrecognized flag");
131  error_flag = true;
132  break;
133  }
134  }
135 
136  if (error_flag) {
137  return -1;
138  }
139 
140 
141  if (!run_as_task) {
142 
143  /* schedule a cycle to start things */
144  int ret = work_queue(HPWORK, &_work, (worker_t)&RCInput::cycle_trampoline, nullptr, 0);
145 
146  if (ret < 0) {
147  return ret;
148  }
149 
150  _task_id = task_id_is_work_queue;
151 
152  } else {
153 
154  /* start the IO interface task */
155 
156  _task_id = px4_task_spawn_cmd("rc_input",
157  SCHED_DEFAULT,
158  SCHED_PRIORITY_SLOW_DRIVER,
159  1000,
160  (px4_main_t)&run_trampoline,
161  nullptr);
162 
163  if (_task_id < 0) {
164  _task_id = -1;
165  return -errno;
166  }
167  }
168 
169  return PX4_OK;
170 }

198 void
199 RCInput::fill_rc_in(uint16_t raw_rc_count_local,
200  uint16_t raw_rc_values_local[input_rc_s::RC_INPUT_MAX_CHANNELS],
201  hrt_abstime now, bool frame_drop, bool failsafe,
202  unsigned frame_drops, int rssi = -1)
203 {
204  // fill rc_in struct for publishing
205  _rc_in.channel_count = raw_rc_count_local;
206 
207  if (_rc_in.channel_count > input_rc_s::RC_INPUT_MAX_CHANNELS) {
208  _rc_in.channel_count = input_rc_s::RC_INPUT_MAX_CHANNELS;
209  }
210 
211  unsigned valid_chans = 0;
212 
213  for (unsigned i = 0; i < _rc_in.channel_count; i++) {
214  _rc_in.values[i] = raw_rc_values_local[i];
215 
216  if (raw_rc_values_local[i] != UINT16_MAX) {
217  valid_chans++;
218  }
219 
220  // once filled, reset values back to default
221  _raw_rc_values[i] = UINT16_MAX;
222  }
223 
224  _rc_in.timestamp = now;
225  _rc_in.timestamp_last_signal = _rc_in.timestamp;
226  _rc_in.rc_ppm_frame_length = 0;
227 
228  /* fake rssi if no value was provided */
229  if (rssi == -1) {
230 
231  /* set RSSI if analog RSSI input is present */
232  if (_analog_rc_rssi_stable) {
233  float rssi_analog = ((_analog_rc_rssi_volt - 0.2f) / 3.0f) * 100.0f;
234 
235  if (rssi_analog > 100.0f) {
236  rssi_analog = 100.0f;
237  }
238 
239  if (rssi_analog < 0.0f) {
240  rssi_analog = 0.0f;
241  }
242 
243  _rc_in.rssi = rssi_analog;
244 
245  } else {
246  _rc_in.rssi = 255;
247  }
248 
249  } else {
250  _rc_in.rssi = rssi;
251  }
252 
253  if (valid_chans == 0) {
254  _rc_in.rssi = 0;
255  }
256 
257  _rc_in.rc_failsafe = failsafe;
258  _rc_in.rc_lost = (valid_chans == 0);
259  _rc_in.rc_lost_frame_count = frame_drops;
260  _rc_in.rc_total_frame_count = 0;
261 }

289 void
290 RCInput::cycle()
291 {
292  while (true) {
293 
294  perf_begin(_cycle_perf);
295 
296  const hrt_abstime cycle_timestamp = hrt_absolute_time();
297 
298 #if defined(SPEKTRUM_POWER)
299  /* vehicle command */
300  bool vehicle_command_updated = false;
301  orb_check(_vehicle_cmd_sub, &vehicle_command_updated);
302 
303  if (vehicle_command_updated) {
304  vehicle_command_s vcmd = {};
305  orb_copy(ORB_ID(vehicle_command), _vehicle_cmd_sub, &vcmd);
306 
307  // Check for a pairing command
308  if ((unsigned int)vcmd.command == vehicle_command_s::VEHICLE_CMD_START_RX_PAIR) {
309  if (!_rc_scan_locked /* !_armed.armed */) { // TODO: add armed check?
310  if ((int)vcmd.param1 == 0) {
311  // DSM binding command
312  int dsm_bind_mode = (int)vcmd.param2;
313 
314  int dsm_bind_pulses = 0;
315 
316  if (dsm_bind_mode == 0) {
317  dsm_bind_pulses = DSM2_BIND_PULSES;
318 
319  } else if (dsm_bind_mode == 1) {
320  dsm_bind_pulses = DSMX_BIND_PULSES;
321 
322  } else {
323  dsm_bind_pulses = DSMX8_BIND_PULSES;
324  }
325 
326  bind_spektrum(dsm_bind_pulses);
327  }
328 
329  } else {
330  PX4_WARN("system armed, bind request rejected");
331  }
332  }
333  }
334 
335 #endif /* SPEKTRUM_POWER */
336 
337  /* update ADC sampling */
338 #ifdef ADC_RC_RSSI_CHANNEL
339  bool adc_updated = false;
340  orb_check(_adc_sub, &adc_updated);
341 
342  if (adc_updated) {
343 
344  struct adc_report_s adc;
345  orb_copy(ORB_ID(adc_report), _adc_sub, &adc);
346  const unsigned adc_chans = sizeof(adc.channel_id) / sizeof(adc.channel_id[0]);
347 
348  for (unsigned i = 0; i < adc_chans; i++) {
349  if (adc.channel_id[i] == ADC_RC_RSSI_CHANNEL) {
350 
351  if (_analog_rc_rssi_volt < 0.0f) {
352  _analog_rc_rssi_volt = adc.channel_value[i];
353  }
354 
355  _analog_rc_rssi_volt = _analog_rc_rssi_volt * 0.995f + adc.channel_value[i] * 0.005f;
356 
357  /* only allow this to be used if we see a high RSSI once */
358  if (_analog_rc_rssi_volt > 2.5f) {
359  _analog_rc_rssi_stable = true;
360  }
361  }
362  }
363  }
364 
365 #endif /* ADC_RC_RSSI_CHANNEL */
366 
367  bool rc_updated = false;
368 
369 #ifdef RC_SERIAL_PORT
370  // This block scans for a supported serial RC input and locks onto the first one found
371  // Scan for 300 msec, then switch protocol
372  constexpr hrt_abstime rc_scan_max = 300_ms;
373 
374  bool sbus_failsafe, sbus_frame_drop;
375  unsigned frame_drops;
376  bool dsm_11_bit;
377 
378  if (_report_lock && _rc_scan_locked) {
379  _report_lock = false;
380  //PX4_WARN("RCscan: %s RC input locked", RC_SCAN_STRING[_rc_scan_state]);
381  }
382 
383  int newBytes = 0;
384 
385  if (_run_as_task) {
386  // TODO: needs work (poll _rcs_fd)
387  // int ret = poll(fds, sizeof(fds) / sizeof(fds[0]), 100);
388  // then update priority to SCHED_PRIORITY_FAST_DRIVER
389  // read all available data from the serial RC input UART
390  newBytes = ::read(_rcs_fd, &_rcs_buf[0], SBUS_BUFFER_SIZE);
391 
392  } else {
393  // read all available data from the serial RC input UART
394  newBytes = ::read(_rcs_fd, &_rcs_buf[0], SBUS_BUFFER_SIZE);
395  }
396 
397  switch (_rc_scan_state) {
398  case RC_SCAN_SBUS:
399  if (_rc_scan_begin == 0) {
400  _rc_scan_begin = cycle_timestamp;
401  // Configure serial port for SBUS
402  sbus_config(_rcs_fd, board_supports_single_wire(RC_UXART_BASE));
403  rc_io_invert(true, RC_UXART_BASE);
404 
405  } else if (_rc_scan_locked
406  || cycle_timestamp - _rc_scan_begin < rc_scan_max) {
407 
408  // parse new data
409  if (newBytes > 0) {
410  rc_updated = sbus_parse(cycle_timestamp, &_rcs_buf[0], newBytes, &_raw_rc_values[0], &_raw_rc_count, &sbus_failsafe,
411  &sbus_frame_drop, &frame_drops, input_rc_s::RC_INPUT_MAX_CHANNELS);
412 
413  if (rc_updated) {
414  // we have a new SBUS frame. Publish it.
415  _rc_in.input_source = input_rc_s::RC_INPUT_SOURCE_PX4FMU_SBUS;
416  fill_rc_in(_raw_rc_count, _raw_rc_values, cycle_timestamp,
417  sbus_frame_drop, sbus_failsafe, frame_drops);
418  _rc_scan_locked = true;
419  }
420  }
421 
422  } else {
423  // Scan the next protocol
424  set_rc_scan_state(RC_SCAN_DSM);
425  }
426 
427  break;
428 
429  case RC_SCAN_DSM:
430  if (_rc_scan_begin == 0) {
431  _rc_scan_begin = cycle_timestamp;
432  // // Configure serial port for DSM
433  dsm_config(_rcs_fd);
434  rc_io_invert(false, RC_UXART_BASE);
435 
436  } else if (_rc_scan_locked
437  || cycle_timestamp - _rc_scan_begin < rc_scan_max) {
438 
439  if (newBytes > 0) {
440  int8_t dsm_rssi;
441 
442  // parse new data
443  rc_updated = dsm_parse(cycle_timestamp, &_rcs_buf[0], newBytes, &_raw_rc_values[0], &_raw_rc_count,
444  &dsm_11_bit, &frame_drops, &dsm_rssi, input_rc_s::RC_INPUT_MAX_CHANNELS);
445 
446  if (rc_updated) {
447  // we have a new DSM frame. Publish it.
448  _rc_in.input_source = input_rc_s::RC_INPUT_SOURCE_PX4FMU_DSM;
449  fill_rc_in(_raw_rc_count, _raw_rc_values, cycle_timestamp,
450  false, false, frame_drops, dsm_rssi);
451  _rc_scan_locked = true;
452  }
453  }
454 
455  } else {
456  // Scan the next protocol
457  set_rc_scan_state(RC_SCAN_ST24);
458  }
459 
460  break;
461 
462  case RC_SCAN_ST24:
463  if (_rc_scan_begin == 0) {
464  _rc_scan_begin = cycle_timestamp;
465  // Configure serial port for DSM
466  dsm_config(_rcs_fd);
467  rc_io_invert(false, RC_UXART_BASE);
468 
469  } else if (_rc_scan_locked
470  || cycle_timestamp - _rc_scan_begin < rc_scan_max) {
471 
472  if (newBytes > 0) {
473  // parse new data
474  uint8_t st24_rssi, lost_count;
475 
476  rc_updated = false;
477 
478  for (unsigned i = 0; i < (unsigned)newBytes; i++) {
479  /* set updated flag if one complete packet was parsed */
480  st24_rssi = RC_INPUT_RSSI_MAX;
481  rc_updated = (OK == st24_decode(_rcs_buf[i], &st24_rssi, &lost_count,
482  &_raw_rc_count, _raw_rc_values, input_rc_s::RC_INPUT_MAX_CHANNELS));
483  }
484 
485  // The st24 will keep outputting RC channels and RSSI even if RC has been lost.
486  // The only way to detect RC loss is therefore to look at the lost_count.
487 
488  if (rc_updated) {
489  if (lost_count == 0) {
490  // we have a new ST24 frame. Publish it.
491  _rc_in.input_source = input_rc_s::RC_INPUT_SOURCE_PX4FMU_ST24;
492  fill_rc_in(_raw_rc_count, _raw_rc_values, cycle_timestamp,
493  false, false, frame_drops, st24_rssi);
494  _rc_scan_locked = true;
495 
496  } else {
497  // if the lost count > 0 means that there is an RC loss
498  _rc_in.rc_lost = true;
499  }
500  }
501  }
502 
503  } else {
504  // Scan the next protocol
505  set_rc_scan_state(RC_SCAN_SUMD);
506  }
507 
508  break;
509 
510  case RC_SCAN_SUMD:
511  if (_rc_scan_begin == 0) {
512  _rc_scan_begin = cycle_timestamp;
513  // Configure serial port for DSM
514  dsm_config(_rcs_fd);
515  rc_io_invert(false, RC_UXART_BASE);
516 
517  } else if (_rc_scan_locked
518  || cycle_timestamp - _rc_scan_begin < rc_scan_max) {
519 
520  if (newBytes > 0) {
521  // parse new data
522  uint8_t sumd_rssi, rx_count;
523  bool sumd_failsafe;
524 
525  rc_updated = false;
526 
527  for (unsigned i = 0; i < (unsigned)newBytes; i++) {
528  /* set updated flag if one complete packet was parsed */
529  sumd_rssi = RC_INPUT_RSSI_MAX;
530  rc_updated = (OK == sumd_decode(_rcs_buf[i], &sumd_rssi, &rx_count,
531  &_raw_rc_count, _raw_rc_values, input_rc_s::RC_INPUT_MAX_CHANNELS, &sumd_failsafe));
532  }
533 
534  if (rc_updated) {
535  // we have a new SUMD frame. Publish it.
536  _rc_in.input_source = input_rc_s::RC_INPUT_SOURCE_PX4FMU_SUMD;
537  fill_rc_in(_raw_rc_count, _raw_rc_values, cycle_timestamp,
538  false, sumd_failsafe, frame_drops, sumd_rssi);
539  _rc_scan_locked = true;
540  }
541  }
542 
543  } else {
544  // Scan the next protocol
545  set_rc_scan_state(RC_SCAN_PPM);
546  }
547 
548  break;
549 
550  case RC_SCAN_PPM:
551  // skip PPM if it's not supported
552 #ifdef HRT_PPM_CHANNEL
553  if (_rc_scan_begin == 0) {
554  _rc_scan_begin = cycle_timestamp;
555  // Configure timer input pin for CPPM
556  px4_arch_configgpio(GPIO_PPM_IN);
557  rc_io_invert(false, RC_UXART_BASE);
558 
559  } else if (_rc_scan_locked || cycle_timestamp - _rc_scan_begin < rc_scan_max) {
560 
561  // see if we have new PPM input data
562  if ((ppm_last_valid_decode != _rc_in.timestamp_last_signal) && ppm_decoded_channels > 3) {
563  // we have a new PPM frame. Publish it.
564  rc_updated = true;
565  _rc_in.input_source = input_rc_s::RC_INPUT_SOURCE_PX4FMU_PPM;
566  fill_rc_in(ppm_decoded_channels, ppm_buffer, cycle_timestamp, false, false, 0);
567  _rc_scan_locked = true;
568  _rc_in.rc_ppm_frame_length = ppm_frame_length;
569  _rc_in.timestamp_last_signal = ppm_last_valid_decode;
570  }
571 
572  } else {
573  // disable CPPM input by mapping it away from the timer capture input
574  px4_arch_unconfiggpio(GPIO_PPM_IN);
575  // Scan the next protocol
576  set_rc_scan_state(RC_SCAN_CRSF);
577  }
578 
579 #else // skip PPM if it's not supported
580  set_rc_scan_state(RC_SCAN_CRSF);
581 
582 #endif // HRT_PPM_CHANNEL
583 
584  break;
585 
586  case RC_SCAN_CRSF:
587  if (_rc_scan_begin == 0) {
588  _rc_scan_begin = cycle_timestamp;
589  // Configure serial port for CRSF
590  crsf_config(_rcs_fd);
591  rc_io_invert(false, RC_UXART_BASE);
592 
593  } else if (_rc_scan_locked
594  || cycle_timestamp - _rc_scan_begin < rc_scan_max) {
595 
596  // parse new data
597  if (newBytes > 0) {
598  rc_updated = crsf_parse(cycle_timestamp, &_rcs_buf[0], newBytes, &_raw_rc_values[0], &_raw_rc_count,
599  input_rc_s::RC_INPUT_MAX_CHANNELS);
600 
601  if (rc_updated) {
602  // we have a new CRSF frame. Publish it.
603  _rc_in.input_source = input_rc_s::RC_INPUT_SOURCE_PX4FMU_CRSF;
604  fill_rc_in(_raw_rc_count, _raw_rc_values, cycle_timestamp, false, false, 0);
605 
606  // Enable CRSF Telemetry only on the Omnibus, because on Pixhawk (-related) boards
607  // we cannot write to the RC UART
608  // It might work on FMU-v5. Or another option is to use a different UART port
609 #ifdef CONFIG_ARCH_BOARD_OMNIBUS_F4SD
610 
611  if (!_rc_scan_locked && !_crsf_telemetry) {
612  _crsf_telemetry = new CRSFTelemetry(_rcs_fd);
613  }
614 
615 #endif /* CONFIG_ARCH_BOARD_OMNIBUS_F4SD */
616 
617  _rc_scan_locked = true;
618 
619  if (_crsf_telemetry) {
620  _crsf_telemetry->update(cycle_timestamp);
621  }
622  }
623  }
624 
625  } else {
626  // Scan the next protocol
627  set_rc_scan_state(RC_SCAN_SBUS);
628  }
629 
630  break;
631  }
632 
633 #else // RC_SERIAL_PORT not defined
634 #ifdef HRT_PPM_CHANNEL
635 
636  // see if we have new PPM input data
637  if ((ppm_last_valid_decode != _rc_in.timestamp_last_signal) && ppm_decoded_channels > 3) {
638  // we have a new PPM frame. Publish it.
639  rc_updated = true;
640  fill_rc_in(ppm_decoded_channels, ppm_buffer, cycle_timestamp, false, false, 0);
641  _rc_in.rc_ppm_frame_length = ppm_frame_length;
642  _rc_in.timestamp_last_signal = ppm_last_valid_decode;
643  }
644 
645 #endif // HRT_PPM_CHANNEL
646 #endif // RC_SERIAL_PORT
647 
648  perf_end(_cycle_perf);
649 
650  if (rc_updated) {
651  perf_count(_publish_interval_perf);
652 
653  int instance;
654  orb_publish_auto(ORB_ID(input_rc), &_to_input_rc, &_rc_in, &instance, ORB_PRIO_DEFAULT);
655 
656  } else if (!rc_updated && ((hrt_absolute_time() - _rc_in.timestamp_last_signal) > 1_s)) {
657  _rc_scan_locked = false;
658  }
659
