112 int
113 LPS25H_I2C::write(unsigned address, void *data, unsigned count)
114 {
115  uint8_t buf[32];
116 
117  if (sizeof(buf) < (count + 1)) {
118  return -EIO;
119  }
120 
121  buf[0] = address;
122  memcpy(&buf[1], data, count);
123 
124  return transfer(&buf[0], count + 1, nullptr, 0);
