47  const bool optional, int32_t &device_id, const bool report_fail)

49  const bool exists = (orb_exists(ORB_ID(sensor_gyro), instance) == PX4_OK);
50  bool calibration_valid = false;
51  bool gyro_valid = false;
52 
53  if (exists) {
54 
55  uORB::SubscriptionData<sensor_gyro_s> gyro{ORB_ID(sensor_gyro), instance};
56 
57  gyro_valid = (hrt_elapsed_time(&gyro.get().timestamp) < 1_s);
58 
59  if (!gyro_valid) {
60  if (report_fail) {
61  mavlink_log_critical(mavlink_log_pub, "Preflight Fail: no valid data from Gyro #%u", instance);
62  }
63  }
64 
65  device_id = gyro.get().device_id;
66 
67  calibration_valid = check_calibration("CAL_GYRO%u_ID", device_id);
68 
69  if (!calibration_valid) {
70  if (report_fail) {
71  mavlink_log_critical(mavlink_log_pub, "Preflight Fail: Gyro #%u uncalibrated", instance);
72  }
73  }
74 
75  } else {
76  if (!optional && report_fail) {
77  mavlink_log_critical(mavlink_log_pub, "Preflight Fail: Gyro Sensor #%u missing", instance);
78  }
79  }
80 
81  if (instance == 0) {
82  set_health_flags(subsystem_info_s::SUBSYSTEM_TYPE_GYRO, exists, !optional, calibration_valid && gyro_valid, status);
83 
84  } else if (instance == 1) {
85  set_health_flags(subsystem_info_s::SUBSYSTEM_TYPE_GYRO2, exists, !optional, calibration_valid && gyro_valid, status);
86  }
