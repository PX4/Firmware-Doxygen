73 SimpleMixer::parse_output_scaler(const char *buf, unsigned &buflen, mixer_scaler_s &scaler)
74 {
75  int ret;
76  int s[5];
77  int n = -1;
78 
79  buf = findtag(buf, buflen, 'O');
80 
81  if ((buf == nullptr) || (buflen < 12)) {
82  debug("output parser failed finding tag, ret: '%s'", buf);
83  return -1;
84  }
85 
86  if ((ret = sscanf(buf, "O: %d %d %d %d %d %n",
87  &s[0], &s[1], &s[2], &s[3], &s[4], &n)) != 5) {
88  debug("out scaler parse failed on '%s' (got %d, consumed %d)", buf, ret, n);
89  return -1;
90  }
91 
92  buf = skipline(buf, buflen);
93 
94  if (buf == nullptr) {
95  debug("no line ending, line is incomplete");
96  return -1;
97  }
98 
99  scaler.negative_scale = s[0] / 10000.0f;
100  scaler.positive_scale = s[1] / 10000.0f;
101  scaler.offset = s[2] / 10000.0f;
102  scaler.min_output = s[3] / 10000.0f;
103  scaler.max_output = s[4] / 10000.0f;
104 
105  return 0;
106 }

146 SimpleMixer *
147 SimpleMixer::from_text(Mixer::ControlCallback control_cb, uintptr_t cb_handle, const char *buf, unsigned &buflen)
148 {
149  SimpleMixer *sm = nullptr;
150  mixer_simple_s *mixinfo = nullptr;
151  unsigned inputs;
152  int used;
153  const char *end = buf + buflen;
154  char next_tag;
155 
156  /* enforce that the mixer ends with a new line */
157  if (!string_well_formed(buf, buflen)) {
158  return nullptr;
159  }
160 
161  /* get the base info for the mixer */
162  if (sscanf(buf, "M: %u%n", &inputs, &used) != 1) {
163  debug("simple parse failed on '%s'", buf);
164  goto out;
165  }
166 
167  /* at least 1 input is required */
168  if (inputs == 0) {
169  debug("simple parse got 0 inputs");
170  goto out;
171  }
172 
173  buf = skipline(buf, buflen);
174 
175  if (buf == nullptr) {
176  debug("no line ending, line is incomplete");
177  goto out;
178  }
179 
180  mixinfo = (mixer_simple_s *)malloc(MIXER_SIMPLE_SIZE(inputs));
181 
182  if (mixinfo == nullptr) {
183  debug("could not allocate memory for mixer info");
184  goto out;
185  }
186 
187  mixinfo->control_count = inputs;
188 
189  /* find the next tag */
190  next_tag = findnexttag(end - buflen, buflen);
191 
192  if (next_tag == 'S') {
193 
194  /* No output scalers specified. Use default values.
195  * Corresponds to:
196  * O: 10000 10000 0 -10000 10000
197  */
198  mixinfo->output_scaler.negative_scale = 1.0f;
199  mixinfo->output_scaler.positive_scale = 1.0f;
200  mixinfo->output_scaler.offset = 0.f;
201  mixinfo->output_scaler.min_output = -1.0f;
202  mixinfo->output_scaler.max_output = 1.0f;
203 
204  } else {
205 
206  if (parse_output_scaler(end - buflen, buflen, mixinfo->output_scaler)) {
207  debug("simple mixer parser failed parsing out scaler tag, ret: '%s'", buf);
208  goto out;
209  }
210  }
211 
212  for (unsigned i = 0; i < inputs; i++) {
213  if (parse_control_scaler(end - buflen, buflen,
214  mixinfo->controls[i].scaler,
215  mixinfo->controls[i].control_group,
216  mixinfo->controls[i].control_index)) {
217  debug("simple mixer parser failed parsing ctrl scaler tag, ret: '%s'", buf);
218  goto out;
219  }
220 
221  }
222 
223  sm = new SimpleMixer(control_cb, cb_handle, mixinfo);
224 
225  if (sm != nullptr) {
226  mixinfo = nullptr;
227  debug("loaded mixer with %d input(s)", inputs);
228 
229  } else {
230  debug("could not allocate memory for mixer");
231  }
232 
233 out:
234 
235  if (mixinfo != nullptr) {
236  free(mixinfo);
237  }
238 
239  return sm;
240 }

242 SimpleMixer *
243 SimpleMixer::pwm_input(Mixer::ControlCallback control_cb, uintptr_t cb_handle, unsigned input, uint16_t min,
244  uint16_t mid, uint16_t max)
245 {
246  SimpleMixer *sm = nullptr;
247  mixer_simple_s *mixinfo = nullptr;
248 
249  mixinfo = (mixer_simple_s *)malloc(MIXER_SIMPLE_SIZE(1));
250 
251  if (mixinfo == nullptr) {
252  debug("could not allocate memory for mixer info");
253  goto out;
254  }
255 
256  mixinfo->control_count = 1;
257 
258  /*
259  * Always pull from group 0, with the input value giving the channel.
260  */
261  mixinfo->controls[0].control_group = 0;
262  mixinfo->controls[0].control_index = input;
263 
264  /*
265  * Conversion uses both the input and output side of the mixer.
266  *
267  * The input side is used to slide the control value such that the min argument
268  * results in a value of zero.
269  *
270  * The output side is used to apply the scaling for the min/max values so that
271  * the resulting output is a -1.0 ... 1.0 value for the min...max range.
272  */
273  mixinfo->controls[0].scaler.negative_scale = 1.0f;
274  mixinfo->controls[0].scaler.positive_scale = 1.0f;
275  mixinfo->controls[0].scaler.offset = -mid;
276  mixinfo->controls[0].scaler.min_output = -(mid - min);
277  mixinfo->controls[0].scaler.max_output = (max - mid);
278 
279  mixinfo->output_scaler.negative_scale = 500.0f / (mid - min);
280  mixinfo->output_scaler.positive_scale = 500.0f / (max - mid);
281  mixinfo->output_scaler.offset = 0.0f;
282  mixinfo->output_scaler.min_output = -1.0f;
283  mixinfo->output_scaler.max_output = 1.0f;
284 
285  sm = new SimpleMixer(control_cb, cb_handle, mixinfo);
286 
287  if (sm != nullptr) {
288  mixinfo = nullptr;
289  debug("PWM input mixer for %d", input);
290 
291  } else {
292  debug("could not allocate memory for PWM input mixer");
293  }
294 
295 out:
296 
297  if (mixinfo != nullptr) {
298  free(mixinfo);
299  }
300 
301  return sm;
302 }

304 unsigned
305 SimpleMixer::mix(float *outputs, unsigned space)
306 {
307  float sum = 0.0f;
308 
309  if (_pinfo == nullptr) {
310  return 0;
311  }
312 
313  if (space < 1) {
314  return 0;
315  }
316 
317  for (unsigned i = 0; i < _pinfo->control_count; i++) {
318  float input = 0.0f;
319 
320  _control_cb(_cb_handle,
321  _pinfo->controls[i].control_group,
322  _pinfo->controls[i].control_index,
323  input);
324 
325  sum += scale(_pinfo->controls[i].scaler, input);
326  }
327 
328  *outputs = scale(_pinfo->output_scaler, sum);
329  return 1;
330 }

346 int
347 SimpleMixer::check()
348 {
349  int ret;
350  float junk;
351 
352  /* sanity that presumes that a mixer includes a control no more than once */
353  /* max of 32 groups due to groups_required API */
354  if (_pinfo->control_count > 32) {
355  return -2;
356  }
357 
358  /* validate the output scaler */
359  ret = scale_check(_pinfo->output_scaler);
360 
361  if (ret != 0) {
362  return ret;
363  }
364 
365  /* validate input scalers */
366  for (unsigned i = 0; i < _pinfo->control_count; i++) {
367 
368  /* verify that we can fetch the control */
369  if (_control_cb(_cb_handle,
370  _pinfo->controls[i].control_group,
371  _pinfo->controls[i].control_index,
372  junk) != 0) {
373  return -3;
374  }
375 
376  /* validate the scaler */
377  ret = scale_check(_pinfo->controls[i].scaler);
378 
379  if (ret != 0) {
380  return (10 * i + ret);
381  }
382  }
383 
384  return 0;
385 }
