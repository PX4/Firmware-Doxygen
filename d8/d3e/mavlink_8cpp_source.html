132 void Mavlink::handle_msg_set_attitude_target(const mavlink_message_t *mmsg)
133 {
134  mavlink_set_attitude_target_t set_attitude_target;
135  mavlink_msg_set_attitude_target_decode(mmsg, &set_attitude_target);
136 
137  /* set correct ignore flags for thrust field: copy from mavlink message */
138  _offboard_control_mode.ignore_thrust = (bool)(set_attitude_target.type_mask & (1 << 6));
139 
140  /*
141  * if attitude or body rate have been used (not ignored) previously and this message only sends
142  * throttle and has the ignore bits set for attitude and rates don't change the flags for attitude and
143  * body rates to keep the controllers running
144  */
145  bool ignore_bodyrate = (bool)(set_attitude_target.type_mask & 0x7);
146  bool ignore_attitude = (bool)(set_attitude_target.type_mask & (1 << 7));
147 
148  if (ignore_bodyrate && ignore_attitude && !_offboard_control_mode.ignore_thrust) {
149  /* Message want's us to ignore everything except thrust: only ignore if previously ignored */
150  _offboard_control_mode.ignore_bodyrate = ignore_bodyrate && _offboard_control_mode.ignore_bodyrate;
151  _offboard_control_mode.ignore_attitude = ignore_attitude && _offboard_control_mode.ignore_attitude;
152 
153  } else {
154  _offboard_control_mode.ignore_bodyrate = ignore_bodyrate;
155  _offboard_control_mode.ignore_attitude = ignore_attitude;
156  }
157 
158  _offboard_control_mode.ignore_position = true;
159  _offboard_control_mode.ignore_velocity = true;
160  _offboard_control_mode.ignore_acceleration_force = true;
161 
162  _offboard_control_mode.timestamp = get_time_micros();
163  _offboard_control_mode_pub.publish(_offboard_control_mode);
164 
165  /* The real mavlink app has a ckeck at this location which makes sure that the attitude setpoint
166  * gets published only if in offboard mode. We leave that out for now.
167  */
168 
169  _att_sp.timestamp = get_time_micros();
170 
171  if (!ignore_attitude) {
172  mavlink_quaternion_to_euler(set_attitude_target.q, &_att_sp.roll_body, &_att_sp.pitch_body,
173  &_att_sp.yaw_body);
174  mavlink_quaternion_to_dcm(set_attitude_target.q, (float(*)[3])_att_sp.R_body.data());
175  _att_sp.R_valid = true;
176  }
177 
178 
179  if (!_offboard_control_mode.ignore_thrust) {
180  _att_sp.thrust = set_attitude_target.thrust;
181  }
182 
183  if (!ignore_attitude) {
184  for (ssize_t i = 0; i < 4; i++) {
185  _att_sp.q_d[i] = set_attitude_target.q[i];
186  }
187 
188  _att_sp.q_d_valid = true;
189  }
190 
191  _v_att_sp_pub.publish(_att_sp);
192 
193 
194  //XXX real mavlink publishes rate sp here
195 
196 }

198 void Mavlink::handle_msg_set_position_target_local_ned(const mavlink_message_t *mmsg)
199 {
200 
201  mavlink_set_position_target_local_ned_t set_position_target_local_ned;
202  mavlink_msg_set_position_target_local_ned_decode(mmsg, &set_position_target_local_ned);
203 
204  offboard_control_mode offboard_control_mode;
205  // memset(&offboard_control_mode, 0, sizeof(offboard_control_mode));//XXX breaks compatibility with multiple setpoints
206 
207  /* Only accept messages which are intended for this system */
208  // XXX removed for sitl, makes maybe sense to re-introduce at some point
209  // if ((mavlink_system.sysid == set_position_target_local_ned.target_system ||
210  // set_position_target_local_ned.target_system == 0) &&
211  // (mavlink_system.compid == set_position_target_local_ned.target_component ||
212  // set_position_target_local_ned.target_component == 0)) {
213 
214  /* convert mavlink type (local, NED) to uORB offboard control struct */
215  offboard_control_mode.ignore_position = (bool)(set_position_target_local_ned.type_mask & 0x7);
216  offboard_control_mode.ignore_velocity = (bool)(set_position_target_local_ned.type_mask & 0x38);
217  offboard_control_mode.ignore_acceleration_force = (bool)(set_position_target_local_ned.type_mask & 0x1C0);
218  bool is_force_sp = (bool)(set_position_target_local_ned.type_mask & (1 << 9));
219  /* yaw ignore flag mapps to ignore_attitude */
220  offboard_control_mode.ignore_attitude = (bool)(set_position_target_local_ned.type_mask & 0x400);
221  /* yawrate ignore flag mapps to ignore_bodyrate */
222  offboard_control_mode.ignore_bodyrate = (bool)(set_position_target_local_ned.type_mask & 0x800);
223 
224 
225 
226  offboard_control_mode.timestamp = get_time_micros();
227  _offboard_control_mode_pub.publish(offboard_control_mode);
228 
229  /* The real mavlink app has a ckeck at this location which makes sure that the position setpoint triplet
230  * gets published only if in offboard mode. We leave that out for now.
231  */
232  if (is_force_sp && offboard_control_mode.ignore_position &&
233  offboard_control_mode.ignore_velocity) {
234  /* The offboard setpoint is a force setpoint only, directly writing to the force
235  * setpoint topic and not publishing the setpoint triplet topic */
236 
237  PX4_WARN("force setpoint not supported");
238 
239  } else {
240  /* It's not a pure force setpoint: publish to setpoint triplet topic */
241  position_setpoint_triplet pos_sp_triplet;
242  pos_sp_triplet.previous.valid = false;
243  pos_sp_triplet.next.valid = false;
244  pos_sp_triplet.current.valid = true;
245  pos_sp_triplet.current.type = position_setpoint::SETPOINT_TYPE_POSITION; //XXX support others
246 
247  /* set the local pos values */
248  if (!offboard_control_mode.ignore_position) {
249  pos_sp_triplet.current.position_valid = true;
250  pos_sp_triplet.current.x = set_position_target_local_ned.x;
251  pos_sp_triplet.current.y = set_position_target_local_ned.y;
252  pos_sp_triplet.current.z = set_position_target_local_ned.z;
253 
254  } else {
255  pos_sp_triplet.current.position_valid = false;
256  }
257 
258  /* set the local vel values */
259  if (!offboard_control_mode.ignore_velocity) {
260  pos_sp_triplet.current.velocity_valid = true;
261  pos_sp_triplet.current.vx = set_position_target_local_ned.vx;
262  pos_sp_triplet.current.vy = set_position_target_local_ned.vy;
263  pos_sp_triplet.current.vz = set_position_target_local_ned.vz;
264 
265  } else {
266  pos_sp_triplet.current.velocity_valid = false;
267  }
268 
269  /* set the local acceleration values if the setpoint type is 'local pos' and none
270  * of the accelerations fields is set to 'ignore' */
271  if (!offboard_control_mode.ignore_acceleration_force) {
272  pos_sp_triplet.current.acceleration_valid = true;
273  pos_sp_triplet.current.a_x = set_position_target_local_ned.afx;
274  pos_sp_triplet.current.a_y = set_position_target_local_ned.afy;
275  pos_sp_triplet.current.a_z = set_position_target_local_ned.afz;
276  pos_sp_triplet.current.acceleration_is_force =
277  is_force_sp;
278 
279  } else {
280  pos_sp_triplet.current.acceleration_valid = false;
281  }
282 
283  /* set the yaw sp value */
284  if (!offboard_control_mode.ignore_attitude && !isnan(set_position_target_local_ned.yaw)) {
285  pos_sp_triplet.current.yaw_valid = true;
286  pos_sp_triplet.current.yaw = set_position_target_local_ned.yaw;
287 
288  } else {
289  pos_sp_triplet.current.yaw_valid = false;
290  }
291 
292  /* set the yawrate sp value */
293  if (!offboard_control_mode.ignore_bodyrate && !isnan(set_position_target_local_ned.yaw)) {
294  pos_sp_triplet.current.yawspeed_valid = true;
295  pos_sp_triplet.current.yawspeed = set_position_target_local_ned.yaw_rate;
296 
297  } else {
298  pos_sp_triplet.current.yawspeed_valid = false;
299  }
300 
301  //XXX handle global pos setpoints (different MAV frames)
302 
303  _pos_sp_triplet_pub.publish(pos_sp_triplet);
304  }
