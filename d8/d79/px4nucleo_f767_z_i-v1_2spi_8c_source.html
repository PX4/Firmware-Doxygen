136 __EXPORT int stm32_spi_bus_initialize(void)
137 {
138  /* Configure SPI-based devices */
139 
140  spi_sensors = stm32_spibus_initialize(PX4_SPI_BUS_SENSORS);
141 
142  if (!spi_sensors) {
143  message("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_SENSORS);
144  return -ENODEV;
145  }
146 
147  /* Default PX4_SPI_BUS_SENSORS to 1MHz and de-assert the known chip selects. */
148  SPI_SETFREQUENCY(spi_sensors, 10000000);
149  SPI_SETBITS(spi_sensors, 8);
150  SPI_SETMODE(spi_sensors, SPIDEV_MODE3);
151 
152  for (int cs = PX4_SENSORS_BUS_FIRST_CS; cs <= PX4_SENSORS_BUS_LAST_CS; cs++) {
153  SPI_SELECT(spi_sensors, cs, false);
154  }
155 
156  /* Get the SPI port for the FRAM */
157 
158  spi_fram = stm32_spibus_initialize(PX4_SPI_BUS_RAMTRON);
159 
160  if (!spi_fram) {
161  message("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_RAMTRON);
162  return -ENODEV;
163  }
164 
165  /* Default SPI_BUS_RAMTRON to 12MHz and de-assert the known chip selects.
166  */
167 
168  SPI_SETFREQUENCY(spi_fram, 12 * 1000 * 1000);
169  SPI_SETBITS(spi_fram, 8);
170  SPI_SETMODE(spi_fram, SPIDEV_MODE3);
171 
172  for (int cs = PX4_RAMTRON_BUS_FIRST_CS; cs <= PX4_RAMTRON_BUS_LAST_CS; cs++) {
173  SPI_SELECT(spi_fram, cs, false);
174  }
175 
176  /* Get the SPI port for the BARO */
177 
178  spi_baro = stm32_spibus_initialize(PX4_SPI_BUS_BARO);
179 
180  if (!spi_baro) {
181  message("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_BARO);
182  return -ENODEV;
183  }
184 
185  /* MS5611 has max SPI clock speed of 20MHz
186  */
187 
188  SPI_SETFREQUENCY(spi_baro, 20 * 1000 * 1000);
189  SPI_SETBITS(spi_baro, 8);
190  SPI_SETMODE(spi_baro, SPIDEV_MODE3);
191 
192  for (int cs = PX4_BARO_BUS_FIRST_CS; cs <= PX4_BARO_BUS_LAST_CS; cs++) {
193  SPI_SELECT(spi_baro, cs, false);
194  }
195 
196  /* Get the SPI port for the PX4_SPI_BUS_ICM */
197 
198  spi_icm = stm32_spibus_initialize(PX4_SPI_BUS_ICM);
199 
200  if (!spi_icm) {
201  message("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_ICM);
202  return -ENODEV;
203  }
204 
205  /* ICM 20608 G has max SPI clock speed of 8MHz
206  */
207 
208  SPI_SETFREQUENCY(spi_icm, 8 * 1000 * 1000);
209  SPI_SETBITS(spi_icm, 8);
210  SPI_SETMODE(spi_icm, SPIDEV_MODE3);
211 
212  for (int cs = PX4_ICM_BUS_FIRST_CS; cs <= PX4_ICM_BUS_LAST_CS; cs++) {
213  SPI_SELECT(spi_icm, cs, false);
214  }
215 
216  return OK;
217 
218 }

224 __EXPORT void stm32_spi1select(FAR struct spi_dev_s *dev, uint32_t devid, bool selected)
225 {
226  /* SPI select is active low, so write !selected to select the device */
227 
228  int sel = (int) devid;
229  ASSERT(PX4_SPI_BUS_ID(sel) == PX4_SPI_BUS_SENSORS);
230 
231  /* Making sure the other peripherals are not selected */
232 
233  for (size_t cs = 0; arraySize(spi1selects_gpio) > 1 && cs < arraySize(spi1selects_gpio); cs++) {
234  if (spi1selects_gpio[cs] != 0) {
235  stm32_gpiowrite(spi1selects_gpio[cs], 1);
236  }
237  }
238 
239  uint32_t gpio = spi1selects_gpio[PX4_SPI_DEV_ID(sel)];
240 
241  if (gpio) {
242  stm32_gpiowrite(gpio, !selected);
243  }
244 }

345  stm32_configgpio(GPIO_SPI_CS_OFF_ICM_20608_G);
346 
347  stm32_gpiowrite(GPIO_SPI_CS_OFF_MPU9250, 0);
348  stm32_gpiowrite(GPIO_SPI_CS_OFF_HMC5983, 0);
349  stm32_gpiowrite(GPIO_SPI_CS_OFF_MS5611, 0);
350  stm32_gpiowrite(GPIO_SPI_CS_OFF_ICM_20608_G, 0);
351 
352  stm32_configgpio(GPIO_SPI1_SCK_OFF);
353  stm32_configgpio(GPIO_SPI1_MISO_OFF);
354  stm32_configgpio(GPIO_SPI1_MOSI_OFF);
355 
356  stm32_gpiowrite(GPIO_SPI1_SCK_OFF, 0);
357  stm32_gpiowrite(GPIO_SPI1_MISO_OFF, 0);
358  stm32_gpiowrite(GPIO_SPI1_MOSI_OFF, 0);
359 
360  stm32_configgpio(GPIO_DRDY_OFF_MPU9250);
361  stm32_configgpio(GPIO_DRDY_OFF_HMC5983);
362  stm32_configgpio(GPIO_DRDY_OFF_ICM_20608_G);
363 
364  stm32_gpiowrite(GPIO_DRDY_OFF_MPU9250, 0);
365  stm32_gpiowrite(GPIO_DRDY_OFF_HMC5983, 0);
366  stm32_gpiowrite(GPIO_DRDY_OFF_ICM_20608_G, 0);
367 
368  /* set the sensor rail off */
369  stm32_configgpio(GPIO_VDD_3V3_SENSORS_EN);
370  stm32_gpiowrite(GPIO_VDD_3V3_SENSORS_EN, 0);
371 
372  /* wait for the sensor rail to reach GND */
373  usleep(ms * 1000);
374  warnx("reset done, %d ms", ms);
375 
376  /* re-enable power */
377 
378  /* switch the sensor rail back on */
379  stm32_gpiowrite(GPIO_VDD_3V3_SENSORS_EN, 1);
380 
381  /* wait a bit before starting SPI, different times didn't influence results */
382  usleep(100);
383 
384  /* reconfigure the SPI pins */
385 #ifdef CONFIG_STM32F7_SPI1
386  stm32_configgpio(GPIO_SPI_CS_MPU9250);
387  stm32_configgpio(GPIO_SPI_CS_HMC5983);
388  stm32_configgpio(GPIO_SPI_CS_MS5611);

393  * state machines
394  */

397  stm32_gpiowrite(GPIO_SPI_CS_MS5611, 1);
398  stm32_gpiowrite(GPIO_SPI_CS_ICM_20608_G, 1);
399 
400  stm32_configgpio(GPIO_SPI1_SCK);
401  stm32_configgpio(GPIO_SPI1_MISO);
402  stm32_configgpio(GPIO_SPI1_MOSI);
403
