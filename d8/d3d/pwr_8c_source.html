108 static int board_button_irq(int irq, FAR void *context, FAR void *args)
109 {
110  if (board_pwr_button_down()) {
111 
112  led_on(BOARD_LED_RED);
113  clock_gettime(CLOCK_REALTIME, &time_down);
114  power_state_notification(PWR_BUTTON_DOWN);
115 
116  } else {
117 
118  power_state_notification(PWR_BUTTON_UP);
119 
120  led_off(BOARD_LED_RED);
121 
122  struct timespec now;
123 
124  clock_gettime(CLOCK_REALTIME, &now);
125 
126  uint64_t tdown_ms = time_down.tv_sec * 1000 + time_down.tv_nsec / 1000000;
127 
128  uint64_t tnow_ms = now.tv_sec * 1000 + now.tv_nsec / 1000000;
129 
130  if (tdown_ms != 0 && (tnow_ms - tdown_ms) >= MS_PWR_BUTTON_DOWN) {
131 
132  led_on(BOARD_LED_BLUE);
133 
134  if (power_state_notification(PWR_BUTTON_REQUEST_SHUT_DOWN) == PWR_BUTTON_RESPONSE_SHUT_DOWN_NOW) {
135  up_mdelay(200);
136  board_shutdown();
137  }
138 
139  } else {
140  power_state_notification(PWR_BUTTON_IDEL);
141  }
142  }
143 
144  return OK;
145 }

162 void board_pwr_init(int stage)
163 {
164  if (stage == 0) {
165  stm32_configgpio(POWER_ON_GPIO);
166  stm32_configgpio(KEY_AD_GPIO);
167  }
168 
169  if (stage == 1) {
