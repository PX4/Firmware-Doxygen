105  usleep(100000);

118  usleep(100000);
119  map_fail_count++;
120  }
121

124 
125  /* give system time to flush error message in case there are more */

196  /* give system time to flush error message in case there are more */

205  /* give system time to flush error message in case there are more */
206  usleep(100000);
207  }
208 
209  /* assert deadzone is sane */

211  if (report_fail) { mavlink_log_critical(mavlink_log_pub, "RC ERROR: RC%d_DZ > %u.", i + 1, RC_INPUT_MAX_DEADZONE_US); }
212 
213  /* give system time to flush error message in case there are more */
214  usleep(100000);

216  }
217

219

223  }
224 
225  if (channels_failed) {

232  }
233 
234  usleep(100000);
