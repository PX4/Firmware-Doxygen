239 int
240 SRF02::init()
241 {
242  int ret = PX4_ERROR;
243 
244  /* do I2C init (and probe) first */
245  if (I2C::init() != OK) {
246  return ret;
247  }
248 
249  /* allocate basic report buffers */
250  _reports = new ringbuffer::RingBuffer(2, sizeof(distance_sensor_s));
251 
252  _index_counter = SRF02_BASEADDR; /* set temp sonar i2c address to base adress */
253  set_device_address(_index_counter); /* set I2c port to temp sonar i2c adress */
254 
255  if (_reports == nullptr) {
256  return ret;
257  }
258 
259  _class_instance = register_class_devname(RANGE_FINDER_BASE_DEVICE_PATH);
260 
261  /* get a publish handle on the range finder topic */
262  struct distance_sensor_s ds_report = {};
263 
264  _distance_sensor_topic = orb_advertise_multi(ORB_ID(distance_sensor), &ds_report,
265  &_orb_class_instance, ORB_PRIO_LOW);
266 
267  if (_distance_sensor_topic == nullptr) {
268  DEVICE_LOG("failed to create distance_sensor object. Did you start uOrb?");
269  }
270 
271  // XXX we should find out why we need to wait 200 ms here
272  usleep(200000);
273 
274  /* check for connected rangefinders on each i2c port:
275  We start from i2c base address (0x70 = 112) and count downwards
276  So second iteration it uses i2c address 111, third iteration 110 and so on*/
277  for (unsigned counter = 0; counter <= MB12XX_MAX_RANGEFINDERS; counter++) {
278  _index_counter = SRF02_BASEADDR - counter; /* set temp sonar i2c address to base adress - counter */
279  set_device_address(_index_counter); /* set I2c port to temp sonar i2c adress */
280  int ret2 = measure();
281 
282  if (ret2 == 0) { /* sonar is present -> store address_index in array */
283  addr_ind.push_back(_index_counter);
284  DEVICE_DEBUG("sonar added");
285  _latest_sonar_measurements.push_back(200);
286  }
287  }
288 
289  _index_counter = SRF02_BASEADDR;
290  set_device_address(_index_counter); /* set i2c port back to base adress for rest of driver */
291 
292  /* if only one sonar detected, no special timing is required between firing, so use default */
293  if (addr_ind.size() == 1) {
294  _cycling_rate = SRF02_CONVERSION_INTERVAL;
295 
296  } else {
297  _cycling_rate = TICKS_BETWEEN_SUCCESIVE_FIRES;
298  }
299 
300  /* show the connected sonars in terminal */
301  for (unsigned i = 0; i < addr_ind.size(); i++) {
302  DEVICE_LOG("sonar %d with address %d added", (i + 1), addr_ind[i]);
303  }
304 
305  DEVICE_DEBUG("Number of sonars connected: %zu", addr_ind.size());
306 
307  ret = OK;
308  /* sensor is ok, but we don't really know if it is within range */
309  _sensor_ok = true;
310 
311  return ret;
312 }

344 int
345 SRF02::ioctl(device::file_t *filp, int cmd, unsigned long arg)
346 {
347  switch (cmd) {
348 
349  case SENSORIOCSPOLLRATE: {
350  switch (arg) {
351 
352  /* switching to manual polling */
353  case SENSOR_POLLRATE_MANUAL:
354  stop();
355  _measure_ticks = 0;
356  return OK;
357 
358  /* external signalling (DRDY) not supported */
359  case SENSOR_POLLRATE_EXTERNAL:
360 
361  /* zero would be bad */
362  case 0:
363  return -EINVAL;
364 
365  /* set default/max polling rate */
366  case SENSOR_POLLRATE_MAX:
367  case SENSOR_POLLRATE_DEFAULT: {
368  /* do we need to start internal polling? */
369  bool want_start = (_measure_ticks == 0);
370 
371  /* set interval for next measurement to minimum legal value */
372  _measure_ticks = USEC2TICK(_cycling_rate);
373 
374  /* if we need to start the poll state machine, do it */
375  if (want_start) {
376  start();
377 
378  }
379 
380  return OK;
381  }
382 
383  /* adjust to a legal polling interval in Hz */
384  default: {
385  /* do we need to start internal polling? */
386  bool want_start = (_measure_ticks == 0);
387 
388  /* convert hz to tick interval via microseconds */
389  int ticks = USEC2TICK(1000000 / arg);
390 
391  /* check against maximum rate */
392  if (ticks < USEC2TICK(_cycling_rate)) {
393  return -EINVAL;
394  }
395 
396  /* update interval for next measurement */
397  _measure_ticks = ticks;
398 
399  /* if we need to start the poll state machine, do it */
400  if (want_start) {
401  start();
402  }
403 
404  return OK;
405  }
406  }
407  }
408 
409  case SENSORIOCGPOLLRATE:
410  if (_measure_ticks == 0) {
411  return SENSOR_POLLRATE_MANUAL;
412  }
413 
414  return (1000 / _measure_ticks);
415 
416  case SENSORIOCSQUEUEDEPTH: {
417  /* lower bound is mandatory, upper bound is a sanity check */
418  if ((arg < 1) || (arg > 100)) {
419  return -EINVAL;
420  }
421 
422  ATOMIC_ENTER;
423 
424  if (!_reports->resize(arg)) {
425  ATOMIC_LEAVE;
426  return -ENOMEM;
427  }
428 
429  ATOMIC_LEAVE;
430 
431  return OK;
432  }
433 
434  case SENSORIOCRESET:
435  /* XXX implement this */
436  return -EINVAL;
437 
438  default:
439  /* give it to the superclass */
440  return I2C::ioctl(filp, cmd, arg);
441  }
442 }

444 ssize_t
445 SRF02::read(device::file_t *filp, char *buffer, size_t buflen)
446 {
447 
448  unsigned count = buflen / sizeof(struct distance_sensor_s);
449  struct distance_sensor_s *rbuf = reinterpret_cast<struct distance_sensor_s *>(buffer);
450  int ret = 0;
451 
452  /* buffer must be large enough */
453  if (count < 1) {
454  return -ENOSPC;
455  }
456 
457  /* if automatic measurement is enabled */
458  if (_measure_ticks > 0) {
459 
460  /*
461  * While there is space in the caller's buffer, and reports, copy them.
462  * Note that we may be pre-empted by the workq thread while we are doing this;
463  * we are careful to avoid racing with them.
464  */
465  while (count--) {
466  if (_reports->get(rbuf)) {
467  ret += sizeof(*rbuf);
468  rbuf++;
469  }
470  }
471 
472  /* if there was no data, warn the caller */
473  return ret ? ret : -EAGAIN;
474  }
475 
476  /* manual measurement - run one conversion */
477  do {
478  _reports->flush();
479 
480  /* trigger a measurement */
481  if (OK != measure()) {
482  ret = -EIO;
483  break;
484  }
485 
486  /* wait for it to complete */
487  usleep(_cycling_rate * 2);
488 
489  /* run the collection phase */
490  if (OK != collect()) {
491  ret = -EIO;
492  break;
493  }
494 
495  /* state machine will have generated a report, copy it out */
496  if (_reports->get(rbuf)) {
497  ret = sizeof(*rbuf);
498  }
499 
500  } while (0);
501 
502  return ret;
503 }

531 int
532 SRF02::collect()
533 {
534  int ret = -EIO;
535 
536  /* read from the sensor */
537  uint8_t val[2] = {0, 0};
538  uint8_t cmd = 0x02;
539  perf_begin(_sample_perf);
540 
541  ret = transfer(&cmd, 1, nullptr, 0);
542  ret = transfer(nullptr, 0, &val[0], 2);
543 
544  if (ret < 0) {
545  DEVICE_DEBUG("error reading from sensor: %d", ret);
546  perf_count(_comms_errors);
547  perf_end(_sample_perf);
548  return ret;
549  }
550 
551  uint16_t distance_cm = val[0] << 8 | val[1];
552  float distance_m = float(distance_cm) * 1e-2f;
553 
554  struct distance_sensor_s report;
555  report.timestamp = hrt_absolute_time();
556  report.type = distance_sensor_s::MAV_DISTANCE_SENSOR_ULTRASOUND;
557  report.orientation = _rotation;
558  report.current_distance = distance_m;
559  report.min_distance = get_minimum_distance();
560  report.max_distance = get_maximum_distance();
561  report.covariance = 0.0f;
562  report.signal_quality = -1;
563  /* TODO: set proper ID */
564  report.id = 0;
565 
566  /* publish it, if we are the primary */
567  if (_distance_sensor_topic != nullptr) {
568  orb_publish(ORB_ID(distance_sensor), _distance_sensor_topic, &report);
569  }
570 
571  _reports->force(&report);
572 
573  /* notify anyone waiting for data */
574  poll_notify(POLLIN);
575 
576  ret = OK;
577 
578  perf_end(_sample_perf);
579  return ret;
580 }

610 void
611 SRF02::cycle()
612 {
613  if (_collect_phase) {
614  _index_counter = addr_ind[_cycle_counter]; /*sonar from previous iteration collect is now read out */
615  set_device_address(_index_counter);
616 
617  /* perform collection */
618  if (OK != collect()) {
619  DEVICE_DEBUG("collection error");
620  /* if error restart the measurement state machine */
621  start();
622  return;
623  }
624 
625  /* next phase is measurement */
626  _collect_phase = false;
627 
628  /* change i2c adress to next sonar */
629  _cycle_counter = _cycle_counter + 1;
630 
631  if (_cycle_counter >= addr_ind.size()) {
632  _cycle_counter = 0;
633  }
634 
635  /* Is there a collect->measure gap? Yes, and the timing is set equal to the cycling_rate
636  Otherwise the next sonar would fire without the first one having received its reflected sonar pulse */
637 
638  if (_measure_ticks > USEC2TICK(_cycling_rate)) {
639 
640  /* schedule a fresh cycle call when we are ready to measure again */
641  work_queue(HPWORK,
642  &_work,
643  (worker_t)&SRF02::cycle_trampoline,
644  this,
645  _measure_ticks - USEC2TICK(_cycling_rate));
646  return;
647  }
648  }
649 
650  /* Measurement (firing) phase */
651 
652  /* ensure sonar i2c adress is still correct */
653  _index_counter = addr_ind[_cycle_counter];
654  set_device_address(_index_counter);
655 
656  /* Perform measurement */
657  if (OK != measure()) {
658  DEVICE_DEBUG("measure error sonar adress %d", _index_counter);
659  }
660 
661  /* next phase is collection */
662  _collect_phase = true;
663 
664  /* schedule a fresh cycle call when the measurement is done */
665  work_queue(HPWORK,
666  &_work,
667  (worker_t)&SRF02::cycle_trampoline,
668  this,
669  USEC2TICK(_cycling_rate));
670 
671 }

731 {
732  int fd = -1;
733 
734  if (g_dev != nullptr) {
735  PX4_ERR("already started");
736  return PX4_ERROR;
737  }
738 
739  /* create the driver */
740  g_dev = new SRF02(rotation, i2c_bus);
741 
742  if (g_dev == nullptr) {
743  goto fail;
744  }
745 
746  if (OK != g_dev->init()) {
747  goto fail;
748  }
749 
750  /* set the poll rate to default, starts automatic data collection */
751  fd = px4_open(SRF02_DEVICE_PATH, O_RDONLY);
752 
753  if (fd < 0) {

756

758  goto fail;
759  }
760

767  px4_close(fd);
768  }

796 /**
797  * Perform some basic functional tests on the driver;
798  * make sure we can collect data from the sensor in polled
799  * and automatic modes.
800  */
801 int
802 test()
803 {
804  struct distance_sensor_s report;
805  ssize_t sz;
806  int ret;
807 
808  int fd = px4_open(SRF02_DEVICE_PATH, O_RDONLY);
809 
810  if (fd < 0) {
811  PX4_ERR("%s open failed (try 'srf02 start' if the driver is not running)", SRF02_DEVICE_PATH);
812  return PX4_ERROR;
813  }
814 
815  /* do a simple demand read */
816  sz = read(fd, &report, sizeof(report));
817 
818  if (sz != sizeof(report)) {
819  PX4_ERR("immediate read failed");
820  return PX4_ERROR;
821  }
822 
823  print_message(report);
824 
825  /* start the sensor polling at 2Hz */
826  if (OK != ioctl(fd, SENSORIOCSPOLLRATE, 2)) {
827  PX4_ERR("failed to set 2Hz poll rate");
828  return PX4_ERROR;
829  }
830 
831  /* read the sensor 5x and report each value */
832  for (unsigned i = 0; i < 5; i++) {
833  struct pollfd fds;
834 
835  /* wait for data to be ready */
836  fds.fd = fd;
837  fds.events = POLLIN;
838  ret = poll(&fds, 1, 2000);
839 
840  if (ret != 1) {
841  PX4_ERR("timed out waiting for sensor data");
842  return PX4_ERROR;
843  }
844 
845  /* now go get it */
846  sz = read(fd, &report, sizeof(report));
847 
848  if (sz != sizeof(report)) {
849  PX4_ERR("periodic read failed");
850  return PX4_ERROR;
851  }
852 
853  print_message(report);
854  }
855 
856  /* reset the sensor polling to default rate */
857  if (OK != ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT)) {
858  PX4_ERR("failed to set default poll rate");
859  return PX4_ERROR;
860  }
861 
862  PX4_INFO("PASS");
863  return PX4_OK;
864 }
