458 int MPU9250::reset_mpu()
459 {
460  write_reg(MPUREG_PWR_MGMT_1, BIT_H_RESET);
461  write_checked_reg(MPUREG_PWR_MGMT_1, MPU_CLK_SEL_AUTO);
462  write_checked_reg(MPUREG_PWR_MGMT_2, 0);
463 
464 
465  usleep(1000);
466 
467  // Enable I2C bus or Disable I2C bus (recommended on data sheet)
468 
469  write_checked_reg(MPUREG_USER_CTRL, is_i2c() ? 0 : BIT_I2C_IF_DIS);
470 
471  // SAMPLE RATE
472  _set_sample_rate(_sample_rate);
473 
474  _set_dlpf_filter(MPU9250_DEFAULT_ONCHIP_FILTER_FREQ);
475 
476  // Gyro scale 2000 deg/s ()
477  write_checked_reg(MPUREG_GYRO_CONFIG, BITS_FS_2000DPS);
478 
479  // correct gyro scale factors
480  // scale to rad/s in SI units
481  // 2000 deg/s = (2000/180)*PI = 34.906585 rad/s
482  // scaling factor:
483  // 1/(2^15)*(2000/180)*PI
484  _gyro_range_scale = (0.0174532 / 16.4);//1.0f / (32768.0f * (2000.0f / 180.0f) * M_PI_F);
485  _gyro_range_rad_s = (2000.0f / 180.0f) * M_PI_F;
486 
487  set_accel_range(ACCEL_RANGE_G);
488 
489  // INT CFG => Interrupt on Data Ready
490  write_checked_reg(MPUREG_INT_ENABLE, BIT_RAW_RDY_EN); // INT: Raw data ready
491 
492 #ifdef USE_I2C
493  bool bypass = !_mag->is_passthrough();
494 #else
495  bool bypass = false;
496 #endif
497 
498  /* INT: Clear on any read.
499  * If this instance is for a device is on I2C bus the Mag will have an i2c interface
500  * that it will use to access the either: a) the internal mag device on the internal I2C bus
501  * or b) it could be used to access a downstream I2C devices connected to the chip on
502  * it's AUX_{ASD|SCL} pins. In either case we need to disconnect (bypass) the internal master
503  * controller that chip provides as a SPI to I2C bridge.
504  * so bypass is true if the mag has an i2c non null interfaces.
505  */
506 
507  write_checked_reg(MPUREG_INT_PIN_CFG, BIT_INT_ANYRD_2CLEAR | (bypass ? BIT_INT_BYPASS_EN : 0));
508 
509  write_checked_reg(MPUREG_ACCEL_CONFIG2, BITS_ACCEL_CONFIG2_41HZ);
510 
511  uint8_t retries = 3;
512  bool all_ok = false;
513 
514  while (!all_ok && retries--) {
515 
516  // Assume all checked values are as expected
517  all_ok = true;
518  uint8_t reg;
519 
520  for (uint8_t i = 0; i < MPU9250_NUM_CHECKED_REGISTERS; i++) {
521  if ((reg = read_reg(_checked_registers[i])) != _checked_values[i]) {
522  write_reg(_checked_registers[i], _checked_values[i]);
523  PX4_ERR("Reg %d is:%d s/b:%d Tries:%d", _checked_registers[i], reg, _checked_values[i], retries);
524  all_ok = false;
525  }
526  }
527  }
528 
529  return all_ok ? OK : -EIO;
530 }

578 void
579 MPU9250::_set_dlpf_filter(uint16_t frequency_hz)
580 {
581  uint8_t filter;
582 
583  /*
584  choose next highest filter frequency available
585  */
586  if (frequency_hz == 0) {
587  _dlpf_freq = 0;
588  filter = BITS_DLPF_CFG_3600HZ;
589 
590  } else if (frequency_hz <= 5) {
591  _dlpf_freq = 5;
592  filter = BITS_DLPF_CFG_5HZ;
593 
594  } else if (frequency_hz <= 10) {
595  _dlpf_freq = 10;
596  filter = BITS_DLPF_CFG_10HZ;
597 
598  } else if (frequency_hz <= 20) {
599  _dlpf_freq = 20;
600  filter = BITS_DLPF_CFG_20HZ;
601 
602  } else if (frequency_hz <= 41) {
603  _dlpf_freq = 41;
604  filter = BITS_DLPF_CFG_41HZ;
605 
606  } else if (frequency_hz <= 92) {
607  _dlpf_freq = 92;

1120 void
1121 MPU9250::check_registers(void)
1122 {
1123  /*
1124  we read the register at full speed, even though it isn't
1125  listed as a high speed register. The low speed requirement
1126  for some registers seems to be a propagation delay
1127  requirement for changing sensor configuration, which should
1128  not apply to reading a single register. It is also a better
1129  test of SPI bus health to read at the same speed as we read
1130  the data registers.
1131  */
1132  uint8_t v;
1133 
1134  if ((v = read_reg(_checked_registers[_checked_next], MPU9250_HIGH_BUS_SPEED)) !=
1135  _checked_values[_checked_next]) {
1136  _checked_bad[_checked_next] = v;
1137 
1138  /*
1139  if we get the wrong value then we know the SPI bus
1140  or sensor is very sick. We set _register_wait to 20
1141  and wait until we have seen 20 good values in a row
1142  before we consider the sensor to be OK again.
1143  */
1144  perf_count(_bad_registers);
1145 
1146  /*
1147  try to fix the bad register value. We only try to
1148  fix one per loop to prevent a bad sensor hogging the
1149  bus.
1150  */
1151  if (_register_wait == 0 || _checked_next == 0) {
1152  // if the product_id is wrong then reset the
1153  // sensor completely
1154  write_reg(MPUREG_PWR_MGMT_1, BIT_H_RESET);
1155  write_reg(MPUREG_PWR_MGMT_2, MPU_CLK_SEL_AUTO);
1156  // after doing a reset we need to wait a long
1157  // time before we do any other register writes
1158  // or we will end up with the mpu9250 in a
1159  // bizarre state where it has all correct
1160  // register values but large offsets on the
1161  // accel axes
1162  _reset_wait = hrt_absolute_time() + 10000;
1163  _checked_next = 0;
1164 
1165  } else {
1166  write_reg(_checked_registers[_checked_next], _checked_values[_checked_next]);
1167  // waiting 3ms between register writes seems
1168  // to raise the chance of the sensor
1169  // recovering considerably
1170  _reset_wait = hrt_absolute_time() + 3000;
1171  }
1172 
1173  _register_wait = 20;
1174  }
1175 
1176  _checked_next = (_checked_next + 1) % MPU9250_NUM_CHECKED_REGISTERS;
1177 }

1211  perf_count(_duplicates);

1222 void
1223 MPU9250::measure()
1224 {
1225  if (hrt_absolute_time() < _reset_wait) {
1226  // we're waiting for a reset to complete
1227  return;
1228  }
1229 
1230  struct MPUReport mpu_report;
1231 
1232  struct Report {
1233  int16_t accel_x;
1234  int16_t accel_y;
1235  int16_t accel_z;
1236  int16_t temp;
1237  int16_t gyro_x;
1238  int16_t gyro_y;
1239  int16_t gyro_z;
1240  } report;
1241 
1242  /* start measuring */
1243  perf_begin(_sample_perf);
1244 
1245  /*
1246  * Fetch the full set of measurements from the MPU9250 in one pass.
1247  */
1248  if (OK != _interface->read(MPU9250_SET_SPEED(MPUREG_INT_STATUS, MPU9250_HIGH_BUS_SPEED),
1249  (uint8_t *)&mpu_report,
1250  sizeof(mpu_report))) {
1251  perf_end(_sample_perf);
1252  return;
1253  }
1254 
1255  check_registers();
1256 
1257  if (check_duplicate(&mpu_report.accel_x[0])) {
1258  return;
1259  }
1260 
1261 #ifdef USE_I2C
1262 
1263  if (_mag->is_passthrough()) {
1264 #endif
1265  _mag->_measure(mpu_report.mag);
1266 #ifdef USE_I2C
1267 
1268  } else {
1269  _mag->measure();
1270  }
1271 
1272 #endif
1273 
1274  /*
1275  * Convert from big to little endian
1276  */
1277  report.accel_x = int16_t_from_bytes(mpu_report.accel_x);
1278  report.accel_y = int16_t_from_bytes(mpu_report.accel_y);
1279  report.accel_z = int16_t_from_bytes(mpu_report.accel_z);
1280  report.temp = int16_t_from_bytes(mpu_report.temp);
1281  report.gyro_x = int16_t_from_bytes(mpu_report.gyro_x);
1282  report.gyro_y = int16_t_from_bytes(mpu_report.gyro_y);
1283  report.gyro_z = int16_t_from_bytes(mpu_report.gyro_z);
1284 
1285  if (check_null_data((uint32_t *)&report, sizeof(report) / 4)) {
1286  return;
1287  }
1288 
1289  if (_register_wait != 0) {
1290  // we are waiting for some good transfers before using the sensor again
1291  // We still increment _good_transfers, but don't return any data yet
1292  _register_wait--;
1293  return;
1294  }
1295 
1296  /*
1297  * Swap axes and negate y
1298  */
1299  int16_t accel_xt = report.accel_y;
1300  int16_t accel_yt = ((report.accel_x == -32768) ? 32767 : -report.accel_x);
1301 
1302  int16_t gyro_xt = report.gyro_y;
1303  int16_t gyro_yt = ((report.gyro_x == -32768) ? 32767 : -report.gyro_x);
1304 
1305  /*
1306  * Apply the swap
1307  */
1308  report.accel_x = accel_xt;
1309  report.accel_y = accel_yt;
1310  report.gyro_x = gyro_xt;
1311  report.gyro_y = gyro_yt;
1312 
1313  /*
1314  * Report buffers.
1315  */
1316  accel_report arb;
1317  gyro_report grb;
1318 
1319  /*
1320  * Adjust and scale results to m/s^2.
1321  */
1322  grb.timestamp = arb.timestamp = hrt_absolute_time();
1323
