227 RCUpdate::set_params_from_rc(const ParameterHandles &parameter_handles)
228 {
229  for (int i = 0; i < rc_parameter_map_s::RC_PARAM_MAP_NCHAN; i++) {
230  if (_rc.function[rc_channels_s::RC_CHANNELS_FUNCTION_PARAM_1 + i] < 0 || !_rc_parameter_map.valid[i]) {
231  /* This RC channel is not mapped to a RC-Parameter Channel (e.g. RC_MAP_PARAM1 == 0)
232  * or no request to map this channel to a param has been sent via mavlink
233  */
234  continue;
235  }
236 
237  float rc_val = get_rc_value((rc_channels_s::RC_CHANNELS_FUNCTION_PARAM_1 + i), -1.0, 1.0);
238 
239  /* Check if the value has changed,
240  * maybe we need to introduce a more aggressive limit here */
241  if (rc_val > _param_rc_values[i] + FLT_EPSILON || rc_val < _param_rc_values[i] - FLT_EPSILON) {
242  _param_rc_values[i] = rc_val;
243  float param_val = math::constrain(
244  _rc_parameter_map.value0[i] + _rc_parameter_map.scale[i] * rc_val,
245  _rc_parameter_map.value_min[i], _rc_parameter_map.value_max[i]);
246  param_set(parameter_handles.rc_param[i], &param_val);

251 void
252 RCUpdate::rc_poll(const ParameterHandles &parameter_handles)
253 {
254  bool rc_updated;
255  orb_check(_rc_sub, &rc_updated);
256 
257  if (rc_updated) {
258  /* read low-level values from FMU or IO RC inputs (PPM, Spektrum, S.Bus) */
259  struct rc_input_values rc_input;
260 
261  orb_copy(ORB_ID(input_rc), _rc_sub, &rc_input);
262 
263  /* detect RC signal loss */
264  bool signal_lost;
265 
266  /* check flags and require at least four channels to consider the signal valid */
267  if (rc_input.rc_lost || rc_input.rc_failsafe || rc_input.channel_count < 4) {
268  /* signal is lost or no enough channels */
269  signal_lost = true;
270 
271  } else {
272  /* signal looks good */
273  signal_lost = false;
274 
275  /* check failsafe */
276  int8_t fs_ch = _rc.function[_parameters.rc_map_failsafe]; // get channel mapped to throttle
277 
278  if (_parameters.rc_map_failsafe > 0) { // if not 0, use channel number instead of rc.function mapping
279  fs_ch = _parameters.rc_map_failsafe - 1;
280  }
281 
282  if (_parameters.rc_fails_thr > 0 && fs_ch >= 0) {
283  /* failsafe configured */
284  if ((_parameters.rc_fails_thr < _parameters.min[fs_ch] && rc_input.values[fs_ch] < _parameters.rc_fails_thr) ||
285  (_parameters.rc_fails_thr > _parameters.max[fs_ch] && rc_input.values[fs_ch] > _parameters.rc_fails_thr)) {
286  /* failsafe triggered, signal is lost by receiver */
287  signal_lost = true;
288  }
289  }
290  }
291 
292  unsigned channel_limit = rc_input.channel_count;
293 
294  if (channel_limit > RC_MAX_CHAN_COUNT) {
295  channel_limit = RC_MAX_CHAN_COUNT;
296  }
297 
298  /* read out and scale values from raw message even if signal is invalid */
299  for (unsigned int i = 0; i < channel_limit; i++) {
300 
301  /*
302  * 1) Constrain to min/max values, as later processing depends on bounds.
303  */
304  if (rc_input.values[i] < _parameters.min[i]) {
305  rc_input.values[i] = _parameters.min[i];
306  }
307 
308  if (rc_input.values[i] > _parameters.max[i]) {
309  rc_input.values[i] = _parameters.max[i];
310  }
311 
312  /*
313  * 2) Scale around the mid point differently for lower and upper range.
314  *
315  * This is necessary as they don't share the same endpoints and slope.
316  *
317  * First normalize to 0..1 range with correct sign (below or above center),
318  * the total range is 2 (-1..1).
319  * If center (trim) == min, scale to 0..1, if center (trim) == max,
320  * scale to -1..0.
321  *
322  * As the min and max bounds were enforced in step 1), division by zero
323  * cannot occur, as for the case of center == min or center == max the if
324  * statement is mutually exclusive with the arithmetic NaN case.
325  *
326  * DO NOT REMOVE OR ALTER STEP 1!
327  */
328  if (rc_input.values[i] > (_parameters.trim[i] + _parameters.dz[i])) {
329  _rc.channels[i] = (rc_input.values[i] - _parameters.trim[i] - _parameters.dz[i]) / (float)(
330  _parameters.max[i] - _parameters.trim[i] - _parameters.dz[i]);
331 
332  } else if (rc_input.values[i] < (_parameters.trim[i] - _parameters.dz[i])) {
333  _rc.channels[i] = (rc_input.values[i] - _parameters.trim[i] + _parameters.dz[i]) / (float)(
334  _parameters.trim[i] - _parameters.min[i] - _parameters.dz[i]);
335 
336  } else {
337  /* in the configured dead zone, output zero */
338  _rc.channels[i] = 0.0f;
339  }
340 
341  _rc.channels[i] *= _parameters.rev[i];
342 
343  /* handle any parameter-induced blowups */
344  if (!PX4_ISFINITE(_rc.channels[i])) {
345  _rc.channels[i] = 0.0f;
346  }
347  }
348 
349  _rc.channel_count = rc_input.channel_count;
350  _rc.rssi = rc_input.rssi;
351  _rc.signal_lost = signal_lost;
352  _rc.timestamp = rc_input.timestamp_last_signal;
353  _rc.frame_drop_count = rc_input.rc_lost_frame_count;
354 
355  /* publish rc_channels topic even if signal is invalid, for debug */
356  int instance;
357  orb_publish_auto(ORB_ID(rc_channels), &_rc_pub, &_rc, &instance, ORB_PRIO_DEFAULT);
358 
359  /* only publish manual control if the signal is still present and was present once */
360  if (!signal_lost && rc_input.timestamp_last_signal > 0) {
361 
362  /* initialize manual setpoint */
363  struct manual_control_setpoint_s manual = {};
364  /* set mode slot to unassigned */
365  manual.mode_slot = manual_control_setpoint_s::MODE_SLOT_NONE;
366  /* set the timestamp to the last signal time */
367  manual.timestamp = rc_input.timestamp_last_signal;
368  manual.data_source = manual_control_setpoint_s::SOURCE_RC;
369 
370  /* limit controls */
371  manual.y = get_rc_value(rc_channels_s::RC_CHANNELS_FUNCTION_ROLL, -1.0, 1.0);
372  manual.x = get_rc_value(rc_channels_s::RC_CHANNELS_FUNCTION_PITCH, -1.0, 1.0);
373  manual.r = get_rc_value(rc_channels_s::RC_CHANNELS_FUNCTION_YAW, -1.0, 1.0);
374  manual.z = get_rc_value(rc_channels_s::RC_CHANNELS_FUNCTION_THROTTLE, 0.0, 1.0);
375  manual.flaps = get_rc_value(rc_channels_s::RC_CHANNELS_FUNCTION_FLAPS, -1.0, 1.0);
376  manual.aux1 = get_rc_value(rc_channels_s::RC_CHANNELS_FUNCTION_AUX_1, -1.0, 1.0);
377  manual.aux2 = get_rc_value(rc_channels_s::RC_CHANNELS_FUNCTION_AUX_2, -1.0, 1.0);
378  manual.aux3 = get_rc_value(rc_channels_s::RC_CHANNELS_FUNCTION_AUX_3, -1.0, 1.0);
379  manual.aux4 = get_rc_value(rc_channels_s::RC_CHANNELS_FUNCTION_AUX_4, -1.0, 1.0);
380  manual.aux5 = get_rc_value(rc_channels_s::RC_CHANNELS_FUNCTION_AUX_5, -1.0, 1.0);
381 
382  /* filter controls */
383  manual.y = math::constrain(_filter_roll.apply(manual.y), -1.f, 1.f);
384  manual.x = math::constrain(_filter_pitch.apply(manual.x), -1.f, 1.f);
385  manual.r = math::constrain(_filter_yaw.apply(manual.r), -1.f, 1.f);
386  manual.z = math::constrain(_filter_throttle.apply(manual.z), 0.f, 1.f);
387 
388  if (_parameters.rc_map_flightmode > 0) {
389 
390  /* the number of valid slots equals the index of the max marker minus one */
391  const int num_slots = manual_control_setpoint_s::MODE_SLOT_MAX;
392 
393  /* the half width of the range of a slot is the total range
394  * divided by the number of slots, again divided by two
395  */
396  const float slot_width_half = 2.0f / num_slots / 2.0f;
397 
398  /* min is -1, max is +1, range is 2. We offset below min and max */
399  const float slot_min = -1.0f - 0.05f;
400  const float slot_max = 1.0f + 0.05f;
401 
402  /* the slot gets mapped by first normalizing into a 0..1 interval using min
403  * and max. Then the right slot is obtained by multiplying with the number of
404  * slots. And finally we add half a slot width to ensure that integer rounding
405  * will take us to the correct final index.
406  */
407  manual.mode_slot = (((((_rc.channels[_parameters.rc_map_flightmode - 1] - slot_min) * num_slots) + slot_width_half) /
408  (slot_max - slot_min)) + (1.0f / num_slots));
409 
410  if (manual.mode_slot >= num_slots) {
411  manual.mode_slot = num_slots - 1;
412  }
413  }
414 
415  /* mode switches */
416  manual.mode_switch = get_rc_sw3pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_MODE, _parameters.rc_auto_th,
417  _parameters.rc_auto_inv, _parameters.rc_assist_th, _parameters.rc_assist_inv);
418  manual.rattitude_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_RATTITUDE,
419  _parameters.rc_rattitude_th,
420  _parameters.rc_rattitude_inv);
421  manual.posctl_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_POSCTL, _parameters.rc_posctl_th,
422  _parameters.rc_posctl_inv);
423  manual.return_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_RETURN, _parameters.rc_return_th,
424  _parameters.rc_return_inv);
425  manual.loiter_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_LOITER, _parameters.rc_loiter_th,
426  _parameters.rc_loiter_inv);
427  manual.acro_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_ACRO, _parameters.rc_acro_th,
428  _parameters.rc_acro_inv);
429  manual.offboard_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_OFFBOARD,
430  _parameters.rc_offboard_th, _parameters.rc_offboard_inv);
431  manual.kill_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_KILLSWITCH,
432  _parameters.rc_killswitch_th, _parameters.rc_killswitch_inv);
433  manual.arm_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_ARMSWITCH,
434  _parameters.rc_armswitch_th, _parameters.rc_armswitch_inv);
435  manual.transition_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_TRANSITION,
436  _parameters.rc_trans_th, _parameters.rc_trans_inv);
437  manual.gear_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_GEAR,
438  _parameters.rc_gear_th, _parameters.rc_gear_inv);
439  manual.stab_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_STAB,
440  _parameters.rc_stab_th, _parameters.rc_stab_inv);
441  manual.man_switch = get_rc_sw2pos_position(rc_channels_s::RC_CHANNELS_FUNCTION_MAN,
442  _parameters.rc_man_th, _parameters.rc_man_inv);
443 
444  /* publish manual_control_setpoint topic */
445  orb_publish_auto(ORB_ID(manual_control_setpoint), &_manual_control_pub, &manual, &instance,
446  ORB_PRIO_HIGH);
447 
448  /* copy from mapped manual control to control group 3 */
449  struct actuator_controls_s actuator_group_3 = {};
450 
451  actuator_group_3.timestamp = rc_input.timestamp_last_signal;
452 
453  actuator_group_3.control[0] = manual.y;
454  actuator_group_3.control[1] = manual.x;
455  actuator_group_3.control[2] = manual.r;
456  actuator_group_3.control[3] = manual.z;
457  actuator_group_3.control[4] = manual.flaps;
458  actuator_group_3.control[5] = manual.aux1;
459  actuator_group_3.control[6] = manual.aux2;
460  actuator_group_3.control[7] = manual.aux3;
461 
462  /* publish actuator_controls_3 topic */
463  orb_publish_auto(ORB_ID(actuator_controls_3), &_actuator_group_3_pub, &actuator_group_3, &instance,
464  ORB_PRIO_DEFAULT);
465 
466  /* Update parameters from RC Channels (tuning with RC) if activated */
467  if (hrt_elapsed_time(&_last_rc_to_param_map_time) > 1e6) {
468  set_params_from_rc(parameter_handles);
469  _last_rc_to_param_map_time = hrt_absolute_time();
470  }
471  }
472  }
