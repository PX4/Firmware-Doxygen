122 __EXPORT px4_hw_mft_item board_query_manifest(px4_hw_mft_item_id_t id)
123 {
124  static px4_hw_mft_list_entry boards_manifest = px4_hw_mft_list_uninitialized;
125 
126  if (boards_manifest == px4_hw_mft_list_uninitialized) {
127  uint32_t ver_rev = board_get_hw_version();
128 
129  for (unsigned i = 0; i < arraySize(mft_lists); i++) {
130  if (mft_lists[i].hw_ver_rev == ver_rev) {
131  boards_manifest = &mft_lists[i];
132  break;
133  }
134  }
135 
136  if (boards_manifest == px4_hw_mft_list_uninitialized) {
137  PX4_ERR("Board %4x is not supported!", ver_rev);
138  }
139  }
140 
141  px4_hw_mft_item rv = &device_unsupported;
142 
143  if (boards_manifest != px4_hw_mft_list_uninitialized &&
144  id < boards_manifest->entries) {
145  rv = &boards_manifest->mft[id];
146  }
147 
148  return rv;
149 }
