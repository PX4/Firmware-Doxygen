58 void
59 Landingslope::calculateSlopeValues()
60 {
61  _H0 = _flare_relative_alt + _H1_virt;
62  _d1 = _flare_relative_alt / tanf(_landing_slope_angle_rad);
63  _flare_constant = (_H0 * _d1) / _flare_relative_alt;
64  _flare_length = -logf(_H1_virt / _H0) * _flare_constant;
65  _horizontal_slope_displacement = (_flare_length - _d1);
66 }
