105 int
106 uORB::DeviceNode::open(device::file_t *filp)
107 {
108  int ret;
109 
110  /* is this a publisher? */
111  if (FILE_FLAGS(filp) == PX4_F_WRONLY) {
112 
113  /* become the publisher if we can */
114  lock();
115 
116  if (_publisher == 0) {
117  _publisher = px4_getpid();
118  ret = PX4_OK;
119 
120  } else {
121  ret = -EBUSY;
122  }
123 
124  unlock();
125 
126  /* now complete the open */
127  if (ret == PX4_OK) {
128  ret = CDev::open(filp);
129 
130  /* open failed - not the publisher anymore */
131  if (ret != PX4_OK) {
132  _publisher = 0;
133  }
134  }
135 
136  return ret;
137  }
138 
139  /* is this a new subscriber? */
140  if (FILE_FLAGS(filp) == PX4_F_RDONLY) {
141 
142  /* allocate subscriber data */
143  SubscriberData *sd = new SubscriberData{};
144 
145  if (nullptr == sd) {
146  return -ENOMEM;
147  }
148 
149  /* If there were any previous publications, allow the subscriber to read them */
150  sd->generation = _generation - (_queue_size < _generation ? _queue_size : _generation);
151 
152  /* set priority */
153  sd->set_priority(_priority);
154 
155  FILE_PRIV(filp) = (void *)sd;
156 
157  ret = CDev::open(filp);
158 
159  add_internal_subscriber();
160 
161  if (ret != PX4_OK) {
162  PX4_ERR("CDev::open failed");
163  delete sd;
164  }
165 
166  return ret;
167  }
168 
169  if (FILE_FLAGS(filp) == 0) {
170  return CDev::open(filp);
171  }
172 
173  /* can only be pub or sub, not both */
174  return -EINVAL;
175 }

202 ssize_t
203 uORB::DeviceNode::read(device::file_t *filp, char *buffer, size_t buflen)
204 {
205  SubscriberData *sd = (SubscriberData *)filp_to_sd(filp);
206 
207  /* if the object has not been written yet, return zero */
208  if (_data == nullptr) {
209  return 0;
210  }
211 
212  /* if the caller's buffer is the wrong size, that's an error */
213  if (buflen != _meta->o_size) {
214  return -EIO;
215  }
216 
217  /*
218  * Perform an atomic copy & state update
219  */
220  ATOMIC_ENTER;
221 
222  if (_generation > sd->generation + _queue_size) {
223  /* Reader is too far behind: some messages are lost */
224  _lost_messages += _generation - (sd->generation + _queue_size);
225  sd->generation = _generation - _queue_size;
226  }
227 
228  if (_generation == sd->generation && sd->generation > 0) {
229  /* The subscriber already read the latest message, but nothing new was published yet.
230  * Return the previous message
231  */
232  --sd->generation;
233  }
234 
235  /* if the caller doesn't want the data, don't give it to them */
236  if (nullptr != buffer) {
237  memcpy(buffer, _data + (_meta->o_size * (sd->generation % _queue_size)), _meta->o_size);
238  }
239 
240  if (sd->generation < _generation) {
241  ++sd->generation;
242  }
243 
244  /* set priority */
245  sd->set_priority(_priority);
246 
247  /*
248  * Clear the flag that indicates that an update has been reported, as
249  * we have just collected it.
250  */
251  sd->set_update_reported(false);
252 
253  ATOMIC_LEAVE;
254 
255  return _meta->o_size;
256 }

258 ssize_t
259 uORB::DeviceNode::write(device::file_t *filp, const char *buffer, size_t buflen)
260 {
261  /*
262  * Writes are legal from interrupt context as long as the
263  * object has already been initialised from thread context.
264  *
265  * Writes outside interrupt context will allocate the object
266  * if it has not yet been allocated.
267  *
268  * Note that filp will usually be NULL.
269  */
270  if (nullptr == _data) {
271 #ifdef __PX4_NUTTX
272 
273  if (!up_interrupt_context()) {
274 
275  lock();
276 
277  /* re-check size */
278  if (nullptr == _data) {
279  _data = new uint8_t[_meta->o_size * _queue_size];
280  }
281 
282  unlock();
283  }
284 
285 #else
286  lock();
287 
288  /* re-check size */
289  if (nullptr == _data) {
290  _data = new uint8_t[_meta->o_size * _queue_size];
291  }
292 
293  unlock();
294 #endif
295 
296  /* failed or could not allocate */
297  if (nullptr == _data) {
298  return -ENOMEM;
299  }
300  }
301 
302  /* If write size does not match, that is an error */
303  if (_meta->o_size != buflen) {
304  return -EIO;
305  }
306 
307  /* Perform an atomic copy. */
308  ATOMIC_ENTER;
309  memcpy(_data + (_meta->o_size * (_generation % _queue_size)), buffer, _meta->o_size);
310 
311  /* update the timestamp and generation count */
312  _last_update = hrt_absolute_time();
313  /* wrap-around happens after ~49 days, assuming a publisher rate of 1 kHz */
314  _generation++;
315 
316  _published = true;
317 
318  ATOMIC_LEAVE;
319 
320  /* notify any poll waiters */
321  poll_notify(POLLIN);
322 
323  return _meta->o_size;
324 }

326 int
327 uORB::DeviceNode::ioctl(device::file_t *filp, int cmd, unsigned long arg)
328 {
329  SubscriberData *sd = filp_to_sd(filp);
330 
331  switch (cmd) {
332  case ORBIOCLASTUPDATE: {
333  ATOMIC_ENTER;
334  *(hrt_abstime *)arg = _last_update;
335  ATOMIC_LEAVE;
336  return PX4_OK;
337  }
338 
339  case ORBIOCUPDATED:
340 #ifndef __PX4_NUTTX
341  lock();
342 #endif
343  *(bool *)arg = appears_updated(sd);
344 #ifndef __PX4_NUTTX
345  unlock();
346 #endif
347  return PX4_OK;
348 
349  case ORBIOCSETINTERVAL: {
350  int ret = PX4_OK;
351  lock();
352 
353  if (arg == 0) {
354  if (sd->update_interval) {
355  delete (sd->update_interval);
356  sd->update_interval = nullptr;
357  }
358 
359  } else {
360  if (sd->update_interval) {
361  sd->update_interval->interval = arg;
362 #ifndef __PX4_NUTTX
363  sd->update_interval->last_update = hrt_absolute_time();
364 #endif
365 
366  } else {
367  sd->update_interval = new UpdateIntervalData();
368 
369  if (sd->update_interval) {
370  memset(&sd->update_interval->update_call, 0, sizeof(hrt_call));
371  sd->update_interval->interval = arg;
372 #ifndef __PX4_NUTTX
373  sd->update_interval->last_update = hrt_absolute_time();
374 #endif
375 
376  } else {
377  ret = -ENOMEM;
378  }
379  }
380  }
381 
382  unlock();
383  return ret;
384  }
385 
386  case ORBIOCGADVERTISER:
387  *(uintptr_t *)arg = (uintptr_t)this;
388  return PX4_OK;
389 
390  case ORBIOCGPRIORITY:
391  *(int *)arg = sd->priority();
392  return PX4_OK;
393 
394  case ORBIOCSETQUEUESIZE:
395  //no need for locking here, since this is used only during the advertisement call,
396  //and only one advertiser is allowed to open the DeviceNode at the same time.
397  return update_queue_size(arg);
398 
399  case ORBIOCGETINTERVAL:
400  if (sd->update_interval) {
401  *(unsigned *)arg = sd->update_interval->interval;
402 
403  } else {
404  *(unsigned *)arg = 0;
405  }
406 
407  return OK;
408 
409  case ORBIOCISPUBLISHED:
410  *(unsigned long *)arg = _published;
411 
412  return OK;
413 
414  default:
415  /* give it to the superclass */
416  return CDev::ioctl(filp, cmd, arg);
417  }
418 }

420 ssize_t
421 uORB::DeviceNode::publish(const orb_metadata *meta, orb_advert_t handle, const void *data)
422 {
423  uORB::DeviceNode *devnode = (uORB::DeviceNode *)handle;
424  int ret;
425 
426  /* check if the device handle is initialized */
427  if ((devnode == nullptr) || (meta == nullptr)) {
428  errno = EFAULT;
429  return PX4_ERROR;
430  }
431 
432  /* check if the orb meta data matches the publication */
433  if (devnode->_meta != meta) {
434  errno = EINVAL;
435  return PX4_ERROR;
436  }
437 
438  /* call the devnode write method with no file pointer */
439  ret = devnode->write(nullptr, (const char *)data, meta->o_size);
440 
441  if (ret < 0) {
442  errno = -ret;
443  return PX4_ERROR;
444  }
445 
446  if (ret != (int)meta->o_size) {
447  errno = EIO;
448  return PX4_ERROR;
449  }
450 
451 #ifdef ORB_COMMUNICATOR
452  /*
453  * if the write is successful, send the data over the Multi-ORB link
454  */
455  uORBCommunicator::IChannel *ch = uORB::Manager::get_instance()->get_uorb_communicator();
456 
457  if (ch != nullptr) {
458  if (ch->send_message(meta->o_name, meta->o_size, (uint8_t *)data) != 0) {
459  PX4_ERR("Error Sending [%s] topic data over comm_channel", meta->o_name);
460  return PX4_ERROR;
461  }
462  }
463 
464 #endif /* ORB_COMMUNICATOR */
465 
466  return PX4_OK;
467 }

469 int uORB::DeviceNode::unadvertise(orb_advert_t handle)
470 {
471  if (handle == nullptr) {
472  return -EINVAL;
473  }
474 
475  uORB::DeviceNode *devnode = (uORB::DeviceNode *)handle;
476 
477  /*
478  * We are cheating a bit here. First, with the current implementation, we can only
479  * have multiple publishers for instance 0. In this case the caller will have
480  * instance=nullptr and _published has no effect at all. Thus no unadvertise is
481  * necessary.
482  * In case of multiple instances, we have at most 1 publisher per instance and
483  * we can signal an instance as 'free' by setting _published to false.
484  * We never really free the DeviceNode, for this we would need reference counting
485  * of subscribers and publishers. But we also do not have a leak since future
486  * publishers reuse the same DeviceNode object.
487  */
488  devnode->_published = false;
489 
490  return PX4_OK;
491 }

546 #ifdef __PX4_NUTTX
547 bool
548 uORB::DeviceNode::appears_updated(SubscriberData *sd)
549 {
550  /* assume it doesn't look updated */
551  bool ret = false;
552 
553  /* avoid racing between interrupt and non-interrupt context calls */
554  irqstate_t state = px4_enter_critical_section();
555 
556  /* check if this topic has been published yet, if not bail out */
557  if (_data == nullptr) {
558  ret = false;
559  goto out;
560  }
561 
562  /*
563  * If the subscriber's generation count matches the update generation
564  * count, there has been no update from their perspective; if they
565  * don't match then we might have a visible update.
566  */
567  while (sd->generation != _generation) {
568 
569  /*
570  * Handle non-rate-limited subscribers.
571  */
572  if (sd->update_interval == nullptr) {
573  ret = true;
574  break;
575  }
576 
577  /*
578  * If we have previously told the subscriber that there is data,
579  * and they have not yet collected it, continue to tell them
580  * that there has been an update. This mimics the non-rate-limited
581  * behaviour where checking / polling continues to report an update
582  * until the topic is read.
583  */
584  if (sd->update_reported()) {
585  ret = true;
586  break;
587  }
588 
589  /*
590  * If the interval timer is still running, the topic should not
591  * appear updated, even though at this point we know that it has.
592  * We have previously been through here, so the subscriber
593  * must have collected the update we reported, otherwise
594  * update_reported would still be true.
595  */
596  if (!hrt_called(&sd->update_interval->update_call)) {
597  break;
598  }
599 
600  /*
601  * Make sure that we don't consider the topic to be updated again
602  * until the interval has passed once more by restarting the interval
603  * timer and thereby re-scheduling a poll notification at that time.
604  */
605  hrt_call_after(&sd->update_interval->update_call,
606  sd->update_interval->interval,
607  &uORB::DeviceNode::update_deferred_trampoline,
608  (void *)this);
609 
610  /*
611  * Remember that we have told the subscriber that there is data.
612  */
613  sd->set_update_reported(true);
614  ret = true;
615 
616  break;
617  }
618 
619 out:
620  px4_leave_critical_section(state);
621 
622  /* consider it updated */
623  return ret;
624 }

628 bool
629 uORB::DeviceNode::appears_updated(SubscriberData *sd)
630 {
631 
632  /* block if in simulation mode */
633  while (px4_sim_delay_enabled()) {
634  usleep(100);
635  }
636 
637  /* assume it doesn't look updated */
638  bool ret = false;
639 
640  /* check if this topic has been published yet, if not bail out */
641  if (_data == nullptr) {
642  return false;
643  }
644 
645  /*
646  * If the subscriber's generation count matches the update generation
647  * count, there has been no update from their perspective; if they
648  * don't match then we might have a visible update.
649  */
650  while (sd->generation != _generation) {
651 
652  /*
653  * Handle non-rate-limited subscribers.
654  */
655  if (sd->update_interval == nullptr) {
656  ret = true;
657  break;
658  }
659 
660  /*
661  * If we have previously told the subscriber that there is data,
662  * and they have not yet collected it, continue to tell them
663  * that there has been an update. This mimics the non-rate-limited
664  * behaviour where checking / polling continues to report an update
665  * until the topic is read.
666  */
667  if (sd->update_reported()) {
668  ret = true;
669  break;
670  }
671 
672  // If we have not yet reached the deadline, then assume that we can ignore any
673  // newly received data.
674  if (sd->update_interval->last_update + sd->update_interval->interval > hrt_absolute_time()) {
675  break;
676  }
677 
678  /*
679  * Remember that we have told the subscriber that there is data.
680  */
681  sd->set_update_reported(true);
682  sd->update_interval->last_update = hrt_absolute_time();
683  ret = true;
684 
685  break;
686  }
687 
688  return ret;
689 }

836 int
837 uORB::DeviceMaster::ioctl(device::file_t *filp, int cmd, unsigned long arg)
838 {
839  int ret;
840 
841  switch (cmd) {
842  case ORBIOCADVERTISE: {
843  const struct orb_advertdata *adv = (const struct orb_advertdata *)arg;
844  const struct orb_metadata *meta = adv->meta;
845  char nodepath[orb_maxpath];
846 
847  /* construct a path to the node - this also checks the node name */
848  ret = uORB::Utils::node_mkpath(nodepath, meta, adv->instance);
849 
850  if (ret != PX4_OK) {
851  return ret;
852  }
853 
854  ret = PX4_ERROR;
855 
856  /* try for topic groups */
857  const unsigned max_group_tries = (adv->instance != nullptr) ? ORB_MULTI_MAX_INSTANCES : 1;
858  unsigned group_tries = 0;
859 
860  if (adv->instance) {
861  /* for an advertiser, this will be 0, but a for subscriber that requests a certain instance,
862  * we do not want to start with 0, but with the instance the subscriber actually requests.
863  */
864  group_tries = *adv->instance;
865 
866  if (group_tries >= max_group_tries) {
867  return -ENOMEM;
868  }
869  }
870 
871  SmartLock smart_lock(_lock);
872 
873  do {
874  /* if path is modifyable change try index */
875  if (adv->instance != nullptr) {
876  /* replace the number at the end of the string */
877  nodepath[strlen(nodepath) - 1] = '0' + group_tries;
878  *(adv->instance) = group_tries;
879  }
880 
881  const char *objname = meta->o_name; //no need for a copy, meta->o_name will never be freed or changed
882 
883  /* driver wants a permanent copy of the path, so make one here */
884  const char *devpath = strdup(nodepath);
885 
886  if (devpath == nullptr) {
887  return -ENOMEM;
888  }
889 
890  /* construct the new node */
891  uORB::DeviceNode *node = new uORB::DeviceNode(meta, objname, devpath, adv->priority);
892 
893  /* if we didn't get a device, that's bad */
894  if (node == nullptr) {
895  free((void *)devpath);
896  return -ENOMEM;
897  }
898 
899  /* initialise the node - this may fail if e.g. a node with this name already exists */
900  ret = node->init();
901 
902  /* if init failed, discard the node and its name */
903  if (ret != PX4_OK) {
904  delete node;
905 
906  if (ret == -EEXIST) {
907  /* if the node exists already, get the existing one and check if
908  * something has been published yet. */
909  uORB::DeviceNode *existing_node = getDeviceNodeLocked(devpath);
910 
911  if ((existing_node != nullptr) && !(existing_node->is_published())) {
912  /* nothing has been published yet, lets claim it */
913  existing_node->set_priority(adv->priority);
914  ret = PX4_OK;
915 
916  } else {
917  /* otherwise: data has already been published, keep looking */
918  }
919  }
920 
921  /* also discard the name now */
922  free((void *)devpath);
923 
924  } else {
925  // add to the node map;.
926 #ifdef __PX4_NUTTX
927  _node_map.insert(devpath, node);
928 #else
929  _node_map[std::string(devpath)] = node;
930 #endif
931  }
932 
933  group_tries++;
934 
935  } while (ret != PX4_OK && (group_tries < max_group_tries));
936 
937  if (ret != PX4_OK && group_tries >= max_group_tries) {
938  ret = -ENOMEM;
939  }
940 
941  return ret;
942  }
943 
944  default:
945  /* give it to the superclass */
946  return CDev::ioctl(filp, cmd, arg);
947  }
948 }
