149 int OcpocADC::read(px4_adc_msg_t(*buf)[PX4_MAX_ADC_CHANNELS], unsigned int len)
150 {
151  uint32_t buff[1];
152  int ret = 0;

154  FILE *xadc_fd = fopen(ADC_VOLTAGE_PATH, "r");
155 
156  if (xadc_fd != NULL) {
157  int ret_tmp = fscanf(xadc_fd, "%d", buff);
158 
159  if (ret_tmp < 0) {
160  ret = ret_tmp;
161  }
162 
163  fclose(xadc_fd);
164 
165  (*buf)[0].am_data = buff[0];
166 
167  } else {
168  (*buf)[0].am_data = 0;
169  ret = -1;
170  }
171 
172  (*buf)[0].am_channel = ADC_BATTERY_VOLTAGE_CHANNEL;
173 
174  return ret;

214  }
215

229 
230  if (ret < 0) {
