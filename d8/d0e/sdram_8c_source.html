82 /****************************************************************************
83  * Name: sam_sdram_config
84  *
85  * Description:
86  * Configures the on-board SDRAM. SAME70 Xplained features one external
87  * IS42S16100E-7BLI, 512Kx16x2, 10ns, SDRAM. SDRAM0 is connected to chip
88  * select NCS1.
89  *
90  * Input Parameters:
91  * None
92  *
93  * Assumptions:
94  * This test runs early in initialization before I- and D-caches are
95  * enabled.
96  *
97  * NOTE: Since the delay loop is calibrate with caches in enabled, the
98  * calls to up_udelay() are wrong ty orders of magnitude.
99  *
100  ****************************************************************************/
101 
102 void sam_sdram_config(void)
103 {
104  volatile uint8_t *psdram = (uint8_t *)SAM_SDRAMCS_BASE;
105  uint32_t regval;
106  int i;
107 
108  /* Configure SDRAM pins */
109 
110  sam_configgpio(GPIO_SMC_D0);
111  sam_configgpio(GPIO_SMC_D1);
112  sam_configgpio(GPIO_SMC_D2);
113  sam_configgpio(GPIO_SMC_D3);
114  sam_configgpio(GPIO_SMC_D4);
115  sam_configgpio(GPIO_SMC_D5);
116  sam_configgpio(GPIO_SMC_D6);
117  sam_configgpio(GPIO_SMC_D7);
118  sam_configgpio(GPIO_SMC_D8);
119  sam_configgpio(GPIO_SMC_D9);
120  sam_configgpio(GPIO_SMC_D10);
121  sam_configgpio(GPIO_SMC_D11);
122  sam_configgpio(GPIO_SMC_D12);
123  sam_configgpio(GPIO_SMC_D13);
124  sam_configgpio(GPIO_SMC_D14);
125  sam_configgpio(GPIO_SMC_D15);
126 
127  /* SAME70 SDRAM
128  * --------------- -----------
129  * PC20 A2 A0
130  * PC21 A3 A1
131  * PC22 A4 A2
132  * PC23 A5 A3
133  * PC24 A6 A4
134  * PC25 A7 A5
135  * PC26 A8 A6
136  * PC27 A9 A7
137  * PC28 A10 A8
138  * PC29 A11 A9
139  * PD13 SDA10 A10
140  * PA20 BA0 A11
141  * PD17 CAS nCAS
142  * PD14 SDCKE CKE
143  * PD23 SDCK CLK
144  * PC15 SDCS nCS
145  * PC18 A0/NBS0 LDQM
146  * PD16 RAS nRAS
147  * PD15 NWR1/NBS1 UDQM
148  * PD29 SDWE nWE
149  */
150 
151  sam_configgpio(GPIO_SMC_A2); /* PC20 A2 -> A0 */
152  sam_configgpio(GPIO_SMC_A3); /* PC21 A3 -> A1 */
153  sam_configgpio(GPIO_SMC_A4); /* PC22 A4 -> A2 */
154  sam_configgpio(GPIO_SMC_A5); /* PC23 A5 -> A3 */
155  sam_configgpio(GPIO_SMC_A6); /* PC24 A6 -> A4 */
156  sam_configgpio(GPIO_SMC_A7); /* PC25 A7 -> A5 */
157  sam_configgpio(GPIO_SMC_A8); /* PC26 A8 -> A6 */
158  sam_configgpio(GPIO_SMC_A9); /* PC27 A9 -> A7 */
159  sam_configgpio(GPIO_SMC_A10); /* PC28 A10 -> A8 */
160  sam_configgpio(GPIO_SMC_A11); /* PC29 A11 -> A9 */
161  sam_configgpio(GPIO_SDRAMC_A10_2); /* PD13 SDA10 -> A10 */
162  sam_configgpio(GPIO_SDRAMC_BA0); /* PA20 BA0 -> A11 */
163 
164  sam_configgpio(GPIO_SDRAMC_CKE); /* PD14 SDCKE -> CKE */
165  sam_configgpio(GPIO_SDRAMC_CK); /* PD23 SDCK -> CLK */
166  sam_configgpio(GPIO_SDRAMC_CS_1); /* PC15 SDCS -> nCS */
167  sam_configgpio(GPIO_SDRAMC_RAS); /* PD16 RAS -> nRAS */
168  sam_configgpio(GPIO_SDRAMC_CAS); /* PD17 CAS -> nCAS */
169  sam_configgpio(GPIO_SDRAMC_WE); /* PD29 SDWE -> nWE */
170  sam_configgpio(GPIO_SMC_NBS0); /* PC18 A0/NBS0 -> LDQM */
171  sam_configgpio(GPIO_SMC_NBS1); /* PD15 NWR1/NBS1 -> UDQM */
172 
173  /* Enable the SDRAMC peripheral */
174 
175  sam_sdramc_enableclk();
176 
177  regval = getreg32(SAM_MATRIX_CCFG_SMCNFCS);
178  regval |= MATRIX_CCFG_SMCNFCS_SDRAMEN;
179  putreg32(regval, SAM_MATRIX_CCFG_SMCNFCS);
180 
181  /* 1. SDRAM features must be set in the configuration register:
182  * asynchronous timings (TRC, TRAS, etc.), number of columns, rows, CAS
183  * latency, and the data bus width.
184  *
185  * SDRAMC_CR_NC_COL8 8 column bits
186  * SDRAMC_CR_NR_ROW11 1 row bits
187  * SDRAMC_CR_NB_BANK2 2 banks
188  * SDRAMC_CR_CAS_LATENCY3 3 cycle CAS latency
189  * SDRAMC_CR_DBW 16 bit
190  * SDRAMC_CR_TWR(4) 4 cycle write recovery delay
191  * SDRAMC_CR_TRCTRFC(11) 63 ns min
192  * SDRAMC_CR_TRP(5) 21 ns min Command period (PRE to ACT)
193  * SDRAMC_CR_TRCD(5) 21 ns min Active Command to read/Write Command delay time
194  * SDRAMC_CR_TRAS(8) 42 ns min Command period (ACT to PRE)
195  * SDRAMC_CR_TXSR(13) 70 ns min Exit self-refresh to active time
196  */
197 
198  regval = SDRAMC_CR_NC_COL8 | /* 8 column bits */
199  SDRAMC_CR_NR_ROW11 | /* 11 row bits */
200  SDRAMC_CR_NB_BANK2 | /* 2 banks */
201  SDRAMC_CR_CAS_LATENCY3 | /* 3 cycle CAS latency */
202  SDRAMC_CR_DBW | /* 16 bit */
203  SDRAMC_CR_TWR(4) | /* 4 cycle write recovery delay */
204  SDRAMC_CR_TRCTRFC(11) | /* 63 ns min */
205  SDRAMC_CR_TRP(5) | /* 21 ns min Command period (PRE to ACT) */
206  SDRAMC_CR_TRCD(5) | /* 21 ns min Active Command to read/Write Command delay time */
207  SDRAMC_CR_TRAS(8) | /* 42 ns min Command period (ACT to PRE) */
208  SDRAMC_CR_TXSR(13); /* 70 ns min Exit self-refresh to active time */
209 
210  putreg32(regval, SAM_SDRAMC_CR);
211 
212  /* 2. For mobile SDRAM, temperature-compensated self refresh (TCSR), drive
213  * strength (DS) and partial array self refresh (PASR) must be set in
214  * the Low Power Register.
215  */
216 
217  putreg32(0, SAM_SDRAMC_LPR);
218 
219  /* 3. The SDRAM memory type must be set in the Memory Device Register.*/
220 
221  putreg32(SDRAMC_MDR_SDRAM, SAM_SDRAMC_MDR);
222 
223  /* 4. A minimum pause of 200 usec is provided to precede any signal toggle.*/
224 
225  up_udelay(200);
226 
227  /* 5. A NOP command is issued to the SDRAM devices. The application must
228  * set Mode to 1 in the Mode Register and perform a write access to any
229  * SDRAM address.
230  */
231 
232  putreg32(SDRAMC_MR_MODE_NOP, SAM_SDRAMC_MR);
233  *psdram = 0;
234  up_udelay(200);
235 
236  /* 6. An All Banks Precharge command is issued to the SDRAM devices. The
237  * application must set Mode to 2 in the Mode Register and perform a
238  * write access to any SDRAM address.
239  */
240 
241  putreg32(SDRAMC_MR_MODE_PRECHARGE, SAM_SDRAMC_MR);
242  *psdram = 0;
243  up_udelay(200);
244 
245  /* 7. Eight auto-refresh (CBR) cycles are provided. The application must
246  * set the Mode to 4 in the Mode Register and perform a write access to
247  * any SDRAM location eight times.
248  */
249 
250  for (i = 0 ; i < 8; i++) {
251  putreg32(SDRAMC_MR_MODE_AUTOREFRESH, SAM_SDRAMC_MR);
252  *psdram = 0;
253  }
254 
255  up_udelay(200);
256 
257  /* 8. A Mode Register set (MRS) cycle is issued to program the parameters
258  * of the SDRAM devices, in particular CAS latency and burst length.
259  * The application must set Mode to 3 in the Mode Register and perform
260  * a write access to the SDRAM. The write address must be chosen so
261  * that BA[1:0] are set to 0. For example, with a 16-bit 128 MB SDRAM
262  * (12 rows, 9 columns, 4 banks) bank address, the SDRAM write access
263  * should be done at the address 0x70000000.
264  */
265 
266  putreg32(SDRAMC_MR_MODE_LOADMODE, SAM_SDRAMC_MR);
267  *psdram = 0;
268  up_udelay(200);
269 
270  /* 9. For mobile SDRAM initialization, an Extended Mode Register set
271  * (EMRS) cycle is issued to program the SDRAM parameters (TCSR, PASR,
272  * DS). The application must set Mode to 5 in the Mode Register and
273  * perform a write access to the SDRAM. The write address must be
274  * chosen so that BA[1] or BA[0] are set to 1.
275  *
276  * For example, with a 16-bit 128 MB SDRAM, (12 rows, 9 columns, 4
277  * banks) bank address the SDRAM write access should be done at the
278  * address 0x70800000 or 0x70400000.
279  */
280 
281  //putreg32(SDRAMC_MR_MODE_EXTLOADMODE, SDRAMC_MR_MODE_EXT_LOAD_MODEREG);
282  // *((uint8_t *)(psdram + SDRAM_BA0)) = 0;
283 
284  /* 10. The application must go into Normal Mode, setting Mode to 0 in the
285  * Mode Register and performing a write access at any location in the
286  * SDRAM.
287  */
288 
289  putreg32(SDRAMC_MR_MODE_NORMAL, SAM_SDRAMC_MR);
290  *psdram = 0;
291  up_udelay(200);
292 
293  /* 11. Write the refresh rate into the count field in the SDRAMC Refresh
294  * Timer register. (Refresh rate = delay between refresh cycles). The
295  * SDRAM device requires a refresh every 15.625 usec or 7.81 usec. With
296  * a 100 MHz frequency, the Refresh Timer Counter Register must be set
297  * with the value 1562(15.625 usec x 100 MHz) or 781(7.81 usec x 100
298  * MHz).
299  *
300  * For IS42S16100E, 2048 refresh cycle every 32ms, every 15.625 usec
301  */
302 
303  regval = (32 * (BOARD_MCK_FREQUENCY / 1000)) / 2048 ;
304  putreg32(regval, SAM_SDRAMC_TR);
305 
306  regval = getreg32(SAM_SDRAMC_CFR1);
307  regval |= SDRAMC_CFR1_UNAL;
308  putreg32(regval, SAM_SDRAMC_CFR1);
309 
310  /* After initialization, the SDRAM devices are fully functional. */
311 }
