73 static int dn_to_ordinal(uint16_t dn)
74 {
75 
76  const struct {
77  uint16_t low; // High(n-1) + 1
78  uint16_t high; // Average High(n)+Low(n+1) EX. 1356 = AVRG(1331,1382)
79  } dn2o[] = {
80  // R1(up) R2(down) V min V Max DN Min DN Max
81  {0, 0 }, // 0 No Resistors
82  {1, 579 }, // 1 24.9K 442K 0.166255191 0.44102252 204 553
83  {580, 967 }, // 2 32.4K 174K 0.492349322 0.770203609 605 966
84  {968, 1356}, // 3 38.3K 115K 0.787901749 1.061597759 968 1331
85  {1357, 1756}, // 4 46.4K 84.5K 1.124833577 1.386007306 1382 1738
86  {1757, 2137}, // 5 51.1K 61.9K 1.443393279 1.685367869 1774 2113
87  {2138, 2519}, // 6 61.9K 51.1K 1.758510242 1.974702534 2161 2476
88  {2520, 2919}, // 7 84.5K 46.4K 2.084546498 2.267198261 2562 2842
89  {2920, 3308}, // 8 115K 38.3K 2.437863827 2.57656294 2996 3230
90  {3309, 3699}, // 9 174K 32.4K 2.755223792 2.847933804 3386 3571
91  {3700, 4095}, // 10 442K 24.9K 3.113737849 3.147347506 3827 3946
92  };
93 
94  for (unsigned int i = 0; i < arraySize(dn2o); i++) {
95  if (dn >= dn2o[i].low && dn <= dn2o[i].high) {
96  return i;
97  }
98  }
99 
100  return -1;
101 }

104  * Name: read_id_dn
105  *
106  * Description:
107  * Read the HW sense set to get a DN of the value formed by
108  * 0 VDD
109  * |
110  * /
111  * \ R1
112  * /
113  * |
114  * +--------------- GPIO_HW_xxx_SENCE | ADC channel N
115  * |
116  * /
117  * \ R2
118  * /
119  * |
120  * |
121  * +--------------- GPIO_HW_xxx_DRIVE
122  *
123  * Input Parameters:
124  * id - pointer to receive the dn for the id set
125  * gpio_drive - gpio that is the drive
126  * gpio_sense - gpio that is the sence
127  * adc_channel - the Channel number associated with gpio_sense
128  *
129  * Returned Value:
130  * 0 - Success and id is set
131  * -EIO - FAiled to init or read the ADC
132  *
133  ************************************************************************************/
134 
135 static int read_id_dn(int *id, uint32_t gpio_drive, uint32_t gpio_sense, int adc_channel)
136 {
137  int rv = -EIO;
138  const unsigned int samples = 16;
139  /*
140  * Step one is there resistors?
141  *
142  * If we set the mid-point of the ladder which is the ADC input to an
143  * output, then whatever state is driven out should be seen by the GPIO
144  * that is on the bottom of the ladder that is switched to an input.
145  * The SENCE line is effectively an output with a high value pullup
146  * resistor on it driving an input through a series resistor with a pull up.
147  * If present the series resistor will form a low pass filter due to stray
148  * capacitance, but this is fine as long as we give it time to settle.
149  */
150 
151  /* Turn the drive lines to digital inputs with No pull up */
152 
153  stm32_configgpio(_MK_GPIO_INPUT(gpio_drive) & ~GPIO_PUPD_MASK);
154 
155  /* Turn the sense lines to digital outputs LOW */
156 
157  stm32_configgpio(_MK_GPIO_OUTPUT(gpio_sense));
158 
159 
160  up_udelay(100); /* About 10 TC assuming 485 K */
161 
162  /* Read Drive lines while sense are driven low */
163 
164  int low = stm32_gpioread(_MK_GPIO_INPUT(gpio_drive));
165 
166 
167  /* Write the sense lines HIGH */
168 
169  stm32_gpiowrite(_MK_GPIO_OUTPUT(gpio_sense), 1);
170 
171  up_udelay(100); /* About 10 TC assuming 485 K */
172 
173  /* Read Drive lines while sense are driven high */
174 
175  int high = stm32_gpioread(_MK_GPIO_INPUT(gpio_drive));
176 
177  /* restore the pins to ANALOG */
178 
179  stm32_configgpio(gpio_sense);
180 
181  /* Turn the drive lines to digital outputs LOW */
182 
183  stm32_configgpio(gpio_drive ^ GPIO_OUTPUT_SET);
184 
185  up_udelay(100); /* About 10 TC assuming 485 K */
186 
187  /* Are Resistors in place ?*/
188 
189  uint32_t dn_sum = 0;
190  uint16_t dn = 0;
191 
192  if ((high ^ low) && low == 0) {
193 
194 
195  /* Yes - Fire up the ADC (it has once control) */
196 
197  if (board_adc_init() == OK) {
198 
199  /* Read the value */
200  for (unsigned av = 0; av < samples; av++) {
201  dn = board_adc_sample(adc_channel);
202 
203  if (dn == 0xffff) {
204  break;
205  }
206 
207  dn_sum += dn;
208  }
209 
210  if (dn != 0xffff) {
211  *id = dn_sum / samples;
212  rv = OK;
213  }
214  }
215 
216  } else {
217  /* No - No Resistors is ID 0 */
218  *id = 0;
219  rv = OK;
220  }
221 
222  /* Turn the drive lines to digital outputs High */
223 
224  stm32_configgpio(gpio_drive);
225  return rv;
226 }

229 static int determine_hw_info(int *revision, int *version)
230 {
231  int dn;
232  int rv = read_id_dn(&dn, GPIO_HW_REV_DRIVE, GPIO_HW_REV_SENSE, ADC_HW_REV_SENSE_CHANNEL);
233 
234  if (rv == OK) {
235  *revision = dn_to_ordinal(dn);
236  rv = read_id_dn(&dn, GPIO_HW_VER_DRIVE, GPIO_HW_VER_SENSE, ADC_HW_VER_SENSE_CHANNEL);
237 
238  if (rv == OK) {
239  *version = dn_to_ordinal(dn);
240  }
241  }
242 
243  return rv;
244 }

332 int board_determine_hw_info()
333 {
334  int rv = determine_hw_info(&hw_revision, &hw_version);
335 
336  if (rv == OK) {

338  hw_info[HW_INFO_INIT_VER] = board_get_hw_version() + '0';
339  }
340 
341  return rv;
