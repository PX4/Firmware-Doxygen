112 int16_t uORB::KraitFastRpcChannel::send_message(const char *messageName, int32_t length, uint8_t *data)
113 {
114  int16_t rc = 0;
115  int32_t status = 0;
116  hrt_abstime t1, t4;
117  hrt_abstime t2 = 0;
118  hrt_abstime t3 = 0;
119  t1 = hrt_absolute_time();
120 
121  if (_AdspSubscriberCache.find(std::string(messageName)) == _AdspSubscriberCache.end()) {
122  // check the status from adsp. as it is not cached.
123  if (_KraitWrapper.IsSubscriberPresent(messageName, &status) == 0) {
124  _AdspSubscriberCache[messageName] = status;
125  _AdspSubscriberSampleTimestamp[messageName] = hrt_absolute_time();
126  }
127 
128  } else {
129  if ((hrt_absolute_time() - _AdspSubscriberSampleTimestamp[messageName]) > _SubCacheRefreshRate) {
130  if (_KraitWrapper.IsSubscriberPresent(messageName, &status) == 0) {
131  _AdspSubscriberCache[messageName] = status;
132  _AdspSubscriberSampleTimestamp[messageName] = hrt_absolute_time();
133  }
134  }
135  }
136 
137  if (_AdspSubscriberCache[messageName] > 0) {// there are remote subscribers
138  t2 = hrt_absolute_time();
139  rc = _KraitWrapper.SendData(messageName, length, data);
140  t3 = hrt_absolute_time();
141  _snd_msg_count++;
142  //PX4_DEBUG( "***** SENDING[%s] topic to remote....\n", messageName.c_str() );
143 
144  } else {
145  //PX4_DEBUG( "******* NO SUBSCRIBER PRESENT ON THE REMOTE FOR topic[%s] \n", messageName.c_str() );
146  }
147 
148  t4 = hrt_absolute_time();
149  _overall_snd_count++;
150 
151  if ((t4 - t1) < _overall_snd_min) { _overall_snd_min = (t4 - t1); }
152 
153  if ((t4 - t1) > _overall_snd_max) { _overall_snd_max = (t4 - t1); }
154 
155  if (_AdspSubscriberCache[messageName] > 0) {
156  if ((t3 - t2) < _snd_msg_min) { _snd_msg_min = (t3 - t2); }
157 
158  if ((t3 - t2) > _snd_msg_max) { _snd_msg_max = (t3 - t2); }
159 
160  _snd_msg_avg = ((double)((_snd_msg_avg * (_snd_msg_count - 1)) +
161  (unsigned long)(t3 - t2))) / (double)(_snd_msg_count);
162  }
163 
164  _overall_snd_avg = ((double)((_overall_snd_avg * (_overall_snd_count - 1)) +
165  (unsigned long)(t4 - t1))) / (double)(_overall_snd_count);
166 
167  if ((t4 - _log_check_time) > _log_check_interval) {
168  /*
169  PX4_DEBUG("SndMsgStats: overall_min: %lu overall_max: %lu snd_msg_min: %lu snd_msg_max: %lu",
170  _overall_snd_min, _overall_snd_max,
171  _snd_msg_min, _snd_msg_max);
172  PX4_DEBUG(".... overall_avg: %f (%lu) snd_msg_avg: %f (%lu)",
173  _overall_snd_avg, _overall_snd_count, _snd_msg_avg, _snd_msg_count);
174  */
175  _log_check_time = t4;
176  _overall_snd_min = _snd_msg_min = 0xFFFFFFF;
177  _overall_snd_max = _snd_msg_max = 0;
178  _overall_snd_count = _snd_msg_count = 0;
179  _overall_snd_avg = _snd_msg_avg = 0.0;
180  }
181 
182  //PX4_DEBUG( "Response for SendMessage for [%s],len[%d] rc[%d]\n", messageName.c_str(), length, rc );
183  return rc;
184 }

229 void uORB::KraitFastRpcChannel::fastrpc_recv_thread()
230 {
231  // sit in while loop.
232  int32_t rc = 0;
233  int32_t data_length = 0;
234  uint8_t *data = nullptr;
235  unsigned long rpc_min, rpc_max;
236  unsigned long orb_min, orb_max;
237  double rpc_avg, orb_avg;
238  unsigned long count = 0;
239  rpc_max = orb_max = 0;
240  rpc_min = orb_min = 0xFFFFFFFF;
241  rpc_avg = orb_avg = 0.0;
242 
243  int32_t num_topics = 0;
244 
245  hrt_abstime check_time = 0;
246 
247  while (!_ThreadShouldExit) {
248  hrt_abstime t1, t2, t3;
249  t1 = hrt_absolute_time();
250  rc = _KraitWrapper.ReceiveBulkData(&data, &data_length, &num_topics);
251 
252  t2 = hrt_absolute_time();
253 
254  if (rc == 0) {
255  //PX4_DEBUG( "Num of topics Received: %d", num_topics );
256  int32_t bytes_processed = 0;
257 
258  for (int i = 0; i < num_topics; ++i) {
259  uint8_t *new_pkt = &(data[bytes_processed]);
260  struct BulkTransferHeader *header = (struct BulkTransferHeader *)new_pkt;
261  char *messageName = (char *)(new_pkt + sizeof(struct BulkTransferHeader));
262  uint16_t check_msg_len = strlen(messageName);
263 
264  if (header->_MsgNameLen != (check_msg_len + 1)) {
265  PX4_ERR("Error: Packing error. Sent Msg Len. of[%d] but strlen returned:[%d]", header->_MsgNameLen, check_msg_len);
266  PX4_ERR("Error: NumTopics: %d processing topic: %d msgLen[%d] dataLen[%d] data_len[%d] bytes processed: %d",
267  num_topics, i, header->_MsgNameLen, header->_DataLen, data_length, bytes_processed);
268  DumpData(data, data_length, num_topics);
269  break;
270  }
271 
272  uint8_t *topic_data = (uint8_t *)(messageName + strlen(messageName) + 1);
273 
274  if (_RxHandler != nullptr) {
275  if (header->_MsgType == _DATA_MSG_TYPE) {
276  //PX4_DEBUG( "Received topic data for: [%s] len[%d]\n", messageName, data_length );
277  _RxHandler->process_received_message(messageName,
278  header->_DataLen, topic_data);
279 
280  } else if (header->_MsgType == _CONTROL_MSG_TYPE_ADVERTISE) {
281  PX4_DEBUG("Received topic advertise message for: [%s] len[%d]\n", messageName, data_length);
282  _RxHandler->process_remote_topic(messageName, true);
283 
284  } else if (header->_MsgType == _CONTROL_MSG_TYPE_UNADVERTISE) {
285  PX4_DEBUG("Received topic unadvertise message for: [%s] len[%d]\n", messageName, data_length);
286  _RxHandler->process_remote_topic(messageName, false);
287  }
288  }
289 
290  bytes_processed += header->_MsgNameLen + header->_DataLen + sizeof(struct BulkTransferHeader);
291  }
292 
293  } else {
294  PX4_DEBUG("Error: Getting data over fastRPC channel\n");
295  break;
296  }
297 
298  t3 = hrt_absolute_time();
299  count++;
300 
301  if ((unsigned long)(t2 - t1) < rpc_min) {
302  rpc_min = (unsigned long)(t2 - t1);
303  }
304 
305  if ((unsigned long)(t2 - t1) > rpc_max) {
306  rpc_max = (unsigned long)(t2 - t1);
307  }
308 
309  if ((unsigned long)(t3 - t2) < orb_min) {
310  orb_min = (unsigned long)(t3 - t2);
311  }
312 
313  if ((unsigned long)(t3 - t2) > orb_max) {
314  orb_max = (unsigned long)(t3 - t2);
315  }
316 
317  rpc_avg = ((double)((rpc_avg * (count - 1)) + (unsigned long)(t2 - t1))) / (double)(count);
318  orb_avg = ((double)((orb_avg * (count - 1)) + (unsigned long)(t3 - t2))) / (double)(count);
319 
320  if ((unsigned long)(t3 - check_time) >= 10000000) {
321  //PX4_DEBUG("Krait RPC Stats : rpc_min: %lu rpc_max: %lu rpc_avg: %f", rpc_min, rpc_max, rpc_avg);
322  //PX4_DEBUG("Krait RPC(orb) Stats: orb_min: %lu orb_max: %lu orb_avg: %f", orb_min, orb_max, orb_avg);
323  check_time = t3;
324  rpc_max = orb_max = 0;
325  rpc_min = orb_min = 0xFFFFFF;
326  orb_avg = 0;
327  rpc_avg = 0;
328  count = 0;
329  }
330 
331  //PX4_DEBUG("MsgName: %30s, t1: %lu, t2: %lu, t3: %lu, dt1: %lu, dt2: %lu",name, (unsigned long) t1, (unsigned long) t2, (unsigned long) t3,
332  // (unsigned long) (t2-t1), (unsigned long) (t3-t2));
333  }
334 
335  PX4_DEBUG("[uORB::KraitFastRpcChannel::fastrpc_recv_thread] Exiting fastrpc_recv_thread\n");
336 }

338 void DumpData(uint8_t *buffer, int32_t length, int32_t num_topics)
339 {
340  FILE *fp = fopen(_log_file_name.c_str(), "a+");
341 
342  if (fp == nullptr) {
343  PX4_ERR("Error unable to open log file[%s]", _log_file_name.c_str());
344  return;
345  }
346 
347  fprintf(fp, "===== Data Len[%d] num_topics[%d] ======\n", length, num_topics);
348 
349  for (int i = 0; i < length; i += 16) {
350  int remaining_chars = length - i;
351  remaining_chars = (remaining_chars >= 16) ? 16 : remaining_chars;
352 
353  fprintf(fp, "%p - ", &(buffer[i]));
354 
355  for (int j = 0; j < remaining_chars; j++) {
356  fprintf(fp, " %02X", buffer[i + j]);
357 
358  if (j == 7) {
359  fprintf(fp, " -");
360  }
361  }
362 
363  fprintf(fp, " ");
364 
365  for (int j = 0; j < remaining_chars; j++) {
366  fprintf(fp, "%c", (char)buffer[i + j ]);
367  }
368 
369  fprintf(fp, "\n");
370  }
371 
372  fclose(fp);
373 }
