75 void FlightTaskManualPosition::_scaleSticks()
76 {
77  /* Use same scaling as for FlightTaskManualAltitude */
78  FlightTaskManualAltitude::_scaleSticks();
79 
80  /* Constrain length of stick inputs to 1 for xy*/
81  Vector2f stick_xy(_sticks_expo(0), _sticks_expo(1));
82 
83  float mag = math::constrain(stick_xy.length(), 0.0f, 1.0f);
84 
85  if (mag > FLT_EPSILON) {
86  stick_xy = stick_xy.normalized() * mag;
87  }
88 
89  // scale the stick inputs
90  if (PX4_ISFINITE(_sub_vehicle_local_position->get().vxy_max)) {
91  // estimator provides vehicle specific max
92 
93  // use the minimum of the estimator and user specified limit
94  _velocity_scale = fminf(_constraints.speed_xy, _sub_vehicle_local_position->get().vxy_max);
95  // Allow for a minimum of 0.3 m/s for repositioning
96  _velocity_scale = fmaxf(_velocity_scale, 0.3f);
97 
98  } else if (stick_xy.length() > 0.5f) {
99  // raise the limit at a constant rate up to the user specified value
100 
101  if (_velocity_scale < _constraints.speed_xy) {
102  _velocity_scale += _deltatime * MPC_ACC_HOR_ESTM.get();
103 
104  } else {
105  _velocity_scale = _constraints.speed_xy;
106 
107  }
108  }
109 
110  // scale velocity to its maximum limits
111  Vector2f vel_sp_xy = stick_xy * _velocity_scale;
112 
113  /* Rotate setpoint into local frame. */
114  _rotateIntoHeadingFrame(vel_sp_xy);
115  _velocity_setpoint(0) = vel_sp_xy(0);
116  _velocity_setpoint(1) = vel_sp_xy(1);
117 }

119 void FlightTaskManualPosition::_updateXYlock()
120 {
121  /* If position lock is not active, position setpoint is set to NAN.*/
122  const float vel_xy_norm = Vector2f(&_velocity(0)).length();
123  const bool apply_brake = Vector2f(&_velocity_setpoint(0)).length() < FLT_EPSILON;
124  const bool stopped = (MPC_HOLD_MAX_XY.get() < FLT_EPSILON || vel_xy_norm < MPC_HOLD_MAX_XY.get());
125 
126  if (apply_brake && stopped && !PX4_ISFINITE(_position_setpoint(0))) {
127  _position_setpoint(0) = _position(0);
128  _position_setpoint(1) = _position(1);
129 
130  } else if (PX4_ISFINITE(_position_setpoint(0)) && apply_brake) {
131  // Position is locked but check if a reset event has happened.
132  // We will shift the setpoints.
133  if (_sub_vehicle_local_position->get().xy_reset_counter
134  != _reset_counter) {
135  _position_setpoint(0) = _position(0);
136  _position_setpoint(1) = _position(1);
137  _reset_counter = _sub_vehicle_local_position->get().xy_reset_counter;
138  }
139 
140  } else {
141  /* don't lock*/
142  _position_setpoint(0) = NAN;
143  _position_setpoint(1) = NAN;
144  }
145 }
