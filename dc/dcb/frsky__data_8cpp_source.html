125 /**
126  * Sends one byte, performing byte-stuffing if necessary.
127  */
128 static void frsky_send_byte(int uart, uint8_t value)
129 {
130  const uint8_t x5E[] = { 0x5D, 0x3E };
131  const uint8_t x5D[] = { 0x5D, 0x3D };
132 
133  switch (value) {
134  case 0x5E:
135  write(uart, x5E, sizeof(x5E));
136  break;
137 
138  case 0x5D:
139  write(uart, x5D, sizeof(x5D));
140  break;
141 
142  default:
143  write(uart, &value, sizeof(value));
144  break;
145  }
146 }

148 /**
149  * Sends one data id/value pair.
150  */
151 static void frsky_send_data(int uart, uint8_t id, int16_t data)
152 {
153  /* Cast data to unsigned, because signed shift might behave incorrectly */
154  uint16_t udata = data;
155 
156  frsky_send_startstop(uart);
157 
158  frsky_send_byte(uart, id);
159  frsky_send_byte(uart, udata); /* LSB */
160  frsky_send_byte(uart, udata >> 8); /* MSB */
161 }

163 void frsky_update_topics()
164 {
165  struct frsky_subscription_data_s *subs = subscription_data;
166  bool updated;
167 
168  /* get a local copy of the current sensor values */
169  orb_check(subs->sensor_sub, &updated);
170 
171  if (updated) {
172  orb_copy(ORB_ID(sensor_combined), subs->sensor_sub, &subs->sensor_combined);
173  }
174 
175  orb_check(subs->vehicle_air_data_sub, &updated);
176 
177  if (updated) {
178  orb_copy(ORB_ID(vehicle_air_data), subs->vehicle_air_data_sub, &subs->airdata);
179  }
180 
181  /* get a local copy of the vehicle status */
182  orb_check(subs->vehicle_status_sub, &updated);
183 
184  if (updated) {
185  struct vehicle_status_s vehicle_status;
186  orb_copy(ORB_ID(vehicle_status), subs->vehicle_status_sub, &vehicle_status);
187  subs->current_flight_mode = vehicle_status.nav_state;
188  }
189 
190  /* get a local copy of the battery data */
191  orb_check(subs->battery_status_sub, &updated);
192 
193  if (updated) {
194  orb_copy(ORB_ID(battery_status), subs->battery_status_sub, &subs->battery_status);
195  }
196 
197  /* get a local copy of the global position data */
198  orb_check(subs->vehicle_global_position_sub, &updated);
199 
200  if (updated) {
201  orb_copy(ORB_ID(vehicle_global_position), subs->vehicle_global_position_sub, &subs->global_pos);
202  }
203 
204  /* get a local copy of the raw GPS data */
205  orb_check(subs->vehicle_gps_position_sub, &updated);
206 
207  if (updated) {
208  orb_copy(ORB_ID(vehicle_gps_position), subs->vehicle_gps_position_sub, &subs->vehicle_gps_position);
209  }
210 }

216 void frsky_send_frame1(int uart)
217 {
218  struct frsky_subscription_data_s *subs = subscription_data;
219 
220  /* send formatted frame */
221  frsky_send_data(uart, FRSKY_ID_ACCEL_X, roundf(subs->sensor_combined.accelerometer_m_s2[0] * 1000.0f));
222  frsky_send_data(uart, FRSKY_ID_ACCEL_Y, roundf(subs->sensor_combined.accelerometer_m_s2[1] * 1000.0f));
223  frsky_send_data(uart, FRSKY_ID_ACCEL_Z, roundf(subs->sensor_combined.accelerometer_m_s2[2] * 1000.0f));
224 
225  frsky_send_data(uart, FRSKY_ID_BARO_ALT_BP, subs->airdata.baro_alt_meter);
226  frsky_send_data(uart, FRSKY_ID_BARO_ALT_AP, roundf(frac(subs->airdata.baro_alt_meter) * 100.0f));
227 
228  frsky_send_data(uart, FRSKY_ID_VFAS,
229  roundf(subs->battery_status.voltage_v * 10.0f));
230  frsky_send_data(uart, FRSKY_ID_CURRENT,
231  (subs->battery_status.current_a < 0) ? 0 : roundf(subs->battery_status.current_a * 10.0f));
232 
233  int16_t telem_flight_mode = get_telemetry_flight_mode(subs->current_flight_mode);
234  frsky_send_data(uart, FRSKY_ID_TEMP1, telem_flight_mode); // send flight mode as TEMP1. This matches with OpenTX & APM
235 
236  frsky_send_data(uart, FRSKY_ID_TEMP2, subs->vehicle_gps_position.satellites_used * 10 +
237  subs->vehicle_gps_position.fix_type);
238 
239  frsky_send_startstop(uart);
240 }

251 /**
252  * Sends frame 2 (every 1000ms):
253  * GPS course, latitude, longitude, ground speed, GPS altitude, remaining battery level
254  */
255 void frsky_send_frame2(int uart)
256 {
257  struct vehicle_global_position_s *global_pos = &subscription_data->global_pos;
258  struct battery_status_s *battery_status = &subscription_data->battery_status;
259  struct vehicle_gps_position_s *gps = &subscription_data->vehicle_gps_position;
260  /* send formatted frame */
261  float course = 0, lat = 0, lon = 0, speed = 0, alt = 0;
262  char lat_ns = 0, lon_ew = 0;
263  int sec = 0;
264 
265  if (global_pos->timestamp != 0 && hrt_absolute_time() < global_pos->timestamp + 20000) {
266  course = global_pos->yaw / M_PI_F * 180.0f;
267 
268  if (course < 0.f) { // course is in range [0, 360], 0=north, CW
269  course += 360.f;
270  }
271 
272  lat = frsky_format_gps(fabsf(global_pos->lat));
273  lat_ns = (global_pos->lat < 0) ? 'S' : 'N';
274  lon = frsky_format_gps(fabsf(global_pos->lon));
275  lon_ew = (global_pos->lon < 0) ? 'W' : 'E';
276  speed = sqrtf(global_pos->vel_n * global_pos->vel_n + global_pos->vel_e * global_pos->vel_e)
277  * 25.0f / 46.0f;
278  alt = global_pos->alt;
279  }
280 
281  if (gps->timestamp != 0 && hrt_absolute_time() < gps->timestamp + 20000) {
282  time_t time_gps = gps->time_utc_usec / 1000000ULL;
283  struct tm *tm_gps = gmtime(&time_gps);
284 
285  sec = tm_gps->tm_sec;
286  }
287 
288  frsky_send_data(uart, FRSKY_ID_GPS_COURS_BP, course);
289  frsky_send_data(uart, FRSKY_ID_GPS_COURS_AP, frac(course) * 1000.0f);
290 
291  frsky_send_data(uart, FRSKY_ID_GPS_LAT_BP, lat);
292  frsky_send_data(uart, FRSKY_ID_GPS_LAT_AP, frac(lat) * 10000.0f);
293  frsky_send_data(uart, FRSKY_ID_GPS_LAT_NS, lat_ns);
294 
295  frsky_send_data(uart, FRSKY_ID_GPS_LONG_BP, lon);
296  frsky_send_data(uart, FRSKY_ID_GPS_LONG_AP, frac(lon) * 10000.0f);
297  frsky_send_data(uart, FRSKY_ID_GPS_LONG_EW, lon_ew);
298 
299  frsky_send_data(uart, FRSKY_ID_GPS_SPEED_BP, speed);
300  frsky_send_data(uart, FRSKY_ID_GPS_SPEED_AP, frac(speed) * 100.0f);
301 
302  frsky_send_data(uart, FRSKY_ID_GPS_ALT_BP, alt);
303  frsky_send_data(uart, FRSKY_ID_GPS_ALT_AP, frac(alt) * 100.0f);
304 
305  frsky_send_data(uart, FRSKY_ID_FUEL,
306  roundf(battery_status->remaining * 100.0f));
307 
308  frsky_send_data(uart, FRSKY_ID_GPS_SEC, sec);
309 
310  frsky_send_startstop(uart);
311 }

313 /**
314  * Sends frame 3 (every 5000ms):
315  * GPS date & time
316  */
317 void frsky_send_frame3(int uart)
318 {
319  /* send formatted frame */
320  time_t time_gps = subscription_data->vehicle_gps_position.time_utc_usec / 1000000ULL;
321  struct tm *tm_gps = gmtime(&time_gps);
322  uint16_t hour_min = (tm_gps->tm_min << 8) | (tm_gps->tm_hour & 0xff);
323  frsky_send_data(uart, FRSKY_ID_GPS_DAY_MONTH, tm_gps->tm_mday);
324  frsky_send_data(uart, FRSKY_ID_GPS_YEAR, tm_gps->tm_year);
325  frsky_send_data(uart, FRSKY_ID_GPS_HOUR_MIN, hour_min);
326  frsky_send_data(uart, FRSKY_ID_GPS_SEC, tm_gps->tm_sec);
327 
328  frsky_send_startstop(uart);
329 }

331 /* parse 11 byte frames */
332 bool frsky_parse_host(uint8_t *sbuf, int nbytes, struct adc_linkquality *v)
333 {
334  bool data_ready = false;
335  static int dcount = 0;
336  static uint8_t type = 0;
337  static uint8_t data[11];
338  static enum {
339  HEADER = 0,
340  TYPE,
341  DATA,
342  TRAILER
343  } state = HEADER;
344 
345  for (int i = 0; i < nbytes; i++) {
346  switch (state) {
347  case HEADER:
348  if (sbuf[i] == 0x7E) {
349  state = TYPE;
350  }
351 
352  break;
353 
354  case TYPE:
355  if (sbuf[i] != 0x7E) {
356  state = DATA;
357  type = sbuf[i];
358  dcount = 0;
359  }
360 
361  break;
362 
363  case DATA:
364 
365  /* read 8 data bytes */
366  if (dcount < 7) {
367  data[dcount++] = sbuf[i];
368 
369  } else {
370  /* received all data bytes */
371  state = TRAILER;
372  }
373 
374  break;
375 
376  case TRAILER:
377  state = HEADER;
378 
379  if (sbuf[i] != 0x7E) {
380 // warnx("host packet error: %x", sbuf[i]);
381 
382  } else {
383  data_ready = true;
384 
385  if (type == 0xFE) {
386  /* this is an adc_linkquality packet */
387  v->ad1 = data[0];
388  v->ad2 = data[1];
389  v->linkq = data[2];
390  }
391  }
392 
393  break;
394  }
395  }
396 
397  return data_ready;
398 }
