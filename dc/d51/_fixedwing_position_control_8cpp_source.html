471 float
472 FixedwingPositionControl::calculate_target_airspeed(float airspeed_demand)
473 {
474  /*
475  * Calculate accelerated stall airspeed factor from commanded bank angle and use it to increase minimum airspeed.
476  *
477  * We don't know the stall speed of the aircraft, but assuming user defined
478  * minimum airspeed (FW_AIRSPD_MIN) is slightly larger than stall speed
479  * this is close enough.
480  *
481  * increase lift vector to balance additional weight in bank
482  * cos(bank angle) = W/L = 1/n
483  * n is the load factor
484  *
485  * lift is proportional to airspeed^2 so the increase in stall speed is
486  * Vsacc = Vs * sqrt(n)
487  *
488  */
489  float adjusted_min_airspeed = _parameters.airspeed_min;
490 
491  if (_airspeed_valid && PX4_ISFINITE(_att_sp.roll_body)) {
492 
493  adjusted_min_airspeed = constrain(_parameters.airspeed_min / sqrtf(cosf(_att_sp.roll_body)), _parameters.airspeed_min,
494  _parameters.airspeed_max);
495  }
496 
497  // add minimum ground speed undershoot (only non-zero in presence of sufficient wind)
498  // sanity check: limit to range
499  return constrain(airspeed_demand + _groundspeed_undershoot, adjusted_min_airspeed, _parameters.airspeed_max);
500 }

502 void
503 FixedwingPositionControl::calculate_gndspeed_undershoot(const Vector2f &curr_pos,
504  const Vector2f &ground_speed,
505  const position_setpoint_s &pos_sp_prev, const position_setpoint_s &pos_sp_curr)
506 {
507  if (pos_sp_curr.valid && !_l1_control.circle_mode()) {
508  /* rotate ground speed vector with current attitude */
509  Vector2f yaw_vector(_R_nb(0, 0), _R_nb(1, 0));
510  yaw_vector.normalize();
511  float ground_speed_body = yaw_vector * ground_speed;
512 
513  /* The minimum desired ground speed is the minimum airspeed projected on to the ground using the altitude and horizontal difference between the waypoints if available*/
514  float distance = 0.0f;
515  float delta_altitude = 0.0f;
516 
517  if (pos_sp_prev.valid) {
518  distance = get_distance_to_next_waypoint(pos_sp_prev.lat, pos_sp_prev.lon, pos_sp_curr.lat, pos_sp_curr.lon);
519  delta_altitude = pos_sp_curr.alt - pos_sp_prev.alt;
520 
521  } else {
522  distance = get_distance_to_next_waypoint((double)curr_pos(0), (double)curr_pos(1), pos_sp_curr.lat, pos_sp_curr.lon);
523  delta_altitude = pos_sp_curr.alt - _global_pos.alt;
524  }
525 
526  float ground_speed_desired = _parameters.airspeed_min * cosf(atan2f(delta_altitude, distance));
527 
528  /*
529  * Ground speed undershoot is the amount of ground velocity not reached
530  * by the plane. Consequently it is zero if airspeed is >= min ground speed
531  * and positive if airspeed < min ground speed.
532  *
533  * This error value ensures that a plane (as long as its throttle capability is
534  * not exceeded) travels towards a waypoint (and is not pushed more and more away
535  * by wind). Not countering this would lead to a fly-away.
536  */
537  _groundspeed_undershoot = max(ground_speed_desired - ground_speed_body, 0.0f);
538 
539  } else {
540  _groundspeed_undershoot = 0.0f;
541  }

605 bool
606 FixedwingPositionControl::update_desired_altitude(float dt)
607 {
608  /*
609  * The complete range is -1..+1, so this is 6%
610  * of the up or down range or 3% of the total range.
611  */
612  const float deadBand = 0.06f;
613 
614  /*
615  * The correct scaling of the complete range needs
616  * to account for the missing part of the slope
617  * due to the deadband
618  */
619  const float factor = 1.0f - deadBand;
620 
621  /* Climbout mode sets maximum throttle and pitch up */
622  bool climbout_mode = false;
623 
624  /*
625  * Reset the hold altitude to the current altitude if the uncertainty
626  * changes significantly.
627  * This is to guard against uncommanded altitude changes
628  * when the altitude certainty increases or decreases.
629  */
630 
631  if (fabsf(_althold_epv - _global_pos.epv) > ALTHOLD_EPV_RESET_THRESH) {
632  _hold_alt = _global_pos.alt;
633  _althold_epv = _global_pos.epv;
634  }
635 
636  /*
637  * Manual control has as convention the rotation around
638  * an axis. Positive X means to rotate positively around
639  * the X axis in NED frame, which is pitching down
640  */
641  if (_manual.x > deadBand) {
642  /* pitching down */
643  float pitch = -(_manual.x - deadBand) / factor;
644  _hold_alt += (_parameters.max_sink_rate * dt) * pitch;
645  _was_in_deadband = false;
646 
647  } else if (_manual.x < - deadBand) {
648  /* pitching up */
649  float pitch = -(_manual.x + deadBand) / factor;
650  _hold_alt += (_parameters.max_climb_rate * dt) * pitch;
651  _was_in_deadband = false;
652  climbout_mode = (pitch > MANUAL_THROTTLE_CLIMBOUT_THRESH);
653 
654  } else if (!_was_in_deadband) {
655  /* store altitude at which manual.x was inside deadBand
656  * The aircraft should immediately try to fly at this altitude
657  * as this is what the pilot expects when he moves the stick to the center */
658  _hold_alt = _global_pos.alt;
659  _althold_epv = _global_pos.epv;
660  _was_in_deadband = true;
661  }
662 
663  if (_vehicle_status.is_vtol) {
664  if (_vehicle_status.is_rotary_wing || _vehicle_status.in_transition_mode) {
665  _hold_alt = _global_pos.alt;
666  }
667  }
668 
669  return climbout_mode;
670 }

695 bool
696 FixedwingPositionControl::control_position(const Vector2f &curr_pos, const Vector2f &ground_speed,
697  const position_setpoint_s &pos_sp_prev, const position_setpoint_s &pos_sp_curr)
698 {
699  float dt = 0.01f;
700 
701  if (_control_position_last_called > 0) {
702  dt = hrt_elapsed_time(&_control_position_last_called) * 1e-6f;
703  }
704 
705  _control_position_last_called = hrt_absolute_time();
706 
707  _l1_control.set_dt(dt);
708 
709  /* only run position controller in fixed-wing mode and during transitions for VTOL */
710  if (_vehicle_status.is_rotary_wing && !_vehicle_status.in_transition_mode) {
711  _control_mode_current = FW_POSCTRL_MODE_OTHER;
712  return false;
713  }
714 
715  bool setpoint = true;
716 
717  _att_sp.fw_control_yaw = false; // by default we don't want yaw to be contoller directly with rudder
718  _att_sp.apply_flaps = vehicle_attitude_setpoint_s::FLAPS_OFF; // by default we don't use flaps
719 
720  calculate_gndspeed_undershoot(curr_pos, ground_speed, pos_sp_prev, pos_sp_curr);
721 
722  // l1 navigation logic breaks down when wind speed exceeds max airspeed
723  // compute 2D groundspeed from airspeed-heading projection
724  Vector2f air_speed_2d{_airspeed * cosf(_yaw), _airspeed * sinf(_yaw)};
725  Vector2f nav_speed_2d{0.0f, 0.0f};
726 
727  // angle between air_speed_2d and ground_speed
728  float air_gnd_angle = acosf((air_speed_2d * ground_speed) / (air_speed_2d.length() * ground_speed.length()));
729 
730  // if angle > 90 degrees or groundspeed is less than threshold, replace groundspeed with airspeed projection
731  if ((fabsf(air_gnd_angle) > M_PI_2_F) || (ground_speed.length() < 3.0f)) {
732  nav_speed_2d = air_speed_2d;
733 
734  } else {
735  nav_speed_2d = ground_speed;
736  }
737 
738  /* no throttle limit as default */
739  float throttle_max = 1.0f;
740 
741  /* save time when airplane is in air */
742  if (!_was_in_air && !_vehicle_land_detected.landed) {
743  _was_in_air = true;
744  _time_went_in_air = hrt_absolute_time();
745  _takeoff_ground_alt = _global_pos.alt;
746  }
747 
748  /* reset flag when airplane landed */
749  if (_vehicle_land_detected.landed) {
750  _was_in_air = false;
751  }
752 
753  /* Reset integrators if switching to this mode from a other mode in which posctl was not active */
754  if (_control_mode_current == FW_POSCTRL_MODE_OTHER) {
755  /* reset integrators */
756  _tecs.reset_state();
757  }
758 
759  if (_control_mode.flag_control_auto_enabled && pos_sp_curr.valid) {
760  /* AUTONOMOUS FLIGHT */
761 
762  _control_mode_current = FW_POSCTRL_MODE_AUTO;
763 
764  /* reset hold altitude */
765  _hold_alt = _global_pos.alt;
766 
767  /* reset hold yaw */
768  _hdg_hold_yaw = _yaw;
769 
770  /* get circle mode */
771  bool was_circle_mode = _l1_control.circle_mode();
772 
773  /* restore speed weight, in case changed intermittently (e.g. in landing handling) */
774  _tecs.set_speed_weight(_parameters.speed_weight);
775 
776  /* current waypoint (the one currently heading for) */
777  Vector2f curr_wp((float)pos_sp_curr.lat, (float)pos_sp_curr.lon);
778 
779  /* Initialize attitude controller integrator reset flags to 0 */
780  _att_sp.roll_reset_integral = false;
781  _att_sp.pitch_reset_integral = false;
782  _att_sp.yaw_reset_integral = false;
783 
784  /* previous waypoint */
785  Vector2f prev_wp{0.0f, 0.0f};
786 
787  if (pos_sp_prev.valid) {
788  prev_wp(0) = (float)pos_sp_prev.lat;
789  prev_wp(1) = (float)pos_sp_prev.lon;
790 
791  } else {
792  /*
793  * No valid previous waypoint, go for the current wp.
794  * This is automatically handled by the L1 library.
795  */
796  prev_wp(0) = (float)pos_sp_curr.lat;
797  prev_wp(1) = (float)pos_sp_curr.lon;
798  }
799 
800  float mission_airspeed = _parameters.airspeed_trim;
801 
802  if (PX4_ISFINITE(pos_sp_curr.cruising_speed) &&
803  pos_sp_curr.cruising_speed > 0.1f) {
804 
805  mission_airspeed = pos_sp_curr.cruising_speed;
806  }
807 
808  float mission_throttle = _parameters.throttle_cruise;
809 
810  if (PX4_ISFINITE(pos_sp_curr.cruising_throttle) &&
811  pos_sp_curr.cruising_throttle > 0.01f) {
812 
813  mission_throttle = pos_sp_curr.cruising_throttle;
814  }
815 
816  if (pos_sp_curr.type == position_setpoint_s::SETPOINT_TYPE_IDLE) {
817  _att_sp.thrust = 0.0f;
818  _att_sp.roll_body = 0.0f;
819  _att_sp.pitch_body = 0.0f;
820 
821  } else if (pos_sp_curr.type == position_setpoint_s::SETPOINT_TYPE_POSITION) {
822  /* waypoint is a plain navigation waypoint */
823  _l1_control.navigate_waypoints(prev_wp, curr_wp, curr_pos, nav_speed_2d);
824  _att_sp.roll_body = _l1_control.get_roll_setpoint();
825  _att_sp.yaw_body = _l1_control.nav_bearing();
826 
827  tecs_update_pitch_throttle(pos_sp_curr.alt,
828  calculate_target_airspeed(mission_airspeed),
829  radians(_parameters.pitch_limit_min) - _parameters.pitchsp_offset_rad,
830  radians(_parameters.pitch_limit_max) - _parameters.pitchsp_offset_rad,
831  _parameters.throttle_min,
832  _parameters.throttle_max,
833  mission_throttle,
834  false,
835  radians(_parameters.pitch_limit_min));
836 
837  } else if (pos_sp_curr.type == position_setpoint_s::SETPOINT_TYPE_LOITER) {
838 
839  /* waypoint is a loiter waypoint */
840  _l1_control.navigate_loiter(curr_wp, curr_pos, pos_sp_curr.loiter_radius,
841  pos_sp_curr.loiter_direction, nav_speed_2d);
842  _att_sp.roll_body = _l1_control.get_roll_setpoint();
843  _att_sp.yaw_body = _l1_control.nav_bearing();
844 
845  float alt_sp = pos_sp_curr.alt;
846 
847  if (in_takeoff_situation()) {
848  alt_sp = max(alt_sp, _takeoff_ground_alt + _parameters.climbout_diff);
849  _att_sp.roll_body = constrain(_att_sp.roll_body, radians(-5.0f), radians(5.0f));
850  }
851 
852  if (_fw_pos_ctrl_status.abort_landing) {
853  if (pos_sp_curr.alt - _global_pos.alt < _parameters.climbout_diff) {
854  // aborted landing complete, normal loiter over landing point
855  _fw_pos_ctrl_status.abort_landing = false;
856 
857  } else {
858  // continue straight until vehicle has sufficient altitude
859  _att_sp.roll_body = 0.0f;
860  }
861  }
862 
863  tecs_update_pitch_throttle(alt_sp,
864  calculate_target_airspeed(mission_airspeed),
865  radians(_parameters.pitch_limit_min) - _parameters.pitchsp_offset_rad,
866  radians(_parameters.pitch_limit_max) - _parameters.pitchsp_offset_rad,
867  _parameters.throttle_min,
868  _parameters.throttle_max,
869  _parameters.throttle_cruise,
870  false,
871  radians(_parameters.pitch_limit_min));
872 
873  } else if (pos_sp_curr.type == position_setpoint_s::SETPOINT_TYPE_LAND) {
874  control_landing(curr_pos, ground_speed, pos_sp_prev, pos_sp_curr);
875 
876  } else if (pos_sp_curr.type == position_setpoint_s::SETPOINT_TYPE_TAKEOFF) {
877  control_takeoff(curr_pos, ground_speed, pos_sp_prev, pos_sp_curr);
878  }
879 
880  /* reset landing state */
881  if (pos_sp_curr.type != position_setpoint_s::SETPOINT_TYPE_LAND) {
882  reset_landing_state();
883  }
884 
885  /* reset takeoff/launch state */
886  if (pos_sp_curr.type != position_setpoint_s::SETPOINT_TYPE_TAKEOFF) {
887  reset_takeoff_state();
888  }
889 
890  if (was_circle_mode && !_l1_control.circle_mode()) {
891  /* just kicked out of loiter, reset roll integrals */
892  _att_sp.roll_reset_integral = true;
893  }
894 
895  } else if (_control_mode.flag_control_velocity_enabled &&
896  _control_mode.flag_control_altitude_enabled) {
897  /* POSITION CONTROL: pitch stick moves altitude setpoint, throttle stick sets airspeed,
898  heading is set to a distant waypoint */
899 
900  if (_control_mode_current != FW_POSCTRL_MODE_POSITION) {
901  /* Need to init because last loop iteration was in a different mode */
902  _hold_alt = _global_pos.alt;
903  _hdg_hold_yaw = _yaw;
904  _hdg_hold_enabled = false; // this makes sure the waypoints are reset below
905  _yaw_lock_engaged = false;
906 
907  /* reset setpoints from other modes (auto) otherwise we won't
908  * level out without new manual input */
909  _att_sp.roll_body = _manual.y * _parameters.man_roll_max_rad;
910  _att_sp.yaw_body = 0;
911  }
912 
913  _control_mode_current = FW_POSCTRL_MODE_POSITION;
914 
915  float altctrl_airspeed = get_demanded_airspeed();
916 
917  /* update desired altitude based on user pitch stick input */
918  bool climbout_requested = update_desired_altitude(dt);
919 
920  /* if we assume that user is taking off then help by demanding altitude setpoint well above ground
921  * and set limit to pitch angle to prevent stearing into ground
922  */
923  float pitch_limit_min{0.0f};
924  do_takeoff_help(&_hold_alt, &pitch_limit_min);
925 
926  /* throttle limiting */
927  throttle_max = _parameters.throttle_max;
928 
929  if (_vehicle_land_detected.landed && (fabsf(_manual.z) < THROTTLE_THRESH)) {
930  throttle_max = 0.0f;
931  }
932 
933  tecs_update_pitch_throttle(_hold_alt,
934  altctrl_airspeed,
935  radians(_parameters.pitch_limit_min),
936  radians(_parameters.pitch_limit_max),
937  _parameters.throttle_min,
938  throttle_max,
939  _parameters.throttle_cruise,
940  climbout_requested,
941  climbout_requested ? radians(10.0f) : pitch_limit_min,
942  tecs_status_s::TECS_MODE_NORMAL);
943 
944  /* heading control */
945  if (fabsf(_manual.y) < HDG_HOLD_MAN_INPUT_THRESH &&
946  fabsf(_manual.r) < HDG_HOLD_MAN_INPUT_THRESH) {
947 
948  /* heading / roll is zero, lock onto current heading */
949  if (fabsf(_att.yawspeed) < HDG_HOLD_YAWRATE_THRESH && !_yaw_lock_engaged) {
950  // little yaw movement, lock to current heading
951  _yaw_lock_engaged = true;
952 
953  }
954 
955  /* user tries to do a takeoff in heading hold mode, reset the yaw setpoint on every iteration
956  to make sure the plane does not start rolling
957  */
958  if (in_takeoff_situation()) {
959  _hdg_hold_enabled = false;
960  _yaw_lock_engaged = true;
961  }
962 
963  if (_yaw_lock_engaged) {
964 
965  /* just switched back from non heading-hold to heading hold */
966  if (!_hdg_hold_enabled) {
967  _hdg_hold_enabled = true;
968  _hdg_hold_yaw = _yaw;
969 
970  get_waypoint_heading_distance(_hdg_hold_yaw, _hdg_hold_prev_wp, _hdg_hold_curr_wp, true);
971  }
972 
973  /* we have a valid heading hold position, are we too close? */
974  float dist = get_distance_to_next_waypoint(_global_pos.lat, _global_pos.lon, _hdg_hold_curr_wp.lat,
975  _hdg_hold_curr_wp.lon);
976 
977  if (dist < HDG_HOLD_REACHED_DIST) {
978  get_waypoint_heading_distance(_hdg_hold_yaw, _hdg_hold_prev_wp, _hdg_hold_curr_wp, false);
979  }
980 
981  Vector2f prev_wp{(float)_hdg_hold_prev_wp.lat, (float)_hdg_hold_prev_wp.lon};
982  Vector2f curr_wp{(float)_hdg_hold_curr_wp.lat, (float)_hdg_hold_curr_wp.lon};
983 
984  /* populate l1 control setpoint */
985  _l1_control.navigate_waypoints(prev_wp, curr_wp, curr_pos, ground_speed);
986 
987  _att_sp.roll_body = _l1_control.get_roll_setpoint();
988  _att_sp.yaw_body = _l1_control.nav_bearing();
989 
990  if (in_takeoff_situation()) {
991  /* limit roll motion to ensure enough lift */
992  _att_sp.roll_body = constrain(_att_sp.roll_body, radians(-15.0f), radians(15.0f));
993  }
994  }
995  }
996 
997  if (!_yaw_lock_engaged || fabsf(_manual.y) >= HDG_HOLD_MAN_INPUT_THRESH ||
998  fabsf(_manual.r) >= HDG_HOLD_MAN_INPUT_THRESH) {
999 
1000  _hdg_hold_enabled = false;
1001  _yaw_lock_engaged = false;
1002  _att_sp.roll_body = _manual.y * _parameters.man_roll_max_rad;
1003  _att_sp.yaw_body = 0;
1004  }
1005 
1006  } else if (_control_mode.flag_control_altitude_enabled) {
1007  /* ALTITUDE CONTROL: pitch stick moves altitude setpoint, throttle stick sets airspeed */
1008 
1009  if (_control_mode_current != FW_POSCTRL_MODE_POSITION && _control_mode_current != FW_POSCTRL_MODE_ALTITUDE) {
1010  /* Need to init because last loop iteration was in a different mode */
1011  _hold_alt = _global_pos.alt;
1012  }
1013 
1014  _control_mode_current = FW_POSCTRL_MODE_ALTITUDE;
1015 
1016  /* Get demanded airspeed */
1017  float altctrl_airspeed = get_demanded_airspeed();
1018 
1019  /* update desired altitude based on user pitch stick input */
1020  bool climbout_requested = update_desired_altitude(dt);
1021 
1022  /* if we assume that user is taking off then help by demanding altitude setpoint well above ground
1023  * and set limit to pitch angle to prevent stearing into ground
1024  */
1025  float pitch_limit_min{0.0f};
1026  do_takeoff_help(&_hold_alt, &pitch_limit_min);
1027 
1028  /* throttle limiting */
1029  throttle_max = _parameters.throttle_max;
1030 
1031  if (_vehicle_land_detected.landed && (fabsf(_manual.z) < THROTTLE_THRESH)) {
1032  throttle_max = 0.0f;
1033  }
1034 
1035  tecs_update_pitch_throttle(_hold_alt,
1036  altctrl_airspeed,
1037  radians(_parameters.pitch_limit_min),
1038  radians(_parameters.pitch_limit_max),
1039  _parameters.throttle_min,
1040  throttle_max,
1041  _parameters.throttle_cruise,
1042  climbout_requested,
1043  climbout_requested ? radians(10.0f) : pitch_limit_min,
1044  tecs_status_s::TECS_MODE_NORMAL);
1045 
1046  _att_sp.roll_body = _manual.y * _parameters.man_roll_max_rad;
1047  _att_sp.yaw_body = 0;
1048 
1049  } else {
1050  _control_mode_current = FW_POSCTRL_MODE_OTHER;
1051 
1052  /* do not publish the setpoint */
1053  setpoint = false;
1054 
1055  // reset hold altitude
1056  _hold_alt = _global_pos.alt;
1057 
1058  /* reset landing and takeoff state */
1059  if (!_last_manual) {
1060  reset_landing_state();
1061  reset_takeoff_state();
1062  }
1063  }
1064 
1065  /* Copy thrust output for publication */
1066  if (_control_mode_current == FW_POSCTRL_MODE_AUTO && // launchdetector only available in auto
1067  pos_sp_curr.type == position_setpoint_s::SETPOINT_TYPE_TAKEOFF &&
1068  _launch_detection_state != LAUNCHDETECTION_RES_DETECTED_ENABLEMOTORS &&
1069  !_runway_takeoff.runwayTakeoffEnabled()) {
1070 
1071  /* making sure again that the correct thrust is used,
1072  * without depending on library calls for safety reasons.
1073  the pre-takeoff throttle and the idle throttle normally map to the same parameter. */
1074  _att_sp.thrust = _parameters.throttle_idle;
1075 
1076  } else if (_control_mode_current == FW_POSCTRL_MODE_AUTO &&
1077  pos_sp_curr.type == position_setpoint_s::SETPOINT_TYPE_TAKEOFF &&
1078  _runway_takeoff.runwayTakeoffEnabled()) {
1079 
1080  _att_sp.thrust = _runway_takeoff.getThrottle(min(get_tecs_thrust(), throttle_max));
1081 
1082  } else if (_control_mode_current == FW_POSCTRL_MODE_AUTO &&

1122 
1123  } else {
1124  _last_manual = true;
1125  }
1126

1302  const position_setpoint_s &pos_sp_prev, const position_setpoint_s &pos_sp_curr)

1309  prev_wp(0) = (float)pos_sp_prev.lat;
1310  prev_wp(1) = (float)pos_sp_prev.lon;
1311 
1312  } else {
1313  /*
1314  * No valid previous waypoint, go for the current wp.
1315  * This is automatically handled by the L1 library.
1316  */

1318  prev_wp(1) = (float)pos_sp_curr.lon;

1322  // if they have been enabled using the corresponding parameter

1324

1326  if (_time_started_landing == 0) {
1327  _time_started_landing = hrt_absolute_time();

1329  _fw_pos_ctrl_status.abort_landing = false;
1330  }
1331 
1332  const float bearing_airplane_currwp = get_bearing_to_next_waypoint((double)curr_pos(0), (double)curr_pos(1),
1333  (double)curr_wp(0), (double)curr_wp(1));
1334 
1335  float bearing_lastwp_currwp = bearing_airplane_currwp;
1336 
1337  if (pos_sp_prev.valid) {
1338  bearing_lastwp_currwp = get_bearing_to_next_waypoint((double)prev_wp(0), (double)prev_wp(1), (double)curr_wp(0),
1339  (double)curr_wp(1));
1340  }
1341 
1342  /* Horizontal landing control */
1343  /* switch to heading hold for the last meters, continue heading hold after */
1344  float wp_distance = get_distance_to_next_waypoint((double)curr_pos(0), (double)curr_pos(1), (double)curr_wp(0),
1345  (double)curr_wp(1));
1346

1348  float wp_distance_save = wp_distance;
1349 
1350  if (fabsf(bearing_airplane_currwp - bearing_lastwp_currwp) >= radians(90.0f)) {
1351  wp_distance_save = 0.0f;
1352  }
1353 
1354  // create virtual waypoint which is on the desired flight path but
1355  // some distance behind landing waypoint. This will make sure that the plane
1356  // will always follow the desired flight path even if we get close or past
1357  // the landing waypoint
1358  if (pos_sp_prev.valid) {
1359  double lat = pos_sp_curr.lat;
1360  double lon = pos_sp_curr.lon;
1361 
1362  create_waypoint_from_line_and_dist(pos_sp_curr.lat, pos_sp_curr.lon,
1363  pos_sp_prev.lat, pos_sp_prev.lon, -1000.0f, &lat, &lon);
1364 
1365  curr_wp(0) = (float)lat;
1366  curr_wp(1) = (float)lon;
1367  }
1368 
1369  // we want the plane to keep tracking the desired flight path until we start flaring
1370  // if we go into heading hold mode earlier then we risk to be pushed away from the runway by cross winds
1371  if ((_parameters.land_heading_hold_horizontal_distance > 0.0f) && !_land_noreturn_horizontal &&
1372  ((wp_distance < _parameters.land_heading_hold_horizontal_distance) || _land_noreturn_vertical)) {
1373 
1374  if (pos_sp_prev.valid) {
1375  /* heading hold, along the line connecting this and the last waypoint */
1376  _target_bearing = bearing_lastwp_currwp;
1377 
1378  } else {
1379  _target_bearing = _yaw;
1380  }
1381 
1382  _land_noreturn_horizontal = true;
1383  mavlink_log_info(&_mavlink_log_pub, "Landing, heading hold");
1384  }
1385 
1386  if (_land_noreturn_horizontal) {
1387  // heading hold
1388  _l1_control.navigate_heading(_target_bearing, _yaw, ground_speed);
1389 
1390  } else {
1391  // normal navigation
1392  _l1_control.navigate_waypoints(prev_wp, curr_wp, curr_pos, ground_speed);
1393  }
1394 
1395  _att_sp.roll_body = _l1_control.get_roll_setpoint();
1396  _att_sp.yaw_body = _l1_control.nav_bearing();
1397 
1398  if (_land_noreturn_horizontal) {
1399  /* limit roll motion to prevent wings from touching the ground first */
1400  _att_sp.roll_body = constrain(_att_sp.roll_body, radians(-10.0f), radians(10.0f));
1401  }
1402 
1403  /* Vertical landing control */
1404  /* apply minimum pitch (flare) and limit roll if close to touch down, altitude error is negative (going down) */
1405 
1406  // default to no terrain estimation, just use landing waypoint altitude
1407  float terrain_alt = pos_sp_curr.alt;
1408 
1409  if (_parameters.land_use_terrain_estimate == 1) {
1410  if (_global_pos.terrain_alt_valid) {
1411  // all good, have valid terrain altitude
1412  terrain_alt = _global_pos.terrain_alt;
1413  _t_alt_prev_valid = terrain_alt;
1414  _time_last_t_alt = hrt_absolute_time();
1415 
1416  } else if (_time_last_t_alt == 0) {
1417  // we have started landing phase but don't have valid terrain
1418  // wait for some time, maybe we will soon get a valid estimate
1419  // until then just use the altitude of the landing waypoint
1420  if (hrt_elapsed_time(&_time_started_landing) < 10 * 1000 * 1000) {
1421  terrain_alt = pos_sp_curr.alt;
1422 
1423  } else {
1424  // still no valid terrain, abort landing
1425  terrain_alt = pos_sp_curr.alt;
1426  _fw_pos_ctrl_status.abort_landing = true;
1427  }
1428 
1429  } else if ((!_global_pos.terrain_alt_valid && hrt_elapsed_time(&_time_last_t_alt) < T_ALT_TIMEOUT * 1000 * 1000)
1430  || _land_noreturn_vertical) {
1431  // use previous terrain estimate for some time and hope to recover
1432  // if we are already flaring (land_noreturn_vertical) then just
1433  // go with the old estimate
1434  terrain_alt = _t_alt_prev_valid;
1435 
1436  } else {
1437  // terrain alt was not valid for long time, abort landing
1438  terrain_alt = _t_alt_prev_valid;
1439  _fw_pos_ctrl_status.abort_landing = true;
1440  }
1441  }
1442 
1443  /* Check if we should start flaring with a vertical and a
1444  * horizontal limit (with some tolerance)
1445  * The horizontal limit is only applied when we are in front of the wp
1446  */
1447  if (((_global_pos.alt < terrain_alt + _landingslope.flare_relative_alt()) &&
1448  (wp_distance_save < _landingslope.flare_length() + 5.0f)) ||
1449  _land_noreturn_vertical) { //checking for land_noreturn to avoid unwanted climb out
1450 
1451  /* land with minimal speed */
1452 
1453  /* force TECS to only control speed with pitch, altitude is only implicitly controlled now */
1454  // _tecs.set_speed_weight(2.0f);
1455 
1456  /* kill the throttle if param requests it */
1457  float throttle_max = _parameters.throttle_max;
1458 
1459  /* enable direct yaw control using rudder/wheel */
1460  if (_land_noreturn_horizontal) {
1461  _att_sp.yaw_body = _target_bearing;
1462  _att_sp.fw_control_yaw = true;
1463  }
1464 
1465  if (_global_pos.alt < terrain_alt + _landingslope.motor_lim_relative_alt() || _land_motor_lim) {
1466  throttle_max = min(throttle_max, _parameters.throttle_land_max);
1467 
1468  if (!_land_motor_lim) {
1469  _land_motor_lim = true;
1470  mavlink_log_info(&_mavlink_log_pub, "Landing, limiting throttle");
1471  }
1472  }
1473 
1474  float flare_curve_alt_rel = _landingslope.getFlareCurveRelativeAltitudeSave(wp_distance, bearing_lastwp_currwp,
1475  bearing_airplane_currwp);
1476 
1477  /* avoid climbout */
1478  if ((_flare_curve_alt_rel_last < flare_curve_alt_rel && _land_noreturn_vertical) || _land_stayonground) {
1479  flare_curve_alt_rel = 0.0f; // stay on ground
1480  _land_stayonground = true;
1481  }
1482 
1483  const float airspeed_land = _parameters.land_airspeed_scale * _parameters.airspeed_min;
1484  const float throttle_land = _parameters.throttle_min + (_parameters.throttle_max - _parameters.throttle_min) * 0.1f;
1485 
1486  tecs_update_pitch_throttle(terrain_alt + flare_curve_alt_rel,
1487  calculate_target_airspeed(airspeed_land),
1488  radians(_parameters.land_flare_pitch_min_deg),
1489  radians(_parameters.land_flare_pitch_max_deg),
1490  0.0f,
1491  throttle_max,
1492  throttle_land,
1493  false,
1494  _land_motor_lim ? radians(_parameters.land_flare_pitch_min_deg) : radians(_parameters.pitch_limit_min),
1495  _land_motor_lim ? tecs_status_s::TECS_MODE_LAND_THROTTLELIM : tecs_status_s::TECS_MODE_LAND);
1496 
1497  if (!_land_noreturn_vertical) {
1498  // just started with the flaring phase
1499  _flare_pitch_sp = 0.0f;
1500  _flare_height = _global_pos.alt - terrain_alt;
1501  mavlink_log_info(&_mavlink_log_pub, "Landing, flaring");
1502  _land_noreturn_vertical = true;
1503 
1504  } else {
1505  if (_global_pos.vel_d > 0.1f) {
1506  _flare_pitch_sp = radians(_parameters.land_flare_pitch_min_deg) *
1507  constrain((_flare_height - (_global_pos.alt - terrain_alt)) / _flare_height, 0.0f, 1.0f);
1508  }
1509 
1510  // otherwise continue using previous _flare_pitch_sp
1511  }
1512 
1513  _att_sp.pitch_body = _flare_pitch_sp;
1514  _flare_curve_alt_rel_last = flare_curve_alt_rel;
1515 
1516  } else {
1517 
1518  /* intersect glide slope:
1519  * minimize speed to approach speed
1520  * if current position is higher than the slope follow the glide slope (sink to the
1521  * glide slope)
1522  * also if the system captures the slope it should stay
1523  * on the slope (bool land_onslope)
1524  * if current position is below the slope continue at previous wp altitude
1525  * until the intersection with slope
1526  * */
1527 
1528  float altitude_desired = terrain_alt;
1529 
1530  const float landing_slope_alt_rel_desired = _landingslope.getLandingSlopeRelativeAltitudeSave(wp_distance,
1531  bearing_lastwp_currwp, bearing_airplane_currwp);
1532 
1533  if (_global_pos.alt > terrain_alt + landing_slope_alt_rel_desired || _land_onslope) {
1534  /* stay on slope */
1535  altitude_desired = terrain_alt + landing_slope_alt_rel_desired;
1536 
1537  if (!_land_onslope) {
1538  mavlink_log_info(&_mavlink_log_pub, "Landing, on slope");
1539  _land_onslope = true;
1540  }
1541 
1542  } else {
1543  /* continue horizontally */
1544  if (pos_sp_prev.valid) {
1545  altitude_desired = pos_sp_prev.alt;
1546 
1547  } else {
1548  altitude_desired = _global_pos.alt;
1549  }
1550  }
1551 
1552  const float airspeed_approach = _parameters.land_airspeed_scale * _parameters.airspeed_min;
1553 
1554  tecs_update_pitch_throttle(altitude_desired,
1555  calculate_target_airspeed(airspeed_approach),
1556  radians(_parameters.pitch_limit_min),
1557  radians(_parameters.pitch_limit_max),
1558  _parameters.throttle_min,
1559  _parameters.throttle_max,
1560  _parameters.throttle_cruise,
1561  false,
1562  radians(_parameters.pitch_limit_min));
1563  }
