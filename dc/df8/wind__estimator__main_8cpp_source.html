184 void
185 WindEstimatorModule::cycle()
186 {
187  perf_count(_perf_interval);
188  perf_begin(_perf_elapsed);
189 
190  bool param_updated;
191  orb_check(_param_sub, &param_updated);
192 
193  if (param_updated) {
194  update_params();
195  }
196 
197  bool lpos_valid = false;
198  bool att_valid = false;
199  bool airspeed_valid = false;
200 
201  const hrt_abstime time_now_usec = hrt_absolute_time();
202 
203  // validate required conditions for the filter to fuse measurements
204 
205  vehicle_attitude_s att = {};
206 
207  if (orb_copy(ORB_ID(vehicle_attitude), _vehicle_attitude_sub, &att) == PX4_OK) {
208  att_valid = (time_now_usec - att.timestamp < 1_s) && (att.timestamp > 0);
209  }
210 
211  vehicle_local_position_s lpos = {};
212 
213  if (orb_copy(ORB_ID(vehicle_local_position), _vehicle_local_position_sub, &lpos) == PX4_OK) {
214  lpos_valid = (time_now_usec - lpos.timestamp < 1_s) && (lpos.timestamp > 0) && lpos.v_xy_valid;
215  }
216 
217  // update wind and airspeed estimator
218  _wind_estimator.update(time_now_usec);
219 
220  if (lpos_valid && att_valid) {
221 
222  Vector3f vI(lpos.vx, lpos.vy, lpos.vz);
223  Quatf q(att.q);
224 
225  // sideslip fusion
226  _wind_estimator.fuse_beta(time_now_usec, vI, q);
227 
228  // additionally, for airspeed fusion we need to have recent measurements
229  airspeed_s airspeed = {};
230 
231  if (orb_copy(ORB_ID(airspeed), _airspeed_sub, &airspeed) == PX4_OK) {
232  airspeed_valid = (time_now_usec - airspeed.timestamp < 1_s) && (airspeed.timestamp > 0);
233  }
234 
235  if (airspeed_valid) {
236  Vector3f vel_var{Dcmf(q) *Vector3f{lpos.evh, lpos.evh, lpos.evv}};
237 
238  // airspeed fusion
239  _wind_estimator.fuse_airspeed(time_now_usec, airspeed.indicated_airspeed_m_s, vI, Vector2f{vel_var(0), vel_var(1)});
240  }
241 
242  // if we fused either airspeed or sideslip we publish a wind_estimate message
243  wind_estimate_s wind_est = {};
244 
245  wind_est.timestamp = time_now_usec;
246  float wind[2];
247  _wind_estimator.get_wind(wind);
248  wind_est.windspeed_north = wind[0];
249  wind_est.windspeed_east = wind[1];
250  float wind_cov[2];
251  _wind_estimator.get_wind_var(wind_cov);
252  wind_est.variance_north = wind_cov[0];
253  wind_est.variance_east = wind_cov[1];
254  wind_est.tas_innov = _wind_estimator.get_tas_innov();
255  wind_est.tas_innov_var = _wind_estimator.get_tas_innov_var();
256  wind_est.beta_innov = _wind_estimator.get_beta_innov();
257  wind_est.beta_innov_var = _wind_estimator.get_beta_innov_var();
258  wind_est.tas_scale = _wind_estimator.get_tas_scale();
259 
260  orb_publish_auto(ORB_ID(wind_estimate), &_wind_est_pub, &wind_est, &_instance, ORB_PRIO_DEFAULT);
261  }
262 
263  perf_end(_perf_elapsed);
264 
265  if (should_exit()) {
266  exit_and_cleanup();
267 
268  } else {
269  /* schedule next cycle */
270  work_queue(LPWORK, &_work, (worker_t)&WindEstimatorModule::cycle_trampoline, this, USEC2TICK(SCHEDULE_INTERVAL));
271  }
272 }

306  PRINT_MODULE_DESCRIPTION(
307  R"DESCR_STR(
308 ### Description
309 This module runs a combined wind and airspeed scale factor estimator.
310 If provided the vehicle NED speed and attitude it can estimate the horizontal wind components based on a zero
311 sideslip assumption. This makes the estimator only suitable for fixed wing vehicles.
312 If provided an airspeed measurement this module also estimates an airspeed scale factor based on the following model:
313 measured_airspeed = scale_factor * real_airspeed.
