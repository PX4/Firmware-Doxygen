117 static void led_pwm_timer_init_timer(unsigned timer)
118 {
119  irqstate_t flags = px4_enter_critical_section();
120 
121  /* enable the timer clock before we try to talk to it */
122 
123  modifyreg32(led_pwm_timers[timer].clock_register, 0, led_pwm_timers[timer].clock_bit);
124 
125  /* disable and configure the timer */
126  rCR1(timer) = 0;
127  rCR2(timer) = 0;
128  rSMCR(timer) = 0;
129  rDIER(timer) = 0;
130  rCCER(timer) = 0;
131  rCCMR1(timer) = 0;
132  rCCMR2(timer) = 0;
133  rCCR1(timer) = 0;
134  rCCR2(timer) = 0;
135  rCCR3(timer) = 0;
136  rCCR4(timer) = 0;
137  rCCER(timer) = 0;
138  rDCR(timer) = 0;
139 
140  if ((led_pwm_timers[timer].base == STM32_TIM1_BASE) || (led_pwm_timers[timer].base == STM32_TIM8_BASE)) {
141 
142  /* master output enable = on */
143 
144  rBDTR(timer) = ATIM_BDTR_MOE;
145  }
146 
147  /* If the timer clock source provided as clock_freq is the STM32_APBx_TIMx_CLKIN
148  * then configure the timer to free-run at 1MHz.
149  * Otherwise, other frequencies are attainable by adjusting .clock_freq accordingly.
150  */
151 
152  rPSC(timer) = (led_pwm_timers[timer].clock_freq / 1000000) - 1;
153 
154  /* configure the timer to update at the desired rate */
155 
156  rARR(timer) = (1000000 / LED_PWM_RATE) - 1;
157 
158  /* generate an update event; reloads the counter and all registers */
159  rEGR(timer) = GTIM_EGR_UG;
160 
161  px4_leave_critical_section(flags);
162 
163 }

166 static void
167 led_pwm_channel_init(unsigned channel)
168 {
169  /* Only initialize used channels */
170 
171  if (led_pwm_channels[channel].timer_channel) {
172 
173  unsigned timer = led_pwm_channels[channel].timer_index;
174 
175  /* configure the GPIO first */
176  px4_arch_configgpio(led_pwm_channels[channel].gpio_out);
177 
178  uint16_t polarity = led_pwm_channels[channel].masks;
179 
180  /* configure the channel */
181  switch (led_pwm_channels[channel].timer_channel) {
182  case 1:
183  rCCMR1(timer) |= (GTIM_CCMR_MODE_PWM1 << GTIM_CCMR1_OC1M_SHIFT) | GTIM_CCMR1_OC1PE;
184  rCCER(timer) |= polarity | GTIM_CCER_CC1E;
185  break;
186 
187  case 2:
188  rCCMR1(timer) |= (GTIM_CCMR_MODE_PWM1 << GTIM_CCMR1_OC2M_SHIFT) | GTIM_CCMR1_OC2PE;
189  rCCER(timer) |= polarity | GTIM_CCER_CC2E;
190  break;
191 
192  case 3:
193  rCCMR2(timer) |= (GTIM_CCMR_MODE_PWM1 << GTIM_CCMR2_OC3M_SHIFT) | GTIM_CCMR2_OC3PE;
194  rCCER(timer) |= polarity | GTIM_CCER_CC3E;
195  break;
196 
197  case 4:
198  rCCMR2(timer) |= (GTIM_CCMR_MODE_PWM1 << GTIM_CCMR2_OC4M_SHIFT) | GTIM_CCMR2_OC4PE;
199  rCCER(timer) |= polarity | GTIM_CCER_CC4E;
200  break;
201  }
202  }
203 }

205 int
206 led_pwm_servo_set(unsigned channel, uint8_t cvalue)
207 {
208  if (channel >= arraySize(led_pwm_channels)) {
209  return -1;
210  }
211 
212  unsigned timer = led_pwm_channels[channel].timer_index;
213 
214  /* test timer for validity */
215  if ((led_pwm_timers[timer].base == 0) ||
216  (led_pwm_channels[channel].gpio_out == 0)) {
217  return -1;
218  }
219 
220  unsigned period = led_pwm_timer_get_period(timer);
221 
222  unsigned value = (unsigned)cvalue * period / 255;
223 
224  /* configure the channel */
225  if (value > 0) {
226  value--;
227  }
228 
229 
230  switch (led_pwm_channels[channel].timer_channel) {
231  case 1:
232  rCCR1(timer) = value;
233  break;
234 
235  case 2:
236  rCCR2(timer) = value;
237  break;
238 
239  case 3:
240  rCCR3(timer) = value;
241  break;
242 
243  case 4:
244  rCCR4(timer) = value;
245  break;
246 
247  default:
248  return -1;
249  }
250 
251  return 0;
252 }

319 led_pwm_servo_arm(bool armed)
320 {
321  /* iterate timers and arm/disarm appropriately */
322  for (unsigned i = 0; i < arraySize(led_pwm_timers); i++) {
323  if (led_pwm_timers[i].base != 0) {
324  if (armed) {
325  /* force an update to preload all registers */
326  rEGR(i) = GTIM_EGR_UG;
327 
328  /* arm requires the timer be enabled */
329  rCR1(i) |= GTIM_CR1_CEN | GTIM_CR1_ARPE;
330 
331  } else {
332  rCR1(i) = 0;
333  }
334  }
