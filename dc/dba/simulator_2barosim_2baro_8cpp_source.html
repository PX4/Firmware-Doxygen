161 int
162 BAROSIM::init()
163 {
164  int ret;
165  struct baro_report brp = {};
166 
167  ret = VirtDevObj::init();
168 
169  if (ret != OK) {
170  PX4_ERR("VirtDevObj init failed");
171  goto out;
172  }
173 
174  /* allocate basic report buffers */
175  _reports = new ringbuffer::RingBuffer(2, sizeof(baro_report));
176 
177  if (_reports == nullptr) {
178  PX4_ERR("can't get memory for reports");
179  ret = -ENOMEM;
180  goto out;
181  }
182 
183  _reports->flush();
184 
185  _baro_topic = orb_advertise_multi(ORB_ID(sensor_baro), &brp,
186  &_orb_class_instance, (is_external()) ? ORB_PRIO_HIGH : ORB_PRIO_DEFAULT);
187 
188  if (_baro_topic == nullptr) {
189  PX4_ERR("failed to create sensor_baro publication");
190  return -ENODEV;
191  }
192 
193  /* fill report structures */
194  start();
195 
196 out:
197  return ret;
198 }

214 int
215 BAROSIM::collect()
216 {
217  bool status;
218 
219  simulator::RawBaroData raw_baro;
220 
221  perf_begin(_sample_perf);
222 
223  report.timestamp = hrt_absolute_time();
224  report.error_count = perf_event_count(_comms_errors);
225 
226  /* read requested */
227  Simulator *sim = Simulator::getInstance();
228 
229  if (sim == nullptr) {
230  PX4_ERR("Error BAROSIM_DEV::transfer no simulator");
231  return -ENODEV;
232  }
233 
234  PX4_DEBUG("BAROSIM_DEV::transfer getting sample");
235  status = sim->getBaroSample((uint8_t *)(&raw_baro), sizeof(raw_baro));
236 
237  if (!status) {
238  perf_count(_comms_errors);
239  perf_end(_sample_perf);
240  return -1;
241  }
242 
243  report.pressure = raw_baro.pressure;
244  report.temperature = raw_baro.temperature;
245 
246  /* fake device ID */
247  report.device_id = 478459;
248 
249  /* publish it */
250  if (!(m_pub_blocked)) {
251  if (_baro_topic != nullptr) {
252  /* publish it */
253  orb_publish(ORB_ID(sensor_baro), _baro_topic, &report);
254 
255  } else {
256  PX4_WARN("BAROSIM::collect _baro_topic not initialized");
257  }
258  }
259 
260  _reports->force(&report);
261 
262  perf_end(_sample_perf);
263 
264  return OK;
265 }
