34 /**
35  * @file MulticopterLandDetector.cpp
36  *
37  *The MC land-detector goes through 3 states before it will detect landed:
38  *
39  *State 1 (=ground_contact):
40  *ground_contact is detected once the vehicle is not moving along the NED-z direction and has
41  *a thrust value below 0.3 of the thrust_range (thrust_hover - thrust_min). The condition has to be true
42  *for GROUND_CONTACT_TRIGGER_TIME_US in order to detect ground_contact
43  *
44  *State 2 (=maybe_landed):
45  *maybe_landed can only occur if the internal ground_contact hysteresis state is true. maybe_landed criteria requires to have no motion in x and y,
46  *no rotation and a thrust below 0.1 of the thrust_range (thrust_hover - thrust_min). In addition, the mc_pos_control turns off the thrust_sp in
47  *body frame along x and y which helps to detect maybe_landed. The criteria for maybe_landed needs to be true for MAYBE_LAND_DETECTOR_TRIGGER_TIME_US.
48  *
49  *State 3 (=landed)
50  *landed can only be detected if maybe_landed is true for LAND_DETECTOR_TRIGGER_TIME_US. No farther criteria is tested, but the mc_pos_control goes into
51  *idle (thrust_sp = 0) which helps to detect landed. By doing this the thrust-criteria of State 2 will always be met, however the remaining criteria of no rotation and no motion still
52  *have to be valid.
53 
54  *It is to note that if one criteria is not met, then vehicle exits the state directly without blocking.
55  *
56  *If the land-detector does not detect ground_contact, then the vehicle is either flying or falling, where free fall detection heavily relies
57  *on the acceleration. TODO: verify that free fall is reliable
58  *
59  * @author Johan Jansen <jnsn.johan@gmail.com>
60  * @author Morten Lysgaard <morten@lysgaard.no>
61  * @author Julian Oes <julian@oes.ch>
62  */

135 bool MulticopterLandDetector::_get_freefall_state()
136 {
137  if (_params.freefall_acc_threshold < 0.1f
138  || _params.freefall_acc_threshold > 10.0f) { //if parameter is set to zero or invalid, disable free-fall detection.
139  return false;
140  }
141 
142  if (_sensors.timestamp == 0) {
143  // _sensors is not valid yet, we have to assume we're not falling.
144  return false;
145  }
146 
147  float acc_norm = _sensors.accel_x * _sensors.accel_x
148  + _sensors.accel_y * _sensors.accel_y
149  + _sensors.accel_z * _sensors.accel_z;
150  acc_norm = sqrtf(acc_norm); //norm of specific force. Should be close to 9.8 m/s^2 when landed.
151 
152  return (acc_norm < _params.freefall_acc_threshold); //true if we are currently falling
153 }

155 bool MulticopterLandDetector::_get_ground_contact_state()
156 {
157  // only trigger flight conditions if we are armed
158  if (!_arming.armed) {
159  return true;
160  }
161 
162  // land speed threshold
163  float land_speed_threshold = 0.9f * math::max(_params.landSpeed, 0.1f);
164 
165  // Check if we are moving vertically - this might see a spike after arming due to
166  // throttle-up vibration. If accelerating fast the throttle thresholds will still give
167  // an accurate in-air indication.
168  bool verticalMovement;
169 
170  if (hrt_elapsed_time(&_landed_time) < LAND_DETECTOR_LAND_PHASE_TIME_US) {
171 
172  // Widen acceptance thresholds for landed state right after arming
173  // so that motor spool-up and other effects do not trigger false negatives.
174  verticalMovement = fabsf(_vehicleLocalPosition.vz) > _params.maxClimbRate * 2.5f;
175 
176  } else {
177 
178  // Adjust maxClimbRate if land_speed is lower than 2x maxClimbrate
179  float maxClimbRate = ((land_speed_threshold * 0.5f) < _params.maxClimbRate) ? (0.5f * land_speed_threshold) :
180  _params.maxClimbRate;
181  verticalMovement = fabsf(_vehicleLocalPosition.z_deriv) > maxClimbRate;
182  }
183 
184  // Check if we are moving horizontally.
185  bool horizontalMovement = sqrtf(_vehicleLocalPosition.vx * _vehicleLocalPosition.vx
186  + _vehicleLocalPosition.vy * _vehicleLocalPosition.vy) > _params.maxVelocity;
187 
188  // if we have a valid velocity setpoint and the vehicle is demanded to go down but no vertical movement present,
189  // we then can assume that the vehicle hit ground
190  bool in_descend = _is_climb_rate_enabled()
191  && (_vehicleLocalPositionSetpoint.vz >= land_speed_threshold);
192  bool hit_ground = in_descend && !verticalMovement;
193 
194  // TODO: we need an accelerometer based check for vertical movement for flying without GPS
195  if ((_has_low_thrust() || hit_ground) && (!horizontalMovement || !_has_position_lock())
196  && (!verticalMovement || !_has_altitude_lock())) {
197  return true;
198  }
199 
200  return false;
201 }

203 bool MulticopterLandDetector::_get_maybe_landed_state()
204 {
205  // Time base for this function
206  const hrt_abstime now = hrt_absolute_time();
207 
208  // only trigger flight conditions if we are armed
209  if (!_arming.armed) {
210  return true;
211  }
212 
213  if (_has_minimal_thrust()) {
214  if (_min_trust_start == 0) {
215  _min_trust_start = now;
216  }
217 
218  } else {
219  _min_trust_start = 0;
220  }
221 
222  float landThresholdFactor = 1.0f;
223 
224  // Widen acceptance thresholds for landed state right after landed
225  if (hrt_elapsed_time(&_landed_time) < LAND_DETECTOR_LAND_PHASE_TIME_US) {
226  landThresholdFactor = 2.5f;
227  }
228 
229  // Next look if all rotation angles are not moving.
230  float maxRotationScaled = _params.maxRotation_rad_s * landThresholdFactor;
231 
232  bool rotating = (fabsf(_vehicleAttitude.rollspeed) > maxRotationScaled) ||
233  (fabsf(_vehicleAttitude.pitchspeed) > maxRotationScaled) ||
234  (fabsf(_vehicleAttitude.yawspeed) > maxRotationScaled);
235 
236  // Return status based on armed state and throttle if no position lock is available.
237  if (!_has_altitude_lock() && !rotating) {
238  // The system has minimum trust set (manual or in failsafe)
239  // if this persists for 8 seconds AND the drone is not
240  // falling consider it to be landed. This should even sustain
241  // quite acrobatic flight.
242  return (_min_trust_start > 0) && (hrt_elapsed_time(&_min_trust_start) > 8_s);
243  }
244 
245  if (_ground_contact_hysteresis.get_state() && _has_minimal_thrust() && !rotating) {
246  // Ground contact, no thrust and no movement -> landed
247  return true;
248  }
249 
250  return false;
251 }

253 bool MulticopterLandDetector::_get_landed_state()
254 {
255  // reset the landed_time
256  if (!_maybe_landed_hysteresis.get_state()) {
257 
258  _landed_time = 0;
259 
260  } else if (_landed_time == 0) {
261 
262  _landed_time = hrt_absolute_time();
263 
264  }
265

268  return _maybe_landed_hysteresis.get_state();

312 bool MulticopterLandDetector::_has_low_thrust()
313 {
314  // 30% of throttle range between min and hover
315  float sys_min_throttle = _params.minThrottle + (_params.hoverThrottle - _params.minThrottle) * 0.3f;
316 
317  // Check if thrust output is less than the minimum auto throttle param.
318  return _actuators.control[actuator_controls_s::INDEX_THROTTLE] <= sys_min_throttle;
319 }

321 bool MulticopterLandDetector::_has_minimal_thrust()
322 {
323  // 10% of throttle range between min and hover once we entered ground contact
324  float sys_min_throttle = _params.minThrottle + (_params.hoverThrottle - _params.minThrottle) * _params.throttleRange;
325 
326  // Determine the system min throttle based on flight mode
327  if (!_control_mode.flag_control_climb_rate_enabled) {
328  sys_min_throttle = (_params.minManThrottle + 0.01f);
329  }
330 
331  // Check if thrust output is less than the minimum auto throttle param.
332  return _actuators.control[actuator_controls_s::INDEX_THROTTLE] <= sys_min_throttle;
333 }
