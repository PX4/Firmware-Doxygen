74 ssize_t UavcanMagnetometerBridge::read(struct file *filp, char *buffer, size_t buflen)
75 {
76  static uint64_t last_read = 0;
77  struct mag_report *mag_buf = reinterpret_cast<struct mag_report *>(buffer);
78 
79  /* buffer must be large enough */
80  unsigned count = buflen / sizeof(struct mag_report);
81 
82  if (count < 1) {
83  return -ENOSPC;
84  }
85 
86  if (last_read < _report.timestamp) {
87  /* copy report */
88  lock();
89  *mag_buf = _report;
90  last_read = _report.timestamp;
91  unlock();
92  return sizeof(struct mag_report);
93 
94  } else {
95  /* no new data available, warn caller */
96  return -EAGAIN;
97  }
98 }

139 void UavcanMagnetometerBridge::mag_sub_cb(const
140  uavcan::ReceivedDataStructure<uavcan::equipment::ahrs::MagneticFieldStrength>
141  &msg)
142 {
143  lock();
144  /*
145  * FIXME HACK
146  * This code used to rely on msg.getMonotonicTimestamp().toUSec() instead of HRT.
147  * It stopped working when the time sync feature has been introduced, because it caused libuavcan
148  * to use an independent time source (based on hardware TIM5) instead of HRT.
149  * The proper solution is to be developed.
150  */
151  _report.timestamp = hrt_absolute_time();
152  _report.device_id = _device_id.devid;
153  _report.is_external = true;
154 
155  _report.x = (msg.magnetic_field_ga[0] - _scale.x_offset) * _scale.x_scale;
156  _report.y = (msg.magnetic_field_ga[1] - _scale.y_offset) * _scale.y_scale;
157  _report.z = (msg.magnetic_field_ga[2] - _scale.z_offset) * _scale.z_scale;
158  unlock();
159 
160  publish(msg.getSrcNodeID().get(), &_report);
161 }
