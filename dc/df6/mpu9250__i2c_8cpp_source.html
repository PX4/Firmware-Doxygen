118 MPU9250_I2C::write(unsigned reg_speed, void *data, unsigned count)
119 {
120  uint8_t cmd[MPU_MAX_WRITE_BUFFER_SIZE];
121 
122  if (sizeof(cmd) < (count + 1)) {
123  return -EIO;
124  }
125 
126  cmd[0] = MPU9250_REG(reg_speed);
127  cmd[1] = *(uint8_t *)data;
128  return transfer(&cmd[0], count + 1, nullptr, 0);
129 }

131 int
132 MPU9250_I2C::read(unsigned reg_speed, void *data, unsigned count)
133 {
134  /* We want to avoid copying the data of MPUReport: So if the caller
135  * supplies a buffer not MPUReport in size, it is assume to be a reg or
136  * reg 16 read
137  * Since MPUReport has a cmd at front, we must return the data
138  * after that. Foe anthing else we must return it
139  */
140  uint32_t offset = count < sizeof(MPUReport) ? 0 : offsetof(MPUReport, status);
141  uint8_t cmd = MPU9250_REG(reg_speed);
142  return transfer(&cmd, 1, &((uint8_t *)data)[offset], count);
143 }
