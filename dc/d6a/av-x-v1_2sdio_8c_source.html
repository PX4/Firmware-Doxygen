118 int stm32_sdio_initialize(void)
119 {
120  int ret;
121 
122 #ifdef HAVE_NCD
123  /* Card detect */
124  bool cd_status;
125 
126  /* Configure the card detect GPIO */
127  stm32_configgpio(GPIO_SDMMC1_NCD);
128 
129  /* Register an interrupt handler for the card detect pin */
130  stm32_gpiosetevent(GPIO_SDMMC1_NCD, true, true, true, stm32_ncd_interrupt);
131 #endif
132 
133  /* Mount the SDIO-based MMC/SD block driver */
134  /* First, get an instance of the SDIO interface */
135  finfo("Initializing SDIO slot %d\n", SDIO_SLOTNO);
136 
137  sdio_dev = sdio_initialize(SDIO_SLOTNO);
138 
139  if (!sdio_dev) {
140  PX4_ERR("[boot] Failed to initialize SDIO slot %d\n", SDIO_SLOTNO);
141  return -ENODEV;
142  }
143 
144  /* Now bind the SDIO interface to the MMC/SD driver */
145 
146  finfo("Bind SDIO to the MMC/SD driver, minor=%d\n", SDIO_MINOR);
147 
148  ret = mmcsd_slotinitialize(SDIO_MINOR, sdio_dev);
149 
150  if (ret != OK) {
151  PX4_ERR("[boot] Failed to bind SDIO to the MMC/SD driver: %d\n", ret);
152  return ret;
153  }
154 
155  finfo("Successfully bound SDIO to the MMC/SD driver\n");
156 
157 #ifdef HAVE_NCD
158  /* Use SD card detect pin to check if a card is g_sd_inserted */
159 
160  cd_status = !stm32_gpioread(GPIO_SDMMC1_NCD);
161  finfo("Card detect : %d\n", cd_status);
162 
163  sdio_mediachange(sdio_dev, cd_status);
164 #else
165  /* Assume that the SD card is inserted. What choice do we have? */
166 
167  sdio_mediachange(sdio_dev, true);
168 #endif
169 
170  return OK;
171 }
172 
173 #endif /* CONFIG_MMCSD */
