48 /**
49  * Base publication wrapper class
50  */
51 template<typename T>
52 class Publication
53 {
54 public:
55 
56  /**
57  * Constructor
58  *
59  * @param meta The uORB metadata (usually from the ORB_ID() macro) for the topic.
60  */
61  Publication(const orb_metadata *meta) : _meta(meta) {}
62  ~Publication() { orb_unadvertise(_handle); }
63 
64  /**
65  * Publish the struct
66  * @param data The uORB message struct we are updating.
67  */
68  bool publish(const T &data)
69  {
70  if (_handle != nullptr) {
71  return (orb_publish(_meta, _handle, &data) == PX4_OK);
72 
73  } else {
74  orb_advert_t handle = orb_advertise(_meta, &data);
75 
76  if (handle != nullptr) {
77  _handle = handle;
78  return true;
79  }
80  }
81 
82  return false;
83  }
84 
85 protected:
86  const orb_metadata *_meta;
87 
88  orb_advert_t _handle{nullptr};
89 };
