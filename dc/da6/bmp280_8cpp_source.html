194 int
195 BMP280::init()
196 {
197  int ret = CDev::init();
198 
199  if (ret != OK) {
200  DEVICE_DEBUG("CDev init failed");
201  return ret;
202  }
203 
204  /* allocate basic report buffers */
205  _reports = new ringbuffer::RingBuffer(2, sizeof(baro_report));
206 
207  if (_reports == nullptr) {
208  DEVICE_DEBUG("can't get memory for reports");
209  ret = -ENOMEM;
210  return ret;
211  }
212 
213  /* register alternate interfaces if we have to */
214  _class_instance = register_class_devname(BARO_BASE_DEVICE_PATH);
215 
216  /* reset sensor */
217  _interface->set_reg(BPM280_VALUE_RESET, BPM280_ADDR_RESET);
218  usleep(10000);
219 
220  /* check id*/
221  if (_interface->get_reg(BPM280_ADDR_ID) != BPM280_VALUE_ID) {
222  PX4_WARN("id of your baro is not: 0x%02x", BPM280_VALUE_ID);
223  return -EIO;
224  }
225 
226  /* set config, recommended settings */
227  _curr_ctrl = BPM280_CTRL_P16 | BPM280_CTRL_T2;
228  _interface->set_reg(_curr_ctrl, BPM280_ADDR_CTRL);
229  _max_mesure_ticks = USEC2TICK(BPM280_MT_INIT + BPM280_MT * (16 - 1 + 2 - 1));
230  _interface->set_reg(BPM280_CONFIG_F16, BPM280_ADDR_CONFIG);
231 
232  /* get calibration and pre process them*/
233  _cal = _interface->get_calibration(BPM280_ADDR_CAL);
234 
235  _fcal.t1 = _cal->t1 * powf(2, 4);
236  _fcal.t2 = _cal->t2 * powf(2, -14);
237  _fcal.t3 = _cal->t3 * powf(2, -34);
238 
239  _fcal.p1 = _cal->p1 * (powf(2, 4) / -100000.0f);
240  _fcal.p2 = _cal->p1 * _cal->p2 * (powf(2, -31) / -100000.0f);
241  _fcal.p3 = _cal->p1 * _cal->p3 * (powf(2, -51) / -100000.0f);
242 
243  _fcal.p4 = _cal->p4 * powf(2, 4) - powf(2, 20);
244  _fcal.p5 = _cal->p5 * powf(2, -14);
245  _fcal.p6 = _cal->p6 * powf(2, -31);
246 
247  _fcal.p7 = _cal->p7 * powf(2, -4);
248  _fcal.p8 = _cal->p8 * powf(2, -19) + 1.0f;
249  _fcal.p9 = _cal->p9 * powf(2, -35);
250 
251  /* do a first measurement cycle to populate reports with valid data */
252  struct baro_report brp;
253  _reports->flush();
254 
255  if (measure()) {
256  return -EIO;
257  }
258 
259  usleep(TICK2USEC(_max_mesure_ticks));
260 
261  if (collect()) {
262  return -EIO;
263  }
264 
265  _reports->get(&brp);
266 
267  _baro_topic = orb_advertise_multi(ORB_ID(sensor_baro), &brp,
268  &_orb_class_instance, _interface->is_external() ? ORB_PRIO_HIGH : ORB_PRIO_DEFAULT);
269 
270  if (_baro_topic == nullptr) {
271  PX4_WARN("failed to create sensor_baro publication");
272  return -ENOMEM;
273  }
274 
275  return OK;
276 
277 }

279 ssize_t
280 BMP280::read(struct file *filp, char *buffer, size_t buflen)
281 {
282  unsigned count = buflen / sizeof(struct baro_report);
283  struct baro_report *brp = reinterpret_cast<struct baro_report *>(buffer);
284  int ret = 0;
285 
286  /* buffer must be large enough */
287  if (count < 1) {
288  return -ENOSPC;
289  }
290 
291  /* if automatic measurement is enabled */
292  if (_report_ticks > 0) {
293 
294  /*
295  * While there is space in the caller's buffer, and reports, copy them.
296  * Note that we may be pre-empted by the workq thread while we are doing this;
297  * we are careful to avoid racing with them.
298  */
299  while (count--) {
300  if (_reports->get(brp)) {
301  ret += sizeof(*brp);
302  brp++;
303  }
304  }
305 
306  /* if there was no data, warn the caller */
307  return ret ? ret : -EAGAIN;
308  }
309 
310  /* manual measurement - run one conversion */
311 
312  _reports->flush();
313 
314  if (measure()) {
315  return -EIO;
316  }
317 
318  usleep(TICK2USEC(_max_mesure_ticks));
319 
320  if (collect()) {
321  return -EIO;
322  }
323 
324  if (_reports->get(brp)) { //get new generated report
325  ret = sizeof(*brp);
326  }
327 
328  return ret;
329 }

331 int
332 BMP280::ioctl(struct file *filp, int cmd, unsigned long arg)
333 {
334  switch (cmd) {
335 
336  case SENSORIOCSPOLLRATE: {
337 
338  unsigned ticks = 0;
339 
340  switch (arg) {
341 
342  case SENSOR_POLLRATE_MANUAL:
343  stop_cycle();
344  _report_ticks = 0;
345  return OK;
346 
347  case SENSOR_POLLRATE_EXTERNAL:
348  case 0:
349  return -EINVAL;
350 
351  case SENSOR_POLLRATE_MAX:
352 
353  /* FALLTHROUGH */
354  case SENSOR_POLLRATE_DEFAULT:
355  ticks = _max_mesure_ticks;
356 
357  /* FALLTHROUGH */
358  default: {
359  if (ticks == 0) {
360  ticks = USEC2TICK(USEC_PER_SEC / arg);
361  }
362 
363  /* do we need to start internal polling? */
364  bool want_start = (_report_ticks == 0);
365 
366  /* check against maximum rate */
367  if (ticks < _max_mesure_ticks) {
368  return -EINVAL;
369  }
370 
371  _report_ticks = ticks;
372 
373  if (want_start) {
374  start_cycle();
375  }
376 
377  return OK;
378  }
379  }
380 
381  break;
382  }
383 
384  case SENSORIOCGPOLLRATE:
385  if (_report_ticks == 0) {
386  return SENSOR_POLLRATE_MANUAL;
387  }
388 
389  return (USEC_PER_SEC / USEC_PER_TICK / _report_ticks);
390 
391  case SENSORIOCSQUEUEDEPTH: {
392  /* lower bound is mandatory, upper bound is a sanity check */
393  if ((arg < 1) || (arg > 100)) {
394  return -EINVAL;
395  }
396 
397  irqstate_t flags = px4_enter_critical_section();
398 
399  if (!_reports->resize(arg)) {
400  px4_leave_critical_section(flags);
401  return -ENOMEM;
402  }
403 
404  px4_leave_critical_section(flags);
405  return OK;
406  }
407 
408  case SENSORIOCRESET:
409  /*
410  * Since we are initialized, we do not need to do anything, since the
411  * PROM is correctly read and the part does not need to be configured.
412  */
413  return OK;
414 
415  default:
416  break;
417  }
418 
419  return CDev::ioctl(filp, cmd, arg);
420 }

452 {

454  collect();

456 
457  if ((wait_gap != 0) && (_running)) {
458  work_queue(HPWORK, &_work, (worker_t)&BMP280::cycle_trampoline, this,

494 int
495 BMP280::collect()
496 {
497  _collect_phase = false;
498 
499  perf_begin(_sample_perf);
500 
501  struct baro_report report;
502  /* this should be fairly close to the end of the conversion, so the best approximation of the time */
503  report.timestamp = hrt_absolute_time();
504  report.error_count = perf_event_count(_comms_errors);
505 
506  bmp280::data_s *data = _interface->get_data(BPM280_ADDR_DATA);
507 
508  if (data == nullptr) {
509  perf_count(_comms_errors);
510  perf_cancel(_sample_perf);
511  return -EIO;
512  }
513 
514  //convert data to number 20 bit
515  uint32_t p_raw = data->p_msb << 12 | data->p_lsb << 4 | data->p_xlsb >> 4;
516  uint32_t t_raw = data->t_msb << 12 | data->t_lsb << 4 | data->t_xlsb >> 4;
517 
518  // Temperature
519  float ofs = (float) t_raw - _fcal.t1;
520  float t_fine = (ofs * _fcal.t3 + _fcal.t2) * ofs;
521  _T = t_fine * (1.0f / 5120.0f);
522 
523  // Pressure
524  float tf = t_fine - 128000.0f;
525  float x1 = (tf * _fcal.p6 + _fcal.p5) * tf + _fcal.p4;
526  float x2 = (tf * _fcal.p3 + _fcal.p2) * tf + _fcal.p1;
527 
528  float pf = ((float) p_raw + x1) / x2;
529  _P = (pf * _fcal.p9 + _fcal.p8) * pf + _fcal.p7;
530 
531 
532  report.temperature = _T;
533  report.pressure = _P / 100.0f; // to mbar
534 
535  /* publish it */
536  if (!(_pub_blocked)) {
537  /* publish it */
538  orb_publish(ORB_ID(sensor_baro), _baro_topic, &report);
539  }
540 
541  _reports->force(&report);
542 
543  /* notify anyone waiting for data */
544  poll_notify(POLLIN);
545 
546  perf_end(_sample_perf);
547 
548  return OK;
549 }

610 /**
611  * Start the driver.
612  */
613 bool
614 start_bus(struct bmp280_bus_option &bus)
615 {
616  if (bus.dev != nullptr) {
617  PX4_ERR("bus option already started");
618  exit(1);
619  }
620 
621  bmp280::IBMP280 *interface = bus.interface_constructor(bus.busnum, bus.device, bus.external);
622 
623  if (interface->init() != OK) {
624  delete interface;
625  PX4_WARN("no device on bus %u", (unsigned)bus.busid);
626  return false;
627  }
628 
629  bus.dev = new BMP280(interface, bus.devpath);
630 
631  if (bus.dev == nullptr) {
632  return false;
633  }
634 
635  if (OK != bus.dev->init()) {
636  delete bus.dev;
637  bus.dev = nullptr;
638  return false;
639  }
640 
641  int fd = open(bus.devpath, O_RDONLY);
642 
643  /* set the poll rate to default, starts automatic data collection */
644  if (fd == -1) {
645  PX4_ERR("can't open baro device");
646  exit(1);
647  }
648 
649  if (ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
650  close(fd);
651  PX4_ERR("failed setting default poll rate");
652  exit(1);
653  }
654 
655  close(fd);
656  return true;
657 }

660 /**
661  * Start the driver.
662  *
663  * This function call only returns once the driver
664  * is either successfully up and running or failed to start.
665  */
666 void
667 start(enum BMP280_BUS busid)
668 {
669  uint8_t i;
670  bool started = false;
671 
672  for (i = 0; i < NUM_BUS_OPTIONS; i++) {
673  if (busid == BMP280_BUS_ALL && bus_options[i].dev != NULL) {
674  // this device is already started
675  continue;
676  }
677 
678  if (busid != BMP280_BUS_ALL && bus_options[i].busid != busid) {
679  // not the one that is asked for
680  continue;
681  }
682 
683  started |= start_bus(bus_options[i]);
684  }
685 
686  if (!started) {
687  PX4_WARN("bus option number is %d", i);
688  PX4_ERR("driver start failed");
689  exit(1);
690  }
691 
692  // one or more drivers started OK
693  exit(0);
694 }

739  PX4_ERR("immediate read failed");
740  exit(1);
741  }
742 
743  print_message(report);
744

746  if (OK != ioctl(fd, SENSORIOCSQUEUEDEPTH, 10)) {
747  PX4_ERR("failed to set queue depth");
748  exit(1);
749  }
750 
751  /* start the sensor polling at 2Hz */
752  if (OK != ioctl(fd, SENSORIOCSPOLLRATE, 2)) {
753  PX4_ERR("failed to set 2Hz poll rate");
754  exit(1);
755  }
756 
757  /* read the sensor 5x and report each value */
758  for (unsigned i = 0; i < 5; i++) {
759  struct pollfd fds;
760 
761  /* wait for data to be ready */
762  fds.fd = fd;
763  fds.events = POLLIN;
764  ret = poll(&fds, 1, 2000);
765 
766  if (ret != 1) {
767  PX4_ERR("timed out waiting for sensor data");
768  exit(1);
769  }
770 
771  /* now go get it */
772  sz = read(fd, &report, sizeof(report));
773 
774  if (sz != sizeof(report)) {
775  PX4_ERR("periodic read failed");
776  exit(1);
777  }
778 
779  print_message(report);
780  }
781 
782  close(fd);
