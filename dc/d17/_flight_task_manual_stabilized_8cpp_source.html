69 void FlightTaskManualStabilized::_updateHeadingSetpoints()
70 {
71  /* Yaw-lock depends on stick input. If not locked,
72  * yaw_sp is set to NAN.
73  * TODO: add yawspeed to get threshold.*/
74  if (fabsf(_yawspeed_setpoint) > FLT_EPSILON) {
75  // no fixed heading when rotating around yaw by stick
76  _yaw_setpoint = NAN;
77 
78  } else {
79  // hold the current heading when no more rotation commanded
80  if (!PX4_ISFINITE(_yaw_setpoint)) {
81  _yaw_setpoint = _yaw;
82 
83  } else {
84  // check reset counter and update yaw setpoint if necessary
85  if (_sub_attitude->get().quat_reset_counter != _heading_reset_counter) {
86  _yaw_setpoint += matrix::Eulerf(matrix::Quatf(_sub_attitude->get().delta_q_reset)).psi();
87  _heading_reset_counter = _sub_attitude->get().quat_reset_counter;
88  }
89  }
90  }

93 void FlightTaskManualStabilized::_updateThrustSetpoints()
94 {
95  /* Rotate setpoint into local frame. */
96  Vector2f sp{_sticks(0), _sticks(1)};
97  _rotateIntoHeadingFrame(sp);
98 
99  /* Ensure that maximum tilt is in [0.001, Pi] */
100  float tilt_max = math::constrain(_constraints.tilt, 0.001f, M_PI_F);
101 
102  const float x = sp(0) * tilt_max;
103  const float y = sp(1) * tilt_max;
104 
105  /* The norm of the xy stick input provides the pointing
106  * direction of the horizontal desired thrust setpoint. The magnitude of the
107  * xy stick inputs represents the desired tilt. Both tilt and magnitude can
108  * be captured through Axis-Angle.
109  */
110  /* The Axis-Angle is the perpendicular vector to xy-stick input */
111  Vector2f v = Vector2f(y, -x);
112  float v_norm = v.norm(); // the norm of v defines the tilt angle
113 
114  if (v_norm > tilt_max) { // limit to the configured maximum tilt angle
115  v *= tilt_max / v_norm;
116  }
117 
118  /* The final thrust setpoint is found by rotating the scaled unit vector pointing
119  * upward by the Axis-Angle.
120  */
121  Quatf q_sp = AxisAnglef(v(0), v(1), 0.0f);
122  _thrust_setpoint = q_sp.conjugate(Vector3f(0.0f, 0.0f, -1.0f)) * _throttle;
123 }

125 void FlightTaskManualStabilized::_rotateIntoHeadingFrame(Vector2f &v)
126 {
127  float yaw_rotate = PX4_ISFINITE(_yaw_setpoint) ? _yaw_setpoint : _yaw;
128  Vector3f v_r = Vector3f(Dcmf(Eulerf(0.0f, 0.0f, yaw_rotate)) * Vector3f(v(0), v(1), 0.0f));
129  v(0) = v_r(0);
130  v(1) = v_r(1);
131 }

139 float FlightTaskManualStabilized::_throttleCurve()
140 {
141  /* Scale stick z from [-1,1] to [min thrust, max thrust]
142  * with hover throttle at 0.5 stick */
143  float throttle = -((_sticks(2) - 1.0f) * 0.5f);
144 
145  if (throttle < 0.5f) {
146  return (_throttle_hover.get() - _throttle_min_stabilized.get()) / 0.5f * throttle + _throttle_min_stabilized.get();
147 
148  } else {
149  return (_throttle_max.get() - _throttle_hover.get()) / 0.5f * (throttle - 1.0f) + _throttle_max.get();
150  }
151 }
