69  if (_main_task != -1) {

122 void
123 CameraFeedback::task_main()
124 {
125 
126  // We only support trigger feedback for now
127  // This will later be extended to support hardware feedback from the camera.
128  if (_camera_feedback_mode != CAMERA_FEEDBACK_MODE_TRIGGER) {
129  return;
130  }
131 
132  // Polling sources
133  _trigger_sub = orb_subscribe(ORB_ID(camera_trigger));
134  struct camera_trigger_s trig = {};
135 
136  px4_pollfd_struct_t fds[1] = {};
137  fds[0].fd = _trigger_sub;
138  fds[0].events = POLLIN;
139 
140  // Geotagging subscriptions
141  _gpos_sub = orb_subscribe(ORB_ID(vehicle_global_position));
142  _att_sub = orb_subscribe(ORB_ID(vehicle_attitude));
143  struct vehicle_global_position_s gpos = {};
144  struct vehicle_attitude_s att = {};
145 
146  bool updated = false;
147 
148  while (!_task_should_exit) {
149 
150  /* wait for up to 20ms for data */
151  int pret = px4_poll(&fds[0], (sizeof(fds) / sizeof(fds[0])), 20);
152 
153  if (pret < 0) {
154  PX4_WARN("poll error %d, %d", pret, errno);
155  continue;
156  }
157 
158  /* trigger subscription updated */
159  if (fds[0].revents & POLLIN) {
160 
161  orb_copy(ORB_ID(camera_trigger), _trigger_sub, &trig);
162 
163  /* update geotagging subscriptions */
164  orb_check(_gpos_sub, &updated);
165 
166  if (updated) {
167  orb_copy(ORB_ID(vehicle_global_position), _gpos_sub, &gpos);
168  }
169 
170  orb_check(_att_sub, &updated);
171 
172  if (updated) {
173  orb_copy(ORB_ID(vehicle_attitude), _att_sub, &att);
174  }
175 
176  if (trig.timestamp == 0 ||
177  gpos.timestamp == 0 ||
178  att.timestamp == 0) {
179  // reject until we have valid data
180  continue;
181  }
182 
183  struct camera_capture_s capture = {};
184 
185  // Fill timestamps
186  capture.timestamp = trig.timestamp;
187 
188  capture.timestamp_utc = trig.timestamp_utc;
189 
190  // Fill image sequence
191  capture.seq = trig.seq;
192 
193  // Fill position data
194  capture.lat = gpos.lat;
195 
196  capture.lon = gpos.lon;
197 
198  capture.alt = gpos.alt;
199 
200  capture.ground_distance = gpos.terrain_alt_valid ? (gpos.alt - gpos.terrain_alt) : -1.0f;
201 
202  // Fill attitude data
203  // TODO : this needs to be rotated by camera orientation or set to gimbal orientation when available
204  capture.q[0] = att.q[0];
205 
206  capture.q[1] = att.q[1];
207 
208  capture.q[2] = att.q[2];
209 
210  capture.q[3] = att.q[3];
211 
212  // Indicate that no capture feedback from camera is available
213  capture.result = -1;
214 
215  int instance_id;
216 
217  orb_publish_auto(ORB_ID(camera_capture), &_capture_pub, &capture, &instance_id, ORB_PRIO_DEFAULT);
218 
219  }
220 
221  }
222 
223  PX4_INFO("Exiting.");
224  _main_task = -1;
225 
226 }

265

267  PX4_WARN("not running");

271  camera_feedback::g_camera_feedback->stop();
