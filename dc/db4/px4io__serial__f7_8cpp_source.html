137  modifyreg32(PX4IO_SERIAL_RCC_REG, 0, PX4IO_SERIAL_RCC_EN);

140  px4_arch_configgpio(PX4IO_SERIAL_TX_GPIO);
141  px4_arch_configgpio(PX4IO_SERIAL_RX_GPIO);
142 
143  /* reset & configure the UART */
144  rCR1 = 0;
145  rCR2 = 0;
146  rCR3 = 0;
147 
148  /* clear data that may be in the RDR and clear overrun error: */
149  if (rISR & USART_ISR_RXNE) {
150  (void)rRDR;
151  }
152 
153  rICR = rISR & rISR_ERR_FLAGS_MASK; /* clear the flags */
154

161  up_enable_irq(PX4IO_SERIAL_VECTOR);

164  rCR3 = USART_CR3_EIE;
165  /* TODO: maybe use DDRE */

247 int
248 PX4IO_serial_f7::_bus_exchange(IOPacket *_packet)
249 {
250  _current_packet = _packet;
251 
252  /* clear data that may be in the RDR and clear overrun error: */
253  if (rISR & USART_ISR_RXNE) {
254  (void)rRDR;
255  }
256 
257  rICR = rISR & rISR_ERR_FLAGS_MASK; /* clear the flags */
258 
259  /* start RX DMA */
260  perf_begin(_pc_txns);
261  perf_begin(_pc_dmasetup);
262 
263  /* DMA setup time ~3µs */
264  _rx_dma_status = _dma_status_waiting;
265 
266  /*
267  * Note that we enable circular buffer mode as a workaround for
268  * there being no API to disable the DMA FIFO. We need direct mode
269  * because otherwise when the line idle interrupt fires there
270  * will be packet bytes still in the DMA FIFO, and we will assume
271  * that the idle was spurious.
272  *
273  * XXX this should be fixed with a NuttX change.
274  */
275  stm32_dmasetup(
276  _rx_dma,
277  PX4IO_SERIAL_BASE + STM32_USART_RDR_OFFSET,
278  reinterpret_cast<uint32_t>(_current_packet),
279  sizeof(*_current_packet),
280  DMA_SCR_CIRC | /* XXX see note above */
281  DMA_SCR_DIR_P2M |
282  DMA_SCR_MINC |
283  DMA_SCR_PSIZE_8BITS |
284  DMA_SCR_MSIZE_8BITS |
285  DMA_SCR_PBURST_SINGLE |
286  DMA_SCR_MBURST_SINGLE);
287  rCR3 |= USART_CR3_DMAR;
288  stm32_dmastart(_rx_dma, _dma_callback, this, false);
289 
290  /* Clean _current_packet, so DMA can see the data */
291  arch_clean_dcache((uintptr_t)_current_packet,
292  (uintptr_t)_current_packet + ALIGNED_IO_BUFFER_SIZE);
293 
294  /* start TX DMA - no callback if we also expect a reply */
295  /* DMA setup time ~3µs */
296  stm32_dmasetup(
297  _tx_dma,
298  PX4IO_SERIAL_BASE + STM32_USART_TDR_OFFSET,
299  reinterpret_cast<uint32_t>(_current_packet),
300  PKT_SIZE(*_current_packet),
301  DMA_SCR_DIR_M2P |
302  DMA_SCR_MINC |
303  DMA_SCR_PSIZE_8BITS |
304  DMA_SCR_MSIZE_8BITS |
305  DMA_SCR_PBURST_SINGLE |
306  DMA_SCR_MBURST_SINGLE);
307  rCR3 |= USART_CR3_DMAT;
308  stm32_dmastart(_tx_dma, nullptr, nullptr, false);
309  //rCR1 &= ~USART_CR1_TE;
310  //rCR1 |= USART_CR1_TE;
311 
312  perf_end(_pc_dmasetup);
313 
314  /* compute the deadline for a 10ms timeout */
315  struct timespec abstime;
316  clock_gettime(CLOCK_REALTIME, &abstime);
317  abstime.tv_nsec += 10 * 1000 * 1000;
318 
319  if (abstime.tv_nsec >= 1000 * 1000 * 1000) {
320  abstime.tv_sec++;
321  abstime.tv_nsec -= 1000 * 1000 * 1000;
322  }
323 
324  /* wait for the transaction to complete - 64 bytes @ 1.5Mbps ~426µs */
325  int ret;
326 
327  for (;;) {
328  ret = sem_timedwait(&_completion_semaphore, &abstime);
329 
330  if (ret == OK) {
331  /* check for DMA errors */
332  if (_rx_dma_status & DMA_STATUS_TEIF) {
333  perf_count(_pc_dmaerrs);
334  ret = -EIO;
335  break;
336  }
337 
338  /* check packet CRC - corrupt packet errors mean IO receive CRC error */
339  uint8_t crc = _current_packet->crc;
340  _current_packet->crc = 0;
341 
342  if ((crc != crc_packet(_current_packet)) || (PKT_CODE(*_current_packet) == PKT_CODE_CORRUPT)) {
343  perf_count(_pc_crcerrs);
344  ret = -EIO;
345  break;
346  }
347 
348  /* successful txn (may still be reporting an error) */
349  break;
350  }
351 
352  if (errno == ETIMEDOUT) {
353  /* something has broken - clear out any partial DMA state and reconfigure */
354  _abort_dma();
355  perf_count(_pc_timeouts);
356  perf_cancel(_pc_txns); /* don't count this as a transaction */
357  break;
358  }
359 
360  /* we might? see this for EINTR */
361  syslog(LOG_ERR, "unexpected ret %d/%d\n", ret, errno);
362  }
363 
364  /* reset DMA status */
365  _rx_dma_status = _dma_status_inactive;
366 
367  /* update counters */
368  perf_end(_pc_txns);
369 
370  return ret;
371 }

383 void
384 PX4IO_serial_f7::_do_rx_dma_callback(unsigned status)
385 {
386  /* on completion of a reply, wake the waiter */
387  if (_rx_dma_status == _dma_status_waiting) {
388 
389  /* check for packet overrun - this will occur after DMA completes */
390  uint32_t sr = rISR;
391 
392  if (sr & (USART_ISR_ORE | USART_ISR_RXNE)) {
393  (void)rRDR;
394  rICR = sr & (USART_ISR_ORE | USART_ISR_RXNE);
395  status = DMA_STATUS_TEIF;
396  }
397 
398  /* save RX status */
399  _rx_dma_status = status;
400 
401  /* disable UART DMA */
402  rCR3 &= ~(USART_CR3_DMAT | USART_CR3_DMAR);
403 
404  /* complete now */
405  px4_sem_post(&_completion_semaphore);
406  }

421 void
422 PX4IO_serial_f7::_do_interrupt()
423 {
424  uint32_t sr = rISR; /* get UART status register */
425 
426  if (sr & USART_ISR_RXNE) {
427  (void)rRDR; /* read DR to clear RXNE */
428  }
429 
430  rICR = sr & rISR_ERR_FLAGS_MASK; /* clear flags */
431 
432  if (sr & (USART_ISR_ORE | /* overrun error - packet was too big for DMA or DMA was too slow */
433  USART_ISR_NF | /* noise error - we have lost a byte due to noise */
434  USART_ISR_FE)) { /* framing error - start/stop bit lost or line break */
435 
436  /*
437  * If we are in the process of listening for something, these are all fatal;
438  * abort the DMA with an error.
439  */
440  if (_rx_dma_status == _dma_status_waiting) {
441  _abort_dma();
442 
443  perf_count(_pc_uerrs);
444  /* complete DMA as though in error */
445  _do_rx_dma_callback(DMA_STATUS_TEIF);
446 
447  return;
448  }
449 
450  /* XXX we might want to use FE / line break as an out-of-band handshake ... handle it here */
451 
452  /* don't attempt to handle IDLE if it's set - things went bad */
453  return;
454  }
455 
456  if (sr & USART_ISR_IDLE) {
457 
458  /* if there is DMA reception going on, this is a short packet */
459  if (_rx_dma_status == _dma_status_waiting) {
460  /* Invalidate _current_packet, so we get fresh data from RAM */
461  arch_invalidate_dcache((uintptr_t)_current_packet,
462  (uintptr_t)_current_packet + ALIGNED_IO_BUFFER_SIZE);
463 
464  /* verify that the received packet is complete */
465  size_t length = sizeof(*_current_packet) - stm32_dmaresidual(_rx_dma);
466 
467  if ((length < 1) || (length < PKT_SIZE(*_current_packet))) {
468  perf_count(_pc_badidle);
469 
470  /* stop the receive DMA */
471  stm32_dmastop(_rx_dma);
472 
473  /* complete the short reception */
474  _do_rx_dma_callback(DMA_STATUS_TEIF);
475  return;
476  }
477 
478  perf_count(_pc_idle);
479 
480  /* stop the receive DMA */
481  stm32_dmastop(_rx_dma);
482 
483  /* complete the short reception */
484  _do_rx_dma_callback(DMA_STATUS_TCIF);
485  }
486  }
