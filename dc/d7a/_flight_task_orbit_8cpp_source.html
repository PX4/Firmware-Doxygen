89 bool FlightTaskOrbit::setRadius(const float r)
90 {
91  if (math::isInRange(r, _radius_min, _radius_max)) {
92  // small radius is more important than high velocity for safety
93  if (!checkAcceleration(r, _v, _acceleration_max)) {
94  _v = math::sign(_v) * sqrtf(_acceleration_max * r);

139 bool FlightTaskOrbit::update()
140 {
141  // update altitude
142  FlightTaskManualAltitudeSmooth::update();
143 
144  // stick input adjusts parameters within a fixed time frame
145  const float r = _r - _sticks_expo(0) * _deltatime * (_radius_max / 8.f);
146  const float v = _v - _sticks_expo(1) * _deltatime * (_velocity_max / 4.f);
147 
148  setRadius(r);
149  setVelocity(v);
150 
151  // xy velocity to go around in a circle
152  Vector2f center_to_position = Vector2f(_position.data()) - _center;
153  Vector2f velocity_xy = Vector2f(-center_to_position(1), center_to_position(0));
154  velocity_xy = velocity_xy.unit_or_zero();
155  velocity_xy *= _v;
156 
157  // xy velocity adjustment to stay on the radius distance
158  velocity_xy += (_r - center_to_position.norm()) * center_to_position.unit_or_zero();
159 
160  _velocity_setpoint(0) = velocity_xy(0);
161  _velocity_setpoint(1) = velocity_xy(1);
162 
163  // make vehicle front always point towards the center
164  _yaw_setpoint = atan2f(center_to_position(1), center_to_position(0)) + M_PI_F;
165  // yawspeed feed-forward because we know the necessary angular rate
166  _yawspeed_setpoint = _v / _r;
167 
168  return true;
169 }
