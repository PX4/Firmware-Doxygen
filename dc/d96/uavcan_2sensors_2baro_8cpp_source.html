76 ssize_t UavcanBarometerBridge::read(struct file *filp, char *buffer, size_t buflen)
77 {
78  unsigned count = buflen / sizeof(struct baro_report);
79  struct baro_report *baro_buf = reinterpret_cast<struct baro_report *>(buffer);
80  int ret = 0;
81 
82  /* buffer must be large enough */
83  if (count < 1) {
84  return -ENOSPC;
85  }
86 
87  while (count--) {
88  if (_reports.get(baro_buf)) {
89  ret += sizeof(*baro_buf);
90  baro_buf++;
91  }
92  }
93 
94  /* if there was no data, warn the caller */
95  return ret ? ret : -EAGAIN;
96 }

98 int UavcanBarometerBridge::ioctl(struct file *filp, int cmd, unsigned long arg)
99 {
100  switch (cmd) {
101  case SENSORIOCSPOLLRATE: {
102  // not supported yet, pretend that everything is ok
103  return OK;
104  }
105 
106  case SENSORIOCSQUEUEDEPTH: {
107  /* lower bound is mandatory, upper bound is a sanity check */
108  if ((arg < 1) || (arg > 100)) {
109  return -EINVAL;
110  }
111 
112  irqstate_t flags = px4_enter_critical_section();
113 
114  if (!_reports.resize(arg)) {
115  px4_leave_critical_section(flags);
116  return -ENOMEM;
117  }
118 
119  px4_leave_critical_section(flags);
120 
121  return OK;
122  }
123 
124  default: {
125  return CDev::ioctl(filp, cmd, arg);
126  }
127  }
128 }

136 void UavcanBarometerBridge::air_pressure_sub_cb(const
137  uavcan::ReceivedDataStructure<uavcan::equipment::air_data::StaticPressure> &msg)
138 {
139  baro_report report;
140 
141  /*
142  * FIXME HACK
143  * This code used to rely on msg.getMonotonicTimestamp().toUSec() instead of HRT.
144  * It stopped working when the time sync feature has been introduced, because it caused libuavcan
145  * to use an independent time source (based on hardware TIM5) instead of HRT.
146  * The proper solution is to be developed.
147  */
148  report.timestamp = hrt_absolute_time();
149  report.temperature = last_temperature_kelvin - 273.15F;
150  report.pressure = msg.static_pressure / 100.0F; // Convert to millibar
151  report.error_count = 0;
152 
153  /* TODO get device ID for sensor */
154  report.device_id = 0;
155 
156  // add to the ring buffer
157  _reports.force(&report);
158 
159  publish(msg.getSrcNodeID().get(), &report);
160 }
