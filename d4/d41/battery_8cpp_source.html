144 void
145 Battery::sumDischarged(hrt_abstime timestamp, float current_a)
146 {
147  // Not a valid measurement
148  if (current_a < 0.f) {
149  // Because the measurement was invalid we need to stop integration
150  // and re-initialize with the next valid measurement
151  _last_timestamp = 0;
152  return;
153  }
154 
155  // Ignore first update because we don't know dt.
156  if (_last_timestamp != 0) {
157  const float dt = (timestamp - _last_timestamp) / 1e6;
158  // mAh since last loop: (current[A] * 1000 = [mA]) * (dt[s] / 3600 = [h])
159  _discharged_mah_loop = (current_a * 1e3f) * (dt / 3600.f);
160  _discharged_mah += _discharged_mah_loop;
161  }
162 
163  _last_timestamp = timestamp;
164 }

166 void
167 Battery::estimateRemaining(float voltage_v, float current_a, float throttle, bool armed)
168 {
169  // remaining battery capacity based on voltage
170  float cell_voltage = voltage_v / _n_cells.get();
171 
172  // correct battery voltage locally for load drop to avoid estimation fluctuations
173  if (_r_internal.get() >= 0.f) {
174  cell_voltage += _r_internal.get() * current_a;
175 
176  } else {
177  // assume linear relation between throttle and voltage drop
178  cell_voltage += throttle * _v_load_drop.get();
179  }
180 
181  _remaining_voltage = math::gradual(cell_voltage, _v_empty.get(), _v_charged.get(), 0.f, 1.f);
182 
183  // choose which quantity we're using for final reporting
184  if (_capacity.get() > 0.f) {
185  // if battery capacity is known, fuse voltage measurement with used capacity
186  if (!_battery_initialized) {
187  // initialization of the estimation state
188  _remaining = _remaining_voltage;
189 
190  } else {
191  // The lower the voltage the more adjust the estimate with it to avoid deep discharge
192  const float weight_v = 3e-4f * (1 - _remaining_voltage);
193  _remaining = (1 - weight_v) * _remaining + weight_v * _remaining_voltage;
194  // directly apply current capacity slope calculated using current
195  _remaining -= _discharged_mah_loop / _capacity.get();
196  _remaining = math::max(_remaining, 0.f);
197  }
198 
199  } else {
200  // else use voltage
201  _remaining = _remaining_voltage;
202  }
203 }

206 Battery::determineWarning(bool connected)
207 {
208  if (connected) {
209  // propagate warning state only if the state is higher, otherwise remain in current warning state
210  if (_remaining < _emergency_thr.get() || (_warning == battery_status_s::BATTERY_WARNING_EMERGENCY)) {
211  _warning = battery_status_s::BATTERY_WARNING_EMERGENCY;
212 
213  } else if (_remaining < _crit_thr.get() || (_warning == battery_status_s::BATTERY_WARNING_CRITICAL)) {
214  _warning = battery_status_s::BATTERY_WARNING_CRITICAL;
215 
216  } else if (_remaining < _low_thr.get() || (_warning == battery_status_s::BATTERY_WARNING_LOW)) {
217  _warning = battery_status_s::BATTERY_WARNING_LOW;
218  }
219  }

222 void
223 Battery::computeScale()
224 {
225  const float voltage_range = (_v_charged.get() - _v_empty.get());
226 
227  // reusing capacity calculation to get single cell voltage before drop
228  const float bat_v = _v_empty.get() + (voltage_range * _remaining_voltage);
229 
230  _scale = _v_charged.get() / bat_v;
231 
232  if (_scale > 1.3f) { // Allow at most 30% compensation
233  _scale = 1.3f;
234 
235  } else if (!PX4_ISFINITE(_scale) || _scale < 1.f) { // Shouldn't ever be more than the power at full battery
236  _scale = 1.f;
237  }
