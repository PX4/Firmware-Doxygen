95 void Tiltrotor::update_vtol_state()
96 {
97 
98  /* simple logic using a two way switch to perform transitions.
99  * after flipping the switch the vehicle will start tilting rotors, picking up
100  * forward speed. After the vehicle has picked up enough speed the rotors are tilted
101  * forward completely. For the backtransition the motors simply rotate back.
102  */
103 
104  if (!_attc->is_fixed_wing_requested()) {
105 
106  // plane is in multicopter mode
107  switch (_vtol_schedule.flight_mode) {
108  case MC_MODE:
109  break;
110 
111  case FW_MODE:
112  _vtol_schedule.flight_mode = TRANSITION_BACK;
113  _vtol_schedule.transition_start = hrt_absolute_time();
114  break;
115 
116  case TRANSITION_FRONT_P1:
117  // failsafe into multicopter mode
118  _vtol_schedule.flight_mode = MC_MODE;
119  break;
120 
121  case TRANSITION_FRONT_P2:
122  // failsafe into multicopter mode
123  _vtol_schedule.flight_mode = MC_MODE;
124  break;
125 
126  case TRANSITION_BACK:
127  if (_tilt_control <= _params_tiltrotor.tilt_mc) {
128  _vtol_schedule.flight_mode = MC_MODE;
129  }
130 
131  break;
132  }
133 
134  } else {
135 
136  switch (_vtol_schedule.flight_mode) {
137  case MC_MODE:
138  // initialise a front transition
139  _vtol_schedule.flight_mode = TRANSITION_FRONT_P1;
140  _vtol_schedule.transition_start = hrt_absolute_time();
141  break;
142 
143  case FW_MODE:
144  break;
145 
146  case TRANSITION_FRONT_P1: {
147  // allow switch if we are not armed for the sake of bench testing
148  bool transition_to_p2 = can_transition_on_ground();
149 
150  float time_since_trans_start = (float)(hrt_absolute_time() - _vtol_schedule.transition_start) * 1e-6f;
151 
152  // check if we have reached airspeed to switch to fw mode
153  transition_to_p2 |= !_params->airspeed_disabled &&
154  _airspeed->indicated_airspeed_m_s >= _params->transition_airspeed &&
155  time_since_trans_start > _params->front_trans_time_min;
156 
157  // check if airspeed is invalid and transition by time
158  transition_to_p2 |= _params->airspeed_disabled &&
159  _tilt_control > _params_tiltrotor.tilt_transition &&
160  time_since_trans_start > _params->front_trans_time_openloop;
161 
162  if (transition_to_p2) {
163  _vtol_schedule.flight_mode = TRANSITION_FRONT_P2;
164  _vtol_schedule.transition_start = hrt_absolute_time();
165  }
166 
167  break;
168  }
169 
170  case TRANSITION_FRONT_P2:
171 
172  // if the rotors have been tilted completely we switch to fw mode
173  if (_tilt_control >= _params_tiltrotor.tilt_fw) {
174  _vtol_schedule.flight_mode = FW_MODE;
175  _tilt_control = _params_tiltrotor.tilt_fw;
176  }
177 
178  break;
179 
180  case TRANSITION_BACK:
181  // failsafe into fixed wing mode
182  _vtol_schedule.flight_mode = FW_MODE;
183  break;
184  }
185  }
186 
187  // map tiltrotor specific control phases to simple control modes
188  switch (_vtol_schedule.flight_mode) {
189  case MC_MODE:
190  _vtol_mode = ROTARY_WING;
191  break;
192 
193  case FW_MODE:
194  _vtol_mode = FIXED_WING;
195  break;
196 
197  case TRANSITION_FRONT_P1:
198  case TRANSITION_FRONT_P2:
199  _vtol_mode = TRANSITION_TO_FW;
200  break;
201 
202  case TRANSITION_BACK:
203  _vtol_mode = TRANSITION_TO_MC;
204  break;
205  }
206 }

224 void Tiltrotor::update_transition_state()
225 {
226  VtolType::update_transition_state();
227 
228  float time_since_trans_start = (float)(hrt_absolute_time() - _vtol_schedule.transition_start) * 1e-6f;
229 
230  if (!_flag_was_in_trans_mode) {
231  // save desired heading for transition and last thrust value
232  _flag_was_in_trans_mode = true;
233  }
234 
235  if (_vtol_schedule.flight_mode == TRANSITION_FRONT_P1) {
236  // for the first part of the transition the rear rotors are enabled
237  if (_motor_state != ENABLED) {
238  _motor_state = set_motor_state(_motor_state, ENABLED);
239  }
240 
241  // tilt rotors forward up to certain angle
242  if (_tilt_control <= _params_tiltrotor.tilt_transition) {
243  _tilt_control = _params_tiltrotor.tilt_mc +
244  fabsf(_params_tiltrotor.tilt_transition - _params_tiltrotor.tilt_mc) * time_since_trans_start /
245  _params->front_trans_duration;
246  }
247 
248 
249  // at low speeds give full weight to MC
250  _mc_roll_weight = 1.0f;
251  _mc_yaw_weight = 1.0f;
252 
253  // reduce MC controls once the plane has picked up speed
254  if (!_params->airspeed_disabled && _airspeed->indicated_airspeed_m_s > ARSP_YAW_CTRL_DISABLE) {
255  _mc_yaw_weight = 0.0f;
256  }
257 
258  if (!_params->airspeed_disabled && _airspeed->indicated_airspeed_m_s >= _params->airspeed_blend) {
259  _mc_roll_weight = 1.0f - (_airspeed->indicated_airspeed_m_s - _params->airspeed_blend) /
260  (_params->transition_airspeed - _params->airspeed_blend);
261  }
262 
263  // without airspeed do timed weight changes
264  if (_params->airspeed_disabled
265  && time_since_trans_start > _params->front_trans_time_min) {
266  _mc_roll_weight = 1.0f - (time_since_trans_start - _params->front_trans_time_min) /
267  (_params->front_trans_time_openloop - _params->front_trans_time_min);
268  _mc_yaw_weight = _mc_roll_weight;
269  }
270 
271  _thrust_transition = _mc_virtual_att_sp->thrust;
272 
273  } else if (_vtol_schedule.flight_mode == TRANSITION_FRONT_P2) {
274  // the plane is ready to go into fixed wing mode, tilt the rotors forward completely
275  _tilt_control = _params_tiltrotor.tilt_transition +
276  fabsf(_params_tiltrotor.tilt_fw - _params_tiltrotor.tilt_transition) * time_since_trans_start /
277  _params_tiltrotor.front_trans_dur_p2;
278 
279  _mc_roll_weight = 0.0f;
280  _mc_yaw_weight = 0.0f;
281 
282  // ramp down rear motors (setting MAX_PWM down scales the given output into the new range)
283  int rear_value = (1.0f - time_since_trans_start / _params_tiltrotor.front_trans_dur_p2) *
284  (PWM_DEFAULT_MAX - PWM_DEFAULT_MIN) + PWM_DEFAULT_MIN;
285 
286 
287  _motor_state = set_motor_state(_motor_state, VALUE, rear_value);
288 
289 
290  _thrust_transition = _mc_virtual_att_sp->thrust;
291 
292  } else if (_vtol_schedule.flight_mode == TRANSITION_BACK) {
293  if (_motor_state != ENABLED) {
294  _motor_state = set_motor_state(_motor_state, ENABLED);
295  }
296 
297 
298  // set idle speed for rotary wing mode
299  if (!flag_idle_mc) {
300  flag_idle_mc = set_idle_mc();
301  }
302 
303  // tilt rotors back
304  if (_tilt_control > _params_tiltrotor.tilt_mc) {
305  _tilt_control = _params_tiltrotor.tilt_fw -
306  fabsf(_params_tiltrotor.tilt_fw - _params_tiltrotor.tilt_mc) * time_since_trans_start / _params->back_trans_duration;
307  }
308 
309  // set zero throttle for backtransition otherwise unwanted moments will be created
310  _actuators_mc_in->control[actuator_controls_s::INDEX_THROTTLE] = 0.0f;
311 
312  _mc_roll_weight = 0.0f;
313 
314  }
315 
316  _mc_roll_weight = math::constrain(_mc_roll_weight, 0.0f, 1.0f);
317  _mc_yaw_weight = math::constrain(_mc_yaw_weight, 0.0f, 1.0f);
318 
319  // copy virtual attitude setpoint to real attitude setpoint (we use multicopter att sp)
320  memcpy(_v_att_sp, _mc_virtual_att_sp, sizeof(vehicle_attitude_setpoint_s));
321 }

330 * Write data to actuator output topic.
331 */
332 void Tiltrotor::fill_actuator_outputs()
333 {
334  _actuators_out_0->timestamp = hrt_absolute_time();
335  _actuators_out_0->timestamp_sample = _actuators_mc_in->timestamp_sample;
336 
337  _actuators_out_0->control[actuator_controls_s::INDEX_ROLL] = _actuators_mc_in->control[actuator_controls_s::INDEX_ROLL]
338  * _mc_roll_weight;
339  _actuators_out_0->control[actuator_controls_s::INDEX_PITCH] =
340  _actuators_mc_in->control[actuator_controls_s::INDEX_PITCH] * _mc_pitch_weight;
341  _actuators_out_0->control[actuator_controls_s::INDEX_YAW] = _actuators_mc_in->control[actuator_controls_s::INDEX_YAW] *
342  _mc_yaw_weight;
343 
344  if (_vtol_schedule.flight_mode == FW_MODE) {
345  _actuators_out_0->control[actuator_controls_s::INDEX_THROTTLE] =
346  _actuators_fw_in->control[actuator_controls_s::INDEX_THROTTLE];
347 
348  /* allow differential thrust if enabled */
349  if (_params_tiltrotor.diff_thrust == 1) {
350  _actuators_out_0->control[actuator_controls_s::INDEX_ROLL] =
351  _actuators_fw_in->control[actuator_controls_s::INDEX_YAW] * _params_tiltrotor.diff_thrust_scale;
352  }
353 
354  } else {
355  _actuators_out_0->control[actuator_controls_s::INDEX_THROTTLE] =
356  _actuators_mc_in->control[actuator_controls_s::INDEX_THROTTLE];
357  }
358 
359  _actuators_out_1->timestamp = hrt_absolute_time();
360  _actuators_out_1->timestamp_sample = _actuators_fw_in->timestamp_sample;
361 
362  _actuators_out_1->control[actuator_controls_s::INDEX_ROLL] =
363  -_actuators_fw_in->control[actuator_controls_s::INDEX_ROLL];
364  _actuators_out_1->control[actuator_controls_s::INDEX_PITCH] =
365  (_actuators_fw_in->control[actuator_controls_s::INDEX_PITCH]);
366  _actuators_out_1->control[actuator_controls_s::INDEX_YAW] =
367  _actuators_fw_in->control[actuator_controls_s::INDEX_YAW]; // yaw
368  _actuators_out_1->control[4] = _tilt_control;
369 }
