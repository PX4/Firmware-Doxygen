197 int DfBebopBusWrapper::_publish(struct bebop_state_data &data)
198 {
199 
200  battery_status_s battery_report;
201  const hrt_abstime timestamp = hrt_absolute_time();
202 
203  // TODO Check if this is the right way for the Bebop
204  // We don't have current measurements
205  _battery.updateBatteryStatus(timestamp, data.battery_voltage_v, 0.0, true, true, 0, _last_throttle, _armed,
206  &battery_report);
207 
208  esc_status_s esc_status = {};
209 
210  uint16_t esc_speed_setpoint_rpm[4] = {};
211  BebopBus::_get_esc_speed_setpoint(esc_speed_setpoint_rpm);
212  esc_status.timestamp = hrt_absolute_time();
213  esc_status.esc_count = 4;
214 
215  for (int i = 0; i < 4; i++) {
216  esc_status.esc[_esc_map[i]].timestamp = esc_status.timestamp;
217  esc_status.esc[_esc_map[i]].esc_rpm = data.rpm[i];
218  esc_status.esc[_esc_map[i]].esc_setpoint_raw = esc_speed_setpoint_rpm[i];
219  }
220 
221  // TODO: when is this ever blocked?
222  if (!(m_pub_blocked)) {
223 
224  if (_battery_topic == nullptr) {
225  _battery_topic = orb_advertise_multi(ORB_ID(battery_status), &battery_report,
226  &_battery_orb_class_instance, ORB_PRIO_LOW);
227 
228  } else {
229  orb_publish(ORB_ID(battery_status), _battery_topic, &battery_report);
230  }
231 
232  if (_esc_topic == nullptr) {
233  _esc_topic = orb_advertise(ORB_ID(esc_status), &esc_status);
234 
235  } else {
236  orb_publish(ORB_ID(esc_status), _esc_topic, &esc_status);
237  }
238 
239  }
240 
241  return 0;
242 }

328 void task_main(int argc, char *argv[])
329 {
330  _is_running = true;
331 
332  // Subscribe for orb topics
333  _controls_sub = orb_subscribe(ORB_ID(actuator_controls_0));
334  _armed_sub = orb_subscribe(ORB_ID(actuator_armed));
335 
336  // Start disarmed
337  _armed.armed = false;
338  _armed.prearmed = false;
339  _motors_running = false;
340 
341  // Set up poll topic
342  px4_pollfd_struct_t fds[1];
343  fds[0].fd = _controls_sub;
344  fds[0].events = POLLIN;
345  /* Don't limit poll intervall for now, 250 Hz should be fine. */
346  //orb_set_interval(_controls_sub, 10);
347 
348  // Set up mixers
349  if (initialize_mixers(MIXER_FILENAME) < 0) {
350  PX4_ERR("Mixer initialization failed.");
351  return;
352  }
353 
354  // Main loop
355  while (!_task_should_exit) {
356 
357  int pret = px4_poll(&fds[0], (sizeof(fds) / sizeof(fds[0])), 10);
358 
359  /* Timed out, do a periodic check for _task_should_exit. */
360  if (pret == 0) {
361  continue;
362  }
363 
364  /* This is undesirable but not much we can do. */
365  if (pret < 0) {
366  PX4_WARN("poll error %d, %d", pret, errno);
367  /* sleep a bit before next try */
368  usleep(100000);
369  continue;
370  }
371 
372  if (fds[0].revents & POLLIN) {
373  orb_copy(ORB_ID(actuator_controls_0), _controls_sub, &_controls[0]);
374 
375  _outputs.timestamp = _controls[0].timestamp;
376 
377 
378  if (_mixers != nullptr) {
379  /* do mixing */
380  _outputs.noutputs = _mixers->mix(_outputs.output, 4);
381  }
382 
383  // Set last throttle for battery calculations
384  g_dev->set_last_throttle(_controls[0].control[3]);
385 
386  /* disable unused ports by setting their output to NaN */
387  for (size_t i = _outputs.noutputs; i < sizeof(_outputs.output) / sizeof(_outputs.output[0]);
388  i++) {
389  _outputs.output[i] = NAN;
390  }
391 
392  // Check if the outputs are in range and rescale
393  for (size_t i = 0; i < _outputs.noutputs; ++i) {
394  if (i < _outputs.noutputs &&
395  PX4_ISFINITE(_outputs.output[i]) &&
396  _outputs.output[i] >= -1.0f &&
397  _outputs.output[i] <= 1.0f) {
398  /* scale for Bebop output 0.0 - 1.0 */
399  _outputs.output[i] = (_outputs.output[i] + 1.0f) / 2.0f;
400 
401  } else {
402  /*
403  * Value is NaN, INF or out of band - set to the minimum value.
404  * This will be clearly visible on the servo status and will limit the risk of accidentally
405  * spinning motors. It would be deadly in flight.
406  */
407  _outputs.output[i] = 0.0;
408  }
409  }
410 
411  // Adjust order of BLDCs from PX4 to Bebop
412  float motor_out[4];
413  motor_out[0] = _outputs.output[2];
414  motor_out[1] = _outputs.output[0];
415  motor_out[2] = _outputs.output[3];
416  motor_out[3] = _outputs.output[1];
417 
418  g_dev->set_esc_speeds(motor_out);
419 
420  if (_outputs_pub != nullptr) {
421  orb_publish(ORB_ID(actuator_outputs), _outputs_pub, &_outputs);
422 
423  } else {
424  _outputs_pub = orb_advertise(ORB_ID(actuator_outputs), &_outputs);
425  }
426  }
427 
428  bool updated;
429  orb_check(_armed_sub, &updated);
430 
431  if (updated) {
432  orb_copy(ORB_ID(actuator_armed), _armed_sub, &_armed);
433  }
434 
435  // Start the motors if armed but not alreay running
436  if (_armed.armed && !_motors_running) {
437  g_dev->start_motors();
438  _motors_running = true;
439  }
440 
441  // Stop motors if not armed but running
442  if (!_armed.armed && _motors_running) {
443  g_dev->stop_motors();
444  _motors_running = false;
445  }
446  }
447 
448  orb_unsubscribe(_controls_sub);
449  orb_unsubscribe(_armed_sub);
450 
451  _is_running = false;
452 
453 }

455 int start()
456 {
457  g_dev = new DfBebopBusWrapper();
458 
459  if (g_dev == nullptr) {
460  PX4_ERR("failed instantiating DfBebopBusWrapper object");
461  return -1;
462  }
463 
464  int ret = g_dev->start();
465 
466  if (ret != 0) {
467  PX4_ERR("DfBebopBusWrapper start failed");
468  return ret;
469  }
470 
471  // Open the Bebop dirver
472  DevHandle h;
473  DevMgr::getHandle(BEBOP_BUS_DEVICE_PATH, h);
474 
475  if (!h.isValid()) {
476  DF_LOG_INFO("Error: unable to obtain a valid handle for the receiver at: %s (%d)",
477  BEBOP_BUS_DEVICE_PATH, h.getError());
478  return -1;
479  }
480 
481  DevMgr::releaseHandle(h);
482 
483  // Start the task to forward the motor control commands
484  ASSERT(_task_handle == -1);
485 
486  /* start the task */
487  _task_handle = px4_task_spawn_cmd("bebop_bus_esc_main",
488  SCHED_DEFAULT,
489  SCHED_PRIORITY_MAX,
490  2000,
491  (px4_main_t)&task_main,
492  nullptr);
493 
494  if (_task_handle < 0) {
495  warn("task start failed");
496  return -1;
497  }
498 
499  _is_running = true;
500  return 0;
501 }
