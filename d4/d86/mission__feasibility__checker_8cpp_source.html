53 bool
54 MissionFeasibilityChecker::checkMissionFeasible(const mission_s &mission,
55  float max_distance_to_1st_waypoint, float max_distance_between_waypoints,
56  bool land_start_req)
57 {
58  bool failed = false;
59  bool warned = false;
60 
61  // first check if we have a valid position
62  const bool home_valid = _navigator->home_position_valid();
63  const bool home_alt_valid = _navigator->home_alt_valid();
64 
65  if (!home_alt_valid) {
66  failed = true;
67  warned = true;
68  mavlink_log_info(_navigator->get_mavlink_log_pub(), "Not yet ready for mission, no position lock.");
69 
70  } else {
71  failed = failed || !checkDistanceToFirstWaypoint(mission, max_distance_to_1st_waypoint);
72  }
73 
74  const float home_alt = _navigator->get_home_position()->alt;
75 
76  // check if all mission item commands are supported
77  failed = failed || !checkMissionItemValidity(mission);
78  failed = failed || !checkDistancesBetweenWaypoints(mission, max_distance_between_waypoints);
79  failed = failed || !checkGeofence(mission, home_alt, home_valid);
80  failed = failed || !checkHomePositionAltitude(mission, home_alt, home_alt_valid, warned);
81 
82  // VTOL always respects rotary wing feasibility
83  if (_navigator->get_vstatus()->is_rotary_wing || _navigator->get_vstatus()->is_vtol) {
84  failed = failed || !checkRotarywing(mission, home_alt, home_alt_valid);
85 
86  } else {
87  failed = failed || !checkFixedwing(mission, home_alt, home_alt_valid, land_start_req);
88  }
89 
90  return !failed;
91 }

99

102  return false;
103  }
104 
105  // look for a takeoff waypoint
106  if (missionitem.nav_cmd == NAV_CMD_TAKEOFF) {

108  // this makes sure that the takeoff waypoint is not reached before we are at least one meter in the air

352 bool
353 MissionFeasibilityChecker::checkFixedWingLanding(const mission_s &mission, bool land_start_req)
354 {
355  /* Go through all mission items and search for a landing waypoint
356  * if landing waypoint is found: the previous waypoint is checked to be at a feasible distance and altitude given the landing slope */
357 
358  bool landing_valid = false;
359 
360  bool land_start_found = false;
361  size_t do_land_start_index = 0;
362  size_t landing_approach_index = 0;
363 
364  for (size_t i = 0; i < mission.count; i++) {
365  struct mission_item_s missionitem;
366  const ssize_t len = sizeof(missionitem);
367 
368  if (dm_read((dm_item_t)mission.dataman_id, i, &missionitem, len) != len) {
369  /* not supposed to happen unless the datamanager can't access the SD card, etc. */
370  return false;
371  }
372 
373  // if DO_LAND_START found then require valid landing AFTER
374  if (missionitem.nav_cmd == NAV_CMD_DO_LAND_START) {
375  if (land_start_found) {
376  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "Mission rejected: more than one land start.");
377  return false;
378 
379  } else {
380  land_start_found = true;
381  do_land_start_index = i;
382  }
383  }
384 
385  if (missionitem.nav_cmd == NAV_CMD_LAND) {
386  mission_item_s missionitem_previous {};
387 
388  if (i > 0) {
389  landing_approach_index = i - 1;
390 
391  if (dm_read((dm_item_t)mission.dataman_id, landing_approach_index, &missionitem_previous, len) != len) {
392  /* not supposed to happen unless the datamanager can't access the SD card, etc. */
393  return false;
394  }
395 
396  if (MissionBlock::item_contains_position(missionitem_previous)) {
397 
398  const bool fw_status_valid = (_navigator->get_fw_pos_ctrl_status()->timestamp > 0);
399  const float wp_distance = get_distance_to_next_waypoint(missionitem_previous.lat, missionitem_previous.lon,
400  missionitem.lat, missionitem.lon);
401 
402  if (fw_status_valid && (wp_distance > _navigator->get_fw_pos_ctrl_status()->landing_flare_length)) {
403  /* Last wp is before flare region */
404 
405  const float delta_altitude = missionitem.altitude - missionitem_previous.altitude;
406 
407  if (delta_altitude < 0) {
408 
409  const float horizontal_slope_displacement = _navigator->get_fw_pos_ctrl_status()->landing_horizontal_slope_displacement;
410  const float slope_angle_rad = _navigator->get_fw_pos_ctrl_status()->landing_slope_angle_rad;
411  const float slope_alt_req = Landingslope::getLandingSlopeAbsoluteAltitude(wp_distance, missionitem.altitude,
412  horizontal_slope_displacement, slope_angle_rad);
413 
414  if (missionitem_previous.altitude > slope_alt_req + 1.0f) {
415  /* Landing waypoint is above altitude of slope at the given waypoint distance (with small tolerance for floating point discrepancies) */
416  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "Mission rejected: adjust landing approach.");
417 
418  const float wp_distance_req = Landingslope::getLandingSlopeWPDistance(missionitem_previous.altitude,
419  missionitem.altitude, horizontal_slope_displacement, slope_angle_rad);
420 
421  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "Move down %d m or move further away by %d m.",
422  (int)ceilf(slope_alt_req - missionitem_previous.altitude),
423  (int)ceilf(wp_distance_req - wp_distance));
424 
425  return false;
426  }
427 
428  } else {
429  /* Landing waypoint is above last waypoint */
430  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "Mission rejected: landing above last waypoint.");
431  return false;
432  }
433 
434  } else {
435  /* Last wp is in flare region */
436  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "Mission rejected: waypoint within landing flare.");
437  return false;
438  }
439 
440  landing_valid = true;
441 
442  } else {
443  // mission item before land doesn't have a position
444  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "Mission rejected: need landing approach.");
445  return false;
446  }
447 
448  } else {
449  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "Mission rejected: starts with land waypoint.");
450  return false;
451  }
452  }
453  }
454 
455  if (land_start_req && !land_start_found) {
456  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "Mission rejected: land start required.");
457  return false;
458  }
459 
460  if (land_start_found && (!landing_valid || (do_land_start_index > landing_approach_index))) {
461  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "Mission rejected: invalid land start.");
462  return false;
463  }
464 
465  /* No landing waypoints or no waypoints */
466  return true;
