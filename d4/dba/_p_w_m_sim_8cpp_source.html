65 int
66 PWMSim::set_mode(Mode mode)
67 {
68  /*
69  * Configure for PWM output.
70  *
71  * Note that regardless of the configured mode, the task is always
72  * listening and mixing; the mode just selects which of the channels
73  * are presented on the output pins.
74  */
75  switch (mode) {
76  case MODE_8PWM:
77  PX4_INFO("MODE_8PWM");
78  /* multi-port as 8 PWM outs */
79  _update_rate = 400; /* default output rate */
80  _num_outputs = 8;
81  break;
82 
83  case MODE_16PWM:
84  PX4_INFO("MODE_16PWM");
85  /* multi-port as 16 PWM outs */
86  _update_rate = 400; /* default output rate */
87  _num_outputs = 16;
88  break;
89 
90  case MODE_NONE:
91  PX4_INFO("MODE_NONE");
92  /* disable servo outputs and set a very low update rate */
93  _update_rate = 10;
94  _num_outputs = 0;
95  break;
96 
97  default:
98  return -EINVAL;
99  }
100 
101  _mode = mode;
102  return OK;
103 }

116 void
117 PWMSim::subscribe()
118 {
119  /* subscribe/unsubscribe to required actuator control groups */
120  uint32_t sub_groups = _groups_required & ~_groups_subscribed;
121  uint32_t unsub_groups = _groups_subscribed & ~_groups_required;
122  _poll_fds_num = 0;
123 
124  for (unsigned i = 0; i < actuator_controls_s::NUM_ACTUATOR_CONTROL_GROUPS; i++) {
125  if (sub_groups & (1 << i)) {
126  PX4_DEBUG("subscribe to actuator_controls_%d", i);
127  _control_subs[i] = orb_subscribe(_control_topics[i]);
128  }
129 
130  if (unsub_groups & (1 << i)) {
131  PX4_DEBUG("unsubscribe from actuator_controls_%d", i);
132  orb_unsubscribe(_control_subs[i]);
133  _control_subs[i] = -1;
134  }
135 
136  if (_control_subs[i] >= 0) {
137  _poll_fds[_poll_fds_num].fd = _control_subs[i];
138  _poll_fds[_poll_fds_num].events = POLLIN;
139  _poll_fds_num++;
140  }
141  }

156 void
157 PWMSim::run()
158 {
159  /* force a reset of the update rate */
160  _current_update_rate = 0;
161 
162  _armed_sub = orb_subscribe(ORB_ID(actuator_armed));
163 
164  /* advertise the mixed control outputs, insist on the first group output */
165  _outputs_pub = orb_advertise(ORB_ID(actuator_outputs), &_actuator_outputs);
166 
167  update_params();
168  int params_sub = orb_subscribe(ORB_ID(parameter_update));
169 
170  /* loop until killed */
171  while (!should_exit()) {
172 
173  if (_groups_subscribed != _groups_required) {
174  subscribe();
175  _groups_subscribed = _groups_required;
176  }
177 
178  /* handle update rate changes */
179  if (_current_update_rate != _update_rate) {
180  int update_rate_in_ms = int(1000 / _update_rate);
181 
182  if (update_rate_in_ms < 2) {
183  update_rate_in_ms = 2;
184  }
185 
186  for (unsigned i = 0; i < actuator_controls_s::NUM_ACTUATOR_CONTROL_GROUPS; i++) {
187  if (_control_subs[i] >= 0) {
188  orb_set_interval(_control_subs[i], update_rate_in_ms);
189  }
190  }
191 
192  // up_pwm_servo_set_rate(_update_rate);
193  _current_update_rate = _update_rate;
194  }
195 
196  if (_mixers) {
197  _mixers->set_airmode(_airmode);
198  }
199 
200  /* this can happen during boot, but after the sleep its likely resolved */
201  if (_poll_fds_num == 0) {
202  sleep(1);
203 
204  PX4_DEBUG("no valid fds");
205  continue;
206  }
207 
208  /* sleep waiting for data, but no more than a second */
209  int ret = px4_poll(&_poll_fds[0], _poll_fds_num, 1000);
210 
211  /* this would be bad... */
212  if (ret < 0) {
213  PX4_ERR("poll error %d", errno);
214  continue;
215  }
216 
217  if (ret == 0) {
218  // timeout
219  continue;
220  }
221 
222  /* get controls for required topics */
223  unsigned poll_id = 0;
224 
225  for (unsigned i = 0; i < actuator_controls_s::NUM_ACTUATOR_CONTROL_GROUPS; i++) {
226  if (_control_subs[i] >= 0) {
227  if (_poll_fds[poll_id].revents & POLLIN) {
228  orb_copy(_control_topics[i], _control_subs[i], &_controls[i]);
229  }
230 
231  poll_id++;
232  }
233  }
234 
235  /* can we mix? */
236  if (_armed && _mixers != nullptr) {
237 
238  /* do mixing */
239  unsigned num_outputs = _mixers->mix(&_actuator_outputs.output[0], _num_outputs);
240  _actuator_outputs.noutputs = num_outputs;
241 
242  /* disable unused ports by setting their output to NaN */
243  for (size_t i = 0; i < sizeof(_actuator_outputs.output) / sizeof(_actuator_outputs.output[0]); i++) {
244  if (i >= num_outputs) {
245  _actuator_outputs.output[i] = NAN;
246  }
247  }
248 
249  /* iterate actuators */
250  for (unsigned i = 0; i < num_outputs; i++) {
251  /* last resort: catch NaN, INF and out-of-band errors */
252  if (i < _actuator_outputs.noutputs &&
253  PX4_ISFINITE(_actuator_outputs.output[i]) &&
254  _actuator_outputs.output[i] >= -1.0f &&
255  _actuator_outputs.output[i] <= 1.0f) {
256  /* scale for PWM output 1000 - 2000us */
257  _actuator_outputs.output[i] = 1500 + (500 * _actuator_outputs.output[i]);
258 
259  if (_actuator_outputs.output[i] > _pwm_max[i]) {
260  _actuator_outputs.output[i] = _pwm_max[i];
261  }
262 
263  if (_actuator_outputs.output[i] < _pwm_min[i]) {
264  _actuator_outputs.output[i] = _pwm_min[i];
265  }
266 
267  } else {
268  /*
269  * Value is NaN, INF or out of band - set to the minimum value.
270  * This will be clearly visible on the servo status and will limit the risk of accidentally
271  * spinning motors. It would be deadly in flight.
272  */
273  _actuator_outputs.output[i] = PWM_SIM_DISARMED_MAGIC;
274  }
275  }
276 
277  /* overwrite outputs in case of force_failsafe */
278  if (_failsafe) {
279  for (size_t i = 0; i < num_outputs; i++) {
280  _actuator_outputs.output[i] = PWM_SIM_FAILSAFE_MAGIC;
281  }
282  }
283 
284  /* overwrite outputs in case of lockdown */
285  if (_lockdown) {
286  for (size_t i = 0; i < num_outputs; i++) {
287  _actuator_outputs.output[i] = 0.0;
288  }
289  }
290 
291  /* and publish for anyone that cares to see */
292  _actuator_outputs.timestamp = hrt_absolute_time();
293  orb_publish(ORB_ID(actuator_outputs), _outputs_pub, &_actuator_outputs);
294 
295  // use first valid timestamp_sample for latency tracking
296  for (int i = 0; i < actuator_controls_s::NUM_ACTUATOR_CONTROL_GROUPS; i++) {
297  const bool required = _groups_required & (1 << i);
298  const hrt_abstime &timestamp_sample = _controls[i].timestamp_sample;
299 
300  if (required && (timestamp_sample > 0)) {
301  perf_set_elapsed(_perf_control_latency, _actuator_outputs.timestamp - timestamp_sample);
302  break;
303  }
304  }
305  }
306 
307  /* how about an arming update? */
308  bool updated;
309 
310  orb_check(_armed_sub, &updated);
311 
312  if (updated) {
313  actuator_armed_s aa = {};
314 
315  if (orb_copy(ORB_ID(actuator_armed), _armed_sub, &aa) == PX4_OK) {
316  /* do not obey the lockdown value, as lockdown is for PWMSim. Only obey manual lockdown */
317  _armed = aa.armed;
318  _failsafe = aa.force_failsafe;
319  _lockdown = aa.manual_lockdown;
320  }
321  }
322 
323  /* check for parameter updates */
324  bool param_updated = false;
325  orb_check(params_sub, &param_updated);
326 
327  if (param_updated) {
328  struct parameter_update_s update;
329  orb_copy(ORB_ID(parameter_update), params_sub, &update);
330  update_params();
331  }
332  }
333 
334  for (unsigned i = 0; i < actuator_controls_s::NUM_ACTUATOR_CONTROL_GROUPS; i++) {
335  if (_control_subs[i] >= 0) {
336  orb_unsubscribe(_control_subs[i]);
337  }
338  }
339 
340  orb_unsubscribe(_armed_sub);
341  orb_unsubscribe(params_sub);
342 }

385

397  set_pwm_rate(arg);

399

401  // PWMSim always outputs at the alternate (usually faster) rate

429

444

446  break;
447  }
448

450  struct pwm_output_values *pwm = (struct pwm_output_values *)arg;
451 
452  for (unsigned i = 0; i < _num_outputs; i++) {
453  pwm->values[i] = (PWM_SIM_PWM_MAX_MAGIC + PWM_SIM_PWM_MIN_MAGIC) / 2;
454  }
455 
456  pwm->channel_count = _num_outputs;
457  break;
458  }
459 
460  case PWM_SERVO_GET_MAX_PWM: {
461  struct pwm_output_values *pwm = (struct pwm_output_values *)arg;
462 
463  for (unsigned i = 0; i < _num_outputs; i++) {
464  pwm->values[i] = PWM_SIM_PWM_MAX_MAGIC;
465  }
466 
467  pwm->channel_count = _num_outputs;
468  break;
469  }
470 
471  case PWM_SERVO_GET(0) ... PWM_SERVO_GET(PWM_OUTPUT_MAX_CHANNELS - 1): {
472 
473  unsigned channel = cmd - PWM_SERVO_GET(0);
474 
475  if (channel >= _num_outputs) {
476  ret = -EINVAL;
477 
478  } else {
479  /* fetch a current PWM value */
480  *(servo_position_t *)arg = _actuator_outputs.output[channel];
481  }
482 
483  break;
484  }
485 
486  case PWM_SERVO_GET_RATEGROUP(0) ... PWM_SERVO_GET_RATEGROUP(PWM_OUTPUT_MAX_CHANNELS - 1): {
487  // no restrictions on output grouping
488  unsigned channel = cmd - PWM_SERVO_GET_RATEGROUP(0);
489 
490  *(uint32_t *)arg = (1 << channel);
491  break;
492  }
493 
494  case PWM_SERVO_GET_COUNT:
495  case MIXERIOCGETOUTPUTCOUNT:
496  if (_mode == MODE_16PWM) {
497  *(unsigned *)arg = 16;
498 
499  } else if (_mode == MODE_8PWM) {
500 
501  *(unsigned *)arg = 8;
502  }
503 
504  break;
505 
506  case MIXERIOCRESET:
507  if (_mixers != nullptr) {
508  delete _mixers;
509  _mixers = nullptr;
510  _groups_required = 0;
511  }
512 
513  break;
514 
515  case MIXERIOCADDSIMPLE: {
516  mixer_simple_s *mixinfo = (mixer_simple_s *)arg;
517 
518  SimpleMixer *mixer = new SimpleMixer(control_callback, (uintptr_t)&_controls, mixinfo);
519 
520  if (mixer->check()) {
521  delete mixer;
522  _groups_required = 0;
523  ret = -EINVAL;
524 
525  } else {
526  if (_mixers == nullptr) {
527  _mixers = new MixerGroup(control_callback, (uintptr_t)&_controls);
528  }
529 
530  _mixers->add_mixer(mixer);
531  _mixers->groups_required(_groups_required);
532  }
533 
534  break;
535  }
536 
537  case MIXERIOCLOADBUF: {
538  const char *buf = (const char *)arg;
539  unsigned buflen = strnlen(buf, 1024);
540 
541  if (_mixers == nullptr) {
542  _mixers = new MixerGroup(control_callback, (uintptr_t)&_controls);
543  }
544 
545  if (_mixers == nullptr) {
546  _groups_required = 0;
547  ret = -ENOMEM;
548 
549  } else {
550 
551  ret = _mixers->load_from_buf(buf, buflen);
552 
553  if (ret != 0) {
554  PX4_ERR("mixer load failed with %d", ret);
555  delete _mixers;
556  _mixers = nullptr;
557  _groups_required = 0;
558  ret = -EINVAL;
559 
560  } else {
561  _mixers->groups_required(_groups_required);
562  }
563  }
564 
565  break;
566  }
567 
568 
569  default:
570  ret = -ENOTTY;
571  break;
572  }
573 
574  unlock();
575 
576  return ret;
577 }
