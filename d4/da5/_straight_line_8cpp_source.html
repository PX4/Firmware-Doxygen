54 void StraightLine::generateSetpoints(matrix::Vector3f &position_setpoint, matrix::Vector3f &velocity_setpoint)
55 {
56  // Check if target position has been reached
57  if (_desired_speed_at_target < VEL_ZERO_THRESHOLD &&
58  (_pos - _target).length() < NAV_ACC_RAD.get()) {
59  // Vehicle has reached target. Lock position
60  position_setpoint = _target;
61  velocity_setpoint = Vector3f(0.0f, 0.0f, 0.0f);
62 
63  return;
64  }
65 
66  // unit vector in the direction of the straight line
67  Vector3f u_orig_to_target = (_target - _origin).unit_or_zero();
68  // vector from origin to current position
69  Vector3f orig_to_pos = _pos - _origin;
70  // current position projected perpendicularly onto desired line
71  Vector3f closest_pt_on_line = _origin + u_orig_to_target * (orig_to_pos * u_orig_to_target);
72  // previous velocity in the direction of the line
73  float speed_sp_prev = math::max(velocity_setpoint * u_orig_to_target, 0.0f);
74 
75  // Calculate accelerating/decelerating distance depending on speed, speed at target and acceleration/deceleration
76  float acc_dec_distance = fabs((_desired_speed * _desired_speed) - (_desired_speed_at_target *
77  _desired_speed_at_target)) / 2.0f;
78  acc_dec_distance /= _desired_speed > _desired_speed_at_target ? _desired_deceleration : _desired_acceleration;
79 
80  float dist_to_target = (_target - _pos).length(); // distance to target
81 
82  // Either accelerate or decelerate
83  float speed_sp = dist_to_target > acc_dec_distance ? _desired_speed : _desired_speed_at_target;
84  float max_acc_dec = speed_sp > speed_sp_prev ? _desired_acceleration : -_desired_deceleration;
85 
86  float acc_track = 0.0f;
87 
88  if (_deltatime > FLT_EPSILON) {
89  acc_track = (speed_sp - speed_sp_prev) / _deltatime;
90  }
91 
92  if (fabs(acc_track) > fabs(max_acc_dec)) {
93  // accelerate/decelerate with desired acceleration/deceleration towards target
94  speed_sp = speed_sp_prev + max_acc_dec * _deltatime;
95  }
96 
97  // constrain the velocity
98  speed_sp = math::constrain(speed_sp, 0.0f, _desired_speed);
99 
100  // set the position and velocity setpoints
101  position_setpoint = closest_pt_on_line;
102  velocity_setpoint = u_orig_to_target * speed_sp;
103 }

105 float StraightLine::getMaxAcc()
106 {
107  // check if origin and target are different points
108  if ((_target - _origin).length() < FLT_EPSILON) {
109  return MPC_ACC_HOR_MAX.get();
110  }
111 
112  // unit vector in the direction of the straight line
113  Vector3f u_orig_to_target = (_target - _origin).unit_or_zero();
114 
115  // calculate the maximal horizontal acceleration
116  float divider = Vector2f(u_orig_to_target.data()).length();
117  float max_acc_hor = MPC_ACC_HOR_MAX.get();
118 
119  if (divider > FLT_EPSILON) {
120  max_acc_hor /= divider;
121 
122  } else {
123  max_acc_hor *= 1000.0f;
124  }
125 
126  // calculate the maximal vertical acceleration
127  float max_acc_vert_original = u_orig_to_target(2) < 0 ? MPC_ACC_UP_MAX.get() : MPC_ACC_DOWN_MAX.get();
128  float max_acc_vert = max_acc_vert_original;
129 
130  if (fabs(u_orig_to_target(2)) > FLT_EPSILON) {
131  max_acc_vert /= fabs(u_orig_to_target(2));
132 
133  } else {
134  max_acc_vert *= 1000.0f;
135  }
136 
137  return math::min(max_acc_hor, max_acc_vert);
138 }
