167  uint32_t crc32(const uint16_t *data, size_t len) const
168  {
169  uint32_t crc = 0xffffffff;
170  uint8_t tbl_idx;
171  uint8_t b;
172 
173  while (len--) {
174  b = *data;
175  tbl_idx = crc ^ b;
176  crc = (crc32_tab[tbl_idx] ^ (crc >> 8));
177 
178  b = *data >> 8;
179  tbl_idx = crc ^ b;
180  crc = (crc32_tab[tbl_idx] ^ (crc >> 8));
181 
182  data++;
183  }
184 
185  return crc ^ 0xffffffff;
186  }
