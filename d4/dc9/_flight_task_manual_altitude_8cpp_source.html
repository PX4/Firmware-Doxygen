89 void FlightTaskManualAltitude::_updateAltitudeLock()
90 {
91  // Depending on stick inputs and velocity, position is locked.
92  // If not locked, altitude setpoint is set to NAN.
93 
94  // Check if user wants to break
95  const bool apply_brake = fabsf(_velocity_setpoint(2)) <= FLT_EPSILON;
96 
97  // Check if vehicle has stopped
98  const bool stopped = (MPC_HOLD_MAX_Z.get() < FLT_EPSILON || fabsf(_velocity(2)) < MPC_HOLD_MAX_Z.get());
99 
100  // Manage transition between use of distance to ground and distance to local origin
101  // when terrain hold behaviour has been selected.
102  if (MPC_ALT_MODE.get() == 2) {
103  // Use horizontal speed as a transition criteria
104  float spd_xy = Vector2f(&_velocity(0)).length();
105 
106  // Use presence of horizontal stick inputs as a transition criteria
107  float stick_xy = Vector2f(&_sticks_expo(0)).length();
108  bool stick_input = stick_xy > 0.001f;
109 
110  if (_terrain_hold) {
111  bool too_fast = spd_xy > MPC_HOLD_MAX_XY.get();
112 
113  if (stick_input || too_fast || !PX4_ISFINITE(_dist_to_bottom)) {
114  // Stop using distance to ground
115  _terrain_hold = false;
116  _terrain_follow = false;
117 
118  // Adjust the setpoint to maintain the same height error to reduce control transients
119  if (PX4_ISFINITE(_dist_to_ground_lock) && PX4_ISFINITE(_dist_to_bottom)) {
120  _position_setpoint(2) = _position(2) + (_dist_to_ground_lock - _dist_to_bottom);
121 
122  } else {
123  _position_setpoint(2) = _position(2);
124  }
125  }
126 
127  } else {
128  bool not_moving = spd_xy < 0.5f * MPC_HOLD_MAX_XY.get();
129 
130  if (!stick_input && not_moving && PX4_ISFINITE(_dist_to_bottom)) {
131  // Start using distance to ground
132  _terrain_hold = true;
133  _terrain_follow = true;
134 
135  // Adjust the setpoint to maintain the same height error to reduce control transients
136  if (PX4_ISFINITE(_position_setpoint(2))) {
137  _dist_to_ground_lock = _dist_to_bottom + (_position_setpoint(2) - _position(2));
138  }
139  }
140  }
141 
142  }
143 
144  if ((MPC_ALT_MODE.get() == 1 || _terrain_follow) && PX4_ISFINITE(_dist_to_bottom)) {
145  // terrain following
146  _terrainFollowing(apply_brake, stopped);
147  // respect maximum altitude
148  _respectMaxAltitude();
149 
150  } else {
151  // normal mode where height is dependent on local frame
152 
153  if (apply_brake && stopped && !PX4_ISFINITE(_position_setpoint(2))) {
154  // lock position
155  _position_setpoint(2) = _position(2);
156 
157  // Ensure that minimum altitude is respected if
158  // there is a distance sensor and distance to bottom is below minimum.
159  if (PX4_ISFINITE(_dist_to_bottom) && _dist_to_bottom < _constraints.min_distance_to_ground) {
160  _terrainFollowing(apply_brake, stopped);
161 
162  } else {
163  _dist_to_ground_lock = NAN;
164  }
165 
166  } else if (PX4_ISFINITE(_position_setpoint(2)) && apply_brake) {
167  // Position is locked but check if a reset event has happened.
168  // We will shift the setpoints.
169  if (_sub_vehicle_local_position->get().z_reset_counter != _reset_counter) {
170  _position_setpoint(2) = _position(2);
171  _reset_counter = _sub_vehicle_local_position->get().z_reset_counter;
172  }
173 
174  } else {
175  // user demands velocity change
176  _position_setpoint(2) = NAN;
177  // ensure that maximum altitude is respected
178  _respectMaxAltitude();
179  }
180  }
181 }

183 void FlightTaskManualAltitude::_respectMinAltitude()
184 {
185  const bool respectAlt = PX4_ISFINITE(_dist_to_bottom)
186  && _dist_to_bottom < _constraints.min_distance_to_ground;
187 
188  // Height above ground needs to be limited (flow / range-finder)
189  if (respectAlt) {
190  // increase altitude to minimum flow distance
191  _position_setpoint(2) = _position(2)
192  - (_constraints.min_distance_to_ground - _dist_to_bottom);
193  }

196 void FlightTaskManualAltitude::_terrainFollowing(bool apply_brake, bool stopped)
197 {
198  if (apply_brake && stopped && !PX4_ISFINITE(_dist_to_ground_lock)) {
199  // User wants to break and vehicle reached zero velocity. Lock height to ground.
200 
201  // lock position
202  _position_setpoint(2) = _position(2);
203  // ensure that minimum altitude is respected
204  _respectMinAltitude();
205  // lock distance to ground but adjust first for minimum altitude
206  _dist_to_ground_lock = _dist_to_bottom - (_position_setpoint(2) - _position(2));
207 
208  } else if (apply_brake && PX4_ISFINITE(_dist_to_ground_lock)) {
209  // vehicle needs to follow terrain
210 
211  // difference between the current distance to ground and the desired distance to ground
212  const float delta_distance_to_ground = _dist_to_ground_lock - _dist_to_bottom;
213  // adjust position setpoint for the delta (note: NED frame)
214  _position_setpoint(2) = _position(2) - delta_distance_to_ground;
215 
216  } else {
217  // user demands velocity change in D-direction
218  _dist_to_ground_lock = _position_setpoint(2) = NAN;
219  }
220 }

222 void FlightTaskManualAltitude::_respectMaxAltitude()
223 {
224  if (PX4_ISFINITE(_dist_to_bottom)) {
225 
226  // if there is a valid maximum distance to ground, linearly increase speed limit with distance
227  // below the maximum, preserving control loop vertical position error gain.
228  if (PX4_ISFINITE(_constraints.max_distance_to_ground)) {
229  _constraints.speed_up = math::constrain(MPC_Z_P.get() * (_constraints.max_distance_to_ground - _dist_to_bottom),
230  -_min_speed_down, _max_speed_up);
231 
232  } else {
233  _constraints.speed_up = _max_speed_up;
234  }
235 
236  // if distance to bottom exceeded maximum distance, slowly approach maximum distance
237  if (_dist_to_bottom > _constraints.max_distance_to_ground) {
238  // difference between current distance to ground and maximum distance to ground
239  const float delta_distance_to_max = _dist_to_bottom - _constraints.max_distance_to_ground;
240  // set position setpoint to maximum distance to ground
241  _position_setpoint(2) = _position(2) + delta_distance_to_max;
242  // limit speed downwards to 0.7m/s
243  _constraints.speed_down = math::min(_min_speed_down, 0.7f);
244 
245  } else {
246  _constraints.speed_down = _min_speed_down;
247 
248  }
249  }

252 void FlightTaskManualAltitude::_updateSetpoints()
253 {
254  FlightTaskManualStabilized::_updateSetpoints(); // get yaw and thrust setpoints
255 
256  _thrust_setpoint *= NAN; // Don't need thrust setpoint from Stabilized mode.
257 
258  // Thrust in xy are extracted directly from stick inputs. A magnitude of
259  // 1 means that maximum thrust along xy is demanded. A magnitude of 0 means no
260  // thrust along xy is demanded. The maximum thrust along xy depends on the thrust
261  // setpoint along z-direction, which is computed in PositionControl.cpp.
262 
263  Vector2f sp{_sticks(0), _sticks(1)};
264  _rotateIntoHeadingFrame(sp);
265 
266  if (sp.length() > 1.0f) {
267  sp.normalize();
268  }
269 
270  _thrust_setpoint(0) = sp(0);
271  _thrust_setpoint(1) = sp(1);
272 
273  _updateAltitudeLock();
274 }
