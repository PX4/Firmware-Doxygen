373 /**
374  * Attitude controller.
375  * Input: 'vehicle_attitude_setpoint' topics (depending on mode)
376  * Output: '_rates_sp' vector, '_thrust_sp'
377  */
378 void
379 MulticopterAttitudeControl::control_attitude(float dt)
380 {
381  vehicle_attitude_setpoint_poll();
382  _thrust_sp = _v_att_sp.thrust;
383 
384  /* prepare yaw weight from the ratio between roll/pitch and yaw gains */
385  Vector3f attitude_gain = _attitude_p;
386  const float roll_pitch_gain = (attitude_gain(0) + attitude_gain(1)) / 2.f;
387  const float yaw_w = math::constrain(attitude_gain(2) / roll_pitch_gain, 0.f, 1.f);
388  attitude_gain(2) = roll_pitch_gain;
389 
390  /* get estimated and desired vehicle attitude */
391  Quatf q(_v_att.q);
392  Quatf qd(_v_att_sp.q_d);
393 
394  /* ensure input quaternions are exactly normalized because acosf(1.00001) == NaN */
395  q.normalize();
396  qd.normalize();
397 
398  /* calculate reduced desired attitude neglecting vehicle's yaw to prioritize roll and pitch */
399  Vector3f e_z = q.dcm_z();
400  Vector3f e_z_d = qd.dcm_z();
401  Quatf qd_red(e_z, e_z_d);
402 
403  if (abs(qd_red(1)) > (1.f - 1e-5f) || abs(qd_red(2)) > (1.f - 1e-5f)) {
404  /* In the infinitesimal corner case where the vehicle and thrust have the completely opposite direction,
405  * full attitude control anyways generates no yaw input and directly takes the combination of
406  * roll and pitch leading to the correct desired yaw. Ignoring this case would still be totally safe and stable. */
407  qd_red = qd;
408 
409  } else {
410  /* transform rotation from current to desired thrust vector into a world frame reduced desired attitude */
411  qd_red *= q;
412  }
413 
414  /* mix full and reduced desired attitude */
415  Quatf q_mix = qd_red.inversed() * qd;
416  q_mix *= math::signNoZero(q_mix(0));
417  /* catch numerical problems with the domain of acosf and asinf */
418  q_mix(0) = math::constrain(q_mix(0), -1.f, 1.f);
419  q_mix(3) = math::constrain(q_mix(3), -1.f, 1.f);
420  qd = qd_red * Quatf(cosf(yaw_w * acosf(q_mix(0))), 0, 0, sinf(yaw_w * asinf(q_mix(3))));
421 
422  /* quaternion attitude control law, qe is rotation from q to qd */
423  Quatf qe = q.inversed() * qd;
424 
425  /* using sin(alpha/2) scaled rotation axis as attitude error (see quaternion definition by axis angle)
426  * also taking care of the antipodal unit quaternion ambiguity */
427  Vector3f eq = 2.f * math::signNoZero(qe(0)) * qe.imag();
428 
429  /* calculate angular rates setpoint */
430  _rates_sp = eq.emult(attitude_gain);
431 
432  /* Feed forward the yaw setpoint rate.
433  * yaw_sp_move_rate is the feed forward commanded rotation around the world z-axis,
434  * but we need to apply it in the body frame (because _rates_sp is expressed in the body frame).
435  * Therefore we infer the world z-axis (expressed in the body frame) by taking the last column of R.transposed (== q.inversed)
436  * and multiply it by the yaw setpoint rate (yaw_sp_move_rate).
437  * This yields a vector representing the commanded rotatation around the world z-axis expressed in the body frame
438  * such that it can be added to the rates setpoint.
439  */
440  _rates_sp += q.inversed().dcm_z() * _v_att_sp.yaw_sp_move_rate;
441 
442 
443  /* limit rates */
444  for (int i = 0; i < 3; i++) {
445  if ((_v_control_mode.flag_control_velocity_enabled || _v_control_mode.flag_control_auto_enabled) &&
446  !_v_control_mode.flag_control_manual_enabled) {
447  _rates_sp(i) = math::constrain(_rates_sp(i), -_auto_rate_max(i), _auto_rate_max(i));
448 
449  } else {
450  _rates_sp(i) = math::constrain(_rates_sp(i), -_mc_rate_max(i), _mc_rate_max(i));
451  }
452  }
453 
454  /* VTOL weather-vane mode, dampen yaw rate */
455  if (_vehicle_status.is_vtol && _v_att_sp.disable_mc_yaw_control) {
456  if (_v_control_mode.flag_control_velocity_enabled || _v_control_mode.flag_control_auto_enabled) {
457 
458  const float wv_yaw_rate_max = _auto_rate_max(2) * _vtol_wv_yaw_rate_scale.get();
459  _rates_sp(2) = math::constrain(_rates_sp(2), -wv_yaw_rate_max, wv_yaw_rate_max);
460 
461  // prevent integrator winding up in weathervane mode
462  _rates_int(2) = 0.0f;
463  }
464  }
465 }

467 /*
468  * Throttle PID attenuation
469  * Function visualization available here https://www.desmos.com/calculator/gn4mfoddje
470  * Input: 'tpa_breakpoint', 'tpa_rate', '_thrust_sp'
471  * Output: 'pidAttenuationPerAxis' vector
472  */
473 Vector3f
474 MulticopterAttitudeControl::pid_attenuations(float tpa_breakpoint, float tpa_rate)
475 {
476  /* throttle pid attenuation factor */
477  float tpa = 1.0f - tpa_rate * (fabsf(_v_rates_sp.thrust) - tpa_breakpoint) / (1.0f - tpa_breakpoint);
478  tpa = fmaxf(TPA_RATE_LOWER_LIMIT, fminf(1.0f, tpa));
479 
480  Vector3f pidAttenuationPerAxis;
481  pidAttenuationPerAxis(AXIS_INDEX_ROLL) = tpa;
482  pidAttenuationPerAxis(AXIS_INDEX_PITCH) = tpa;
483  pidAttenuationPerAxis(AXIS_INDEX_YAW) = 1.0;
484 
485  return pidAttenuationPerAxis;
486 }

488 /*
489  * Attitude rates controller.
490  * Input: '_rates_sp' vector, '_thrust_sp'
491  * Output: '_att_control' vector
492  */
493 void
494 MulticopterAttitudeControl::control_attitude_rates(float dt)
495 {
496  /* reset integral if disarmed */
497  if (!_v_control_mode.flag_armed || !_vehicle_status.is_rotary_wing) {
498  _rates_int.zero();
499  }
500 
501  // get the raw gyro data and correct for thermal errors
502  Vector3f rates;
503 
504  if (_selected_gyro == 0) {
505  rates(0) = (_sensor_gyro.x - _sensor_correction.gyro_offset_0[0]) * _sensor_correction.gyro_scale_0[0];
506  rates(1) = (_sensor_gyro.y - _sensor_correction.gyro_offset_0[1]) * _sensor_correction.gyro_scale_0[1];
507  rates(2) = (_sensor_gyro.z - _sensor_correction.gyro_offset_0[2]) * _sensor_correction.gyro_scale_0[2];
508 
509  } else if (_selected_gyro == 1) {
510  rates(0) = (_sensor_gyro.x - _sensor_correction.gyro_offset_1[0]) * _sensor_correction.gyro_scale_1[0];
511  rates(1) = (_sensor_gyro.y - _sensor_correction.gyro_offset_1[1]) * _sensor_correction.gyro_scale_1[1];
512  rates(2) = (_sensor_gyro.z - _sensor_correction.gyro_offset_1[2]) * _sensor_correction.gyro_scale_1[2];
513 
514  } else if (_selected_gyro == 2) {
515  rates(0) = (_sensor_gyro.x - _sensor_correction.gyro_offset_2[0]) * _sensor_correction.gyro_scale_2[0];
516  rates(1) = (_sensor_gyro.y - _sensor_correction.gyro_offset_2[1]) * _sensor_correction.gyro_scale_2[1];
517  rates(2) = (_sensor_gyro.z - _sensor_correction.gyro_offset_2[2]) * _sensor_correction.gyro_scale_2[2];
518 
519  } else {
520  rates(0) = _sensor_gyro.x;
521  rates(1) = _sensor_gyro.y;
522  rates(2) = _sensor_gyro.z;
523  }
524 
525  // rotate corrected measurements from sensor to body frame
526  rates = _board_rotation * rates;
527 
528  // correct for in-run bias errors
529  rates(0) -= _sensor_bias.gyro_x_bias;
530  rates(1) -= _sensor_bias.gyro_y_bias;
531  rates(2) -= _sensor_bias.gyro_z_bias;
532 
533  Vector3f rates_p_scaled = _rate_p.emult(pid_attenuations(_tpa_breakpoint_p.get(), _tpa_rate_p.get()));
534  Vector3f rates_i_scaled = _rate_i.emult(pid_attenuations(_tpa_breakpoint_i.get(), _tpa_rate_i.get()));
535  Vector3f rates_d_scaled = _rate_d.emult(pid_attenuations(_tpa_breakpoint_d.get(), _tpa_rate_d.get()));
536 
537  /* angular rates error */
538  Vector3f rates_err = _rates_sp - rates;
539 
540  /* apply low-pass filtering to the rates for D-term */
541  Vector3f rates_filtered(
542  _lp_filters_d[0].apply(rates(0)),
543  _lp_filters_d[1].apply(rates(1)),
544  _lp_filters_d[2].apply(rates(2)));
545 
546  _att_control = rates_p_scaled.emult(rates_err) +
547  _rates_int -
548  rates_d_scaled.emult(rates_filtered - _rates_prev_filtered) / dt +
549  _rate_ff.emult(_rates_sp);
550 
551  _rates_prev = rates;
552  _rates_prev_filtered = rates_filtered;
553 
554  /* update integral only if we are not landed */
555  if (!_vehicle_land_detected.maybe_landed && !_vehicle_land_detected.landed) {
556  for (int i = AXIS_INDEX_ROLL; i < AXIS_COUNT; i++) {
557  // Check for positive control saturation
558  bool positive_saturation =
559  ((i == AXIS_INDEX_ROLL) && _saturation_status.flags.roll_pos) ||
560  ((i == AXIS_INDEX_PITCH) && _saturation_status.flags.pitch_pos) ||
561  ((i == AXIS_INDEX_YAW) && _saturation_status.flags.yaw_pos);
562 
563  // Check for negative control saturation
564  bool negative_saturation =
565  ((i == AXIS_INDEX_ROLL) && _saturation_status.flags.roll_neg) ||
566  ((i == AXIS_INDEX_PITCH) && _saturation_status.flags.pitch_neg) ||
567  ((i == AXIS_INDEX_YAW) && _saturation_status.flags.yaw_neg);
568 
569  // prevent further positive control saturation
570  if (positive_saturation) {
571  rates_err(i) = math::min(rates_err(i), 0.0f);
572 
573  }
574 
575  // prevent further negative control saturation
576  if (negative_saturation) {
577  rates_err(i) = math::max(rates_err(i), 0.0f);
578 
579  }
580 
581  // Perform the integration using a first order method and do not propagate the result if out of range or invalid
582  float rate_i = _rates_int(i) + rates_i_scaled(i) * rates_err(i) * dt;
583 
584  if (PX4_ISFINITE(rate_i) && rate_i > -_rate_int_lim(i) && rate_i < _rate_int_lim(i)) {
585  _rates_int(i) = rate_i;
586 
587  }
588  }
589  }
590 
591  /* explicitly limit the integrator state */
592  for (int i = AXIS_INDEX_ROLL; i < AXIS_COUNT; i++) {
593  _rates_int(i) = math::constrain(_rates_int(i), -_rate_int_lim(i), _rate_int_lim(i));
594 
595  }
596 }

599 MulticopterAttitudeControl::run()
600 {
601 
602  /*
603  * do subscriptions
604  */
605  _v_att_sub = orb_subscribe(ORB_ID(vehicle_attitude));
606  _v_att_sp_sub = orb_subscribe(ORB_ID(vehicle_attitude_setpoint));
607  _v_rates_sp_sub = orb_subscribe(ORB_ID(vehicle_rates_setpoint));
608  _v_control_mode_sub = orb_subscribe(ORB_ID(vehicle_control_mode));
609  _params_sub = orb_subscribe(ORB_ID(parameter_update));
610  _manual_control_sp_sub = orb_subscribe(ORB_ID(manual_control_setpoint));
611  _vehicle_status_sub = orb_subscribe(ORB_ID(vehicle_status));
612  _motor_limits_sub = orb_subscribe(ORB_ID(multirotor_motor_limits));
613  _battery_status_sub = orb_subscribe(ORB_ID(battery_status));
614 
615  _gyro_count = math::min(orb_group_count(ORB_ID(sensor_gyro)), MAX_GYRO_COUNT);
616 
617  if (_gyro_count == 0) {
618  _gyro_count = 1;
619  }
620 
621  for (unsigned s = 0; s < _gyro_count; s++) {
622  _sensor_gyro_sub[s] = orb_subscribe_multi(ORB_ID(sensor_gyro), s);
623  }
624 
625  _sensor_correction_sub = orb_subscribe(ORB_ID(sensor_correction));
626  _sensor_bias_sub = orb_subscribe(ORB_ID(sensor_bias));
627  _vehicle_land_detected_sub = orb_subscribe(ORB_ID(vehicle_land_detected));
628 
629  /* wakeup source: gyro data from sensor selected by the sensor app */
630  px4_pollfd_struct_t poll_fds = {};
631  poll_fds.events = POLLIN;
632 
633  const hrt_abstime task_start = hrt_absolute_time();
634  hrt_abstime last_run = task_start;
635  float dt_accumulator = 0.f;
636  int loop_counter = 0;
637 
638  while (!should_exit()) {
639 
640  poll_fds.fd = _sensor_gyro_sub[_selected_gyro];
641 
642  /* wait for up to 100ms for data */
643  int pret = px4_poll(&poll_fds, 1, 100);
644 
645  /* timed out - periodic check for should_exit() */
646  if (pret == 0) {
647  continue;
648  }
649 
650  /* this is undesirable but not much we can do - might want to flag unhappy status */
651  if (pret < 0) {
652  PX4_ERR("poll error %d, %d", pret, errno);
653  /* sleep a bit before next try */
654  usleep(100000);
655  continue;
656  }
657 
658  perf_begin(_loop_perf);
659 
660  /* run controller on gyro changes */
661  if (poll_fds.revents & POLLIN) {
662  const hrt_abstime now = hrt_absolute_time();
663  float dt = (now - last_run) / 1e6f;
664  last_run = now;
665 
666  /* guard against too small (< 2ms) and too large (> 20ms) dt's */
667  if (dt < 0.002f) {
668  dt = 0.002f;
669 
670  } else if (dt > 0.02f) {
671  dt = 0.02f;
672  }
673 
674  /* copy gyro data */
675  orb_copy(ORB_ID(sensor_gyro), _sensor_gyro_sub[_selected_gyro], &_sensor_gyro);
676 
677  /* check for updates in other topics */
678  parameter_update_poll();
679  vehicle_control_mode_poll();
680  vehicle_manual_poll();
681  vehicle_status_poll();
682  vehicle_motor_limits_poll();
683  battery_status_poll();
684  vehicle_attitude_poll();
685  sensor_correction_poll();
686  sensor_bias_poll();
687  vehicle_land_detected_poll();
688 
689  /* Check if we are in rattitude mode and the pilot is above the threshold on pitch
690  * or roll (yaw can rotate 360 in normal att control). If both are true don't
691  * even bother running the attitude controllers */
692  if (_v_control_mode.flag_control_rattitude_enabled) {
693  if (fabsf(_manual_control_sp.y) > _rattitude_thres.get() ||
694  fabsf(_manual_control_sp.x) > _rattitude_thres.get()) {
695  _v_control_mode.flag_control_attitude_enabled = false;
696  }
697  }
698 
699  if (_v_control_mode.flag_control_attitude_enabled) {
700 
701  control_attitude(dt);
702 
703  /* publish attitude rates setpoint */
704  _v_rates_sp.roll = _rates_sp(0);
705  _v_rates_sp.pitch = _rates_sp(1);
706  _v_rates_sp.yaw = _rates_sp(2);
707  _v_rates_sp.thrust = _thrust_sp;
708  _v_rates_sp.timestamp = hrt_absolute_time();
709 
710  if (_v_rates_sp_pub != nullptr) {
711  orb_publish(_rates_sp_id, _v_rates_sp_pub, &_v_rates_sp);
712 
713  } else if (_rates_sp_id) {
714  _v_rates_sp_pub = orb_advertise(_rates_sp_id, &_v_rates_sp);
715  }
716 
717  } else {
718  /* attitude controller disabled, poll rates setpoint topic */
719  if (_v_control_mode.flag_control_manual_enabled) {
720  /* manual rates control - ACRO mode */
721  Vector3f man_rate_sp(
722  math::superexpo(_manual_control_sp.y, _acro_expo_rp.get(), _acro_superexpo_rp.get()),
723  math::superexpo(-_manual_control_sp.x, _acro_expo_rp.get(), _acro_superexpo_rp.get()),
724  math::superexpo(_manual_control_sp.r, _acro_expo_y.get(), _acro_superexpo_y.get()));
725  _rates_sp = man_rate_sp.emult(_acro_rate_max);
726  _thrust_sp = _manual_control_sp.z;
727 
728  /* publish attitude rates setpoint */
729  _v_rates_sp.roll = _rates_sp(0);
730  _v_rates_sp.pitch = _rates_sp(1);
731  _v_rates_sp.yaw = _rates_sp(2);
732  _v_rates_sp.thrust = _thrust_sp;
733  _v_rates_sp.timestamp = hrt_absolute_time();
734 
735  if (_v_rates_sp_pub != nullptr) {
736  orb_publish(_rates_sp_id, _v_rates_sp_pub, &_v_rates_sp);
737 
738  } else if (_rates_sp_id) {
739  _v_rates_sp_pub = orb_advertise(_rates_sp_id, &_v_rates_sp);
740  }
741 
742  } else {
743  /* attitude controller disabled, poll rates setpoint topic */
744  vehicle_rates_setpoint_poll();
745  _rates_sp(0) = _v_rates_sp.roll;
746  _rates_sp(1) = _v_rates_sp.pitch;
747  _rates_sp(2) = _v_rates_sp.yaw;
748  _thrust_sp = _v_rates_sp.thrust;
749  }
750  }
751 
752  if (_v_control_mode.flag_control_rates_enabled) {
753  control_attitude_rates(dt);
754 
755  /* publish actuator controls */
756  _actuators.control[0] = (PX4_ISFINITE(_att_control(0))) ? _att_control(0) : 0.0f;
757  _actuators.control[1] = (PX4_ISFINITE(_att_control(1))) ? _att_control(1) : 0.0f;
758  _actuators.control[2] = (PX4_ISFINITE(_att_control(2))) ? _att_control(2) : 0.0f;
759  _actuators.control[3] = (PX4_ISFINITE(_thrust_sp)) ? _thrust_sp : 0.0f;
760  _actuators.control[7] = _v_att_sp.landing_gear;
761  _actuators.timestamp = hrt_absolute_time();
762  _actuators.timestamp_sample = _sensor_gyro.timestamp;
763 
764  /* scale effort by battery status */
765  if (_bat_scale_en.get() && _battery_status.scale > 0.0f) {
766  for (int i = 0; i < 4; i++) {
767  _actuators.control[i] *= _battery_status.scale;
768  }
769  }
770 
771  if (!_actuators_0_circuit_breaker_enabled) {
772  if (_actuators_0_pub != nullptr) {
773 
774  orb_publish(_actuators_id, _actuators_0_pub, &_actuators);
775 
776  } else if (_actuators_id) {
777  _actuators_0_pub = orb_advertise(_actuators_id, &_actuators);
778  }
779 
780  }
781 
782  /* publish controller status */
783  rate_ctrl_status_s rate_ctrl_status;
784  rate_ctrl_status.timestamp = hrt_absolute_time();
785  rate_ctrl_status.rollspeed = _rates_prev(0);
786  rate_ctrl_status.pitchspeed = _rates_prev(1);
787  rate_ctrl_status.yawspeed = _rates_prev(2);
788  rate_ctrl_status.rollspeed_integ = _rates_int(0);
789  rate_ctrl_status.pitchspeed_integ = _rates_int(1);
790  rate_ctrl_status.yawspeed_integ = _rates_int(2);
791 
792  int instance;
793  orb_publish_auto(ORB_ID(rate_ctrl_status), &_controller_status_pub, &rate_ctrl_status, &instance, ORB_PRIO_DEFAULT);
794  }
795 
796  if (_v_control_mode.flag_control_termination_enabled) {
797  if (!_vehicle_status.is_vtol) {
798 
799  _rates_sp.zero();
800  _rates_int.zero();
801  _thrust_sp = 0.0f;
802  _att_control.zero();
803 
804  /* publish actuator controls */
805  _actuators.control[0] = 0.0f;
806  _actuators.control[1] = 0.0f;
807  _actuators.control[2] = 0.0f;
808  _actuators.control[3] = 0.0f;
809  _actuators.timestamp = hrt_absolute_time();
810  _actuators.timestamp_sample = _sensor_gyro.timestamp;
811 
812  if (!_actuators_0_circuit_breaker_enabled) {
813  if (_actuators_0_pub != nullptr) {
814 
815  orb_publish(_actuators_id, _actuators_0_pub, &_actuators);
816 
817  } else if (_actuators_id) {
818  _actuators_0_pub = orb_advertise(_actuators_id, &_actuators);
819  }
820  }
821  }
822  }
823 
824  /* calculate loop update rate while disarmed or at least a few times (updating the filter is expensive) */
825  if (!_v_control_mode.flag_armed || (now - task_start) < 3300000) {
826  dt_accumulator += dt;
827  ++loop_counter;
828 
829  if (dt_accumulator > 1.f) {
830  const float loop_update_rate = (float)loop_counter / dt_accumulator;
831  _loop_update_rate_hz = _loop_update_rate_hz * 0.5f + loop_update_rate * 0.5f;
832  dt_accumulator = 0;
833  loop_counter = 0;
834  _lp_filters_d[0].set_cutoff_frequency(_loop_update_rate_hz, _d_term_cutoff_freq.get());
835  _lp_filters_d[1].set_cutoff_frequency(_loop_update_rate_hz, _d_term_cutoff_freq.get());
836  _lp_filters_d[2].set_cutoff_frequency(_loop_update_rate_hz, _d_term_cutoff_freq.get());
837  }
838  }
839 
840  }
841 
842  perf_end(_loop_perf);
843  }
844 
845  orb_unsubscribe(_v_att_sub);
846  orb_unsubscribe(_v_att_sp_sub);
847  orb_unsubscribe(_v_rates_sp_sub);
848  orb_unsubscribe(_v_control_mode_sub);
849  orb_unsubscribe(_params_sub);
850  orb_unsubscribe(_manual_control_sp_sub);
851  orb_unsubscribe(_vehicle_status_sub);
852  orb_unsubscribe(_motor_limits_sub);
853  orb_unsubscribe(_battery_status_sub);
854 
855  for (unsigned s = 0; s < _gyro_count; s++) {
856  orb_unsubscribe(_sensor_gyro_sub[s]);
857  }
858 
859  orb_unsubscribe(_sensor_correction_sub);
860  orb_unsubscribe(_sensor_bias_sub);
861  orb_unsubscribe(_vehicle_land_detected_sub);
862 }
