74 int board_mcu_version(char *rev, const char **revstr, const char **errata)
75 {
76  uint32_t abc = getreg32(STM32_DEBUGMCU_BASE);
77 
78  int32_t chip_version = abc & DEVID_MASK;
79  enum MCU_REV revid = (abc & REVID_MASK) >> 16;
80  const char *chip_errata = NULL;
81 
82  switch (chip_version) {
83 
84  case STM32F74xxx_75xxx:
85  *revstr = "STM32F74xxx";
86  break;
87 
88  case STM32F76xxx_77xxx:
89  *revstr = "STM32F76xxx";
90  break;
91 
92  case STM32F42x_43x:
93  *revstr = "STM32F42x";
94  /* Set possible errata */
95  chip_errata = STM32_F4_ERRATA;
96  break;
97 
98  case STM32F103_LD:
99  *revstr = "STM32F1xx Low";
100  break;
101 
102  case STM32F103_MD:
103  *revstr = "STM32F1xx Med";
104  break;
105 
106  case STM32F103_HD:
107  *revstr = "STM32F1xx Hi";
108  break;
109 
110  case STM32F103_XLD:
111  *revstr = "STM32F1xx XL";
112  break;
113 
114  case STM32F103_CON:
115  *revstr = "STM32F1xx Con";
116  break;
117 
118  default:
119  *revstr = "STM32F???";
120  break;
121  }
122 
123  switch (revid) {
124 
125  case MCU_REV_STM32F4_REV_A:
126  *rev = 'A';
127  break;
128 
129  case MCU_REV_STM32F4_REV_Z:
130  *rev = 'Z';
131  break;
132 
133  case MCU_REV_STM32F4_REV_Y:
134  *rev = 'Y';
135  break;
136 
137  case MCU_REV_STM32F4_REV_1:
138  *rev = '1';
139  break;
140 
141  case MCU_REV_STM32F4_REV_3:
142  *rev = '3';
143  chip_errata = NULL;
144  break;
145 
146  default:
147  // todo add rev for 103 - if needed
148  *rev = '?';
149  revid = -1;
150  break;
151  }
152 
153  if (errata) {
154  *errata = chip_errata;
155  }
156 
157  return revid;
158 }
