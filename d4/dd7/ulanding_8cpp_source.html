198  int ret = 0;
199 
200  do { /* create a scope to handle exit conditions using break */

207  }
208 
209  /* open fd */
210  int fd = ::open(_port, O_RDWR | O_NOCTTY);
211  PX4_INFO("passed open port");
212 
213  if (fd < 0) {
214  PX4_WARN("failed to open serial device");
215  ret = 1;
216  break;
217  }
218 
219  struct termios uart_config;
220 
221  int termios_state;
222 
223  /* fill the struct for the new configuration */
224  tcgetattr(fd, &uart_config);
225 
226  /** Input flags - Turn off input processing
227  *
228  * convert break to null byte, no CR to NL translation,
229  * no NL to CR translation, don't mark parity errors or breaks
230  * no input parity check, don't strip high bit off,
231  * no XON/XOFF software flow control
232  *
233  */
234 
235  uart_config.c_iflag &= ~(IGNBRK | BRKINT | ICRNL |
236  INLCR | PARMRK | INPCK | ISTRIP | IXON);
237 
238  /** No line processing
239  *
240  * echo off, echo newline off, canonical mode off,
241  * extended input processing off, signal chars off
242  *
243  */
244 
245  uart_config.c_lflag &= ~(ECHO | ECHONL | ICANON | IEXTEN | ISIG);
246 
247  /* clear ONLCR flag (which appends a CR for every LF) */
248  uart_config.c_oflag &= ~ONLCR;
249 
250  /* no parity, one stop bit */
251  uart_config.c_cflag &= ~(CSTOPB | PARENB);
252 
253 
254 
255  unsigned speed = B115200;
256 
257  /* set baud rate */
258  if ((termios_state = cfsetispeed(&uart_config, speed)) < 0) {
259  PX4_WARN("ERR CFG: %d ISPD", termios_state);
260  ret = 1;
261  break;
262  }
263 
264  if ((termios_state = cfsetospeed(&uart_config, speed)) < 0) {
265  PX4_WARN("ERR CFG: %d OSPD\n", termios_state);
266  ret = 1;
267  break;
268  }
269 
270  if ((termios_state = tcsetattr(fd, TCSANOW, &uart_config)) < 0) {
271  PX4_WARN("ERR baud %d ATTR", termios_state);
272  ret = 1;
273  break;

330 bool Radar::read_and_parse(uint8_t *buf, int len, float *range)
331 {
332  bool ret = false;
333 
334  // write new data into a ring buffer
335  for (int i = 0; i < len; i++) {
336 
337  _head++;
338 
339  if (_head >= BUF_LEN) {
340  _head = 0;
341  }
342 
343  if (_tail == _head) {
344  _tail = (_tail == BUF_LEN - 1) ? 0 : _head + 1;
345  }
346 
347  _buf[_head] = buf[i];
348  }
349 
350  // check how many bytes are in the buffer, return if it's lower than the size of one package
351  int num_bytes = _head >= _tail ? (_head - _tail + 1) : (_head + 1 + BUF_LEN - _tail);
352 
353  if (num_bytes < BUF_LEN / 3) {
354  PX4_DEBUG("not enough bytes");
355  return false;
356  }
357 
358  int index = _head;
359  uint8_t no_header_counter = 0; // counter for bytes which are non header bytes
360 
361  // go through the buffer backwards starting from the newest byte
362  // if we find a header byte and the previous two bytes weren't header bytes
363  // then we found the newest package.
364  for (int i = 0; i < num_bytes; i++) {
365  if (is_header_byte(_buf[index])) {
366  if (no_header_counter >= BUF_LEN / 3 - 1) {
367  if (ULANDING_VERSION == 1) {
368  if (((_buf[index + 1] + _buf[index + 2] + _buf[index + 3] + _buf[index + 4])) != (_buf[index + 5])
369  || (_buf[index + 1] <= 0)) {
370  ret = false;
371  break;
372  }
373 
374  int raw = _buf[index + 3] * 256 + _buf[index + 2];
375 
376  *range = ((float)(raw / 100.)); // raw is in cm, converts range to meters;
377 
378  } else {
379  int raw = (_buf[index + 1] & 0x7F);
380  raw += ((_buf[index + 2] & 0x7F) << 7);
381 
382  *range = ((float)raw) * 0.045f;
383  }
384 
385  // set the tail to one after the index because we neglect
386  // any data before the one we just read
387  _tail = index == BUF_LEN - 1 ? 0 : index++;
388  ret = true;
389  break;
390 
391  }
392 
393  no_header_counter = 0;
394 
395  } else {
396  no_header_counter++;
397  }
398 
399  index--;
400 
401  if (index < 0) {
402  index = BUF_LEN - 1;
403  }
404 
405  }
406 
407  return ret;
408 }

410 void
411 Radar::task_main()
412 {
413  int fd = ::open(_port, O_RDWR | O_NOCTTY);
414 
415  if (fd < 0) {
416  PX4_WARN("serial port not open");
417  }
418 
419  if (!isatty(fd)) {
420  PX4_WARN("not a serial device");
421  }
422 
423 
424  // we poll on data from the serial port
425  pollfd fds[1];
426  fds[0].fd = fd;
427  fds[0].events = POLLIN;
428 
429  // read buffer
430  uint8_t buf[BUF_LEN];
431 
432  while (!_task_should_exit) {
433  // wait for up to 100ms for data
434  int pret = ::poll(fds, (sizeof(fds) / sizeof(fds[0])), 100);
435 
436  // timed out
437  if (pret == 0) {
438  continue;
439  }
440 
441  if (pret < 0) {
442  PX4_DEBUG("radar serial port poll error");
443  // sleep a bit before next try
444  usleep(100000);
445  continue;
446  }
447 
448  if (fds[0].revents & POLLIN) {
449  memset(&buf[0], 0, sizeof(buf));
450  int len = ::read(fd, &buf[0], sizeof(buf));
451 
452  if (len <= 0) {
453  PX4_DEBUG("error reading radar");
454  }
455 
456  float range = 0;
457 
458  if (read_and_parse(&buf[0], len, &range)) {
459 
460  struct distance_sensor_s report = {};
461  report.timestamp = hrt_absolute_time();
462  report.type = distance_sensor_s::MAV_DISTANCE_SENSOR_ULTRASOUND;
463  report.orientation = _rotation;
464  report.current_distance = range;
465  report.current_distance = report.current_distance > ULANDING_MAX_DISTANCE ? ULANDING_MAX_DISTANCE :
466  report.current_distance;
467  report.current_distance = report.current_distance < ULANDING_MIN_DISTANCE ? ULANDING_MIN_DISTANCE :
468  report.current_distance;
469  report.min_distance = ULANDING_MIN_DISTANCE;
470  report.max_distance = ULANDING_MAX_DISTANCE;
471  report.covariance = SENS_VARIANCE;
472  report.type = distance_sensor_s::MAV_DISTANCE_SENSOR_RADAR;
473  report.id = 0;
474 
475  // publish radar data
476  orb_publish(ORB_ID(distance_sensor), _distance_sensor_topic, &report);
477  }
478 
479  }
480  }
481 
482  ::close(fd);
483 }

485 int ulanding_radar_main(int argc, char *argv[])
486 {
487  if (argc <= 1) {
488  PX4_WARN("not enough arguments, usage: start [device_path], stop, info ");
489  return 1;
490  }
491 
492  // check for optional arguments
493  int ch;
494  int myoptind = 1;
495  const char *myoptarg = nullptr;
496  uint8_t rotation = distance_sensor_s::ROTATION_DOWNWARD_FACING;
497 
498 
499  while ((ch = px4_getopt(argc, argv, "R:", &myoptind, &myoptarg)) != EOF) {
500  switch (ch) {
501  case 'R':
502  rotation = (uint8_t)atoi(myoptarg);
503  PX4_INFO("Setting distance sensor orientation to %d", (int)rotation);
504  break;
505 
506  default:
507  PX4_WARN("Unknown option!");
508  }
509  }
510 
511  /*

513  */
514  if (!strcmp(argv[myoptind], "start")) {
515  if (radar::g_dev != nullptr) {
516  PX4_WARN("driver already started");
517  return 0;
518  }
519 
520  if (argc > myoptind + 1) {
521  radar::g_dev = new Radar(rotation, argv[myoptind + 1]);
522 
523  } else {
524  radar::g_dev = new Radar(rotation, RADAR_DEFAULT_PORT);
525  }
526

528  PX4_ERR("failed to create instance of Radar");

530  }

533  delete radar::g_dev;
534  radar::g_dev = nullptr;
535  return 1;
536  }
537

539  delete radar::g_dev;
540  radar::g_dev = nullptr;
541  return 1;
542  }
543 
544  return 0;
545  }
546 
547  /*
548  * Stop the driver
549  */
550  if (!strcmp(argv[myoptind], "stop")) {
551  if (radar::g_dev != nullptr) {
552  delete radar::g_dev;
553  radar::g_dev = nullptr;
554 
555  } else {
556  PX4_WARN("driver not running");

558 
559  return 0;

562  if (!strcmp(argv[myoptind], "info")) {
563  PX4_INFO("ulanding radar from Aerotenna");

565  PX4_INFO("max distance %.2f m", (double)ULANDING_MAX_DISTANCE);
566  PX4_INFO("update rate: 500 Hz");
567

569 
570  }
