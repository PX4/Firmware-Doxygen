295  * Main loop function

297 void
298 PCA9685::i2cpwm()
299 {
300  if (_mode == IOX_MODE_TEST_OUT) {
301  setPin(0, PCA9685_PWMCENTER);
302  _should_run = true;
303 
304  } else if (_mode == IOX_MODE_OFF) {
305  _should_run = false;
306 
307  } else {
308  if (!_mode_on_initialized) {
309  /* Subscribe to actuator control 2 (payload group for gimbal) */
310  _actuator_controls_sub = orb_subscribe(ORB_ID(actuator_controls_2));
311  /* set the uorb update interval lower than the driver pwm interval */
312  orb_set_interval(_actuator_controls_sub, 1000.0f / PCA9685_PWMFREQ - 5);
313 
314  _mode_on_initialized = true;
315  }
316 
317  /* Read the servo setpoints from the actuator control topics (gimbal) */
318  bool updated;
319  orb_check(_actuator_controls_sub, &updated);
320 
321  if (updated) {
322  orb_copy(ORB_ID(actuator_controls_2), _actuator_controls_sub, &_actuator_controls);
323 
324  for (int i = 0; i < actuator_controls_s::NUM_ACTUATOR_CONTROLS; i++) {
325  /* Scale the controls to PWM, first multiply by pi to get rad,
326  * the control[i] values are on the range -1 ... 1 */
327  uint16_t new_value = PCA9685_PWMCENTER +
328  (_actuator_controls.control[i] * M_PI_F * PCA9685_SCALE);
329  DEVICE_DEBUG("%d: current: %u, new %u, control %.2f", i, _current_values[i], new_value,
330  (double)_actuator_controls.control[i]);
331 
332  if (new_value != _current_values[i] &&
333  isfinite(new_value) &&
334  new_value >= PCA9685_PWMMIN &&
335  new_value <= PCA9685_PWMMAX) {
336  /* This value was updated, send the command to adjust the PWM value */
337  setPin(i, new_value);
338  _current_values[i] = new_value;
339  }
340  }
341  }
342 
343  _should_run = true;
344  }
345 
346  // check if any activity remains, else stop
347  if (!_should_run) {
348  _running = false;
349  return;
350  }
351 
352  // re-queue ourselves to run again later
353  _running = true;
354  work_queue(LPWORK, &_work, (worker_t)&PCA9685::i2cpwm_trampoline, this, _i2cpwm_interval);
355 }

359 {
360  int ret;
361  /* convert to correct message */
362  _msg[0] = LED0_ON_L + 4 * num;
363  _msg[1] = on;
364  _msg[2] = on >> 8;
365  _msg[3] = off;
366  _msg[4] = off >> 8;
367 
368  /* try i2c transfer */

370 
371  if (OK != ret) {
372  perf_count(_comms_errors);

374  }
375

379 int
380 PCA9685::setPin(uint8_t num, uint16_t val, bool invert)
381 {
382  // Clamp value between 0 and 4095 inclusive.
383  if (val > 4095) {
384  val = 4095;
385  }
386 
387  if (invert) {
388  if (val == 0) {
389  // Special value for signal fully on.
390  return setPWM(num, 4096, 0);
391 
392  } else if (val == 4095) {
393  // Special value for signal fully off.
394  return setPWM(num, 0, 4096);
395 
396  } else {
397  return setPWM(num, 0, 4095 - val);
398  }
399 
400  } else {
401  if (val == 4095) {
402  // Special value for signal fully on.
403  return setPWM(num, 4096, 0);
404 
405  } else if (val == 0) {
406  // Special value for signal fully off.
407  return setPWM(num, 0, 4096);
408 
409  } else {
410  return setPWM(num, 0, val);
411  }
412  }
413 
414  return PX4_ERROR;
415 }

417 int
418 PCA9685::setPWMFreq(float freq)
419 {
420  int ret = OK;
421  freq *= 0.9f; /* Correct for overshoot in the frequency setting (see issue
422  https://github.com/adafruit/Adafruit-PWM-Servo-Driver-Library/issues/11). */
423  float prescaleval = 25000000;
424  prescaleval /= 4096;
425  prescaleval /= freq;
426  prescaleval -= 1;
427  uint8_t prescale = uint8_t(prescaleval + 0.5f); //implicit floor()
428  uint8_t oldmode;
429  ret = read8(PCA9685_MODE1, oldmode);
430 
431  if (ret != OK) {
432  return ret;
433  }
434 
435  uint8_t newmode = (oldmode & 0x7F) | 0x10; // sleep
436 
437  ret = write8(PCA9685_MODE1, newmode); // go to sleep
438 
439  if (ret != OK) {
440  return ret;
441  }
442 
443  ret = write8(PCA9685_PRESCALE, prescale); // set the prescaler
444 
445  if (ret != OK) {
446  return ret;
447  }
448 
449  ret = write8(PCA9685_MODE1, oldmode);
450 
451  if (ret != OK) {
452  return ret;
453  }
454 
455  usleep(5000); //5ms delay (from arduino driver)
456 
457  ret = write8(PCA9685_MODE1, oldmode | 0xa1); // This sets the MODE1 register to turn on auto increment.
458 
459  if (ret != OK) {
460  return ret;
461  }
462 
463  return ret;
464 }
