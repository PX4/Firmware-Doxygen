63 int uORBTest::UnitTest::pubsublatency_main()
64 {
65  /* poll on test topic and output latency */
66  float latency_integral = 0.0f;
67 
68  /* wakeup source(s) */
69  px4_pollfd_struct_t fds[3];
70 
71  int test_multi_sub = orb_subscribe_multi(ORB_ID(orb_test), 0);
72  int test_multi_sub_medium = orb_subscribe_multi(ORB_ID(orb_test_medium), 0);
73  int test_multi_sub_large = orb_subscribe_multi(ORB_ID(orb_test_large), 0);
74 
75  struct orb_test_large t;
76 
77  /* clear all ready flags */
78  orb_copy(ORB_ID(orb_test), test_multi_sub, &t);
79  orb_copy(ORB_ID(orb_test_medium), test_multi_sub_medium, &t);
80  orb_copy(ORB_ID(orb_test_large), test_multi_sub_large, &t);
81 
82  fds[0].fd = test_multi_sub;
83  fds[0].events = POLLIN;
84  fds[1].fd = test_multi_sub_medium;
85  fds[1].events = POLLIN;
86  fds[2].fd = test_multi_sub_large;
87  fds[2].events = POLLIN;
88 
89  const unsigned maxruns = 1000;
90  unsigned timingsgroup = 0;
91  int current_value = t.val;
92  int num_missed = 0;
93 
94  // timings has to be on the heap to keep frame size below 2048 bytes
95  unsigned *timings = new unsigned[maxruns];
96  unsigned timing_min = 9999999, timing_max = 0;
97 
98  for (unsigned i = 0; i < maxruns; i++) {
99  /* wait for up to 500ms for data */
100  int pret = px4_poll(&fds[0], (sizeof(fds) / sizeof(fds[0])), 500);
101 
102  if (fds[0].revents & POLLIN) {
103  orb_copy(ORB_ID(orb_test), test_multi_sub, &t);
104  timingsgroup = 0;
105 
106  } else if (fds[1].revents & POLLIN) {
107  orb_copy(ORB_ID(orb_test_medium), test_multi_sub_medium, &t);
108  timingsgroup = 1;
109 
110  } else if (fds[2].revents & POLLIN) {
111  orb_copy(ORB_ID(orb_test_large), test_multi_sub_large, &t);
112  timingsgroup = 2;
113  }
114 
115  if (pret < 0) {
116  PX4_ERR("poll error %d, %d", pret, errno);
117  continue;
118  }
119 
120  num_missed += t.val - current_value - 1;
121  current_value = t.val;
122 
123  unsigned elt = (unsigned)hrt_elapsed_time(&t.time);
124  latency_integral += elt;
125  timings[i] = elt;
126 
127  if (elt > timing_max) {
128  timing_max = elt;
129  }
130 
131  if (elt < timing_min) {
132  timing_min = elt;
133  }
134  }
135 
136  orb_unsubscribe(test_multi_sub);
137  orb_unsubscribe(test_multi_sub_medium);
138  orb_unsubscribe(test_multi_sub_large);
139 
140  if (pubsubtest_print) {
141  char fname[32];
142  sprintf(fname, PX4_STORAGEDIR"/uorb_timings%u.txt", timingsgroup);
143  FILE *f = fopen(fname, "w");
144 
145  if (f == nullptr) {
146  PX4_ERR("Error opening file!");
147  delete[] timings;
148  return PX4_ERROR;
149  }
150 
151  for (unsigned i = 0; i < maxruns; i++) {
152  fprintf(f, "%u\n", timings[i]);
153  }
154 
155  fclose(f);
156  }
157 
158 
159  float std_dev = 0.f;
160  float mean = latency_integral / maxruns;
161 
162  for (unsigned i = 0; i < maxruns; i++) {
163  float diff = (float)timings[i] - mean;
164  std_dev += diff * diff;
165  }
166 
167  delete[] timings;
168 
169  PX4_INFO("mean: %8.4f us", static_cast<double>(mean));
170  PX4_INFO("std dev: %8.4f us", static_cast<double>(sqrtf(std_dev / (maxruns - 1))));
171  PX4_INFO("min: %3i us", timing_min);
172  PX4_INFO("max: %3i us", timing_max);
173  PX4_INFO("missed topic updates: %i", num_missed);
174 
175  pubsubtest_passed = true;
176 
177  if (static_cast<float>(latency_integral / maxruns) > 100.0f) {
178  pubsubtest_res = PX4_ERROR;
179 
180  } else {
181  pubsubtest_res = PX4_OK;
182  }
183 
184  return pubsubtest_res;
185 }
