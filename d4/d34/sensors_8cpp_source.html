100 /**
101  * Analog layout:
102  * FMU:
103  * IN2 - battery voltage
104  * IN3 - battery current
105  * IN4 - 5V sense
106  * IN10 - spare (we could actually trim these from the set)
107  * IN11 - spare on FMUv2 & v3, RC RSSI on FMUv4
108  * IN12 - spare (we could actually trim these from the set)
109  * IN13 - aux1 on FMUv2, unavaible on v3 & v4
110  * IN14 - aux2 on FMUv2, unavaible on v3 & v4
111  * IN15 - pressure sensor on FMUv2, unavaible on v3 & v4
112  *
113  * IO:
114  * IN4 - servo supply rail

116  *
117  * The channel definitions (e.g., ADC_BATTERY_VOLTAGE_CHANNEL, ADC_BATTERY_CURRENT_CHANNEL, and ADC_AIRSPEED_VOLTAGE_CHANNEL) are defined in board_config.h

298 void
299 Sensors::diff_pres_poll(const vehicle_air_data_s &raw)
300 {
301  bool updated;
302  orb_check(_diff_pres_sub, &updated);
303 
304  if (updated) {
305  differential_pressure_s diff_pres;
306  int ret = orb_copy(ORB_ID(differential_pressure), _diff_pres_sub, &diff_pres);
307 
308  if (ret != PX4_OK) {
309  return;
310  }
311 
312  float air_temperature_celsius = (diff_pres.temperature > -300.0f) ? diff_pres.temperature :
313  (raw.baro_temp_celcius - PCB_TEMP_ESTIMATE_DEG);
314 
315  airspeed_s airspeed;
316  airspeed.timestamp = diff_pres.timestamp;
317 
318  /* push data into validator */
319  float airspeed_input[3] = { diff_pres.differential_pressure_raw_pa, diff_pres.temperature, 0.0f };
320 
321  _airspeed_validator.put(airspeed.timestamp, airspeed_input, diff_pres.error_count,
322  ORB_PRIO_HIGH);
323 
324  airspeed.confidence = _airspeed_validator.confidence(hrt_absolute_time());
325 
326  enum AIRSPEED_SENSOR_MODEL smodel;
327 
328  switch ((diff_pres.device_id >> 16) & 0xFF) {
329  case DRV_DIFF_PRESS_DEVTYPE_SDP31:
330 
331  /* fallthrough */
332  case DRV_DIFF_PRESS_DEVTYPE_SDP32:
333 
334  /* fallthrough */
335  case DRV_DIFF_PRESS_DEVTYPE_SDP33:
336  /* fallthrough */
337  smodel = AIRSPEED_SENSOR_MODEL_SDP3X;
338  break;
339 
340  default:
341  smodel = AIRSPEED_SENSOR_MODEL_MEMBRANE;
342  break;
343  }
344 
345  /* don't risk to feed negative airspeed into the system */
346  airspeed.indicated_airspeed_m_s = calc_indicated_airspeed_corrected((enum AIRSPEED_COMPENSATION_MODEL)
347  _parameters.air_cmodel,
348  smodel, _parameters.air_tube_length, _parameters.air_tube_diameter_mm,
349  diff_pres.differential_pressure_filtered_pa, raw.baro_pressure_pa,
350  air_temperature_celsius);
351 
352  airspeed.true_airspeed_m_s = calc_true_airspeed_from_indicated(airspeed.indicated_airspeed_m_s, raw.baro_pressure_pa,
353  air_temperature_celsius);
354 
355  airspeed.air_temperature_celsius = air_temperature_celsius;
356 
357  if (PX4_ISFINITE(airspeed.indicated_airspeed_m_s) && PX4_ISFINITE(airspeed.true_airspeed_m_s)) {
358  int instance;
359  orb_publish_auto(ORB_ID(airspeed), &_airspeed_pub, &airspeed, &instance, ORB_PRIO_DEFAULT);
360  }
361  }

414 void
415 Sensors::adc_poll()
416 {
417  /* only read if not in HIL mode */
418  if (_hil_enabled) {
419  return;
420  }
421 
422  hrt_abstime t = hrt_absolute_time();
423 
424  /* rate limit to 100 Hz */
425  if (t - _last_adc >= 10000) {
426  /* make space for a maximum of twelve channels (to ensure reading all channels at once) */
427  px4_adc_msg_t buf_adc[PX4_MAX_ADC_CHANNELS];
428  /* read all channels available */
429  int ret = _h_adc.read(&buf_adc, sizeof(buf_adc));
430 
431 #if BOARD_NUMBER_BRICKS > 0
432  //todo:abosorb into new class Power
433 
434  /* For legacy support we publish the battery_status for the Battery that is
435  * associated with the Brick that is the selected source for VDD_5V_IN
436  * Selection is done in HW ala a LTC4417 or similar, or may be hard coded
437  * Like in the FMUv4
438  */
439 
440  /* The ADC channels that are associated with each brick, in power controller
441  * priority order highest to lowest, as defined by the board config.
442  */
443  int bat_voltage_v_chan[BOARD_NUMBER_BRICKS] = BOARD_BATT_V_LIST;
444  int bat_voltage_i_chan[BOARD_NUMBER_BRICKS] = BOARD_BATT_I_LIST;
445 
446  if (_parameters.battery_adc_channel >= 0) { // overwrite default
447  bat_voltage_v_chan[0] = _parameters.battery_adc_channel;
448  }
449 
450  /* The valid signals (HW dependent) are associated with each brick */
451  bool valid_chan[BOARD_NUMBER_BRICKS] = BOARD_BRICK_VALID_LIST;
452 
453  /* Per Brick readings with default unread channels at 0 */
454  float bat_current_a[BOARD_NUMBER_BRICKS] = {0.0f};
455  float bat_voltage_v[BOARD_NUMBER_BRICKS] = {0.0f};
456 
457  /* Based on the valid_chan, used to indicate the selected the lowest index
458  * (highest priority) supply that is the source for the VDD_5V_IN
459  * When < 0 none selected
460  */
461 
462  int selected_source = -1;
463 
464 #endif /* BOARD_NUMBER_BRICKS > 0 */
465 
466  if (ret >= (int)sizeof(buf_adc[0])) {
467 
468  /* Read add channels we got */
469  for (unsigned i = 0; i < ret / sizeof(buf_adc[0]); i++) {
470 #ifdef ADC_AIRSPEED_VOLTAGE_CHANNEL
471 
472  if (ADC_AIRSPEED_VOLTAGE_CHANNEL == buf_adc[i].am_channel) {
473 
474  /* calculate airspeed, raw is the difference from */
475  const float voltage = (float)(buf_adc[i].am_data) * 3.3f / 4096.0f * 2.0f; // V_ref/4096 * (voltage divider factor)
476 
477  /**
478  * The voltage divider pulls the signal down, only act on
479  * a valid voltage from a connected sensor. Also assume a non-
480  * zero offset from the sensor if its connected.
481  */
482  if (voltage > 0.4f && (_parameters.diff_pres_analog_scale > 0.0f)) {
483 
484  const float diff_pres_pa_raw = voltage * _parameters.diff_pres_analog_scale - _parameters.diff_pres_offset_pa;
485 
486  _diff_pres.timestamp = t;
487  _diff_pres.differential_pressure_raw_pa = diff_pres_pa_raw;
488  _diff_pres.differential_pressure_filtered_pa = (_diff_pres.differential_pressure_filtered_pa * 0.9f) +
489  (diff_pres_pa_raw * 0.1f);
490  _diff_pres.temperature = -1000.0f;
491 
492  int instance;
493  orb_publish_auto(ORB_ID(differential_pressure), &_diff_pres_pub, &_diff_pres, &instance, ORB_PRIO_DEFAULT);
494  }
495 
496  } else
497 #endif /* ADC_AIRSPEED_VOLTAGE_CHANNEL */
498  {
499 
500 #if BOARD_NUMBER_BRICKS > 0
501 
502  for (int b = 0; b < BOARD_NUMBER_BRICKS; b++) {
503 
504  /* Once we have subscriptions, Do this once for the lowest (highest priority
505  * supply on power controller) that is valid.
506  */
507  if (_battery_pub[b] != nullptr && selected_source < 0 && valid_chan[b]) {
508  /* Indicate the lowest brick (highest priority supply on power controller)
509  * that is valid as the one that is the selected source for the
510  * VDD_5V_IN
511  */
512  selected_source = b;
513 
514 #if BOARD_NUMBER_BRICKS > 1
515 
516  /* Move the selected_source to instance 0 */
517  if (_battery_pub_intance0ndx != selected_source) {
518 
519  orb_advert_t tmp_h = _battery_pub[_battery_pub_intance0ndx];
520  _battery_pub[_battery_pub_intance0ndx] = _battery_pub[selected_source];
521  _battery_pub[selected_source] = tmp_h;
522  _battery_pub_intance0ndx = selected_source;
523  }
524 
525 #endif /* BOARD_NUMBER_BRICKS > 1 */
526  }
527 
528  // todo:per brick scaling
529  /* look for specific channels and process the raw voltage to measurement data */
530  if (bat_voltage_v_chan[b] == buf_adc[i].am_channel) {
531  /* Voltage in volts */
532  bat_voltage_v[b] = (buf_adc[i].am_data * _parameters.battery_voltage_scaling) * _parameters.battery_v_div;
533 
534  } else if (bat_voltage_i_chan[b] == buf_adc[i].am_channel) {
535  bat_current_a[b] = ((buf_adc[i].am_data * _parameters.battery_current_scaling)
536  - _parameters.battery_current_offset) * _parameters.battery_a_per_v;
537  }
538  }
539 
540 #endif /* BOARD_NUMBER_BRICKS > 0 */
541  }
542  }
543 
544 #if BOARD_NUMBER_BRICKS > 0
545 
546  if (_parameters.battery_source == 0) {
547 
548  for (int b = 0; b < BOARD_NUMBER_BRICKS; b++) {
549 
550  /* Consider the brick connected if there is a voltage */
551  bool connected = bat_voltage_v[b] > BOARD_ADC_OPEN_CIRCUIT_V;
552 
553  /* In the case where the BOARD_ADC_OPEN_CIRCUIT_V is
554  * greater than the BOARD_VALID_UV let the HW qualify that it
555  * is connected.
556  */
557  if (BOARD_ADC_OPEN_CIRCUIT_V > BOARD_VALID_UV) {
558  connected &= valid_chan[b];
559  }
560 
561  actuator_controls_s ctrl;
562  orb_copy(ORB_ID(actuator_controls_0), _actuator_ctrl_0_sub, &ctrl);
563 
564  battery_status_s battery_status;
565  _battery[b].updateBatteryStatus(t, bat_voltage_v[b], bat_current_a[b],
566  connected, selected_source == b, b,
567  ctrl.control[actuator_controls_s::INDEX_THROTTLE],
568  _armed, &battery_status);
569  int instance;
570  orb_publish_auto(ORB_ID(battery_status), &_battery_pub[b], &battery_status, &instance, ORB_PRIO_DEFAULT);
571  }
572  }
573 
574 #endif /* BOARD_NUMBER_BRICKS > 0 */
575 
576  _last_adc = t;
577  }
578  }

582 void
583 Sensors::run()
584 {
585  if (!_hil_enabled) {
586 #if !defined(__PX4_QURT) && BOARD_NUMBER_BRICKS > 0
587  adc_init();
588 #endif
589  }
590 
591  sensor_combined_s raw = {};
592  sensor_preflight_s preflt = {};
593  vehicle_air_data_s airdata = {};
594  vehicle_magnetometer_s magnetometer = {};
595 
596  _rc_update.init();
597  _voted_sensors_update.init(raw);
598 
599  /* (re)load params and calibration */
600  parameter_update_poll(true);
601 
602  /*
603  * do subscriptions
604  */
605  _diff_pres_sub = orb_subscribe(ORB_ID(differential_pressure));
606  _vcontrol_mode_sub = orb_subscribe(ORB_ID(vehicle_control_mode));
607  _params_sub = orb_subscribe(ORB_ID(parameter_update));
608  _actuator_ctrl_0_sub = orb_subscribe(ORB_ID(actuator_controls_0));
609 
610  /* get a set of initial values */
611  _voted_sensors_update.sensors_poll(raw, airdata, magnetometer);
612 
613  diff_pres_poll(airdata);
614 
615  _rc_update.rc_parameter_map_poll(_parameter_handles, true /* forced */);
616 
617  _sensor_preflight = orb_advertise(ORB_ID(sensor_preflight), &preflt);
618 
619  /* wakeup source */
620  px4_pollfd_struct_t poll_fds = {};
621  poll_fds.events = POLLIN;
622 
623  uint64_t last_config_update = hrt_absolute_time();
624 
625  while (!should_exit()) {
626 
627  /* use the best-voted gyro to pace output */
628  poll_fds.fd = _voted_sensors_update.best_gyro_fd();
629 
630  /* wait for up to 50ms for data (Note that this implies, we can have a fail-over time of 50ms,
631  * if a gyro fails) */
632  int pret = px4_poll(&poll_fds, 1, 50);
633 
634  /* if pret == 0 it timed out - periodic check for should_exit(), etc. */
635 
636  /* this is undesirable but not much we can do - might want to flag unhappy status */
637  if (pret < 0) {
638  /* if the polling operation failed because no gyro sensor is available yet,
639  * then attempt to subscribe once again
640  */
641  if (_voted_sensors_update.num_gyros() == 0) {
642  _voted_sensors_update.initialize_sensors();
643  }
644 
645  usleep(1000);
646

648  }
649 
650  perf_begin(_loop_perf);
651 
652  /* check vehicle status for changes to publication state */
653  vehicle_control_mode_poll();
654 
655  /* the timestamp of the raw struct is updated by the gyro_poll() method (this makes the gyro
656  * a mandatory sensor) */
657  const uint64_t airdata_prev_timestamp = airdata.timestamp;
658  const uint64_t magnetometer_prev_timestamp = magnetometer.timestamp;

661 
662  /* check battery voltage */
663  adc_poll();
664 
665  diff_pres_poll(airdata);
666 
667  if (raw.timestamp > 0) {
668 
669  _voted_sensors_update.set_relative_timestamps(raw);
670 
671  int instance;
672  orb_publish_auto(ORB_ID(sensor_combined), &_sensor_pub, &raw, &instance, ORB_PRIO_DEFAULT);
673

675  orb_publish_auto(ORB_ID(vehicle_air_data), &_airdata_pub, &airdata, &instance, ORB_PRIO_DEFAULT);
676  }
677 
678  if (magnetometer.timestamp != magnetometer_prev_timestamp) {
679  orb_publish_auto(ORB_ID(vehicle_magnetometer), &_magnetometer_pub, &magnetometer, &instance, ORB_PRIO_DEFAULT);
680  }
681 
682  _voted_sensors_update.check_failover();
683 
684  /* If the the vehicle is disarmed calculate the length of the maximum difference between
685  * IMU units as a consistency metric and publish to the sensor preflight topic
686  */
687  if (!_armed) {
688  preflt.timestamp = hrt_absolute_time();
689  _voted_sensors_update.calc_accel_inconsistency(preflt);
690  _voted_sensors_update.calc_gyro_inconsistency(preflt);
691  _voted_sensors_update.calc_mag_inconsistency(preflt);
692  orb_publish(ORB_ID(sensor_preflight), _sensor_preflight, &preflt);
693  }
694  }
695 
696  /* keep adding sensors as long as we are not armed,
697  * when not adding sensors poll for param updates
698  */
699  if (!_armed && hrt_elapsed_time(&last_config_update) > 500_ms) {
700  _voted_sensors_update.initialize_sensors();
701  last_config_update = hrt_absolute_time();
702 
703  } else {
704 
705  /* check parameters for updates */
706  parameter_update_poll();
707 
708  /* check rc parameter map for updates */
709  _rc_update.rc_parameter_map_poll(_parameter_handles);
710  }
711 
712  /* Look for new r/c input data */
713  _rc_update.rc_poll(_parameter_handles);
714 
715  perf_end(_loop_perf);
716  }
717 
718  orb_unsubscribe(_diff_pres_sub);
719  orb_unsubscribe(_vcontrol_mode_sub);
720  orb_unsubscribe(_params_sub);
721  orb_unsubscribe(_actuator_ctrl_0_sub);
722 
723  if (_sensor_pub) {
724  orb_unadvertise(_sensor_pub);
725  }
726 
727  if (_airdata_pub) {
728  orb_unadvertise(_airdata_pub);
729  }
730 
731  if (_magnetometer_pub) {
732  orb_unadvertise(_magnetometer_pub);
733  }
734 
735  _rc_update.deinit();
736  _voted_sensors_update.deinit();
737 }
