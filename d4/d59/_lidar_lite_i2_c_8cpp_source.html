170 int LidarLiteI2C::probe()
171 {
172  // cope with both old and new I2C bus address
173  const uint8_t addresses[2] = {LL40LS_BASEADDR, LL40LS_BASEADDR_OLD};
174 
175  // more retries for detection
176  _retries = 10;
177 
178  for (uint8_t i = 0; i < sizeof(addresses); i++) {
179  /*
180  check for hw and sw versions. It would be better if
181  we had a proper WHOAMI register
182  */
183  if (read_reg(LL40LS_HW_VERSION, _hw_version) == OK && _hw_version > 0 &&
184  read_reg(LL40LS_SW_VERSION, _sw_version) == OK && _sw_version > 0) {
185  goto ok;
186  }
187 
188  DEVICE_DEBUG("probe failed hw_version=0x%02x sw_version=0x%02x\n",
189  (unsigned)_hw_version,
190  (unsigned)_sw_version);
191  }
192 
193  // not found on any address
194  return -EIO;
195 
196 ok:
197  _retries = 3;
198  return reset_sensor();
199 }

234 ssize_t LidarLiteI2C::read(device::file_t *filp, char *buffer, size_t buflen)
235 {
236  unsigned count = buflen / sizeof(struct distance_sensor_s);
237  struct distance_sensor_s *rbuf = reinterpret_cast<struct distance_sensor_s *>(buffer);
238  int ret = 0;
239 
240  /* buffer must be large enough */
241  if (count < 1) {
242  return -ENOSPC;
243  }
244 
245  /* if automatic measurement is enabled */
246  if (getMeasureTicks() > 0) {
247 
248  /*
249  * While there is space in the caller's buffer, and reports, copy them.
250  * Note that we may be pre-empted by the workq thread while we are doing this;
251  * we are careful to avoid racing with them.
252  */
253  while (count--) {
254  if (_reports->get(rbuf)) {
255  ret += sizeof(*rbuf);
256  rbuf++;
257  }
258  }
259 
260  /* if there was no data, warn the caller */
261  return ret ? ret : -EAGAIN;
262  }
263 
264  /* manual measurement - run one conversion */
265  do {
266  _reports->flush();
267 
268  /* trigger a measurement */
269  if (OK != measure()) {
270  ret = -EIO;
271  break;
272  }
273 
274  /* wait for it to complete */
275  usleep(LL40LS_CONVERSION_INTERVAL);
276 
277  /* run the collection phase */
278  if (OK != collect()) {
279  ret = -EIO;
280  break;
281  }
282 
283  /* state machine will have generated a report, copy it out */
284  if (_reports->get(rbuf)) {
285  ret = sizeof(*rbuf);
286  }
287 
288  } while (0);
289 
290  return ret;
291 }

293 int LidarLiteI2C::measure()
294 {
295  int ret;
296 
297  if (_pause_measurements) {
298  // we are in print_registers() and need to avoid
299  // acquisition to keep the I2C peripheral on the
300  // sensor active
301  return OK;
302  }
303 
304  /*
305  * Send the command to begin a measurement.
306  */
307  const uint8_t cmd[2] = { LL40LS_MEASURE_REG, LL40LS_MSRREG_ACQUIRE };
308  ret = lidar_transfer(cmd, sizeof(cmd), nullptr, 0);
309 
310  if (OK != ret) {
311  perf_count(_comms_errors);
312  DEVICE_DEBUG("i2c::transfer returned %d", ret);
313 
314  // if we are getting lots of I2C transfer errors try
315  // resetting the sensor
316  if (perf_event_count(_comms_errors) % 10 == 0) {
317  perf_count(_sensor_resets);
318  reset_sensor();
319  }
320 
321  return ret;
322  }
323 
324  // remember when we sent the acquire so we can know when the
325  // acquisition has timed out
326  _acquire_time_usec = hrt_absolute_time();
327  ret = OK;
328 
329  return ret;
330 }

335 int LidarLiteI2C::reset_sensor()
336 {
337  int ret;
338  ret = write_reg(LL40LS_MEASURE_REG, LL40LS_MSRREG_RESET);
339 
340  if (ret != OK) {
341  return ret;
342  }
343 
344  // wait for sensor reset to complete
345  usleep(50000);
346  ret = write_reg(LL40LS_SIG_COUNT_VAL_REG, LL40LS_SIG_COUNT_VAL_MAX);
347 
348  if (ret != OK) {
349  return ret;
350  }
351 
352  // wait for register write to complete
353  usleep(1000);
354 
355  return OK;

388 int LidarLiteI2C::collect()
389 {
390  int ret = -EIO;
391 
392  /* read from the sensor */
393  uint8_t val[2] = {0, 0};
394 
395  perf_begin(_sample_perf);
396 
397  // read the high and low byte distance registers
398  uint8_t distance_reg = LL40LS_DISTHIGH_REG | LL40LS_AUTO_INCREMENT;
399  ret = lidar_transfer(&distance_reg, 1, &val[0], sizeof(val));
400 
401  // if the transfer failed or if the high bit of distance is
402  // set then the distance is invalid
403  if (ret < 0 || (val[0] & 0x80)) {
404  if (hrt_absolute_time() - _acquire_time_usec > LL40LS_CONVERSION_TIMEOUT) {
405  /*
406  NACKs from the sensor are expected when we
407  read before it is ready, so only consider it
408  an error if more than 100ms has elapsed.
409  */
410  DEVICE_DEBUG("error reading from sensor: %d", ret);
411  perf_count(_comms_errors);
412 
413  if (perf_event_count(_comms_errors) % 10 == 0) {
414  perf_count(_sensor_resets);
415  reset_sensor();
416  }
417  }
418 
419  perf_end(_sample_perf);
420  // if we are getting lots of I2C transfer errors try
421  // resetting the sensor
422  return ret;
423  }
424 
425  uint16_t distance_cm = (val[0] << 8) | val[1];
426  float distance_m = float(distance_cm) * 1e-2f;
427  struct distance_sensor_s report;
428 
429  if (distance_cm == 0) {
430  _zero_counter++;
431 
432  if (_zero_counter == 20) {
433  /* we have had 20 zeros in a row - reset the
434  sensor. This is a known bad state of the
435  sensor where it returns 16 bits of zero for
436  the distance with a trailing NACK, and
437  keeps doing that even when the target comes
438  into a valid range.
439  */
440  _zero_counter = 0;
441  perf_end(_sample_perf);
442  perf_count(_sensor_zero_resets);
443  return reset_sensor();
444  }
445 
446  } else {
447  _zero_counter = 0;
448  }
449 
450  _last_distance = distance_cm;
451 
452  /* Relative signal strength measurement, i.e. the strength of
453  * the main signal peak compared to the general noise level*/
454  uint8_t signal_strength_reg = LL40LS_SIGNAL_STRENGTH_REG;
455  ret = lidar_transfer(&signal_strength_reg, 1, &val[0], 1);
456 
457  // check if the transfer failed
458  if (ret < 0) {
459  if (hrt_absolute_time() - _acquire_time_usec > LL40LS_CONVERSION_TIMEOUT) {
460  /*
461  NACKs from the sensor are expected when we
462  read before it is ready, so only consider it
463  an error if more than 100ms has elapsed.
464  */
465  PX4_INFO("signal strength read failed");
466 
467  DEVICE_DEBUG("error reading signal strength from sensor: %d", ret);
468  perf_count(_comms_errors);
469 
470  if (perf_event_count(_comms_errors) % 10 == 0) {
471  perf_count(_sensor_resets);
472  reset_sensor();
473  }
474  }
475 
476  perf_end(_sample_perf);
477  // if we are getting lots of I2C transfer errors try
478  // resetting the sensor
479  return ret;
480  }
481 
482  uint8_t ll40ls_signal_strength = val[0];
483 
484 
485  /* Absolute peak strength measurement, i.e. absolute strength of main signal peak*/
486  uint8_t peak_strength_reg = LL40LS_PEAK_STRENGTH_REG;
487  ret = lidar_transfer(&peak_strength_reg, 1, &val[0], 1);
488 
489  // check if the transfer failed
490  if (ret < 0) {
491  if (hrt_absolute_time() - _acquire_time_usec > LL40LS_CONVERSION_TIMEOUT) {
492  /*
493  NACKs from the sensor are expected when we
494  read before it is ready, so only consider it
495  an error if more than 100ms has elapsed.
496  */
497  PX4_INFO("peak strenght read failed");
498 
499  DEVICE_DEBUG("error reading peak strength from sensor: %d", ret);
500  perf_count(_comms_errors);
501 
502  if (perf_event_count(_comms_errors) % 10 == 0) {
503  perf_count(_sensor_resets);
504  reset_sensor();
505  }
506  }
507 
508  perf_end(_sample_perf);
509  // if we are getting lots of I2C transfer errors try
510  // resetting the sensor
511  return ret;
512  }
513 
514  uint8_t ll40ls_peak_strength = val[0];
515 
516  /* Final data quality evaluation. This is based on the datasheet and simple heuristics retrieved from experiments*/
517  // Step 1: Normalize signal strength to 0...100 percent using the absolute signal peak strength.
518  uint8_t signal_quality = 100 * math::max(ll40ls_peak_strength - LL40LS_PEAK_STRENGTH_LOW,
519  0) / (LL40LS_PEAK_STRENGTH_HIGH - LL40LS_PEAK_STRENGTH_LOW);
520 
521  // Step 2: Also use ll40ls_signal_strength (a relative measure, i.e. peak strength to noise!) to reject potentially ambiguous measurements
522  if (ll40ls_signal_strength <= LL40LS_SIGNAL_STRENGTH_LOW) { signal_quality = 0; }
523 
524  // Step 3: Filter physically impossible measurements, which removes some crazy outliers that appear on LL40LS.
525  if (distance_m < LL40LS_MIN_DISTANCE) { signal_quality = 0; }
526 
527  /* this should be fairly close to the end of the measurement, so the best approximation of the time */
528  report.timestamp = hrt_absolute_time();
529  report.current_distance = distance_m;
530  report.min_distance = get_minimum_distance();
531  report.max_distance = get_maximum_distance();
532  report.covariance = 0.0f;
533  report.signal_quality = signal_quality;
534  report.type =
535  distance_sensor_s::MAV_DISTANCE_SENSOR_LASER; // the sensor is in fact a laser + sonar but there is no enum for this
536  report.orientation = _rotation;
537  report.id = 0; // TODO: set proper ID
538 
539  /* publish it, if we are the primary */
540  if (_distance_sensor_topic != nullptr) {
541  orb_publish(ORB_ID(distance_sensor), _distance_sensor_topic, &report);
542  }
543 
544  _reports->force(&report);
545 
546  /* notify anyone waiting for data */
547  poll_notify(POLLIN);
548 
549  ret = OK;
550 
551  perf_end(_sample_perf);
552  return ret;
553 }

577 void LidarLiteI2C::cycle()
578 {
579  /* collection phase? */
580  if (_collect_phase) {
581 
582  /* try a collection */
583  if (OK != collect()) {
584  DEVICE_DEBUG("collection error");
585 
586  /* if we've been waiting more than 200ms then
587  send a new acquire */
588  if (hrt_absolute_time() - _acquire_time_usec > LL40LS_CONVERSION_TIMEOUT * 2) {
589  _collect_phase = false;
590  }
591 
592  } else {
593  /* next phase is measurement */
594  _collect_phase = false;
595 
596  /*
597  * Is there a collect->measure gap?
598  */
599  if (getMeasureTicks() > USEC2TICK(LL40LS_CONVERSION_INTERVAL)) {
600 
601  /* schedule a fresh cycle call when we are ready to measure again */
602  work_queue(HPWORK,
603  &_work,
604  (worker_t)&LidarLiteI2C::cycle_trampoline,
605  this,
606  getMeasureTicks() - USEC2TICK(LL40LS_CONVERSION_INTERVAL));
607 
608  return;
609  }
610  }
611  }
612 
613  if (_collect_phase == false) {
614  /* measurement phase */
615  if (OK != measure()) {
616  DEVICE_DEBUG("measure error");
617 
618  } else {
619  /* next phase is collection. Don't switch to
620  collection phase until we have a successful
621  acquire request I2C transfer */
622  _collect_phase = true;
623  }
624  }
625 
626  /* schedule a fresh cycle call when the measurement is done */
627  work_queue(HPWORK,
628  &_work,
629  (worker_t)&LidarLiteI2C::cycle_trampoline,
630  this,
631  USEC2TICK(LL40LS_CONVERSION_INTERVAL));
632 }
