56  tune_control_s::ORB_QUEUE_LENGTH);
57 
58  if (argc == 1) {
59  PX4_INFO("Volume silenced for testing predefined tunes 0-20.");

61

63  tune_control.tune_id = i;

65 
66  if (result != PX4_OK) {

68  return result;
69  }

73  }
74

80  }
