285 pthread_addr_t UavcanServers::run(pthread_addr_t)
286 {
287  prctl(PR_SET_NAME, "uavcan fw srv", 0);
288 
289  Lock lock(_subnode_mutex);
290 
291  /*
292  Copy any firmware bundled in the ROMFS to the appropriate location on the
293  SD card, unless the user has copied other firmware for that device.
294  */
295  unpackFwFromROMFS(UAVCAN_FIRMWARE_PATH, UAVCAN_ROMFS_FW_PATH);
296 
297  /* the subscribe call needs to happen in the same thread,
298  * so not in the constructor */
299  int cmd_sub = orb_subscribe(ORB_ID(vehicle_command));
300  int param_request_sub = orb_subscribe(ORB_ID(uavcan_parameter_request));
301  int armed_sub = orb_subscribe(ORB_ID(actuator_armed));
302 
303  /* Set up shared service clients */
304  _param_getset_client.setCallback(GetSetCallback(this, &UavcanServers::cb_getset));
305  _param_opcode_client.setCallback(ExecuteOpcodeCallback(this, &UavcanServers::cb_opcode));
306  _param_restartnode_client.setCallback(RestartNodeCallback(this, &UavcanServers::cb_restart));
307  _enumeration_client.setCallback(EnumerationBeginCallback(this, &UavcanServers::cb_enumeration_begin));
308  _enumeration_indication_sub.start(EnumerationIndicationCallback(this, &UavcanServers::cb_enumeration_indication));
309  _enumeration_getset_client.setCallback(GetSetCallback(this, &UavcanServers::cb_enumeration_getset));
310  _enumeration_save_client.setCallback(ExecuteOpcodeCallback(this, &UavcanServers::cb_enumeration_save));
311 
312  _count_in_progress = _param_in_progress = _param_list_in_progress = _cmd_in_progress = _param_list_all_nodes = false;
313  memset(_param_counts, 0, sizeof(_param_counts));
314 
315  _esc_enumeration_active = false;
316  memset(_esc_enumeration_ids, 0, sizeof(_esc_enumeration_ids));
317  _esc_enumeration_index = 0;
318 
319  while (!_subnode_thread_should_exit) {
320 
321  if (_check_fw == true) {
322  _check_fw = false;
323  _node_info_retriever.invalidateAll();
324  }
325 
326  const int spin_res = _subnode.spin(uavcan::MonotonicDuration::fromMSec(10));
327 
328  if (spin_res < 0) {
329  warnx("node spin error %i", spin_res);
330  }
331 
332  // Check for parameter requests (get/set/list)
333  bool param_request_ready;
334  orb_check(param_request_sub, &param_request_ready);
335 
336  if (param_request_ready && !_param_list_in_progress && !_param_in_progress && !_count_in_progress) {
337  struct uavcan_parameter_request_s request;
338  orb_copy(ORB_ID(uavcan_parameter_request), param_request_sub, &request);
339 
340  if (_param_counts[request.node_id]) {
341  /*
342  * We know how many parameters are exposed by this node, so
343  * process the request.
344  */
345  if (request.message_type == MAVLINK_MSG_ID_PARAM_REQUEST_READ) {
346  uavcan::protocol::param::GetSet::Request req;
347 
348  if (request.param_index >= 0) {
349  req.index = request.param_index;
350 
351  } else {
352  req.name = (char *)request.param_id;
353  }
354 
355  int call_res = _param_getset_client.call(request.node_id, req);
356 
357  if (call_res < 0) {
358  warnx("UAVCAN command bridge: couldn't send GetSet: %d", call_res);
359 
360  } else {
361  _param_in_progress = true;
362  _param_index = request.param_index;
363  }
364 
365  } else if (request.message_type == MAVLINK_MSG_ID_PARAM_SET) {
366  uavcan::protocol::param::GetSet::Request req;
367 
368  if (request.param_index >= 0) {
369  req.index = request.param_index;
370 
371  } else {
372  req.name = (char *)request.param_id;
373  }
374 
375  if (request.param_type == MAV_PARAM_TYPE_REAL32) {
376  req.value.to<uavcan::protocol::param::Value::Tag::real_value>() = request.real_value;
377 
378  } else if (request.param_type == MAV_PARAM_TYPE_UINT8) {
379  req.value.to<uavcan::protocol::param::Value::Tag::boolean_value>() = request.int_value;
380 
381  } else {
382  req.value.to<uavcan::protocol::param::Value::Tag::integer_value>() = request.int_value;
383  }
384 
385  // Set the dirty bit for this node
386  set_node_params_dirty(request.node_id);
387 
388  int call_res = _param_getset_client.call(request.node_id, req);
389 
390  if (call_res < 0) {
391  warnx("UAVCAN command bridge: couldn't send GetSet: %d", call_res);
392 
393  } else {
394  _param_in_progress = true;
395  _param_index = request.param_index;
396  }
397 
398  } else if (request.message_type == MAVLINK_MSG_ID_PARAM_REQUEST_LIST) {
399  // This triggers the _param_list_in_progress case below.
400  _param_index = 0;
401  _param_list_in_progress = true;
402  _param_list_node_id = request.node_id;
403  _param_list_all_nodes = false;
404 
405  warnx("UAVCAN command bridge: starting component-specific param list");
406  }
407 
408  } else if (request.node_id == MAV_COMP_ID_ALL) {
409  if (request.message_type == MAVLINK_MSG_ID_PARAM_REQUEST_LIST) {
410  /*
411  * This triggers the _param_list_in_progress case below,
412  * but additionally iterates over all active nodes.
413  */
414  _param_index = 0;
415  _param_list_in_progress = true;
416  _param_list_node_id = get_next_active_node_id(0);
417  _param_list_all_nodes = true;
418 
419  warnx("UAVCAN command bridge: starting global param list with node %hhu", _param_list_node_id);
420 
421  if (_param_counts[_param_list_node_id] == 0) {
422  param_count(_param_list_node_id);
423  }
424  }
425 
426  } else {
427  /*
428  * Need to know how many parameters this node has before we can
429  * continue; count them now and then process the request.
430  */
431  param_count(request.node_id);
432  }
433  }
434 
435  // Handle parameter listing index/node ID advancement
436  if (_param_list_in_progress && !_param_in_progress && !_count_in_progress) {
437  if (_param_index >= _param_counts[_param_list_node_id]) {
438  warnx("UAVCAN command bridge: completed param list for node %hhu", _param_list_node_id);
439  // Reached the end of the current node's parameter set.
440  _param_list_in_progress = false;
441 
442  if (_param_list_all_nodes) {
443  // We're listing all parameters for all nodes -- get the next node ID
444  uint8_t next_id = get_next_active_node_id(_param_list_node_id);
445 
446  if (next_id < 128) {
447  _param_list_node_id = next_id;
448 
449  /*
450  * If there is a next node ID, check if that node's parameters
451  * have been counted before. If not, do it now.
452  */
453  if (_param_counts[_param_list_node_id] == 0) {
454  param_count(_param_list_node_id);
455  }
456 
457  // Keep on listing.
458  _param_index = 0;
459  _param_list_in_progress = true;
460  warnx("UAVCAN command bridge: started param list for node %hhu", _param_list_node_id);
461  }
462  }
463  }
464  }
465 
466  // Check if we're still listing, and need to get the next parameter
467  if (_param_list_in_progress && !_param_in_progress && !_count_in_progress) {
468  // Ready to request the next value -- _param_index is incremented
469  // after each successful fetch by cb_getset
470  uavcan::protocol::param::GetSet::Request req;
471  req.index = _param_index;
472 
473  int call_res = _param_getset_client.call(_param_list_node_id, req);
474 
475  if (call_res < 0) {
476  _param_list_in_progress = false;
477  warnx("UAVCAN command bridge: couldn't send param list GetSet: %d", call_res);
478 
479  } else {
480  _param_in_progress = true;
481  }
482  }
483 
484  // Check for ESC enumeration commands
485  bool cmd_ready;
486  orb_check(cmd_sub, &cmd_ready);
487 
488  if (cmd_ready && !_cmd_in_progress) {
489  struct vehicle_command_s cmd;
490  orb_copy(ORB_ID(vehicle_command), cmd_sub, &cmd);
491 
492  uint8_t cmd_ack_result = vehicle_command_ack_s::VEHICLE_RESULT_ACCEPTED;
493 
494  if (cmd.command == vehicle_command_s::VEHICLE_CMD_PREFLIGHT_UAVCAN) {
495  int command_id = static_cast<int>(cmd.param1 + 0.5f);
496  int node_id = static_cast<int>(cmd.param2 + 0.5f);
497  int call_res;
498 
499  warnx("UAVCAN command bridge: received UAVCAN command ID %d, node ID %d", command_id, node_id);
500 
501  switch (command_id) {
502  case 0:
503  case 1: {
504  _esc_enumeration_active = command_id;
505  _esc_enumeration_index = 0;
506  _esc_count = 0;
507  uavcan::protocol::enumeration::Begin::Request req;
508  // TODO: Incorrect implementation; the parameter name field should be left empty.
509  // Leaving it as-is to avoid breaking compatibility with non-compliant nodes.
510  req.parameter_name = "esc_index";
511  req.timeout_sec = _esc_enumeration_active ? 65535 : 0;
512  call_res = _enumeration_client.call(get_next_active_node_id(0), req);
513 
514  if (call_res < 0) {
515  warnx("UAVCAN ESC enumeration: couldn't send initial Begin request: %d", call_res);
516  beep(BeepFrequencyError);
517  cmd_ack_result = vehicle_command_ack_s::VEHICLE_RESULT_FAILED;
518 
519  } else {
520  beep(BeepFrequencyGenericIndication);
521  }
522 
523  break;
524  }
525 
526  default: {
527  warnx("UAVCAN command bridge: unknown command ID %d", command_id);
528  cmd_ack_result = vehicle_command_ack_s::VEHICLE_RESULT_UNSUPPORTED;
529  break;
530  }
531  }
532 
533  } else if (cmd.command == vehicle_command_s::VEHICLE_CMD_PREFLIGHT_STORAGE) {
534  int command_id = static_cast<int>(cmd.param1 + 0.5f);
535 
536  warnx("UAVCAN command bridge: received storage command ID %d", command_id);
537 
538  switch (command_id) {
539  case 1: {
540  // Param save request
541  int node_id;
542  node_id = get_next_dirty_node_id(1);
543 
544  if (node_id < 128) {
545  _param_save_opcode = uavcan::protocol::param::ExecuteOpcode::Request::OPCODE_SAVE;
546  param_opcode(node_id);
547  }
548 
549  break;
550  }
551 
552  case 2: {
553  // Command is a param erase request -- apply it to all active nodes by setting the dirty bit
554  _param_save_opcode = uavcan::protocol::param::ExecuteOpcode::Request::OPCODE_ERASE;
555 
556  for (int i = 1; i < 128; i = get_next_active_node_id(i)) {
557  set_node_params_dirty(i);
558  }
559 
560  param_opcode(get_next_dirty_node_id(1));
561  break;
562  }
563  }
564  }
565 
566  // Acknowledge the received command
567  vehicle_command_ack_s ack = {};
568  ack.timestamp = hrt_absolute_time();
569  ack.command = cmd.command;
570  ack.result = cmd_ack_result;
571  ack.target_system = cmd.source_system;
572  ack.target_component = cmd.source_component;
573 
574  if (_command_ack_pub == nullptr) {
575  _command_ack_pub = orb_advertise_queue(ORB_ID(vehicle_command_ack), &ack, vehicle_command_ack_s::ORB_QUEUE_LENGTH);
576 
577  } else {
578  orb_publish(ORB_ID(vehicle_command_ack), _command_ack_pub, &ack);
579  }
580 
581  }
582 
583  // Shut down once armed
584  // TODO (elsewhere): start up again once disarmed?
585  bool updated;
586  orb_check(armed_sub, &updated);
587 
588  if (updated) {
589  struct actuator_armed_s armed;
590  orb_copy(ORB_ID(actuator_armed), armed_sub, &armed);
591 
592  if (armed.armed && !(armed.lockdown || armed.manual_lockdown)) {
593  warnx("UAVCAN command bridge: system armed, exiting now.");
594  break;
595  }
596  }
597  }
598 
599  _subnode_thread_should_exit = false;
600 
601  warnx("exiting");
602  return (pthread_addr_t) 0;

605 void UavcanServers::cb_getset(const uavcan::ServiceCallResult<uavcan::protocol::param::GetSet> &result)
606 {
607  if (_count_in_progress) {
608  /*
609  * Currently in parameter count mode:
610  * Iterate over all parameters for the node to which the request was
611  * originally sent, in order to find the maximum parameter ID. If a
612  * request fails, set the node's parameter count to zero.
613  */
614  uint8_t node_id = result.getCallID().server_node_id.get();
615 
616  if (result.isSuccessful()) {
617  uavcan::protocol::param::GetSet::Response resp = result.getResponse();
618 
619  if (resp.name.size()) {
620  _count_index++;
621  _param_counts[node_id] = _count_index;
622 
623  uavcan::protocol::param::GetSet::Request req;
624  req.index = _count_index;
625 
626  int call_res = _param_getset_client.call(result.getCallID().server_node_id, req);
627 
628  if (call_res < 0) {
629  _count_in_progress = false;
630  _count_index = 0;
631  warnx("UAVCAN command bridge: couldn't send GetSet during param count: %d", call_res);
632  beep(BeepFrequencyError);
633  }
634 
635  } else {
636  _count_in_progress = false;
637  _count_index = 0;
638  warnx("UAVCAN command bridge: completed param count for node %hhu: %hhu", node_id, _param_counts[node_id]);
639  beep(BeepFrequencyGenericIndication);
640  }
641 
642  } else {
643  _param_counts[node_id] = 0;
644  _count_in_progress = false;
645  _count_index = 0;
646  warnx("UAVCAN command bridge: GetSet error during param count");
647  }
648 
649  } else {
650  /*
651  * Currently in parameter get/set mode:
652  * Publish a uORB uavcan_parameter_value message containing the current value
653  * of the parameter.

966 void UavcanServers::unpackFwFromROMFS(const char *sd_path, const char *romfs_path)
967 {
968  /*
969  Copy the ROMFS firmware directory to the appropriate location on SD, without
970  overriding any firmware the user has already loaded there.
971 
972  The SD firmware directory structure is along the lines of:
973 
974  /fs/microsd/fw
975  - /c
976  - /nodename-v1-1.0.25d0137d.bin cache file (copy of /fs/microsd/fw/org.pixhawk.nodename-v1/1.1/nodename-v1-1.0.25d0137d.bin)
977  - /othernode-v3-1.6.25d0137d.bin cache file (copy of /fs/microsd/fw/com.example.othernode-v3/1.6/othernode-v3-1.6.25d0137d.bin)
978  - /org.pixhawk.nodename-v1 device directory for org.pixhawk.nodename-v1
979  - - /1.0 version directory for hardware 1.0
980  - - /1.1 version directory for hardware 1.1
981  - - - /nodename-v1-1.0.25d0137d.bin firmware file for org.pixhawk.nodename-v1 nodes, hardware version 1.1
982  - /com.example.othernode-v3 device directory for com.example.othernode-v3
983  - - /1.0 version directory for hardawre 1.0
984  - - - /othernode-v3-1.6.25d0137d.bin firmware file for com.example.othernode-v3, hardware version 1.6
985 
986  The ROMFS directory structure is the same, but located at /etc/uavcan/fw
987  Files located there are prefixed with _ to identify them a comming from the rom
988  file system.
989 
990  We iterate over all device directories in the ROMFS base directory, and create
991  corresponding device directories on the SD card if they don't already exist.
992 
993  In each device directory, we iterate over each version directory and create a
994  corresponding version directory on the SD card if it doesn't already exist.
995 
996  In each version directory, we remove any files with a name starting with "_"
997  in the corresponding directory on the SD card that don't match the bundled firmware
998  filename; if the directory is empty after that process, we copy the bundled firmware.
999 
1000  todo:This code would benefit from the use of strcat.
1001 
1002  */
1003  const size_t maxlen = UAVCAN_MAX_PATH_LENGTH;
1004  const size_t sd_path_len = strlen(sd_path);
1005  const size_t romfs_path_len = strlen(romfs_path);
1006  struct stat sb;
1007  int rv;
1008  char dstpath[maxlen + 1];
1009  char srcpath[maxlen + 1];
1010 
1011  DIR *const romfs_dir = opendir(romfs_path);
1012 
1013  if (!romfs_dir) {
1014  return;
1015  }
1016 
1017  memcpy(dstpath, sd_path, sd_path_len + 1);
1018  memcpy(srcpath, romfs_path, romfs_path_len + 1);
1019 
1020  // Iterate over all device directories in ROMFS
1021  struct dirent *dev_dirent = NULL;
1022 
1023  while ((dev_dirent = readdir(romfs_dir)) != NULL) {
1024  // Skip if not a directory
1025  if (!DIRENT_ISDIRECTORY(dev_dirent->d_type)) {
1026  continue;
1027  }
1028 
1029  // Make sure the path fits
1030  size_t dev_dirname_len = strlen(dev_dirent->d_name);
1031  size_t srcpath_dev_len = romfs_path_len + 1 + dev_dirname_len;
1032 
1033  if (srcpath_dev_len > maxlen) {
1034  warnx("dev: srcpath '%s/%s' too long", romfs_path, dev_dirent->d_name);
1035  continue;
1036  }
1037 
1038  size_t dstpath_dev_len = sd_path_len + 1 + dev_dirname_len;
1039 
1040  if (dstpath_dev_len > maxlen) {
1041  warnx("dev: dstpath '%s/%s' too long", sd_path, dev_dirent->d_name);
1042  continue;
1043  }
1044 
1045  // Create the device name directory on the SD card if it doesn't already exist
1046  dstpath[sd_path_len] = '/';
1047  memcpy(&dstpath[sd_path_len + 1], dev_dirent->d_name, dev_dirname_len + 1);
1048 
1049  if (stat(dstpath, &sb) != 0 || !S_ISDIR(sb.st_mode)) {
1050  rv = mkdir(dstpath, S_IRWXU | S_IRWXG | S_IRWXO);
1051 
1052  if (rv != 0) {
1053  warnx("dev: couldn't create '%s'", dstpath);
1054  continue;
1055  }
1056  }
1057 
1058  // Set up the source path
1059  srcpath[romfs_path_len] = '/';
1060  memcpy(&srcpath[romfs_path_len + 1], dev_dirent->d_name, dev_dirname_len + 1);
1061 
1062  DIR *const dev_dir = opendir(srcpath);
1063 
1064  if (!dev_dir) {
1065  warnx("dev: couldn't open '%s'", srcpath);
1066  continue;
1067  }
1068 
1069  // Iterate over all version directories in the current ROMFS device directory
1070  struct dirent *ver_dirent = NULL;
1071 
1072  while ((ver_dirent = readdir(dev_dir)) != NULL) {
1073  // Skip if not a directory
1074  if (!DIRENT_ISDIRECTORY(ver_dirent->d_type)) {
1075  continue;
1076  }
1077 
1078  // Make sure the path fits
1079  size_t ver_dirname_len = strlen(ver_dirent->d_name);
1080  size_t srcpath_ver_len = srcpath_dev_len + 1 + ver_dirname_len;
1081 
1082  if (srcpath_ver_len > maxlen) {
1083  warnx("ver: srcpath '%s/%s' too long", srcpath, ver_dirent->d_name);
1084  continue;
1085  }
1086 
1087  size_t dstpath_ver_len = dstpath_dev_len + 1 + ver_dirname_len;
1088 
1089  if (dstpath_ver_len > maxlen) {
1090  warnx("ver: dstpath '%s/%s' too long", dstpath, ver_dirent->d_name);
1091  continue;
1092  }
1093

1095  dstpath[dstpath_dev_len] = '/';
1096  memcpy(&dstpath[dstpath_dev_len + 1], ver_dirent->d_name, ver_dirname_len + 1);
1097

1103  continue;
1104  }
1105  }
1106

1109  memcpy(&srcpath[srcpath_dev_len + 1], ver_dirent->d_name, ver_dirname_len + 1);

1114

1116  warnx("ver: couldn't open '%s'", srcpath);
1117  continue;
1118  }

1121 
1122  while ((src_fw_dirent = readdir(src_ver_dir)) != NULL &&
1123  !DIRENT_ISFILE(src_fw_dirent->d_type));
1124

1126  (void)closedir(src_ver_dir);
1127  continue;

1131

1133

1135  DIR *const dst_ver_dir = opendir(dstpath);
1136

1138  warnx("unlink: couldn't open '%s'", dstpath);
1139

1155 
1156  } else if (!memcmp(fw_dirent->d_name, UAVCAN_ROMFS_FW_PREFIX, sizeof(UAVCAN_ROMFS_FW_PREFIX) - 1)) {

1166  dstpath[dstpath_ver_len] = '/';
1167  memcpy(&dstpath[dstpath_ver_len + 1], fw_dirent->d_name, dst_fw_len + 1);
1168  unlink(dstpath);
1169  warnx("unlink: removed '%s'", dstpath);
1170  }
1171 
1172  } else {
1173  // User file, don't copy firmware
1174  copy_fw = false;
1175  }
1176  }
1177 
1178  (void)closedir(dst_ver_dir);
1179  }
1180 
1181  // If we need to, copy the file from ROMFS to the SD card
1182  if (copy_fw) {
1183  size_t srcpath_fw_len = srcpath_ver_len + 1 + fw_len;

1185 
1186  if (srcpath_fw_len > maxlen) {
1187  warnx("copy: srcpath '%s/%s' too long", srcpath, src_fw_dirent->d_name);
1188 
1189  } else if (dstpath_fw_len > maxlen) {
1190  warnx("copy: dstpath '%s/%s' too long", dstpath, src_fw_dirent->d_name);
1191 
1192  } else {
1193  // All OK, make the paths and copy the file
1194  srcpath[srcpath_ver_len] = '/';
1195  memcpy(&srcpath[srcpath_ver_len + 1], src_fw_dirent->d_name, fw_len + 1);
1196 
1197  dstpath[dstpath_ver_len] = '/';
1198  memcpy(&dstpath[dstpath_ver_len + 1], src_fw_dirent->d_name, fw_len + 1);
1199 
1200  rv = copyFw(dstpath, srcpath);
1201

1204

1206  warnx("copy: '%s' -> '%s' succeeded", srcpath, dstpath);
