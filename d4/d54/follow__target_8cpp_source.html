99 void FollowTarget::on_active()
100 {
101  struct map_projection_reference_s target_ref;
102  follow_target_s target_motion_with_offset = {};
103  uint64_t current_time = hrt_absolute_time();
104  bool _radius_entered = false;
105  bool _radius_exited = false;
106  bool updated = false;
107  float dt_ms = 0;
108 
109  orb_check(_follow_target_sub, &updated);
110 
111  if (updated) {
112  follow_target_s target_motion;
113 
114  _target_updates++;
115 
116  // save last known motion topic
117 
118  _previous_target_motion = _current_target_motion;
119 
120  orb_copy(ORB_ID(follow_target), _follow_target_sub, &target_motion);
121 
122  if (_current_target_motion.timestamp == 0) {
123  _current_target_motion = target_motion;
124  }
125 
126  _current_target_motion.timestamp = target_motion.timestamp;
127  _current_target_motion.lat = (_current_target_motion.lat * (double)_responsiveness) + target_motion.lat * (double)(
128  1 - _responsiveness);
129  _current_target_motion.lon = (_current_target_motion.lon * (double)_responsiveness) + target_motion.lon * (double)(
130  1 - _responsiveness);
131 
132  } else if (((current_time - _current_target_motion.timestamp) / 1000) > TARGET_TIMEOUT_MS && target_velocity_valid()) {
133  reset_target_validity();
134  }
135 
136  // update distance to target
137 
138  if (target_position_valid()) {
139 
140  // get distance to target
141 
142  map_projection_init(&target_ref, _navigator->get_global_position()->lat, _navigator->get_global_position()->lon);
143  map_projection_project(&target_ref, _current_target_motion.lat, _current_target_motion.lon, &_target_distance(0),
144  &_target_distance(1));
145 
146  }
147 
148  // update target velocity
149 
150  if (target_velocity_valid() && updated) {
151 
152  dt_ms = ((_current_target_motion.timestamp - _previous_target_motion.timestamp) / 1000);
153 
154  // ignore a small dt
155  if (dt_ms > 10.0F) {
156  // get last gps known reference for target
157  map_projection_init(&target_ref, _previous_target_motion.lat, _previous_target_motion.lon);
158 
159  // calculate distance the target has moved
160  map_projection_project(&target_ref, _current_target_motion.lat, _current_target_motion.lon,
161  &(_target_position_delta(0)), &(_target_position_delta(1)));
162 
163  // update the average velocity of the target based on the position
164  _est_target_vel = _target_position_delta / (dt_ms / 1000.0f);
165 
166  // if the target is moving add an offset and rotation
167  if (_est_target_vel.length() > .5F) {
168  _target_position_offset = _rot_matrix * _est_target_vel.normalized() * _follow_offset;
169  }
170 
171  // are we within the target acceptance radius?
172  // give a buffer to exit/enter the radius to give the velocity controller
173  // a chance to catch up
174 
175  _radius_exited = ((_target_position_offset + _target_distance).length() > (float) TARGET_ACCEPTANCE_RADIUS_M * 1.5f);
176  _radius_entered = ((_target_position_offset + _target_distance).length() < (float) TARGET_ACCEPTANCE_RADIUS_M);
177 
178  // to keep the velocity increase/decrease smooth
179  // calculate how many velocity increments/decrements
180  // it will take to reach the targets velocity
181  // with the given amount of steps also add a feed forward input that adjusts the
182  // velocity as the position gap increases since
183  // just traveling at the exact velocity of the target will not
184  // get any closer or farther from the target
185 
186  _step_vel = (_est_target_vel - _current_vel) + (_target_position_offset + _target_distance) * FF_K;
187  _step_vel /= (dt_ms / 1000.0F * (float) INTERPOLATION_PNTS);
188  _step_time_in_ms = (dt_ms / (float) INTERPOLATION_PNTS);
189 
190  // if we are less than 1 meter from the target don't worry about trying to yaw
191  // lock the yaw until we are at a distance that makes sense
192 
193  if ((_target_distance).length() > 1.0F) {
194 
195  // yaw rate smoothing
196 
197  // this really needs to control the yaw rate directly in the attitude pid controller
198  // but seems to work ok for now since the yaw rate cannot be controlled directly in auto mode
199 
200  _yaw_angle = get_bearing_to_next_waypoint(_navigator->get_global_position()->lat,
201  _navigator->get_global_position()->lon,
202  _current_target_motion.lat,
203  _current_target_motion.lon);
204 
205  _yaw_rate = wrap_pi((_yaw_angle - _navigator->get_global_position()->yaw) / (dt_ms / 1000.0f));
206 
207  } else {
208  _yaw_angle = _yaw_rate = NAN;
209  }
210  }
211 
212 // warnx(" _step_vel x %3.6f y %3.6f cur vel %3.6f %3.6f tar vel %3.6f %3.6f dist = %3.6f (%3.6f) mode = %d yaw rate = %3.6f",
213 // (double) _step_vel(0),
214 // (double) _step_vel(1),
215 // (double) _current_vel(0),
216 // (double) _current_vel(1),
217 // (double) _est_target_vel(0),
218 // (double) _est_target_vel(1),
219 // (double) (_target_distance).length(),
220 // (double) (_target_position_offset + _target_distance).length(),
221 // _follow_target_state,
222 // (double) _yaw_rate);
223  }
224 
225  if (target_position_valid()) {
226 
227  // get the target position using the calculated offset
228 
229  map_projection_init(&target_ref, _current_target_motion.lat, _current_target_motion.lon);
230  map_projection_reproject(&target_ref, _target_position_offset(0), _target_position_offset(1),
231  &target_motion_with_offset.lat, &target_motion_with_offset.lon);
232  }
233 
234  // clamp yaw rate smoothing if we are with in
235  // 3 degrees of facing target
236 
237  if (PX4_ISFINITE(_yaw_rate)) {
238  if (fabsf(fabsf(_yaw_angle) - fabsf(_navigator->get_global_position()->yaw)) < math::radians(3.0F)) {
239  _yaw_rate = NAN;
240  }
241  }
242 
243  // update state machine
244 
245  switch (_follow_target_state) {
246 
247  case TRACK_POSITION: {
248 
249  if (_radius_entered == true) {
250  _follow_target_state = TRACK_VELOCITY;
251 
252  } else if (target_velocity_valid()) {
253  set_follow_target_item(&_mission_item, _param_min_alt.get(), target_motion_with_offset, _yaw_angle);
254  // keep the current velocity updated with the target velocity for when it's needed
255  _current_vel = _est_target_vel;
256 
257  update_position_sp(true, true, _yaw_rate);
258 
259  } else {
260  _follow_target_state = SET_WAIT_FOR_TARGET_POSITION;
261  }
262 
263  break;
264  }
265 
266  case TRACK_VELOCITY: {
267 
268  if (_radius_exited == true) {
269  _follow_target_state = TRACK_POSITION;
270 
271  } else if (target_velocity_valid()) {
272 
273  if ((float)(current_time - _last_update_time) / 1000.0f >= _step_time_in_ms) {
274  _current_vel += _step_vel;
275  _last_update_time = current_time;
276  }
277 
278  set_follow_target_item(&_mission_item, _param_min_alt.get(), target_motion_with_offset, _yaw_angle);
279 
280  update_position_sp(true, false, _yaw_rate);
281 
282  } else {
283  _follow_target_state = SET_WAIT_FOR_TARGET_POSITION;
284  }
285 
286  break;
287  }
288 
289  case SET_WAIT_FOR_TARGET_POSITION: {
290 
291  // Climb to the minimum altitude
292  // and wait until a position is received
293 
294  follow_target_s target = {};
295 
296  // for now set the target at the minimum height above the uav
297 
298  target.lat = _navigator->get_global_position()->lat;
299  target.lon = _navigator->get_global_position()->lon;
300  target.alt = 0.0F;
301 
302  set_follow_target_item(&_mission_item, _param_min_alt.get(), target, _yaw_angle);
303 
304  update_position_sp(false, false, _yaw_rate);
305 
306  _follow_target_state = WAIT_FOR_TARGET_POSITION;
307  }
308 
309  /* FALLTHROUGH */
310 
311  case WAIT_FOR_TARGET_POSITION: {
312 
313  if (is_mission_item_reached() && target_velocity_valid()) {
314  _target_position_offset(0) = _follow_offset;
315  _follow_target_state = TRACK_POSITION;
316  }
317 
318  break;
319  }
320  }
321 }

323 void FollowTarget::update_position_sp(bool use_velocity, bool use_position, float yaw_rate)
324 {
325  // convert mission item to current setpoint
326 
327  struct position_setpoint_triplet_s *pos_sp_triplet = _navigator->get_position_setpoint_triplet();
328 
329  // activate line following in pos control if position is valid
330 
331  pos_sp_triplet->previous.valid = use_position;
332  pos_sp_triplet->previous = pos_sp_triplet->current;
333  mission_apply_limitation(_mission_item);
334  mission_item_to_position_setpoint(_mission_item, &pos_sp_triplet->current);
335  pos_sp_triplet->current.type = position_setpoint_s::SETPOINT_TYPE_FOLLOW_TARGET;
336  pos_sp_triplet->current.position_valid = use_position;
337  pos_sp_triplet->current.velocity_valid = use_velocity;
338  pos_sp_triplet->current.vx = _current_vel(0);
339  pos_sp_triplet->current.vy = _current_vel(1);
340  pos_sp_triplet->next.valid = false;
341  pos_sp_triplet->current.yawspeed_valid = PX4_ISFINITE(yaw_rate);
342  pos_sp_triplet->current.yawspeed = yaw_rate;
343  _navigator->set_position_setpoint_triplet_updated();
