121 int
122 I2C::transfer(const uint8_t *send, unsigned send_len, uint8_t *recv, unsigned recv_len)
123 {
124 #ifndef __PX4_LINUX
125  return PX4_ERROR;
126 #else
127  struct i2c_msg msgv[2];
128  unsigned msgs;
129  int ret = PX4_ERROR;
130  unsigned retry_count = 0;
131 
132  if (_fd < 0) {
133  PX4_ERR("I2C device not opened");
134  return 1;
135  }
136 
137  do {
138  DEVICE_DEBUG("transfer out %p/%u in %p/%u", send, send_len, recv, recv_len);
139  msgs = 0;
140 
141  if (send_len > 0) {
142  msgv[msgs].addr = get_device_address();
143  msgv[msgs].flags = 0;
144  msgv[msgs].buf = const_cast<uint8_t *>(send);
145  msgv[msgs].len = send_len;
146  msgs++;
147  }
148 
149  if (recv_len > 0) {
150  msgv[msgs].addr = get_device_address();
151  msgv[msgs].flags = I2C_M_READ;
152  msgv[msgs].buf = recv;
153  msgv[msgs].len = recv_len;
154  msgs++;
155  }
156 
157  if (msgs == 0) {
158  return -EINVAL;
159  }
160 
161  struct i2c_rdwr_ioctl_data packets;
162 
163  packets.msgs = msgv;
164 
165  packets.nmsgs = msgs;
166 
167  if (simulate) {
168  DEVICE_DEBUG("I2C SIM: transfer_4 on %s", get_devname());
169  ret = PX4_OK;
170 
171  } else {
172  ret = ::ioctl(_fd, I2C_RDWR, (unsigned long)&packets);
173 
174  if (ret == -1) {
175  DEVICE_DEBUG("I2C transfer failed");
176  ret = PX4_ERROR;
177 
178  } else {
179  ret = PX4_OK;
180  }
181  }
182 
183  /* success */
184  if (ret == PX4_OK) {
185  break;
186  }
187 
188  } while (retry_count++ < _retries);
189 
190  return ret;
191 #endif
192 }
