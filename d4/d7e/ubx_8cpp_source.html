113

116  uint16_t out_proto_mask = output_mode == OutputMode::GPS ?
117  UBX_TX_CFG_PRT_OUTPROTOMASK_GPS :
118  UBX_TX_CFG_PRT_OUTPROTOMASK_RTCM;
119  uint16_t in_proto_mask = output_mode == OutputMode::GPS ?
120  UBX_TX_CFG_PRT_INPROTOMASK_GPS :
121  UBX_TX_CFG_PRT_INPROTOMASK_RTCM;
122  //FIXME: RTCM3 output needs at least protocol version 20. The protocol version can be checked via the version
123  //output:
124  //WARN VER ext " PROTVER=20.00"
125  //However this is a string and it is not well documented, that PROTVER is always contained. Maybe there is a
126  //better way to check the protocol version?
127 
128 
129  if (_interface == Interface::UART) {
130  for (baud_i = 0; baud_i < sizeof(baudrates) / sizeof(baudrates[0]); baud_i++) {
131  unsigned test_baudrate = baudrates[baud_i];
132 
133  UBX_DEBUG("baudrate set to %i", test_baudrate);
134 
135  if (baudrate > 0 && baudrate != test_baudrate) {
136  continue; // skip to next baudrate
137  }
138 
139  setBaudrate(test_baudrate);
140 
141  /* reset all configuration on the module - this is necessary as some vendors lock
142  * lock bad configurations
143  */
144  ubx_payload_tx_cfg_cfg_t cfg_cfg = {};
145  cfg_cfg.clearMask = ((1 << 12) | (1 << 11) | (1 << 10) | (1 << 9) |
146  (1 << 8) | (1 << 4) | (1 << 3) | (1 << 2) | (1 << 1) | (1 << 0));
147  cfg_cfg.deviceMask = (1 << 2) | (1 << 1) | (1 << 0);
148 
149  if (!sendMessage(UBX_MSG_CFG_CFG, (uint8_t *)&cfg_cfg, sizeof(ubx_payload_tx_cfg_cfg_t))) {
150  UBX_DEBUG("cfg reset: UART TX failed");
151  }
152 
153  if (waitForAck(UBX_MSG_CFG_CFG, UBX_CONFIG_TIMEOUT, true) < 0) {
154  UBX_DEBUG("cfg reset failed");
155 
156  } else {
157  UBX_DEBUG("cfg reset ACK");
158  }
159 
160  /* allow the module to re-initialize */
161  usleep(100000);
162 
163  /* flush input and wait for at least 20 ms silence */
164  decodeInit();
165  receive(20);
166  decodeInit();
167 
168  /* Send a CFG-PRT message to set the UBX protocol for in and out
169  * and leave the baudrate as it is, we just want an ACK-ACK for this */
170  memset(cfg_prt, 0, 2 * sizeof(ubx_payload_tx_cfg_prt_t));
171  cfg_prt[0].portID = UBX_TX_CFG_PRT_PORTID;
172  cfg_prt[0].mode = UBX_TX_CFG_PRT_MODE;
173  cfg_prt[0].baudRate = test_baudrate;
174  cfg_prt[0].inProtoMask = in_proto_mask;
175  cfg_prt[0].outProtoMask = out_proto_mask;
176  cfg_prt[1].portID = UBX_TX_CFG_PRT_PORTID_USB;
177  cfg_prt[1].mode = UBX_TX_CFG_PRT_MODE;
178  cfg_prt[1].baudRate = test_baudrate;
179  cfg_prt[1].inProtoMask = in_proto_mask;
180  cfg_prt[1].outProtoMask = out_proto_mask;
181 
182  if (!sendMessage(UBX_MSG_CFG_PRT, (uint8_t *)cfg_prt, 2 * sizeof(ubx_payload_tx_cfg_prt_t))) {
183  continue;
184  }
185 
186  if (waitForAck(UBX_MSG_CFG_PRT, UBX_CONFIG_TIMEOUT, false) < 0) {
187  /* try next baudrate */
188  continue;
189  }
190 
191  /* Send a CFG-PRT message again, this time change the baudrate */
192  memset(cfg_prt, 0, 2 * sizeof(ubx_payload_tx_cfg_prt_t));
193  cfg_prt[0].portID = UBX_TX_CFG_PRT_PORTID;
194  cfg_prt[0].mode = UBX_TX_CFG_PRT_MODE;
195  cfg_prt[0].baudRate = UBX_TX_CFG_PRT_BAUDRATE;
196  cfg_prt[0].inProtoMask = in_proto_mask;
197  cfg_prt[0].outProtoMask = out_proto_mask;
198  cfg_prt[1].portID = UBX_TX_CFG_PRT_PORTID_USB;
199  cfg_prt[1].mode = UBX_TX_CFG_PRT_MODE;
200  cfg_prt[1].baudRate = UBX_TX_CFG_PRT_BAUDRATE;
201  cfg_prt[1].inProtoMask = in_proto_mask;
202  cfg_prt[1].outProtoMask = out_proto_mask;
203 
204  if (!sendMessage(UBX_MSG_CFG_PRT, (uint8_t *)cfg_prt, 2 * sizeof(ubx_payload_tx_cfg_prt_t))) {
205  continue;
206  }
207 
208  /* no ACK is expected here, but read the buffer anyway in case we actually get an ACK */
209  waitForAck(UBX_MSG_CFG_PRT, UBX_CONFIG_TIMEOUT, false);
210 
211  if (UBX_TX_CFG_PRT_BAUDRATE != test_baudrate) {
212  setBaudrate(UBX_TX_CFG_PRT_BAUDRATE);
213  }
214 
215  /* at this point we have correct baudrate on both ends */
216  baudrate = UBX_TX_CFG_PRT_BAUDRATE;
217  break;
218  }
219 
220  if (baud_i >= sizeof(baudrates) / sizeof(baudrates[0])) {
221  return -1; // connection and/or baudrate detection failed
222  }
223 
224  } else if (_interface == Interface::SPI) {
225  memset(cfg_prt, 0, 2 * sizeof(ubx_payload_tx_cfg_prt_t));
226  cfg_prt[0].portID = UBX_TX_CFG_PRT_PORTID_SPI;
227  cfg_prt[0].mode = UBX_TX_CFG_PRT_MODE_SPI;
228  cfg_prt[0].inProtoMask = in_proto_mask;
229  cfg_prt[0].outProtoMask = out_proto_mask;
230 
231  if (!sendMessage(UBX_MSG_CFG_PRT, (uint8_t *)cfg_prt, sizeof(ubx_payload_tx_cfg_prt_t))) {
232  return -1;
233  }
234 
235  /* no ACK is expected here, but read the buffer anyway in case we actually get an ACK */
236  waitForAck(UBX_MSG_CFG_PRT, UBX_CONFIG_TIMEOUT, false);
237 
238  } else {
239  return -1;
240  }
241 
242  /* Send a CFG-RATE message to define update rate */
243  memset(&_buf.payload_tx_cfg_rate, 0, sizeof(_buf.payload_tx_cfg_rate));
244  _buf.payload_tx_cfg_rate.measRate = UBX_TX_CFG_RATE_MEASINTERVAL;
245  _buf.payload_tx_cfg_rate.navRate = UBX_TX_CFG_RATE_NAVRATE;
246  _buf.payload_tx_cfg_rate.timeRef = UBX_TX_CFG_RATE_TIMEREF;
247 
248  if (!sendMessage(UBX_MSG_CFG_RATE, (uint8_t *)&_buf, sizeof(_buf.payload_tx_cfg_rate))) {
249  return -1;
250  }
251 
252  if (waitForAck(UBX_MSG_CFG_RATE, UBX_CONFIG_TIMEOUT, true) < 0) {
253  return -1;
254  }
255 
256  if (output_mode != OutputMode::GPS) {
257  // RTCM mode force stationary dynamic model
258  _dyn_model = 2;
259  }
260

269  }

273  }

276  /* send a SBAS message to set the SBAS options */
277  memset(&_buf.payload_tx_cfg_sbas, 0, sizeof(_buf.payload_tx_cfg_sbas));

400 int // -1 = NAK, error or timeout, 0 = ACK
401 GPSDriverUBX::waitForAck(const uint16_t msg, const unsigned timeout, const bool report)
402 {
403  int ret = -1;
404 
405  _ack_state = UBX_ACK_WAITING;
406  _ack_waiting_msg = msg; // memorize sent msg class&ID for ACK check
407 
408  gps_abstime time_started = gps_absolute_time();
409 
410  while ((_ack_state == UBX_ACK_WAITING) && (gps_absolute_time() < time_started + timeout * 1000)) {
411  receive(timeout);
412  }
413 
414  if (_ack_state == UBX_ACK_GOT_ACK) {
415  ret = 0; // ACK received ok
416 
417  } else if (report) {
418  if (_ack_state == UBX_ACK_GOT_NAK) {
419  UBX_DEBUG("ubx msg 0x%04x NAK", SWAP16((unsigned)msg));
420 
421  } else {
422  UBX_DEBUG("ubx msg 0x%04x ACK timeout", SWAP16((unsigned)msg));
423  }
424  }
425 
426  _ack_state = UBX_ACK_IDLE;
427  return ret;
428 }

430 int // -1 = error, 0 = no message handled, 1 = message handled, 2 = sat info message handled
431 GPSDriverUBX::receive(unsigned timeout)
432 {
433  uint8_t buf[GPS_READ_BUFFER_SIZE];
434 
435  /* timeout additional to poll */
436  gps_abstime time_started = gps_absolute_time();
437 
438  int handled = 0;
439 
440  while (true) {
441  bool ready_to_return = _configured ? (_got_posllh && _got_velned) : handled;
442 
443  /* Wait for only UBX_PACKET_TIMEOUT if something already received. */
444  int ret = read(buf, sizeof(buf), ready_to_return ? UBX_PACKET_TIMEOUT : timeout);
445 
446  if (ret < 0) {
447  /* something went wrong when polling or reading */
448  UBX_WARN("ubx poll_or_read err");
449  return -1;
450 
451  } else if (ret == 0) {
452  /* return success if ready */
453  if (ready_to_return) {
454  _got_posllh = false;
455  _got_velned = false;
456  return handled;
457  }
458 
459  } else {
460  //UBX_DEBUG("read %d bytes", ret);
461 
462  /* pass received bytes to the packet decoder */
463  for (int i = 0; i < ret; i++) {
464  handled |= parseChar(buf[i]);
465  //UBX_DEBUG("parsed %d: 0x%x", i, buf[i]);
466  }
467 
468  if (_interface == Interface::SPI) {
469  if (buf[ret - 1] == 0xff) {
470  if (ready_to_return) {
471  _got_posllh = false;
472  _got_velned = false;
473  return handled;
474  }
475  }
476  }
477  }
478 
479  /* abort after timeout if no useful packets received */
480  if (time_started + timeout * 1000 < gps_absolute_time()) {
481  UBX_DEBUG("timed out, returning");
482  return -1;
483  }
484  }

487 int // 0 = decoding, 1 = message handled, 2 = sat info message handled
488 GPSDriverUBX::parseChar(const uint8_t b)
489 {
490  int ret = 0;
491 
492  switch (_decode_state) {
493 
494  /* Expecting Sync1 */
495  case UBX_DECODE_SYNC1:
496  if (b == UBX_SYNC1) { // Sync1 found --> expecting Sync2
497  UBX_TRACE_PARSER("A");
498  _decode_state = UBX_DECODE_SYNC2;
499 
500  } else if (b == RTCM3_PREAMBLE && _rtcm_parsing) {
501  UBX_TRACE_PARSER("RTCM");
502  _decode_state = UBX_DECODE_RTCM3;
503  _rtcm_parsing->addByte(b);
504  }
505 
506  break;
507 
508  /* Expecting Sync2 */
509  case UBX_DECODE_SYNC2:
510  if (b == UBX_SYNC2) { // Sync2 found --> expecting Class
511  UBX_TRACE_PARSER("B");
512  _decode_state = UBX_DECODE_CLASS;
513 
514  } else { // Sync1 not followed by Sync2: reset parser
515  decodeInit();
516  }
517 
518  break;
519 
520  /* Expecting Class */
521  case UBX_DECODE_CLASS:
522  UBX_TRACE_PARSER("C");
523  addByteToChecksum(b); // checksum is calculated for everything except Sync and Checksum bytes
524  _rx_msg = b;
525  _decode_state = UBX_DECODE_ID;
526  break;
527 
528  /* Expecting ID */
529  case UBX_DECODE_ID:
530  UBX_TRACE_PARSER("D");
531  addByteToChecksum(b);
532  _rx_msg |= b << 8;
533  _decode_state = UBX_DECODE_LENGTH1;
534  break;
535 
536  /* Expecting first length byte */
537  case UBX_DECODE_LENGTH1:
538  UBX_TRACE_PARSER("E");
539  addByteToChecksum(b);
540  _rx_payload_length = b;
541  _decode_state = UBX_DECODE_LENGTH2;
542  break;
543 
544  /* Expecting second length byte */
545  case UBX_DECODE_LENGTH2:
546  UBX_TRACE_PARSER("F");
547  addByteToChecksum(b);
548  _rx_payload_length |= b << 8; // calculate payload size
549 
550  if (payloadRxInit() != 0) { // start payload reception
551  // payload will not be handled, discard message
552  decodeInit();
553 
554  } else {
555  _decode_state = (_rx_payload_length > 0) ? UBX_DECODE_PAYLOAD : UBX_DECODE_CHKSUM1;
556  }
557 
558  break;
559 
560  /* Expecting payload */
561  case UBX_DECODE_PAYLOAD:
562  UBX_TRACE_PARSER(".");
563  addByteToChecksum(b);
564 
565  switch (_rx_msg) {
566  case UBX_MSG_NAV_SVINFO:
567  ret = payloadRxAddNavSvinfo(b); // add a NAV-SVINFO payload byte
568  break;
569 
570  case UBX_MSG_MON_VER:
571  ret = payloadRxAddMonVer(b); // add a MON-VER payload byte
572  break;
573 
574  default:
575  ret = payloadRxAdd(b); // add a payload byte
576  break;
577  }
578 
579  if (ret < 0) {
580  // payload not handled, discard message
581  decodeInit();
582 
583  } else if (ret > 0) {
584  // payload complete, expecting checksum
585  _decode_state = UBX_DECODE_CHKSUM1;
586 
587  } else {
588  // expecting more payload, stay in state UBX_DECODE_PAYLOAD
589  }
590 
591  ret = 0;
592  break;
593 
594  /* Expecting first checksum byte */
595  case UBX_DECODE_CHKSUM1:
596  if (_rx_ck_a != b) {
597  UBX_DEBUG("ubx checksum err");
598  decodeInit();
599 
600  } else {
601  _decode_state = UBX_DECODE_CHKSUM2;
602  }
603 
604  break;
605 
606  /* Expecting second checksum byte */
607  case UBX_DECODE_CHKSUM2:
608  if (_rx_ck_b != b) {
609  UBX_DEBUG("ubx checksum err");
610 
611  } else {
612  ret = payloadRxDone(); // finish payload processing
613  }
614 
615  decodeInit();
616  break;
617 
618  case UBX_DECODE_RTCM3:
619  if (_rtcm_parsing->addByte(b)) {
620  UBX_DEBUG("got RTCM message with length %i", (int)_rtcm_parsing->messageLength());
621  gotRTCMMessage(_rtcm_parsing->message(), _rtcm_parsing->messageLength());
622  decodeInit();
623  }
624 
625  break;
626 
627  default:
628  break;
629  }
