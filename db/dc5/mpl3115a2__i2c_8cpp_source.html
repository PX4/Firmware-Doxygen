110 int
111 MPL3115A2_I2C::reset()
112 {
113  int max = 10;
114  reg_write(MPL3115A2_CTRL_REG1, CTRL_REG1_RST);
115  int rv = CTRL_REG1_RST;
116  int ret = 1;
117 
118  while (ret == 1 && (rv & CTRL_REG1_RST) && max--) {
119  usleep(4000);
120  ret = reg_read(MPL3115A2_CTRL_REG1, &rv);
121  }
122 
123  return ret == 1 ? PX4_OK : ret;
124 }

126 int
127 MPL3115A2_I2C::read(unsigned offset, void *data, unsigned count)
128 {
129 
130  int ret = -EINVAL;
131 
132  switch (offset) {
133  case MPL3115A2_CTRL_REG1:
134  ret = reg_read(offset, data, count);
135  break;
136 
137  case OUT_P_MSB: {
138  union _cvt {
139  MPL3115A2_data_t reading;
140  } *cvt = (_cvt *)data;
141 
142  /* read the most recent measurement
143  * 3 Pressure and 2 temprtture
144  */
145  uint8_t b[3 + 2];
146  uint8_t reg = (uint8_t) offset;
147 
148  ret = transfer(&reg, 1, &b[0], sizeof(b));
149 
150  if (ret == PX4_OK) {
151  cvt->reading.pressure.q = ((uint32_t)b[0]) << 18 | ((uint32_t) b[1]) << 10 | (((uint32_t)b[2]) & 0xc0) << 2 | ((
152  b[2] & 0x30) >> 4);
153  cvt->reading.temperature.w = ((uint16_t) b[3]) << 8 | (b[4] >> 4);
154 
155  }
156  }
157  break;
158  }
159 
160  return ret;
161 }

184 int
185 MPL3115A2_I2C::probe()
186 {
187  _retries = 10;
188  uint8_t whoami = 0;
189 
190  if ((reg_read(MPL3115A2_REG_WHO_AM_I, &whoami) > 0) && (whoami == MPL3115A2_WHO_AM_I)) {
191  /*
192  * Disable retries; we may enable them selectively in some cases,
193  * but the device gets confused if we retry some of the commands.
194  */
195  _retries = 0;
196  return PX4_OK;
197  }
198 
199  return -EIO;
200 }
