247  /**
248  * Calls all callback waiting to be called
249  */
250  void spinOnce()
251  {
252  /* Loop through subscriptions, call callback for updated subscriptions */
253  SubscriberNode *sub = _subs.getHead();
254  int count = 0;
255 
256  while (sub != nullptr) {
257  if (count++ > kMaxSubscriptions) {
258  PX4_WARN("exceeded max subscriptions");
259  break;
260  }
261 
262  sub->update();
263  sub = sub->getSibling();
264  }
265  }

267  /**
268  * Keeps calling callbacks for incoming messages, returns when module is terminated
269  */
270  void spin()
271  {
272  while (!_appState.exitRequested()) {
273  const int timeout_ms = 100;
274 
275  /* Only continue in the loop if the nodehandle has subscriptions */
276  if (_sub_min_interval == nullptr) {
277  usleep(timeout_ms * 1000);
278  continue;
279  }
280 
281  /* Poll fd with smallest interval */
282  px4_pollfd_struct_t pfd;
283  pfd.fd = _sub_min_interval->getUORBHandle();
284  pfd.events = POLLIN;
285  px4_poll(&pfd, 1, timeout_ms);
286  spinOnce();
287  }
288  }
