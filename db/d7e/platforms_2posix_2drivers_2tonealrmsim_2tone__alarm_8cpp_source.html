34 /**
35  * Driver for the PX4 audio alarm port, /dev/tone_alarm.
36  *
37  * The tone_alarm driver supports a set of predefined "alarm"
38  * tunes and one user-supplied tune.
39  *
40  * The TONE_SET_ALARM ioctl can be used to select a predefined
41  * alarm tune, from 1 - <TBD>. Selecting tune zero silences
42  * the alarm.
43  *
44  * Tunes follow the syntax of the Microsoft GWBasic/QBasic PLAY
45  * statement, with some exceptions and extensions.
46  *
47  * From Wikibooks:
48  *
49  * PLAY "[string expression]"
50  *
51  * Used to play notes and a score ... The tones are indicated by letters A through G.
52  * Accidentals are indicated with a "+" or "#" (for sharp) or "-" (for flat)
53  * immediately after the note letter. See this example:
54  *
55  * PLAY "C C# C C#"
56  *
57  * Whitespaces are ignored inside the string expression. There are also codes that
58  * set the duration, octave and tempo. They are all case-insensitive. PLAY executes
59  * the commands or notes the order in which they appear in the string. Any indicators
60  * that change the properties are effective for the notes following that indicator.
61  *
62  * Ln Sets the duration (length) of the notes. The variable n does not indicate an actual duration
63  * amount but rather a note type; L1 - whole note, L2 - half note, L4 - quarter note, etc.
64  * (L8, L16, L32, L64, ...). By default, n = 4.
65  * For triplets and quintets, use L3, L6, L12, ... and L5, L10, L20, ... series respectively.
66  * The shorthand notation of length is also provided for a note. For example, "L4 CDE L8 FG L4 AB"
67  * can be shortened to "L4 CDE F8G8 AB". F and G play as eighth notes while others play as quarter notes.
68  * On Sets the current octave. Valid values for n are 0 through 6. An octave begins with C and ends with B.
69  * Remember that C- is equivalent to B.
70  * < > Changes the current octave respectively down or up one level.
71  * Nn Plays a specified note in the seven-octave range. Valid values are from 0 to 84. (0 is a pause.)
72  * Cannot use with sharp and flat. Cannot use with the shorthand notation neither.
73  * MN Stand for Music Normal. Note duration is 7/8ths of the length indicated by Ln. It is the default mode.
74  * ML Stand for Music Legato. Note duration is full length of that indicated by Ln.
75  * MS Stand for Music Staccato. Note duration is 3/4ths of the length indicated by Ln.
76  * Pn Causes a silence (pause) for the length of note indicated (same as Ln).
77  * Tn Sets the number of "L4"s per minute (tempo). Valid values are from 32 to 255. The default value is T120.
78  * . When placed after a note, it causes the duration of the note to be 3/2 of the set duration.
79  * This is how to get "dotted" notes. "L4 C#." would play C sharp as a dotted quarter note.
80  * It can be used for a pause as well.
81  *
82  * Extensions/variations:
83  *
84  * MB MF The MF command causes the tune to play once and then stop. The MB command causes the
85  * tune to repeat when it ends.
86  *
87  */

253 unsigned
254 ToneAlarm::note_to_divisor(unsigned note)
255 {
256  const int TONE_ALARM_CLOCK = 120000000ul / 4;
257 
258  // compute the frequency first (Hz)
259  float freq = 880.0f * expf(logf(2.0f) * ((int)note - 46) / 12.0f);
260 
261  float period = 0.5f / freq;
262 
263  // and the divisor, rounded to the nearest integer
264  unsigned divisor = (period * TONE_ALARM_CLOCK) + 0.5f;
265 
266  return divisor;
267 }

269 unsigned
270 ToneAlarm::note_duration(unsigned &silence, unsigned note_length, unsigned dots)
271 {
272  unsigned whole_note_period = (60 * 1000000 * 4) / _tempo;
273 
274  if (note_length == 0) {
275  note_length = 1;
276  }
277 
278  unsigned note_period = whole_note_period / note_length;
279 
280  switch (_note_mode) {
281  case MODE_NORMAL:
282  silence = note_period / 8;
283  break;
284 
285  case MODE_STACCATO:
286  silence = note_period / 4;
287  break;
288 
289  default:
290  case MODE_LEGATO:
291  silence = 0;
292  break;
293  }
294 
295  note_period -= silence;
296 
297  unsigned dot_extension = note_period / 2;
298 
299  while (dots--) {
300  note_period += dot_extension;
301  dot_extension /= 2;
302  }
303 
304  return note_period;
305 }

333 ToneAlarm::start_note(unsigned note)
334 {
335  // compute the divisor
336  unsigned divisor = note_to_divisor(note);
337 
338  // pick the lowest prescaler value that we can use
339  // (note that the effective prescale value is 1 greater)
340  unsigned prescale = divisor / 65536;
341 
342  // calculate the timer period for the selected prescaler value
343  unsigned period = (divisor / (prescale + 1)) - 1;
344 
345  // Silence warning of unused var
346  do_something(period);
347  PX4_DEBUG("ToneAlarm::start_note %u", period);

379 void
380 ToneAlarm::next_note()
381 {
382  // do we have an inter-note gap to wait for?
383  if (_silence_length > 0) {
384  stop_note();
385  _note_call = {};
386  hrt_call_after(&_note_call, (hrt_abstime)_silence_length, (hrt_callout)next_trampoline, this);
387  _silence_length = 0;
388  return;
389  }
390 
391  // make sure we still have a tune - may be removed by the write / ioctl handler
392  if ((_next == nullptr) || (_tune == nullptr)) {
393  stop_note();
394  return;
395  }
396 
397  // parse characters out of the string until we have resolved a note
398  unsigned note = 0;
399  unsigned note_length = _note_length;
400  unsigned duration;
401 
402  while (note == 0) {
403  // we always need at least one character from the string
404  int c = next_char();
405 
406  if (c == 0) {
407  goto tune_end;
408  }
409 
410  _next++;
411 
412  switch (c) {
413  case 'L': // select note length
414  _note_length = next_number();
415 
416  if (_note_length < 1) {
417  goto tune_error;
418  }
419 
420  break;
421 
422  case 'O': // select octave
423  _octave = next_number();
424 
425  if (_octave > 6) {
426  _octave = 6;
427  }
428 
429  break;
430 
431  case '<': // decrease octave
432  if (_octave > 0) {
433  _octave--;
434  }
435 
436  break;
437 
438  case '>': // increase octave
439  if (_octave < 6) {
440  _octave++;
441  }
442 
443  break;
444 
445  case 'M': // select inter-note gap
446  c = next_char();
447 
448  if (c == 0) {
449  goto tune_error;
450  }
451 
452  _next++;
453 
454  switch (c) {
455  case 'N':
456  _note_mode = MODE_NORMAL;
457  break;
458 
459  case 'L':
460  _note_mode = MODE_LEGATO;
461  break;
462 
463  case 'S':
464  _note_mode = MODE_STACCATO;
465  break;
466 
467  case 'F':
468  _repeat = false;
469  break;
470 
471  case 'B':
472  _repeat = true;
473  break;
474 
475  default:
476  goto tune_error;
477  }
478 
479  break;
480 
481  case 'P': // pause for a note length
482  stop_note();
483  _note_call = {};
484  hrt_call_after(&_note_call,
485  (hrt_abstime)rest_duration(next_number(), next_dots()),
486  (hrt_callout)next_trampoline,
487  this);
488  return;
489 
490  case 'T': { // change tempo
491  unsigned nt = next_number();
492 
493  if ((nt >= 32) && (nt <= 255)) {
494  _tempo = nt;
495 
496  } else {
497  goto tune_error;
498  }
499 
500  break;
501  }
502 
503  case 'N': // play an arbitrary note
504  note = next_number();
505 
506  if (note > 84) {
507  goto tune_error;
508  }
509 
510  if (note == 0) {
511  // this is a rest - pause for the current note length
512  _note_call = {};
513  hrt_call_after(&_note_call,
514  (hrt_abstime)rest_duration(_note_length, next_dots()),
515  (hrt_callout)next_trampoline,
516  this);
517  return;
518  }
519 
520  break;
521 
522  case 'A'...'G': // play a note in the current octave
523  note = _note_tab[c - 'A'] + (_octave * 12) + 1;
524  c = next_char();
525 
526  switch (c) {
527  case '#': // up a semitone
528  case '+':
529  if (note < 84) {
530  note++;
531  }
532 
533  _next++;
534  break;
535 
536  case '-': // down a semitone
537  if (note > 1) {
538  note--;
539  }
540 
541  _next++;
542  break;
543 
544  default:
545  // 0 / no next char here is OK
546  break;
547  }
548 
549  // shorthand length notation
550  note_length = next_number();
551 
552  if (note_length == 0) {
553  note_length = _note_length;
554  }
555 
556  break;
557 
558  default:
559  goto tune_error;
560  }
561  }
562 
563  // compute the duration of the note and the following silence (if any)
564  duration = note_duration(_silence_length, note_length, next_dots());
565 
566  // start playing the note
567  start_note(note);
568 
569  // and arrange a callback when the note should stop
570  _note_call = {};
571  hrt_call_after(&_note_call, (hrt_abstime)duration, (hrt_callout)next_trampoline, this);
572  return;
573 
574  // tune looks bad (unexpected EOF, bad character, etc.)
575 tune_error:
576  PX4_ERR("tune error\n");
577  _repeat = false; // don't loop on error
578 
579  // stop (and potentially restart) the tune
580 tune_end:
581  stop_note();
582 
583  if (_repeat) {
584  start_tune(_tune);
585 
586  } else {
587  _tune = nullptr;
588  _default_tune_number = 0;
589  }
590 
591 }

643 int
644 ToneAlarm::devIOCTL(unsigned long cmd, unsigned long arg)
645 {
646  int result = OK;
647 
648  PX4_DEBUG("ToneAlarm::devIOCTL %i %lu", cmd, arg);
649 
650  /* decide whether to increase the alarm level to cmd or leave it alone */
651  switch (cmd) {
652  case TONE_SET_ALARM:
653  if (arg < TONE_NUMBER_OF_TUNES) {
654  if (arg == TONE_STOP_TUNE) {
655  // stop the tune
656  _tune = nullptr;
657  _next = nullptr;
658  _repeat = false;
659  _default_tune_number = 0;
660 
661  } else {
662  /* always interrupt alarms, unless they are repeating and already playing */
663  if (!(_repeat && _default_tune_number == arg)) {
664  /* play the selected tune */
665  _default_tune_number = arg;
666  start_tune(_default_tunes[arg]);
667  PX4_INFO("%s", _tune_names[arg]);
668  }
669  }
670 
671  } else {
672  result = -EINVAL;
673  }
674 
675  break;
676 
677  default:
678  result = -ENOTTY;
679  break;
680  }
681 
682  /* give it to the superclass if we didn't like it */
683  if (result == -ENOTTY) {
684  result = VirtDevObj::devIOCTL(cmd, arg);
685  }
686 
687  return result;
688 }

690 ssize_t
691 ToneAlarm::devWrite(const void *buffer, size_t len)
692 {
693  // sanity-check the buffer for length and nul-termination
694  if (len > _tune_max) {
695  return -EFBIG;
696  }
697 
698  // if we have an existing user tune, free it
699  if (_user_tune != nullptr) {
700 
701  // if we are playing the user tune, stop
702  if (_tune == _user_tune) {
703  _tune = nullptr;
704  _next = nullptr;
705  }
706 
707  // free the old user tune
708  free((void *)_user_tune);
709  _user_tune = nullptr;
710  }
711 
712  const char *buf = reinterpret_cast<const char *>(buffer);
713 
714  // if the new tune is empty, we're done
715  if (buf[0] == '\0') {
716  return OK;
717  }
718 
719  // allocate a copy of the new tune
