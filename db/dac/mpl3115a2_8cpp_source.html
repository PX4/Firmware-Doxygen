329 ssize_t
330 MPL3115A2::read(struct file *filp, char *buffer, size_t buflen)
331 {
332  unsigned count = buflen / sizeof(struct baro_report);
333  struct baro_report *brp = reinterpret_cast<struct baro_report *>(buffer);
334  int ret = 0;
335 
336  /* buffer must be large enough */
337  if (count < 1) {
338  return -ENOSPC;
339  }
340 
341  /* if automatic measurement is enabled */
342  if (_measure_ticks > 0) {
343 
344  /*
345  * While there is space in the caller's buffer, and reports, copy them.
346  * Note that we may be pre-empted by the workq thread while we are doing this;
347  * we are careful to avoid racing with them.
348  */
349  while (count--) {
350  if (_reports->get(brp)) {
351  ret += sizeof(*brp);
352  brp++;
353  }
354  }
355 
356  /* if there was no data, warn the caller */
357  return ret ? ret : -EAGAIN;
358  }
359 
360  /* manual measurement - run one conversion */
361  do {
362  _reports->flush();
363 
364  /* First reading */
365  do {
366  ret = measure();
367 
368  if (ret == -EAGAIN) {
369  usleep(1000);
370  }
371  } while (ret == -EAGAIN);
372 
373  if (ret != OK) {
374  ret = -EIO;
375  break;
376  }
377 
378  if (OK != collect()) {
379  ret = -EIO;
380  break;
381  }
382 
383  /* state machine will have generated a report, copy it out */
384  if (_reports->get(brp)) {
385  ret = sizeof(*brp);
386  }
387 
388  } while (0);
389 
390  return ret;
391 }

393 int
394 MPL3115A2::ioctl(struct file *filp, int cmd, unsigned long arg)
395 {
396  switch (cmd) {
397 
398  case SENSORIOCSPOLLRATE: {
399  switch (arg) {
400 
401  /* switching to manual polling */
402  case SENSOR_POLLRATE_MANUAL:
403  stop_cycle();
404  _measure_ticks = 0;
405  return OK;
406 
407  /* external signalling not supported */
408  case SENSOR_POLLRATE_EXTERNAL:
409 
410  /* zero would be bad */
411  case 0:
412  return -EINVAL;
413 
414  /* set default/max polling rate */
415  case SENSOR_POLLRATE_MAX:
416  case SENSOR_POLLRATE_DEFAULT: {
417  /* do we need to start internal polling? */
418  bool want_start = (_measure_ticks == 0);
419 
420  /* set interval for next measurement to minimum legal value */
421  _measure_ticks = USEC2TICK(MPL3115A2_CONVERSION_INTERVAL);
422 
423  /* if we need to start the poll state machine, do it */
424  if (want_start) {
425  start_cycle();
426  }
427 
428  return OK;
429  }
430 
431  /* adjust to a legal polling interval in Hz */
432  default: {
433  /* do we need to start internal polling? */
434  bool want_start = (_measure_ticks == 0);
435 
436  /* convert hz to tick interval via microseconds */
437  unsigned ticks = USEC2TICK(1000000 / arg);
438 
439  /* check against maximum rate */
440  if (ticks < USEC2TICK(MPL3115A2_CONVERSION_INTERVAL)) {
441  return -EINVAL;
442  }
443 
444  /* update interval for next measurement */
445  _measure_ticks = ticks;
446 
447  /* if we need to start the poll state machine, do it */
448  if (want_start) {
449  start_cycle();
450  }
451 
452  return OK;
453  }
454  }
455  }
456 
457  case SENSORIOCGPOLLRATE:
458  if (_measure_ticks == 0) {
459  return SENSOR_POLLRATE_MANUAL;
460  }
461 
462  return (1000 / _measure_ticks);
463 
464  case SENSORIOCSQUEUEDEPTH: {
465  /* lower bound is mandatory, upper bound is a sanity check */
466  if ((arg < 1) || (arg > 100)) {
467  return -EINVAL;
468  }
469 
470  irqstate_t flags = px4_enter_critical_section();
471 
472  if (!_reports->resize(arg)) {
473  px4_leave_critical_section(flags);
474  return -ENOMEM;
475  }
476 
477  px4_leave_critical_section(flags);
478  return OK;
479  }
480 
481  case SENSORIOCRESET: {
482  /*
483  * Since we are initialized, we do not need to do anything, since the
484  * PROM is correctly read and the part does not need to be configured.
485  */
486  unsigned int dummy;
487  _interface->ioctl(IOCTL_RESET, dummy);
488  return OK;
489  }
490 
491  default:
492  break;
493  }
494 
495  /* give it to the bus-specific superclass */
496  // return bus_ioctl(filp, cmd, arg);
497  return CDev::ioctl(filp, cmd, arg);
498 }

526 void
527 MPL3115A2::cycle()
528 {
529  int ret;
530  unsigned dummy;
531 
532  /* collection phase? */
533  if (_collect_phase) {
534 
535  /* perform collection */
536  ret = collect();
537 
538  if (ret == -EIO) {
539  /* issue a reset command to the sensor */
540  _interface->ioctl(IOCTL_RESET, dummy);
541  /* reset the collection state machine and try again - we need
542  * to wait 2.8 ms after issuing the sensor reset command
543  * according to the MPL3115A2 datasheet
544  */
545  start_cycle(USEC2TICK(2800));
546  return;
547  }
548 
549 
550  if (ret == -EAGAIN) {
551 
552  /* Ready read it on next cycle */
553  work_queue(HPWORK,
554  &_work,
555  (worker_t)&MPL3115A2::cycle_trampoline,
556  this,
557  USEC2TICK(MPL3115A2_CONVERSION_INTERVAL));
558  return;
559  }
560 
561  /* next phase is measurement */
562  _collect_phase = false;
563  }
564 
565  /* Look for a ready condition */
566 
567  ret = measure();
568 
569  if (ret == -EIO) {
570  /* issue a reset command to the sensor */
571  _interface->ioctl(IOCTL_RESET, dummy);
572  /* reset the collection state machine and try again */
573  start_cycle();
574  return;
575  }
576 
577  /* next phase is measurement */
578  _collect_phase = true;
579 
580  /* schedule a fresh cycle call when the measurement is done */
581  work_queue(HPWORK,
582  &_work,
583  (worker_t)&MPL3115A2::cycle_trampoline,
584  this,
585  USEC2TICK(MPL3115A2_CONVERSION_INTERVAL));
586 }

610 int
611 MPL3115A2::collect()
612 {
613  int ret;
614  MPL3115A2_data_t reading;
615  uint8_t ctrl;
616 
617  perf_begin(_sample_perf);
618 
619  ret = _interface->read(MPL3115A2_CTRL_REG1, (void *)&ctrl, 1);
620 
621  if (ret == -EIO) {
622  perf_end(_sample_perf);
623  return ret;
624  }
625 
626  if (ctrl & CTRL_REG1_OST) {
627  perf_end(_sample_perf);
628  return -EAGAIN;
629  }
630 
631  struct baro_report report;
632 
633  /* this should be fairly close to the end of the conversion, so the best approximation of the time */
634  report.timestamp = hrt_absolute_time();
635 
636  report.error_count = perf_event_count(_comms_errors);
637 
638  /* read the most recent measurement - read offset/size are hardcoded in the interface */
639  ret = _interface->read(OUT_P_MSB, (void *)&reading, sizeof(reading));
640 
641  if (ret == -EIO) {
642  perf_count(_comms_errors);
643  perf_end(_sample_perf);
644  return ret;
645  }
646 
647 
648  _T = (float) reading.temperature.b[1] + ((float)(reading.temperature.b[0]) / 16.0f);
649 
650  _P = (float)(reading.pressure.q >> 8) + ((float)(reading.pressure.b[0]) / 4.0f);
651 
652  report.temperature = _T;
653  report.pressure = _P / 100.0f; /* convert to millibar */
654 
655  /* return device ID */
656  report.device_id = _device_id.devid;
657 
658  /* altitude calculations based on http://www.kansasflyer.org/index.asp?nav=Avi&sec=Alti&tab=Theory&pg=1 */
659 
660  /* publish it */
661  if (!(_pub_blocked) && _baro_topic != nullptr) {
662  /* publish it */
663  orb_publish(ORB_ID(sensor_baro), _baro_topic, &report);
664  }
665 
666  _reports->force(&report);
667 
668  /* notify anyone waiting for data */
669  poll_notify(POLLIN);
670 
671  perf_end(_sample_perf);
672 
673  return OK;
674 }

829  * make sure we can collect data from the sensor in polled

832 void
833 test(enum MPL3115A2_BUS busid)
834 {
835  struct mpl3115a2_bus_option &bus = find_bus(busid);
836  struct baro_report report;
837  ssize_t sz;
838  int ret;
839 
840  int fd;
841 
842  fd = open(bus.devpath, O_RDONLY);
843 
844  if (fd < 0) {
845  err(1, "open failed (try 'mpl3115a2 start' if the driver is not running)");
846  }
847 
848  /* do a simple demand read */
849  sz = read(fd, &report, sizeof(report));
850 
851  if (sz != sizeof(report)) {
852  err(1, "immediate read failed");
853  }
854 
855  print_message(report);
856 
857  /* set the queue depth to 10 */
858  if (OK != ioctl(fd, SENSORIOCSQUEUEDEPTH, 10)) {
859  errx(1, "failed to set queue depth");
860  }
861 
862  /* start the sensor polling at 2Hz */
863  if (OK != ioctl(fd, SENSORIOCSPOLLRATE, 2)) {
864  errx(1, "failed to set 2Hz poll rate");
865  }
866 
867  /* read the sensor 5x and report each value */
868  for (unsigned i = 0; i < 5; i++) {
869  struct pollfd fds;
870 
871  /* wait for data to be ready */
872  fds.fd = fd;
873  fds.events = POLLIN;
874  ret = poll(&fds, 1, 2000);
875 
876  if (ret != 1) {
877  errx(1, "timed out waiting for sensor data");
878  }
879 
880  /* now go get it */
881  sz = read(fd, &report, sizeof(report));
882 
883  if (sz != sizeof(report)) {
884  err(1, "periodic read failed");
885  }
886 
887  print_message(report);
888  }
889 
890  close(fd);
891  errx(0, "PASS");
892 }
