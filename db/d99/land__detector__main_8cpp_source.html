61 int LandDetector::task_spawn(int argc, char *argv[])
62 {
63  if (argc < 2) {
64  print_usage();
65  return -1;
66  }
67 
68  LandDetector *obj;
69 
70  if (strcmp(argv[1], "fixedwing") == 0) {
71  obj = new FixedwingLandDetector();
72 
73  } else if (strcmp(argv[1], "multicopter") == 0) {
74  obj = new MulticopterLandDetector();
75 
76  } else if (strcmp(argv[1], "vtol") == 0) {
77  obj = new VtolLandDetector();
78 
79  } else if (strcmp(argv[1], "ugv") == 0) {
80  obj = new RoverLandDetector();
81 
82  } else {
83  print_usage("unknown mode");
84  return -1;
85  }
86 
87  if (obj == nullptr) {
88  PX4_ERR("alloc failed");
89  return -1;
90  }
91 
92  int ret = obj->start();
93 
94  if (ret < 0) {
95  delete obj;
96  return ret;
97  }
98 
99  // Remember current active mode
100  strncpy(_currentMode, argv[1], sizeof(_currentMode) - 1);
101  _currentMode[sizeof(_currentMode) - 1] = '\0';
102 
103  wait_until_running(); // this will wait until _object is set from the cycle method
104  _task_id = task_id_is_work_queue;
105  return 0;
106 }

134 int LandDetector::print_usage(const char *reason)
135 {
136  if (reason != nullptr) {

139 
140  PRINT_MODULE_DESCRIPTION(
141  R"DESCR_STR(
142 ### Description
143 Module to detect the freefall and landed state of the vehicle, and publishing the `vehicle_land_detected` topic.
144 Each vehicle type (multirotor, fixedwing, vtol, ...) provides its own algorithm, taking into account various
145 states, such as commanded thrust, arming state and vehicle motion.
146 
147 ### Implementation
148 Every type is implemented in its own class with a common base class. The base class maintains a state (landed,
149 maybe_landed, ground_contact). Each possible state is implemented in the derived classes. A hysteresis and a fixed
150 priority of each internal state determines the actual land_detector state.
151 
152 #### Multicopter Land Detector
153 **ground_contact**: thrust setpoint and velocity in z-direction must be below a defined threshold for time
154 GROUND_CONTACT_TRIGGER_TIME_US. When ground_contact is detected, the position controller turns off the thrust setpoint
155 in body x and y.
156 
157 **maybe_landed**: it requires ground_contact together with a tighter thrust setpoint threshold and no velocity in the
158 horizontal direction. The trigger time is defined by MAYBE_LAND_TRIGGER_TIME. When maybe_landed is detected, the
159 position controller sets the thrust setpoint to zero.
160 
161 **landed**: it requires maybe_landed to be true for time LAND_DETECTOR_TRIGGER_TIME_US.
162 
163 The module runs periodically on the HP work queue.
164 )DESCR_STR");
165 
166  PRINT_MODULE_USAGE_NAME("land_detector", "system");
167  PRINT_MODULE_USAGE_COMMAND_DESCR("start", "Start the background task");
168  PRINT_MODULE_USAGE_ARG("fixedwing|multicopter|vtol|ugv", "Select vehicle type", false);
169  PRINT_MODULE_USAGE_DEFAULT_COMMANDS();
170  return 0;
171 }
