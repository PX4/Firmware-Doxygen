88 void LogWriterFile::start_log(const char *filename)
89 {
90  // register the current file with the hardfault handler: if the system crashes,
91  // the hardfault handler will append the crash log to that file on the next reboot.
92  // Note that we don't deregister it when closing the log, so that crashes after disarming
93  // are appended as well (the same holds for crashes before arming, which can be a bit misleading)
94  int ret = hardfault_store_filename(filename);
95 
96  if (ret) {
97  PX4_ERR("Failed to register ULog file to the hardfault handler (%i)", ret);
98  }
99 
100  _fd = ::open(filename, O_CREAT | O_WRONLY, PX4_O_MODE_666);
101 
102  if (_fd < 0) {
103  PX4_ERR("Can't open log file %s, errno: %d", filename, errno);
104  _should_run = false;
105  return;
106  }
107 
108  if (_buffer == nullptr) {
109  _buffer = new uint8_t[_buffer_size];
110 
111  if (_buffer == nullptr) {
112  PX4_ERR("Can't create log buffer");
113  ::close(_fd);
114  _fd = -1;
115  _should_run = false;
116  return;
117  }
118  }
119 
120  PX4_INFO("Opened log file: %s", filename);
121  _should_run = true;
122  _running = true;
123 
124  // Clear buffer and counters
125  _head = 0;
126  _count = 0;
127  _total_written = 0;
128  notify();
129 }

212 void LogWriterFile::run()
213 {
214  while (!_exit_thread) {
215  // Outer endless loop
216  // Wait for _should_run flag
217  while (!_exit_thread) {
218  bool start = false;
219  pthread_mutex_lock(&_mtx);
220  pthread_cond_wait(&_cv, &_mtx);
221  start = _should_run;
222  pthread_mutex_unlock(&_mtx);
223 
224  if (start) {
225  break;
226  }
227  }
228 
229  if (_exit_thread) {
230  break;
231  }
232 
233  int poll_count = 0;
234  int written = 0;
235  hrt_abstime last_fsync = hrt_absolute_time();
236 
237  while (true) {
238  size_t available = 0;
239  void *read_ptr = nullptr;
240  bool is_part = false;
241 
242  /* lock _buffer
243  * wait for sufficient data, cycle on notify()
244  */
245  pthread_mutex_lock(&_mtx);
246 
247  while (true) {
248  available = get_read_ptr(&read_ptr, &is_part);
249 
250  /* if sufficient data available or partial read or terminating, exit this wait loop */
251  if ((available >= _min_write_chunk) || is_part || !_should_run) {
252  /* GOTO end of block */
253  break;
254  }
255 
256  /* wait for a call to notify()
257  * this call unlocks the mutex while waiting, and returns with the mutex locked
258  */
259  pthread_cond_wait(&_cv, &_mtx);
260  }
261 
262  pthread_mutex_unlock(&_mtx);
263  written = 0;
264 
265  if (available > 0) {
266  perf_begin(_perf_write);
267  written = ::write(_fd, read_ptr, available);
268  perf_end(_perf_write);
269 
270  const hrt_abstime now = hrt_absolute_time();
271 
272  /* call fsync periodically to minimize potential loss of data */
273  if (++poll_count >= 100 || now - last_fsync > 1_s) {
274  perf_begin(_perf_fsync);
275  ::fsync(_fd);
276  perf_end(_perf_fsync);
277  poll_count = 0;
278  last_fsync = now;
279  }
280 
281  if (written < 0) {
282  PX4_WARN("error writing log file");
283  _should_run = false;
284  /* GOTO end of block */
285  break;
286  }
287 
288  pthread_mutex_lock(&_mtx);
289  /* subtract bytes written from number in _buffer (_count -= written) */
290  mark_read(written);
291  pthread_mutex_unlock(&_mtx);
292 
293  _total_written += written;
294  }
295 
296  if (!_should_run && written == static_cast<int>(available) && !is_part) {
297  // Stop only when all data written
298  _running = false;
299  _head = 0;
300  _count = 0;
301 
302  if (_fd >= 0) {
303  int res = ::close(_fd);
304  _fd = -1;
305 
306  if (res) {
307  PX4_WARN("error closing log file");
308 
309  } else {
310  PX4_INFO("closed logfile, bytes written: %zu", _total_written);
311  }
312  }
313 
314  break;
315  }
316  }
317  }

338 int LogWriterFile::write(void *ptr, size_t size, uint64_t dropout_start)
339 {
340  if (!is_started()) {
341  return 0;
342  }
343 
344  // Bytes available to write
345  size_t available = _buffer_size - _count;
346  size_t dropout_size = 0;
347 
348  if (dropout_start) {
349  dropout_size = sizeof(ulog_message_dropout_s);
350  }
351 
352  if (size + dropout_size > available) {
353  // buffer overflow
354  return -1;
355  }
356 
357  if (dropout_start) {
358  //write dropout msg
359  ulog_message_dropout_s dropout_msg;
360  dropout_msg.duration = (uint16_t)(hrt_elapsed_time(&dropout_start) / 1000);
361  write_no_check(&dropout_msg, sizeof(dropout_msg));
362  }
363 
364  write_no_check(ptr, size);
365  return 0;
366 }

368 void LogWriterFile::write_no_check(void *ptr, size_t size)
369 {
370  size_t n = _buffer_size - _head; // bytes to end of the buffer
371 
372  uint8_t *buffer_c = reinterpret_cast<uint8_t *>(ptr);
373 
374  if (size > n) {
375  // Message goes over the end of the buffer
376  memcpy(&(_buffer[_head]), buffer_c, n);
377  _head = 0;
378 
379  } else {
380  n = 0;
381  }
382 
383  // now: n = bytes already written
384  size_t p = size - n; // number of bytes to write
385  memcpy(&(_buffer[_head]), &(buffer_c[n]), p);
386  _head = (_head + p) % _buffer_size;
387  _count += size;
388 }
