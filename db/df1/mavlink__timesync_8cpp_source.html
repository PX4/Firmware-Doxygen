56 void
57 MavlinkTimesync::handle_message(const mavlink_message_t *msg)
58 {
59  switch (msg->msgid) {
60  case MAVLINK_MSG_ID_TIMESYNC: {
61 
62  mavlink_timesync_t tsync = {};
63  mavlink_msg_timesync_decode(msg, &tsync);
64 
65  const uint64_t now = hrt_absolute_time();
66 
67  if (tsync.tc1 == 0) { // Message originating from remote system, timestamp and return it
68 
69  mavlink_timesync_t rsync;
70 
71  rsync.tc1 = now * 1000ULL;
72  rsync.ts1 = tsync.ts1;
73 
74  mavlink_msg_timesync_send_struct(_mavlink->get_channel(), &rsync);
75 
76  return;
77 
78  } else if (tsync.tc1 > 0) { // Message originating from this system, compute time offset from it
79 
80  // Calculate time offset between this system and the remote system, assuming RTT for
81  // the timesync packet is roughly equal both ways.
82  int64_t offset_us = (int64_t)((tsync.ts1 / 1000ULL) + now - (tsync.tc1 / 1000ULL) * 2) / 2 ;
83 
84  // Calculate the round trip time (RTT) it took the timesync packet to bounce back to us from remote system
85  uint64_t rtt_us = now - (tsync.ts1 / 1000ULL);
86 
87  // Calculate the difference of this sample from the current estimate
88  uint64_t deviation = llabs((int64_t)_time_offset - offset_us);
89 
90  if (rtt_us < MAX_RTT_SAMPLE) { // Only use samples with low RTT
91 
92  if (sync_converged() && (deviation > MAX_DEVIATION_SAMPLE)) {
93 
94  // Increment the counter if we have a good estimate and are getting samples far from the estimate
95  _high_deviation_count++;
96 
97  // We reset the filter if we received 5 consecutive samples which violate our present estimate.
98  // This is most likely due to a time jump on the offboard system.
99  if (_high_deviation_count > MAX_CONSECUTIVE_HIGH_DEVIATION) {
100  PX4_ERR("[timesync] Time jump detected. Resetting time synchroniser.");
101  // Reset the filter
102  reset_filter();
103  }
104 
105  } else {
106 
107  // Filter gain scheduling
108  if (!sync_converged()) {
109  // Interpolate with a sigmoid function
110  double progress = (double)_sequence / (double)CONVERGENCE_WINDOW;
111  double p = 1.0 - exp(0.5 * (1.0 - 1.0 / (1.0 - progress)));
112  _filter_alpha = p * ALPHA_GAIN_FINAL + (1.0 - p) * ALPHA_GAIN_INITIAL;
113  _filter_beta = p * BETA_GAIN_FINAL + (1.0 - p) * BETA_GAIN_INITIAL;
114 
115  } else {
116  _filter_alpha = ALPHA_GAIN_FINAL;
117  _filter_beta = BETA_GAIN_FINAL;
118  }
119 
120  // Perform filter update
121  add_sample(offset_us);
122 
123  // Increment sequence counter after filter update
124  _sequence++;
125 
126  // Reset high deviation count after filter update
127  _high_deviation_count = 0;
128 
129  // Reset high RTT count after filter update
130  _high_rtt_count = 0;
131  }
132 
133  } else {
134  // Increment counter if round trip time is too high for accurate timesync
135  _high_rtt_count++;
136 
137  if (_high_rtt_count > MAX_CONSECUTIVE_HIGH_RTT) {
138  PX4_WARN("[timesync] RTT too high for timesync: %llu ms", rtt_us / 1000ULL);
139  // Reset counter to rate-limit warnings
140  _high_rtt_count = 0;
141  }
142 
143  }
144 
145  // Publish status message
146  struct timesync_status_s tsync_status = {};
147 
148  tsync_status.timestamp = hrt_absolute_time();
149  tsync_status.remote_timestamp = tsync.tc1 / 1000ULL;
150  tsync_status.observed_offset = offset_us;
151  tsync_status.estimated_offset = (int64_t)_time_offset;
152  tsync_status.round_trip_time = rtt_us;
153 
154  if (_timesync_status_pub == nullptr) {
155  int instance;
156  _timesync_status_pub = orb_advertise_multi(ORB_ID(timesync_status), &tsync_status, &instance, ORB_PRIO_DEFAULT);
157 
158  } else {
159  orb_publish(ORB_ID(timesync_status), _timesync_status_pub, &tsync_status);
160  }
161 
162  }
163 
164  break;
165  }
166 
167  case MAVLINK_MSG_ID_SYSTEM_TIME: {
168 
169  mavlink_system_time_t time;
170  mavlink_msg_system_time_decode(msg, &time);
171 
172  timespec tv = {};
173  px4_clock_gettime(CLOCK_REALTIME, &tv);
174 
175  // date -d @1234567890: Sat Feb 14 02:31:30 MSK 2009
176  bool onb_unix_valid = (unsigned long long)tv.tv_sec > PX4_EPOCH_SECS;
177  bool ofb_unix_valid = time.time_unix_usec > PX4_EPOCH_SECS * 1000ULL;
178 
179  if (!onb_unix_valid && ofb_unix_valid) {
180  tv.tv_sec = time.time_unix_usec / 1000000ULL;
181  tv.tv_nsec = (time.time_unix_usec % 1000000ULL) * 1000ULL;
182 
183  if (px4_clock_settime(CLOCK_REALTIME, &tv)) {
184  PX4_ERR("[timesync] Failed setting realtime clock");
185  }
186  }
187 
188  break;
189  }
190 
191  default:
192  break;
193  }
194 }

214 void
215 MavlinkTimesync::add_sample(int64_t offset_us)
216 {
217  /* Online exponential smoothing filter. The derivative of the estimate is also
218  * estimated in order to produce an estimate without steady state lag:
219  * https://en.wikipedia.org/wiki/Exponential_smoothing#Double_exponential_smoothing
220  */
221 
222  double time_offset_prev = _time_offset;
223 
224  if (_sequence == 0) { // First offset sample
225  _time_offset = offset_us;
226 
227  } else {
228  // Update the clock offset estimate
229  _time_offset = _filter_alpha * offset_us + (1.0 - _filter_alpha) * (_time_offset + _time_skew);
230 
231  // Update the clock skew estimate
232  _time_skew = _filter_beta * (_time_offset - time_offset_prev) + (1.0 - _filter_beta) * _time_skew;
233  }
234 
235 }
