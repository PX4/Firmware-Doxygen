73 #define PERF(name, op, count) do { \
74  px4_usleep(1000); \
75  reset(); \
76  perf_counter_t p = perf_alloc(PC_ELAPSED, name); \
77  for (int i = 0; i < count; i++) { \
78  lock(); \
79  perf_begin(p); \
80  op; \
81  perf_end(p); \
82  unlock(); \
83  reset(); \
84  } \
85  perf_print_counter(p); \
86  perf_free(p); \
87  } while (0)
