218 /*
219  The MPU9250 can only handle high bus speeds on the sensor and
220  interrupt status registers. All other registers have a maximum 1MHz
221  Communication with all registers of the device is performed using either
222  I2C at 400kHz or SPI at 1MHz. For applications requiring faster communications,
223  the sensor and interrupt registers may be read using SPI at 20MHz
224  */
225 #define MPU9250_LOW_BUS_SPEED 0
226 #define MPU9250_HIGH_BUS_SPEED 0x8000
227 # define MPU9250_IS_HIGH_SPEED(r) ((r) & MPU9250_HIGH_BUS_SPEED)
228 # define MPU9250_REG(r) ((r) &~MPU9250_HIGH_BUS_SPEED)
229 # define MPU9250_SET_SPEED(r, s) ((r)|(s))
230 # define MPU9250_HIGH_SPEED_OP(r) MPU9250_SET_SPEED((r), MPU9250_HIGH_BUS_SPEED)
231 # define MPU9250_LOW_SPEED_OP(r) MPU9250_REG((r))
