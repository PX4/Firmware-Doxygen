87 static int open_uart(const char *device)
88 {
89  /* baud rate */
90  int speed = B19200;
91 
92  /* open uart */
93  int uart = open(device, O_RDWR | O_NOCTTY);
94 
95  if (uart < 0) {
96  PX4_ERR("FAIL: Error opening port");
97  return ERROR;
98  }
99 
100  /* Try to set baud rate */
101  struct termios uart_config;
102 
103  /* Fill the struct for the new configuration */
104  tcgetattr(uart, &uart_config);
105 
106  /* Clear ONLCR flag (which appends a CR for every LF) */
107  uart_config.c_oflag &= ~ONLCR;
108 
109  /* Set baud rate */
110  if (cfsetispeed(&uart_config, speed) < 0 || cfsetospeed(&uart_config, speed) < 0) {
111  PX4_ERR("FAIL: Error setting baudrate / termios config for cfsetispeed, cfsetospeed");
112  return ERROR;
113  }
114 
115  if (tcsetattr(uart, TCSANOW, &uart_config) < 0) {
116  PX4_ERR("FAIL: Error setting baudrate / termios config for tcsetattr");
117  return ERROR;
118  }
119 
120  return uart;
121 }

131 int test_hott_telemetry(int argc, char *argv[])
132 {
133  PX4_INFO("HoTT Telemetry Test Requirements:");
134  PX4_INFO("- Radio on and Electric Air. Mod on (telemetry -> sensor select).");
135  PX4_INFO("- Receiver telemetry port must be in telemetry mode.");
136  PX4_INFO("- Connect telemetry wire to /dev/ttyS1 (USART2).");
137  PX4_INFO("Testing...");
138 
139  const char device[] = "/dev/ttyS1";
140  int fd = open_uart(device);
141 
142  if (fd < 0) {
143  close(fd);
144  return ERROR;
145  }
146 
147 #ifdef TIOCSSINGLEWIRE
148  /* Activate single wire mode */
149  ioctl(fd, TIOCSSINGLEWIRE, SER_SINGLEWIRE_ENABLED);
150 #endif
151 
152  char send = 'a';
153  write(fd, &send, 1);
154 
155  /* Since TX and RX are now connected we should be able to read in what we wrote */
156  const int timeout = 1000;
157  struct pollfd fds[] = { { .fd = fd, .events = POLLIN } };
158 
159  if (poll(fds, 1, timeout) == 0) {
160  PX4_ERR("FAIL: Could not read sent data.");
161  return 1;
162  }
163 
164  char receive;
165  read(fd, &receive, 1);
166  PX4_INFO("PASS: Single wire enabled. Sent %x and received %x", send, receive);
167 
168 
169  /* Attempt to read HoTT poll messages from the HoTT receiver */
170  int received_count = 0;
171  int valid_count = 0;
172  const int max_polls = 5;
173  uint8_t byte;
174 
175  for (; received_count < 5; received_count++) {
176  if (poll(fds, 1, timeout) == 0) {
177  PX4_ERR("FAIL: Could not read sent data. Is your HoTT receiver plugged in on %s?", device);
178  return 1;
179 
180  } else {
181  read(fd, &byte, 1);
182 
183  if (byte == 0x80) {
184  valid_count++;
185  }
186 
187  /* Read the device ID being polled */
188  read(fd, &byte, 1);
189  }
190  }
191 
192  if (received_count > 0 && valid_count > 0) {
193  if (received_count == max_polls && valid_count == max_polls) {
194  PX4_INFO("PASS: Received %d out of %d valid byte pairs from the HoTT receiver device.", received_count, max_polls);
195 
196  } else {
197  PX4_WARN("WARN: Received %d out of %d byte pairs of which %d were valid from the HoTT receiver device.", received_count,
198  max_polls, valid_count);
199  }
200 
201  } else {
202  /* Let's work out what went wrong */
203  if (received_count == 0) {
204  PX4_ERR("FAIL: Could not read any polls from HoTT receiver device.");
205  return 1;
206  }
207 
208  if (valid_count == 0) {
209  PX4_ERR("FAIL: Received unexpected values from the HoTT receiver device.");
210  return 1;
211  }
212  }
213 
214 
215  /* Attempt to send a HoTT response messages */
216  uint8_t response[] = {0x7c, 0x8e, 0x00, 0xe0, 0x00, 0x80, 0x00, 0x00, 0x00, 0x00, \
217  0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, \
218  0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0xf4, 0x01, 0x00, 0x00, \
219  0x19, 0x00, 0x00, 0x00, 0x30, 0x75, 0x78, 0x00, 0x00, 0x00, \
220  0x00, 0x00, 0x00, 0x7d, 0x12
221  };
222 
223  usleep(5000);
224 
225  for (unsigned int i = 0; i < sizeof(response); i++) {
226  write(fd, &response[i], 1);
227  usleep(1000);
228  }
229 
230  PX4_INFO("PASS: Response sent to the HoTT receiver device. Voltage should now show 2.5V.");
231 
232 
233 #ifdef TIOCSSINGLEWIRE
234  /* Disable single wire */
235  ioctl(fd, TIOCSSINGLEWIRE, ~SER_SINGLEWIRE_ENABLED);
236 #endif
237 
238  write(fd, &send, 1);
239 
240  /* We should timeout as there will be nothing to read (TX and RX no longer connected) */
241  if (poll(fds, 1, timeout) == 0) {
242  PX4_ERR("FAIL: timeout expected.");
243  return 1;
244  }
245 
246  PX4_INFO("PASS: Single wire disabled.");
247 
248  close(fd);
249  return 0;
250 }
