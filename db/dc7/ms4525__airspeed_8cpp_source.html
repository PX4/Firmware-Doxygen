141 int
142 MEASAirspeed::collect()
143 {
144  /* read from the sensor */
145  uint8_t val[4] = {0, 0, 0, 0};
146 
147  perf_begin(_sample_perf);
148 
149  int ret = transfer(nullptr, 0, &val[0], 4);
150 
151  if (ret < 0) {
152  perf_count(_comms_errors);
153  perf_end(_sample_perf);
154  return ret;
155  }
156 
157  uint8_t status = (val[0] & 0xC0) >> 6;
158 
159  switch (status) {
160  case 0:
161  // Normal Operation. Good Data Packet
162  break;
163 
164  case 1:
165  // Reserved
166  return -EAGAIN;
167 
168  case 2:
169  // Stale Data. Data has been fetched since last measurement cycle.
170  return -EAGAIN;
171 
172  case 3:
173  // Fault Detected
174  perf_count(_comms_errors);
175  perf_end(_sample_perf);
176  return -EAGAIN;
177  }
178 
179  int16_t dp_raw = 0, dT_raw = 0;
180  dp_raw = (val[0] << 8) + val[1];
181  /* mask the used bits */
182  dp_raw = 0x3FFF & dp_raw;
183  dT_raw = (val[2] << 8) + val[3];
184  dT_raw = (0xFFE0 & dT_raw) >> 5;
185 
186  // dT max is almost certainly an invalid reading
187  if (dT_raw == 2047) {
188  perf_count(_comms_errors);
189  return -EAGAIN;
190  }
191 
192  float temperature = ((200.0f * dT_raw) / 2047) - 50;
193 
194  // Calculate differential pressure. As its centered around 8000
195  // and can go positive or negative
196  const float P_min = -1.0f;
197  const float P_max = 1.0f;
198  const float PSI_to_Pa = 6894.757f;
199  /*
200  this equation is an inversion of the equation in the
201  pressure transfer function figure on page 4 of the datasheet
202 
203  We negate the result so that positive differential pressures
204  are generated when the bottom port is used as the static
205  port on the pitot and top port is used as the dynamic port
206  */
207  float diff_press_PSI = -((dp_raw - 0.1f * 16383) * (P_max - P_min) / (0.8f * 16383) + P_min);
208  float diff_press_pa_raw = diff_press_PSI * PSI_to_Pa;
209 
210  // correct for 5V rail voltage if possible
211  voltage_correction(diff_press_pa_raw, temperature);
212 
213  /*
214  With the above calculation the MS4525 sensor will produce a
215  positive number when the top port is used as a dynamic port
216  and bottom port is used as the static port
217  */
218 
219  struct differential_pressure_s report;
220 
221  report.timestamp = hrt_absolute_time();
222  report.error_count = perf_event_count(_comms_errors);
223  report.temperature = temperature;
224  report.differential_pressure_filtered_pa = _filter.apply(diff_press_pa_raw) - _diff_pres_offset;
225  report.differential_pressure_raw_pa = diff_press_pa_raw - _diff_pres_offset;
226  report.device_id = _device_id.devid;
227 
228  if (_airspeed_pub != nullptr && !(_pub_blocked)) {
229  /* publish it */
230  orb_publish(ORB_ID(differential_pressure), _airspeed_pub, &report);
231  }
232 
233  ret = OK;
234 
235  perf_end(_sample_perf);
236 
237  return ret;
238 }

240 void
241 MEASAirspeed::cycle()
242 {
243  int ret;
244 
245  /* collection phase? */
246  if (_collect_phase) {
247 
248  /* perform collection */
249  ret = collect();
250 
251  if (OK != ret) {
252  /* restart the measurement state machine */
253  start();
254  _sensor_ok = false;
255  return;
256  }
257 
258  /* next phase is measurement */
259  _collect_phase = false;
260 
261  /*
262  * Is there a collect->measure gap?
263  */
264  if (_measure_ticks > USEC2TICK(CONVERSION_INTERVAL)) {
265 
266  /* schedule a fresh cycle call when we are ready to measure again */
267  work_queue(HPWORK,
268  &_work,
269  (worker_t)&Airspeed::cycle_trampoline,
270  this,
271  _measure_ticks - USEC2TICK(CONVERSION_INTERVAL));
272 
273  return;
274  }
275  }
276 
277  /* measurement phase */
278  ret = measure();
279 
280  if (OK != ret) {
281  DEVICE_DEBUG("measure error");
282  }
283 
284  _sensor_ok = (ret == OK);
285 
286  /* next phase is collection */
287  _collect_phase = true;
288 
289  /* schedule a fresh cycle call when the measurement is done */
290  work_queue(HPWORK,
291  &_work,
292  (worker_t)&Airspeed::cycle_trampoline,
293  this,
294  USEC2TICK(CONVERSION_INTERVAL));
295 }

297 /**
298  correct for 5V rail voltage if the system_power ORB topic is
299  available
300 
301  See http://uav.tridgell.net/MS4525/MS4525-offset.png for a graph of
302  offset versus voltage for 3 sensors
303  */
304 void
305 MEASAirspeed::voltage_correction(float &diff_press_pa, float &temperature)
306 {
307 #if defined(ADC_SCALED_V5_SENSE)
308 
309  if (_t_system_power == -1) {
310  _t_system_power = orb_subscribe(ORB_ID(system_power));
311  }
312 
313  if (_t_system_power == -1) {
314  // not available
315  return;
316  }
317 
318  bool updated = false;
319  orb_check(_t_system_power, &updated);
320 
321  if (updated) {
322  orb_copy(ORB_ID(system_power), _t_system_power, &system_power);
323  }
324 
325  if (system_power.voltage5v_v < 3.0f || system_power.voltage5v_v > 6.0f) {
326  // not valid, skip correction
327  return;
328  }
329 
330  const float slope = 65.0f;
331  /*
332  apply a piecewise linear correction, flattening at 0.5V from 5V
333  */
334  float voltage_diff = system_power.voltage5v_v - 5.0f;
335 
336  if (voltage_diff > 0.5f) {
337  voltage_diff = 0.5f;
338  }
339 
340  if (voltage_diff < -0.5f) {
341  voltage_diff = -0.5f;
342  }
343 
344  diff_press_pa -= voltage_diff * slope;
345 
346  /*
347  the temperature masurement varies as well
348  */
349  const float temp_slope = 0.887f;
350  voltage_diff = system_power.voltage5v_v - 5.0f;
351 
352  if (voltage_diff > 0.5f) {
353  voltage_diff = 0.5f;
354  }
355 
356  if (voltage_diff < -1.0f) {
357  voltage_diff = -1.0f;
358  }
359 
360  temperature -= voltage_diff * temp_slope;
361 #endif // defined(ADC_SCALED_V5_SENSE)
362 }

398 /**
399  * Start the driver on a specific bus.
400  *
401  * This function call only returns once the driver is up and running
402  * or failed to detect the sensor.
403  */
404 int
405 start_bus(int i2c_bus, int address)
406 {
407  int fd;
408 
409  if (g_dev != nullptr) {
410  PX4_ERR("already started");
411  return PX4_ERROR;
412  }
413 
414  /* create the driver, try the MS4525DO first */
415  g_dev = new MEASAirspeed(i2c_bus, address, PATH_MS4525);
416 
417  /* check if the MS4525DO was instantiated */
418  if (g_dev == nullptr) {
419  goto fail;
420  }
421 
422  if (OK != g_dev->init()) {
423  goto fail;
424  }
425 
426  /* set the poll rate to default, starts automatic data collection */
427  fd = px4_open(PATH_MS4525, O_RDONLY);
428 
429  if (fd < 0) {
430  goto fail;
431  }
432 
433  if (px4_ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
434  goto fail;
435  }
436 
437  return PX4_OK;
438 
439 fail:
440 
441  if (g_dev != nullptr) {
442  delete g_dev;
443  g_dev = nullptr;
444  }
445 
446  return PX4_ERROR;
447 }

507 int
508 ms4525_airspeed_main(int argc, char *argv[])
509 {
510  int i2c_bus = PX4_I2C_BUS_DEFAULT;
511 
512  int myoptind = 1;
513  int ch;
514  const char *myoptarg = nullptr;
515 
516  int device_type = DEVICE_TYPE_MS4525;
517 
518  bool start_all = false;
519 
520  while ((ch = px4_getopt(argc, argv, "ab:T:", &myoptind, &myoptarg)) != EOF) {
521  switch (ch) {
522  case 'b':
523  i2c_bus = atoi(myoptarg);
524  break;
525 
526  case 'a':
527  start_all = true;
528  break;
529 
530  case 'T':
531  device_type = atoi(myoptarg);
532  break;
533 
534  default:
535  meas_airspeed_usage();
536  return 0;
537  }
538  }
539 
540  if (myoptind >= argc) {
541  meas_airspeed_usage();
542  return -1;
543  }
544 
545  /*
546  * Start/load the driver.
547  */
548  if (!strcmp(argv[myoptind], "start")) {
549  if (start_all) {
550  return meas_airspeed::start();
551 
552  } else if (device_type == DEVICE_TYPE_MS4515) {
553  return meas_airspeed::start_bus(i2c_bus, I2C_ADDRESS_MS4515DO);
554 
555  } else if (device_type == DEVICE_TYPE_MS4525) {
556  return meas_airspeed::start_bus(i2c_bus, I2C_ADDRESS_MS4525DO);
557  }
558  }
559 
560  /*
561  * Stop the driver
562  */
563  if (!strcmp(argv[myoptind], "stop")) {
564  return meas_airspeed::stop();
565  }
566 
567  /*
568  * Reset the driver.
569  */
570  if (!strcmp(argv[myoptind], "reset")) {
571  return meas_airspeed::reset();
572  }
573 
574  meas_airspeed_usage();
575  return 0;
576 }
