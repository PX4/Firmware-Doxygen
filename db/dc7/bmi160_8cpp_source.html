932 void
933 BMI160::check_registers(void)
934 {
935  uint8_t v;
936 
937  if ((v = read_reg(_checked_registers[_checked_next])) !=
938  _checked_values[_checked_next]) {
939  _checked_bad[_checked_next] = v;
940 
941  /*
942  if we get the wrong value then we know the SPI bus
943  or sensor is very sick. We set _register_wait to 20
944  and wait until we have seen 20 good values in a row
945  before we consider the sensor to be OK again.
946  */
947  perf_count(_bad_registers);
948 
949  /*
950  try to fix the bad register value. We only try to
951  fix one per loop to prevent a bad sensor hogging the
952  bus.
953  */
954  if (_register_wait == 0 || _checked_next == 0) {
955  // if the product_id is wrong then reset the
956  // sensor completely
957  write_reg(BMIREG_CMD, BMI160_SOFT_RESET);
958  _reset_wait = hrt_absolute_time() + 10000;
959  _checked_next = 0;
960 
961  } else {
962  write_reg(_checked_registers[_checked_next], _checked_values[_checked_next]);
963  // waiting 3ms between register writes seems
964  // to raise the chance of the sensor
965  // recovering considerably
966  _reset_wait = hrt_absolute_time() + 3000;
967  }
968 
969  _register_wait = 20;
970  }
971 
972  _checked_next = (_checked_next + 1) % BMI160_NUM_CHECKED_REGISTERS;
973 }

1015  return;

1019  _last_accel[1] = bmi_report.accel_y;

1024  uint8_t temp_h = read_reg(BMIREG_TEMP_1);

1033  report.gyro_y = bmi_report.gyro_y;

1037  report.accel_y == 0 &&

1040  report.gyro_x == 0 &&
1041  report.gyro_y == 0 &&
1042  report.gyro_z == 0) {
