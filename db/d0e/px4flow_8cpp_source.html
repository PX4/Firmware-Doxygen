313 PX4FLOW::probe()
314 {
315  uint8_t val[I2C_FRAME_SIZE];
316 
317  // to be sure this is not a ll40ls Lidar (which can also be on
318  // 0x42) we check if a I2C_FRAME_SIZE byte transfer works from address

320  // happily returns some data

322  return -EIO;
323  }
324 
325  // that worked, so start a measurement cycle
326  return measure();

329 int
330 PX4FLOW::ioctl(struct file *filp, int cmd, unsigned long arg)
331 {
332  switch (cmd) {
333 
334  case SENSORIOCSPOLLRATE: {
335  switch (arg) {
336 
337  /* switching to manual polling */
338  case SENSOR_POLLRATE_MANUAL:
339  stop();
340  _measure_ticks = 0;
341  return OK;
342 
343  /* external signalling (DRDY) not supported */
344  case SENSOR_POLLRATE_EXTERNAL:
345 
346  /* zero would be bad */
347  case 0:
348  return -EINVAL;
349 
350  /* set default/max polling rate */
351  case SENSOR_POLLRATE_MAX:
352  case SENSOR_POLLRATE_DEFAULT: {
353  /* do we need to start internal polling? */
354  bool want_start = (_measure_ticks == 0);
355 
356  /* set interval for next measurement to minimum legal value */
357  _measure_ticks = USEC2TICK(_conversion_interval);
358 
359  /* if we need to start the poll state machine, do it */
360  if (want_start) {
361  start();
362  }
363 
364  return OK;
365  }
366 
367  /* adjust to a legal polling interval in Hz */
368  default: {
369  /* do we need to start internal polling? */
370  bool want_start = (_measure_ticks == 0);
371 
372  /* convert hz to tick interval via microseconds */
373  unsigned ticks = USEC2TICK(1000000 / arg);
374 
375  /* check against maximum rate */
376  if (ticks < USEC2TICK(_conversion_interval)) {
377  return -EINVAL;
378  }
379 
380  /* update interval for next measurement */
381  _measure_ticks = ticks;
382 
383  /* if we need to start the poll state machine, do it */
384  if (want_start) {
385  start();
386  }
387 
388  return OK;
389  }
390  }
391  }
392 
393  case SENSORIOCGPOLLRATE:
394  if (_measure_ticks == 0) {
395  return SENSOR_POLLRATE_MANUAL;
396  }
397 
398  return (1000 / _measure_ticks);
399 
400  case SENSORIOCSQUEUEDEPTH: {
401  /* lower bound is mandatory, upper bound is a sanity check */
402  if ((arg < 1) || (arg > 100)) {
403  return -EINVAL;
404  }
405 
406  irqstate_t flags = px4_enter_critical_section();
407 
408  if (!_reports->resize(arg)) {
409  px4_leave_critical_section(flags);
410  return -ENOMEM;
411  }
412 
413  px4_leave_critical_section(flags);
414 
415  return OK;
416  }
417 
418  case SENSORIOCRESET:
419  /* XXX implement this */
420  return -EINVAL;
421 
422  default:
423  /* give it to the superclass */
424  return I2C::ioctl(filp, cmd, arg);
425  }
426 }

428 ssize_t
429 PX4FLOW::read(struct file *filp, char *buffer, size_t buflen)
430 {
431  unsigned count = buflen / sizeof(struct optical_flow_s);
432  struct optical_flow_s *rbuf = reinterpret_cast<struct optical_flow_s *>(buffer);
433  int ret = 0;
434 
435  /* buffer must be large enough */
436  if (count < 1) {
437  return -ENOSPC;
438  }
439 
440  /* if automatic measurement is enabled */
441  if (_measure_ticks > 0) {
442 
443  /*
444  * While there is space in the caller's buffer, and reports, copy them.
445  * Note that we may be pre-empted by the workq thread while we are doing this;
446  * we are careful to avoid racing with them.
447  */
448  while (count--) {
449  if (_reports->get(rbuf)) {
450  ret += sizeof(*rbuf);
451  rbuf++;
452  }
453  }
454 
455  /* if there was no data, warn the caller */
456  return ret ? ret : -EAGAIN;
457  }
458 
459  /* manual measurement - run one conversion */
460  do {
461  _reports->flush();
462 
463  /* trigger a measurement */
464  if (OK != measure()) {
465  ret = -EIO;
466  break;
467  }
468 
469  /* run the collection phase */
470  if (OK != collect()) {
471  ret = -EIO;
472  break;
473  }
474 
475  /* state machine will have generated a report, copy it out */
476  if (_reports->get(rbuf)) {
477  ret = sizeof(*rbuf);
478  }
479 
480  } while (0);
481 
482  return ret;
483 }

507 int
508 PX4FLOW::collect()
509 {
510  int ret = -EIO;
511 
512  /* read from the sensor */
513  uint8_t val[I2C_FRAME_SIZE + I2C_INTEGRAL_FRAME_SIZE] = { };
514 
515  perf_begin(_sample_perf);
516 
517  if (PX4FLOW_REG == 0x00) {
518  ret = transfer(nullptr, 0, &val[0], I2C_FRAME_SIZE + I2C_INTEGRAL_FRAME_SIZE);
519  }
520 
521  if (PX4FLOW_REG == 0x16) {
522  ret = transfer(nullptr, 0, &val[0], I2C_INTEGRAL_FRAME_SIZE);
523  }
524 
525  if (ret < 0) {
526  DEVICE_DEBUG("error reading from sensor: %d", ret);
527  perf_count(_comms_errors);
528  perf_end(_sample_perf);
529  return ret;
530  }
531 
532  if (PX4FLOW_REG == 0) {
533  memcpy(&_frame, val, I2C_FRAME_SIZE);
534  memcpy(&_frame_integral, &(val[I2C_FRAME_SIZE]), I2C_INTEGRAL_FRAME_SIZE);
535  }
536 
537  if (PX4FLOW_REG == 0x16) {
538  memcpy(&_frame_integral, val, I2C_INTEGRAL_FRAME_SIZE);
539  }
540 
541 
542  struct optical_flow_s report;
543 
544  report.timestamp = hrt_absolute_time();
545 
546  report.pixel_flow_x_integral = static_cast<float>(_frame_integral.pixel_flow_x_integral) / 10000.0f;//convert to radians
547 
548  report.pixel_flow_y_integral = static_cast<float>(_frame_integral.pixel_flow_y_integral) / 10000.0f;//convert to radians
549 
550  report.frame_count_since_last_readout = _frame_integral.frame_count_since_last_readout;
551 
552  report.ground_distance_m = static_cast<float>(_frame_integral.ground_distance) / 1000.0f;//convert to meters
553 
554  report.quality = _frame_integral.qual; //0:bad ; 255 max quality
555 
556  report.gyro_x_rate_integral = static_cast<float>(_frame_integral.gyro_x_rate_integral) / 10000.0f; //convert to radians
557 
558  report.gyro_y_rate_integral = static_cast<float>(_frame_integral.gyro_y_rate_integral) / 10000.0f; //convert to radians
559 
560  report.gyro_z_rate_integral = static_cast<float>(_frame_integral.gyro_z_rate_integral) / 10000.0f; //convert to radians
561 
562  report.integration_timespan = _frame_integral.integration_timespan; //microseconds
563 
564  report.time_since_last_sonar_update = _frame_integral.sonar_timestamp;//microseconds
565 
566  report.gyro_temperature = _frame_integral.gyro_temperature;//Temperature * 100 in centi-degrees Celsius
567 
568  report.sensor_id = 0;
569 
570  report.max_flow_rate = _sensor_max_flow_rate;
571 
572  report.min_ground_distance = _sensor_min_range;
573 
574  report.max_ground_distance = _sensor_max_range;
575 
576  /* rotate measurements in yaw from sensor frame to body frame according to parameter SENS_FLOW_ROT */
577  float zeroval = 0.0f;
578 
579  rotate_3f(_sensor_rotation, report.pixel_flow_x_integral, report.pixel_flow_y_integral, zeroval);
580 
581  rotate_3f(_sensor_rotation, report.gyro_x_rate_integral, report.gyro_y_rate_integral, report.gyro_z_rate_integral);
582 
583  if (_px4flow_topic == nullptr) {
584  _px4flow_topic = orb_advertise(ORB_ID(optical_flow), &report);
585 
586  } else {
587  /* publish it */
588  orb_publish(ORB_ID(optical_flow), _px4flow_topic, &report);
589  }
590 
591  /* publish to the distance_sensor topic as well */
592  struct distance_sensor_s distance_report;
593  distance_report.timestamp = report.timestamp;
594  distance_report.min_distance = PX4FLOW_MIN_DISTANCE;
595  distance_report.max_distance = PX4FLOW_MAX_DISTANCE;
596  distance_report.current_distance = report.ground_distance_m;
597  distance_report.covariance = 0.0f;
598  distance_report.signal_quality = -1;
599  distance_report.type = distance_sensor_s::MAV_DISTANCE_SENSOR_ULTRASOUND;
600  /* TODO: the ID needs to be properly set */
601  distance_report.id = 0;
602  distance_report.orientation = _sonar_rotation;
603 
604  orb_publish(ORB_ID(distance_sensor), _distance_sensor_topic, &distance_report);
605 
606  /* post a report to the ring */
607  _reports->force(&report);
608 
609  /* notify anyone waiting for data */
610  poll_notify(POLLIN);
611 
612  ret = OK;
613 
614  perf_end(_sample_perf);
615  return ret;

691 /**
692  * Start the driver.
693  */
694 int
695 start(int argc, char *argv[])
696 {
697  int fd;
698 
699  /* entry check: */
700  if (start_in_progress) {
701  PX4_WARN("start already in progress");
702  return 1;
703  }
704 
705  if (g_dev != nullptr) {
706  start_in_progress = false;
707  PX4_WARN("already started");
708  return 1;
709  }
710 
711  /* parse command line options */
712  int address = I2C_FLOW_ADDRESS_DEFAULT;
713  int conversion_interval = PX4FLOW_CONVERSION_INTERVAL_DEFAULT;
714 
715  /* don't exit from getopt loop to leave getopt global variables in consistent state,
716  * set error flag instead */
717  bool err_flag = false;
718  int ch;
719  int myoptind = 1;
720  const char *myoptarg = nullptr;
721  uint8_t sonar_rotation = distance_sensor_s::ROTATION_DOWNWARD_FACING;
722 
723  while ((ch = px4_getopt(argc, argv, "a:i:R:", &myoptind, &myoptarg)) != EOF) {
724  switch (ch) {
725  case 'a':
726  address = strtoul(myoptarg, nullptr, 16);
727
