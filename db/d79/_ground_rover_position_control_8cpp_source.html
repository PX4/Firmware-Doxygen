207 bool
208 GroundRoverPositionControl::control_position(const matrix::Vector2f &current_position,
209  const matrix::Vector3f &ground_speed, const position_setpoint_triplet_s &pos_sp_triplet)
210 {
211  float dt = 0.01; // Using non zero value to a avoid division by zero
212 
213  if (_control_position_last_called > 0) {
214  dt = hrt_elapsed_time(&_control_position_last_called) * 1e-6f;
215  }
216 
217  _control_position_last_called = hrt_absolute_time();
218 
219  bool setpoint = true;
220 
221  if (_control_mode.flag_control_auto_enabled && pos_sp_triplet.current.valid) {
222  /* AUTONOMOUS FLIGHT */
223 
224  _control_mode_current = UGV_POSCTRL_MODE_AUTO;
225 
226  /* get circle mode */
227  bool was_circle_mode = _gnd_control.circle_mode();
228 
229  /* current waypoint (the one currently heading for) */
230  matrix::Vector2f curr_wp((float)pos_sp_triplet.current.lat, (float)pos_sp_triplet.current.lon);
231 
232  /* previous waypoint */
233  matrix::Vector2f prev_wp = curr_wp;
234 
235  if (pos_sp_triplet.previous.valid) {
236  prev_wp(0) = (float)pos_sp_triplet.previous.lat;
237  prev_wp(1) = (float)pos_sp_triplet.previous.lon;
238  }
239 
240  matrix::Vector2f ground_speed_2d = {ground_speed(0), ground_speed(1)};
241 
242  float mission_throttle = _parameters.throttle_cruise;
243 
244  /* Just control the throttle */
245  if (_parameters.speed_control_mode == 1) {
246  /* control the speed in closed loop */
247 
248  float mission_target_speed = _parameters.gndspeed_trim;
249 
250  if (PX4_ISFINITE(_pos_sp_triplet.current.cruising_speed) &&
251  _pos_sp_triplet.current.cruising_speed > 0.1f) {
252  mission_target_speed = _pos_sp_triplet.current.cruising_speed;
253  }
254 
255  // Velocity in body frame
256  const Dcmf R_to_body(Quatf(_sub_attitude.get().q).inversed());
257  const Vector3f vel = R_to_body * Vector3f(ground_speed(0), ground_speed(1), ground_speed(2));
258 
259  const float x_vel = vel(0);
260  const float x_acc = _sub_sensors.get().accel_x;
261 
262  // Compute airspeed control out and just scale it as a constant
263  mission_throttle = _parameters.throttle_speed_scaler
264  * pid_calculate(&_speed_ctrl, mission_target_speed, x_vel, x_acc, dt);
265 
266  // Constrain throttle between min and max
267  mission_throttle = math::constrain(mission_throttle, _parameters.throttle_min, _parameters.throttle_max);
268 
269  } else {
270  /* Just control throttle in open loop */
271  if (PX4_ISFINITE(_pos_sp_triplet.current.cruising_throttle) &&
272  _pos_sp_triplet.current.cruising_throttle > 0.01f) {
273 
274  mission_throttle = _pos_sp_triplet.current.cruising_throttle;
275  }
276  }
277 
278  if (pos_sp_triplet.current.type == position_setpoint_s::SETPOINT_TYPE_IDLE) {
279  _att_sp.roll_body = 0.0f;
280  _att_sp.pitch_body = 0.0f;
281  _att_sp.yaw_body = 0.0f;
282  _att_sp.thrust = 0.0f;
283 
284  } else if ((pos_sp_triplet.current.type == position_setpoint_s::SETPOINT_TYPE_POSITION)
285  || (pos_sp_triplet.current.type == position_setpoint_s::SETPOINT_TYPE_TAKEOFF)) {
286 
287  /* waypoint is a plain navigation waypoint or the takeoff waypoint, does not matter */
288  _gnd_control.navigate_waypoints(prev_wp, curr_wp, current_position, ground_speed_2d);
289  _att_sp.roll_body = _gnd_control.get_roll_setpoint();
290  _att_sp.pitch_body = 0.0f;
291  _att_sp.yaw_body = _gnd_control.nav_bearing();
292  _att_sp.fw_control_yaw = true;
293  _att_sp.thrust = mission_throttle;
294 
295  } else if (pos_sp_triplet.current.type == position_setpoint_s::SETPOINT_TYPE_LOITER) {
296 
297  /* waypoint is a loiter waypoint so we want to stop*/
298  _gnd_control.navigate_loiter(curr_wp, current_position, pos_sp_triplet.current.loiter_radius,
299  pos_sp_triplet.current.loiter_direction, ground_speed_2d);
300 
301  _att_sp.roll_body = _gnd_control.get_roll_setpoint();
302  _att_sp.pitch_body = 0.0f;
303  _att_sp.yaw_body = _gnd_control.nav_bearing();
304  _att_sp.fw_control_yaw = true;
305  _att_sp.thrust = 0.0f;
306  }
307 
308  if (was_circle_mode && !_gnd_control.circle_mode()) {
309  /* just kicked out of loiter, reset integrals */
310  _att_sp.yaw_reset_integral = true;
311  }
312 
313  } else {
314  _control_mode_current = UGV_POSCTRL_MODE_OTHER;
315 
316  _att_sp.roll_body = 0.0f;
317  _att_sp.pitch_body = 0.0f;
318  _att_sp.yaw_body = 0.0f;
319  _att_sp.fw_control_yaw = true;
320  _att_sp.thrust = 0.0f;
321 
322  /* do not publish the setpoint */
323  setpoint = false;
324  }
325 
326  return setpoint;
327 }

329 void
330 GroundRoverPositionControl::task_main()
331 {
332  _control_mode_sub = orb_subscribe(ORB_ID(vehicle_control_mode));
333  _global_pos_sub = orb_subscribe(ORB_ID(vehicle_global_position));
334  _manual_control_sub = orb_subscribe(ORB_ID(manual_control_setpoint));
335  _params_sub = orb_subscribe(ORB_ID(parameter_update));
336  _pos_sp_triplet_sub = orb_subscribe(ORB_ID(position_setpoint_triplet));
337 
338  /* rate limit control mode updates to 5Hz */
339  orb_set_interval(_control_mode_sub, 200);
340 
341  /* rate limit position updates to 50 Hz */
342  orb_set_interval(_global_pos_sub, 20);
343 
344  /* abort on a nonzero return value from the parameter init */
345  if (parameters_update()) {
346  /* parameter setup went wrong, abort */
347  warnx("aborting startup due to errors.");
348  _task_should_exit = true;
349  }
350 
351  /* wakeup source(s) */
352  px4_pollfd_struct_t fds[2];
353 
354  /* Setup of loop */
355  fds[0].fd = _params_sub;
356  fds[0].events = POLLIN;
357  fds[1].fd = _global_pos_sub;
358  fds[1].events = POLLIN;
359 
360  _task_running = true;
361 
362  while (!_task_should_exit) {
363 
364  /* wait for up to 500ms for data */
365  int pret = px4_poll(&fds[0], (sizeof(fds) / sizeof(fds[0])), 100);
366 
367  /* timed out - periodic check for _task_should_exit, etc. */
368  if (pret == 0) {
369  continue;
370  }
371 
372  /* this is undesirable but not much we can do - might want to flag unhappy status */
373  if (pret < 0) {
374  warn("poll error %d, %d", pret, errno);
375  continue;
376  }
377 
378  /* check vehicle control mode for changes to publication state */
379  vehicle_control_mode_poll();
380 
381  /* only update parameters if they changed */
382  if (fds[0].revents & POLLIN) {
383  /* read from param to clear updated flag */
384  struct parameter_update_s update;
385  orb_copy(ORB_ID(parameter_update), _params_sub, &update);
386 
387  /* update parameters from storage */
388  parameters_update();
389  }
390 
391  /* only run controller if position changed */
392  if (fds[1].revents & POLLIN) {
393  perf_begin(_loop_perf);
394 
395  /* load local copies */
396  orb_copy(ORB_ID(vehicle_global_position), _global_pos_sub, &_global_pos);
397 
398  // handle estimator reset events. we only adjust setpoins for manual modes
399  if (_control_mode.flag_control_manual_enabled) {
400 
401  // adjust navigation waypoints in position control mode
402  if (_control_mode.flag_control_altitude_enabled && _control_mode.flag_control_velocity_enabled
403  && _global_pos.lat_lon_reset_counter != _pos_reset_counter) {
404  }
405  }
406 
407  // update the reset counters in any case
408  _pos_reset_counter = _global_pos.lat_lon_reset_counter;
409 
410  manual_control_setpoint_poll();
411  position_setpoint_triplet_poll();
412  _sub_attitude.update();
413  _sub_sensors.update();
414 
415  matrix::Vector3f ground_speed(_global_pos.vel_n, _global_pos.vel_e, _global_pos.vel_d);
416  matrix::Vector2f current_position((float)_global_pos.lat, (float)_global_pos.lon);
417 
418  /*
419  * Attempt to control position, on success (= sensors present and not in manual mode),
420  * publish setpoint.
421  */
422  if (control_position(current_position, ground_speed, _pos_sp_triplet)) {
423  _att_sp.timestamp = hrt_absolute_time();
424 
425  Quatf q(Eulerf(_att_sp.roll_body, _att_sp.pitch_body, _att_sp.yaw_body));
426  q.copyTo(_att_sp.q_d);
427  _att_sp.q_d_valid = true;
428 
429  if (!_control_mode.flag_control_offboard_enabled ||
430  _control_mode.flag_control_position_enabled ||
431  _control_mode.flag_control_velocity_enabled ||
432  _control_mode.flag_control_acceleration_enabled) {
433 
434  /* lazily publish the setpoint only once available */
435  if (_attitude_sp_pub != nullptr) {
436  /* publish the attitude setpoint */
437  orb_publish(ORB_ID(vehicle_attitude_setpoint), _attitude_sp_pub, &_att_sp);
438 
439  } else {
440  /* advertise and publish */
441  _attitude_sp_pub = orb_advertise(ORB_ID(vehicle_attitude_setpoint), &_att_sp);
442  }
443  }
444 
445  /* XXX check if radius makes sense here */
446  float turn_distance = _parameters.l1_distance; //_gnd_control.switch_distance(100.0f);
447 
448  /* lazily publish navigation capabilities */
449  if ((hrt_elapsed_time(&_gnd_pos_ctrl_status.timestamp) > 1000000)
450  || (fabsf(turn_distance - _gnd_pos_ctrl_status.turn_distance) > FLT_EPSILON
451  && turn_distance > 0)) {
452 
453  /* set new turn distance */
454  _gnd_pos_ctrl_status.turn_distance = turn_distance;
455 
456  _gnd_pos_ctrl_status.nav_roll = _gnd_control.get_roll_setpoint();
457  _gnd_pos_ctrl_status.nav_pitch = 0.0f;
458  _gnd_pos_ctrl_status.nav_bearing = _gnd_control.nav_bearing();
459 
460  _gnd_pos_ctrl_status.target_bearing = _gnd_control.target_bearing();
461  _gnd_pos_ctrl_status.xtrack_error = _gnd_control.crosstrack_error();
462 
463  matrix::Vector2f curr_wp((float)_pos_sp_triplet.current.lat, (float)_pos_sp_triplet.current.lon);
464  _gnd_pos_ctrl_status.wp_dist = get_distance_to_next_waypoint((double)current_position(0), (double)current_position(1),
465  (double)curr_wp(0), (double)curr_wp(1));
466 
467  gnd_pos_ctrl_status_publish();
468  }
469  }
470 
471  perf_end(_loop_perf);
472  }
473  }
474 
475  _task_running = false;
476 
477  warnx("exiting.\n");
478 
479  _control_task = -1;
480 }
