218 void
219 MEASAirspeedSim::cycle()
220 {
221  int ret;
222 
223  /* collection phase? */
224  if (_collect_phase) {
225 
226  /* perform collection */
227  ret = collect();
228 
229  if (OK != ret) {
230  /* restart the measurement state machine */
231  start();
232  _sensor_ok = false;
233  return;
234  }
235 
236  /* next phase is measurement */
237  _collect_phase = false;
238 
239  /*
240  * Is there a collect->measure gap?
241  */
242  if (_measure_ticks > USEC2TICK(CONVERSION_INTERVAL)) {
243 
244  /* schedule a fresh cycle call when we are ready to measure again */
245  work_queue(HPWORK,
246  &_work,
247  (worker_t)&AirspeedSim::cycle_trampoline,
248  this,
249  _measure_ticks - USEC2TICK(CONVERSION_INTERVAL));
250 
251  return;
252  }
253  }
254 
255  /* measurement phase */
256  ret = measure();
257 
258  if (OK != ret) {
259  DEVICE_DEBUG("measure error");
260  }
261 
262  _sensor_ok = (ret == OK);
263 
264  /* next phase is collection */
265  _collect_phase = true;
266 
267  /* schedule a fresh cycle call when the measurement is done */
268  work_queue(HPWORK,
269  &_work,
270  (worker_t)&AirspeedSim::cycle_trampoline,
271  this,
272  USEC2TICK(CONVERSION_INTERVAL));
273 }

289 /**
290  * Start the driver.
291  *
292  * This function call only returns once the driver is up and running
293  * or failed to detect the sensor.
294  */
295 int
296 start(int i2c_bus)
297 {
298  int fd;
299 
300  if (g_dev != nullptr) {
301  PX4_WARN("already started");
302  }
303 
304  /* create the driver, try the MS4525DO first */
305  g_dev = new MEASAirspeedSim(i2c_bus, I2C_ADDRESS_MS4525DO, PATH_MS4525);
306 
307  /* check if the MS4525DO was instantiated */
308  if (g_dev == nullptr) {
309  goto fail;
310  }
311 
312  /* try the MS5525DSO next if init fails */
313  if (OK != g_dev->AirspeedSim::init()) {
314  delete g_dev;
315  g_dev = new MEASAirspeedSim(i2c_bus, I2C_ADDRESS_MS5525DSO, PATH_MS5525);
316 
317  /* check if the MS5525DSO was instantiated */
318  if (g_dev == nullptr) {
319  goto fail;
320  }
321 
322  /* both versions failed if the init for the MS5525DSO fails, give up */
323  if (OK != g_dev->AirspeedSim::init()) {
324  goto fail;
325  }
326  }
327 
328  /* set the poll rate to default, starts automatic data collection */
329  fd = px4_open(PATH_MS4525, O_RDONLY);
330 
331  if (fd < 0) {
332  goto fail;
333  }
334 
335  if (px4_ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
336  goto fail;
337  }
338 
339  return 0;
340 
341 fail:
342 
343  if (g_dev != nullptr) {
344  delete g_dev;
345  g_dev = nullptr;
346  }
347 
348  PX4_ERR("no MS4525 airspeedSim sensor connected");
349  return 1;
350 }

369 /**
370  * Perform some basic functional tests on the driver;
371  * make sure we can collect data from the sensor in polled
372  * and automatic modes.
373  */
374 int
375 test()
376 {
377  struct differential_pressure_s report;
378  ssize_t sz;
379  int ret;
380 
381  int fd = px4_open(PATH_MS4525, O_RDONLY);
382 
383  if (fd < 0) {
384  PX4_ERR("%s open failed (try 'meas_airspeed_sim start' if the driver is not running", PATH_MS4525);
385  return 1;
386  }
387 
388  /* do a simple demand read */
389  sz = px4_read(fd, &report, sizeof(report));
390 
391  if (sz != sizeof(report)) {
392  PX4_ERR("immediate read failed");
393  return 1;
394  }
395 
396  print_message(report);
397 
398  /* start the sensor polling at 2Hz */
399  if (OK != px4_ioctl(fd, SENSORIOCSPOLLRATE, 2)) {
400  PX4_WARN("failed to set 2Hz poll rate");
401  }
402 
403  /* read the sensor 5x and report each value */
404  for (unsigned i = 0; i < 5; i++) {
405  struct pollfd fds;
406 
407  /* wait for data to be ready */
408  fds.fd = fd;
409  fds.events = POLLIN;
410  ret = poll(&fds, 1, 2000);
411 
412  if (ret != 1) {
413  warnx("timed out");
414  }
415 
416  /* now go get it */
417  sz = read(fd, &report, sizeof(report));
418 
419  if (sz != sizeof(report)) {
420  PX4_WARN("periodic read failed");
421  }
422 
423  print_message(report);
424  }
425 
426  /* reset the sensor polling to its default rate */
427  if (OK != ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT)) {
428  PX4_ERR("failed to set default rate");
429  return 1;
430  }
431 
432  PX4_WARN("PASS");
433 
434  return 0;
435 }
