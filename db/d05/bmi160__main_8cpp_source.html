51 #if defined(PX4_SPI_BUS_EXT) && defined(PX4_SPIDEV_EXT_BMI)
52  *g_dev_ptr = new BMI160(PX4_SPI_BUS_EXT, path_accel, path_gyro, PX4_SPIDEV_EXT_BMI, rotation);

55 #endif
56 
57  } else {

61  errx(0, "No Internal SPI CS");
62 #endif
63  }

70  goto fail;
