79 struct LoggerSubscription {
80  int fd[ORB_MULTI_MAX_INSTANCES]; ///< uorb subscription. The first fd is also used to store the interval if
81  /// not subscribed yet (-interval - 1)
82  uint16_t msg_ids[ORB_MULTI_MAX_INSTANCES];
83  const orb_metadata *metadata = nullptr;
84 
85  LoggerSubscription() {}
86 
87  LoggerSubscription(int fd_, const orb_metadata *metadata_) :
88  metadata(metadata_)
89  {
90  fd[0] = fd_;
91 
92  for (int i = 1; i < ORB_MULTI_MAX_INSTANCES; i++) {
93  fd[i] = -1;
94  }
95 
96  for (int i = 0; i < ORB_MULTI_MAX_INSTANCES; i++) {
97  msg_ids[i] = (uint16_t) - 1;
98  }
99  }
100 };
