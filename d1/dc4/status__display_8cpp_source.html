65 bool StatusDisplay::check_for_updates()
66 {

68 
69  if (_subscriber_handler.battery_status_updated()) {
70  orb_copy(ORB_ID(battery_status), _subscriber_handler.get_battery_status_sub(), &_battery_status);
71  got_updates = true;

73 
74  if (_subscriber_handler.cpuload_updated()) {
75  orb_copy(ORB_ID(cpuload), _subscriber_handler.get_cpuload_sub(), &_cpu_load);
76  got_updates = true;
77  }
78 
79  if (_subscriber_handler.vehicle_status_flags_updated()) {
80  orb_copy(ORB_ID(vehicle_status_flags), _subscriber_handler.get_vehicle_status_flags_sub(), &_vehicle_status_flags);
81  got_updates = true;
82  }
83 
84  if (_subscriber_handler.vehicle_status_updated()) {
85  orb_copy(ORB_ID(vehicle_status), _subscriber_handler.get_vehicle_status_sub(), &_vehicle_status);
86  got_updates = true;
87  }
88 
89  return got_updates;
90 }
