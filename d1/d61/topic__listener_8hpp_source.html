55 void listener(const orb_id_t &id, unsigned num_msgs, unsigned topic_instance, unsigned topic_interval)
56 {
57  if (orb_exists(id, topic_instance) != 0) {
58  PX4_INFO_RAW("never published\n");
59  return;
60  }
61 
62  int sub = orb_subscribe_multi(id, topic_instance);
63  orb_set_interval(sub, topic_interval);
64 
65  bool updated = false;
66  unsigned i = 0;
67  hrt_abstime start_time = hrt_absolute_time();
68 
69  while (i < num_msgs) {
70  orb_check(sub, &updated);
71 
72  if (i == 0) {
73  updated = true;
74 
75  } else {
76  usleep(500);
77  }
78 
79  if (updated) {
80  start_time = hrt_absolute_time();
81  i++;
82 
83  PX4_INFO_RAW("\nTOPIC: %s instance %d #%d\n", id->o_name, topic_instance, i);
84 
85  T container;
86 
87  if (orb_copy(id, sub, &container) == PX4_OK) {
88  print_message(container);
89 
90  } else {
91  PX4_ERR("orb_copy failed");
92  }
93 
94  } else {
95  if (hrt_elapsed_time(&start_time) > 2 * 1000 * 1000) {
96  PX4_INFO_RAW("Waited for 2 seconds without a message. Giving up.\n");
97  break;
98  }
99  }
100  }
101 
102  orb_unsubscribe(sub);
103 }
