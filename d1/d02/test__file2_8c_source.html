62 static uint8_t get_value(uint32_t ofs)
63 {
64  union {
65  uint32_t ofs;
66  uint8_t buf[4];
67  } u;
68  u.ofs = ofs;
69  return u.buf[ofs % 4];
70 }

72 static void test_corruption(const char *filename, uint32_t write_chunk, uint32_t write_size, uint16_t flags)
73 {
74  printf("Testing on %s with write_chunk=%u write_size=%u\n",
75  filename, (unsigned)write_chunk, (unsigned)write_size);
76 
77  uint32_t ofs = 0;
78  int fd = open(filename, O_CREAT | O_RDWR | O_TRUNC, PX4_O_MODE_666);
79 
80  if (fd == -1) {
81  perror(filename);
82  exit(1);

85  // create a file of size write_size, in write_chunk blocks
86  uint8_t counter = 0;
87 
88  while (ofs < write_size) {
89  uint8_t buffer[write_chunk];
90 
91  for (uint16_t j = 0; j < write_chunk; j++) {
92  buffer[j] = get_value(ofs);
93  ofs++;
94  }
95 
96  if (write(fd, buffer, sizeof(buffer)) != (int)sizeof(buffer)) {
97  printf("write failed at offset %u\n", ofs);
98  exit(1);
99  }
100 
101  if (flags & FLAG_FSYNC) {
102  fsync(fd);
103  }
104 
105  if (counter % 100 == 0) {
106  printf("write ofs=%u\r", ofs);
107  }
108 
109  counter++;
110  }
111 
112  close(fd);
113 
114  printf("write ofs=%u\n", ofs);
115 
116  // read and check
117  fd = open(filename, O_RDONLY);
118 
119  if (fd == -1) {
120  perror(filename);
121  exit(1);
122  }
123 
124  counter = 0;
125  ofs = 0;
126 
127  while (ofs < write_size) {
128  uint8_t buffer[write_chunk];
129 
130  if (counter % 100 == 0) {
131  printf("read ofs=%u\r", ofs);
132  }
133 
134  counter++;
135 
136  if (read(fd, buffer, sizeof(buffer)) != (int)sizeof(buffer)) {
137  printf("read failed at offset %u\n", ofs);
138  close(fd);
139  return;
140  }
141 
142  for (uint16_t j = 0; j < write_chunk; j++) {
143  if (buffer[j] != get_value(ofs)) {
144  printf("corruption at ofs=%u got %u\n", ofs, buffer[j]);
145  close(fd);
146  return;
147  }
148 
149  ofs++;

155  }
