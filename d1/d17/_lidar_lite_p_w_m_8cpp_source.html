88 int LidarLitePWM::init()
89 {
90  /* do regular cdev init */
91  int ret = CDev::init();
92 
93  if (ret != PX4_OK) {
94  return PX4_ERROR;
95  }
96 
97  /* allocate basic report buffers */
98  _reports = new ringbuffer::RingBuffer(2, sizeof(struct distance_sensor_s));
99 
100  if (_reports == nullptr) {
101  return PX4_ERROR;
102  }
103 
104  _class_instance = register_class_devname(RANGE_FINDER_BASE_DEVICE_PATH);
105 
106  /* get a publish handle on the distance_sensor topic */
107  struct distance_sensor_s ds_report = {};
108  measure();
109  _reports->get(&ds_report);
110  _distance_sensor_topic = orb_advertise_multi(ORB_ID(distance_sensor), &ds_report,
111  &_orb_class_instance, ORB_PRIO_LOW);
112 
113  if (_distance_sensor_topic == nullptr) {
114  DEVICE_DEBUG("failed to create distance_sensor object. Did you start uOrb?");
115  }
116 
117  return PX4_OK;
118 }

165 int LidarLitePWM::measure()
166 {
167  perf_begin(_sample_perf);
168 
169  if (PX4_OK != collect()) {
170  DEVICE_DEBUG("collection error");
171  perf_count(_read_errors);
172  perf_end(_sample_perf);
173  return PX4_ERROR;
174  }
175 
176  _range.timestamp = hrt_absolute_time();
177  _range.type = distance_sensor_s::MAV_DISTANCE_SENSOR_LASER;
178  _range.max_distance = get_maximum_distance();
179  _range.min_distance = get_minimum_distance();
180  _range.current_distance = float(_pwm.pulse_width) * 1e-3f; /* 10 usec = 1 cm distance for LIDAR-Lite */
181  _range.covariance = 0.0f;
182  _range.orientation = _rotation;
183  /* TODO: set proper ID */
184  _range.id = 0;
185 
186  /* Due to a bug in older versions of the LidarLite firmware, we have to reset sensor on (distance == 0) */
187  if (_range.current_distance <= 0.0f) {
188  perf_count(_sensor_zero_resets);
189  perf_end(_sample_perf);
190  return reset_sensor();
191  }
192 
193  if (_distance_sensor_topic != nullptr) {
194  orb_publish(ORB_ID(distance_sensor), _distance_sensor_topic, &_range);
195  }
196 
197  _reports->force(&_range);
198 
199  poll_notify(POLLIN);
200  perf_end(_sample_perf);
201  return PX4_OK;
202 }

204 ssize_t LidarLitePWM::read(device::file_t *filp, char *buffer, size_t buflen)
205 {
206  unsigned count = buflen / sizeof(struct distance_sensor_s);
207  struct distance_sensor_s *rbuf = reinterpret_cast<struct distance_sensor_s *>(buffer);
208  int ret = 0;
209 
210  /* buffer must be large enough */
211  if (count < 1) {
212  return -ENOSPC;
213  }
214 
215  /* if automatic measurement is enabled */
216  if (getMeasureTicks() > 0) {
217  /*
218  * While there is space in the caller's buffer, and reports, copy them.
219  * Note that we may be pre-empted by the workq thread while we are doing this;
220  * we are careful to avoid racing with them.
221  */
222  while (count--) {
223  if (_reports->get(rbuf)) {
224  ret += sizeof(*rbuf);
225  rbuf++;
226  }
227  }
228 
229  /* if there was no data, warn the caller */
230  return ret ? ret : -EAGAIN;
231 
232  } else {
233 
234  _reports->flush();
235  measure();
236 
237  if (_reports->get(rbuf)) {
238  ret = sizeof(*rbuf);
239  }
240  }
241 
242  return ret;
243 }
