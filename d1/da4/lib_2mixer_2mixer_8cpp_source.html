144 const char *
145 Mixer::skipline(const char *buf, unsigned &buflen)
146 {
147  const char *p;
148 
149  /* if we can find a CR or NL in the buffer, skip up to it */
150  if ((p = (const char *)memchr(buf, '\r', buflen)) || (p = (const char *)memchr(buf, '\n', buflen))) {
151  /* skip up to it AND one beyond - could be on the NUL symbol now */
152  buflen -= (p - buf) + 1;
153  return p + 1;
154  }
155 
156  return nullptr;

159 bool
160 Mixer::string_well_formed(const char *buf, unsigned &buflen)
161 {
162  /* enforce that the mixer ends with a new line */
163  for (int i = buflen - 1; i >= 0; i--) {
164  if (buf[i] == '\0') {
165  continue;
166  }
167 
168  /* require a space or newline at the end of the buffer, fail on printable chars */
169  if (buf[i] == '\n' || buf[i] == '\r') {
170  /* found a line ending, so no split symbols / numbers. good. */
171  return true;
172  }
173 
174  }
175 
176  debug("pre-parser rejected: No newline in buf");
177 
178  return false;
