66 /**
67  * Start the driver.
68  *
69  * This function only returns if the driver is up and running
70  * or failed to detect the sensor.
71  */
72 void
73 start(bool external_bus, enum Rotation rotation, enum sensor_type sensor)
74 {
75 
76  int fd_acc, fd_gyr;
77  BMI055_accel **g_dev_acc_ptr = external_bus ? &g_acc_dev_ext : &g_acc_dev_int;
78  const char *path_accel = external_bus ? BMI055_DEVICE_PATH_ACCEL_EXT : BMI055_DEVICE_PATH_ACCEL;
79  BMI055_gyro **g_dev_gyr_ptr = external_bus ? &g_gyr_dev_ext : &g_gyr_dev_int;
80  const char *path_gyro = external_bus ? BMI055_DEVICE_PATH_GYRO_EXT : BMI055_DEVICE_PATH_GYRO;
81 
82  if (sensor == BMI055_ACCEL) {
83  if (*g_dev_acc_ptr != nullptr)
84  /* if already started, the still command succeeded */
85  {
86  errx(0, "bmi055 accel sensor already started");
87  }
88 
89  /* create the driver */
90  if (external_bus) {
91 #if defined(PX4_SPI_BUS_EXT) && defined(PX4_SPIDEV_EXT_BMI)
92  *g_dev_acc_ptr = new BMI055_accel(PX4_SPI_BUS_EXT, path_accel, PX4_SPIDEV_EXT_BMI, rotation);
93 #else
94  errx(0, "External SPI not available");
95 #endif
96 
97  } else {
98  *g_dev_acc_ptr = new BMI055_accel(PX4_SPI_BUS_SENSORS, path_accel, PX4_SPIDEV_BMI055_ACC, rotation);
99  }
100 
101  if (*g_dev_acc_ptr == nullptr) {
102  goto fail_accel;
103  }
104 
105  if (OK != (*g_dev_acc_ptr)->init()) {
106  goto fail_accel;
107  }
108 
109  /* set the poll rate to default, starts automatic data collection */
110  fd_acc = open(path_accel, O_RDONLY);
111 
112  if (fd_acc < 0) {
113  goto fail_accel;
114  }
115 
116  if (ioctl(fd_acc, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
117  goto fail_accel;
118  }
119 
120  close(fd_acc);
121  }
122 
123  if (sensor == BMI055_GYRO) {
124 
125  if (*g_dev_gyr_ptr != nullptr) {
126  errx(0, "bmi055 gyro sensor already started");
127  }
128 
129  /* create the driver */
130  if (external_bus) {
131 #if defined(PX4_SPI_BUS_EXT) && defined(PX4_SPIDEV_EXT_BMI)
132  *g_dev_ptr = new BMI055_gyro(PX4_SPI_BUS_EXT, path_gyro, PX4_SPIDEV_EXT_BMI, rotation);
133 #else
134  errx(0, "External SPI not available");
135 #endif
136 
137  } else {
138  *g_dev_gyr_ptr = new BMI055_gyro(PX4_SPI_BUS_SENSORS, path_gyro, PX4_SPIDEV_BMI055_GYR, rotation);
139  }
140 
141  if (*g_dev_gyr_ptr == nullptr) {
142  goto fail_gyro;
143  }
144 
145  if (OK != (*g_dev_gyr_ptr)->init()) {
146  goto fail_gyro;
147  }
148 
149  /* set the poll rate to default, starts automatic data collection */
150  fd_gyr = open(path_gyro, O_RDONLY);
151 
152  if (fd_gyr < 0) {
153  goto fail_gyro;
154  }
155 
156  if (ioctl(fd_gyr, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
157  goto fail_gyro;
158  }
159 
160  close(fd_gyr);
161  }
162 
163  exit(PX4_OK);
164 
165 fail_accel:
166 
167  if (*g_dev_acc_ptr != nullptr) {
168  delete (*g_dev_acc_ptr);
169  *g_dev_acc_ptr = nullptr;
170  }
171 
172  PX4_WARN("No BMI055 accel found");
173  exit(PX4_ERROR);
174 
175 fail_gyro:
176 
177  if (*g_dev_gyr_ptr != nullptr) {
178  delete (*g_dev_gyr_ptr);
179  *g_dev_gyr_ptr = nullptr;
180  }
181 
182  PX4_WARN("No BMI055 gyro found");
183  exit(PX4_ERROR);
184 }

218 /**
219  * Perform some basic functional tests on the driver;
220  * make sure we can collect data from the sensor in polled
221  * and automatic modes.
222  */
223 void
224 test(bool external_bus, enum sensor_type sensor)
225 {
226  const char *path_accel = external_bus ? BMI055_DEVICE_PATH_ACCEL_EXT : BMI055_DEVICE_PATH_ACCEL;
227  const char *path_gyro = external_bus ? BMI055_DEVICE_PATH_GYRO_EXT : BMI055_DEVICE_PATH_GYRO;
228  accel_report a_report;
229  gyro_report g_report;
230  ssize_t sz;
231 
232  if (sensor == BMI055_ACCEL) {
233  /* get the accel driver */
234  int fd_acc = open(path_accel, O_RDONLY);
235 
236  if (fd_acc < 0)
237  err(1, "%s Accel file open failed (try 'bmi055 -A start')",
238  path_accel);
239 
240 
241  /* reset to manual polling */
242  if (ioctl(fd_acc, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_MANUAL) < 0) {
243  err(1, "accel reset to manual polling");
244  }
245 
246  /* do a simple demand read */
247  sz = read(fd_acc, &a_report, sizeof(a_report));
248 
249  if (sz != sizeof(a_report)) {
250  warnx("ret: %d, expected: %d", sz, sizeof(a_report));
251  err(1, "immediate accel read failed");
252  }
253 
254  print_message(a_report);
255 
256  /* reset to default polling */
257  if (ioctl(fd_acc, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
258  err(1, "accel reset to default polling");
259  }
260 
261  close(fd_acc);
262  }
263 
264  if (sensor == BMI055_GYRO) {
265 
266  /* get the gyro driver */
267  int fd_gyr = open(path_gyro, O_RDONLY);
268 
269  if (fd_gyr < 0) {
270  err(1, "%s Gyro file open failed (try 'bmi055 -G start')", path_gyro);
271  }
272 
273  /* reset to manual polling */
274  if (ioctl(fd_gyr, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_MANUAL) < 0) {
275  err(1, "gyro reset to manual polling");
276  }
277 
278  /* do a simple demand read */
279  sz = read(fd_gyr, &g_report, sizeof(g_report));
280 
281  if (sz != sizeof(g_report)) {
282  warnx("ret: %d, expected: %d", sz, sizeof(g_report));
283  err(1, "immediate gyro read failed");
284  }
285 
286  print_message(g_report);
287 
288  /* reset to default polling */
289  if (ioctl(fd_gyr, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
290  err(1, "gyro reset to default polling");
291  }
292 
293  close(fd_gyr);
294  }
295 
296  if ((sensor == BMI055_ACCEL) || (sensor == BMI055_GYRO)) {
297  /* XXX add poll-rate tests here too */
298  reset(external_bus, sensor);
299  }
300 
301  errx(0, "PASS");
302 }
