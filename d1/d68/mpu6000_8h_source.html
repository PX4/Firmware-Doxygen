245 /*
246  The MPU6000 can only handle high bus speeds on the sensor and
247  interrupt status registers. All other registers have a maximum 1MHz
248  Communication with all registers of the device is performed using either
249  I2C at 400kHz or SPI at 1MHz. For applications requiring faster communications,
250  the sensor and interrupt registers may be read using SPI at 20MHz
251  */
252 #define MPU6000_LOW_BUS_SPEED 0
253 #define MPU6000_HIGH_BUS_SPEED 0x8000
254 # define MPU6000_IS_HIGH_SPEED(r) ((r) & MPU6000_HIGH_BUS_SPEED)
255 # define MPU6000_REG(r) ((r) &~MPU6000_HIGH_BUS_SPEED)
256 # define MPU6000_SET_SPEED(r, s) ((r)|(s))
257 # define MPU6000_HIGH_SPEED_OP(r) MPU6000_SET_SPEED((r), MPU6000_HIGH_BUS_SPEED)
258 # define MPU6000_LOW_SPEED_OP(r) MPU6000_REG((r))
