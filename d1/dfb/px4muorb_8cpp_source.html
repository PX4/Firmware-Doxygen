88 int px4muorb_param_update_to_shmem(uint32_t param, const uint8_t *value,
89  int data_len_in_bytes)
90 {
91  unsigned int byte_changed, bit_changed;
92  union param_value_u *param_value = (union param_value_u *) value;
93 
94  if (!shmem_info_p) {
95  init_shared_memory();
96  }
97 
98  if (get_shmem_lock(__FILE__, __LINE__) != 0) {
99  PX4_INFO("Could not get shmem lock\n");
100  return -1;
101  }
102 
103  shmem_info_p->params_val[param] = *param_value;
104 
105  byte_changed = param / 8;
106  bit_changed = 1 << param % 8;
107  shmem_info_p->krait_changed_index[byte_changed] |= bit_changed;
108 
109  release_shmem_lock(__FILE__, __LINE__);
110 
111  return 0;
112 }
