183 void
184 MPU9250_mag::_measure(struct ak8963_regs data)
185 {
186  bool mag_notify = true;
187 
188  if (check_duplicate((uint8_t *)&data.x) && !(data.st1 & 0x02)) {
189  perf_count(_mag_duplicates);
190  return;
191  }
192 
193  /* monitor for if data overrun flag is ever set */
194  if (data.st1 & 0x02) {
195  perf_count(_mag_overruns);
196  }
197 
198  /* monitor for if magnetic sensor overflow flag is ever set noting that st2
199  * is usually not even refreshed, but will always be in the same place in the
200  * mpu's buffers regardless, hence the actual count would be bogus
201  */
202  if (data.st2 & 0x08) {
203  perf_count(_mag_overflows);
204  }
205 
206  mag_report mrb;
207  mrb.timestamp = hrt_absolute_time();
208  mrb.is_external = false;
209 
210  /*
211  * Align axes - note the accel & gryo are also re-aligned so this
212  * doesn't look obvious with the datasheet
213  */
214  mrb.x_raw = data.x;
215  mrb.y_raw = -data.y;
216  mrb.z_raw = -data.z;
217 
218  float xraw_f = data.x;
219  float yraw_f = -data.y;
220  float zraw_f = -data.z;
221 
222  /* apply user specified rotation */
223  rotate_3f(_parent->_rotation, xraw_f, yraw_f, zraw_f);
224 
225  mrb.x = ((xraw_f * _mag_range_scale * _mag_asa_x) - _mag_scale.x_offset) * _mag_scale.x_scale;
226  mrb.y = ((yraw_f * _mag_range_scale * _mag_asa_y) - _mag_scale.y_offset) * _mag_scale.y_scale;
227  mrb.z = ((zraw_f * _mag_range_scale * _mag_asa_z) - _mag_scale.z_offset) * _mag_scale.z_scale;
228  mrb.scaling = _mag_range_scale;
229  mrb.temperature = _parent->_last_temperature;
230  mrb.device_id = _parent->_mag->_device_id.devid;
231 
232  mrb.error_count = perf_event_count(_mag_errors);
233 
234  _mag_reports->force(&mrb);
235 
236  /* notify anyone waiting for data */
237  if (mag_notify) {
238  poll_notify(POLLIN);
239  }
240 
241  if (mag_notify && !(_pub_blocked)) {
242  /* publish it */
243  orb_publish(ORB_ID(sensor_mag), _mag_topic, &mrb);
244  }
245 }

247 ssize_t
248 MPU9250_mag::read(struct file *filp, char *buffer, size_t buflen)
249 {
250  unsigned count = buflen / sizeof(mag_report);
251 
252  /* buffer must be large enough */
253  if (count < 1) {
254  return -ENOSPC;
255  }
256 
257  /* if automatic measurement is not enabled, get a fresh measurement into the buffer */
258  if (_parent->_call_interval == 0) {
259  _mag_reports->flush();
260  /* TODO: this won't work as getting valid magnetometer
261  * data requires more than one measure cycle
262  */
263  _parent->measure();
264  }
265 
266  /* if no data, error (we could block here) */
267  if (_mag_reports->empty()) {
268  return -EAGAIN;
269  }
270 
271  perf_count(_mag_reads);
272 
273  /* copy reports out of our buffer to the caller */
274  mag_report *mrp = reinterpret_cast<mag_report *>(buffer);
275  int transferred = 0;
276 
277  while (count--) {
278  if (!_mag_reports->get(mrp)) {
279  break;
280  }
281 
282  transferred++;
283  mrp++;
284  }
285 
286  /* return the number of bytes transferred */
287  return (transferred * sizeof(mag_report));
288 }

290 int
291 MPU9250_mag::ioctl(struct file *filp, int cmd, unsigned long arg)
292 {
293  switch (cmd) {
294 
295  case SENSORIOCRESET:
296  return ak8963_reset();
297 
298  case SENSORIOCSPOLLRATE: {
299  switch (arg) {
300 
301  /* switching to manual polling */
302  case SENSOR_POLLRATE_MANUAL:
303  /*
304  * TODO: investigate being able to stop
305  * the continuous sampling
306  */
307  //stop();
308  return OK;
309 
310  /* external signalling not supported */
311  case SENSOR_POLLRATE_EXTERNAL:
312 
313  /* zero would be bad */
314  case 0:
315  return -EINVAL;
316 
317  /* set default/max polling rate */
318  case SENSOR_POLLRATE_MAX:
319  return ioctl(filp, SENSORIOCSPOLLRATE, 100);
320 
321  case SENSOR_POLLRATE_DEFAULT:
322  return ioctl(filp, SENSORIOCSPOLLRATE, MPU9250_AK8963_SAMPLE_RATE);
323 
324  /* adjust to a legal polling interval in Hz */
325  default: {
326  if (MPU9250_AK8963_SAMPLE_RATE != arg) {
327  return -EINVAL;
328  }
329 
330  return OK;
331  }
332  }
333  }
334 
335  case SENSORIOCGPOLLRATE:
336  return MPU9250_AK8963_SAMPLE_RATE;
337 
338  case SENSORIOCSQUEUEDEPTH: {
339  /* lower bound is mandatory, upper bound is a sanity check */
340  if ((arg < 1) || (arg > 100)) {
341  return -EINVAL;
342  }
343 
344  irqstate_t flags = px4_enter_critical_section();
345 
346  if (!_mag_reports->resize(arg)) {
347  px4_leave_critical_section(flags);
348  return -ENOMEM;
349  }
350 
351  px4_leave_critical_section(flags);
352 
353  return OK;
354  }
355 
356  case MAGIOCGSAMPLERATE:
357  return MPU9250_AK8963_SAMPLE_RATE;
358 
359  case MAGIOCSSAMPLERATE:
360 
361  /*
362  * We don't currently support any means of changing
363  * the sampling rate of the mag
364  */
365  if (MPU9250_AK8963_SAMPLE_RATE != arg) {
366  return -EINVAL;
367  }
368 
369  return OK;
370 
371  case MAGIOCSSCALE:
372  /* copy scale in */
373  memcpy(&_mag_scale, (struct mag_scale *) arg, sizeof(_mag_scale));
374  return OK;
375 
376  case MAGIOCGSCALE:
377  /* copy scale out */
378  memcpy((struct mag_scale *) arg, &_mag_scale, sizeof(_mag_scale));
379  return OK;
380 
381  default:
382  return (int)CDev::ioctl(filp, cmd, arg);
383  }

492 bool
493 MPU9250_mag::ak8963_read_adjustments(void)
494 {
495  uint8_t response[3];
496  float ak8963_ASA[3];
497 
498  write_reg(AK8963REG_CNTL1, AK8963_FUZE_MODE | AK8963_16BIT_ADC);
499  usleep(50);
500 
501  if (_interface != nullptr) {
502  _interface->read(AK8963REG_ASAX, response, 3);
503 
504  } else {
505  passthrough_read(AK8963REG_ASAX, response, 3);
506  }
507 
508  write_reg(AK8963REG_CNTL1, AK8963_POWERDOWN_MODE);
509 
510  for (int i = 0; i < 3; i++) {
511  if (0 != response[i] && 0xff != response[i]) {
512  ak8963_ASA[i] = ((float)(response[i] - 128) / 256.0f) + 1.0f;
513 
514  } else {
515  return false;
516  }
517  }
518 
519  _mag_asa_x = ak8963_ASA[0];
520  _mag_asa_y = ak8963_ASA[1];
521  _mag_asa_z = ak8963_ASA[2];
522 
523  return true;
524 }

543 int
544 MPU9250_mag::ak8963_setup(void)
545 {
546  int retries = 10;
547 
548  do {
549 
550  ak8963_setup_master_i2c();
551  write_reg(AK8963REG_CNTL2, AK8963_RESET);
552 
553  uint8_t id = 0;
554 
555  if (ak8963_check_id(id)) {
556  break;
557  }
558 
559  retries--;
560  PX4_ERR("AK8963: bad id %d retries %d", id, retries);
561  _parent->modify_reg(MPUREG_USER_CTRL, 0, BIT_I2C_MST_RST);
562  up_udelay(100);
563  } while (retries > 0);
564 
565  if (retries > 0) {
566  retries = 10;
567 
568  while (!ak8963_read_adjustments() && retries) {
569  retries--;
570  PX4_ERR("AK8963: failed to read adjustment data. Retries %d", retries);
571 
572  _parent->modify_reg(MPUREG_USER_CTRL, 0, BIT_I2C_MST_RST);
573  up_udelay(100);
574  ak8963_setup_master_i2c();
575  write_reg(AK8963REG_CNTL2, AK8963_RESET);
576  }
577  }
578 
579  if (retries == 0) {
580  PX4_ERR("AK8963: failed to initialize, disabled!");
581  _parent->modify_checked_reg(MPUREG_USER_CTRL, BIT_I2C_MST_EN, 0);
582  _parent->write_reg(MPUREG_I2C_MST_CTRL, 0);
583  return -EIO;
584  }
585 
586  write_reg(AK8963REG_CNTL1, AK8963_CONTINUOUS_MODE2 | AK8963_16BIT_ADC);
587 
588 
589  if (_interface == NULL) {
590 
591  /* Configure mpu' I2c Master interface to read ak8963 data
592  * Into to fifo
593  */
594  set_passthrough(AK8963REG_ST1, sizeof(struct ak8963_regs));
595  }
596 
597  return OK;
598 }
