66 int do_esc_calibration(orb_advert_t *mavlink_log_pub, struct actuator_armed_s* armed)
67 {
68  int return_code = PX4_OK;
69 
70 #if defined(__PX4_POSIX_OCPOC) || defined(__PX4_POSIX_BBBLUE)
71  hrt_abstime timeout_start = 0;
72  hrt_abstime timeout_wait = 60_s;
73  armed->in_esc_calibration_mode = true;
74  calibration_log_info(mavlink_log_pub, CAL_QGC_DONE_MSG, "begin esc");
75  timeout_start = hrt_absolute_time();
76 
77  while (true) {
78  if (hrt_absolute_time() - timeout_start > timeout_wait) {
79  break;
80  }else{
81  usleep(50000);
82  }
83  }
84 
85  armed->in_esc_calibration_mode = false;
86  calibration_log_info(mavlink_log_pub, CAL_QGC_DONE_MSG, "end esc");
87 
88  if (return_code == OK) {
89  calibration_log_info(mavlink_log_pub, CAL_QGC_DONE_MSG, "esc");
90  }
91 
92  return return_code;
93 
94 #else
95  int fd = -1;
96 
97  struct battery_status_s battery = {};
98  int batt_sub = -1;
99  bool batt_updated = false;
100  bool batt_connected = true; // for safety resons assume battery is connected, will be cleared below if not the case
101 
102  hrt_abstime battery_connect_wait_timeout = 20_s;
103  hrt_abstime pwm_high_timeout = 3_s;
104  hrt_abstime timeout_start = 0;
105 
106  calibration_log_info(mavlink_log_pub, CAL_QGC_STARTED_MSG, "esc");
107 
108  batt_sub = orb_subscribe(ORB_ID(battery_status));
109  if (batt_sub < 0) {
110  calibration_log_critical(mavlink_log_pub, CAL_QGC_FAILED_MSG, "Subscribe to battery");
111  goto Error;
112  }
113 
114  // Make sure battery is disconnected
115  if (orb_copy(ORB_ID(battery_status), batt_sub, &battery) == PX4_OK) {
116 
117  // battery is not connected if the connected flag is not set and we have a recent battery measurement
118  if (!battery.connected && (hrt_absolute_time() - battery.timestamp < 500_ms)) {
119  batt_connected = false;
120  }
121  }
122 
123  if (batt_connected) {
124  calibration_log_critical(mavlink_log_pub, CAL_QGC_FAILED_MSG, "Disconnect battery and try again");
125  goto Error;
126  }
127 
128  armed->in_esc_calibration_mode = true;
129 
130  fd = px4_open(PWM_OUTPUT0_DEVICE_PATH, 0);
131 
132  if (fd < 0) {
133  calibration_log_critical(mavlink_log_pub, CAL_QGC_FAILED_MSG, "Can't open PWM device");
134  goto Error;
135  }
136 
137  /* tell IO/FMU that its ok to disable its safety with the switch */
138  if (px4_ioctl(fd, PWM_SERVO_SET_ARM_OK, 0) != PX4_OK) {
139  calibration_log_critical(mavlink_log_pub, CAL_QGC_FAILED_MSG, "Unable to disable safety switch");
140  goto Error;
141  }
142 
143  /* tell IO/FMU that the system is armed (it will output values if safety is off) */
144  if (px4_ioctl(fd, PWM_SERVO_ARM, 0) != PX4_OK) {
145  calibration_log_critical(mavlink_log_pub, CAL_QGC_FAILED_MSG, "Unable to arm system");
146  goto Error;
147  }
148 
149  /* tell IO to switch off safety without using the safety switch */
150  if (px4_ioctl(fd, PWM_SERVO_SET_FORCE_SAFETY_OFF, 0) != PX4_OK) {
151  calibration_log_critical(mavlink_log_pub, CAL_QGC_FAILED_MSG, "Unable to force safety off");
152  goto Error;
153  }
154 
155  calibration_log_info(mavlink_log_pub, "[cal] Connect battery now");
156 
157  timeout_start = hrt_absolute_time();
158 
159  while (true) {
160  // We are either waiting for the user to connect the battery. Or we are waiting to let the PWM
161  // sit high.
162  hrt_abstime timeout_wait = batt_connected ? pwm_high_timeout : battery_connect_wait_timeout;
163 
164  if (hrt_elapsed_time(&timeout_start) > timeout_wait) {
165  if (!batt_connected) {
166  calibration_log_critical(mavlink_log_pub, CAL_QGC_FAILED_MSG, "Timeout waiting for battery");
167  goto Error;
168  }
169 
170  // PWM was high long enough
171  break;
172  }
173 
174  if (!batt_connected) {
175  orb_check(batt_sub, &batt_updated);
176  if (batt_updated) {
177  orb_copy(ORB_ID(battery_status), batt_sub, &battery);
178  if (battery.connected) {
179  // Battery is connected, signal to user and start waiting again
180  batt_connected = true;
181  timeout_start = hrt_absolute_time();
182  calibration_log_info(mavlink_log_pub, "[cal] Battery connected");
183  }
184  }
185  }
186  usleep(50_ms);
187  }
188 
189 Out:
190  if (batt_sub != -1) {
191  orb_unsubscribe(batt_sub);
192  }
193  if (fd != -1) {
194  if (px4_ioctl(fd, PWM_SERVO_SET_FORCE_SAFETY_ON, 0) != PX4_OK) {
195  calibration_log_info(mavlink_log_pub, CAL_QGC_FAILED_MSG, "Safety switch still off");
196  }
197  if (px4_ioctl(fd, PWM_SERVO_DISARM, 0) != PX4_OK) {
198  calibration_log_info(mavlink_log_pub, CAL_QGC_FAILED_MSG, "Servos still armed");
199  }
200  if (px4_ioctl(fd, PWM_SERVO_CLEAR_ARM_OK, 0) != PX4_OK) {
201  calibration_log_info(mavlink_log_pub, CAL_QGC_FAILED_MSG, "Safety switch still deactivated");
202  }
203  px4_close(fd);
204  }
205  armed->in_esc_calibration_mode = false;
206 
207  if (return_code == PX4_OK) {
208  calibration_log_info(mavlink_log_pub, CAL_QGC_DONE_MSG, "esc");
209  }
210 
211  return return_code;
212 
213 Error:
214  return_code = PX4_ERROR;
215  goto Out;
216 #endif
217 }
