54 // golden section search to find extremeum for function with minimum
55 template<typename _Tp>
56 inline const _Tp goldensection(const _Tp &arg1, const _Tp &arg2, _Tp(*fun)(_Tp), const _Tp &tol)
57 {
58  _Tp a = arg1;
59  _Tp b = arg2;
60  _Tp c = b - (b - a) / GOLDEN_RATIO;
61  _Tp d = a + (b - a) / GOLDEN_RATIO;
62 
63  while (abs_t(c - d) > tol) {
64 
65  if (fun(c) < fun(d)) {
66  b = d;
67 
68  } else {
69  a = c;
70  }
71 
72  c = b - (b - a) / GOLDEN_RATIO;
73  d = a + (b - a) / GOLDEN_RATIO;
74 
75  }
76 
77  return ((b + a) / (_Tp)2);
78 }
