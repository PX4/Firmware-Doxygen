98 __EXPORT void stm32_spi1select(FAR struct spi_dev_s *dev, uint32_t devid, bool selected)
99 {
100  /* SPI select is active low, so write !selected to select the device */
101 
102  switch (devid) {
103 
104  case PX4_SPIDEV_ICM:
105 
106  /* Intended fallthrough */
107 
108  case PX4_SPIDEV_ICM_20602:
109 
110  /* Intended fallthrough */
111 
112  case PX4_SPIDEV_ICM_20608:
113 
114  /* Making sure the other peripherals are not selected */
115  stm32_gpiowrite(GPIO_SPI_CS_MPU9250, 1);
116  stm32_gpiowrite(GPIO_SPI_CS_LIS3MDL, 1);
117  stm32_gpiowrite(GPIO_SPI_CS_MS5611, 1);
118  stm32_gpiowrite(GPIO_SPI_CS_ICM_2060X, !selected);
119  stm32_gpiowrite(GPIO_SPI_CS_TEMPCAL_EEPROM, 1);
120  break;
121 
122  case PX4_SPIDEV_BARO:
123  /* Making sure the other peripherals are not selected */
124  stm32_gpiowrite(GPIO_SPI_CS_MPU9250, 1);
125  stm32_gpiowrite(GPIO_SPI_CS_LIS3MDL, 1);
126  stm32_gpiowrite(GPIO_SPI_CS_MS5611, !selected);
127  stm32_gpiowrite(GPIO_SPI_CS_ICM_2060X, 1);
128  stm32_gpiowrite(GPIO_SPI_CS_TEMPCAL_EEPROM, 1);
129  break;
130 
131  case PX4_SPIDEV_LIS:
132  /* Making sure the other peripherals are not selected */
133  stm32_gpiowrite(GPIO_SPI_CS_MPU9250, 1);
134  stm32_gpiowrite(GPIO_SPI_CS_LIS3MDL, !selected);
135  stm32_gpiowrite(GPIO_SPI_CS_MS5611, 1);
136  stm32_gpiowrite(GPIO_SPI_CS_ICM_2060X, 1);
137  stm32_gpiowrite(GPIO_SPI_CS_TEMPCAL_EEPROM, 1);
138  break;
139 
140  case PX4_SPIDEV_MPU:
141  /* Making sure the other peripherals are not selected */
142  stm32_gpiowrite(GPIO_SPI_CS_MPU9250, !selected);
143  stm32_gpiowrite(GPIO_SPI_CS_LIS3MDL, 1);
144  stm32_gpiowrite(GPIO_SPI_CS_MS5611, 1);
145  stm32_gpiowrite(GPIO_SPI_CS_ICM_2060X, 1);
146  stm32_gpiowrite(GPIO_SPI_CS_TEMPCAL_EEPROM, 1);
147  break;
148 
149  case PX4_SPIDEV_EEPROM:
150  /* Making sure the other peripherals are not selected */
151  stm32_gpiowrite(GPIO_SPI_CS_MPU9250, 1);
152  stm32_gpiowrite(GPIO_SPI_CS_LIS3MDL, 1);
153  stm32_gpiowrite(GPIO_SPI_CS_MS5611, 1);
154  stm32_gpiowrite(GPIO_SPI_CS_ICM_2060X, 1);
155  stm32_gpiowrite(GPIO_SPI_CS_TEMPCAL_EEPROM, !selected);
156  break;
157 
158  default:
159  break;
160  }
161 }

224 __EXPORT void board_spi_reset(int ms)
225 {
226  /* disable SPI bus */
227  stm32_configgpio(GPIO_SPI_CS_OFF_MPU9250);
228  stm32_configgpio(GPIO_SPI_CS_OFF_LIS3MDL);
229  stm32_configgpio(GPIO_SPI_CS_OFF_MS5611);
230  stm32_configgpio(GPIO_SPI_CS_OFF_ICM_2060X);
231  stm32_configgpio(GPIO_SPI_CS_TEMPCAL_EEPROM);
232 
233  stm32_gpiowrite(GPIO_SPI_CS_OFF_MPU9250, 0);
234  stm32_gpiowrite(GPIO_SPI_CS_OFF_LIS3MDL, 0);
235  stm32_gpiowrite(GPIO_SPI_CS_OFF_MS5611, 0);
236  stm32_gpiowrite(GPIO_SPI_CS_OFF_ICM_2060X, 0);
237  stm32_gpiowrite(GPIO_SPI_CS_TEMPCAL_EEPROM, 0);
238 
239  stm32_configgpio(GPIO_SPI1_SCK_OFF);
240  stm32_configgpio(GPIO_SPI1_MISO_OFF);
241  stm32_configgpio(GPIO_SPI1_MOSI_OFF);
242 
243  stm32_gpiowrite(GPIO_SPI1_SCK_OFF, 0);
244  stm32_gpiowrite(GPIO_SPI1_MISO_OFF, 0);
245  stm32_gpiowrite(GPIO_SPI1_MOSI_OFF, 0);
246 
247  stm32_configgpio(GPIO_DRDY_OFF_MPU9250);
248  stm32_configgpio(GPIO_DRDY_OFF_LIS3MDL);
249  stm32_configgpio(GPIO_DRDY_OFF_ICM_2060X);
250 
251  stm32_gpiowrite(GPIO_DRDY_OFF_MPU9250, 0);
252  stm32_gpiowrite(GPIO_DRDY_OFF_LIS3MDL, 0);
253  stm32_gpiowrite(GPIO_DRDY_OFF_ICM_2060X, 0);
254 
255  /* set the sensor rail off */
256  stm32_configgpio(GPIO_VDD_3V3_SENSORS_EN);
257  stm32_gpiowrite(GPIO_VDD_3V3_SENSORS_EN, 0);
258 
259  /* wait for the sensor rail to reach GND */
260  usleep(ms * 1000);
261  warnx("reset done, %d ms", ms);
262 
263  /* re-enable power */
264 
265  /* switch the sensor rail back on */
266  stm32_gpiowrite(GPIO_VDD_3V3_SENSORS_EN, 1);
267 
268  /* wait a bit before starting SPI, different times didn't influence results */
269  usleep(100);
270 
271  /* reconfigure the SPI pins */
272 #ifdef CONFIG_STM32_SPI1
273  stm32_configgpio(GPIO_SPI_CS_MPU9250);
274  stm32_configgpio(GPIO_SPI_CS_LIS3MDL);
275  stm32_configgpio(GPIO_SPI_CS_MS5611);
276  stm32_configgpio(GPIO_SPI_CS_ICM_2060X);
277  stm32_configgpio(GPIO_SPI_CS_TEMPCAL_EEPROM);
278 
279  stm32_configgpio(GPIO_SPI1_SCK);
280  stm32_configgpio(GPIO_SPI1_MISO);
281  stm32_configgpio(GPIO_SPI1_MOSI);
282 
283  /* bring up the EXTI pins again */
284  stm32_configgpio(GPIO_DRDY_MPU9250);
285  stm32_configgpio(GPIO_DRDY_LIS3MDL);
286  stm32_configgpio(GPIO_DRDY_ICM_2060X);
287 
288 #endif
289 
290 }
