67 void FlightTaskAutoLine::_generateXYsetpoints()
68 {
69  Vector2f pos_sp_to_dest = Vector2f(&(_target - _position_setpoint)(0));
70  const bool has_reached_altitude = fabsf(_target(2) - _position(2)) < NAV_ACC_RAD.get();
71 
72  if ((_speed_at_target < 0.001f && pos_sp_to_dest.length() < NAV_ACC_RAD.get()) ||
73  (!has_reached_altitude && pos_sp_to_dest.length() < NAV_ACC_RAD.get())) {
74 
75  // Vehicle reached target in xy and no passing required. Lock position */
76  _position_setpoint(0) = _target(0);
77  _position_setpoint(1) = _target(1);
78  _velocity_setpoint(0) = _velocity_setpoint(1) = 0.0f;
79 
80  } else {
81 
82  // Get various path specific vectors. */
83  Vector2f u_prev_to_dest = Vector2f(&(_target - _prev_wp)(0)).unit_or_zero();
84  Vector2f prev_to_pos(&(_position - _prev_wp)(0));
85  Vector2f closest_pt = Vector2f(&_prev_wp(0)) + u_prev_to_dest * (prev_to_pos * u_prev_to_dest);
86  Vector2f closest_to_dest = Vector2f(&(_target - _position)(0));
87  Vector2f prev_to_dest = Vector2f(&(_target - _prev_wp)(0));
88  float speed_sp_track = _mc_cruise_speed;
89  float speed_sp_prev_track = math::max(Vector2f(&_velocity_setpoint(0)) * u_prev_to_dest, 0.0f);
90 
91  // Distance to target when brake should occur. The assumption is made that
92  // 1.5 * cruising speed is enough to break.
93  float target_threshold = 1.5f * _mc_cruise_speed;
94  float speed_threshold = _mc_cruise_speed;
95  const float threshold_max = target_threshold;
96 
97  if (target_threshold > 0.5f * prev_to_dest.length()) {
98  // Target threshold cannot be more than distance from previous to target
99  target_threshold = 0.5f * prev_to_dest.length();
100  }
101 
102  // Compute maximum speed at target threshold */
103  if (threshold_max > NAV_ACC_RAD.get()) {
104  float m = (_mc_cruise_speed - _speed_at_target) / (threshold_max - NAV_ACC_RAD.get());
105  speed_threshold = m * (target_threshold - NAV_ACC_RAD.get()) + _speed_at_target; // speed at transition
106  }
107 
108  // Either accelerate or decelerate
109  if (closest_to_dest.length() < target_threshold) {
110 
111  // Vehicle is close to destination. Start to decelerate
112 
113  if (!has_reached_altitude) {
114  // Altitude is not reached yet. Vehicle has to stop first before proceeding
115  _speed_at_target = 0.0f;
116  }
117 
118  float acceptance_radius = NAV_ACC_RAD.get();
119 
120  if (_speed_at_target < 0.01f) {
121  // If vehicle wants to stop at the target, then set acceptance radius to zero as well.
122  acceptance_radius = 0.0f;
123  }
124 
125  if ((target_threshold - acceptance_radius) >= SIGMA_NORM) {
126 
127  // Slow down depending on distance to target minus acceptance radius.
128  float m = (speed_threshold - _speed_at_target) / (target_threshold - acceptance_radius);
129  speed_sp_track = m * (closest_to_dest.length() - acceptance_radius) + _speed_at_target; // speed at transition
130 
131  } else {
132  speed_sp_track = _speed_at_target;
133  }
134 
135  // If we are close to target and the previous speed setpoint along track was smaller than
136  // current speed setpoint along track, then take over the previous one.
137  // This ensures smoothness since we anyway want to slow down.
138  if ((speed_sp_prev_track < speed_sp_track)
139  && (speed_sp_track * speed_sp_prev_track > 0.0f)
140  && (speed_sp_prev_track > _speed_at_target)) {
141  speed_sp_track = speed_sp_prev_track;
142  }
143 
144  } else {
145 
146  // Vehicle is still far from destination. Accelerate or keep maximum target speed.
147  float acc_track = (speed_sp_track - speed_sp_prev_track) / _deltatime;
148 
149  float yaw_diff = 0.0f;
150 
151  if (PX4_ISFINITE(_yaw_setpoint)) {
152  yaw_diff = wrap_pi(_yaw_setpoint - _yaw);
153  }
154 
155  // If yaw offset is large, only accelerate with 0.5 m/s^2.
156  float acc_max = (fabsf(yaw_diff) > math::radians(MIS_YAW_ERR.get())) ? 0.5f : MPC_ACC_HOR.get();
157 
158  if (acc_track > acc_max) {
159  // accelerate towards target
160  speed_sp_track = acc_max * _deltatime + speed_sp_prev_track;
161  }
162  }
163 
164  speed_sp_track = math::constrain(speed_sp_track, 0.0f, _mc_cruise_speed);
165 
166  _position_setpoint(0) = closest_pt(0);
167  _position_setpoint(1) = closest_pt(1);
168  Vector2f velocity_sp_xy = u_prev_to_dest * speed_sp_track;
169  _velocity_setpoint(0) = velocity_sp_xy(0);
170  _velocity_setpoint(1) = velocity_sp_xy(1);
171  }
172 }

174 void FlightTaskAutoLine::_generateAltitudeSetpoints()
175 {
176  // Total distance between previous and target set-point.
177  const float dist = fabsf(_target(2) - _prev_wp(2));
178 
179  // If target has not been reached, then compute set-point depending on maximum velocity.
180  if ((dist > SIGMA_NORM) && (fabsf(_position(2) - _target(2)) > 0.1f)) {
181 
182  // get various distances */
183  const float dist_to_prev = fabsf(_position(2) - _prev_wp(2));
184  const float dist_to_target = fabsf(_target(2) - _position(2));
185 
186  // check sign
187  const bool flying_upward = _target(2) < _position(2);
188 
189  // limit vertical downwards speed (positive z) close to ground
190  // for now we use the altitude above home and assume that we want to land at same height as we took off
191  float vel_limit = math::gradual(_alt_above_ground,
192  MPC_LAND_ALT2.get(), MPC_LAND_ALT1.get(),
193  MPC_LAND_SPEED.get(), _constraints.speed_down);
194 
195  // Speed at threshold is by default maximum speed. Threshold defines
196  // the point in z at which vehicle slows down to reach target altitude.
197  float speed_sp = (flying_upward) ? _constraints.speed_up : vel_limit;
198 
199  // Target threshold defines the distance to target(2) at which
200  // the vehicle starts to slow down to approach the target smoothly.
201  float target_threshold = speed_sp * 1.5f;
202 
203  // If the total distance in z is NOT 2x distance of target_threshold, we
204  // will need to adjust the final_velocity in z.
205  const bool is_2_target_threshold = dist >= 2.0f * target_threshold;
206  const float min_vel = 0.2f; // minimum velocity: this is needed since estimation is not perfect
207  const float slope = (speed_sp - min_vel) / (target_threshold); // defines the the acceleration when slowing down */
208 
209  if (!is_2_target_threshold) {
210  // Adjust final_velocity since we are already are close to target and therefore it is not necessary to accelerate
211  // upwards with full speed.
212  target_threshold = dist * 0.5f;
213  // get the velocity at target_threshold
214  speed_sp = slope * (target_threshold) + min_vel;
215  }
216 
217  // we want to slow down
218  if (dist_to_target < target_threshold) {
219 
220  speed_sp = slope * dist_to_target + min_vel;
221 
222  } else if (dist_to_prev < target_threshold) {
223  // we want to accelerate
224 
225  const float acc = (speed_sp - fabsf(_velocity_setpoint(2))) / _deltatime;
226  const float acc_max = (flying_upward) ? (MPC_ACC_UP_MAX.get() * 0.5f) : (MPC_ACC_DOWN_MAX.get() * 0.5f);
227 
228  if (acc > acc_max) {
229  speed_sp = acc_max * _deltatime + fabsf(_velocity_setpoint(2));
230  }
231  }
232 
233  // make sure vel_sp_z is always positive
234  if (speed_sp < 0.0f) {
235  PX4_WARN("speed cannot be smaller than 0");
236  speed_sp = 0.0f;
237  }
238 
239  // get the sign of vel_sp_z
240  _velocity_setpoint(2) = (flying_upward) ? -speed_sp : speed_sp;
241  _position_setpoint(2) = NAN; // We don't care about position setpoint
242 
243  } else {
244 
245  // vehicle reached desired target altitude
246  _velocity_setpoint(2) = 0.0f;
247  _position_setpoint(2) = _target(2);
248  }
249 }
