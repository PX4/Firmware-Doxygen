274 int
275 CDev::poll(file_t *filep, px4_pollfd_struct_t *fds, bool setup)
276 {
277  DEVICE_POLL_DEBUG("CDev::Poll %s", setup ? "setup" : "teardown");
278  int ret = PX4_OK;
279 
280  if (setup) {
281  /*
282  * Save the file pointer in the pollfd for the subclass'
283  * benefit.
284  */
285  fds->priv = (void *)filep;
286  DEVICE_POLL_DEBUG("CDev::poll: fds->priv = %p", filep);
287 
288  /*
289  * Lock against poll_notify() and possibly other callers (protect _pollset).
290  */
291  ATOMIC_ENTER;
292 
293  /*
294  * Try to store the fds for later use and handle array resizing.
295  */
296  while ((ret = store_poll_waiter(fds)) == -ENFILE) {
297 
298  // No free slot found. Resize the pollset. This is expensive, but it's only needed initially.
299 
300  if (_max_pollwaiters >= 256 / 2) { //_max_pollwaiters is uint8_t
301  ret = -ENOMEM;
302  break;
303  }
304 
305  const uint8_t new_count = _max_pollwaiters > 0 ? _max_pollwaiters * 2 : 1;
306  px4_pollfd_struct_t **prev_pollset = _pollset;
307 
308 #ifdef __PX4_NUTTX
309  // malloc uses a semaphore, we need to call it enabled IRQ's
310  px4_leave_critical_section(flags);
311 #endif
312  px4_pollfd_struct_t **new_pollset = new px4_pollfd_struct_t *[new_count];
313 
314 #ifdef __PX4_NUTTX
315  flags = px4_enter_critical_section();
316 #endif
317 
318  if (prev_pollset == _pollset) {
319  // no one else updated the _pollset meanwhile, so we're good to go
320  if (!new_pollset) {
321  ret = -ENOMEM;
322  break;
323  }
324 
325  if (_max_pollwaiters > 0) {
326  memset(new_pollset + _max_pollwaiters, 0, sizeof(px4_pollfd_struct_t *) * (new_count - _max_pollwaiters));
327  memcpy(new_pollset, _pollset, sizeof(px4_pollfd_struct_t *) * _max_pollwaiters);
328  }
329 
330  _pollset = new_pollset;
331  _pollset[_max_pollwaiters] = fds;
332  _max_pollwaiters = new_count;
333 
334  // free the previous _pollset (we need to unlock here which is fine because we don't access _pollset anymore)
335 #ifdef __PX4_NUTTX
336  px4_leave_critical_section(flags);
337 #endif
338 
339  if (prev_pollset) {
340  delete[](prev_pollset);
341  }
342 
343 #ifdef __PX4_NUTTX
344  flags = px4_enter_critical_section();
345 #endif
346 
347  // Success
348  ret = PX4_OK;
349  break;
350  }
351 
352 #ifdef __PX4_NUTTX
353  px4_leave_critical_section(flags);
354 #endif
355  // We have to retry
356  delete[] new_pollset;
357 #ifdef __PX4_NUTTX
358  flags = px4_enter_critical_section();
359 #endif
360  }
361 
362  if (ret == PX4_OK) {
363 
364  /*
365  * Check to see whether we should send a poll notification
366  * immediately.
367  */
368  fds->revents |= fds->events & poll_state(filep);
369 
370  /* yes? post the notification */
371  if (fds->revents != 0) {
372  px4_sem_post(fds->sem);
373  }
374 
375  }
376 
377  ATOMIC_LEAVE;
378 
379  } else {
380  ATOMIC_ENTER;
381  /*
382  * Handle a teardown request.
383  */
384  ret = remove_poll_waiter(fds);
385  ATOMIC_LEAVE;
386  }
387 
388  return ret;
389 }

393 {
394  DEVICE_POLL_DEBUG("CDev::poll_notify events = %0x", events);
395 
396  /* lock against poll() as well as other wakeups */
397  ATOMIC_ENTER;
398 
399  for (unsigned i = 0; i < _max_pollwaiters; i++) {
