388 int
389 PMW3901::ioctl(device::file_t *filp, int cmd, unsigned long arg)
390 {
391  switch (cmd) {
392 
393  case SENSORIOCSPOLLRATE: {
394  switch (arg) {
395 
396  case 0:
397  return -EINVAL;
398 
399  case SENSOR_POLLRATE_DEFAULT: {
400 
401  /* do we need to start internal polling? */
402  bool want_start = (_measure_ticks == 0);
403 
404  /* set interval for next measurement to minimum legal value */
405  _measure_ticks = USEC2TICK(PMW3901_SAMPLE_RATE);
406 
407  /* if we need to start the poll state machine, do it */
408  if (want_start) {
409  start();
410  }
411 
412  return OK;
413  }
414 
415  case SENSOR_POLLRATE_MANUAL: {
416 
417  stop();
418  _measure_ticks = 0;
419  return OK;
420  }
421 
422  /* adjust to a legal polling interval in Hz */
423  default: {
424  /* do we need to start internal polling? */
425  bool want_start = (_measure_ticks == 0);
426 
427  /* convert hz to tick interval via microseconds */
428  unsigned ticks = USEC2TICK(1000000 / arg);
429 
430  /* check against maximum rate */
431  if (ticks < USEC2TICK(PMW3901_SAMPLE_RATE)) {
432  return -EINVAL;
433  }
434 
435  /* update interval for next measurement */
436  _measure_ticks = ticks;
437 
438  /* if we need to start the poll state machine, do it */
439  if (want_start) {
440  start();
441  }
442 
443  return OK;
444  }
445  }
446  }
447 
448  default:
449  /* give it to the superclass */
450  return SPI::ioctl(filp, cmd, arg);
451  }

454 ssize_t
455 PMW3901::read(device::file_t *filp, char *buffer, size_t buflen)
456 {
457  unsigned count = buflen / sizeof(struct optical_flow_s);
458  struct optical_flow_s *rbuf = reinterpret_cast<struct optical_flow_s *>(buffer);
459  int ret = 0;
460 
461  /* buffer must be large enough */
462  if (count < 1) {
463  return -ENOSPC;
464  }
465 
466  /* if automatic measurement is enabled */
467  if (_measure_ticks > 0) {
468 
469  /*
470  * While there is space in the caller's buffer, and reports, copy them.
471  * Note that we may be pre-empted by the workq thread while we are doing this;
472  * we are careful to avoid racing with them.
473  */
474  while (count--) {
475  if (_reports->get(rbuf)) {
476  ret += sizeof(*rbuf);
477  rbuf++;
478  }
479  }
480 
481  /* if there was no data, warn the caller */
482  return ret ? ret : -EAGAIN;
483  }
484 
485  /* manual measurement - run one conversion */
486  do {
487  _reports->flush();
488 
489  /* trigger a measurement */
490  if (OK != collect()) {
491  ret = -EIO;
492  break;
493  }
494 
495  /* state machine will have generated a report, copy it out */
496  if (_reports->get(rbuf)) {
497  ret = sizeof(*rbuf);
498  }
499 
500  } while (0);
501 
502  return ret;
503 }

551 int
552 PMW3901::collect()
553 {
554  int ret = OK;
555  int16_t delta_x_raw, delta_y_raw;
556  float delta_x, delta_y;
557 
558  perf_begin(_sample_perf);
559 
560  uint64_t timestamp = hrt_absolute_time();
561  uint64_t dt_flow = timestamp - _previous_collect_timestamp;
562  _previous_collect_timestamp = timestamp;
563 
564  _flow_dt_sum_usec += dt_flow;
565 
566  readMotionCount(delta_x_raw, delta_y_raw);
567 
568  _flow_sum_x += delta_x_raw;
569  _flow_sum_y += delta_y_raw;
570 
571  if (_flow_dt_sum_usec < 45000) {
572 
573  return ret;
574  }
575 
576  delta_x = (float)_flow_sum_x / 500.0f; // proportional factor + convert from pixels to radians
577  delta_y = (float)_flow_sum_y / 500.0f; // proportional factor + convert from pixels to radians
578 
579  struct optical_flow_s report;
580 
581  report.timestamp = timestamp;
582 
583  report.pixel_flow_x_integral = static_cast<float>(delta_x);
584  report.pixel_flow_y_integral = static_cast<float>(delta_y);
585 
586  report.frame_count_since_last_readout = 4; //microseconds
587  report.integration_timespan = _flow_dt_sum_usec; //microseconds
588 
589  report.sensor_id = 0;
590  report.quality = 255;
591 
592  /* No gyro on this board */
593  report.gyro_x_rate_integral = NAN;
594  report.gyro_y_rate_integral = NAN;
595  report.gyro_z_rate_integral = NAN;
596 
597  _flow_dt_sum_usec = 0;
598  _flow_sum_x = 0;
599  _flow_sum_y = 0;
600 
601  if (_optical_flow_pub == nullptr) {
602 
603  _optical_flow_pub = orb_advertise(ORB_ID(optical_flow), &report);
604 
605  } else {
606 
607  orb_publish(ORB_ID(optical_flow), _optical_flow_pub, &report);
608  }
609 
610  /* post a report to the ring */
611  _reports->force(&report);
612 
613  /* notify anyone waiting for data */
614  poll_notify(POLLIN);
615 
616  perf_end(_sample_perf);
617 
618  return ret;
619

746  }
747

754

758

764

766
