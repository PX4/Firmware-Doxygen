97  switch (ch) {
98  case 'b':
99  block_size = strtol(myoptarg, NULL, 0);
100  break;
101 
102  case 'r':
103  num_runs = strtol(myoptarg, NULL, 0);
104  break;
105 
106  case 'd':
107  run_duration = strtol(myoptarg, NULL, 0);
108  break;
109 
110  case 's':
111  synchronized = true;
112  break;
113 
114  default:
115  usage();
116  return -1;
117  break;
118  }
119  }
120 
121  if (block_size <= 0 || num_runs <= 0) {
122  PX4_ERR("invalid argument");
123  return -1;
124  }
125 
126  int bench_fd = open(BENCHMARK_FILE, O_CREAT | O_WRONLY | O_TRUNC, PX4_O_MODE_666);
127 
128  if (bench_fd < 0) {
129  PX4_ERR("Can't open benchmark file %s", BENCHMARK_FILE);
130  return -1;
131  }
132 
133  //create some data block
134  uint8_t *block = (uint8_t *)malloc(block_size);
135 
136  if (!block) {
137  PX4_ERR("Failed to allocate memory block");
138  close(bench_fd);
139  return -1;
140  }
141 
142  for (int i = 0; i < block_size; ++i) {
143  block[i] = (uint8_t)i;
144  }
145 
146  PX4_INFO("Using block size = %i bytes, sync=%i", block_size, (int)synchronized);
147  write_test(bench_fd, block, block_size);

149  free(block);

151  unlink(BENCHMARK_FILE);

153  return 0;

163 void write_test(int fd, uint8_t *block, int block_size)
164 {
165  PX4_INFO("");
166  PX4_INFO("Testing Sequential Write Speed...");
167  double total_elapsed = 0.;
168  unsigned int total_blocks = 0;
169 
170  for (int run = 0; run < num_runs; ++run) {
171  hrt_abstime start = hrt_absolute_time();
172  unsigned int num_blocks = 0;
173  unsigned int max_write_time = 0;
174  unsigned int fsync_time = 0;
175 
176  while ((int64_t)hrt_elapsed_time(&start) < run_duration * 1000) {
177 
178  hrt_abstime write_start = hrt_absolute_time();
179  size_t written = write(fd, block, block_size);
180  unsigned int write_time = hrt_elapsed_time(&write_start) / 1000;
181 
182  if (write_time > max_write_time) {
183  max_write_time = write_time;
184  }
185 
186  if ((int)written != block_size) {
187  PX4_ERR("Write error");
188  return;
189  }
190 
191  if (synchronized) {
192  fsync_time += time_fsync(fd);
193  }
194 
195  ++num_blocks;
196  }
197 
198  //Note: if testing a slow device (SD Card) and the OS buffers a lot (eg. Linux),
199  //fsync can take really long, and it looks like the process hangs. But it does
200  //not and the reported result will still be correct.
201  fsync_time += time_fsync(fd);
202 
203  //report
204  double elapsed = hrt_elapsed_time(&start) / 1.e6;
205  PX4_INFO(" Run %2i: %8.2lf KB/s, max write time: %i ms (=%7.2lf KB/s), fsync: %i ms", run,
206  (double)block_size * num_blocks / elapsed / 1024.,
207  max_write_time, (double)block_size / max_write_time * 1000. / 1024., fsync_time);
208 
209  total_elapsed += elapsed;
210  total_blocks += num_blocks;
211  }
212 
213  PX4_INFO(" Avg : %8.2lf KB/s", (double)block_size * total_blocks / total_elapsed / 1024.);
214 }
