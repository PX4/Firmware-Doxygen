116 __EXPORT void board_autoled_on(int led)
117 {
118  switch (led) {
119  default:
120  case LED_STARTED:
121  case LED_HEAPALLOCATE:
122  case LED_IRQSENABLED:
123  phy_set_led(BOARD_LED_GREEN, false);
124  phy_set_led(BOARD_LED_YELLOW, false);
125  break;
126 
127  case LED_STACKCREATED:
128  phy_set_led(BOARD_LED_GREEN, true);
129  phy_set_led(BOARD_LED_YELLOW, false);
130  g_initialized = true;
131  break;
132 
133  case LED_INIRQ:
134  case LED_SIGNAL:
135  case LED_ASSERTION:
136  case LED_PANIC:
137  phy_set_led(BOARD_LED_YELLOW, true);
138  break;
139 
140  case LED_IDLE : /* IDLE */
141  phy_set_led(BOARD_LED_GREEN, false);
142  break;
143  }
144 }
