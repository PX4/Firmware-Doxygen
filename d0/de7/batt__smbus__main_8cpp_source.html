78 /**
79  * start driver for a specific bus option
80  */
81 bool start_bus(struct batt_smbus_bus_option &bus)
82 {
83  PX4_INFO("starting %s", bus.devpath);
84 
85  if (bus.dev != nullptr) {
86  PX4_WARN("bus option already started");
87  return false;
88  }
89 
90  device::Device *interface = bus.interface_constructor(bus.busnum);
91 
92  if (interface->init() != OK) {
93  delete interface;
94  PX4_WARN("no device on bus %u", (unsigned)bus.busid);
95  return false;
96  }
97 
98  bus.dev = new BATT_SMBUS(interface, bus.devpath);
99 
100  if (bus.dev != nullptr && PX4_OK != bus.dev->init()) {
101  PX4_WARN("init failed");
102  delete bus.dev;
103  bus.dev = nullptr;
104  return false;
105  }
106 
107  return true;
108 }

110 /**
111  * Start the driver.
112  *
113  * This function call only returns once the driver
114  * is either successfully up and running or failed to start.
115  */
116 int start(enum BATT_SMBUS_BUS busid)
117 {
118  for (unsigned i = 0; i < NUM_BUS_OPTIONS; i++) {
119  if (busid == BATT_SMBUS_BUS_ALL && bus_options[i].dev != nullptr) {
120  // This device is already started.
121  PX4_INFO("Smart battery %d already started", bus_options[i].dev);
122  continue;
123  }
124 
125  if (busid != BATT_SMBUS_BUS_ALL && bus_options[i].busid != busid) {
126  // Not the one that is asked for.
127  continue;
128  }
129 
130  if (start_bus(bus_options[i])) {
131  return PX4_OK;
132  }
133 
134  PX4_INFO("Smart battery failed to start");
135  return PX4_ERROR;
136  }
137 
138  return PX4_ERROR;
139 }

160 int manufacture_date()
161 {
162  int result = -1;
163  uint16_t man_date = 0;
164 
165  for (uint8_t i = 0; i < NUM_BUS_OPTIONS; i++) {
166  struct batt_smbus_bus_option &bus = bus_options[i];
167 
168  if (bus.dev != nullptr) {
169  PX4_INFO("%s", bus.devpath);
170  result = bus.dev->manufacture_date(&man_date);
171  }
172  }
173 
174  if (PX4_OK == result) {
175  // Convert the uint16_t into human-readable date format.
176  uint16_t year = ((man_date >> 9) & 0xFF) + 1980;
177  uint8_t month = (man_date >> 5) & 0xF;
178  uint8_t day = man_date & 0x1F;
179  PX4_INFO("The manufacturer date is: %4d-%02d-%02d", year, month, day);
180  return PX4_OK;
181 
182  } else {
183  PX4_WARN("Unable to read the manufacturer date.");
184  return PX4_ERROR;
185  }

285 batt_smbus_main(int argc, char *argv[])
286 {
287  enum BATT_SMBUS_BUS busid = BATT_SMBUS_BUS_I2C_EXTERNAL;
288  int ch;
289 
290 
291  // Jump over start/off/etc and look at options first.
292  while ((ch = getopt(argc, argv, "XIA:")) != EOF) {
293  switch (ch) {
294  case 'X':
295  busid = BATT_SMBUS_BUS_I2C_EXTERNAL;
296  break;
297 
298  case 'I':
299  busid = BATT_SMBUS_BUS_I2C_INTERNAL;
300  break;
301 
302  case 'A':
303  busid = BATT_SMBUS_BUS_ALL;
304  break;
305 
306  default:
307  batt_smbus::usage("unrecognized argument");
308  return 0;
309  }
310  }
311 
312  const char *input = argv[optind];
313 
314  if(!input) {
315  batt_smbus::usage("Please enter an appropriate command.");
316  return 1;
317  }
318 
319  if (!strcmp(input, "start")) {
320  return batt_smbus::start(busid);
321  }
322 
323  struct batt_smbus::batt_smbus_bus_option &bus = batt_smbus::find_bus(busid);
324 
325  if (!strcmp(input, "stop")) {
326  delete bus.dev;
327  bus.dev = nullptr;
328  return 0;
329  }
330 
331  if (!strcmp(input, "suspend")) {
332  bus.dev->stop();
333  return 0;
334  }
335 
336  if (!strcmp(input, "resume")) {
337  bus.dev->start();
338  return 0;
339  }
340 
341  if (!strcmp(input, "man_name")) {
342  batt_smbus::manufacturer_name();
343  return 0;
344  }
345 
346  if (!strcmp(input, "man_date")) {
347  batt_smbus::manufacture_date();
348  return 0;
349  }
350 
351  if (!strcmp(input, "serial_num")) {
352  batt_smbus::serial_number();
353  return 0;
354  }
355 
356  if (!strcmp(input, "sbs_info")) {
357  batt_smbus::manufacturer_name();
358  batt_smbus::manufacture_date();
359  batt_smbus::serial_number();
360  return 0;
361  }
362 
363  if (!strcmp(input, "info")) {
364  bus.dev->info();
365  return 0;
366  }
367 
368  if (!strcmp(input, "unseal")) {
369  bus.dev->unseal();
370  return 0;
371  }
372 
373 
374 
375  if (!strcmp(argv[2], "read_word")) {
376  if (argv[3]) {
377  uint8_t data[2];
378 
379  if (PX4_OK != bus.dev->read_word((uint8_t)atoi(argv[3]), data)) {
380  PX4_INFO("Register read failed");
381  return 1;
382 
383  } else {
384  PX4_INFO("Register value: %d %d", data[1], data[0]);
385  return 0;
386  }
387  }
388 
389  return 1;
390  }
391 
392  // cmd code (u16), length (32bytes max)
393  if (!strcmp(argv[2], "man_read")) {
394  if (argv[3] && argv[4]) {
395  uint8_t data[32] = {0};
396  uint16_t cmd_code = atoi(argv[3]);
397  unsigned length = atoi(argv[4]);
398 
399  if (length > 32) {
400  PX4_WARN("Data length out of range: Max 32 bytes");
401  return 1;
402  }
403 
404  if (PX4_OK != bus.dev->manufacturer_read(cmd_code, data, length)) {
405  PX4_INFO("Block write failed");
406  return 1;
407 
408  } else {
409  return 0;
410  }
411  }
412 
413  return 1;
414  }
415 
416  // cmd code (u16), length (32bytes max)
417  if (!strcmp(argv[2], "block_read")) {
418  if (argv[3] && argv[4]) {
419  uint8_t data[32] = {0};
420  uint8_t cmd_code = atoi(argv[3]);
421  unsigned length = atoi(argv[4]);
422 
423  if (length > 32) {
424  PX4_WARN("Data length out of range: Max 32 bytes");
425  return 1;
426  }
427 
428  if (PX4_OK != bus.dev->block_read(cmd_code, data, length)) {
429  PX4_INFO("Block read failed");
430  return 1;
431 
432  } else {
433  for (unsigned i = 0; i < length; i++) {
434  PX4_INFO("%d", data[i]);
435  }
436 
437  return 0;
438  }
439  }
440 
441  return 1;
442  }
443 
444  // address, length, data1, data2, data3, etc
445  if (!strcmp(argv[2], "block_write")) {
446  if (argv[3] && argv[4]) {
447  uint8_t cmd_code = atoi(argv[3]);
448  unsigned length = atoi(argv[4]);
449  uint8_t tx_buf[32] = {0};
450 
451  if (length > 32) {
452  PX4_WARN("Data length out of range: Max 32 bytes");
453  return 1;
454  }
455 
456  // Data needs to be fed in 1 byte (0x01) at a time.
457  for (unsigned i = 0; i < length; i++) {
458  tx_buf[i] = atoi(argv[5 + i]);
459  PX4_INFO("Read in: %d", tx_buf[i]);
460  }
461 
462  if (PX4_OK != bus.dev->block_write(cmd_code, tx_buf, length)) {
463  PX4_INFO("Dataflash read failed");
464  return 1;
465 
466  } else {
467  return 0;
468  }
469  }
470 
471  return 1;
472  }
473

475  if (argv[3]) {
476  uint16_t address = atoi(argv[3]);
477  uint8_t rx_buf[32] = {0};
478 
479  if (PX4_OK != bus.dev->dataflash_read(address, rx_buf)) {
480  PX4_INFO("Dataflash read failed");

511 
512  } else {

515  }
516  }
517  }

519  batt_smbus::usage("unrecognized argument");
