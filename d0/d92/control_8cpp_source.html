52  calcMagDeclination();

54  // monitor the tilt alignment
55  if (!_control_status.flags.tilt_align) {

57  Vector3f angle_err_var_vec = calcRotVecVariances();
58

71 
72  } else if (_control_status.flags.gps_hgt) {

74 
75  } else if (_control_status.flags.rng_hgt) {

79  ECL_ERR("EKF aligned, (unknown height, IMU buf: %i, OBS buf: %i)", (int)_imu_buffer_length, (int)_obs_buffer_length);

82  }

162 
163  // if the ev data is not in a NED reference frame, then the transformation between EV and EKF navigation frames
164  // needs to be calculated and the observations rotated into the EKF frame of reference

172  && _control_status.flags.yaw_align) {
173 
174  // check for a exernal vision measurement that has fallen behind the fusion time horizon
175  if (_time_last_imu - _time_last_ext_vision < 2 * EV_MAX_INTERVAL) {
176  // turn on use of external vision measurements for position
177  _control_status.flags.ev_pos = true;
178  ECL_INFO("EKF commencing external vision position fusion");
179

181  // Reset transformation between EV and EKF navigation frames when starting fusion
182  resetExtVisRotMat();
183  }
184 
185  // reset the position if we are not already aiding using GPS, else use a relative position
186  // method for fusing the position data
187  if (_control_status.flags.gps) {
188  _fuse_hpos_as_odom = true;
189 
190  } else {
191  resetPosition();
192  resetVelocity();
193 
194  }
195  }
196  }
197 
198  // external vision yaw aiding selection logic
199  if (!_control_status.flags.gps && (_params.fusion_mode & MASK_USE_EVYAW) && !_control_status.flags.ev_yaw && _control_status.flags.tilt_align) {
200  // don't start using EV data unless daa is arriving frequently
201  if (_time_last_imu - _time_last_ext_vision < 2 * EV_MAX_INTERVAL) {
202  // reset the yaw angle to the value from the observaton quaternion
203  // get the roll, pitch, yaw estimates from the quaternion states
204  Quatf q_init(_state.quat_nominal);
205  Eulerf euler_init(q_init);
206 
207  // get initial yaw from the observation quaternion
208  const extVisionSample &ev_newest = _ext_vision_buffer.get_newest();
209  Quatf q_obs(ev_newest.quat);
210  Eulerf euler_obs(q_obs);
211  euler_init(2) = euler_obs(2);
212 
213  // save a copy of the quaternion state for later use in calculating the amount of reset change
214  Quatf quat_before_reset = _state.quat_nominal;
215 
216  // calculate initial quaternion states for the ekf
217  _state.quat_nominal = Quatf(euler_init);
218 
219  // calculate the amount that the quaternion has changed by
220  _state_reset_status.quat_change = quat_before_reset.inversed() * _state.quat_nominal;
221 
222  // add the reset amount to the output observer buffered data
223  // Note q1 *= q2 is equivalent to q1 = q2 * q1
224  for (uint8_t i = 0; i < _output_buffer.get_length(); i++) {
225  _output_buffer[i].quat_nominal *= _state_reset_status.quat_change;
226  }
227 
228  // apply the change in attitude quaternion to our newest quaternion estimate
229  // which was already taken out from the output buffer
230  _output_new.quat_nominal = _state_reset_status.quat_change * _output_new.quat_nominal;
231 
232  // capture the reset event
233  _state_reset_status.quat_counter++;
234 
235  // flag the yaw as aligned
236  _control_status.flags.yaw_align = true;
237 
238  // turn on fusion of external vision yaw measurements and disable all magnetoemter fusion
239  _control_status.flags.ev_yaw = true;
240  _control_status.flags.mag_hdg = false;
241  _control_status.flags.mag_3D = false;
242  _control_status.flags.mag_dec = false;
243 
244  ECL_INFO("EKF commencing external vision yaw fusion");
245  }
246  }
247 
248  // determine if we should start using the height observations
249  if (_params.vdist_sensor_type == VDIST_SENSOR_EV) {
250  // don't start using EV data unless data is arriving frequently
251  if (!_control_status.flags.ev_hgt && (_time_last_imu - _time_last_ext_vision < 2 * EV_MAX_INTERVAL)) {
252  setControlEVHeight();
253  resetHeight();
254  }
255  }
256 
257  // determine if we should use the vertical position observation
258  if (_control_status.flags.ev_hgt) {
259  _fuse_height = true;
260  }
261 
262  // determine if we should use the horizontal position observations
263  if (_control_status.flags.ev_pos) {
264  _fuse_pos = true;
265 
266  // correct position and height for offset relative to IMU
267  Vector3f pos_offset_body = _params.ev_pos_body - _params.imu_pos_body;
268  Vector3f pos_offset_earth = _R_to_earth * pos_offset_body;
269  _ev_sample_delayed.posNED(0) -= pos_offset_earth(0);
270  _ev_sample_delayed.posNED(1) -= pos_offset_earth(1);
271  _ev_sample_delayed.posNED(2) -= pos_offset_earth(2);
272 
273  // Use an incremental position fusion method for EV data if using GPS or if the observations are not in NED
274  if (_control_status.flags.gps || (_params.fusion_mode & MASK_ROTATE_EV)) {
275  _fuse_hpos_as_odom = true;
276 
277  } else {
278  _fuse_hpos_as_odom = false;
279  }
280 
281  if (_fuse_hpos_as_odom) {
282  if (!_hpos_prev_available) {
283  // no previous observation available to calculate position change
284  _fuse_pos = false;
285  _hpos_prev_available = true;
286 
287  } else {
288  // calculate the change in position since the last measurement
289  Vector3f ev_delta_pos = _ev_sample_delayed.posNED - _pos_meas_prev;
290 
291  // rotate measurement into body frame if required
292  if (_params.fusion_mode & MASK_ROTATE_EV) {
293  ev_delta_pos = _ev_rot_mat * ev_delta_pos;
294  }
295 
296  // use the change in position since the last measurement
297  _vel_pos_innov[3] = _state.pos(0) - _hpos_pred_prev(0) - ev_delta_pos(0);
298  _vel_pos_innov[4] = _state.pos(1) - _hpos_pred_prev(1) - ev_delta_pos(1);
299 
300  }
301 
302  // record observation and estimate for use next time
303  _pos_meas_prev = _ev_sample_delayed.posNED;
304  _hpos_pred_prev(0) = _state.pos(0);
305  _hpos_pred_prev(1) = _state.pos(1);
306 
307  } else {
308  // use the absolute position
309  _vel_pos_innov[3] = _state.pos(0) - _ev_sample_delayed.posNED(0);
310  _vel_pos_innov[4] = _state.pos(1) - _ev_sample_delayed.posNED(1);
311 
312  // check if we have been deadreckoning too long
313  if (_time_last_imu - _time_last_pos_fuse > _params.no_gps_timeout_max) {
314  // don't reset velocity if we have another source of aiding constraining it
315  if (_time_last_imu - _time_last_of_fuse > (uint64_t)1E6) {
316  resetVelocity();
317  }
318 
319  resetPosition();
320  }
321  }
322 
323  // observation 1-STD error
324  _posObsNoiseNE = fmaxf(_ev_sample_delayed.posErr, 0.01f);
325 
326  // innovation gate size
327  _posInnovGateNE = fmaxf(_params.ev_innov_gate, 1.0f);
328  }
329 
330  // Fuse available NED position data into the main filter
331  if (_fuse_height || _fuse_pos) {
332  fuseVelPosHeight();
333  _fuse_pos = _fuse_height = false;
334  _fuse_hpos_as_odom = false;
335 
336  }
337 
338  // determine if we should use the yaw observation
339  if (_control_status.flags.ev_yaw) {
340  fuseHeading();
341 
342  }
343 
344  } else if (_control_status.flags.ev_pos
345  && (_time_last_imu >= _time_last_ext_vision)
346  && (_time_last_imu - _time_last_ext_vision > (uint64_t)_params.no_gps_timeout_max)) {
347 
348  // Turn off EV fusion mode if no data has been received
349  _control_status.flags.ev_pos = false;
350  ECL_INFO("EKF External Vision Data Stopped");
351 
352  }

355 void Ekf::controlOpticalFlowFusion()
356 {
357  // Check if on ground motion is un-suitable for use of optical flow
358  if (!_control_status.flags.in_air) {
359  // When on ground check if the vehicle is being shaken or moved in a way that could cause a loss of navigation
360  const float accel_norm = _accel_vec_filt.norm();
361 
362  const bool motion_is_excessive = ((accel_norm > (CONSTANTS_ONE_G * 1.5f)) // upper g limit
363  || (accel_norm < (CONSTANTS_ONE_G * 0.5f)) // lower g limit
364  || (_ang_rate_mag_filt > _flow_max_rate) // angular rate exceeds flow sensor limit
365  || (_R_to_earth(2,2) < cosf(math::radians(30.0f)))); // tilted excessively
366 
367  if (motion_is_excessive) {
368  _time_bad_motion_us = _imu_sample_delayed.time_us;
369 
370  } else {
371  _time_good_motion_us = _imu_sample_delayed.time_us;
372  }
373 
374  } else {
375  _time_bad_motion_us = 0;
376  _time_good_motion_us = _imu_sample_delayed.time_us;
377  }
378 
379  // Accumulate autopilot gyro data across the same time interval as the flow sensor
380  _imu_del_ang_of += _imu_sample_delayed.delta_ang - _state.gyro_bias;
381  _delta_time_of += _imu_sample_delayed.delta_ang_dt;
382 
383  // New optical flow data is available and is ready to be fused when the midpoint of the sample falls behind the fusion time horizon
384  if (_flow_data_ready) {
385  // Inhibit flow use if motion is un-suitable or we have good quality GPS
386  // Apply hysteresis to prevent rapid mode switching
387  float gps_err_norm_lim;
388  if (_control_status.flags.opt_flow) {
389  gps_err_norm_lim = 0.7f;
390  } else {
391  gps_err_norm_lim = 1.0f;
392  }
393 
394  // Check if we are in-air and require optical flow to control position drift
395  bool flow_required = _control_status.flags.in_air &&
396  (_is_dead_reckoning // is doing inertial dead-reckoning so must constrain drift urgently
397  || (_control_status.flags.opt_flow && !_control_status.flags.gps && !_control_status.flags.ev_pos) // is completely reliant on optical flow
398  || (_control_status.flags.gps && (_gps_error_norm > gps_err_norm_lim))); // is using GPS, but GPS is bad
399 
400  if (!_inhibit_flow_use && _control_status.flags.opt_flow) {
401  // inhibit use of optical flow if motion is unsuitable and we are not reliant on it for flight navigation
402  bool preflight_motion_not_ok = !_control_status.flags.in_air && ((_imu_sample_delayed.time_us - _time_good_motion_us) > (uint64_t)1E5);
403  bool flight_motion_not_ok = _control_status.flags.in_air && !_range_aid_mode_enabled;
404  if ((preflight_motion_not_ok || flight_motion_not_ok) && !flow_required) {
405  _inhibit_flow_use = true;
406  }
407  } else if (_inhibit_flow_use && !_control_status.flags.opt_flow){
408  // allow use of optical flow if motion is suitable or we are reliant on it for flight navigation
409  bool preflight_motion_ok = !_control_status.flags.in_air && ((_imu_sample_delayed.time_us - _time_bad_motion_us) > (uint64_t)5E6);
410  bool flight_motion_ok = _control_status.flags.in_air && _range_aid_mode_enabled;
411  if (preflight_motion_ok || flight_motion_ok || flow_required) {
412  _inhibit_flow_use = false;
413  }
414  }
415 
416  // Handle cases where we are using optical flow but are no longer able to because data is old
417  // or its use has been inhibited.
418  if (_control_status.flags.opt_flow) {
419  if (_inhibit_flow_use) {
420  _control_status.flags.opt_flow = false;
421  _time_last_of_fuse = 0;
422 
423  } else if (_time_last_imu - _flow_sample_delayed.time_us > (uint64_t)_params.no_gps_timeout_max) {
424  _control_status.flags.opt_flow = false;
425 
426  }
427  }
428 
429  // optical flow fusion mode selection logic
430  if ((_params.fusion_mode & MASK_USE_OF) // optical flow has been selected by the user
431  && !_control_status.flags.opt_flow // we are not yet using flow data
432  && _control_status.flags.tilt_align // we know our tilt attitude
433  && !_inhibit_flow_use
434  && get_terrain_valid()) // we have a valid distance to ground estimate
435  {
436  // If the heading is not aligned, reset the yaw and magnetic field states
437  if (!_control_status.flags.yaw_align) {
438  _control_status.flags.yaw_align = resetMagHeading(_mag_sample_delayed.mag);
439  }
440 
441  // If the heading is valid and use is not inhibited , start using optical flow aiding
442  if (_control_status.flags.yaw_align) {
443  // set the flag and reset the fusion timeout
444  _control_status.flags.opt_flow = true;
445  _time_last_of_fuse = _time_last_imu;
446 
447  // if we are not using GPS then the velocity and position states and covariances need to be set
448  if (!_control_status.flags.gps || !_control_status.flags.ev_pos) {
449  resetVelocity();
450  resetPosition();
451 
452  // align the output observer to the EKF states
453  alignOutputFilter();
454 
455  }
456  }
457 
458  } else if (!(_params.fusion_mode & MASK_USE_OF)) {
459  _control_status.flags.opt_flow = false;
460 
461  }
462 
463  // handle the case when we have optical flow, are reliant on it, but have not been using it for an extended period
464  if (_control_status.flags.opt_flow
465  && !_control_status.flags.gps
466  && !_control_status.flags.ev_pos) {
467 
468  bool do_reset = _time_last_imu - _time_last_of_fuse > _params.no_gps_timeout_max;
469 
470  if (do_reset) {
471  resetVelocity();
472  resetPosition();
473  }
474  }
475 
476  // Only fuse optical flow if valid body rate compensation data is available
477  if (calcOptFlowBodyRateComp()) {
478 
479  bool flow_quality_good = (_flow_sample_delayed.quality >= _params.flow_qual_min);
480 
481  if (!flow_quality_good && !_control_status.flags.in_air) {
482  // when on the ground with poor flow quality, assume zero ground relative velocity and LOS rate
483  _flowRadXYcomp.zero();
484  } else {
485  // compensate for body motion to give a LOS rate
486  _flowRadXYcomp(0) = _flow_sample_delayed.flowRadXY(0) - _flow_sample_delayed.gyroXYZ(0);
487  _flowRadXYcomp(1) = _flow_sample_delayed.flowRadXY(1) - _flow_sample_delayed.gyroXYZ(1);
488  }
489  } else {
490  // don't use this flow data and wait for the next data to arrive
491  _flow_data_ready = false;
492  }
493  }
494 
495  // Wait until the midpoint of the flow sample has fallen behind the fusion time horizon
496  if (_flow_data_ready && (_imu_sample_delayed.time_us > _flow_sample_delayed.time_us - uint32_t(1e6f * _flow_sample_delayed.dt) / 2)) {
497  // Fuse optical flow LOS rate observations into the main filter only if height above ground has been updated recently
498  // but use a relaxed time criteria to enable it to coast through bad range finder data
499  if (_control_status.flags.opt_flow && (_time_last_imu - _time_last_hagl_fuse < (uint64_t)10e6)) {
500  fuseOptFlow();
501  _last_known_posNE(0) = _state.pos(0);
502  _last_known_posNE(1) = _state.pos(1);
503  }
504 
505  _flow_data_ready = false;
506  }
507 }

509 void Ekf::controlGpsFusion()
510 {
511  // Check for new GPS data that has fallen behind the fusion time horizon
512  if (_gps_data_ready) {
513 
514  // Determine if we should use GPS aiding for velocity and horizontal position
515  // To start using GPS we need angular alignment completed, the local NED origin set and GPS data that has not failed checks recently
516  bool gps_checks_passing = (_time_last_imu - _last_gps_fail_us > (uint64_t)5e6);
517  bool gps_checks_failing = (_time_last_imu - _last_gps_pass_us > (uint64_t)5e6);

521  // Do not use external vision for yaw if using GPS because yaw needs to be
522  // defined relative to an NED reference frame
523  if (!_control_status.flags.yaw_align || _control_status.flags.ev_yaw || _mag_inhibit_yaw_reset_req) {
524  _control_status.flags.yaw_align = false;
525  _control_status.flags.ev_yaw = false;
526  _control_status.flags.yaw_align = resetMagHeading(_mag_sample_delayed.mag);
527  // Handle the special case where we have not been constraining yaw drift or learning yaw bias due
528  // to assumed invalid mag field associated with indoor operation with a downwards looking flow sensor.
529  if (_mag_inhibit_yaw_reset_req) {
530  _mag_inhibit_yaw_reset_req = false;
531  // Zero the yaw bias covariance and set the variance to the initial alignment uncertainty
532  setDiag(P, 12, 12, sq(_params.switch_on_gyro_bias * FILTER_UPDATE_PERIOD_S));
533  }
534  }
535 
536  // If the heading is valid start using gps aiding
537  if (_control_status.flags.yaw_align) {
538  // if we are not already aiding with optical flow, then we need to reset the position and velocity
539  // otherwise we only need to reset the position
540  _control_status.flags.gps = true;
541 
542  if (!_control_status.flags.opt_flow) {
543  if (!resetPosition() || !resetVelocity()) {
544  _control_status.flags.gps = false;
545 
546  }
547 
548  } else if (!resetPosition()) {
549  _control_status.flags.gps = false;
550 
551  }
552 
553  if (_control_status.flags.gps) {
554  ECL_INFO("EKF commencing GPS fusion");
555  _time_last_gps = _time_last_imu;
556  }
557  }
558  }
559 
560  } else if (!(_params.fusion_mode & MASK_USE_GPS)) {
561  _control_status.flags.gps = false;
562 
563  }
564 
565  // Handle the case where we are using GPS and another source of aiding and GPS is failing checks
566  if (_control_status.flags.gps && gps_checks_failing && (_control_status.flags.opt_flow || _control_status.flags.ev_pos)) {
567  _control_status.flags.gps = false;
568  ECL_WARN("EKF GPS data quality poor - stopping use");
569  }
570 
571  // handle the case when we now have GPS, but have not been using it for an extended period
572  if (_control_status.flags.gps) {
573  // We are relying on aiding to constrain drift so after a specified time
574  // with no aiding we need to do something
575  bool do_reset = (_time_last_imu - _time_last_pos_fuse > _params.no_gps_timeout_max)
576  && (_time_last_imu - _time_last_delpos_fuse > _params.no_gps_timeout_max)
577  && (_time_last_imu - _time_last_vel_fuse > _params.no_gps_timeout_max)
578  && (_time_last_imu - _time_last_of_fuse > _params.no_gps_timeout_max);
579 
580  // We haven't had an absolute position fix for a longer time so need to do something
581  do_reset = do_reset || (_time_last_imu - _time_last_pos_fuse > 2 * _params.no_gps_timeout_max);
582 
583  if (do_reset) {
584  // use GPS velocity data to check and correct yaw angle if a FW vehicle
585  if (_control_status.flags.fixed_wing && _control_status.flags.in_air) {
586  // if flying a fixed wing aircraft, do a complete reset that includes yaw
587  realignYawGPS();
588  }
589 
590  resetVelocity();
591  resetPosition();
592  _velpos_reset_request = false;
593  ECL_WARN("EKF GPS fusion timeout - reset to GPS");
594 
595  // Reset the timeout counters
596  _time_last_pos_fuse = _time_last_imu;
597  _time_last_vel_fuse = _time_last_imu;
598 
599  }
600  }
601 
602  // Only use GPS data for position and velocity aiding if enabled
603  if (_control_status.flags.gps) {
604  _fuse_pos = true;
605  _fuse_vert_vel = true;
606  _fuse_hor_vel = true;
607

609  Vector3f ang_rate = _imu_sample_delayed.delta_ang * (1.0f / _imu_sample_delayed.delta_ang_dt);
610  Vector3f pos_offset_body = _params.gps_pos_body - _params.imu_pos_body;
611  Vector3f vel_offset_body = cross_product(ang_rate, pos_offset_body);
612  Vector3f vel_offset_earth = _R_to_earth * vel_offset_body;
613  _gps_sample_delayed.vel -= vel_offset_earth;
614 
615  // correct position and height for offset relative to IMU
616  Vector3f pos_offset_earth = _R_to_earth * pos_offset_body;
617  _gps_sample_delayed.pos(0) -= pos_offset_earth(0);
618  _gps_sample_delayed.pos(1) -= pos_offset_earth(1);
619  _gps_sample_delayed.hgt += pos_offset_earth(2);
620

622  float lower_limit = fmaxf(_params.gps_pos_noise, 0.01f);
623 
624  if (_control_status.flags.opt_flow || _control_status.flags.ev_pos) {
625  // if we are using other sources of aiding, then relax the upper observation
626  // noise limit which prevents bad GPS perturbing the position estimate
627  _posObsNoiseNE = fmaxf(_gps_sample_delayed.hacc, lower_limit);
628 
629  } else {
630  // if we are not using another source of aiding, then we are reliant on the GPS
631  // observations to constrain attitude errors and must limit the observation noise value.
632  float upper_limit = fmaxf(_params.pos_noaid_noise, lower_limit);
633  _posObsNoiseNE = math::constrain(_gps_sample_delayed.hacc, lower_limit, upper_limit);
634  }
635 
636  _velObsVarNE(1) = _velObsVarNE(0) = sq(fmaxf(_gps_sample_delayed.sacc, _params.gps_vel_noise));
637 
638  // calculate innovations
639  _vel_pos_innov[0] = _state.vel(0) - _gps_sample_delayed.vel(0);
640  _vel_pos_innov[1] = _state.vel(1) - _gps_sample_delayed.vel(1);
641  _vel_pos_innov[2] = _state.vel(2) - _gps_sample_delayed.vel(2);
642  _vel_pos_innov[3] = _state.pos(0) - _gps_sample_delayed.pos(0);
643  _vel_pos_innov[4] = _state.pos(1) - _gps_sample_delayed.pos(1);
644 
645  // set innovation gate size
646  _posInnovGateNE = fmaxf(_params.posNE_innov_gate, 1.0f);
647  _hvelInnovGate = fmaxf(_params.vel_innov_gate, 1.0f);
648  }
649 
650  } else if (_control_status.flags.gps && (_imu_sample_delayed.time_us - _gps_sample_delayed.time_us > (uint64_t)10e6)) {
651  _control_status.flags.gps = false;
652  ECL_WARN("EKF GPS data stopped");
653  }

659  * Handle the case where we have not fused height measurements recently and
660  * uncertainty exceeds the max allowable. Reset using the best available height
661  * measurement source, continue using it after the reset and declare the current
662  * source failed if we have switched.
663  */
664 
665  // Check for IMU accelerometer vibration induced clipping as evidenced by the vertical innovations being positive and not stale.
666  // Clipping causes the average accel reading to move towards zero which makes the INS think it is falling and produces positive vertical innovations
667  float var_product_lim = sq(_params.vert_innov_test_lim) * sq(_params.vert_innov_test_lim);
668  bool bad_vert_accel = (_control_status.flags.baro_hgt && // we can only run this check if vertical position and velocity observations are indepedant
669  (sq(_vel_pos_innov[5] * _vel_pos_innov[2]) > var_product_lim * (_vel_pos_innov_var[5] * _vel_pos_innov_var[2])) && // vertical position and velocity sensors are in agreement that we have a significant error
670  (_vel_pos_innov[2] > 0.0f) && // positive innovation indicates that the inertial nav thinks it is falling
671  ((_imu_sample_delayed.time_us - _baro_sample_delayed.time_us) < 2 * BARO_MAX_INTERVAL) && // vertical position data is fresh
672  ((_imu_sample_delayed.time_us - _gps_sample_delayed.time_us) < 2 * GPS_MAX_INTERVAL)); // vertical velocity data is fresh
673 
674  // record time of last bad vert accel
675  if (bad_vert_accel) {
676  _time_bad_vert_accel = _time_last_imu;
677 
678  } else {
679  _time_good_vert_accel = _time_last_imu;
680  }
681 
682  // declare a bad vertical acceleration measurement and make the declaration persist
683  // for a minimum of 10 seconds
684  if (_bad_vert_accel_detected) {
685  _bad_vert_accel_detected = (_time_last_imu - _time_bad_vert_accel < BADACC_PROBATION);

696

698  if (hgt_fusion_timeout || continuous_bad_accel_hgt) {

704  // check if GPS height is available
705  const gpsSample &gps_init = _gps_buffer.get_newest();

717 
718  // reset to GPS if GPS data is available and there is no Baro data
719  reset_to_gps = reset_to_gps || (gps_hgt_available && !baro_hgt_available);

752  reset_height = false;
753

755  }

759  // check if GPS height is available
