135 {
136  syslink_ow_read_t *data = (syslink_ow_read_t *) &msgbuf.data;
137  msgbuf.type = SYSLINK_OW_READ;
138 
139  int nread = 0;
140 
141  while (nread < length) {
142 
143  msgbuf.length = 3;
144  data->idx = i;
145  data->addr = addr;
146  sendAndWait();
147 
148  // Number of bytes actually read
149  int n = MIN(length - nread, msgbuf.length - 3);
150 
151  if (n == 0) {
152  break;
153  }
154 
155  memcpy(buf, data->data, n);
156  nread += n;
157  buf += n;
158  addr += n;
159  }
160

162 }
