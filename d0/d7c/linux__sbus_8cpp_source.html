38 int RcInput::init()
39 {
40  int i;
41 
42  /**
43  * initialize the data of each channel
44  */
45  for (i = 0; i < input_rc_s::RC_INPUT_MAX_CHANNELS; ++i) {
46  _data.values[i] = UINT16_MAX;
47  }
48 
49  _rcinput_pub = orb_advertise(ORB_ID(input_rc), &_data);
50 
51  if (nullptr == _rcinput_pub) {
52  PX4_WARN("error: advertise failed");
53  return -1;
54  }
55 
56  /**
57  * open the serial port
58  */
59  _device_fd = open(_device, O_RDWR | O_NONBLOCK | O_CLOEXEC);
60 
61  if (-1 == _device_fd) {
62  PX4_ERR("Open SBUS input %s failed, status %d \n", _device,
63  (int) _device_fd);
64  fflush(stdout);
65  return -1;
66  }
67 
68  struct termios2 tio { };
69 
70  if (0 != ioctl(_device_fd, TCGETS2, &tio)) {
71  close(_device_fd);
72  _device_fd = -1;
73  return -1;
74  }
75 
76  /**
77  * Setting serial port,8E2, non-blocking.100Kbps
78  */
79  tio.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR | IGNCR | ICRNL
80  | IXON);
81  tio.c_iflag |= (INPCK | IGNPAR);
82  tio.c_oflag &= ~OPOST;
83  tio.c_lflag &= ~(ECHO | ECHONL | ICANON | ISIG | IEXTEN);
84  tio.c_cflag &= ~(CSIZE | CRTSCTS | PARODD | CBAUD);
85  /**
86  * use BOTHER to specify speed directly in c_[io]speed member
87  */
88  tio.c_cflag |= (CS8 | CSTOPB | CLOCAL | PARENB | BOTHER | CREAD);
89  tio.c_ispeed = 100000;
90  tio.c_ospeed = 100000;
91  tio.c_cc[VMIN] = 25;
92  tio.c_cc[VTIME] = 0;
93 
94  if (0 != ioctl(_device_fd, TCSETS2, &tio)) {
95  close(_device_fd);
96  _device_fd = -1;
97  return -1;
98  }
99 
100  return 0;
101 }

103 int RcInput::start(char *device, int channels)
104 {
105  int result = 0;
106  strcpy(_device, device);
107  PX4_INFO("Device %s , channels: %d \n", device, channels);
108  _channels = channels;
109  result = init();
110 
111  if (0 != result) {
112  PX4_WARN("error: RC initialization failed");
113  return -1;
114  }
115 
116  _isRunning = true;
117  result = work_queue(HPWORK, &_work, (worker_t) & RcInput::cycle_trampoline,
118  this, 0);
119 
120  if (result == -1) {
121  _isRunning = false;
122  }
123 
124  return result;
125 }

149 void RcInput::_measure(void)
150 {
151  uint64_t ts;
152  int nread;
153  fd_set fds;
154  FD_ZERO(&fds);
155  FD_SET(_device_fd, &fds);
156  /**
157  *error counter to count the lost frame
158  */
159  int count = 0; //
160 
161  while (1) {
162  nread = read(_device_fd, &_sbusData, sizeof(_sbusData));
163 
164  if (25 == nread) {
165  /**
166  * Notice: most sbus rx device support sbus1
167  */
168  if (0x0f == _sbusData[0] && 0x00 == _sbusData[24]) {
169  break;
170  }
171  }
172 
173  ++count;
174  usleep(RCINPUT_MEASURE_INTERVAL_US);
175  }
176 
177  /**
178  * parse sbus data to pwm
179  */
180  _channels_data[0] =
181  (uint16_t)(((_sbusData[1] | _sbusData[2] << 8) & 0x07FF)
182  * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET;
183  _channels_data[1] = (uint16_t)(((_sbusData[2] >> 3 | _sbusData[3] << 5)
184  & 0x07FF) * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET;
185  _channels_data[2] = (uint16_t)(((_sbusData[3] >> 6 | _sbusData[4] << 2
186  | _sbusData[5] << 10) & 0x07FF) * SBUS_SCALE_FACTOR + .5f)
187  + SBUS_SCALE_OFFSET;
188  _channels_data[3] = (uint16_t)(((_sbusData[5] >> 1 | _sbusData[6] << 7)
189  & 0x07FF) * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET;
190  _channels_data[4] = (uint16_t)(((_sbusData[6] >> 4 | _sbusData[7] << 4)
191  & 0x07FF) * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET;
192  _channels_data[5] = (uint16_t)(((_sbusData[7] >> 7 | _sbusData[8] << 1
193  | _sbusData[9] << 9) & 0x07FF) * SBUS_SCALE_FACTOR + .5f)
194  + SBUS_SCALE_OFFSET;
195  _channels_data[6] = (uint16_t)(((_sbusData[9] >> 2 | _sbusData[10] << 6)
196  & 0x07FF) * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET;
197  _channels_data[7] = (uint16_t)(((_sbusData[10] >> 5 | _sbusData[11] << 3)
198  & 0x07FF) * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET; // & the other 8 + 2 channels if you need them
199  _channels_data[8] = (uint16_t)(((_sbusData[12] | _sbusData[13] << 8)
200  & 0x07FF) * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET;
201  _channels_data[9] = (uint16_t)(((_sbusData[13] >> 3 | _sbusData[14] << 5)
202  & 0x07FF) * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET;
203  _channels_data[10] = (uint16_t)(((_sbusData[14] >> 6 | _sbusData[15] << 2
204  | _sbusData[16] << 10) & 0x07FF) * SBUS_SCALE_FACTOR + .5f)
205  + SBUS_SCALE_OFFSET;
206  _channels_data[11] = (uint16_t)(((_sbusData[16] >> 1 | _sbusData[17] << 7)
207  & 0x07FF) * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET;
208  _channels_data[12] = (uint16_t)(((_sbusData[17] >> 4 | _sbusData[18] << 4)
209  & 0x07FF) * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET;
210  _channels_data[13] = (uint16_t)(((_sbusData[18] >> 7 | _sbusData[19] << 1
211  | _sbusData[20] << 9) & 0x07FF) * SBUS_SCALE_FACTOR + .5f)
212  + SBUS_SCALE_OFFSET;
213  _channels_data[14] = (uint16_t)(((_sbusData[20] >> 2 | _sbusData[21] << 6)
214  & 0x07FF) * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET;
215  _channels_data[15] = (uint16_t)(((_sbusData[21] >> 5 | _sbusData[22] << 3)
216  & 0x07FF) * SBUS_SCALE_FACTOR + .5f) + SBUS_SCALE_OFFSET;
217  int i = 0;
218 
219  for (i = 0; i < _channels; ++i) {
220  _data.values[i] = _channels_data[i];
221  }
222 
223  ts = hrt_absolute_time();
224  _data.timestamp = ts;
225  _data.timestamp_last_signal = ts;
226  _data.channel_count = _channels;
227  _data.rssi = 100;
228  _data.rc_lost_frame_count = count;
229  _data.rc_total_frame_count = 1;
230  _data.rc_ppm_frame_length = 0;
231  _data.rc_failsafe = (_sbusData[23] & (1 << 3)) ? true : false;
232  _data.rc_lost = (_sbusData[23] & (1 << 2)) ? true : false;
233  _data.input_source = input_rc_s::RC_INPUT_SOURCE_PX4IO_SBUS;
234 
235  orb_publish(ORB_ID(input_rc), _rcinput_pub, &_data);
236 }

250 int linux_sbus_main(int argc, char **argv)
251 {
252  int start;
253  int command = -1;
254  /**
255  * ttyS1 for default, it can be changed through -d parameter
256  */
257  char device[30] = "/dev/ttyS1";
258  /**
259  * 8 channel for default setting, it can be changed through -c parameter
260  */
261  int max_channel = 8;
262 
263  /**
264  * Parse command line and get device and channels count from consolex
265  */
266  for (start = 0; start < argc; ++start) {
267  if (0 == strcmp(argv[start], "start")) {
268  command = 0;
269  continue;
270  }
271 
272  if (0 == strcmp(argv[start], "stop")) {
273  command = 1;
274  continue;
275  }
276 
277  if (0 == strcmp(argv[start], "status")) {
278  command = 2;
279  continue;
280  }
281 
282  if (0 == strcmp(argv[start], "-d")) {
283  if (argc > (start + 1)) {
284  strcpy(device, argv[start + 1]);
285  }
286 
287  continue;
288  }
289 
290  if (0 == strcmp(argv[start], "-c")) {
291  if (argc > (start + 1)) {
292  max_channel = atoi(argv[start + 1]);
293  }
294 
295  continue;
296  }
297  }
298 
299  /**
300  * Channels count can't be higher than 16;
301  */
302  max_channel = (max_channel > 16) ? 16 : max_channel;
303 
304  if (0 == command) {
305  if (nullptr != rc_input && rc_input->isRunning()) {
306  PX4_WARN("running");
307  return 0;
308  }
309 
310  rc_input = new RcInput();
311 
312  /** Check if alloc worked. */
313  if (nullptr == rc_input) {
314  PX4_ERR("Sbus driver initialization failed");
315  return -1;
316  }
317 
318  int ret = rc_input->start(device, max_channel);
319 
320  if (ret != 0) {
321  PX4_ERR("Linux sbus module failure");
322  }
323 
324  return 0;
325  }
326 
327  if (1 == command) {
328  if (rc_input == nullptr || !rc_input->isRunning()) {
329  PX4_WARN("Not running \n");
330  /* this is not an error */
331  return 0;
332  }
333 
334  rc_input->stop();
335  /**
336  * Wait for task to die
337  */
338  int i = 0;
339 
340  do {
341  /* wait up to 100ms */
342  usleep(100000);
343  } while (rc_input->isRunning() && ++i < 30);
344 
345  delete rc_input;
346  rc_input = nullptr;
347  return 0;
348  }
349 
350  if (2 == command) {
351  if (rc_input != nullptr && rc_input->isRunning()) {
352  PX4_INFO("running");
353 
354  } else {
355  PX4_INFO("Not running \n");
356  }
357 
358  return 0;
359  }
360 
361  linux_sbus::usage(
362  "Usage: linux_sbus start|stop|status -d <device> -c <channel>");
363  return 0;
364 }
