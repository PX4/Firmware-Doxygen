85 int BATT_SMBUS::block_read(const uint8_t cmd_code, void *data, const unsigned length)
86 {
87  unsigned byte_count = 0;
88  // Length of data (32max). byte_count(1), cmd_code(2), pec(1)
89  uint8_t rx_data[DATA_BUFFER_SIZE + 4];
90 
91  // If this is a ManufacturerBlockAccess() command then the first 2 data bytes will be the cmd_code.
92  if (cmd_code == BATT_SMBUS_MANUFACTURER_BLOCK_ACCESS) {
93  _interface->read(cmd_code, rx_data, length + 4);
94 
95  byte_count = rx_data[0];
96 
97  // addr1, addr2, byte_count,
98  memcpy(data, &rx_data[3], byte_count);
99 
100  } else {
101  // byte_count(1) + data[32max] + pec(1)
102  _interface->read(cmd_code, rx_data, length + 2);
103 
104  byte_count = rx_data[0];
105 
106  memcpy(data, &rx_data[1], byte_count);
107  }
108 
109  // addr(wr), cmd_code, addr(r), byte_count, rx_data[]
110  uint8_t device_address = get_device_address();
111  uint8_t full_data_packet[DATA_BUFFER_SIZE + 4] = {0};
112 
113  full_data_packet[0] = (device_address << 1) | 0x00;
114  full_data_packet[1] = cmd_code;
115  full_data_packet[2] = (device_address << 1) | 0x01;
116  full_data_packet[3] = byte_count;
117 
118  memcpy(&full_data_packet[4], &rx_data[1], byte_count);
119 
120  uint8_t pec = get_pec(full_data_packet, byte_count + 4);
121 
122  // First byte is byte count, followed by data.
123  if (pec != ((uint8_t *)rx_data)[byte_count + 1]) {
124  PX4_INFO("bad PEC from block_read");
125  return PX4_ERROR;
126 
127  } else {
128  return PX4_OK;
129  }
130 }

132 int BATT_SMBUS::block_write(const uint8_t cmd_code, void *data, const unsigned byte_count)
133 {
134  // cmd code, byte count, data[byte_count], pec
135  uint8_t buf[byte_count + 2];
136 
137  buf[0] = cmd_code;
138  buf[1] = (uint8_t)byte_count;
139  memcpy(&buf[2], data, byte_count);
140 
141  uint8_t pec = get_pec(buf, sizeof(buf));
142  buf[byte_count + 2] = pec;
143 
144  unsigned i = 0;
145 
146  // If block_write fails, try up to 10 times.
147  while (i < 10) {
148  if (PX4_OK != _interface->write(0, buf, sizeof(buf))) {
149  i++;
150  PX4_WARN("block_write failed: %d", i);
151  usleep(100000);
152 
153  } else {
154  return PX4_OK;
155  }
156  }
157 
158  return PX4_ERROR;
159 }

161 void BATT_SMBUS::cycle()
162 {
163  // Get the current time.
164  uint64_t now = hrt_absolute_time();
165 
166  // Read data from sensor.
167  battery_status_s new_report = {};
168 
169  // Set time of reading.
170  new_report.timestamp = now;
171 
172  // Don't publish if any reads fail.
173  bool success = true;
174 
175  // Temporary variable for storing SMBUS reads.
176  uint16_t tmp;
177 
178  if (read_word(BATT_SMBUS_VOLTAGE, &tmp) == PX4_OK) {
179 
180  new_report.connected = true;
181 
182  // Convert millivolts to volts.
183  new_report.voltage_v = ((float)tmp) / 1000.0f;
184  new_report.voltage_filtered_v = new_report.voltage_v;
185 
186  // Read current.
187  if (read_word(BATT_SMBUS_CURRENT, &tmp) == PX4_OK) {
188  new_report.current_a = (-1.0f * ((float)(*(int16_t *)&tmp)) / 1000.0f);
189  new_report.current_filtered_a = new_report.current_a;
190 
191  } else {
192  success = false;
193  }
194 
195  // Read average current.
196  if (read_word(BATT_SMBUS_AVERAGE_CURRENT, &tmp) == PX4_OK) {
197  new_report.average_current_a = (-1.0f * ((float)(*(int16_t *)&tmp)) / 1000.0f);
198 
199  } else {
200  success = false;
201  }
202 
203  // Read run time to empty.
204  if (read_word(BATT_SMBUS_RUN_TIME_TO_EMPTY, &tmp) == PX4_OK) {
205  new_report.run_time_to_empty = tmp;
206 
207  } else {
208  success = false;
209  }
210 
211  // Read average time to empty.
212  if (read_word(BATT_SMBUS_AVERAGE_TIME_TO_EMPTY, &tmp) == PX4_OK) {
213  new_report.average_time_to_empty = tmp;
214 
215  } else {
216  success = false;
217  }
218 
219  // Read remaining capacity.
220  if (read_word(BATT_SMBUS_REMAINING_CAPACITY, &tmp) == PX4_OK) {
221 
222  if (tmp > _batt_capacity) {
223  PX4_WARN("Remaining capacity greater than total: Capacity:%hu \tRemaining Capacity:%hu",
224  (uint16_t)_batt_capacity, (uint16_t)tmp);
225  _batt_capacity = (uint16_t)tmp;
226  }
227 
228  // Calculate remaining capacity percent with complementary filter
229  new_report.remaining = ((float)_last_report.remaining * 0.8f) + (0.2f * (1.0f -
230  (((float)_batt_capacity - (float)tmp) / (float)_batt_capacity)));
231 
232  // Calculate total discharged amount.
233  new_report.discharged_mah = (float)_batt_startup_capacity - (float)tmp;
234 
235  // Check if remaining % is out of range.
236  if ((new_report.remaining > 1.00f) || (new_report.remaining <= 0.00f)) {
237  new_report.warning = battery_status_s::BATTERY_WARNING_EMERGENCY;
238  PX4_INFO("Percent out of range: %4.2f", (double)new_report.remaining);
239  }
240 
241  // Check if discharged amount is greater than the starting capacity.
242  else if (new_report.discharged_mah > (float)_batt_startup_capacity) {
243  new_report.warning = battery_status_s::BATTERY_WARNING_EMERGENCY;
244  PX4_INFO("Discharged greater than startup capacity: %4.2f", (double)new_report.discharged_mah);
245  }
246 
247  // Propagate warning state.
248  else {
249  if (new_report.remaining > _low_thr) {
250  new_report.warning = battery_status_s::BATTERY_WARNING_NONE;
251 
252  } else if (new_report.remaining > _crit_thr) {
253  new_report.warning = battery_status_s::BATTERY_WARNING_LOW;
254 
255  } else if (new_report.remaining > _emergency_thr) {
256  new_report.warning = battery_status_s::BATTERY_WARNING_CRITICAL;
257 
258  } else {
259  uint64_t timer = hrt_absolute_time() - now;
260  new_report.warning = battery_status_s::BATTERY_WARNING_EMERGENCY;
261 
262  /* Only warn every 5 seconds */
263  if (timer > 5000000) {
264  PX4_WARN("Battery Warning Emergency: %4.2f", (double)new_report.remaining);
265  }
266  }
267  }
268 
269  } else {
270  success = false;
271  }
272 
273  // Read battery temperature and covert to Celsius.
274  if (read_word(BATT_SMBUS_TEMP, &tmp) == PX4_OK) {
275  new_report.temperature = (float)(((float)tmp / 10.0f) + CONSTANTS_ABSOLUTE_NULL_CELSIUS);
276 
277  } else {
278  success = false;
279  }
280 
281  new_report.capacity = _batt_capacity;
282  new_report.cycle_count = _cycle_count;
283  new_report.serial_number = _serial_number;
284 
285  // Publish to orb.
286  if (_batt_topic != nullptr && success) {
287 
288  orb_publish(_batt_orb_id, _batt_topic, &new_report);
289 
290  // Copy report for test().
291  _last_report = new_report;
292 
293 
294  } else {
295  _batt_topic = orb_advertise(_batt_orb_id, &new_report);
296 
297  if (_batt_topic == nullptr) {
298  PX4_ERR("ADVERT FAIL");
299  return;
300  }
301  }
302 
303  }
304 
305  // Schedule a fresh cycle call when the measurement is done.
306  work_queue(HPWORK, &_work, (worker_t)&BATT_SMBUS::cycle_trampoline, this,
307  USEC2TICK(BATT_SMBUS_MEASUREMENT_INTERVAL_US));
308 }

351 uint8_t BATT_SMBUS::get_pec(uint8_t *buff, const uint8_t len)
352 {
353  // Initialise CRC to zero.
354  uint8_t crc = 0;
355  uint8_t shift_register = 0;
356  bool invert_crc;
357 
358  // Calculate crc for each byte in the stream.
359  for (uint8_t i = 0; i < len; i++) {
360  // Load next data byte into the shift register
361  shift_register = buff[i];
362 
363  // Calculate crc for each bit in the current byte.
364  for (uint8_t j = 0; j < 8; j++) {
365  invert_crc = (crc ^ shift_register) & 0x80;
366  crc <<= 1;
367  shift_register <<= 1;
368 
369  if (invert_crc) {
370  crc ^= BATT_SMBUS_PEC_POLYNOMIAL;
371  }
372  }
373  }
374 
375  return crc;
376 }
