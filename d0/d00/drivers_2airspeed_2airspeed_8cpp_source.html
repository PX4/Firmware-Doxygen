95 int
96 Airspeed::init()
97 {
98  /* do I2C init (and probe) first */
99  if (I2C::init() != PX4_OK) {
100  return PX4_ERROR;
101  }
102 
103  /* register alternate interfaces if we have to */
104  _class_instance = register_class_devname(AIRSPEED_BASE_DEVICE_PATH);
105 
106  /* advertise sensor topic, measure manually to initialize valid report */
107  measure();
108  differential_pressure_s arp = {};
109 
110  /* measurement will have generated a report, publish */
111  _airspeed_pub = orb_advertise_multi(ORB_ID(differential_pressure), &arp, &_airspeed_orb_class_instance,
112  ORB_PRIO_HIGH - _class_instance);
113 
114  if (_airspeed_pub == nullptr) {
115  PX4_WARN("uORB started?");
116  }
117 
118  return PX4_OK;
119 }

136 int
137 Airspeed::ioctl(device::file_t *filp, int cmd, unsigned long arg)
138 {
139  switch (cmd) {
140 
141  case SENSORIOCSPOLLRATE: {
142  switch (arg) {
143 
144  /* switching to manual polling */
145  case SENSOR_POLLRATE_MANUAL:
146  stop();
147  _measure_ticks = 0;
148  return OK;
149 
150  /* external signaling (DRDY) not supported */
151  case SENSOR_POLLRATE_EXTERNAL:
152 
153  /* zero would be bad */
154  case 0:
155  return -EINVAL;
156 
157  /* set default/max polling rate */
158  case SENSOR_POLLRATE_MAX:
159  case SENSOR_POLLRATE_DEFAULT: {
160  /* do we need to start internal polling? */
161  bool want_start = (_measure_ticks == 0);
162 
163  /* set interval for next measurement to minimum legal value */
164  _measure_ticks = USEC2TICK(_conversion_interval);
165 
166  /* if we need to start the poll state machine, do it */
167  if (want_start) {
168  start();
169  }
170 
171  return OK;
172  }
173 
174  /* adjust to a legal polling interval in Hz */
175  default: {
176  /* do we need to start internal polling? */
177  bool want_start = (_measure_ticks == 0);
178 
179  /* convert hz to tick interval via microseconds */
180  unsigned ticks = USEC2TICK(1000000 / arg);
181 
182  /* check against maximum rate */
183  if (ticks < USEC2TICK(_conversion_interval)) {
184  return -EINVAL;
185  }
186 
187  /* update interval for next measurement */
188  _measure_ticks = ticks;
189 
190  /* if we need to start the poll state machine, do it */
191  if (want_start) {
192  start();
193  }
194 
195  return OK;
196  }
197  }
198  }
199  break;
200 
201  case SENSORIOCGPOLLRATE:
202  if (_measure_ticks == 0) {
203  return SENSOR_POLLRATE_MANUAL;
204  }
205 
206  return (1000 / _measure_ticks);
207 
208  case SENSORIOCRESET:
209  /* XXX implement this */
210  return -EINVAL;
211 
212  case AIRSPEEDIOCSSCALE: {
213  struct airspeed_scale *s = (struct airspeed_scale *)arg;
214  _diff_pres_offset = s->offset_pa;
215  return OK;
216  }
217 
218  case AIRSPEEDIOCGSCALE: {
219  struct airspeed_scale *s = (struct airspeed_scale *)arg;
220  s->offset_pa = _diff_pres_offset;
221  s->scale = 1.0f;
222  return OK;
223  }
224 
225  default:
226  /* give it to the superclass */
227  return I2C::ioctl(filp, cmd, arg);
228  }
229 }
