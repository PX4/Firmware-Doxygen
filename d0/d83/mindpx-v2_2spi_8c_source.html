100 __EXPORT void stm32_spi4select(FAR struct spi_dev_s *dev, uint32_t devid, bool selected)
101 {
102  /* SPI select is active low, so write !selected to select the device */
103 
104  switch (devid) {
105  case PX4_SPIDEV_GYRO:
106  /* Making sure the other peripherals are not selected */
107  stm32_gpiowrite(GPIO_SPI_CS_GYRO, !selected);
108  stm32_gpiowrite(GPIO_SPI_CS_ACCEL_MAG, 1);
109  stm32_gpiowrite(GPIO_SPI_CS_BARO, 1);
110  stm32_gpiowrite(GPIO_SPI_CS_MPU, 1);
111  break;
112 
113  case PX4_SPIDEV_ACCEL_MAG:
114  /* Making sure the other peripherals are not selected */
115  stm32_gpiowrite(GPIO_SPI_CS_GYRO, 1);
116  stm32_gpiowrite(GPIO_SPI_CS_ACCEL_MAG, !selected);
117  stm32_gpiowrite(GPIO_SPI_CS_BARO, 1);
118  stm32_gpiowrite(GPIO_SPI_CS_MPU, 1);
119  break;
120 
121  case PX4_SPIDEV_BARO:
122  /* Making sure the other peripherals are not selected */
123 
124  px4_arch_gpiowrite(GPIO_SPI_CS_GYRO, 1);
125  px4_arch_gpiowrite(GPIO_SPI_CS_ACCEL_MAG, 1);
126  px4_arch_gpiowrite(GPIO_SPI_CS_BARO, !selected);
127  px4_arch_gpiowrite(GPIO_SPI_CS_MPU, 1);
128  break;
129 
130  case PX4_SPIDEV_MPU:
131  /* Making sure the other peripherals are not selected */
132  stm32_gpiowrite(GPIO_SPI_CS_GYRO, 1);
133  stm32_gpiowrite(GPIO_SPI_CS_ACCEL_MAG, 1);
134  stm32_gpiowrite(GPIO_SPI_CS_BARO, 1);
135  stm32_gpiowrite(GPIO_SPI_CS_MPU, !selected);
136  break;
137 
138  default:
139  break;
140  }
141 }
