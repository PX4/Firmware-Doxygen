122 int VCDevNode::open(device::file_t *handlep)
123 {
124  // Only allow one writer
125  if (_is_open_for_write && (handlep->flags & PX4_F_WRONLY)) {
126  errno = EBUSY;
127  return -1;

217 {

227  bool mustblock = (timeout < 0);
228 
229  // Test indefinte blocking poll
230  while ((!appState.exitRequested()) && (loop_count < iterations)) {
231  pollret = px4_poll(fds, 1, timeout);
232 
233  if (pollret < 0) {
234  PX4_ERR("Reader: px4_poll failed %d %d FAIL", pollret, px4_errno);
235  goto fail;
236  }
237 
238  PX4_INFO("Reader: px4_poll returned %d", pollret);
239 
240  if (pollret) {
241  readret = px4_read(fd, readbuf, 10);
242 
243  if (readret != 1) {
244  if (mustblock) {
245  PX4_ERR("Reader: read failed %d FAIL", readret);
246  goto fail;
247 
248  } else {
249  PX4_INFO("Reader: read failed %d FAIL", readret);
250  }
251 
252  } else {
253  readbuf[readret] = '\0';
254  PX4_INFO("Reader: px4_poll returned %d, read '%s' PASS", pollret, readbuf);
255  }
256  }
257 
258  if (delayms_after_poll) {
259  usleep(delayms_after_poll * 1000);
260  }
261 
262  loop_count++;
263  }
264 
265  return 0;
266 fail:
267  return 1;
