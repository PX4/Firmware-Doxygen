93 void ManualInput::JoyCallback(const sensor_msgs::JoyConstPtr &msg)
94 {
95 
96  /* Fill px4 manual control setpoint topic with contents from ros joystick */
97  /* Map sticks to x, y, z, r */
98  MapAxis(msg, _param_axes_map[0], _param_axes_scale[0], _param_axes_offset[0], _param_axes_dz[0], _msg_mc_sp.x);
99  MapAxis(msg, _param_axes_map[1], _param_axes_scale[1], _param_axes_offset[1], _param_axes_dz[1], _msg_mc_sp.y);
100  MapAxis(msg, _param_axes_map[2], _param_axes_scale[2], _param_axes_offset[2], _param_axes_dz[2], _msg_mc_sp.z);
101  MapAxis(msg, _param_axes_map[3], _param_axes_scale[3], _param_axes_offset[3], _param_axes_dz[3], _msg_mc_sp.r);
102  //ROS_INFO("x: %1.4f y: %1.4f z: %1.4f r: %1.4f", msg_mc_sp.x, msg_out.y, msg_out.z, msg_out.r);
103 
104  /* Map buttons to switches */
105  MapButtons(msg, _msg_mc_sp);
106 
107  _msg_mc_sp.timestamp = px4::get_time_micros();
108 
109  _man_ctrl_sp_pub.publish(_msg_mc_sp);
110 }

112 void ManualInput::MapAxis(const sensor_msgs::JoyConstPtr &msg, int map_index, double scale, double offset,
113  double deadzone, float &out)
114 {
115  double val = msg->axes[map_index];
116 
117  if (val + offset > deadzone || val + offset < -deadzone) {
118  out = (float)((val + offset)) * scale;
119 
120  } else {
121  out = 0.0f;
122  }

125 void ManualInput::MapButtons(const sensor_msgs::JoyConstPtr &msg, px4::manual_control_setpoint &msg_mc_sp)
126 {
127  msg_mc_sp.acro_switch = px4::manual_control_setpoint::SWITCH_POS_NONE;
128 
129  if (_buttons_state[MAIN_STATE_MANUAL] != msg->buttons[_param_buttons_map[MAIN_STATE_MANUAL]] == true) {
130  msg_mc_sp.mode_switch = px4::manual_control_setpoint::SWITCH_POS_OFF;
131  msg_mc_sp.return_switch = px4::manual_control_setpoint::SWITCH_POS_OFF;
132  msg_mc_sp.posctl_switch = px4::manual_control_setpoint::SWITCH_POS_OFF;
133  msg_mc_sp.loiter_switch = px4::manual_control_setpoint::SWITCH_POS_OFF;
134  msg_mc_sp.return_switch = px4::manual_control_setpoint::SWITCH_POS_OFF;
135  msg_mc_sp.offboard_switch = px4::manual_control_setpoint::SWITCH_POS_OFF;
136  return;
137 
138  } else if (_buttons_state[MAIN_STATE_ALTCTL] != msg->buttons[_param_buttons_map[MAIN_STATE_ALTCTL]] == true) {
139  msg_mc_sp.mode_switch = px4::manual_control_setpoint::SWITCH_POS_MIDDLE;
140  msg_mc_sp.return_switch = px4::manual_control_setpoint::SWITCH_POS_OFF;
141  msg_mc_sp.posctl_switch = px4::manual_control_setpoint::SWITCH_POS_OFF;
142  msg_mc_sp.loiter_switch = px4::manual_control_setpoint::SWITCH_POS_OFF;

147  } else if (_buttons_state[MAIN_STATE_POSCTL] != msg->buttons[_param_buttons_map[MAIN_STATE_POSCTL]] == true) {
148  msg_mc_sp.mode_switch = px4::manual_control_setpoint::SWITCH_POS_MIDDLE;
