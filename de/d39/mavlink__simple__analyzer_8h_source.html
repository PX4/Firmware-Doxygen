44 /**
45  * SimpleAnalyzer
46  *
47  * Class used for simple analysis of data streams.
48  * The data can be analyzed in the following three modes:
49  *
50  * AVERAGE:
51  * The average value is computed at the beginning. Based on the number of analyzed values,
52  * the update rate and the window size the switch to the moving average is determined.
53  *
54  * MIN:
55  * The minimum value is tracked.
56  *
57  * MAX:
58  * The maximum value is tracked.
59  */
60 class SimpleAnalyzer
61 {
62 public:
63  enum Mode {
64  AVERAGE = 0,
65  MIN,
66  MAX,
67  };

111  /**
112  * Get the rounded scaled value casted to the input template type.
113  * Should only be used to return integer types.
114  *
115  * @param[out] ret: The scaled and rounded value of the current analyzer result.
116  * @parma[in] scalingfactor: The factor which is used to scale the result.
117  */
118  template <typename T>
119  void get_scaled(T &ret, float scalingfactor) const
120  {
121  float avg = get_scaled(scalingfactor);
122  int_round(avg);
