34 /*
35  * @file drv_input_capture.c
36  *
37  * Servo driver supporting input capture connected to STM32 timer blocks.
38  *
39  * Works with any of the 'generic' or 'advanced' STM32 timers that
40  * have input pins.
41  *
42  * Require an interrupt.
43  *
44  * The use of thie interface is mutually exclusive with the pwm
45  * because the same timers are used and there is a resource contention
46  * with the ARR as it sets the pwm rate and in this driver needs to match
47  * that of the hrt to back calculate the actual point in time the edge
48  * was detected.
49  *
50  * This is accomplished by taking the difference between the current
51  * count rCNT snapped at the time interrupt and the rCCRx captured on the
52  * edge transition. This delta is applied to hrt time and the resulting
53  * value is the absolute system time the edge occured.
54  *
55  *
56  */

102 static void input_capture_chan_handler(void *context, const io_timers_t *timer, uint32_t chan_index,
103  const timer_io_channels_t *chan,
104  hrt_abstime isrs_time, uint16_t isrs_rcnt,
105  uint16_t capture)
106 {
107  channel_stats[chan_index].last_edge = px4_arch_gpioread(chan->gpio_in);
108 
109  if ((isrs_rcnt - capture) > channel_stats[chan_index].latnecy) {
110  channel_stats[chan_index].latnecy = (isrs_rcnt - capture);
111  }
112 
113  channel_stats[chan_index].chan_in_edges_out++;
114  channel_stats[chan_index].last_time = isrs_time - (isrs_rcnt - capture);
115  uint32_t overflow = _REG32(timer, KINETIS_FTM_CSC_OFFSET(chan->timer_channel - 1)) & FTM_CSC_CHF;
116 
117  if (overflow) {
118 
119  /* Error we has a second edge before we cleared CCxR */
120 
121  channel_stats[chan_index].overflows++;
122  }
123 
124  if (channel_handlers[chan_index].callback) {
125  channel_handlers[chan_index].callback(channel_handlers[chan_index].context, chan_index,
126  channel_stats[chan_index].last_time,
127  channel_stats[chan_index].last_edge, overflow);
128  }
129 }

144 int up_input_capture_set(unsigned channel, input_capture_edge edge, capture_filter_t filter,
145  capture_callback_t callback, void *context)
146 {
147  if (filter > FTM_FILTER_CH0FVAL_MASK) {
148  return -EINVAL;
149  }
150 
151  if (edge > Both) {
152  return -EINVAL;
153  }
154 
155  int rv = io_timer_validate_channel_index(channel);
156 
157  if (rv == 0) {
158 
159  /* This register selects the filter value for the inputs of channels.
160  Channels 4, 5, 6 and 7 do not have an input filter.
161  */
162  if (filter && timer_io_channels[channel].timer_channel - 1 > 3) {
163  return -EINVAL;
164  }
165 
166 
167  if (edge == Disabled) {
168 
169  io_timer_set_enable(false, IOTimerChanMode_Capture, 1 << channel);
170  input_capture_unbind(channel);
171 
172  } else {
173 
174  if (-EBUSY == io_timer_is_channel_free(channel)) {
175  io_timer_free_channel(channel);
176  }
177 
178  input_capture_bind(channel, callback, context);
179 
180  rv = io_timer_channel_init(channel, IOTimerChanMode_Capture, input_capture_chan_handler, context);
181 
182  if (rv != 0) {
183  return rv;
184  }
185 
186  rv = up_input_capture_set_filter(channel, filter);
187 
188  if (rv == 0) {
189  rv = up_input_capture_set_trigger(channel, edge);
190 
191  if (rv == 0) {
192  rv = io_timer_set_enable(true, IOTimerChanMode_Capture, 1 << channel);
193  }
194  }
195  }
196  }
197 
198  return rv;
199 }

201 int up_input_capture_get_filter(unsigned channel, capture_filter_t *filter)
202 {
203  int rv = io_timer_validate_channel_index(channel);
204 
205  if (rv == 0) {
206 
207  rv = -EINVAL;
208 
209  if (timer_io_channels[channel].timer_channel - 1 <= 3) {
210  rv = -ENXIO;
211 
212  /* Any pins in capture mode */
213 
214  if (io_timer_get_channel_mode(channel) == IOTimerChanMode_Capture) {
215 
216  uint32_t timer = timer_io_channels[channel].timer_index;
217  uint16_t rvalue;
218  rv = OK;
219 
220  switch (timer_io_channels[channel].timer_channel) {
221 
222  case 1:
223  rvalue = rFILTER(timer) & FTM_FILTER_CH0FVAL_MASK;
224  *filter = (rvalue >> FTM_FILTER_CH0FVAL_SHIFT);
225  break;
226 
227  case 2:
228  rvalue = rFILTER(timer) & FTM_FILTER_CH1FVAL_MASK;
229  *filter = (rvalue >> FTM_FILTER_CH1FVAL_SHIFT);
230  break;
231 
232  case 3:
233  rvalue = rFILTER(timer) & FTM_FILTER_CH2FVAL_MASK;
234  *filter = (rvalue >> FTM_FILTER_CH2FVAL_SHIFT);
235  break;
236 
237  case 4:
238  rvalue = rFILTER(timer) & FTM_FILTER_CH3FVAL_MASK;
239  *filter = (rvalue >> FTM_FILTER_CH3FVAL_SHIFT);

244  }

344  }

350 int up_input_capture_set_trigger(unsigned channel, input_capture_edge edge)
351 {
352  int rv = io_timer_validate_channel_index(channel);
353 
354  if (rv == 0) {
355 
356  rv = -ENXIO;
357 
358  /* Any pins in capture mode */
359 
360  if (io_timer_get_channel_mode(channel) == IOTimerChanMode_Capture) {
361 
362  uint16_t edge_bits = 0;
363 
364  switch (edge) {
365  case Disabled:
366  break;
367 
368  case Rising:
369  edge_bits = FTM_CSC_MSA;
370  break;
371 
372  case Falling:
373  edge_bits = FTM_CSC_MSB;
374  break;
375 
376  case Both:
377  edge_bits = (FTM_CSC_MSB | FTM_CSC_MSA);
378  break;
379 
380  default:
381  return -EINVAL;;
382  }
383 
384  uint32_t timer = timer_io_channels[channel].timer_index;
385  irqstate_t flags = px4_enter_critical_section();
386  uint32_t rvalue = _REG32(timer, KINETIS_FTM_CSC_OFFSET(timer_io_channels[channel].timer_channel - 1));
387  rvalue &= (FTM_CSC_MSB | FTM_CSC_MSA);
388  rvalue |= edge_bits;
389  _REG32(timer, KINETIS_FTM_CSC_OFFSET(timer_io_channels[channel].timer_channel - 1)) = rvalue;
390  px4_leave_critical_section(flags);
391  rv = OK;
392  }
393  }
394 
395  return rv;
