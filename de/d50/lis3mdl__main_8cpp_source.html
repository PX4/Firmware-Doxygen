175 bool
176 lis3mdl::test(LIS3MDL_BUS bus_id)
177 {
178  struct lis3mdl_bus_option &bus = find_bus(bus_id);
179  struct mag_report report;
180  ssize_t sz;
181  int ret;
182  const char *path = bus.devpath;
183 
184  int fd = open(path, O_RDONLY);
185 
186  if (fd < 0) {
187  PX4_WARN("%s open failed (try 'lis3mdl start')", path);
188  return 1;
189  }
190 
191  /* do a simple demand read */
192  sz = read(fd, &report, sizeof(report));
193 
194  if (sz != sizeof(report)) {
195  PX4_WARN("immediate read failed");
196  return 1;
197  }
198 
199  print_message(report);
200 
201  /* check if mag is onboard or external */
202  if (ioctl(fd, MAGIOCGEXTERNAL, 0) < 0) {
203  PX4_WARN("failed to get if mag is onboard or external");
204  return 1;
205  }
206 
207  /* set the queue depth to 5 */
208  if (ioctl(fd, SENSORIOCSQUEUEDEPTH, 10) != OK) {
209  PX4_WARN("failed to set queue depth");
210  return 1;
211  }
212 
213  /* start the sensor polling at 2Hz */
214  if (ioctl(fd, SENSORIOCSPOLLRATE, 2) != OK) {
215  PX4_WARN("failed to set 2Hz poll rate");
216  return 1;
217  }
218 
219  struct pollfd fds;
220 
221  /* read the sensor 5x and report each value */
222  for (unsigned i = 0; i < 5; i++) {
223 
224  /* wait for data to be ready */
225  fds.fd = fd;
226  fds.events = POLLIN;
227  ret = poll(&fds, 1, 2000);
228 
229  if (ret != 1) {
230  PX4_WARN("timed out waiting for sensor data");
231  return 1;
232  }
233 
234  /* now go get it */
235  sz = read(fd, &report, sizeof(report));
236 
237  if (sz != sizeof(report)) {
238  PX4_WARN("periodic read failed");
239  return 1;
240  }
241 
242  print_message(report);
243  }
244 
245  PX4_INFO("PASS");
246  return 1;
247 }

288 int
289 lis3mdl_main(int argc, char *argv[])
290 {
291  int myoptind = 1;
292  int ch;
293  const char *myoptarg = nullptr;
294 
295  bool calibrate = false;
296  enum LIS3MDL_BUS bus_id = LIS3MDL_BUS_ALL;
297  enum Rotation rotation = ROTATION_NONE;
298 
299  while ((ch = px4_getopt(argc, argv, "XISR:CT", &myoptind, &myoptarg)) != EOF) {
300  switch (ch) {
301  case 'R':
302  rotation = (enum Rotation)atoi(myoptarg);
303  break;
304 #if (PX4_I2C_BUS_ONBOARD || PX4_SPIDEV_LIS)
305 
306  case 'I':
307  bus_id = LIS3MDL_BUS_I2C_INTERNAL;
308  break;
309 #endif
310 
311  case 'X':
312  bus_id = LIS3MDL_BUS_I2C_EXTERNAL;
313  break;
314 
315  case 'S':
316  bus_id = LIS3MDL_BUS_SPI;
317  break;
318 
319  case 'C':
320  calibrate = true;
321  break;
322 
323  default:
324  lis3mdl::usage();
325  return 0;
326  }
327  }
328 
329  if (myoptind >= argc) {
330  lis3mdl::usage();
331  return -1;
332  }
333 
334  const char *verb = argv[myoptind];
335 
336  // Start/load the driver
337  if (!strcmp(verb, "start")) {
338 
339  if (lis3mdl::start(bus_id, rotation)) {
340  if (calibrate) {
341  if (OK != lis3mdl::calibrate(bus_id)) {
342  PX4_WARN("calibration failed");
343  return 1;
344  }
345  }
346 
347  lis3mdl::init(bus_id);
348 
349  return 0;
350 
351  } else {
352  return 1;
353  }
354  }
355 
356  // Stop the driver
357  if (!strcmp(verb, "stop")) {
358  return lis3mdl::stop();
359  }
360 
361  // Test the driver/device
362  if (!strcmp(verb, "test")) {
363  return lis3mdl::test(bus_id);
364  }
365 
366  // Reset the driver
367  if (!strcmp(verb, "reset")) {
368  return lis3mdl::reset(bus_id);
369  }
370 
371  // Print driver information
372  if (!strcmp(verb, "info") ||
373  !strcmp(verb, "status")) {
374  return lis3mdl::info(bus_id);
375  }
376 
377  // Autocalibrate the scaling
378  if (!strcmp(verb, "calibrate")) {
379  if (lis3mdl::calibrate(bus_id) == 0) {
380  PX4_INFO("calibration successful");
381  return 0;
382 
383  } else {
384  PX4_INFO("calibration failed");
385  return 1;
386  }
387 
388  }
389 
390  PX4_INFO("unrecognized command, try 'start', 'test', 'reset', 'calibrate' 'or 'info'");
391  return 1;
392 }
