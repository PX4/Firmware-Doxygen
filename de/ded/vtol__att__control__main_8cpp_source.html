115 VtolAttitudeControl::~VtolAttitudeControl()
116 {
117  if (_control_task != -1) {
118  /* task wakes up every 100ms or so at the longest */
119  _task_should_exit = true;
120 
121  /* wait for a second for the task to quit at our request */
122  unsigned i = 0;
123 
124  do {
125  /* wait 20ms */
126  usleep(20000);
127 
128  /* if we have given up, kill it */
129  if (++i > 50) {
130  px4_task_delete(_control_task);
131  break;
132  }
133  } while (_control_task != -1);
134  }
135 
136  // free memory used by instances of base class VtolType
137  if (_vtol_type != nullptr) {
138  delete _vtol_type;
139  }
140 
141  VTOL_att_control::g_control = nullptr;
142 }

352 /**
353 * Check received command
354 */
355 void
356 VtolAttitudeControl::handle_command()
357 {
358  // update transition command if necessary
359  if (_vehicle_cmd.command == vehicle_command_s::VEHICLE_CMD_DO_VTOL_TRANSITION) {
360  _transition_command = int(_vehicle_cmd.param1 + 0.5f);
361 
362  // Report that we have received the command no matter what we actually do with it.
363  // This might not be optimal but is better than no response at all.
364 
365  if (_vehicle_cmd.from_external) {
366  vehicle_command_ack_s command_ack = {};
367  command_ack.timestamp = hrt_absolute_time();
368  command_ack.command = _vehicle_cmd.command;
369  command_ack.result = (uint8_t)vehicle_command_ack_s::VEHICLE_RESULT_ACCEPTED;
370  command_ack.target_system = _vehicle_cmd.source_system;
371  command_ack.target_component = _vehicle_cmd.source_component;
372 
373  if (_v_cmd_ack_pub == nullptr) {
374  _v_cmd_ack_pub = orb_advertise_queue(ORB_ID(vehicle_command_ack), &command_ack,
375  vehicle_command_ack_s::ORB_QUEUE_LENGTH);
376 
377  } else {
378  orb_publish(ORB_ID(vehicle_command_ack), _v_cmd_ack_pub, &command_ack);
379  }
380  }
381  }

384 /*
385  * Returns true if fixed-wing mode is requested.
386  * Changed either via switch or via command.
387  */
388 bool
389 VtolAttitudeControl::is_fixed_wing_requested()
390 {
391  bool to_fw = false;
392 
393  if (_manual_control_sp.transition_switch != manual_control_setpoint_s::SWITCH_POS_NONE &&
394  _v_control_mode.flag_control_manual_enabled) {
395  to_fw = (_manual_control_sp.transition_switch == manual_control_setpoint_s::SWITCH_POS_ON);
396 
397  } else {
398  // listen to transition commands if not in manual or mode switch is not mapped
399  to_fw = (_transition_command == vtol_vehicle_status_s::VEHICLE_VTOL_STATE_FW);
400  }
401 
402  // handle abort request
403  if (_abort_front_transition) {
404  if (to_fw) {
405  to_fw = false;
406 
407  } else {
408  // the state changed to mc mode, reset the abort request
409  _abort_front_transition = false;
410  _vtol_vehicle_status.vtol_transition_failsafe = false;
411  }
412  }
413 
414  return to_fw;
415 }

430 /**
431 * Update parameters.

433 int
434 VtolAttitudeControl::parameters_update()
435 {
436  float v;
437  int32_t l;
438  /* idle pwm for mc mode */
439  param_get(_params_handles.idle_pwm_mc, &_params.idle_pwm_mc);
440 
441  /* vtol motor count */
442  param_get(_params_handles.vtol_motor_count, &_params.vtol_motor_count);
443 
444  /* vtol fw permanent stabilization */
445  param_get(_params_handles.vtol_fw_permanent_stab, &l);
446  _vtol_vehicle_status.fw_permanent_stab = (l == 1);
447 
448  param_get(_params_handles.vtol_type, &l);
449  _params.vtol_type = l;
450 
451  /* vtol lock elevons in multicopter */
452  param_get(_params_handles.elevons_mc_lock, &l);
453  _params.elevons_mc_lock = (l == 1);
454 
455  /* minimum relative altitude for FW mode (QuadChute) */
456  param_get(_params_handles.fw_min_alt, &v);
457  _params.fw_min_alt = v;
458 
459  /* maximum negative altitude error for FW mode (Adaptive QuadChute) */
460  param_get(_params_handles.fw_alt_err, &v);
461  _params.fw_alt_err = v;
462 
463  /* maximum pitch angle (QuadChute) */
464  param_get(_params_handles.fw_qc_max_pitch, &l);
465  _params.fw_qc_max_pitch = l;
466 
467  /* maximum roll angle (QuadChute) */
468  param_get(_params_handles.fw_qc_max_roll, &l);
469  _params.fw_qc_max_roll = l;
470 
471  param_get(_params_handles.front_trans_time_openloop, &_params.front_trans_time_openloop);
472 
473  param_get(_params_handles.front_trans_time_min, &_params.front_trans_time_min);
474 
475  /*
476  * Minimum transition time can be maximum 90 percent of the open loop transition time,
477  * anything else makes no sense and can potentially lead to numerical problems.
478  */
479  _params.front_trans_time_min = math::min(_params.front_trans_time_openloop * 0.9f,
480  _params.front_trans_time_min);
481 
482  /* weathervane */
483  param_get(_params_handles.wv_takeoff, &l);
484  _params.wv_takeoff = (l == 1);
485 
486  param_get(_params_handles.wv_loiter, &l);
487  _params.wv_loiter = (l == 1);
488 
489  param_get(_params_handles.wv_land, &l);
490  _params.wv_land = (l == 1);
491 
492 
493  param_get(_params_handles.front_trans_duration, &_params.front_trans_duration);
494  param_get(_params_handles.back_trans_duration, &_params.back_trans_duration);
495  param_get(_params_handles.transition_airspeed, &_params.transition_airspeed);
496  param_get(_params_handles.front_trans_throttle, &_params.front_trans_throttle);
497  param_get(_params_handles.back_trans_throttle, &_params.back_trans_throttle);
498  param_get(_params_handles.airspeed_blend, &_params.airspeed_blend);
499  param_get(_params_handles.airspeed_mode, &l);
500  _params.airspeed_disabled = l != 0;
501  param_get(_params_handles.front_trans_timeout, &_params.front_trans_timeout);
502  param_get(_params_handles.mpc_xy_cruise, &_params.mpc_xy_cruise);
503  param_get(_params_handles.fw_motors_off, &_params.fw_motors_off);
504 
505  // standard vtol always needs to turn all mc motors off when going into fixed wing mode
506  // normally the parameter fw_motors_off can be used to specify this, however, since historically standard vtol code
507  // did not use the interface of the VtolType class to disable motors we will have users flying around with a wrong
508  // parameter value. Therefore, explicitly set it here such that all motors will be disabled as expected.
509  if (_params.vtol_type == vtol_type::STANDARD) {
510  _params.fw_motors_off = 12345678;
511  }
512 
513  // make sure parameters are feasible, require at least 1 m/s difference between transition and blend airspeed
514  _params.airspeed_blend = math::min(_params.airspeed_blend, _params.transition_airspeed - 1.0f);
515 
516  // update the parameters of the instances of base VtolType
517  if (_vtol_type != nullptr) {
518  _vtol_type->parameters_update();
519  }
520 
521  return OK;
522 }

577 void VtolAttitudeControl::task_main()
578 {
579  fflush(stdout);
580 
581  /* do subscriptions */
582  _v_att_sp_sub = orb_subscribe(ORB_ID(vehicle_attitude_setpoint));
583  _mc_virtual_att_sp_sub = orb_subscribe(ORB_ID(mc_virtual_attitude_setpoint));
584  _fw_virtual_att_sp_sub = orb_subscribe(ORB_ID(fw_virtual_attitude_setpoint));
585  _mc_virtual_v_rates_sp_sub = orb_subscribe(ORB_ID(mc_virtual_rates_setpoint));
586  _fw_virtual_v_rates_sp_sub = orb_subscribe(ORB_ID(fw_virtual_rates_setpoint));
587  _v_att_sub = orb_subscribe(ORB_ID(vehicle_attitude));
588  _v_control_mode_sub = orb_subscribe(ORB_ID(vehicle_control_mode));
589  _params_sub = orb_subscribe(ORB_ID(parameter_update));
590  _manual_control_sp_sub = orb_subscribe(ORB_ID(manual_control_setpoint));
591  _local_pos_sub = orb_subscribe(ORB_ID(vehicle_local_position));
592  _local_pos_sp_sub = orb_subscribe(ORB_ID(vehicle_local_position_setpoint));
593  _pos_sp_triplet_sub = orb_subscribe(ORB_ID(position_setpoint_triplet));
594  _airspeed_sub = orb_subscribe(ORB_ID(airspeed));
595  _vehicle_cmd_sub = orb_subscribe(ORB_ID(vehicle_command));
596  _tecs_status_sub = orb_subscribe(ORB_ID(tecs_status));
597  _land_detected_sub = orb_subscribe(ORB_ID(vehicle_land_detected));
598 
599  _actuator_inputs_mc = orb_subscribe(ORB_ID(actuator_controls_virtual_mc));
600  _actuator_inputs_fw = orb_subscribe(ORB_ID(actuator_controls_virtual_fw));
601 
602  parameters_update(); // initialize parameter cache
603 
604  _task_should_exit = !_vtol_type->init();
605 
606  /* wakeup source*/
607  px4_pollfd_struct_t fds[1] = {};
608  fds[0].fd = _actuator_inputs_mc;
609  fds[0].events = POLLIN;
610 
611  while (!_task_should_exit) {
612  /* only update parameters if they changed */
613  bool params_updated = false;
614  orb_check(_params_sub, &params_updated);
615 
616  if (params_updated) {
617  /* read from param to clear updated flag */
618  parameter_update_s update;
619  orb_copy(ORB_ID(parameter_update), _params_sub, &update);
620 
621  /* update parameters from storage */
622  parameters_update();
623  }
624 
625  // run vtol_att on MC actuator publications, unless in full FW mode
626  switch (_vtol_type->get_mode()) {
627  case TRANSITION_TO_FW:
628  case TRANSITION_TO_MC:
629  case ROTARY_WING:

631  break;
632 
633  case FIXED_WING:
634  fds[0].fd = _actuator_inputs_fw;
635  break;
636  }
637

639  int pret = px4_poll(&fds[0], sizeof(fds) / sizeof(fds[0]), 100);
640 
641  /* timed out - periodic check for _task_should_exit */
642  if (pret == 0) {
643  continue;
644  }
645 
646  /* this is undesirable but not much we can do - might want to flag unhappy status */
647  if (pret < 0) {
648  PX4_ERR("poll error %d, %d", pret, errno);
649  /* sleep a bit before next try */
650  usleep(100000);
651  continue;
652  }

655  vehicle_manual_poll();
656  vehicle_attitude_poll();

664  actuator_controls_fw_poll();
665  actuator_controls_mc_poll();

668  _vtol_type->update_vtol_state();
669

671  if (_v_control_mode.flag_control_manual_enabled) {
672  if (_vtol_type->get_mode() == ROTARY_WING) {
673  _transition_command = vtol_vehicle_status_s::VEHICLE_VTOL_STATE_MC;
674 
675  } else if (_vtol_type->get_mode() == FIXED_WING) {
676  _transition_command = vtol_vehicle_status_s::VEHICLE_VTOL_STATE_FW;
677 
678  } else if (_vtol_type->get_mode() == TRANSITION_TO_MC) {
679  /* We want to make sure that a mode change (manual>auto) during the back transition
680  * doesn't result in an unsafe state. This prevents the instant fall back to
681  * fixed-wing on the switch from manual to auto */
682  _transition_command = vtol_vehicle_status_s::VEHICLE_VTOL_STATE_MC;
683  }
684  }
685

687  if (_vtol_type->get_mode() == ROTARY_WING) {
688 
689  mc_virtual_att_sp_poll();
690 
691  // vehicle is in rotary wing mode
692  _vtol_vehicle_status.vtol_in_rw_mode = true;
693  _vtol_vehicle_status.vtol_in_trans_mode = false;
694  _vtol_vehicle_status.in_transition_to_fw = false;
695 
696  // got data from mc attitude controller
697  _vtol_type->update_mc_state();
698  fill_mc_att_rates_sp();
699 
700  } else if (_vtol_type->get_mode() == FIXED_WING) {
701 
702  fw_virtual_att_sp_poll();
703 
704  // vehicle is in fw mode
705  _vtol_vehicle_status.vtol_in_rw_mode = false;
706  _vtol_vehicle_status.vtol_in_trans_mode = false;
707  _vtol_vehicle_status.in_transition_to_fw = false;
708 
709  _vtol_type->update_fw_state();
710  fill_fw_att_rates_sp();
711 
712  } else if (_vtol_type->get_mode() == TRANSITION_TO_MC || _vtol_type->get_mode() == TRANSITION_TO_FW) {
713 
714  mc_virtual_att_sp_poll();
715  fw_virtual_att_sp_poll();
716 
717  // vehicle is doing a transition
718  _vtol_vehicle_status.vtol_in_trans_mode = true;
719  _vtol_vehicle_status.vtol_in_rw_mode = true; //making mc attitude controller work during transition
720  _vtol_vehicle_status.in_transition_to_fw = (_vtol_type->get_mode() == TRANSITION_TO_FW);
721 
722  _vtol_type->update_transition_state();
723  fill_mc_att_rates_sp();
724  }
725 
726  _vtol_type->fill_actuator_outputs();
727 
728  /* Only publish if the proper mode(s) are enabled */
729  if (_v_control_mode.flag_control_attitude_enabled ||
730  _v_control_mode.flag_control_rates_enabled ||
731  _v_control_mode.flag_control_manual_enabled) {
732 
733  if (_v_att_sp_pub != nullptr) {
734  /* publish the attitude setpoint */
735  orb_publish(ORB_ID(vehicle_attitude_setpoint), _v_att_sp_pub, &_v_att_sp);
736 
737  } else {
738  /* advertise and publish */
739  _v_att_sp_pub = orb_advertise(ORB_ID(vehicle_attitude_setpoint), &_v_att_sp);
740  }
741 
742  if (_actuators_0_pub != nullptr) {
743  orb_publish(ORB_ID(actuator_controls_0), _actuators_0_pub, &_actuators_out_0);
744 
745  } else {
746  _actuators_0_pub = orb_advertise(ORB_ID(actuator_controls_0), &_actuators_out_0);
747  }
748 
749  if (_actuators_1_pub != nullptr) {
750  orb_publish(ORB_ID(actuator_controls_1), _actuators_1_pub, &_actuators_out_1);
751 
752  } else {
753  _actuators_1_pub = orb_advertise(ORB_ID(actuator_controls_1), &_actuators_out_1);
754  }
755  }
756 
757  /*Advertise/Publish vtol vehicle status*/
758  _vtol_vehicle_status.timestamp = hrt_absolute_time();
759 
760  if (_vtol_vehicle_status_pub != nullptr) {
761  orb_publish(ORB_ID(vtol_vehicle_status), _vtol_vehicle_status_pub, &_vtol_vehicle_status);
762 
763  } else {
764  _vtol_vehicle_status_pub = orb_advertise(ORB_ID(vtol_vehicle_status), &_vtol_vehicle_status);
765  }
766  }
767 
768  PX4_WARN("exit");
769  _control_task = -1;
770 }
