234 int
235 Syslink::open_serial(const char *dev)
236 {
237 #ifndef B1000000
238 #define B1000000 1000000
239 #endif
240 
241  int rate = B1000000;
242 
243  // open uart
244  int fd = px4_open(dev, O_RDWR | O_NOCTTY);
245  int termios_state = -1;
246 
247  if (fd < 0) {
248  PX4_ERR("failed to open uart device!");
249  return -1;
250  }
251 
252  // set baud rate
253  struct termios config;
254  tcgetattr(fd, &config);
255 
256  // clear ONLCR flag (which appends a CR for every LF)
257  config.c_oflag = 0;
258  config.c_lflag &= ~(ECHO | ECHONL | ICANON | IEXTEN | ISIG);
259 
260  // Disable hardware flow control
261  config.c_cflag &= ~CRTSCTS;
262 
263 
264  /* Set baud rate */
265  if (cfsetispeed(&config, rate) < 0 || cfsetospeed(&config, rate) < 0) {
266  warnx("ERR SET BAUD %s: %d\n", dev, termios_state);
267  px4_close(fd);
268  return -1;
269  }
270 
271  if ((termios_state = tcsetattr(fd, TCSANOW, &config)) < 0) {
272  PX4_WARN("ERR SET CONF %s\n", dev);
273  px4_close(fd);
274  return -1;
275  }
276 
277  return fd;
278 }

289 void
290 Syslink::task_main()
291 {
292  _bridge = new SyslinkBridge(this);
293  _bridge->init();
294 
295  _memory = new SyslinkMemory(this);
296  _memory->init();
297 
298  _battery.reset(&_battery_status);
299 
300 
301  // int ret;
302 
303  /* Open serial port */
304  const char *dev = "/dev/ttyS2";
305  _fd = open_serial(dev);
306 
307  if (_fd < 0) {
308  err(1, "can't open %s", dev);
309  return;
310  }
311 
312 
313  /* Set non-blocking */
314  /*
315  int flags = fcntl(_fd, F_GETFL, 0);
316  fcntl(_fd, F_SETFL, flags | O_NONBLOCK);
317  */
318 
319  px4_arch_configgpio(GPIO_NRF_TXEN);
320 
321  px4_pollfd_struct_t fds[2];
322  fds[0].fd = _fd;
323  fds[0].events = POLLIN;
324 
325  _params_sub = orb_subscribe(ORB_ID(parameter_update));
326  fds[1].fd = _params_sub;
327  fds[1].events = POLLIN;
328 
329  int error_counter = 0;
330 
331  char buf[64];
332  int nread;
333 
334  syslink_parse_state state;
335  syslink_message_t msg;
336 
337  syslink_parse_init(&state);
338 
339  //setup initial parameters
340  update_params(true);
341 
342  while (_task_running) {
343  int poll_ret = px4_poll(fds, 2, 500);
344 
345  /* handle the poll result */
346  if (poll_ret == 0) {
347  /* timeout: this means none of our providers is giving us data */
348 
349  } else if (poll_ret < 0) {
350  /* this is seriously bad - should be an emergency */
351  if (error_counter < 10 || error_counter % 50 == 0) {
352  /* use a counter to prevent flooding (and slowing us down) */
353  PX4_ERR("[syslink] ERROR return value from poll(): %d"
354  , poll_ret);
355  }
356 
357  error_counter++;
358 
359  } else {
360  if (fds[0].revents & POLLIN) {
361  if ((nread = read(_fd, buf, sizeof(buf))) < 0) {
362  continue;
363  }
364 
365  for (int i = 0; i < nread; i++) {
366  if (syslink_parse_char(&state, buf[i], &msg)) {
367  handle_message(&msg);
368  }
369  }
370  }
371 
372  if (fds[1].revents & POLLIN) {
373  struct parameter_update_s update;
374  orb_copy(ORB_ID(parameter_update), _params_sub, &update);
375  update_params(false);
376  }
377  }
378 
379  }
380 
381  close(_fd);
382

385 void
386 Syslink::handle_message(syslink_message_t *msg)
387 {
388  hrt_abstime t = hrt_absolute_time();
389 
390  if (t - _lasttime > 1000000) {
391  pktrate = _count;
392  rxrate = _count_in;
393  txrate = _count_out;
394  nullrate = _null_count;
395 
396  _lasttime = t;
397  _count = 0;
398  _null_count = 0;
399  _count_in = 0;
400  _count_out = 0;
401  }
402 
403  _count++;
404 
405  if (msg->type == SYSLINK_PM_ONOFF_SWITCHOFF) {
406  // When the power button is hit
407  } else if (msg->type == SYSLINK_PM_BATTERY_STATE) {
408 
409  if (msg->length != 9) {
410  return;
411  }
412 
413  uint8_t flags = msg->data[0];
414  int charging = flags & 1;
415  int powered = flags & 2;
416 
417  float vbat; //, iset;
418  memcpy(&vbat, &msg->data[1], sizeof(float));
419  //memcpy(&iset, &msg->data[5], sizeof(float));
420 
421  _battery.updateBatteryStatus(t, vbat, -1, true, true, 0, 0, false, &_battery_status);
422 
423 
424  // Update battery charge state
425  if (charging) {
426  _bstate = BAT_CHARGING;
427  }
428 
429  /* With the usb plugged in and battery disconnected, it appears to be charged. The voltage check ensures that a battery is connected */
430  else if (powered && !charging && _battery_status.voltage_filtered_v > 3.7f) {
431  _bstate = BAT_CHARGED;
432 
433  } else {
434  _bstate = BAT_DISCHARGING;
435  }
436 
437 
438  // announce the battery status if needed, just publish else
439  if (_battery_pub != nullptr) {
440  orb_publish(ORB_ID(battery_status), _battery_pub, &_battery_status);
441 
442  } else {
443  _battery_pub = orb_advertise(ORB_ID(battery_status), &_battery_status);
444  }
445 
446  } else if (msg->type == SYSLINK_RADIO_RSSI) {
447  uint8_t rssi = msg->data[0]; // Between 40 and 100 meaning -40 dBm to -100 dBm
448  _rssi = 140 - rssi * 100 / (100 - 40);
449 
450  } else if (msg->type == SYSLINK_RADIO_RAW) {
451  handle_raw(msg);
452  _lastrxtime = t;
453 
454  } else if ((msg->type & SYSLINK_GROUP) == SYSLINK_RADIO) {
455  handle_radio(msg);
456 
457  } else if ((msg->type & SYSLINK_GROUP) == SYSLINK_OW) {
458  memcpy(&_memory->msgbuf, msg, sizeof(syslink_message_t));
459  px4_sem_post(&memory_sem);
460 
461  } else {
462  PX4_INFO("GOT %d", msg->type);
463  }
464 
465  //Send queued messages
466  if (!_queue.empty()) {
467  _queue.get(msg, sizeof(syslink_message_t));
468  send_message(msg);
469  }
470 
471 
472  float p = (t % 500000) / 500000.0f;
473 
474  /* Use LED_GREEN for charging indicator */
475  if (_bstate == BAT_CHARGED) {
476  led_on(LED_GREEN);
477 
478  } else if (_bstate == BAT_CHARGING && p < 0.25f) {
479  led_on(LED_GREEN);
480 
481  } else {
482  led_off(LED_GREEN);
483  }
484 
485  /* Alternate RX/TX LEDS when transfering */
486  bool rx = t - _lastrxtime < 200000,
487  tx = t - _lasttxtime < 200000;
488 
489 
490  if (rx && p < 0.25f) {
491  led_on(LED_RX);
492 
493  } else {
494  led_off(LED_RX);
495  }
496 
497  if (tx && p > 0.5f && p > 0.75f) {
498  led_on(LED_TX);
499 
500  } else {
501  led_off(LED_TX);
502  }
503 
504 
505  // resend parameters if they haven't been acknowledged
506  if (_params_ack[0] == 0 && t - _params_update[0] > 10000) {
507  set_channel(_channel);
508 
509  } else if (_params_ack[1] == 0 && t - _params_update[1] > 10000) {
510  set_datarate(_rate);
511 
512  } else if (_params_ack[2] == 0 && t - _params_update[2] > 10000) {
513  set_address(_addr);
514  }
515

536 void
537 Syslink::handle_raw(syslink_message_t *sys)
538 {
539  crtp_message_t *c = (crtp_message_t *) &sys->length;
540 
541  if (CRTP_NULL(*c)) {
542  if (c->size >= 3) {
543  handle_bootloader(sys);
544  }
545 
546  _null_count++;
547 
548  } else if (c->port == CRTP_PORT_COMMANDER) {
549 
550  crtp_commander *cmd = (crtp_commander *) &c->data[0];
551 
552  struct rc_input_values rc = {};
553 
554  rc.timestamp = hrt_absolute_time();
555  rc.timestamp_last_signal = rc.timestamp;
556  rc.channel_count = 5;
557  rc.rc_failsafe = false;
558  rc.rc_lost = false;
559  rc.rc_lost_frame_count = 0;
560  rc.rc_total_frame_count = 1;
561  rc.rc_ppm_frame_length = 0;
562  rc.input_source = input_rc_s::RC_INPUT_SOURCE_MAVLINK;
563  rc.rssi = _rssi;
564 
565 
566  double pitch = cmd->pitch, roll = cmd->roll, yaw = cmd->yaw;
567 
568  /* channels (scaled to rc limits) */
569  rc.values[0] = pitch * 500 / 20 + 1500;
570  rc.values[1] = roll * 500 / 20 + 1500;
571  rc.values[2] = yaw * 500 / 150 + 1500;
572  rc.values[3] = cmd->thrust * 1000 / USHRT_MAX + 1000;
573  rc.values[4] = 1000; // Dummy channel as px4 needs at least 5
574 
575  if (_rc_pub == nullptr) {
576  _rc_pub = orb_advertise(ORB_ID(input_rc), &rc);
577 
578  } else {
579  orb_publish(ORB_ID(input_rc), _rc_pub, &rc);
580  }
581 
582  } else if (c->port == CRTP_PORT_MAVLINK) {
583  _count_in++;
584  /* Pipe to Mavlink bridge */
585  _bridge->pipe_message(c);
586 
587  } else {
588  handle_raw_other(sys);
589  }
590 
591  // Block all non-requested messaged in bootloader mode
592  if (_bootloader_mode) {
593  return;
594  }
595 
596  // Allow one raw message to be sent from the queue
597  send_queued_raw_message();
598 }

600 void
601 Syslink::handle_bootloader(syslink_message_t *sys)
602 {
603  // Minimal bootloader emulation for being detectable
604  // To the bitcraze utilities, the STM32 will appear to have no flashable pages
605  // Upon receiving a bootloader message, all outbound packets are blocked in 'bootloader mode' due to how fragile the aforementioned utilities are to extra packets
606 
607  crtp_message_t *c = (crtp_message_t *) &sys->length;
608 
609  uint8_t target = c->data[0];
610  uint8_t cmd = c->data[1];
611 
612  if (target != 0xFF) { // CF2 STM32 target
613  return;
614  }
615 
616  _bootloader_mode = true;
617 
618  if (cmd == 0x10) { // GET_INFO
619 
620  c->size = 1 + 23;
621  memset(&c->data[2], 0, 21);
622  c->data[22] = 0x10; // Protocol version
623  send_message(sys);
624  }

629 Syslink::handle_raw_other(syslink_message_t *sys)
630 {
631  // This function doesn't actually do anything
632  // It is just here to return null responses to most standard messages
633 
634  crtp_message_t *c = (crtp_message_t *) &sys->length;
635 
636  if (c->port == CRTP_PORT_LOG) {
637 
638  PX4_INFO("Log: %d %d", c->channel, c->data[0]);
639 
640  if (c->channel == 0) { // Table of Contents Access
641 
642  uint8_t cmd = c->data[0];
643 
644  if (cmd == 0) { // GET_ITEM
645  //int id = c->data[1];
646  memset(&c->data[2], 0, 3);
647  c->data[2] = 1; // type
648  c->size = 1 + 5;
649  send_message(sys);
650 
651  } else if (cmd == 1) { // GET_INFO
652  memset(&c->data[1], 0, 7);
653  c->size = 1 + 8;
654  send_message(sys);
655  }
656 
657  } else if (c->channel == 1) { // Log control
658 
659  uint8_t cmd = c->data[0];
660 
661  PX4_INFO("Responding to cmd: %d", cmd);
662  c->data[2] = 0; // Success
663  c->size = 3 + 1;
664 
665  // resend message
666  send_message(sys);
667 
668  } else if (c->channel == 2) { // Log data
669 
670  }
671  } else if (c->port == CRTP_PORT_MEM) {
672  if (c->channel == 0) { // Info
673  int cmd = c->data[0];
674 
675  if (cmd == 1) { // GET_NBR_OF_MEMS
676  c->data[1] = 0;
677  c->size = 2 + 1;
678 
679  // resend message
680  send_message(sys);
681  }
682  }
683 
684  } else if (c->port == CRTP_PORT_PARAM) {
685  if (c->channel == 0) { // TOC Access
686  // uint8_t msgId = c->data[0];
687 
688  c->data[1] = 0; // Last parameter (id = 0)
689  memset(&c->data[2], 0, 10);
690  c->size = 1 + 8;
691  send_message(sys);
692  }
693 
694  else if (c->channel == 1) { // Param read
695  // 0 is ok
696  c->data[1] = 0; // value
697  c->size = 1 + 2;
698  send_message(sys);
699  }
700 
701  } else {
702  PX4_INFO("Got raw: %d", c->port);
703  }
