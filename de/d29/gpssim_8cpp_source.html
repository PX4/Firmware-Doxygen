191 GPSSIM::~GPSSIM()
192 {
193  delete _Sat_Info;
194 
195  /* tell the task we want it to go away */
196  _task_should_exit = true;
197 
198  /* spin waiting for the task to stop */
199  for (unsigned i = 0; (i < 10) && (_task != -1); i++) {
200  /* give it another 100ms */
201  usleep(100000);
202  }
203 
204  /* well, kill it anyway, though this will probably crash */
205  if (_task != -1) {
206  px4_task_delete(_task);
207  }
208 
209  g_dev = nullptr;

266 int
267 GPSSIM::receive(int timeout)
268 {
269  Simulator *sim = Simulator::getInstance();
270  simulator::RawGPSData gps;
271  sim->getGPSSample((uint8_t *)&gps, sizeof(gps));
272 
273  static int64_t timestamp_last = 0;
274 
275  if (gps.timestamp != timestamp_last) {
276  _report_gps_pos.timestamp = hrt_absolute_time();
277  _report_gps_pos.lat = gps.lat;
278  _report_gps_pos.lon = gps.lon;
279  _report_gps_pos.alt = gps.alt;
280  _report_gps_pos.eph = (float)gps.eph * 1e-2f;
281  _report_gps_pos.epv = (float)gps.epv * 1e-2f;
282  _report_gps_pos.vel_m_s = (float)(gps.vel) / 100.0f;
283  _report_gps_pos.vel_n_m_s = (float)(gps.vn) / 100.0f;
284  _report_gps_pos.vel_e_m_s = (float)(gps.ve) / 100.0f;
285  _report_gps_pos.vel_d_m_s = (float)(gps.vd) / 100.0f;
286  _report_gps_pos.cog_rad = (float)(gps.cog) * 3.1415f / (100.0f * 180.0f);
287  _report_gps_pos.fix_type = gps.fix_type;
288  _report_gps_pos.satellites_used = gps.satellites_visible;
289 
290  timestamp_last = gps.timestamp;
291 
292  // check for data set by the user
293  _report_gps_pos.fix_type = (_fix_type < 0) ? _report_gps_pos.fix_type : _fix_type;
294  _report_gps_pos.satellites_used = (_num_sat < 0) ? _report_gps_pos.satellites_used : _num_sat;
295  // use normal distribution for noise
296  std::normal_distribution<float> normal_distribution(0.0f, 1.0f);
297  _report_gps_pos.lat += (int32_t)(_noise_multiplier * normal_distribution(_gen));
298  _report_gps_pos.lon += (int32_t)(_noise_multiplier * normal_distribution(_gen));
299 
300  return 1;
301 
302  } else {
303 
304  usleep(timeout);
305  return 0;
306  }

309 void
310 GPSSIM::task_main()
311 {
312  /* loop handling received serial bytes and also configuring in between */
313  while (!_task_should_exit) {
314 
315  int recv_ret = receive(TIMEOUT_100MS);
316 
317  if (recv_ret > 0) {
318 
319  /* opportunistic publishing - else invalid data would end up on the bus */
320  if (_report_gps_pos_pub != nullptr) {
321  orb_publish(ORB_ID(vehicle_gps_position), _report_gps_pos_pub, &_report_gps_pos);
322 
323  } else {
324  _report_gps_pos_pub = orb_advertise(ORB_ID(vehicle_gps_position), &_report_gps_pos);
325  }
326 
327  if (_p_report_sat_info) {
328  if (_report_sat_info_pub != nullptr) {
329  orb_publish(ORB_ID(satellite_info), _report_sat_info_pub, _p_report_sat_info);
330 
331  } else {
332  _report_sat_info_pub = orb_advertise(ORB_ID(satellite_info), _p_report_sat_info);
333  }
334  }
335  }
336  }
337 
338  if (_report_gps_pos_pub) {
339  orb_unadvertise(_report_gps_pos_pub);
340  }
341 
342  if (_report_sat_info_pub) {
343  orb_unadvertise(_report_sat_info_pub);
344  }
345 
346  PX4_INFO("exiting");
347 
348  /* tell the dtor that we are exiting */
349  _task = -1;
350 }

511 int
512 gpssim_main(int argc, char *argv[])
513 {
514  // set to default
515  const char *device_name = GPS_DEFAULT_UART_PORT;
516  bool fake_gps = false;
517  bool enable_sat_info = false;
518  int fix_type = -1;
519  int num_sat = -1;
520  int noise_multiplier = 0;
521 
522  // check for optional arguments
523  int ch;
524  int myoptind = 1;
525  const char *myoptarg = nullptr;
526 
527  while ((ch = px4_getopt(argc, argv, "d:fst:n:m:", &myoptind, &myoptarg)) != EOF) {
528  switch (ch) {
529  case 'd':
530  device_name = myoptarg;
531  PX4_INFO("Using device %s", device_name);
532  break;
533 
534  case 'f':
535  fake_gps = true;
536  PX4_INFO("Using fake GPS");
537  break;
538 
539  case 's':
540  enable_sat_info = true;
541  PX4_INFO("Satellite info enabled");
542  break;
543 
544  case 't':
545  fix_type = atoi(myoptarg);
546  PX4_INFO("Setting fix_type to %d", fix_type);
547  break;
548 
549  case 'n':
550  num_sat = atoi(myoptarg);
551  PX4_INFO("Setting number of satellites to %d", num_sat);
552  break;
553 
554  case 'm':
555  noise_multiplier = atoi(myoptarg);
556  PX4_INFO("Setting noise multiplier to %d", noise_multiplier);
557  break;
558 
559  default:
560  PX4_WARN("Unknown option!");
561  }
562  }
563 
564  if (myoptind >= argc) {
565  gpssim::usage("not enough arguments supplied");
566  return 1;
567  }
568 
569  /*
570  * Start/load the driver.
571  */
572  if (!strcmp(argv[myoptind], "start")) {
573  if (g_dev != nullptr) {
574  PX4_WARN("already started");
575  return 0;
576  }
577 
578  gpssim::start(device_name, fake_gps, enable_sat_info, fix_type, num_sat, noise_multiplier);
579  return 0;
580  }
581 
582  /* The following need gpssim running. */
583  if (g_dev == nullptr) {
584  PX4_WARN("not running");
585  return 1;
586  }
587 
588  if (!strcmp(argv[myoptind], "stop")) {
589  gpssim::stop();
590  }
591 
592  /*
593  * Test the driver/device.
594  */
595  if (!strcmp(argv[myoptind], "test")) {
596  gpssim::test();
597  }
598 
599  /*
600  * Reset the driver.
601  */
602  if (!strcmp(argv[myoptind], "reset")) {
603  gpssim::reset();
604  }
605 
606  /*
607  * Print driver status.
608  */
609  if (!strcmp(argv[myoptind], "status")) {
610  gpssim::info();
611  }
612 
613  /*
614  * Set parameters
615  */
616  if (!strcmp(argv[myoptind], "set")) {
617  g_dev->set(fix_type, num_sat, noise_multiplier);
618  }
619 
620  return 0;
621 }
