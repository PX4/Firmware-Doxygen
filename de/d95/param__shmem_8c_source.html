618 /**
619  * Automatically save the parameters after a timeout and limited rate.
620  *
621  * This needs to be called with the writer lock held (it's not necessary that it's the writer lock, but it
622  * needs to be the same lock as autosave_worker() and param_control_autosave() use).
623  */
624 static void param_autosave(void)
625 {
626  if (autosave_scheduled || autosave_disabled) {
627  return;
628  }
629 
630  // wait at least 300ms before saving, because:
631  // - tasks often call param_set() for multiple params, so this avoids unnecessary save calls
632  // - the logger stores changed params. He gets notified on a param change via uORB and then
633  // looks at all unsaved params.
634  hrt_abstime delay = 300 * 1000;
635 
636  const hrt_abstime rate_limit = 2000 * 1000; // rate-limit saving to 2 seconds
637  hrt_abstime last_save_elapsed = hrt_elapsed_time(&last_autosave_timestamp);
638 
639  if (last_save_elapsed < rate_limit && rate_limit > last_save_elapsed + delay) {
640  delay = rate_limit - last_save_elapsed;
641  }
642 
643  autosave_scheduled = true;
644  work_queue(LPWORK, &autosave_work, (worker_t)&autosave_worker, NULL, USEC2TICK(delay));
645 }

661 static int
662 param_set_internal(param_t param, const void *val, bool mark_saved, bool notify_changes)
663 {
664  int result = -1;
665  bool params_changed = false;
666 
667  PX4_DEBUG("param_set_internal params: param = %d, val = 0x%X, mark_saved: %d, notify_changes: %d",
668  param, val, (int)mark_saved, (int)notify_changes);
669 
670  param_lock();
671 
672  if (!handle_in_range(param)) {
673  return result;
674  }
675 
676  if (param_values == NULL) {
677  utarray_new(param_values, &param_icd);
678  }
679 
680  if (param_values == NULL) {
681  debug("failed to allocate modified values array");
682  goto out;
683  }
684 
685  if (handle_in_range(param)) {
686 
687  struct param_wbuf_s *s = param_find_changed(param);
688 
689  if (s == NULL) {
690 
691  /* construct a new parameter */
692  struct param_wbuf_s buf = {
693  .param = param,
694  .val.p = NULL,
695  .unsaved = false
696  };
697 
698  /* add it to the array and sort */
699  utarray_push_back(param_values, &buf);
700  utarray_sort(param_values, param_compare_values);
701 
702  /* find it after sorting */
703  s = param_find_changed(param);
704  }
705 
706  /* update the changed value */
707  switch (param_type(param)) {
708 
709  case PARAM_TYPE_INT32:
710  s->val.i = *(int32_t *)val;
711  break;
712 
713  case PARAM_TYPE_FLOAT:
714  s->val.f = *(float *)val;
715  break;
716 
717  case PARAM_TYPE_STRUCT ... PARAM_TYPE_STRUCT_MAX:
718  if (s->val.p == NULL) {
719  s->val.p = malloc(param_size(param));
720 
721  if (s->val.p == NULL) {
722  debug("failed to allocate parameter storage");
723  goto out;
724  }
725  }
726 
727  memcpy(s->val.p, val, param_size(param));
728  break;
729 
730  default:
731  goto out;
732  }
733 
734  s->unsaved = !mark_saved;
735  params_changed = true;
736  result = 0;
737 
738  if (!mark_saved) { // this is false when importing parameters
739  param_autosave();
740  }
741  }
742 
743 out:
744  param_unlock();
745 
746  /*
747  * If we set something, now that we have unlocked, go ahead and advertise that
748  * a thing has been set.
749  */
750 
751  if (!param_import_done) { notify_changes = 0; }
752 
753  if (params_changed && notify_changes) {
754  _param_notify_changes();
755  }
756 
757  if (result == 0 && !set_called_from_get) {
758  update_to_shmem(param, *(union param_value_u *)val);
759  }
760 
761 #ifdef ENABLE_SHMEM_DEBUG
762 
763  if (param_type(param) == PARAM_TYPE_INT32) {
764  PX4_INFO("param_set for %s : %d", param_name(param), ((union param_value_u *)val)->i);
765  }
766 
767  else if (param_type(param) == PARAM_TYPE_FLOAT) {
768  PX4_INFO("param_set for %s : %f", param_name(param), (double)((union param_value_u *)val)->f);
769  }
770 
771  else {
772  PX4_INFO("Unknown param type for %s", param_name(param));
773  }
774 
775 #endif
776 
777  return result;
778 }

1040 int
1041 param_export(int fd, bool only_unsaved)
1042 {
1043  struct param_wbuf_s *s = NULL;
1044  struct bson_encoder_s encoder;
1045  int result = -1;
1046 
1047  int shutdown_lock_ret = px4_shutdown_lock();
1048 
1049  if (shutdown_lock_ret) {
1050  PX4_ERR("px4_shutdown_lock() failed (%i)", shutdown_lock_ret);
1051  }
1052 
1053  param_lock();
1054 
1055  bson_encoder_init_file(&encoder, fd);
1056 
1057  /* no modified parameters -> we are done */
1058  if (param_values == NULL) {
1059  result = 0;
1060  goto out;
1061  }
1062 
1063  /* First of all, update the index which will call param_get for params
1064  * that have recently been changed. */
1065  update_index_from_shmem();
1066 
1067  while ((s = (struct param_wbuf_s *)utarray_next(param_values, s)) != NULL) {
1068 
1069  int32_t i;
1070  float f;
1071 
1072  /*
1073  * If we are only saving values changed since last save, and this
1074  * one hasn't, then skip it
1075  */
1076  if (only_unsaved && !s->unsaved) {
1077  continue;
1078  }
1079 
1080  s->unsaved = false;
1081 
1082  /* Make sure to get latest from shmem before saving. */
1083  update_from_shmem(s->param, &s->val);
1084 
1085  /* append the appropriate BSON type object */
1086 
1087  switch (param_type(s->param)) {
1088 
1089  case PARAM_TYPE_INT32:
1090  i = s->val.i;
1091 
1092  if (bson_encoder_append_int(&encoder, param_name(s->param), i)) {
1093  PX4_DEBUG("BSON append failed for '%s'", param_name(s->param));
1094  goto out;
1095  }
1096 
1097  break;
1098 
1099  case PARAM_TYPE_FLOAT:
1100  f = s->val.f;
1101 
1102  if (bson_encoder_append_double(&encoder, param_name(s->param), f)) {
1103  PX4_DEBUG("BSON append failed for '%s'", param_name(s->param));
1104  goto out;
1105  }
1106 
1107  break;
1108 
1109  case PARAM_TYPE_STRUCT ... PARAM_TYPE_STRUCT_MAX:
1110  if (bson_encoder_append_binary(&encoder,
1111  param_name(s->param),
1112  BSON_BIN_BINARY,
1113  param_size(s->param),
1114  param_get_value_ptr(s->param))) {
1115  PX4_DEBUG("BSON append failed for '%s'", param_name(s->param));
1116  goto out;
1117  }
1118 
1119  break;
1120 
1121  default:
1122  PX4_DEBUG("unrecognized parameter type");
1123  goto out;
1124  }
1125  }
1126 
1127  result = 0;
1128 
1129 out:
1130  param_unlock();
1131 
1132  fsync(fd); // make sure the data is flushed before releasing the shutdown lock
1133 
1134  if (shutdown_lock_ret == 0) {
1135  px4_shutdown_unlock();
1136  }
1137 
1138  if (result == 0) {
1139  result = bson_encoder_fini(&encoder);
1140  }
1141 
1142  return result;
1143 }

1149 static int
1150 param_import_callback(bson_decoder_t decoder, void *private, bson_node_t node)
1151 {
1152  float f;
1153  int32_t i;
1154  void *v, *tmp = NULL;
1155  int result = -1;
1156  struct param_import_state *state = (struct param_import_state *)private;
1157 
1158  /*
1159  * EOO means the end of the parameter object. (Currently not supporting
1160  * nested BSON objects).
1161  */
1162  if (node->type == BSON_EOO) {
1163  PX4_DEBUG("end of parameters");
1164  return 0;
1165  }
1166 
1167  /*
1168  * Find the parameter this node represents. If we don't know it,
1169  * ignore the node.
1170  */
1171  param_t param = param_find_no_notification(node->name);
1172 
1173  if (param == PARAM_INVALID) {
1174  PX4_DEBUG("ignoring unrecognised parameter '%s'", node->name);
1175  return 1;
1176  }
1177 
1178  /*
1179  * Handle setting the parameter from the node
1180  */
1181 
1182  switch (node->type) {
1183  case BSON_INT32:
1184  if (param_type(param) != PARAM_TYPE_INT32) {
1185  PX4_WARN("unexpected type for %s", node->name);
1186  result = 1; // just skip this entry
1187  goto out;
1188  }
1189 
1190  i = node->i;
1191  v = &i;
1192  PX4_DEBUG("Imported %s with value %d", param_name(param), i);
1193  break;
1194 
1195  case BSON_DOUBLE:
1196  if (param_type(param) != PARAM_TYPE_FLOAT) {
1197  PX4_WARN("unexpected type for %s", node->name);
1198  result = 1; // just skip this entry
1199  goto out;
1200  }
1201 
1202  f = node->d;
1203  v = &f;
1204  PX4_DEBUG("Imported %s with value %f", param_name(param), (double)f);
1205  break;
1206 
1207  case BSON_BINDATA:
1208  if (node->subtype != BSON_BIN_BINARY) {
1209  PX4_WARN("unexpected type for %s", node->name);
1210  result = 1; // just skip this entry
1211  goto out;
1212  }
1213 
1214  if (bson_decoder_data_pending(decoder) != param_size(param)) {
1215  PX4_WARN("bad size for '%s'", node->name);
1216  result = 1; // just skip this entry
1217  goto out;
1218  }
1219 
1220  /* XXX check actual file data size? */
1221  tmp = malloc(param_size(param));
1222 
1223  if (tmp == NULL) {
1224  PX4_DEBUG("failed allocating for '%s'", node->name);
1225  goto out;
1226  }
1227 
1228  if (bson_decoder_copy_data(decoder, tmp)) {
1229  PX4_DEBUG("failed copying data for '%s'", node->name);
1230  goto out;
1231  }
1232 
1233  v = tmp;
1234  break;
1235 
1236  default:
1237  PX4_DEBUG("unrecognised node type");
1238  goto out;
1239  }
1240 
1241  if (param_set_internal(param, v, state->mark_saved, true)) {
1242  PX4_DEBUG("error setting value for '%s'", node->name);
1243  goto out;
1244  }
1245 
1246  if (tmp != NULL) {
1247  free(tmp);
1248  tmp = NULL;
1249  }
1250 
1251  /* don't return zero, that means EOF */
1252  result = 1;
1253 
1254 out:
1255 
1256  if (tmp != NULL) {
1257  free(tmp);
1258  }
1259 
1260  return result;
1261 }
