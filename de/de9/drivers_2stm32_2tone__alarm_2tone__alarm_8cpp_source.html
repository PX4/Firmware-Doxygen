34 /**
35  * Driver for the PX4 audio alarm port, /dev/tone_alarm.
36  *
37  * The tone_alarm driver supports a set of predefined "alarm"
38  * tunes and one user-supplied tune.
39  *
40  * The TONE_SET_ALARM ioctl can be used to select a predefined
41  * alarm tune, from 1 - <TBD>. Selecting tune zero silences
42  * the alarm.
43  *
44  * Tunes follow the syntax of the Microsoft GWBasic/QBasic PLAY
45  * statement, with some exceptions and extensions.
46  *
47  * From Wikibooks:
48  *
49  * PLAY "[string expression]"
50  *
51  * Used to play notes and a score ... The tones are indicated by letters A through G.
52  * Accidentals are indicated with a "+" or "#" (for sharp) or "-" (for flat)
53  * immediately after the note letter. See this example:
54  *
55  * PLAY "C C# C C#"
56  *
57  * Whitespaces are ignored inside the string expression. There are also codes that
58  * set the duration, octave and tempo. They are all case-insensitive. PLAY executes
59  * the commands or notes the order in which they appear in the string. Any indicators
60  * that change the properties are effective for the notes following that indicator.
61  *
62  * Ln Sets the duration (length) of the notes. The variable n does not indicate an actual duration
63  * amount but rather a note type; L1 - whole note, L2 - half note, L4 - quarter note, etc.
64  * (L8, L16, L32, L64, ...). By default, n = 4.
65  * For triplets and quintets, use L3, L6, L12, ... and L5, L10, L20, ... series respectively.
66  * The shorthand notation of length is also provided for a note. For example, "L4 CDE L8 FG L4 AB"
67  * can be shortened to "L4 CDE F8G8 AB". F and G play as eighth notes while others play as quarter notes.
68  * On Sets the current octave. Valid values for n are 0 through 6. An octave begins with C and ends with B.
69  * Remember that C- is equivalent to B.
70  * < > Changes the current octave respectively down or up one level.
71  * Nn Plays a specified note in the seven-octave range. Valid values are from 0 to 84. (0 is a pause.)
72  * Cannot use with sharp and flat. Cannot use with the shorthand notation neither.
73  * MN Stand for Music Normal. Note duration is 7/8ths of the length indicated by Ln. It is the default mode.
74  * ML Stand for Music Legato. Note duration is full length of that indicated by Ln.
75  * MS Stand for Music Staccato. Note duration is 3/4ths of the length indicated by Ln.
76  * Pn Causes a silence (pause) for the length of note indicated (same as Ln).
77  * Tn Sets the number of "L4"s per minute (tempo). Valid values are from 32 to 255. The default value is T120.
78  * . When placed after a note, it causes the duration of the note to be 3/2 of the set duration.
79  * This is how to get "dotted" notes. "L4 C#." would play C sharp as a dotted quarter note.
80  * It can be used for a pause as well.
81  *
82  * Extensions/variations:
83  *
84  * MB MF The MF command causes the tune to play once and then stop. The MB command causes the
85  * tune to repeat when it ends.
86  *
87  */

387 int ToneAlarm::init()
388 {
389  int ret;
390 
391  ret = CDev::init();
392 
393  if (ret != OK) {
394  return ret;
395  }
396 
397  /* configure the GPIO to the idle state */
398  px4_arch_configgpio(GPIO_TONE_ALARM_IDLE);
399 
400 #ifdef GPIO_TONE_ALARM_NEG
401 
402  px4_arch_configgpio(GPIO_TONE_ALARM_NEG);
403 #endif
404 
405  /* clock/power on our timer */
406  modifyreg32(TONE_ALARM_CLOCK_POWER_REG, 0, TONE_ALARM_CLOCK_ENABLE);
407 
408  /* initialise the timer */
409  rCR1 = 0;
410  rCR2 = 0;
411  rSMCR = 0;
412  rDIER = 0;
413  rCCER &= TONE_CCER; /* unlock CCMR* registers */
414  rCCMR1 = TONE_CCMR1;
415  rCCMR2 = TONE_CCMR2;
416  rCCER = TONE_CCER;
417  rDCR = 0;
418 
419 #ifdef rBDTR // If using an advanced timer, you need to activate the output
420  rBDTR = ATIM_BDTR_MOE; // enable the main output of the advanced timer
421 #endif
422 
423  /* toggle the CC output each time the count passes 1 */
424  TONE_rCCR = 1;
425 
426  /* default the timer to a prescale value of 1; playing notes will change this */
427  rPSC = 0;
428 
429  /* make sure the timer is running */
430  rCR1 = GTIM_CR1_CEN;
431 
432  DEVICE_DEBUG("ready");
433 
434  _running = true;
435  work_queue(HPWORK, &_work, (worker_t)&ToneAlarm::next_trampoline, this, 0);
436  return OK;
437 }

459 void ToneAlarm::start_note(unsigned frequency)
460 {
461  // check if circuit breaker is enabled
462  if (_cbrk == CBRK_UNINIT) {
463  _cbrk = circuit_breaker_enabled("CBRK_BUZZER", CBRK_BUZZER_KEY);
464  }
465 
466  if (_cbrk != CBRK_OFF) { return; }
467 
468  // compute the divisor
469  unsigned divisor = frequency_to_divisor(frequency);
470 
471  // pick the lowest prescaler value that we can use
472  // (note that the effective prescale value is 1 greater)
473  unsigned prescale = divisor / 65536;
474 
475  // calculate the timer period for the selected prescaler value
476  unsigned period = (divisor / (prescale + 1)) - 1;
477 
478  rPSC = prescale; // load new prescaler
479  rARR = period; // load new toggle period
480  rEGR = GTIM_EGR_UG; // force a reload of the period
481  rCCER |= TONE_CCER; // enable the output
482 
483  // configure the GPIO to enable timer output
484  px4_arch_configgpio(GPIO_TONE_ALARM);
485 }

498 void ToneAlarm::next_note()
499 {
500  if (!_should_run) {
501  if (_tune_control_sub >= 0) {
502  orb_unsubscribe(_tune_control_sub);
503  }
504 
505  _running = false;
506  return;
507  }
508 
509  // subscribe to tune_control
510  if (_tune_control_sub < 0) {
511  _tune_control_sub = orb_subscribe(ORB_ID(tune_control));
512  }
513 
514  // do we have an inter-note gap to wait for?
515  if (_silence_length > 0) {
516  stop_note();
517  work_queue(HPWORK, &_work, (worker_t)&ToneAlarm::next_trampoline, this, USEC2TICK(_silence_length));
518  _silence_length = 0;
519  return;
520  }
521 
522  // check for updates
523  bool updated = false;
524  orb_check(_tune_control_sub, &updated);
525 
526  if (updated) {
527  orb_copy(ORB_ID(tune_control), _tune_control_sub, &_tune);
528 
529  if (_tunes.set_control(_tune) == 0) {
530  _play_tone = true;
531 
532  } else {
533  _play_tone = false;
534  }
535  }
536 
537  unsigned frequency = 0, duration = 0;
538 
539  if (_play_tone) {
540  _play_tone = false;
541  int parse_ret_val = _tunes.get_next_tune(frequency, duration, _silence_length);
542 
543  if (parse_ret_val >= 0) {
544  // a frequency of 0 correspond to stop_note
545  if (frequency > 0) {
546  // start playing the note
547  start_note(frequency);
548 
549  } else {
550  stop_note();
551  }
552 
553 
554  if (parse_ret_val > 0) {
555  // continue playing
556  _play_tone = true;
557  }
558  }
559 
560  } else {
561  // schedule a call with the tunes max interval
562  duration = _tunes.get_maximum_update_interval();
563  // stop playing the last note after the duration elapsed
564  stop_note();
565  }
566 
567  // and arrange a callback when the note should stop
568  work_queue(HPWORK, &_work, (worker_t)&ToneAlarm::next_trampoline, this, USEC2TICK(duration));
569 }
