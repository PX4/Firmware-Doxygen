234  struct vehicle_status_s status = {};
235  struct vehicle_status_flags_s status_flags = {};
236  struct safety_s safety = {};
237  struct actuator_armed_s armed = {};
238  struct battery_status_s battery = {};
239 
240  size_t cArmingTransitionTests = sizeof(rgArmingTransitionTests) / sizeof(rgArmingTransitionTests[0]);
241  for (size_t i=0; i<cArmingTransitionTests; i++) {
242  const ArmingTransitionTest_t* test = &rgArmingTransitionTests[i];
243 
244  const bool check_gps = false;
245 
246  // Setup initial machine state
247  status.arming_state = test->current_state.arming_state;
248  status_flags.condition_system_sensors_initialized = test->condition_system_sensors_initialized;
249  status.hil_state = test->hil_state;
250  // The power status of the test unit is not relevant for the unit test
251  status_flags.circuit_breaker_engaged_power_check = true;
252  safety.safety_switch_available = test->safety_switch_available;
253  safety.safety_off = test->safety_off;
254  armed.armed = test->current_state.armed;
255  armed.ready_to_arm = test->current_state.ready_to_arm;
256 
257  // Attempt transition
258  transition_result_t result = arming_state_transition(&status, battery, safety, test->requested_state, &armed,
259  false /* no pre-arm checks */,
260  nullptr /* no mavlink_log_pub */,
261  &status_flags,
262  (check_gps ? ARM_REQ_GPS_BIT : 0),
263  2e6 /* 2 seconds after boot, everything should be checked */
264  );
265 
266  // Validate result of transition
267  ut_compare(test->assertMsg, test->expected_transition_result, result);
268  ut_compare(test->assertMsg, status.arming_state, test->expected_state.arming_state);
269  ut_compare(test->assertMsg, armed.armed, test->expected_state.armed);
270  ut_compare(test->assertMsg, armed.ready_to_arm, test->expected_state.ready_to_arm);
271  }
272 
273  return true;
