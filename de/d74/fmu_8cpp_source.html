445 void
446 PX4FMU::safety_check_button(void)
447 {
448 #ifdef GPIO_BTN_SAFETY
449 
450  if (!PX4_MFT_HW_SUPPORTED(PX4_MFT_PX4IO)) {
451 
452  static int counter = 0;
453  /*
454  * Debounce the safety button, change state if it has been held for long enough.
455  *
456  */
457  bool safety_button_pressed = px4_arch_gpioread(GPIO_BTN_SAFETY);
458 
459  /*
460  * Keep pressed for a while to arm.
461  *
462  * Note that the counting sequence has to be same length
463  * for arming / disarming in order to end up as proper
464  * state machine, keep ARM_COUNTER_THRESHOLD the same
465  * length in all cases of the if/else struct below.
466  */
467  if (safety_button_pressed && !_safety_btn_off) {
468 
469  if (counter < CYCLE_COUNT) {
470  counter++;
471 
472  } else if (counter == CYCLE_COUNT) {
473  /* switch to armed state */
474  _safety_btn_off = true;
475  counter++;
476  }
477 
478  } else if (safety_button_pressed && _safety_btn_off) {
479 
480  if (counter < CYCLE_COUNT) {
481  counter++;
482 
483  } else if (counter == CYCLE_COUNT) {
484  /* change to disarmed state and notify the FMU */
485  _safety_btn_off = false;
486  counter++;
487  }
488 
489  } else {
490  counter = 0;
491  }
492  }
493 
494 #endif
495 }

497 void
498 PX4FMU::flash_safety_button()
499 {
500 #if defined(GPIO_BTN_SAFETY) && defined(GPIO_LED_SAFETY)
501 
502  if (!PX4_MFT_HW_SUPPORTED(PX4_MFT_PX4IO)) {
503  /* Select the appropriate LED flash pattern depending on the current arm state */
504  uint16_t pattern = LED_PATTERN_FMU_REFUSE_TO_ARM;
505 
506  /* cycle the blink state machine at 10Hz */
507  static int blink_counter = 0;
508 
509  if (_safety_btn_off) {
510  if (_armed.armed) {
511  pattern = LED_PATTERN_IO_FMU_ARMED;
512 
513  } else {
514  pattern = LED_PATTERN_IO_ARMED;
515  }
516 
517  } else if (_armed.armed) {
518  pattern = LED_PATTERN_FMU_ARMED;
519 
520  } else {
521  pattern = LED_PATTERN_FMU_OK_TO_ARM;
522 
523  }
524 
525  /* Turn the LED on if we have a 1 at the current bit position */
526  px4_arch_gpiowrite(GPIO_LED_SAFETY, !(pattern & (1 << blink_counter++)));
527 
528  if (blink_counter > 15) {
529  blink_counter = 0;
530  }
531  }
532 
533 #endif

688 /* When set_pwm_rate is called from either of the 2 IOCTLs:
689  *
690  * PWM_SERVO_SET_UPDATE_RATE - Sets the "alternate" channel's rate to the callers's rate specified
691  * and the non "alternate" channels to the _pwm_default_rate.
692  *
693  * rate_map = _pwm_alt_rate_channels
694  * default_rate = _pwm_default_rate
695  * alt_rate = arg of IOCTL (see rates)
696  *
697  * PWM_SERVO_SET_SELECT_UPDATE_RATE - The caller's specified rate map selects the "alternate" channels
698  * to be set to the alt rate. (_pwm_alt_rate)
699  * All other channels are set to the default rate. (_pwm_default_rate)
700  *
701  * rate_map = arg of IOCTL
702  * default_rate = _pwm_default_rate
703  * alt_rate = _pwm_alt_rate
704 
705  * rate_map - A mask of 1's for the channels to be set to the
706  * alternate rate.
707  * N.B. All channels is a given group must be set
708  * to the same rate/mode. (default or alt)
709  * rates:
710  * alt_rate, default_rate For PWM is 25 or 400Hz
711  * For Oneshot there is no rate, 0 is therefore used
712  * to select Oneshot mode
713  */
714 int
715 PX4FMU::set_pwm_rate(uint32_t rate_map, unsigned default_rate, unsigned alt_rate)
716 {
717  PX4_DEBUG("set_pwm_rate %x %u %u", rate_map, default_rate, alt_rate);
718 
719  for (unsigned pass = 0; pass < 2; pass++) {
720 
721  /* We should note that group is iterated over from 0 to _max_actuators.
722  * This allows for the ideal worlds situation: 1 channel per group
723  * configuration.
724  *
725  * This is typically not what HW supports. A group represents a timer
726  * and channels belongs to a timer.
727  * Therefore all channels in a group are dependent on the timer's
728  * common settings and can not be independent in terms of count frequency
729  * (granularity of pulse width) and rate (period of repetition).
730  *
731  * To say it another way, all channels in a group moust have the same
732  * rate and mode. (See rates above.)
733  */
734 
735  for (unsigned group = 0; group < _max_actuators; group++) {
736 
737  // get the channel mask for this rate group
738  uint32_t mask = up_pwm_servo_get_rate_group(group);
739 
740  if (mask == 0) {
741  continue;
742  }
743 
744  // all channels in the group must be either default or alt-rate
745  uint32_t alt = rate_map & mask;
746 
747  if (pass == 0) {
748  // preflight
749  if ((alt != 0) && (alt != mask)) {
750  PX4_WARN("rate group %u mask %x bad overlap %x", group, mask, alt);
751  // not a legal map, bail
752  return -EINVAL;
753  }
754 
755  } else {
756  // set it - errors here are unexpected
757  if (alt != 0) {
758  if (up_pwm_servo_set_rate_group_update(group, alt_rate) != OK) {
759  PX4_WARN("rate group set alt failed");
760  return -EINVAL;
761  }
762 
763  } else {
764  if (up_pwm_servo_set_rate_group_update(group, default_rate) != OK) {
765  PX4_WARN("rate group set default failed");
766  return -EINVAL;
767  }
768  }
769  }
770  }
771  }
772 
773  _pwm_alt_rate_channels = rate_map;
774  _pwm_default_rate = default_rate;
775  _pwm_alt_rate = alt_rate;
776 
777  return OK;
778 }

1036 void
1037 PX4FMU::cycle()
1038 {
1039  while (true) {
1040 
1041  if (_groups_subscribed != _groups_required) {
1042  subscribe();
1043  _groups_subscribed = _groups_required;
1044  /* force setting update rate */
1045  _current_update_rate = 0;
1046  }
1047 
1048  int poll_timeout = 10;
1049 
1050  if (!_run_as_task) {
1051  /*
1052  * Adjust actuator topic update rate to keep up with
1053  * the highest servo update rate configured.
1054  *
1055  * We always mix at max rate; some channels may update slower.
1056  */
1057  unsigned max_rate = (_pwm_default_rate > _pwm_alt_rate) ? _pwm_default_rate : _pwm_alt_rate;
1058 
1059  if (_current_update_rate != max_rate) {
1060  _current_update_rate = max_rate;
1061  int update_rate_in_ms = int(1000 / _current_update_rate);
1062 
1063  /* reject faster than 500 Hz updates */
1064  if (update_rate_in_ms < 2) {
1065  update_rate_in_ms = 2;
1066  }
1067 
1068  /* reject slower than 10 Hz updates */
1069  if (update_rate_in_ms > 100) {
1070  update_rate_in_ms = 100;
1071  }
1072 
1073  PX4_DEBUG("adjusted actuator update interval to %ums", update_rate_in_ms);
1074 
1075  for (unsigned i = 0; i < actuator_controls_s::NUM_ACTUATOR_CONTROL_GROUPS; i++) {
1076  if (_control_subs[i] >= 0) {
1077  orb_set_interval(_control_subs[i], update_rate_in_ms);
1078  }
1079  }
1080 
1081  // set to current max rate, even if we are actually checking slower/faster
1082  _current_update_rate = max_rate;
1083  }
1084 
1085  /* check if anything updated */
1086  poll_timeout = 0;
1087  }
1088 
1089  /* wait for an update */
1090  unsigned n_updates = 0;
1091  int ret = px4_poll(_poll_fds, _poll_fds_num, poll_timeout);
1092 
1093  /* this would be bad... */
1094  if (ret < 0) {
1095  DEVICE_LOG("poll error %d", errno);
1096 
1097  } else if (ret == 0) {
1098  /* timeout: no control data, switch to failsafe values */
1099  // PX4_WARN("no PWM: failsafe");
1100 
1101  } else {
1102  if (_mixers != nullptr) {
1103  /* get controls for required topics */
1104  unsigned poll_id = 0;
1105 
1106  for (unsigned i = 0; i < actuator_controls_s::NUM_ACTUATOR_CONTROL_GROUPS; i++) {
1107  if (_control_subs[i] >= 0) {
1108 
1109  if (_poll_fds[poll_id].revents & POLLIN) {
1110  if (i == 0) {
1111  n_updates++;
1112  }
1113 
1114  orb_copy(_control_topics[i], _control_subs[i], &_controls[i]);
1115  }
1116 
1117  poll_id++;
1118  }
1119 
1120  /* During ESC calibration, we overwrite the throttle value. */
1121  if (i == 0 && _armed.in_esc_calibration_mode) {
1122 
1123  /* Set all controls to 0 */
1124  memset(&_controls[i], 0, sizeof(_controls[i]));
1125 
1126  /* except thrust to maximum. */
1127  _controls[i].control[actuator_controls_s::INDEX_THROTTLE] = 1.0f;
1128 
1129  /* Switch off the PWM limit ramp for the calibration. */
1130  _pwm_limit.state = PWM_LIMIT_STATE_ON;
1131  }
1132  }
1133  }
1134  } // poll_fds
1135 
1136  /* run the mixers on every cycle */
1137  {
1138  if (_mixers != nullptr) {
1139 
1140  if (_mot_t_max > FLT_EPSILON) {
1141  hrt_abstime now = hrt_absolute_time();
1142  float dt = (now - _time_last_mix) / 1e6f;
1143  _time_last_mix = now;
1144 
1145  if (dt < 0.0001f) {
1146  dt = 0.0001f;
1147 
1148  } else if (dt > 0.02f) {
1149  dt = 0.02f;
1150  }
1151 
1152  // maximum value the outputs of the multirotor mixer are allowed to change in this cycle
1153  // factor 2 is needed because actuator outputs are in the range [-1,1]
1154  const float delta_out_max = 2.0f * 1000.0f * dt / (_max_pwm[0] - _min_pwm[0]) / _mot_t_max;
1155  _mixers->set_max_delta_out_once(delta_out_max);
1156  }
1157 
1158  if (_thr_mdl_fac > FLT_EPSILON) {
1159  _mixers->set_thrust_factor(_thr_mdl_fac);
1160  }
1161 
1162  /* do mixing */
1163  float outputs[_max_actuators];
1164  const unsigned mixed_num_outputs = _mixers->mix(outputs, _num_outputs);
1165 
1166  /* the PWM limit call takes care of out of band errors, NaN and constrains */
1167  uint16_t pwm_limited[MAX_ACTUATORS];
1168 
1169  pwm_limit_calc(_throttle_armed, arm_nothrottle(), mixed_num_outputs, _reverse_pwm_mask,
1170  _disarmed_pwm, _min_pwm, _max_pwm, outputs, pwm_limited, &_pwm_limit);
1171 
1172  /* overwrite outputs in case of force_failsafe with _failsafe_pwm PWM values */
1173  if (_armed.force_failsafe) {
1174  for (size_t i = 0; i < mixed_num_outputs; i++) {
1175  pwm_limited[i] = _failsafe_pwm[i];
1176  }
1177  }
1178 
1179  /* overwrite outputs in case of lockdown with disarmed PWM values */
1180  if (_armed.lockdown || _armed.manual_lockdown) {
1181  for (size_t i = 0; i < mixed_num_outputs; i++) {
1182  pwm_limited[i] = _disarmed_pwm[i];
1183  }
1184  }
1185 
1186  /* apply _motor_ordering */
1187  reorder_outputs(pwm_limited);
1188 
1189  /* output to the servos */
1190  if (_pwm_initialized) {
1191  for (size_t i = 0; i < mixed_num_outputs; i++) {
1192  up_pwm_servo_set(i, pwm_limited[i]);
1193  }
1194  }
1195 
1196  /* Trigger all timer's channels in Oneshot mode to fire
1197  * the oneshots with updated values.
1198  */
1199  if (n_updates > 0) {
1200  up_pwm_update();
1201  }
1202 
1203  actuator_outputs_s actuator_outputs = {};
1204  actuator_outputs.timestamp = hrt_absolute_time();
1205  actuator_outputs.noutputs = mixed_num_outputs;
1206 
1207  // zero unused outputs
1208  for (size_t i = 0; i < mixed_num_outputs; ++i) {
1209  actuator_outputs.output[i] = pwm_limited[i];
1210  }
1211 
1212  orb_publish_auto(ORB_ID(actuator_outputs), &_outputs_pub, &actuator_outputs, &_class_instance, ORB_PRIO_DEFAULT);
1213 
1214  /* publish mixer status */
1215  MultirotorMixer::saturation_status saturation_status;
1216  saturation_status.value = _mixers->get_saturation_status();
1217 
1218  if (saturation_status.flags.valid) {
1219  multirotor_motor_limits_s motor_limits;
1220  motor_limits.timestamp = hrt_absolute_time();
1221  motor_limits.saturation_status = saturation_status.value;
1222 
1223  orb_publish_auto(ORB_ID(multirotor_motor_limits), &_to_mixer_status, &motor_limits, &_class_instance, ORB_PRIO_DEFAULT);
1224  }
1225 
1226  _mixers->set_airmode(_airmode);
1227 
1228  // use first valid timestamp_sample for latency tracking
1229  for (int i = 0; i < actuator_controls_s::NUM_ACTUATOR_CONTROL_GROUPS; i++) {
1230  const bool required = _groups_required & (1 << i);
1231  const hrt_abstime &timestamp_sample = _controls[i].timestamp_sample;
1232 
1233  if (required && (timestamp_sample > 0)) {
1234  perf_set_elapsed(_perf_control_latency, actuator_outputs.timestamp - timestamp_sample);
1235  break;
1236  }
1237  }
1238  }
1239  }
1240 
1241  _cycle_timestamp = hrt_absolute_time();
1242 
1243 #ifdef GPIO_BTN_SAFETY
1244 
1245  if (!PX4_MFT_HW_SUPPORTED(PX4_MFT_PX4IO)) {
1246 
1247  if (_cycle_timestamp - _last_safety_check >= (unsigned int)1e5) {
1248  _last_safety_check = _cycle_timestamp;
1249 
1250  /**
1251  * Get and handle the safety status at 10Hz
1252  */
1253  struct safety_s safety = {};
1254 
1255  if (!_safety_disabled) {
1256  /* read safety switch input and control safety switch LED at 10Hz */
1257  safety_check_button();
1258  }
1259 
1260  /* Make the safety button flash anyway, no matter if it's used or not. */
1261  flash_safety_button();
1262 
1263  safety.timestamp = hrt_absolute_time();
1264  safety.safety_switch_available = true;
1265  safety.safety_off = _safety_btn_off;
1266 
1267  /* lazily publish the safety status */
1268  if (_to_safety != nullptr) {
1269  orb_publish(ORB_ID(safety), _to_safety, &safety);
1270 
1271  } else {
1272  int instance;
1273  _to_safety = orb_advertise_multi(ORB_ID(safety), &safety, &instance, ORB_PRIO_DEFAULT);
1274  }
1275  }
1276  }
1277 
1278 #endif
1279 
1280  /* check safety button state */
1281  bool updated = false;
1282  orb_check(_safety_sub, &updated);
1283 
1284  if (updated) {
1285  safety_s safety;
1286 
1287  if (orb_copy(ORB_ID(actuator_armed), _safety_sub, &safety) == 0) {
1288  _safety_off = !safety.safety_switch_available || safety.safety_off;
1289  }
1290  }
1291 
1292  /* check arming state */
1293  orb_check(_armed_sub, &updated);
1294 
1295  if (updated) {
1296  orb_copy(ORB_ID(actuator_armed), _armed_sub, &_armed);
1297 
1298  /* Update the armed status and check that we're not locked down.
1299  * We also need to arm throttle for the ESC calibration. */
1300  _throttle_armed = (_safety_off && _armed.armed && !_armed.lockdown) ||
1301  (_safety_off && _armed.in_esc_calibration_mode);
1302  }
1303 
1304  /* update PWM status if armed or if disarmed PWM values are set */
1305  bool pwm_on = _armed.armed || _num_disarmed_set > 0 || _armed.in_esc_calibration_mode;
1306 
1307  if (_pwm_on != pwm_on) {
1308  _pwm_on = pwm_on;
1309 
1310  update_pwm_out_state(pwm_on);
1311  }
1312 
1313  orb_check(_param_sub, &updated);
1314 
1315  if (updated) {
1316  this->update_params();
1317  }
1318 
1319  if (_run_as_task) {
1320  if (should_exit()) {
1321  break;
1322  }
1323 
1324  } else {
1325  if (should_exit()) {
1326  exit_and_cleanup();
1327 
1328  } else {
1329  /* schedule next cycle */
1330  work_queue(HPWORK, &_work, (worker_t)&PX4FMU::cycle_trampoline, this, USEC2TICK(SCHEDULE_INTERVAL));
1331  }
1332 
1333  break;
1334  }
1335  }
1336 }
