60 /**
61  * Start the driver.
62  *
63  * This function only returns if the driver is up and running
64  * or failed to detect the sensor.
65  */
66 void start(bool external_bus, enum Rotation rotation)
67 {
68  int fd;
69  BMM150 **g_dev_ptr = (external_bus ? &g_dev_ext : &g_dev_int);
70  const char *path = (external_bus ? BMM150_DEVICE_PATH_MAG_EXT : BMM150_DEVICE_PATH_MAG);
71 
72  if (*g_dev_ptr != nullptr)
73  /* if already started, the still command succeeded */
74  {
75  PX4_ERR("already started");
76  exit(0);
77  }
78 
79  /* create the driver */
80  if (external_bus) {
81 #if defined(PX4_I2C_BUS_EXPANSION)
82  *g_dev_ptr = new BMM150(PX4_I2C_BUS_EXPANSION, path, rotation);
83 #else
84  PX4_ERR("External I2C not available");
85  exit(0);
86 #endif
87 
88  } else {
89  PX4_ERR("Internal I2C not available");
90  exit(0);
91  }
92 
93  if (*g_dev_ptr == nullptr) {
94  goto fail;
95  }
96 
97 
98  if (OK != (*g_dev_ptr)->init()) {
99  goto fail;
100  }
101 
102  /* set the poll rate to default, starts automatic data collection */
103  fd = open(path, O_RDONLY);
104 
105 
106  if (fd < 0) {
107  goto fail;
108  }
109 
110  if (ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
111  goto fail;
112  }
113 
114  close(fd);
115 
116  exit(0);
117 fail:
118 
119  if (*g_dev_ptr != nullptr) {
120  delete (*g_dev_ptr);
121  *g_dev_ptr = nullptr;
122  }
123 
124  PX4_ERR("driver start failed");
125  exit(1);
126 
127 }

327 int BMM150::init()
328 {
329  int ret = OK;
330 
331  /* do I2C init (and probe) first */
332  ret = I2C::init();
333 
334  /* if probe/setup failed, bail now */
335  if (ret != OK) {
336  DEVICE_DEBUG("I2C setup failed");
337  return ret;
338  }
339 
340  /* allocate basic report buffers */
341  _reports = new ringbuffer::RingBuffer(2, sizeof(mag_report));
342 
343  if (_reports == nullptr) {
344  goto out;
345  }
346 
347  /* Bring the device to sleep mode */
348  modify_reg(BMM150_POWER_CTRL_REG, 1, 1);
349  up_udelay(10000);
350 
351 
352  /* check id*/
353  if (read_reg(BMM150_CHIP_ID_REG) != BMM150_CHIP_ID) {
354  PX4_WARN("id of magnetometer is not: 0x%02x", BMM150_CHIP_ID);
355  return -EIO;
356  }
357 
358  if (reset() != OK) {
359  goto out;
360  }
361 
362  init_trim_registers();
363 
364  _class_instance = register_class_devname(MAG_BASE_DEVICE_PATH);
365 
366  if (measure()) {
367  return -EIO;
368  }
369 
370  up_udelay(10000);
371 
372  if (collect()) {
373  return -EIO;
374  }
375 
376  /* advertise sensor topic, measure manually to initialize valid report */
377  struct mag_report mrb;
378  _reports->get(&mrb);
379 
380  /* measurement will have generated a report, publish */
381  _topic = orb_advertise_multi(ORB_ID(sensor_mag), &mrb,
382  &_orb_class_instance, (external()) ? ORB_PRIO_HIGH : ORB_PRIO_MAX);
383 
384  if (_topic == nullptr) {
385  PX4_WARN("ADVERT FAIL");
386  }
387 
388 out:
389  return ret;
390 }

427 ssize_t
428 BMM150::read(struct file *filp, char *buffer, size_t buflen)
429 {
430  unsigned count = buflen / sizeof(mag_report);
431  struct mag_report *mag_buf = reinterpret_cast<struct mag_report *>(buffer);
432  int ret = 0;
433 
434  /* buffer must be large enough */
435  if (count < 1) {
436  return -ENOSPC;
437  }
438 
439  /* if automatic measurement is enabled */
440  if (_call_interval > 0) {
441  /*
442  * While there is space in the caller's buffer, and reports, copy them.
443  * Note that we may be pre-empted by the workq thread while we are doing this;
444  * we are careful to avoid racing with them.
445  */
446  while (count--) {
447  if (_reports->get(mag_buf)) {
448  ret += sizeof(struct mag_report);
449  mag_buf++;
450  }
451  }
452 
453  /* if there was no data, warn the caller */
454  return ret ? ret : -EAGAIN;
455  }
456 
457  /* manual measurement - run one conversion */
458  /* XXX really it'd be nice to lock against other readers here */
459  do {
460  _reports->flush();
461 
462  /* trigger a measurement */
463  if (OK != measure()) {
464  ret = -EIO;
465  break;
466  }
467 
468  /* wait for it to complete */
469  usleep(BMM150_CONVERSION_INTERVAL);
470 
471  /* run the collection phase */
472  if (OK != collect()) {
473  ret = -EIO;
474  break;
475  }
476 
477 
478  if (_reports->get(mag_buf)) {
479  ret = sizeof(struct mag_report);
480  }
481  } while (0);
482 
483  /* return the number of bytes transferred */
484  return ret;
485

498 void
499 BMM150::cycle()
500 {
501  if (_collect_phase) {
502  collect();
503  unsigned wait_gap = _call_interval - USEC2TICK(BMM150_CONVERSION_INTERVAL);
504 
505  if ((wait_gap != 0) && (_running)) {
506  work_queue(HPWORK, &_work, (worker_t)&BMM150::cycle_trampoline, this,
507  wait_gap); //need to wait some time before new measurement
508  return;
509  }
510 
511  }
512 
513  measure();
514 
515  if ((_running)) {
516  /* schedule a fresh cycle call when the measurement is done */
517  work_queue(HPWORK,
518  &_work,
519  (worker_t)&BMM150::cycle_trampoline,
520  this,
521  USEC2TICK(BMM150_CONVERSION_INTERVAL));
522  }

550 int
551 BMM150::collect()
552 {
553  _collect_phase = false;
554 
555  bool mag_notify = true;
556  uint8_t mag_data[8], status;
557  uint16_t resistance, lsb, msb, msblsb;
558  mag_report mrb;
559 
560 
561  /* start collecting data */
562  perf_begin(_sample_perf);
563 
564  status = read_reg(BMM150_R_LSB);
565 
566  /* Read Magnetometer data*/
567  if (OK != get_data(BMM150_DATA_X_LSB_REG, mag_data, sizeof(mag_data))) {
568  return -EIO;
569  }
570 
571  /* Array holding the mag XYZ and R data
572  mag_data[0] - X LSB
573  mag_data[1] - X MSB
574  mag_data[2] - Y LSB
575  mag_data[3] - Y MSB
576  mag_data[4] - Z LSB
577  mag_data[5] - Z MSB
578  mag_data[6] - R LSB
579  mag_data[7] - R MSB
580  */
581 
582  /* Extract X axis data */
583  lsb = ((mag_data[0] & 0xF8) >> 3);
584  msb = (((int8_t)mag_data[1]) << 5);
585  msblsb = (msb | lsb);
586  mrb.x_raw = (int16_t)msblsb;
587 
588 
589  /* Extract Y axis data */
590  lsb = ((mag_data[2] & 0xF8) >> 3);
591  msb = (((int8_t)mag_data[3]) << 5);
592  msblsb = (msb | lsb);
593  mrb.y_raw = (int16_t)msblsb;
594 
595  /* Extract Z axis data */
596  lsb = ((mag_data[4] & 0xFE) >> 1);
597  msb = (((int8_t)mag_data[5]) << 7);
598  msblsb = (msb | lsb);
599  mrb.z_raw = (int16_t)msblsb;
600 
601  /* Extract Resistance data */
602  lsb = ((mag_data[6] & 0xFC) >> 2);
603  msb = (mag_data[7] << 6);
604  msblsb = (msb | lsb);
605  resistance = (uint16_t)msblsb;
606 
607  /* Check whether data is new or old */
608  if (!(status & 0x01)) {
609  perf_end(_sample_perf);
610  perf_count(_duplicates);
611  _got_duplicate = true;
612  return -EIO;
613  }
614 
615  _got_duplicate = false;
616 
617  if (mrb.x_raw == 0 &&
618  mrb.y_raw == 0 &&
619  mrb.z_raw == 0 &&
620  resistance == 0) {
621  // all zero data - probably a I2C bus error
622  perf_count(_comms_errors);
623  perf_count(_bad_transfers);
624  perf_end(_sample_perf);
625  return -EIO;
626  }
627 
628  perf_count(_good_transfers);
629 
630  /* Compensation for X axis */
631  if (mrb.x_raw != BMM150_FLIP_OVERFLOW_ADCVAL) {
632  /* no overflow */
633  if ((resistance != 0) && (dig_xyz1 != 0)) {
634  mrb.x = ((dig_xyz1 * 16384.0 / resistance) - 16384.0);
635  mrb.x = (((mrb.x_raw * ((((dig_xy2 * ((float)(mrb.x * mrb.x / (float)268435456.0)) + mrb.x * dig_xy1 /
636  (float)16384.0)) + (float)256.0) * (dig_x2 + (float)160.0))) / (float)8192.0) + (dig_x1 * (float)8.0)) / (float)16.0;
637 
638  } else {
639  mrb.x = BMM150_OVERFLOW_OUTPUT_FLOAT;
640  }
641 
642  } else {
643  mrb.x = BMM150_OVERFLOW_OUTPUT_FLOAT;
644  }
645 
646  /* Compensation for Y axis */
647  if (mrb.y_raw != BMM150_FLIP_OVERFLOW_ADCVAL) {
648  /* no overflow */
649  if ((resistance != 0) && (dig_xyz1 != 0)) {
650 
651  mrb.y = ((((float)dig_xyz1) * (float)16384.0 / resistance) - (float)16384.0);
652  mrb.y = (((mrb.y_raw * ((((dig_xy2 * (mrb.y * mrb.y / (float)268435456.0) + mrb.y * dig_xy1 / (float)16384.0)) +
653  (float)256.0) * (dig_y2 + (float)160.0))) / (float)8192.0) + (dig_y1 * (float)8.0)) / (float)16.0;
654 
655 
656  } else {
657  mrb.y = BMM150_OVERFLOW_OUTPUT_FLOAT;
658  }
659 
660  } else {
661  /* overflow, set output to 0.0f */
662  mrb.y = BMM150_OVERFLOW_OUTPUT_FLOAT;
663  }
664 
665 
666  /* Compensation for Z axis */
667  if (mrb.z_raw != BMM150_HALL_OVERFLOW_ADCVAL) {
668  /* no overflow */
669  if ((dig_z2 != 0) && (dig_z1 != 0) && (dig_xyz1 != 0) && (resistance != 0)) {
670  mrb.z = ((((mrb.z_raw - dig_z4) * (float)131072.0) - (dig_z3 * (resistance - dig_xyz1))) / ((
671  dig_z2 + dig_z1 * resistance / (float)32768.0) * (float)4.0)) / (float)16.0;
672  }
673 
674  } else {
675  /* overflow, set output to 0.0f */
676  mrb.z = BMM150_OVERFLOW_OUTPUT_FLOAT;
677  }
678 
679 
680  mrb.timestamp = hrt_absolute_time();
681  mrb.is_external = external();
682 
683  // report the error count as the number of bad transfers.
684  // This allows the higher level code to decide if it
685  // should use this sensor based on whether it has had failures
686  mrb.error_count = perf_event_count(_bad_transfers);
687 
688  // apply user specified rotation
689  rotate_3f(_rotation, mrb.x, mrb.y, mrb.z);
690 
691 
692  /* Scaling the data */
693  mrb.x = ((mrb.x * _range_scale) - _scale.x_offset) * _scale.x_scale;
694  mrb.y = ((mrb.y * _range_scale) - _scale.y_offset) * _scale.y_scale;
695  mrb.z = ((mrb.z * _range_scale) - _scale.z_offset) * _scale.z_scale;
696 
697  mrb.scaling = _range_scale;
698  mrb.device_id = _device_id.devid;
699 
700  _last_report.x = mrb.x;
701  _last_report.y = mrb.y;
702  _last_report.z = mrb.z;
703 
704  _reports->force(&mrb);
705 
706  /* notify anyone waiting for data */
707  if (mag_notify) {
708  poll_notify(POLLIN);
709  }
710 
711  if (mag_notify && !(_pub_blocked)) {
712  /* publish it */
713  orb_publish(ORB_ID(sensor_mag), _topic, &mrb);
714  }
715 
716  perf_end(_sample_perf);
717  return OK;
718

723 {
724 
725  switch (cmd) {
726  case SENSORIOCSPOLLRATE: {
727  switch (arg) {
728 
729  /* switching to manual polling */

736  case SENSOR_POLLRATE_EXTERNAL:

739  case 0:

741

764  if (want_start) {
765  start();
766  }

769  }

784  }

787

789  px4_leave_critical_section(flags);

791  }
792 
793  px4_leave_critical_section(flags);
794 
795  return OK;
796  }

798  case SENSORIOCRESET:
799  return reset();

801  case MAGIOCSSCALE:
802  return OK;
803 
804  case MAGIOCGSCALE:
805  return OK;
806 
807  case MAGIOCCALIBRATE:
808  return OK;
809 
810  case MAGIOCEXSTRAP:
811  return OK;
812 
813  case MAGIOCSSAMPLERATE:
814  return ioctl(filp, SENSORIOCSPOLLRATE, arg);
815 
816  case MAGIOCGSAMPLERATE:
