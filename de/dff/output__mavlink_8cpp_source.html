62 int OutputMavlink::update(const ControlData *control_data)
63 {
64  vehicle_command_s vehicle_command = {};
65  vehicle_command.timestamp = hrt_absolute_time();
66  vehicle_command.target_system = (uint8_t)_config.mavlink_sys_id;
67  vehicle_command.target_component = (uint8_t)_config.mavlink_comp_id;
68 
69  if (control_data) {
70  //got new command
71  _set_angle_setpoints(control_data);
72 
73  vehicle_command.command = vehicle_command_s::VEHICLE_CMD_DO_MOUNT_CONFIGURE;
74  vehicle_command.timestamp = hrt_absolute_time();
75 
76  if (control_data->type == ControlData::Type::Neutral) {
77  vehicle_command.param1 = vehicle_command_s::VEHICLE_MOUNT_MODE_NEUTRAL;
78 
79  } else {
80  vehicle_command.param1 = vehicle_command_s::VEHICLE_MOUNT_MODE_MAVLINK_TARGETING;
81  }
82 
83  if (_vehicle_command_pub) {
84  orb_publish(ORB_ID(vehicle_command), _vehicle_command_pub, &vehicle_command);
85 
86  } else {
87  _vehicle_command_pub = orb_advertise_queue(ORB_ID(vehicle_command), &vehicle_command,
88  vehicle_command_s::ORB_QUEUE_LENGTH);
89  }
90 
91  }
92 
93  if (!_vehicle_command_pub) {
94  return 0;
95  }
96 
97  _handle_position_update();
98 
99  hrt_abstime t = hrt_absolute_time();
100  _calculate_output_angles(t);
101 
102  vehicle_command.timestamp = t;
103  vehicle_command.command = vehicle_command_s::VEHICLE_CMD_DO_MOUNT_CONTROL;
104 
105  // vmount spec has roll, pitch on channels 0, 1, respectively; MAVLink spec has roll, pitch on channels 1, 0, respectively
106  // vmount uses radians, MAVLink uses degrees
107  vehicle_command.param1 = (_angle_outputs[1] + _config.pitch_offset) * M_RAD_TO_DEG_F;
108  vehicle_command.param2 = (_angle_outputs[0] + _config.roll_offset) * M_RAD_TO_DEG_F;
109  vehicle_command.param3 = (_angle_outputs[2] + _config.yaw_offset) * M_RAD_TO_DEG_F;
110 
111  orb_publish(ORB_ID(vehicle_command), _vehicle_command_pub, &vehicle_command);
112 
113  _last_update = t;
114 
115  return 0;
116 }
