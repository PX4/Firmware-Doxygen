54 bool CRSFTelemetry::update(const hrt_abstime &now)
55 {
56  const int update_rate_hz = 10;
57 
58  if (now - _last_update <= 1_s / (update_rate_hz * num_data_types)) {
59  return false;
60  }
61 
62  bool sent = false;
63 
64  switch (_next_type) {
65  case 0:
66  sent = send_battery();
67  break;
68 
69  case 1:
70  sent = send_gps();
71  break;
72 
73  case 2:
74  sent = send_attitude();
75  break;
76 
77  case 3:
78  sent = send_flight_mode();
79  break;
80  }
81 
82  _last_update = now;
83  _next_type = (_next_type + 1) % num_data_types;
84 
85  return sent;
86 }

88 bool CRSFTelemetry::send_battery()
89 {
90  battery_status_s battery_status;
91 
92  if (orb_copy(ORB_ID(battery_status), _battery_status_sub, &battery_status) != 0) {
93  return false;
94  }
95 
96  uint16_t voltage = battery_status.voltage_filtered_v * 10;
97  uint16_t current = battery_status.current_filtered_a * 10;
98  int fuel = battery_status.discharged_mah;
99  uint8_t remaining = battery_status.remaining * 100;
100  return crsf_send_telemetry_battery(_uart_fd, voltage, current, fuel, remaining);
101 }

103 bool CRSFTelemetry::send_gps()
104 {
105  vehicle_gps_position_s vehicle_gps_position;
106 
107  if (orb_copy(ORB_ID(vehicle_gps_position), _vehicle_gps_position_sub, &vehicle_gps_position) != 0) {
108  return false;
109  }
110 
111  int32_t latitude = vehicle_gps_position.lat;
112  int32_t longitude = vehicle_gps_position.lon;
113  uint16_t groundspeed = vehicle_gps_position.vel_d_m_s / 3.6f * 10.f;
114  uint16_t gps_heading = math::degrees(vehicle_gps_position.cog_rad) * 100.f;
115  uint16_t altitude = vehicle_gps_position.alt + 1000;
116  uint8_t num_satellites = vehicle_gps_position.satellites_used;
117 
118  return crsf_send_telemetry_gps(_uart_fd, latitude, longitude, groundspeed,
119  gps_heading, altitude, num_satellites);
120 }
