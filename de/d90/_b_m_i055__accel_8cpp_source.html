157 int BMI055_accel::reset()
158 {
159  write_reg(BMI055_ACC_SOFTRESET, BMI055_SOFT_RESET);//Soft-reset
160  up_udelay(5000);
161 
162  write_checked_reg(BMI055_ACC_BW, BMI055_ACCEL_BW_1000); //Write accel bandwidth
163  write_checked_reg(BMI055_ACC_RANGE, BMI055_ACCEL_RANGE_2_G);//Write range
164  write_checked_reg(BMI055_ACC_INT_EN_1, BMI055_ACC_DRDY_INT_EN); //Enable DRDY interrupt
165  write_checked_reg(BMI055_ACC_INT_MAP_1, BMI055_ACC_DRDY_INT1); //Map DRDY interrupt on pin INT1
166 
167  set_accel_range(BMI055_ACCEL_DEFAULT_RANGE_G);//set accel range
168  accel_set_sample_rate(BMI055_ACCEL_DEFAULT_RATE);//set accel ODR
169 
170  //Enable Accelerometer in normal mode
171  write_reg(BMI055_ACC_PMU_LPW, BMI055_ACCEL_NORMAL);
172  up_udelay(1000);
173 
174  uint8_t retries = 10;
175 
176  while (retries--) {
177  bool all_ok = true;
178 
179  for (uint8_t i = 0; i < BMI055_ACCEL_NUM_CHECKED_REGISTERS; i++) {
180  if (read_reg(_checked_registers[i]) != _checked_values[i]) {
181  write_reg(_checked_registers[i], _checked_values[i]);
182  all_ok = false;
183  }
184  }
185 
186  if (all_ok) {
187  break;
188  }
189  }
190 
191  return OK;
192 }

262 ssize_t
263 BMI055_accel::read(struct file *filp, char *buffer, size_t buflen)
264 {
265  unsigned count = buflen / sizeof(accel_report);
266 
267  /* buffer must be large enough */
268  if (count < 1) {
269  return -ENOSPC;
270  }
271 
272  /* if automatic measurement is not enabled, get a fresh measurement into the buffer */
273  if (_call_interval == 0) {
274  _accel_reports->flush();
275  measure();
276  }
277 
278  /* if no data, error (we could block here) */
279  if (_accel_reports->empty()) {
280  return -EAGAIN;
281  }
282 
283  /* copy reports out of our buffer to the caller */
284  accel_report *arp = reinterpret_cast<accel_report *>(buffer);
285  int transferred = 0;
286 
287  while (count--) {
288  if (!_accel_reports->get(arp)) {
289  break;
290  }
291 
292  transferred++;
293  arp++;
294  }
295 
296  /* return the number of bytes transferred */
297  return (transferred * sizeof(accel_report));
298 }

322 int
323 BMI055_accel::ioctl(struct file *filp, int cmd, unsigned long arg)
324 {
325  switch (cmd) {
326 
327  case SENSORIOCRESET:
328  return reset();
329 
330  case SENSORIOCSPOLLRATE: {
331  switch (arg) {
332 
333  /* switching to manual polling */
334  case SENSOR_POLLRATE_MANUAL:
335  stop();
336  _call_interval = 0;
337  return OK;
338 
339  /* external signalling not supported */
340  case SENSOR_POLLRATE_EXTERNAL:
341 
342  /* zero would be bad */
343  case 0:
344  return -EINVAL;
345 
346  /* set default/max polling rate */
347  case SENSOR_POLLRATE_MAX:
348  return ioctl(filp, SENSORIOCSPOLLRATE, BMI055_ACCEL_MAX_RATE);
349 
350  case SENSOR_POLLRATE_DEFAULT:
351  // Polling at the highest frequency. We may get duplicate values on the sensors
352  return ioctl(filp, SENSORIOCSPOLLRATE, BMI055_ACCEL_DEFAULT_RATE);
353 
354  /* adjust to a legal polling interval in Hz */
355  default: {
356  /* do we need to start internal polling? */
357  bool want_start = (_call_interval == 0);
358 
359  /* convert hz to hrt interval via microseconds */
360  unsigned ticks = 1000000 / arg;
361 
362  /* check against maximum rate */
363  if (ticks < 1000) {
364  return -EINVAL;
365  }
366 
367  // adjust filters
368  float cutoff_freq_hz = _accel_filter_x.get_cutoff_freq();
369  float sample_rate = 1.0e6f / ticks;
370 
371  _accel_filter_x.set_cutoff_frequency(sample_rate, cutoff_freq_hz);
372  _accel_filter_y.set_cutoff_frequency(sample_rate, cutoff_freq_hz);
373  _accel_filter_z.set_cutoff_frequency(sample_rate, cutoff_freq_hz);
374 
375  /* update interval for next measurement */
376  _call_interval = ticks;
377 
378  /*
379  set call interval faster than the sample time. We
380  then detect when we have duplicate samples and reject
381  them. This prevents aliasing due to a beat between the
382  stm32 clock and the bmi055 clock
383  */
384  _call.period = _call_interval - BMI055_TIMER_REDUCTION;
385 
386  /* if we need to start the poll state machine, do it */
387  if (want_start) {
388  start();
389  }
390 
391  return OK;
392  }
393  }
394  }
395 
396  case SENSORIOCGPOLLRATE:
397  if (_call_interval == 0) {
398  return SENSOR_POLLRATE_MANUAL;
399  }
400 
401  return 1000000 / _call_interval;
402 
403  case SENSORIOCSQUEUEDEPTH: {
404  /* lower bound is mandatory, upper bound is a sanity check */
405  if ((arg < 1) || (arg > 100)) {
406  return -EINVAL;
407  }
408 
409  irqstate_t flags = px4_enter_critical_section();
410 
411  if (!_accel_reports->resize(arg)) {
412  px4_leave_critical_section(flags);
413  return -ENOMEM;
414  }
415 
416  px4_leave_critical_section(flags);
417 
418  return OK;
419  }
420 
421  case ACCELIOCGSAMPLERATE:
422  return _accel_sample_rate;
423 
424  case ACCELIOCSSAMPLERATE:
425  return accel_set_sample_rate(arg);
426 
427  case ACCELIOCSSCALE: {
428  /* copy scale, but only if off by a few percent */
429  struct accel_calibration_s *s = (struct accel_calibration_s *) arg;
430  float sum = s->x_scale + s->y_scale + s->z_scale;
431 
432  if (sum > 2.0f && sum < 4.0f) {
433  memcpy(&_accel_scale, s, sizeof(_accel_scale));
434  return OK;
435 
436  } else {
437  return -EINVAL;
438  }
439  }
440 
441  case ACCELIOCGSCALE:
442  /* copy scale out */
443  memcpy((struct accel_calibration_s *) arg, &_accel_scale, sizeof(_accel_scale));
444  return OK;
445 
446  case ACCELIOCSRANGE:
447  return set_accel_range(arg);
448 
449  case ACCELIOCGRANGE:
450  return (unsigned long)((_accel_range_m_s2) / CONSTANTS_ONE_G + 0.5f);
451 
452  default:
453  /* give it to the superclass */
454  return SPI::ioctl(filp, cmd, arg);
455  }
456 }

558 void
559 BMI055_accel::check_registers(void)
560 {
561  uint8_t v;
562 
563  if ((v = read_reg(_checked_registers[_checked_next])) !=
564  _checked_values[_checked_next]) {
565  _checked_bad[_checked_next] = v;
566 
567  /*
568  if we get the wrong value then we know the SPI bus
569  or sensor is very sick. We set _register_wait to 20
570  and wait until we have seen 20 good values in a row
571  before we consider the sensor to be OK again.
572  */
573  perf_count(_bad_registers);
574 
575  /*
576  try to fix the bad register value. We only try to
577  fix one per loop to prevent a bad sensor hogging the
578  bus.
579  */
580  if (_register_wait == 0 || _checked_next == 0) {
581  // if the product_id is wrong then reset the
582  // sensor completely
583  write_reg(BMI055_ACC_SOFTRESET, BMI055_SOFT_RESET);
584  _reset_wait = hrt_absolute_time() + 10000;
585  _checked_next = 0;
586 
587  } else {
588  write_reg(_checked_registers[_checked_next], _checked_values[_checked_next]);
589  // waiting 3ms between register writes seems
590  // to raise the chance of the sensor
591  // recovering considerably
592  _reset_wait = hrt_absolute_time() + 3000;
593  }
594 
595  _register_wait = 20;
596  }
597 
598  _checked_next = (_checked_next + 1) % BMI055_ACCEL_NUM_CHECKED_REGISTERS;
599 }

601 void
602 BMI055_accel::measure()
603 {
604  perf_count(_measure_interval);
605 
606  uint8_t index = 0, accel_data[7];
607  uint16_t lsb, msb, msblsb;
608  uint8_t status_x, status_y, status_z;
609 
610  if (hrt_absolute_time() < _reset_wait) {
611  // we're waiting for a reset to complete
612  return;
613  }
614 
615  struct Report {
616  int16_t accel_x;
617  int16_t accel_y;
618  int16_t accel_z;
619  int16_t temp;
620  } report;
621 
622  /* start measuring */
623  perf_begin(_sample_perf);
624 
625  /*
626  * Fetch the full set of measurements from the BMI055 in one pass.
627  */
628  accel_data[index] = BMI055_ACC_X_L | DIR_READ;
629 
630  if (OK != transfer(accel_data, accel_data, sizeof(accel_data))) {
631  return;
632  }
633 
634  check_registers();
635 
636  /* Extracting accel data from the read data */
637  index = 1;
638  lsb = (uint16_t)accel_data[index++];
639  status_x = (lsb & BMI055_NEW_DATA_MASK);
640  msb = (uint16_t)accel_data[index++];
641  msblsb = (msb << 8) | lsb;
642  report.accel_x = ((int16_t)msblsb >> 4); /* Data in X axis */
643 
644  lsb = (uint16_t)accel_data[index++];
645  status_y = (lsb & BMI055_NEW_DATA_MASK);
646  msb = (uint16_t)accel_data[index++];
647  msblsb = (msb << 8) | lsb;
648  report.accel_y = ((int16_t)msblsb >> 4); /* Data in Y axis */
649 
650  lsb = (uint16_t)accel_data[index++];
651  status_z = (lsb & BMI055_NEW_DATA_MASK);
652  msb = (uint16_t)accel_data[index++];
653  msblsb = (msb << 8) | lsb;
654  report.accel_z = ((int16_t)msblsb >> 4); /* Data in Z axis */
655 
656  // Checking the status of new data
657  if ((!status_x) || (!status_y) || (!status_z)) {
658  perf_end(_sample_perf);
659  perf_count(_duplicates);
660  _got_duplicate = true;
661  return;
662  }
663 
664 
665  _got_duplicate = false;
666 
667  uint8_t temp = read_reg(BMI055_ACC_TEMP);
668  report.temp = temp;
669 
670  if (report.accel_x == 0 &&
671  report.accel_y == 0 &&
672  report.accel_z == 0 &&
673  report.temp == 0) {
674  // all zero data - probably a SPI bus error
675  perf_count(_bad_transfers);
676  perf_end(_sample_perf);
677  // note that we don't call reset() here as a reset()
678  // costs 20ms with interrupts disabled. That means if
679  // the bmi055 accel does go bad it would cause a FMU failure,
680  // regardless of whether another sensor is available,
681  return;
682  }
683 
684  if (_register_wait != 0) {
685  // we are waiting for some good transfers before using
686  // the sensor again, but don't return any data yet
687  _register_wait--;
688  return;
689  }
690 
691  /*
692  * Report buffers.
693  */
694  accel_report arb;
695 
696  arb.timestamp = hrt_absolute_time();
697 
698  // report the error count as the sum of the number of bad
699  // transfers and bad register reads. This allows the higher
700  // level code to decide if it should use this sensor based on
701  // whether it has had failures
702  arb.error_count = perf_event_count(_bad_transfers) + perf_event_count(_bad_registers);
703 
704  /*
705  * 1) Scale raw value to SI units using scaling from datasheet.
706  * 2) Subtract static offset (in SI units)
707  * 3) Scale the statically calibrated values with a linear
708  * dynamically obtained factor
709  *
710  * Note: the static sensor offset is the number the sensor outputs
711  * at a nominally 'zero' input. Therefore the offset has to
712  * be subtracted.
713  *
714  */
715 
716  arb.x_raw = report.accel_x;
717  arb.y_raw = report.accel_y;
718  arb.z_raw = report.accel_z;
719 
720  float xraw_f = report.accel_x;
721  float yraw_f = report.accel_y;
722  float zraw_f = report.accel_z;
723 
724  // apply user specified rotation
725  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
726 
727  float x_in_new = ((xraw_f * _accel_range_scale) - _accel_scale.x_offset) * _accel_scale.x_scale;
728  float y_in_new = ((yraw_f * _accel_range_scale) - _accel_scale.y_offset) * _accel_scale.y_scale;
729  float z_in_new = ((zraw_f * _accel_range_scale) - _accel_scale.z_offset) * _accel_scale.z_scale;
730 
731  arb.x = _accel_filter_x.apply(x_in_new);
732  arb.y = _accel_filter_y.apply(y_in_new);
733  arb.z = _accel_filter_z.apply(z_in_new);
734 
735  matrix::Vector3f aval(x_in_new, y_in_new, z_in_new);
736  matrix::Vector3f aval_integrated;
737 
738  bool accel_notify = _accel_int.put(arb.timestamp, aval, aval_integrated, arb.integral_dt);
739  arb.x_integral = aval_integrated(0);
740  arb.y_integral = aval_integrated(1);
741  arb.z_integral = aval_integrated(2);
742 
743  arb.scaling = _accel_range_scale;
744 
745  _last_temperature = 23 + report.temp * 1.0f / 512.0f;
746 
747  arb.temperature = _last_temperature;
748  arb.device_id = _device_id.devid;
749 
750  _accel_reports->force(&arb);
751 
752  /* notify anyone waiting for data */
753  if (accel_notify) {
754  poll_notify(POLLIN);
755  }
756 
757  if (accel_notify && !(_pub_blocked)) {
758  /* publish it */
759  orb_publish(ORB_ID(sensor_accel), _accel_topic, &arb);
760  }
761 
762  /* stop measuring */
763  perf_end(_sample_perf);
764 }
