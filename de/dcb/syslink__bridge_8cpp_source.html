108  buflen -= size;

114 ssize_t
115 SyslinkBridge::write(struct file *filp, const char *buffer, size_t buflen)
116 {
117  int buflen_rem = buflen;
118 
119  while (buflen_rem > 0) {
120 
121  int datasize = MIN(_msg_to_send_size_remaining, buflen_rem);
122  _msg_to_send.size += datasize;
123  memcpy(&_msg_to_send.data[CRTP_MAX_DATA_SIZE - 1 - _msg_to_send_size_remaining], buffer, datasize);
124 
125  buffer += datasize;
126  _msg_to_send_size_remaining -= datasize;
127  buflen_rem -= datasize;
128 
129 
130  if (_msg_to_send_size_remaining == 0) {
131 
132  if (_link->_writebuffer.force(&_msg_to_send, sizeof(crtp_message_t))) {
133  PX4_WARN("write buffer overflow");
134  }
135 
136  _msg_to_send.size = sizeof(_msg_to_send.header);
137  _msg_to_send_size_remaining = CRTP_MAX_DATA_SIZE - 1;
138  }
139  }
140 
141  return buflen;
142 }
