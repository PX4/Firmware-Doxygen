72 void
73 Hysteresis::update(const hrt_abstime &now_us)
74 {
75  if (_requested_state != _state) {
76 
77  const hrt_abstime elapsed = now_us - _last_time_to_change_state;
78 
79  if (_state && !_requested_state) {
80  // true -> false
81  if (elapsed >= _time_from_true_us) {
82  _state = false;
83  }
84 
85  } else if (!_state && _requested_state) {
86  // false -> true
87  if (elapsed >= _time_from_false_us) {
88  _state = true;
89  }
90  }
91  }
92 }
