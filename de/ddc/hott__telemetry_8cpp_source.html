98 int
99 recv_req_id(int uart, uint8_t *id)
100 {
101  uint8_t mode;
102 
103  struct pollfd fds;
104  fds.fd = uart;
105  fds.events = POLLIN;
106 
107  if (poll(&fds, 1, timeout_ms) > 0) {
108  /* Get the mode: binary or text */
109  read(uart, &mode, sizeof(mode));
110 
111  perf_count(reqs_count);
112 
113  // Debug log
114  for (int x = 15; x > 0; x--) {
115  read_log[x] = read_log[x - 1];
116  }
117 
118  read_log[0] = mode;
119 
120  /* if we have a binary mode request */
121  if (mode != BINARY_MODE_REQUEST_ID) {
122  return PX4_ERROR;
123  }
124 
125  /* Read the device ID being polled */
126  read(uart, id, sizeof(*id));
127 
128  } else {
129  warnx("UART timeout on TX/RX port");
130  return PX4_ERROR;
131  }
132 
133  return PX4_OK;
134 }

136 int
137 send_data(int uart, uint8_t *buffer, size_t size)
138 {
139  usleep(read_delay_us);
140 
141  uint16_t checksum = 0;
142 
143  for (size_t i = 0; i < size; i++) {
144  if (i == size - 1) {
145  /* Set the checksum: the first uint8_t is taken as the checksum. */
146  buffer[i] = checksum & 0xff;
147 
148  } else {
149  checksum += buffer[i];
150  }
151 
152  write(uart, &buffer[i], sizeof(buffer[i]));
153 
154  /* Sleep before sending the next byte. */
155  usleep(write_delay_us);
156  }
157 
158  /* A hack the reads out what was written so the next read from the receiver doesn't get it. */
159  /* TODO: Fix this!! */
160  uint8_t dummy[size];
161  read(uart, &dummy, size);
162 
163  return PX4_OK;
164 }

166 int
167 hott_telemetry_thread_main(int argc, char *argv[])
168 {
169  warnx("starting");
170 
171  connect_count = perf_alloc(PC_COUNT, "reconnects ");
172  recon_port = perf_alloc(PC_COUNT, "reopen port ");
173  reqs_count = perf_alloc(PC_COUNT, "requests ");
174  bin_reply = perf_alloc(PC_COUNT, "bin replies ");
175  txt_reply = perf_alloc(PC_COUNT, "text replies ");
176  bad_reply = perf_alloc(PC_COUNT, "unknown replies ");
177  dead_reply = perf_alloc(PC_COUNT, "dead replies ");
178 
179  thread_running = true;
180 
181  const char *device = DEFAULT_UART;
182 
183  /* read commandline arguments */
184  for (int i = 0; i < argc && argv[i]; i++) {
185  if (strcmp(argv[i], "-d") == 0 || strcmp(argv[i], "--device") == 0) { //device set
186  if (argc > i + 1) {
187  device = argv[i + 1];
188 
189  } else {
190  thread_running = false;
191  errx(1, "missing parameter to -d\n%s", commandline_usage);
192  }
193  }
194  }
195 
196  /* enable UART, writes potentially an empty buffer, but multiplexing is disabled */
197  int uart = open_uart(device);
198 
199  if (uart < 0) {
200  errx(1, "Failed opening HoTT UART, exiting.");
201  thread_running = false;
202  }
203 
204  init_sub_messages();
205 
206  uint8_t buffer[MAX_MESSAGE_BUFFER_SIZE];
207  size_t size = 0;
208  uint8_t id = 0;
209 
210  bool connected = true;
211  int recon = 0;
212 
213  while (!thread_should_exit) {
214  // Listen for and serve poll from the receiver.
215  if (recv_req_id(uart, &id) == PX4_OK) {
216  if (!connected) {
217  connected = true;
218  }
219 
220  switch (id) {
221  case EAM_SENSOR_ID:
222  build_eam_response(buffer, &size);
223  perf_count(bin_reply);
224  break;
225 
226  case GAM_SENSOR_ID:
227  build_gam_response(buffer, &size);
228  perf_count(bin_reply);
229  break;
230 
231  case GPS_SENSOR_ID:
232  build_gps_response(buffer, &size);
233  perf_count(bin_reply);
234  break;
235 
236  case BINARY_MODE_REQUEST_ID:
237  perf_count(dead_reply);
238  break;
239 
240  default:
241  perf_count(bad_reply);
242  continue; // Not a module we support.
243  }
244 
245  send_data(uart, buffer, size);
246 
247  } else {
248  if (connected) {
249  connected = false;
250 
251  } else {
252  recon++;
253  }
254 
255  if (recon > 100) {
256  perf_count(recon_port);
257  close(uart);
258  uart = open_uart(device);
259  perf_reset(reqs_count);
260  perf_reset(bin_reply);
261  perf_reset(txt_reply);
262  perf_reset(dead_reply);
263  perf_reset(bad_reply);
264  }
265  }
266  }
267 
268  warnx("exiting");
269 
270  close(uart);
271 
272  thread_running = false;
273 
274  perf_free(connect_count);
275  perf_free(recon_port);
276  perf_free(reqs_count);
277  perf_free(bin_reply);
278  perf_free(txt_reply);
279  perf_free(bad_reply);
280  perf_free(dead_reply);
281 
282  return 0;
283 }
