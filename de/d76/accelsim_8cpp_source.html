324 
325  /* do SIM init first */
326  if (VirtDevObj::init() != 0) {
327  PX4_WARN("SIM init failed");
328  goto out;
329  }
330 
331  /* allocate basic report buffers */
332  _accel_reports = new ringbuffer::RingBuffer(2, sizeof(accel_report));
333 
334  if (_accel_reports == nullptr) {
335  goto out;
336  }
337 
338  _mag_reports = new ringbuffer::RingBuffer(2, sizeof(mag_report));
339 
340  if (_mag_reports == nullptr) {
341  goto out;
342  }
343 
344  /* do init for the mag device node */
345  ret = _mag->init();
346 
347  if (ret != OK) {
348  PX4_WARN("MAG init failed");
349  goto out;
350  }
351 
352  /* fill report structures */
353  _measure();
354 
355  /* advertise sensor topic, measure manually to initialize valid report */
356  _mag_reports->get(&mrp);
357 
358  /* measurement will have generated a report, publish */
359  _mag->_mag_topic = orb_advertise_multi(ORB_ID(sensor_mag), &mrp,
360  &_mag->_mag_orb_class_instance, ORB_PRIO_LOW);
361 
362  if (_mag->_mag_topic == nullptr) {
363  PX4_WARN("ADVERT ERR");
364  }
365 
366  /* advertise sensor topic, measure manually to initialize valid report */
367  _accel_reports->get(&arp);
368 
369  /* measurement will have generated a report, publish */
370  _accel_topic = orb_advertise_multi(ORB_ID(sensor_accel), &arp,

373  if (_accel_topic == nullptr) {

381 int
382 ACCELSIM::transfer(uint8_t *send, uint8_t *recv, unsigned len)
383 {
384  uint8_t cmd = send[0];
385 
386  if (cmd & DIR_READ) {
387  // Get data from the simulator
388  Simulator *sim = Simulator::getInstance();
389 
390  if (sim == nullptr) {
391  return ENODEV;
392  }
393 
394  // FIXME - not sure what interrupt status should be
395  recv[1] = 0;
396 
397  // skip cmd and status bytes
398  if (cmd & ACC_READ) {
399  sim->getRawAccelReport(&recv[2], len - 2);
400 
401  } else if (cmd & MAG_READ) {
402  sim->getMagReport(&recv[2], len - 2);
403  }
404  }
405 
406  return PX4_OK;
407 }

409 ssize_t
410 ACCELSIM::devRead(void *buffer, size_t buflen)
411 {
412  unsigned count = buflen / sizeof(struct accel_report);
413  accel_report *arb = reinterpret_cast<accel_report *>(buffer);
414  int ret = 0;
415 
416  /* buffer must be large enough */
417  if (count < 1) {
418  return -ENOSPC;
419  }
420 
421  /* if automatic measurement is enabled */
422  if (m_sample_interval_usecs > 0) {
423  /*
424  * While there is space in the caller's buffer, and reports, copy them.
425  */
426  while (count--) {
427  if (_accel_reports->get(arb)) {
428  ret += sizeof(*arb);
429  arb++;
430  }
431  }
432 
433  /* if there was no data, warn the caller */
434  return ret ? ret : -EAGAIN;
435  }
436 
437  /* manual measurement */
438  _measure();
439 
440  /* measurement will have generated a report, copy it out */
441  if (_accel_reports->get(arb)) {
442  ret = sizeof(*arb);
443  }
444 
445  return ret;
446 }

489 int
490 ACCELSIM::devIOCTL(unsigned long cmd, unsigned long arg)
491 {
492  unsigned long ul_arg = (unsigned long)arg;
493 
494  switch (cmd) {
495 
496  case SENSORIOCSPOLLRATE: {
497  switch (ul_arg) {
498 
499  /* switching to manual polling */
500  case SENSOR_POLLRATE_MANUAL:
501  stop();
502  m_sample_interval_usecs = 0;
503  return OK;
504 
505  /* external signalling not supported */
506  case SENSOR_POLLRATE_EXTERNAL:
507 
508  /* zero would be bad */
509  case 0:
510  return -EINVAL;
511 
512  /* set default/max polling rate */
513  case SENSOR_POLLRATE_MAX:
514  return devIOCTL(SENSORIOCSPOLLRATE, 1600);
515 
516  case SENSOR_POLLRATE_DEFAULT:
517  return devIOCTL(SENSORIOCSPOLLRATE, ACCELSIM_ACCEL_DEFAULT_RATE);
518 
519  /* adjust to a legal polling interval in Hz */
520  default: {
521  /* convert hz to hrt interval via microseconds */
522  unsigned interval = 1000000 / ul_arg;
523 
524  /* check against maximum sane rate */
525  if (interval < 500) {
526  return -EINVAL;
527  }
528 
529  /* adjust filters */
530  accel_set_driver_lowpass_filter((float)ul_arg, _accel_filter_x.get_cutoff_freq());
531 
532  bool want_start = (m_sample_interval_usecs == 0);
533 
534  /* update interval for next measurement */
535  setSampleInterval(interval);
536 
537  if (want_start) {
538  start();
539  }
540 
541  return OK;
542  }
543  }
544  }
545 
546  case SENSORIOCGPOLLRATE:
547  if (m_sample_interval_usecs == 0) {
548  return SENSOR_POLLRATE_MANUAL;
549  }
550 
551  return 1000000 / m_sample_interval_usecs;
552 
553  case SENSORIOCSQUEUEDEPTH: {
554  /* lower bound is mandatory, upper bound is a sanity check */
555  if ((ul_arg < 1) || (ul_arg > 100)) {
556  return -EINVAL;
557  }
558 
559  if (!_accel_reports->resize(ul_arg)) {
560  return -ENOMEM;
561  }
562 
563  return OK;
564  }
565 
566  case SENSORIOCRESET:
567  // Nothing to do for simulator
568  return OK;
569 
570  case ACCELIOCSSAMPLERATE:
571  // No need to set internal sampling rate for simulator
572  return OK;
573 
574  case ACCELIOCGSAMPLERATE:
575  return _accel_samplerate;
576 
577  case ACCELIOCSSCALE: {
578  /* copy scale, but only if off by a few percent */
579  struct accel_calibration_s *s = (struct accel_calibration_s *) arg;
580  float sum = s->x_scale + s->y_scale + s->z_scale;
581 
582  if (sum > 2.0f && sum < 4.0f) {
583  memcpy(&_accel_scale, s, sizeof(_accel_scale));
584  return OK;
585 
586  } else {
587  return -EINVAL;
588  }
589  }
590 
591  case ACCELIOCSRANGE:
592  /* arg needs to be in G */
593  return accel_set_range(ul_arg);
594 
595  case ACCELIOCGRANGE:
596  /* convert to m/s^2 and return rounded in G */
597  return (unsigned long)((_accel_range_m_s2) / CONSTANTS_ONE_G + 0.5f);
598 
599  case ACCELIOCGSCALE:
600  /* copy scale out */
601  memcpy((struct accel_calibration_s *) arg, &(_accel_scale), sizeof(_accel_scale));
602  return OK;
603 
604  default:
605  /* give it to the superclass */
606  return VirtDevObj::devIOCTL(cmd, arg);
607  }
608 }

666  }
667 
668  return 1000000 / _mag->m_sample_interval_usecs;

670  case SENSORIOCSQUEUEDEPTH: {
671  /* lower bound is mandatory, upper bound is a sanity check */

674  }
675

677  return -ENOMEM;
678  }
679 
680  return OK;
681  }
682 
683  case SENSORIOCRESET:
684  // Nothing to do for simulator
685  return OK;
686 
687  case MAGIOCSSAMPLERATE:
688  // No need to set internal sampling rate for simulator
689  return OK;
690 
691  case MAGIOCGSAMPLERATE:
692  return _mag_samplerate;
693 
694  case MAGIOCSSCALE:
695  /* copy scale in */
696  memcpy(&_mag_scale, (struct mag_calibration_s *) arg, sizeof(_mag_scale));
697  return OK;
698 
699  case MAGIOCGSCALE:
700  /* copy scale out */
701  memcpy((struct mag_calibration_s *) arg, &_mag_scale, sizeof(_mag_scale));
702  return OK;
703 
704  case MAGIOCSRANGE:
705  return mag_set_range(arg);
706 
707  case MAGIOCGRANGE:
708  return _mag_range_ga;
709 
710  case MAGIOCGEXTERNAL:
711  /* Even if this sensor is on the "external" SPI bus
712  * it is still fixed to the autopilot assembly,
713  * so always return 0.
714  */
715  return 0;
716
