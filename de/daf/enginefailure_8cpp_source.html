84 void
85 EngineFailure::set_ef_item()
86 {
87  struct position_setpoint_triplet_s *pos_sp_triplet = _navigator->get_position_setpoint_triplet();
88 
89  pos_sp_triplet->previous = pos_sp_triplet->current;
90  _navigator->set_can_loiter_at_sp(false);
91 
92  switch (_ef_state) {
93  case EF_STATE_LOITERDOWN: {
94  //XXX create mission item at ground (below?) here
95 
96  _mission_item.lat = _navigator->get_global_position()->lat;
97  _mission_item.lon = _navigator->get_global_position()->lon;
98  _mission_item.altitude_is_relative = false;
99  //XXX setting altitude to a very low value, evaluate other options
100  _mission_item.altitude = _navigator->get_home_position()->alt - 1000.0f;
101  _mission_item.yaw = NAN;
102  _mission_item.loiter_radius = _navigator->get_loiter_radius();
103  _mission_item.nav_cmd = NAV_CMD_LOITER_UNLIMITED;
104  _mission_item.acceptance_radius = _navigator->get_acceptance_radius();
105  _mission_item.autocontinue = true;
106  _mission_item.origin = ORIGIN_ONBOARD;
107 
108  _navigator->set_can_loiter_at_sp(true);
109  break;
110  }
111 
112  default:
113  break;
114  }
115 
116  reset_mission_item_reached();
117 
118  /* convert mission item to current position setpoint and make it valid */
119  mission_apply_limitation(_mission_item);
120  mission_item_to_position_setpoint(_mission_item, &pos_sp_triplet->current);
121  pos_sp_triplet->next.valid = false;
122 
123  _navigator->set_position_setpoint_triplet_updated();
124 }

134 
135  default:
136  break;
137  }
