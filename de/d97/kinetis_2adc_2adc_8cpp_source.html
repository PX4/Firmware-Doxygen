163 ADC::ADC(adc_chan_t channels) :
164  CDev("adc", ADC0_DEVICE_PATH),
165  _sample_perf(perf_alloc(PC_ELAPSED, "adc_samples")),
166  _channels(channels),
167  _channel_count(0),
168  _samples(nullptr),
169  _to_system_power(nullptr),
170  _to_adc_report(nullptr)
171 {
172  _debug_enabled = true;
173 
174  /* always enable the temperature sensor */
175  channels |= 1 << (ADC_SC1_ADCH_TEMP >> ADC_SC1_ADCH_SHIFT);
176 
177  /* allocate the sample array */
178  for (unsigned i = 0; i < ADC_TOTAL_CHANNELS; i++) {
179  if (channels & (1 << i)) {
180  _channel_count++;
181  }
182  }
183 
184  _samples = new adc_msg_s[_channel_count];
185 
186  /* prefill the channel numbers in the sample array */
187 
188  if (_samples != nullptr) {
189  unsigned index = 0;
190 
191  for (unsigned i = 0; i < ADC_TOTAL_CHANNELS; i++) {
192  if (channels & (1 << i)) {
193  _samples[index].am_channel = i;
194  _samples[index].am_data = 0;
195  index++;
196  }
197  }
198  }
199 }

212 int
213 ADC::init()
214 {
215  /* Input is Buss Clock 56 Mhz We will use /8 for 7 Mhz */
216 
217  irqstate_t flags = px4_enter_critical_section();
218 
219  _REG(KINETIS_SIM_SCGC3) |= SIM_SCGC3_ADC1;
220  rCFG1(1) = ADC_CFG1_ADICLK_BUSCLK | ADC_CFG1_MODE_1213BIT | ADC_CFG1_ADIV_DIV8;
221  rCFG2(1) = 0;
222  rSC2(1) = ADC_SC2_REFSEL_DEFAULT;
223 
224  px4_leave_critical_section(flags);
225 
226  /* Clear the CALF and begin the calibration */
227 
228  rSC3(1) = ADC_SC3_CAL | ADC_SC3_CALF;
229 
230  while ((rSC1A(1) & ADC_SC1_COCO) == 0) {
231  usleep(100);
232 
233  if (rSC3(1) & ADC_SC3_CALF) {
234  return -1;
235  }
236  }
237 
238  /* dummy read to clear COCO of calibration */
239 
240  int32_t r = rRA(1);
241 
242  /* Check the state of CALF at the end of calibration */
243 
244  if (rSC3(1) & ADC_SC3_CALF) {
245  return -1;
246  }
247 
248  /* Calculate the calibration values for single ended positive */
249 
250  r = rCLP0(1) + rCLP1(1) + rCLP2(1) + rCLP3(1) + rCLP4(1) + rCLPS(1) ;
251  r = 0x8000U | (r >> 1U);
252  rPG(1) = r;
253 
254  /* Calculate the calibration values for double ended Negitive */
255 
256  r = rCLM0(1) + rCLM1(1) + rCLM2(1) + rCLM3(1) + rCLM4(1) + rCLMS(1) ;
257  r = 0x8000U | (r >> 1U);
258  rMG(1) = r;
259 
260  /* kick off a sample and wait for it to complete */
261  hrt_abstime now = hrt_absolute_time();
262 
263  rSC1A(1) = ADC_SC1_ADCH(ADC_SC1_ADCH_TEMP);
264 
265  while (!(rSC1A(1) & ADC_SC1_COCO)) {
266 
267  /* don't wait for more than 500us, since that means something broke - should reset here if we see this */
268  if ((hrt_absolute_time() - now) > 500) {
269  DEVICE_LOG("sample timeout");
270  return -1;
271  }
272 
273  break;
274  }
275 
276 
277  /* create the device node */
278  return CDev::init();
279 }

349  unsigned max_num = _channel_count;

419 uint16_t
420 ADC::_sample(unsigned channel)
421 {
422  perf_begin(_sample_perf);
423 
424  /* clear any previous COCC */
425  uint16_t result = rRA(1);
426 
427  /* run a single conversion right now - should take about 35 cycles (5 microseconds) max */
428 
429  rSC1A(1) = ADC_SC1_ADCH(channel);
430 
431  /* wait for the conversion to complete */
432  hrt_abstime now = hrt_absolute_time();
433 
434  while (!(rSC1A(1) & ADC_SC1_COCO)) {
435 
436  /* don't wait for more than 10us, since that means something broke - should reset here if we see this */
437  if ((hrt_absolute_time() - now) > 10) {
438  DEVICE_LOG("sample timeout");
439  return 0xffff;
440  }
441  }
442 
443  /* read the result and clear EOC */
444  result = rRA(1);
445 
446  perf_end(_sample_perf);
447  return result;
448 }
