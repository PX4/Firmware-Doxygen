578 int
579 PX4IO::init()
580 {
581  int ret;
582  param_t sys_restart_param;
583  int32_t sys_restart_val = DM_INIT_REASON_VOLATILE;
584 
585  sys_restart_param = param_find("SYS_RESTART_TYPE");
586 
587  if (sys_restart_param != PARAM_INVALID) {
588  /* Indicate restart type is unknown */
589  int32_t prev_val;
590  param_get(sys_restart_param, &prev_val);
591 
592  if (prev_val != DM_INIT_REASON_POWER_ON) {
593  param_set_no_notification(sys_restart_param, &sys_restart_val);
594  }
595  }
596 
597  /* do regular cdev init */
598  ret = CDev::init();
599 
600  if (ret != OK) {
601  return ret;
602  }
603 
604  /* get some parameters */
605  unsigned protocol;
606  hrt_abstime start_try_time = hrt_absolute_time();
607 
608  do {
609  usleep(2000);
610  protocol = io_reg_get(PX4IO_PAGE_CONFIG, PX4IO_P_CONFIG_PROTOCOL_VERSION);
611  } while (protocol == _io_reg_get_error && (hrt_elapsed_time(&start_try_time) < 700U * 1000U));
612 
613  /* if the error still persists after timing out, we give up */
614  if (protocol == _io_reg_get_error) {
615  mavlink_log_emergency(&_mavlink_log_pub, "Failed to communicate with IO, abort.");
616  return -1;
617  }
618 
619  if (protocol != PX4IO_PROTOCOL_VERSION) {
620  mavlink_log_emergency(&_mavlink_log_pub, "IO protocol/firmware mismatch, abort.");
621  return -1;
622  }
623 
624  _hardware = io_reg_get(PX4IO_PAGE_CONFIG, PX4IO_P_CONFIG_HARDWARE_VERSION);
625  _max_actuators = io_reg_get(PX4IO_PAGE_CONFIG, PX4IO_P_CONFIG_ACTUATOR_COUNT);
626  _max_controls = io_reg_get(PX4IO_PAGE_CONFIG, PX4IO_P_CONFIG_CONTROL_COUNT);
627  _max_relays = io_reg_get(PX4IO_PAGE_CONFIG, PX4IO_P_CONFIG_RELAY_COUNT);
628  _max_transfer = io_reg_get(PX4IO_PAGE_CONFIG, PX4IO_P_CONFIG_MAX_TRANSFER) - 2;
629  _max_rc_input = io_reg_get(PX4IO_PAGE_CONFIG, PX4IO_P_CONFIG_RC_INPUT_COUNT);
630 
631  if ((_max_actuators < 1) || (_max_actuators > 16) ||
632  (_max_relays > 32) ||
633  (_max_transfer < 16) || (_max_transfer > 255) ||
634  (_max_rc_input < 1) || (_max_rc_input > 255)) {
635 
636  DEVICE_LOG("config read error");
637  mavlink_log_emergency(&_mavlink_log_pub, "[IO] config read fail, abort.");
638 
639  // ask IO to reboot into bootloader as the failure may
640  // be due to mismatched firmware versions and we want
641  // the startup script to be able to load a new IO
642  // firmware
643 
644  // If IO has already safety off it won't accept going into bootloader mode,
645  // therefore we need to set safety on first.
646  io_reg_set(PX4IO_PAGE_SETUP, PX4IO_P_SETUP_FORCE_SAFETY_ON, PX4IO_FORCE_SAFETY_MAGIC);
647 
648  // Now the reboot into bootloader mode should succeed.
649  io_reg_set(PX4IO_PAGE_SETUP, PX4IO_P_SETUP_REBOOT_BL, PX4IO_REBOOT_BL_MAGIC);
650  return -1;
651  }
652 
653  if (_max_rc_input > input_rc_s::RC_INPUT_MAX_CHANNELS) {
654  _max_rc_input = input_rc_s::RC_INPUT_MAX_CHANNELS;
655  }
656 
657  param_get(param_find("RC_RSSI_PWM_CHAN"), &_rssi_pwm_chan);
658  param_get(param_find("RC_RSSI_PWM_MAX"), &_rssi_pwm_max);
659  param_get(param_find("RC_RSSI_PWM_MIN"), &_rssi_pwm_min);
660 
661  /*
662  * Check for IO flight state - if FMU was flagged to be in
663  * armed state, FMU is recovering from an in-air reset.
664  * Read back status and request the commander to arm
665  * in this case.
666  */
667 
668  uint16_t reg;
669 
670  /* get IO's last seen FMU state */
671  ret = io_reg_get(PX4IO_PAGE_SETUP, PX4IO_P_SETUP_ARMING, &reg, sizeof(reg));
672 
673  if (ret != OK) {
674  return ret;
675  }
676 
677  /*
678  * in-air restart is only tried if the IO board reports it is
679  * already armed, and has been configured for in-air restart
680  */
681  if ((reg & PX4IO_P_SETUP_ARMING_INAIR_RESTART_OK) &&
682  (reg & PX4IO_P_SETUP_ARMING_FMU_ARMED)) {
683 
684  /* get a status update from IO */
685  io_get_status();
686 
687  mavlink_log_emergency(&_mavlink_log_pub, "RECOVERING FROM FMU IN-AIR RESTART");
688 
689  /* WARNING: COMMANDER app/vehicle status must be initialized.
690  * If this fails (or the app is not started), worst-case IO
691  * remains untouched (so manual override is still available).
692  */
693 
694  int safety_sub = orb_subscribe(ORB_ID(actuator_armed));
695  /* fill with initial values, clear updated flag */
696  struct actuator_armed_s safety;
697  uint64_t try_start_time = hrt_absolute_time();
698  bool updated = false;
699 
700  /* keep checking for an update, ensure we got a arming information,
701  not something that was published a long time ago. */
702  do {
703  orb_check(safety_sub, &updated);
704 
705  if (updated) {
706  /* got data, copy and exit loop */
707  orb_copy(ORB_ID(actuator_armed), safety_sub, &safety);
708  break;
709  }
710 
711  /* wait 10 ms */
712  usleep(10000);
713 
714  /* abort after 5s */
715  if ((hrt_absolute_time() - try_start_time) / 1000 > 3000) {
716  mavlink_log_emergency(&_mavlink_log_pub, "Failed to recover from in-air restart (1), abort");
717  return 1;
718  }
719 
720  } while (true);
721 
722  /* send this to itself */
723  param_t sys_id_param = param_find("MAV_SYS_ID");
724  param_t comp_id_param = param_find("MAV_COMP_ID");
725 
726  int32_t sys_id;
727  int32_t comp_id;
728 
729  if (param_get(sys_id_param, &sys_id)) {
730  errx(1, "PRM SYSID");
731  }
732 
733  if (param_get(comp_id_param, &comp_id)) {
734  errx(1, "PRM CMPID");
735  }
736 
737  /* send command to arm system via command API */
738  vehicle_command_s vcmd = {};
739  vcmd.timestamp = hrt_absolute_time();
740  vcmd.param1 = 1.0f; /* request arming */
741  vcmd.command = vehicle_command_s::VEHICLE_CMD_COMPONENT_ARM_DISARM;
742  vcmd.target_system = (uint8_t)sys_id;
743  vcmd.target_component = (uint8_t)comp_id;
744  vcmd.source_system = (uint8_t)sys_id;
745  vcmd.source_component = (uint8_t)comp_id;
746  vcmd.confirmation = true; /* ask to confirm command */
747 
748  /* send command once */
749  orb_advert_t pub = orb_advertise_queue(ORB_ID(vehicle_command), &vcmd, vehicle_command_s::ORB_QUEUE_LENGTH);
750 
751  /* spin here until IO's state has propagated into the system */
752  do {
753  orb_check(safety_sub, &updated);
754 
755  if (updated) {
756  orb_copy(ORB_ID(actuator_armed), safety_sub, &safety);
757  }
758 
759  /* wait 50 ms */
760  usleep(50000);
761 
762  /* abort after 5s */
763  if ((hrt_absolute_time() - try_start_time) / 1000 > 2000) {
764  mavlink_log_emergency(&_mavlink_log_pub, "Failed to recover from in-air restart (2), abort");
765  return 1;
766  }
767 
768  /* re-send if necessary */
769  if (!safety.armed) {
770  orb_publish(ORB_ID(vehicle_command), pub, &vcmd);
771  DEVICE_LOG("re-sending arm cmd");
772  }
773 
774  /* keep waiting for state change for 2 s */
775  } while (!safety.armed);
776 
777  /* Indicate restart type is in-flight */
778  sys_restart_val = DM_INIT_REASON_IN_FLIGHT;
779  int32_t prev_val;
780  param_get(sys_restart_param, &prev_val);
781 
782  if (prev_val != sys_restart_val) {
783  param_set(sys_restart_param, &sys_restart_val);
784  }
785 
786  /* regular boot, no in-air restart, init IO */
787 
788  } else {
789 
790  /* dis-arm IO before touching anything */
791  io_reg_modify(PX4IO_PAGE_SETUP, PX4IO_P_SETUP_ARMING,
792  PX4IO_P_SETUP_ARMING_FMU_ARMED |
793  PX4IO_P_SETUP_ARMING_INAIR_RESTART_OK |
794  PX4IO_P_SETUP_ARMING_MANUAL_OVERRIDE_OK |
795  PX4IO_P_SETUP_ARMING_ALWAYS_PWM_ENABLE |
796  PX4IO_P_SETUP_ARMING_LOCKDOWN, 0);
797 
798  if (_rc_handling_disabled) {
799  ret = io_disable_rc_handling();
800 
801  if (ret != OK) {
802  DEVICE_LOG("failed disabling RC handling");
803  return ret;
804  }
805 
806  } else {
807  /* publish RC config to IO */
808  ret = io_set_rc_config();
809 
810  if (ret != OK) {
811  mavlink_log_critical(&_mavlink_log_pub, "IO RC config upload fail");
812  return ret;
813  }
814  }
815 
816  /* Indicate restart type is power on */
817  sys_restart_val = DM_INIT_REASON_POWER_ON;
818  int32_t prev_val;
819  param_get(sys_restart_param, &prev_val);
820 
821  if (prev_val != sys_restart_val) {
822  param_set(sys_restart_param, &sys_restart_val);
823  }
824 
825  }
826 
827  /* set safety to off if circuit breaker enabled */
828  if (circuit_breaker_enabled("CBRK_IO_SAFETY", CBRK_IO_SAFETY_KEY)) {
829  (void)io_reg_set(PX4IO_PAGE_SETUP, PX4IO_P_SETUP_FORCE_SAFETY_OFF, PX4IO_FORCE_SAFETY_MAGIC);
830  }
831 
832  /* try to claim the generic PWM output device node as well - it's OK if we fail at this */
833  ret = register_driver(PWM_OUTPUT0_DEVICE_PATH, &fops, 0666, (void *)this);
834 
835  if (ret == OK) {
836  PX4_INFO("default PWM output device");
837  _primary_pwm_device = true;
838  }
839 
840  /* start the IO interface task */
841  _task = px4_task_spawn_cmd("px4io",
842  SCHED_DEFAULT,
843  SCHED_PRIORITY_ACTUATOR_OUTPUTS,
844  1500,
845  (main_t)&PX4IO::task_main_trampoline,
846  nullptr);
847 
848  if (_task < 0) {
849  DEVICE_DEBUG("task start failed: %d", errno);
850  return -errno;
851  }
852 
853  return OK;
854 }
