78 void task_main(int argc, char *argv[])
79 {
80  const char *device_path = SPEKTRUM_UART_DEVICE_PATH;
81  int ch;
82  int myoptind = 1;
83  const char *myoptarg = NULL;
84 
85  while ((ch = px4_getopt(argc, argv, "d:", &myoptind, &myoptarg)) != EOF) {
86  switch (ch) {
87  case 'd':
88  device_path = myoptarg;
89  break;
90 
91  default:
92  break;
93  }
94  }
95 
96  int uart_fd = dsm_init(device_path);
97 
98  if (uart_fd < 1) {
99  PX4_ERR("dsm init failed");
100  return;
101  }
102 
103  orb_advert_t rc_pub = nullptr;
104 
105  // Use a buffer size of the double of the minimum, just to be safe.
106  uint8_t rx_buf[2 * DSM_BUFFER_SIZE];
107 
108  _is_running = true;
109  uint16_t raw_rc_values[input_rc_s::RC_INPUT_MAX_CHANNELS];
110  uint16_t raw_rc_count = 0;
111 
112  // Main loop
113  while (!_task_should_exit) {
114 
115  int newbytes = ::read(uart_fd, &rx_buf[0], sizeof(rx_buf));
116 
117  if (newbytes < 0) {
118  PX4_WARN("read failed");
119  continue;
120  }
121 
122  if (newbytes == 0) {
123  continue;
124  }
125 
126  const hrt_abstime now = hrt_absolute_time();
127 
128  bool dsm_11_bit;
129  unsigned frame_drops;
130  int8_t dsm_rssi;
131 
132  // parse new data
133  bool rc_updated = dsm_parse(now, rx_buf, newbytes, &raw_rc_values[0], &raw_rc_count,
134  &dsm_11_bit, &frame_drops, &dsm_rssi, input_rc_s::RC_INPUT_MAX_CHANNELS);
135  UNUSED(dsm_11_bit);
136 
137  if (rc_updated) {
138 
139  input_rc_s input_rc = {};
140 
141  fill_input_rc(raw_rc_count, raw_rc_values, now, false, false, frame_drops, dsm_rssi,
142  input_rc);
143 
144  if (rc_pub == nullptr) {
145  rc_pub = orb_advertise(ORB_ID(input_rc), &input_rc);
146 
147  } else {
148  orb_publish(ORB_ID(input_rc), rc_pub, &input_rc);
149  }
150  }
151 
152  // sleep since no poll for qurt
153  usleep(10000);
154 
155  }
156 
157  orb_unadvertise(rc_pub);
158  dsm_deinit();
159 
160  _is_running = false;
161 }

163 void fill_input_rc(uint16_t raw_rc_count, uint16_t raw_rc_values[input_rc_s::RC_INPUT_MAX_CHANNELS],
164  hrt_abstime now, bool frame_drop, bool failsafe, unsigned frame_drops, int rssi,
165  input_rc_s &input_rc)
166 {
167  input_rc.input_source = input_rc_s::RC_INPUT_SOURCE_QURT;
168 
169  input_rc.channel_count = raw_rc_count;
170 
171  if (input_rc.channel_count > input_rc_s::RC_INPUT_MAX_CHANNELS) {
172  input_rc.channel_count = input_rc_s::RC_INPUT_MAX_CHANNELS;
173  }
174 
175  unsigned valid_chans = 0;
176 
177  for (unsigned i = 0; i < input_rc.channel_count; ++i) {
178  input_rc.values[i] = raw_rc_values[i];
179 
180  if (raw_rc_values[i] != UINT16_MAX) {
181  valid_chans++;
182  }
183  }
184 
185  input_rc.timestamp = now;
186  input_rc.timestamp_last_signal = input_rc.timestamp;
187  input_rc.rc_ppm_frame_length = 0;
188 
189  /* fake rssi if no value was provided */
190  if (rssi == -1) {
191 
192  input_rc.rssi = 255;
193 
194  } else {
195  input_rc.rssi = rssi;
196  }
197 
198  if (valid_chans == 0) {
199  input_rc.rssi = 0;
200  }
201 
202  input_rc.rc_failsafe = failsafe;
203  input_rc.rc_lost = (valid_chans == 0);
204  input_rc.rc_lost_frame_count = frame_drops;
205  input_rc.rc_total_frame_count = 0;
206 }

224  (char *const *)argv);
