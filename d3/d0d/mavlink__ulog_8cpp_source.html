52 MavlinkULog::MavlinkULog(int datarate, float max_rate_factor, uint8_t target_system, uint8_t target_component)
53  : _target_system(target_system), _target_component(target_component),
54  _max_rate_factor(max_rate_factor),
55  _max_num_messages(math::max(1, (int)ceilf(_rate_calculation_delta_t *_max_rate_factor * datarate /
56  (MAVLINK_MSG_ID_LOGGING_DATA_LEN + MAVLINK_NUM_NON_PAYLOAD_BYTES)))),
57  _current_rate_factor(max_rate_factor)
58 {
59  _ulog_stream_sub = orb_subscribe(ORB_ID(ulog_stream));
60 
61  if (_ulog_stream_sub < 0) {
62  PX4_ERR("orb_subscribe failed (%i)", errno);
63 
64  } else {
65  // make sure we won't read any old messages
66  struct ulog_stream_s stream_msg;
67  bool update;
68 
69  while (orb_check(_ulog_stream_sub, &update) == 0 && update) {
70  orb_copy(ORB_ID(ulog_stream), _ulog_stream_sub, &stream_msg);
71  }
72  }
73 
74  _waiting_for_initial_ack = true;
75  _last_sent_time = hrt_absolute_time(); //(ab)use this timestamp during initialization
76  _next_rate_check = _last_sent_time + _rate_calculation_delta_t * 1.e6f;
77 }

99 int MavlinkULog::handle_update(mavlink_channel_t channel)
100 {
101  static_assert(sizeof(ulog_stream_s::data) == MAVLINK_MSG_LOGGING_DATA_FIELD_DATA_LEN,
102  "Invalid uorb ulog_stream.data length");
103  static_assert(sizeof(ulog_stream_s::data) == MAVLINK_MSG_LOGGING_DATA_ACKED_FIELD_DATA_LEN,
104  "Invalid uorb ulog_stream.data length");
105 
106  if (_waiting_for_initial_ack) {
107  if (hrt_elapsed_time(&_last_sent_time) > 3e5) {
108  PX4_WARN("no ack from logger (is it running?)");
109  return -1;
110  }
111 
112  return 0;
113  }
114 
115  // check if we're waiting for an ACK
116  if (_last_sent_time) {
117  bool check_for_updates = false;
118 
119  if (_ack_received) {
120  _last_sent_time = 0;
121  check_for_updates = true;
122 
123  } else {
124 
125  if (hrt_elapsed_time(&_last_sent_time) > ulog_stream_ack_s::ACK_TIMEOUT * 1000) {
126  if (++_sent_tries > ulog_stream_ack_s::ACK_MAX_TRIES) {
127  return -ETIMEDOUT;
128 
129  } else {
130  PX4_DEBUG("re-sending ulog mavlink message (try=%i)", _sent_tries);
131  _last_sent_time = hrt_absolute_time();
132  mavlink_logging_data_acked_t msg;
133  msg.sequence = _ulog_data.sequence;
134  msg.length = _ulog_data.length;
135  msg.first_message_offset = _ulog_data.first_message_offset;
136  msg.target_system = _target_system;
137  msg.target_component = _target_component;
138  memcpy(msg.data, _ulog_data.data, sizeof(msg.data));
139  mavlink_msg_logging_data_acked_send_struct(channel, &msg);
140  }
141  }
142  }
143 
144  if (!check_for_updates) {
145  return 0;
146  }
147  }
148 
149  bool updated = false;
150  int ret = orb_check(_ulog_stream_sub, &updated);
151 
152  while (updated && !ret && _current_num_msgs < _max_num_messages) {
153  orb_copy(ORB_ID(ulog_stream), _ulog_stream_sub, &_ulog_data);
154 
155  if (_ulog_data.timestamp > 0) {
156  if (_ulog_data.flags & ulog_stream_s::FLAGS_NEED_ACK) {
157  _sent_tries = 1;
158  _last_sent_time = hrt_absolute_time();
159  lock();
160  _wait_for_ack_sequence = _ulog_data.sequence;
161  _ack_received = false;
162  unlock();
163 
164  mavlink_logging_data_acked_t msg;
165  msg.sequence = _ulog_data.sequence;
166  msg.length = _ulog_data.length;
167  msg.first_message_offset = _ulog_data.first_message_offset;
168  msg.target_system = _target_system;
169  msg.target_component = _target_component;
170  memcpy(msg.data, _ulog_data.data, sizeof(msg.data));
171  mavlink_msg_logging_data_acked_send_struct(channel, &msg);
172 
173  } else {
174  mavlink_logging_data_t msg;
175  msg.sequence = _ulog_data.sequence;
176  msg.length = _ulog_data.length;
177  msg.first_message_offset = _ulog_data.first_message_offset;
178  msg.target_system = _target_system;
179  msg.target_component = _target_component;
180  memcpy(msg.data, _ulog_data.data, sizeof(msg.data));
181  mavlink_msg_logging_data_send_struct(channel, &msg);
182  }
183  }
184 
185  ++_current_num_msgs;
186  ret = orb_check(_ulog_stream_sub, &updated);
187  }
188 
189  //need to update the rate?
190  hrt_abstime t = hrt_absolute_time();
191 
192  if (t > _next_rate_check) {
193  if (_current_num_msgs < _max_num_messages) {
194  _current_rate_factor = _max_rate_factor * (float)_current_num_msgs / _max_num_messages;
195 
196  } else {
197  _current_rate_factor = _max_rate_factor;
198  }
199 
200  _current_num_msgs = 0;
201  _next_rate_check = t + _rate_calculation_delta_t * 1.e6f;
202  PX4_DEBUG("current rate=%.3f (max=%i msgs in %.3fs)", (double)_current_rate_factor, _max_num_messages,
203  (double)_rate_calculation_delta_t);
204  }
205 
206  return 0;
207 }
