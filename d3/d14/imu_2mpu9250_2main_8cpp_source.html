171 struct mpu9250_bus_option &find_bus(enum MPU9250_BUS busid)
172 {
173  for (uint8_t i = 0; i < NUM_BUS_OPTIONS; i++) {
174  if ((busid == MPU9250_BUS_ALL ||
175  busid == bus_options[i].busid) && bus_options[i].dev != NULL) {

177  }
178  }

180  errx(1, "bus %u not started", (unsigned)busid);
181 }

183 /**
184  * start driver for a specific bus option
185  */
186 bool
187 start_bus(struct mpu9250_bus_option &bus, enum Rotation rotation, bool external)
188 {
189  int fd = -1;
190 
191  if (bus.dev != nullptr) {
192  warnx("%s SPI not available", external ? "External" : "Internal");
193  return false;
194  }
195 
196  device::Device *interface = bus.interface_constructor(bus.busnum, bus.address, external);
197 
198  if (interface == nullptr) {
199  warnx("no device on bus %u", (unsigned)bus.busid);
200  return false;
201  }
202 
203  if (interface->init() != OK) {
204  delete interface;
205  warnx("no device on bus %u", (unsigned)bus.busid);
206  return false;
207  }
208 
209  device::Device *mag_interface = nullptr;
210 
211 #ifdef USE_I2C
212  /* For i2c interfaces, connect to the magnetomer directly */
213  bool is_i2c = bus.busid == MPU9250_BUS_I2C_INTERNAL || bus.busid == MPU9250_BUS_I2C_EXTERNAL;
214 
215  if (is_i2c) {
216  mag_interface = AK8963_I2C_interface(bus.busnum, external);
217  }
218 
219 #endif
220 
221  bus.dev = new MPU9250(interface, mag_interface, bus.accelpath, bus.gyropath, bus.magpath, rotation);
222 
223  if (bus.dev == nullptr) {
224  delete interface;
225  return false;
226  }
227 
228  if (OK != bus.dev->init()) {
229  goto fail;
230  }
231 
232  /* set the poll rate to default, starts automatic data collection */
233  fd = open(bus.accelpath, O_RDONLY);
234 
235  if (fd < 0) {
236  goto fail;
237  }
238 
239  if (ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
240  goto fail;
241  }
242 
243 
244  close(fd);
245 
246  return true;
247 
248 fail:
249 
250  if (fd >= 0) {
251  close(fd);
252  }
253 
254  if (bus.dev != nullptr) {
255  delete (bus.dev);
256  bus.dev = nullptr;
257  }
258 
259  errx(1, "driver start failed");
260 }

262 /**
263  * Start the driver.
264  *
265  * This function only returns if the driver is up and running
266  * or failed to detect the sensor.
267  */
268 void
269 start(enum MPU9250_BUS busid, enum Rotation rotation, bool external)
270 {
271 
272  bool started = false;
273 
274  for (unsigned i = 0; i < NUM_BUS_OPTIONS; i++) {
275  if (busid == MPU9250_BUS_ALL && bus_options[i].dev != NULL) {
276  // this device is already started
277  continue;
278  }
279 
280  if (busid != MPU9250_BUS_ALL && bus_options[i].busid != busid) {
281  // not the one that is asked for
282  continue;
283  }
284 
285  started |= start_bus(bus_options[i], rotation, external);
286  }
287 
288  exit(started ? 0 : 1);
289 
290 }

310 /**
311  * Perform some basic functional tests on the driver;
312  * make sure we can collect data from the sensor in polled
313  * and automatic modes.
314  */
315 void
316 test(enum MPU9250_BUS busid)
317 {
318  struct mpu9250_bus_option &bus = find_bus(busid);
319  accel_report a_report;
320  gyro_report g_report;
321  mag_report m_report;
322  ssize_t sz;
323 
324  /* get the driver */
325  int fd = open(bus.accelpath, O_RDONLY);
326 
327  if (fd < 0) {
328  err(1, "%s open failed (try 'm start')", bus.accelpath);
329  }
330 
331  /* get the driver */
332  int fd_gyro = open(bus.gyropath, O_RDONLY);
333 
334  if (fd_gyro < 0) {
335  err(1, "%s open failed", bus.gyropath);
336  }
337 
338  /* get the driver */
339  int fd_mag = open(bus.magpath, O_RDONLY);
340 
341  if (fd_mag < 0) {
342  err(1, "%s open failed", bus.magpath);
343  }
344 
345  /* reset to manual polling */
346  if (ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_MANUAL) < 0) {
347  err(1, "reset to manual polling");
348  }
349 
350  /* do a simple demand read */
351  sz = read(fd, &a_report, sizeof(a_report));
352 
353  if (sz != sizeof(a_report)) {
354  warnx("ret: %d, expected: %d", sz, sizeof(a_report));
355  err(1, "immediate acc read failed");
356  }
357 
358  print_message(a_report);
359 
360  /* do a simple demand read */
361  sz = read(fd_gyro, &g_report, sizeof(g_report));
362 
363  if (sz != sizeof(g_report)) {
364  warnx("ret: %d, expected: %d", sz, sizeof(g_report));
365  err(1, "immediate gyro read failed");
366  }
367 
368  print_message(g_report);
369 
370  /* do a simple demand read */
371  sz = read(fd_mag, &m_report, sizeof(m_report));
372 
373  if (sz != sizeof(m_report)) {
374  warnx("ret: %d, expected: %d", sz, sizeof(m_report));
375  err(1, "immediate mag read failed");
376  }
377 
378  print_message(m_report);
379 
380  /* reset to default polling */
381  if (ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
382  err(1, "reset to default polling");
383  }
384 
385  close(fd);
386  close(fd_gyro);
387  close(fd_mag);
388 
389  /* XXX add poll-rate tests here too */
390 
391  reset(busid);
392  errx(0, "PASS");
393 }

493 int
494 mpu9250_main(int argc, char *argv[])
495 {
496  int myoptind = 1;

498  const char *myoptarg = nullptr;
499 
500  enum MPU9250_BUS busid = MPU9250_BUS_ALL;

502 
503  while ((ch = px4_getopt(argc, argv, "XISstR:", &myoptind, &myoptarg)) != EOF) {
504  switch (ch) {
505  case 'X':
506  busid = MPU9250_BUS_I2C_EXTERNAL;
507  break;
508 
509  case 'I':
510  busid = MPU9250_BUS_I2C_INTERNAL;
511  break;
512 
513  case 'S':
514  busid = MPU9250_BUS_SPI_EXTERNAL;
515  break;
516 
517  case 's':
518  busid = MPU9250_BUS_SPI_INTERNAL;
519  break;
520 
521  case 't':
522  busid = MPU9250_BUS_SPI_INTERNAL2;
523  break;
524 
525  case 'R':
526  rotation = (enum Rotation)atoi(myoptarg);
527  break;
528 
529  default:
530  mpu9250::usage();
531  return 0;
532  }
533  }
534 
535  if (myoptind >= argc) {
536  mpu9250::usage();
537  return -1;
538  }
539 
540  bool external = busid == MPU9250_BUS_I2C_EXTERNAL || busid == MPU9250_BUS_SPI_EXTERNAL;
541  const char *verb = argv[myoptind];
542 
543  /*
544  * Start/load the driver.
545  */
546  if (!strcmp(verb, "start")) {
547  mpu9250::start(busid, rotation, external);
548  }
549 
550  if (!strcmp(verb, "stop")) {
551  mpu9250::stop(busid);
552  }
553 
554  /*
555  * Test the driver/device.
556  */
557  if (!strcmp(verb, "test")) {
558  mpu9250::test(busid);
559  }
560 
561  /*
562  * Reset the driver.
563  */
564  if (!strcmp(verb, "reset")) {
565  mpu9250::reset(busid);
566  }
567 
568  /*
569  * Print driver information.
570  */
571  if (!strcmp(verb, "info")) {
572  mpu9250::info(busid);
573  }
574 
575  /*
576  * Print register information.
577  */
578  if (!strcmp(verb, "regdump")) {
579  mpu9250::regdump(busid);
580  }
581 
582  if (!strcmp(verb, "testerror")) {
583  mpu9250::testerror(busid);
584  }
585 
586  mpu9250::usage();

588 }
