80 bool VtolLandDetector::_get_landed_state()
81 {
82  // Only trigger in RW mode
83  if ((_vehicle_status.timestamp != 0) && !_vehicle_status.is_rotary_wing) {
84  return false;
85  }
86 
87  // this is returned from the mutlicopter land detector
88  bool landed = MulticopterLandDetector::_get_landed_state();
89 
90  // for vtol we additionally consider airspeed
91  if (hrt_elapsed_time(&_airspeed.timestamp) < 500 * 1000 && _airspeed.confidence > 0.99f
92  && PX4_ISFINITE(_airspeed.indicated_airspeed_m_s)) {
93 
94  _airspeed_filtered = 0.95f * _airspeed_filtered + 0.05f * _airspeed.indicated_airspeed_m_s;
95 
96  } else {
97  // if airspeed does not update, set it to zero and rely on multicopter land detector
98  _airspeed_filtered = 0.0f;
99  }
100 
101  // only consider airspeed if we have been in air before to avoid false
102  // detections in the case of wind on the ground
103  if (_was_in_air && (_airspeed_filtered > _params.maxAirSpeed)) {
104  landed = false;
105  }
106 
107  _was_in_air = !landed;
108 
109  return landed;
110 }
