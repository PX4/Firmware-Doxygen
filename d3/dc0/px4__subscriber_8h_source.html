134  /**
135  * Called on topic update, saves the current message and then calls the provided callback function
136  * needs to use the native type as it is called by ROS
137  */
138  void callback(const typename std::remove_reference < decltype(((T *)nullptr)->data()) >::type &msg)
139  {
140  /* Store data */
141  this->_msg_current.data() = msg;
142 
143  /* Call callback */
144  if (_cbf != NULL) {
145  _cbf(this->get());
146  }
147 
148  }

151  std::function<void(const T &)> _cbf; /**< Callback that the user provided on the subscription */
