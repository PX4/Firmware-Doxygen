534 ssize_t
535 IST8310::read(struct file *filp, char *buffer, size_t buflen)
536 {
537  unsigned count = buflen / sizeof(struct mag_report);
538  struct mag_report *mag_buf = reinterpret_cast<struct mag_report *>(buffer);
539  int ret = 0;
540 
541  /* buffer must be large enough */
542  if (count < 1) {
543  return -ENOSPC;
544  }
545 
546  /* if automatic measurement is enabled */
547  if (_measure_ticks > 0) {
548  /*
549  * While there is space in the caller's buffer, and reports, copy them.
550  * Note that we may be pre-empted by the workq thread while we are doing this;
551  * we are careful to avoid racing with them.
552  */
553  while (count--) {
554  if (_reports->get(mag_buf)) {
555  ret += sizeof(struct mag_report);
556  mag_buf++;
557  }
558  }
559 
560  /* if there was no data, warn the caller */
561  return ret ? ret : -EAGAIN;
562  }
563 
564  /* manual measurement - run one conversion */
565  /* XXX really it'd be nice to lock against other readers here */
566  do {
567  _reports->flush();
568 
569  /* trigger a measurement */
570  if (OK != measure()) {
571  ret = -EIO;
572  break;
573  }
574 
575  /* wait for it to complete */
576  usleep(IST8310_CONVERSION_INTERVAL);
577 
578  /* run the collection phase */
579  if (OK != collect()) {
580  ret = -EIO;
581  break;
582  }
583 
584  if (_reports->get(mag_buf)) {
585  ret = sizeof(struct mag_report);
586  }
587  } while (0);
588 
589  return ret;
590 }

592 int
593 IST8310::ioctl(struct file *filp, int cmd, unsigned long arg)
594 {
595  switch (cmd) {
596  case SENSORIOCSPOLLRATE: {
597  switch (arg) {
598 
599  /* switching to manual polling */
600  case SENSOR_POLLRATE_MANUAL:
601  stop();
602  _measure_ticks = 0;
603  return OK;
604 
605  /* external signalling (DRDY) not supported */
606  case SENSOR_POLLRATE_EXTERNAL:
607 
608  /* zero would be bad */
609  case 0:
610  return -EINVAL;
611 
612  /* set default/max polling rate */
613  case SENSOR_POLLRATE_MAX:
614  case SENSOR_POLLRATE_DEFAULT: {
615  /* do we need to start internal polling? */
616  bool want_start = (_measure_ticks == 0);
617 
618  /* set interval for next measurement to minimum legal value */
619  _measure_ticks = USEC2TICK(IST8310_CONVERSION_INTERVAL);
620 
621  /* if we need to start the poll state machine, do it */
622  if (want_start) {
623  start();
624  }
625 
626  return OK;
627  }
628 
629  /* adjust to a legal polling interval in Hz */
630  default: {
631  /* do we need to start internal polling? */
632  bool want_start = (_measure_ticks == 0);
633 
634  /* convert hz to tick interval via microseconds */
635  unsigned ticks = USEC2TICK(1000000 / arg);
636 
637  /* check against maximum rate */
638  if (ticks < USEC2TICK(IST8310_CONVERSION_INTERVAL)) {
639  return -EINVAL;
640  }
641 
642  /* update interval for next measurement */
643  _measure_ticks = ticks;
644 
645  /* if we need to start the poll state machine, do it */
646  if (want_start) {
647  start();
648  }
649 
650  return OK;
651  }
652  }
653  }
654 
655  case SENSORIOCGPOLLRATE:
656  if (_measure_ticks == 0) {
657  return SENSOR_POLLRATE_MANUAL;
658  }
659 
660  return 1000000 / TICK2USEC(_measure_ticks);
661 
662  case SENSORIOCSQUEUEDEPTH: {
663  /* lower bound is mandatory, upper bound is a sanity check */
664  if ((arg < 1) || (arg > 100)) {
665  return -EINVAL;
666  }
667 
668  irqstate_t flags = px4_enter_critical_section();
669 
670  if (!_reports->resize(arg)) {
671  px4_leave_critical_section(flags);
672  return -ENOMEM;
673  }
674 
675  px4_leave_critical_section(flags);
676 
677  return OK;
678  }
679 
680  case SENSORIOCRESET:
681  return reset();
682 
683  case MAGIOCSSAMPLERATE:
684  /* same as pollrate because device is in single measurement mode*/
685  return ioctl(filp, SENSORIOCSPOLLRATE, arg);
686 
687  case MAGIOCGSAMPLERATE:
688  /* same as pollrate because device is in single measurement mode*/
689  return 1000000 / TICK2USEC(_measure_ticks);
690 
691  case MAGIOCEXSTRAP:
692  return set_selftest(arg);
693 
694  case MAGIOCSSCALE:
695  /* set new scale factors */
696  memcpy(&_scale, (struct mag_calibration_s *)arg, sizeof(_scale));
697  return 0;
698 
699  case MAGIOCGSCALE:
700  /* copy out scale factors */
701  memcpy((struct mag_calibration_s *)arg, &_scale, sizeof(_scale));
702  return 0;
703 
704  case MAGIOCCALIBRATE:
705  return calibrate(filp, arg);
706 
707  case MAGIOCGEXTERNAL:
708  return external();
709 
710  default:
711  /* give it to the superclass */
712  return CDev::ioctl(filp, cmd, arg);
713  }

847 IST8310::collect()
848 {

850  struct { /* status register and data as read back from the device */
851  uint8_t x[2];
852  uint8_t y[2];
853  uint8_t z[2];
854  } report_buffer;

856  struct {
857  int16_t x, y, z;
858  } report;
859 
860  int ret;
861  uint8_t check_counter;
862 
863  perf_begin(_sample_perf);
864  struct mag_report new_report;
865  const bool sensor_is_external = external();
866 
867  float xraw_f;
868  float yraw_f;
869  float zraw_f;
870 
871  /* this should be fairly close to the end of the measurement, so the best approximation of the time */
872  new_report.timestamp = hrt_absolute_time();
873  new_report.is_external = sensor_is_external;
874  new_report.error_count = perf_event_count(_comms_errors);
875  new_report.scaling = _range_scale;
876  new_report.device_id = _device_id.devid;
877 
878  /*
879  * @note We could read the status register here, which could tell us that
880  * we were too early and that the output registers are still being
881  * written. In the common case that would just slow us down, and
882  * we're better off just never being early.
883  */
884 
885  /* get measurements from the device */
886  ret = read(ADDR_DATA_OUT_X_LSB, (uint8_t *)&report_buffer, sizeof(report_buffer));
887 
888  if (ret != OK) {
889  perf_count(_comms_errors);
890  DEVICE_DEBUG("I2C read error");
891  goto out;
892  }
893 
894  /* swap the data we just received */
895  report.x = (((int16_t)report_buffer.x[1]) << 8) | (int16_t)report_buffer.x[0];
896  report.y = (((int16_t)report_buffer.y[1]) << 8) | (int16_t)report_buffer.y[0];
897  report.z = (((int16_t)report_buffer.z[1]) << 8) | (int16_t)report_buffer.z[0];
898 
899 
900  /*
901  * Check if value makes sense according to the FSR and Resolution of
902  * this sensor, discarding outliers
903  */

905  report.y > IST8310_MAX_VAL_XY || report.y < IST8310_MIN_VAL_XY ||
906  report.z > IST8310_MAX_VAL_Z || report.z < IST8310_MIN_VAL_Z) {
907  perf_count(_range_errors);
908  DEVICE_DEBUG("data/status read error");
909  goto out;
910  }
911 
912  /* temperature measurement is not available on IST8310 */
913  new_report.temperature = 0;
914 
915  /*
916  * raw outputs
917  *
918  * Sensor doesn't follow right hand rule, swap x and y to make it obey
919  * it.
920  */
921  new_report.x_raw = report.y;
922  new_report.y_raw = report.x;
923  new_report.z_raw = report.z;
924 
925  /* scale values for output */
926  xraw_f = report.y;
927  yraw_f = report.x;
928  zraw_f = report.z;
929 
930  /* apply user specified rotation */
931  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
932  new_report.x = ((xraw_f * _range_scale) - _scale.x_offset) * _scale.x_scale;
933  new_report.y = ((yraw_f * _range_scale) - _scale.y_offset) * _scale.y_scale;
934  new_report.z = ((zraw_f * _range_scale) - _scale.z_offset) * _scale.z_scale;
935 
936  if (!(_pub_blocked)) {
937 
938  if (_mag_topic != nullptr) {
939  /* publish it */
940  orb_publish(ORB_ID(sensor_mag), _mag_topic, &new_report);
941 
942  } else {
943  _mag_topic = orb_advertise_multi(ORB_ID(sensor_mag), &new_report,
944  &_orb_class_instance, sensor_is_external ? ORB_PRIO_MAX : ORB_PRIO_HIGH);
945 
946  if (_mag_topic == nullptr) {
947  DEVICE_DEBUG("ADVERT FAIL");
948  }
949  }
950  }
951 
952  _last_report = new_report;
953 
954  /* post a report to the ring */
955  _reports->force(&new_report);
956 
957  /* notify anyone waiting for data */
958  poll_notify(POLLIN);
959 
960  /*
961  periodically check the range register and configuration
962  registers. With a bad I2C cable it is possible for the
963  registers to become corrupt, leading to bad readings. It
964  doesn't happen often, but given the poor cables some
965  vehicles have it is worth checking for.
966  */
967  check_counter = perf_event_count(_sample_perf) % 256;
968 
969  if (check_counter == 128) {
970  check_conf();
971  }
972 
973  ret = OK;
974 
975 out:
976  perf_end(_sample_perf);
977  return ret;
