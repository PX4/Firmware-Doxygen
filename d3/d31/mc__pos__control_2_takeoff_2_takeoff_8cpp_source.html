48  const float takeoff_desired_vz, const bool skip_takeoff, const hrt_abstime &now_us)
49 {

51 
52  switch (_takeoff_state) {
53  case TakeoffState::disarmed:
54  if (armed) {
55  _takeoff_state = TakeoffState::spoolup;
56 
57  } else {
58  break;
59  }
60 
61  // FALLTHROUGH
62  case TakeoffState::spoolup:
63  if (_spoolup_time_hysteresis.get_state()) {
64  _takeoff_state = TakeoffState::ready_for_takeoff;
65 
66  } else {
67  break;
68  }
69 
70  // FALLTHROUGH
71  case TakeoffState::ready_for_takeoff:
72  if (want_takeoff) {
73  _takeoff_state = TakeoffState::rampup;
74  _takeoff_ramp_vz = _takeoff_ramp_vz_init;
75 
76  } else {
77  break;
78  }
79 
80  // FALLTHROUGH
81  case TakeoffState::rampup:
82  if (_takeoff_ramp_vz >= takeoff_desired_vz) {
83  _takeoff_state = TakeoffState::flight;
84 
85  } else {
86  break;
87  }
88 
89  // FALLTHROUGH
90  case TakeoffState::flight:
91  if (landed) {
92  _takeoff_state = TakeoffState::ready_for_takeoff;
93  }
94 
95  break;
96 
97  default:
98  break;
99  }
100 
101  if (armed && skip_takeoff) {
102  _takeoff_state = TakeoffState::flight;
103  }
104 
105  // TODO: need to consider free fall here
106  if (!armed) {
107  _takeoff_state = TakeoffState::disarmed;
108  }

111 float Takeoff::updateRamp(const float dt, const float takeoff_desired_vz)
112 {
113  if (_takeoff_state < TakeoffState::rampup) {
114  return _takeoff_ramp_vz_init;
115  }
116 
117  if (_takeoff_state == TakeoffState::rampup) {
118  if (_takeoff_ramp_time > dt) {
119  _takeoff_ramp_vz += (takeoff_desired_vz - _takeoff_ramp_vz_init) * dt / _takeoff_ramp_time;
120 
121  } else {
122  _takeoff_ramp_vz = takeoff_desired_vz;
123  }
124 
125  if (_takeoff_ramp_vz < takeoff_desired_vz) {
126  return _takeoff_ramp_vz;
127  }
128  }
129 
130  return takeoff_desired_vz;
131 }
