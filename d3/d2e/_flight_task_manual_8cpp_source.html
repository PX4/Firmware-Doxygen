68 bool FlightTaskManual::_evaluateSticks()
69 {
70  /* Sticks are rescaled linearly and exponentially to [-1,1] */
71  if ((_time_stamp_current - _sub_manual_control_setpoint->get().timestamp) < _timeout) {
72 
73  /* Linear scale */
74  _sticks(0) = _sub_manual_control_setpoint->get().x; /* NED x, "pitch" [-1,1] */
75  _sticks(1) = _sub_manual_control_setpoint->get().y; /* NED y, "roll" [-1,1] */
76  _sticks(2) = -(_sub_manual_control_setpoint->get().z - 0.5f) * 2.f; /* NED z, "thrust" resacaled from [0,1] to [-1,1] */
77  _sticks(3) = _sub_manual_control_setpoint->get().r; /* "yaw" [-1,1] */
78 
79  /* Exponential scale */
80  _sticks_expo(0) = math::expo_deadzone(_sticks(0), _xy_vel_man_expo.get(), _stick_dz.get());
81  _sticks_expo(1) = math::expo_deadzone(_sticks(1), _xy_vel_man_expo.get(), _stick_dz.get());
82  _sticks_expo(2) = math::expo_deadzone(_sticks(2), _z_vel_man_expo.get(), _stick_dz.get());
83  _sticks_expo(3) = math::expo_deadzone(_sticks(3), _yaw_expo.get(), _stick_dz.get());
84 
85  // Only switch the landing gear up if the user switched from gear down to gear up.
86  // If the user had the switch in the gear up position and took off ignore it
87  // until he toggles the switch to avoid retracting the gear immediately on takeoff.
88  int8_t gear_switch = _sub_manual_control_setpoint->get().gear_switch;
89 
90  if (!_constraints.landing_gear) {
91  if (gear_switch == manual_control_setpoint_s::SWITCH_POS_OFF) {
92  _applyGearSwitch(gear_switch);
93  }
94 
95  } else {
96  _applyGearSwitch(gear_switch);
97  }
98 
99  // valid stick inputs are required
100  const bool valid_sticks = PX4_ISFINITE(_sticks(0))
101  && PX4_ISFINITE(_sticks(1))
102  && PX4_ISFINITE(_sticks(2))
103  && PX4_ISFINITE(_sticks(3));
104 
105  return valid_sticks;
106 
107  } else {
108  /* Timeout: set all sticks to zero */
109  _sticks.zero();
110  _sticks_expo.zero();
111  _constraints.landing_gear = vehicle_constraints_s::GEAR_KEEP;
112  return false;
113  }
114 }
