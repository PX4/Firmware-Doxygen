102 /* This is used to schedule work for the logger (periodic scan for updated topics) */
103 static void timer_callback(void *arg)
104 {
105  /* Note: we are in IRQ context here (on NuttX) */
106 
107  timer_callback_data_s *data = (timer_callback_data_s *)arg;
108 
109  if (watchdog_update(data->watchdog_data)) {
110  data->watchdog_triggered = true;
111  }
112 
113  /* check the value of the semaphore: if the logger cannot keep up with running it's main loop as fast
114  * as the timer_callback here increases the semaphore count, the counter would increase unbounded,
115  * leading to an overflow at some point. This case we want to avoid here, so we check the current
116  * value against a (somewhat arbitrary) threshold, and avoid calling sem_post() if it's exceeded.
117  * (it's not a problem if the threshold is a bit too large, it just means the logger will do
118  * multiple iterations at once, the next time it's scheduled). */
119  int semaphore_value;
120 
121  if (px4_sem_getvalue(&data->semaphore, &semaphore_value) == 0 && semaphore_value > 1) {
122  return;
123  }
124 
125  px4_sem_post(&data->semaphore);
126 
127 }

497 bool Logger::add_topic(const char *name, unsigned interval)
498 {
499  const orb_metadata *const*topics = orb_get_topics();
500  LoggerSubscription *subscription = nullptr;
501 
502  for (size_t i = 0; i < orb_topics_count(); i++) {
503  if (strcmp(name, topics[i]->o_name) == 0) {
504  bool already_added = false;
505 
506  // check if already added: if so, only update the interval
507  for (size_t j = 0; j < _subscriptions.size(); ++j) {
508  if (_subscriptions[j].metadata == topics[i]) {
509  PX4_DEBUG("logging topic %s, interval: %i, already added, only setting interval",
510  topics[i]->o_name, interval);
511  subscription = &_subscriptions[j];
512  already_added = true;
513  break;
514  }
515  }
516 
517  if (!already_added) {
518  subscription = add_topic(topics[i]);
519  PX4_DEBUG("logging topic: %s, interval: %i", topics[i]->o_name, interval);
520  break;
521  }
522  }
523  }
524 
525  // if we poll on a topic, we don't use the interval and let the polled topic define the maximum interval
526  if (_polling_topic_meta) {
527  interval = 0;
528  }
529 
530  if (subscription) {
531  if (subscription->fd[0] >= 0) {
532  orb_set_interval(subscription->fd[0], interval);
533  } else {
534  // store the interval: use a value < 0 to know it's not a valid fd
535  subscription->fd[0] = -interval - 1;
536  }
537  }
538 
539  return subscription;
540 }

570 bool Logger::try_to_subscribe_topic(LoggerSubscription &sub, int multi_instance)
571 {
572  bool ret = false;
573  if (OK == orb_exists(sub.metadata, multi_instance)) {
574 
575  unsigned int interval;
576 
577  if (multi_instance == 0) {
578  // the first instance and no subscription yet: this means we stored the negative interval as fd
579  interval = (unsigned int) (-(sub.fd[0] + 1));
580  } else {
581  // set to the same interval as the first instance
582  if (orb_get_interval(sub.fd[0], &interval) != 0) {
583  interval = 0;
584  }
585  }
586 
587  int &handle = sub.fd[multi_instance];
588  handle = orb_subscribe_multi(sub.metadata, multi_instance);
589 
590  if (handle >= 0) {
591  PX4_DEBUG("subscribed to instance %d of topic %s", multi_instance, sub.metadata->o_name);
592  if (interval > 0) {
593  orb_set_interval(handle, interval);
594  }
595  ret = true;
596  } else {
597  PX4_ERR("orb_subscribe_multi %s failed (%i)", sub.metadata->o_name, errno);
598  }
599  }
600  return ret;

732 int Logger::add_topics_from_file(const char *fname)
733 {
734  FILE *fp;
735  char line[80];
736  char topic_name[80];
737  unsigned interval;
738  int ntopics = 0;
739 
740  /* open the topic list file */
741  fp = fopen(fname, "r");
742 
743  if (fp == nullptr) {
744  return -1;
745  }
746 
747  /* call add_topic for each topic line in the file */
748  for (;;) {
749 
750  /* get a line, bail on error/EOF */
751  line[0] = '\0';
752 
753  if (fgets(line, sizeof(line), fp) == nullptr) {
754  break;
755  }
756 
757  /* skip comment lines */
758  if ((strlen(line) < 2) || (line[0] == '#')) {
759  continue;
760  }
761 
762  // read line with format: <topic_name>[, <interval>]
763  interval = 0;
764  int nfields = sscanf(line, "%s %u", topic_name, &interval);
765 
766  if (nfields > 0) {
767  int name_len = strlen(topic_name);
768 
769  if (name_len > 0 && topic_name[name_len - 1] == ',') {
770  topic_name[name_len - 1] = '\0';
771  }
772 
773  /* add topic with specified interval */
774  if (add_topic(topic_name, interval)) {
775  ntopics++;
776 
777  } else {
778  PX4_ERR("Failed to add topic %s", topic_name);
779  }
780  }
781  }
782 
783  fclose(fp);
784  return ntopics;
785 }

1067

1117

1121  }
1122  }

1125  }

1188  }
1189 
1190 #endif /* DBGPRINT */
1191 
1192  } else { // not logging
1193 
1194  // try to subscribe to new topics, even if we don't log, so that:
1195  // - we avoid subscribing to many topics at once, when logging starts
1196  // - we'll get the data immediately once we start logging (no need to wait for the next subscribe timeout)

1202  }

1206  }
1207 
1208  } else if (loop_time > next_subscribe_check) {
1209  next_subscribe_topic_index = 0;
1210  }
1211  }
1212 
1213  // wait for next loop iteration...
1214  if (polling_topic_sub >= 0) {
1215  px4_pollfd_struct_t fds[1];
1216  fds[0].fd = polling_topic_sub;
1217  fds[0].events = POLLIN;
1218  int pret = px4_poll(fds, 1, 1000);
1219 
1220  if (pret < 0) {
1221  PX4_ERR("poll failed (%i)", pret);
1222 
1223  } else if (pret != 0) {
1224  if (fds[0].revents & POLLIN) {
1225  // need to to an orb_copy so that poll will not return immediately
1226  orb_copy(_polling_topic_meta, polling_topic_sub, _msg_buffer);
1227  }
1228  }
1229 
1230  } else {
1231  /*
1232  * We wait on the semaphore, which periodically gets updated by a high-resolution timer.
1233  * The simpler alternative would be:
1234  * usleep(max(300, _log_interval - elapsed_time_since_loop_start));
1235  * And on linux this is quite accurate as well, but under NuttX it is not accurate,
1236  * because usleep() has only a granularity of CONFIG_MSEC_PER_TICK (=1ms).
1237  */
1238  while (px4_sem_wait(&timer_callback_data.semaphore) != 0);
1239  }
1240  }
1241 
1242  stop_log_file();
1243 
1244  hrt_cancel(&timer_call);
1245  px4_sem_destroy(&timer_callback_data.semaphore);
1246 
1247  // stop the writer thread
1248  _writer.thread_stop();
1249 
1250  //unsubscribe
1251  for (LoggerSubscription &sub : _subscriptions) {
1252  for (int instance = 0; instance < ORB_MULTI_MAX_INSTANCES; instance++) {
1253  if (sub.fd[instance] >= 0) {
1254  orb_unsubscribe(sub.fd[instance]);
1255  sub.fd[instance] = -1;
1256  }
1257  }
1258  }
1259 
1260  if (polling_topic_sub >= 0) {
1261  orb_unsubscribe(polling_topic_sub);
1262  }
1263 
1264  if (_mavlink_log_pub) {
1265  orb_unadvertise(_mavlink_log_pub);
1266  _mavlink_log_pub = nullptr;
1267  }
1268 
1269  if (vehicle_command_ack_pub) {
1270  orb_unadvertise(vehicle_command_ack_pub);
1271  }
1272 
1273  if (vehicle_command_sub != -1) {
1274  orb_unsubscribe(vehicle_command_sub);
1275  }
1276 
1277  px4_unregister_shutdown_hook(&Logger::request_stop_static);
1278 }
