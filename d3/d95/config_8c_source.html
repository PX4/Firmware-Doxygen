71 int
72 config_main(int argc, char *argv[])
73 {
74  bool is_device_cmd = argc >= 2 && (!strcmp(argv[1], "block") || !strcmp(argv[1], "unblock"));
75 
76  if (argc >= 3) {
77  if (!is_device_cmd && !strncmp(argv[2], "/dev/gyro", 9)) {
78  return do_gyro(argc - 1, argv + 1);
79 
80  } else if (!is_device_cmd && !strncmp(argv[2], "/dev/accel", 10)) {
81  return do_accel(argc - 1, argv + 1);
82 
83  } else if (!is_device_cmd && !strncmp(argv[2], "/dev/mag", 8)) {
84  return do_mag(argc - 1, argv + 1);
85 
86  } else {
87  return do_device(argc - 1, argv + 1);
88  }
89  }
90 
91  print_usage();
92  return 1;
93 }

117 static int
118 do_device(int argc, char *argv[])
119 {
120  if (argc < 2) {
121  print_usage();
122  return 1;
123  }
124 
125  int fd;
126 
127  fd = open(argv[1], 0);
128 
129  if (fd < 0) {
130  PX4_ERR("open %s failed (%i)", argv[1], errno);
131  return 1;
132 
133  } else {
134 
135  int ret;
136 
137  if (argc == 2 && !strcmp(argv[0], "block")) {
138 
139  /* disable the device publications */
140  ret = ioctl(fd, DEVIOCSPUBBLOCK, 1);
141 
142  if (ret) {
143  PX4_ERR("uORB publications could not be blocked");
144  return 1;
145  }
146 
147  } else if (argc == 2 && !strcmp(argv[0], "unblock")) {
148 
149  /* enable the device publications */
150  ret = ioctl(fd, DEVIOCSPUBBLOCK, 0);
151 
152  if (ret) {
153  PX4_ERR("uORB publications could not be unblocked");
154  return 1;
155  }
156 
157  } else {
158  print_usage();
159  return 1;
160  }
161  }
162 
163  return 0;
164 }

166 static int
167 do_gyro(int argc, char *argv[])
168 {
169  int fd;
170 
171  fd = open(argv[1], 0);
172 
173  if (fd < 0) {
174  PX4_ERR("open %s failed (%i)", argv[1], errno);
175  return 1;
176 
177  } else {
178 
179  int ret;
180 
181  if (argc == 3 && !strcmp(argv[0], "sampling")) {
182 
183  /* set the gyro internal sampling rate up to at least i Hz */
184  ret = ioctl(fd, GYROIOCSSAMPLERATE, strtoul(argv[2], NULL, 0));
185 
186  if (ret) {
187  PX4_ERR("sampling rate could not be set");
188  return 1;
189  }
190 
191  } else if (argc == 3 && !strcmp(argv[0], "rate")) {
192 
193  /* set the driver to poll at i Hz */
194  ret = ioctl(fd, SENSORIOCSPOLLRATE, strtoul(argv[2], NULL, 0));
195 
196  if (ret) {
197  PX4_ERR("pollrate could not be set");
198  return 1;
199  }
200 
201  } else if (argc == 3 && !strcmp(argv[0], "range")) {
202 
203  /* set the range to i dps */
204  ret = ioctl(fd, GYROIOCSRANGE, strtoul(argv[2], NULL, 0));
205 
206  if (ret) {
207  PX4_ERR("range could not be set");
208  return 1;
209  }
210 
211  } else {
212  print_usage();
213  return 1;
214  }
215 
216  int srate = ioctl(fd, GYROIOCGSAMPLERATE, 0);
217  int prate = ioctl(fd, SENSORIOCGPOLLRATE, 0);
218  int range = ioctl(fd, GYROIOCGRANGE, 0);
219  int id = ioctl(fd, DEVIOCGDEVICEID, 0);
220  int32_t calibration_id = 0;
221 
222  param_get(param_find("CAL_GYRO0_ID"), &(calibration_id));
223 
224  PX4_INFO("gyro: \n\tdevice id:\t0x%X\t(calibration is for device id 0x%X)\n\tsample rate:\t%d Hz\n\tread rate:\t%d Hz\n\trange:\t%d dps",
225  id, calibration_id, srate, prate, range);
226 
227  close(fd);
228  }
229 
230  return 0;
231 }
