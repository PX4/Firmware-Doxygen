131 template <typename RType> class Report
132 {
133 public:
134  Report(int readers) :
135  _readidx(0),
136  _max_readers(readers),
137  _report_len(sizeof(RType))
138  {
139  memset(_buf, 0, sizeof(_buf));
140  px4_sem_init(&_lock, 0, _max_readers);
141  }
142 
143  ~Report() {}
144 
145  bool copyData(void *outbuf, int len)
146  {
147  if (len != _report_len) {
148  return false;
149  }
150 
151  read_lock();
152  memcpy(outbuf, &_buf[_readidx], _report_len);
153  read_unlock();
154  return true;
155  }
156  void writeData(void *inbuf)
157  {
158  write_lock();
159  memcpy(&_buf[!_readidx], inbuf, _report_len);
160  _readidx = !_readidx;
161  write_unlock();
162  }
163 
164 protected:
165  void read_lock() { px4_sem_wait(&_lock); }
166  void read_unlock() { px4_sem_post(&_lock); }
167  void write_lock()
168  {
169  for (int i = 0; i < _max_readers; i++) {
170  px4_sem_wait(&_lock);
171  }
172  }
173  void write_unlock()
174  {
175  for (int i = 0; i < _max_readers; i++) {
176  px4_sem_post(&_lock);
177  }
178  }
179 
180  int _readidx;
181  px4_sem_t _lock;
182  const int _max_readers;
183  const int _report_len;
184  RType _buf[2];
185 };
