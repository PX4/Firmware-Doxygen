538 void VotedSensorsUpdate::accel_poll(struct sensor_combined_s &raw)
539 {
540  float *offsets[] = {_corrections.accel_offset_0, _corrections.accel_offset_1, _corrections.accel_offset_2 };
541  float *scales[] = {_corrections.accel_scale_0, _corrections.accel_scale_1, _corrections.accel_scale_2 };
542 
543  for (int uorb_index = 0; uorb_index < _accel.subscription_count; uorb_index++) {
544  bool accel_updated;
545  orb_check(_accel.subscription[uorb_index], &accel_updated);
546 
547  if (accel_updated) {
548  struct accel_report accel_report;
549 
550  int ret = orb_copy(ORB_ID(sensor_accel), _accel.subscription[uorb_index], &accel_report);
551 
552  if (ret != PX4_OK || accel_report.timestamp == 0) {
553  continue; //ignore invalid data
554  }
555 
556  if (!_accel.enabled[uorb_index]) {
557  continue;
558  }
559 
560  // First publication with data
561  if (_accel.priority[uorb_index] == 0) {
562  int32_t priority = 0;
563  orb_priority(_accel.subscription[uorb_index], &priority);
564  _accel.priority[uorb_index] = (uint8_t)priority;
565  }
566 
567  _accel_device_id[uorb_index] = accel_report.device_id;
568 
569  matrix::Vector3f accel_data;
570 
571  if (accel_report.integral_dt != 0) {
572  /*
573  * Using data that has been integrated in the driver before downsampling is preferred
574  * becasue it reduces aliasing errors. Correct the raw sensor data for scale factor errors
575  * and offsets due to temperature variation. It is assumed that any filtering of input
576  * data required is performed in the sensor driver, preferably before downsampling.
577  */
578 
579  // convert the delta velocities to an equivalent acceleration before application of corrections
580  float dt_inv = 1.e6f / accel_report.integral_dt;
581  accel_data = matrix::Vector3f(accel_report.x_integral * dt_inv,
582  accel_report.y_integral * dt_inv,
583  accel_report.z_integral * dt_inv);
584 
585  _last_sensor_data[uorb_index].accelerometer_integral_dt = accel_report.integral_dt;
586 
587  } else {
588  // using the value instead of the integral (the integral is the prefered choice)
589 
590  // Correct each sensor for temperature effects
591  // Filtering and/or downsampling of temperature should be performed in the driver layer
592  accel_data = matrix::Vector3f(accel_report.x, accel_report.y, accel_report.z);
593 
594  // handle the cse where this is our first output
595  if (_last_accel_timestamp[uorb_index] == 0) {
596  _last_accel_timestamp[uorb_index] = accel_report.timestamp - 1000;
597  }
598 
599  // approximate the delta time using the difference in accel data time stamps
600  _last_sensor_data[uorb_index].accelerometer_integral_dt =
601  (accel_report.timestamp - _last_accel_timestamp[uorb_index]);
602  }

655  struct gyro_report gyro_report;
656 
657  int ret = orb_copy(ORB_ID(sensor_gyro), _gyro.subscription[uorb_index], &gyro_report);
658

660  continue; //ignore invalid data

664  continue;
665  }

683  * data required is performed in the sensor driver, preferably before downsampling.

704  }
705

707  _last_sensor_data[uorb_index].gyro_integral_dt =
708  (gyro_report.timestamp - _last_sensor_data[uorb_index].timestamp);
709  }
710 
711  // handle temperature compensation
712  if (_temperature_compensation.apply_corrections_gyro(uorb_index, gyro_rate, gyro_report.temperature,
713  offsets[uorb_index], scales[uorb_index]) == 2) {
714  _corrections_changed = true;
715  }
716

718  gyro_rate = _board_rotation * gyro_rate;
719

721  _last_sensor_data[uorb_index].gyro_rad[1] = gyro_rate(1);

727  }
728  }

733

1122 VotedSensorsUpdate::calc_accel_inconsistency(sensor_preflight_s &preflt)
1123 {

1125  unsigned check_index = 0; // the number of sensors the primary has been checked against
1126 
1127  // Check each sensor against the primary
1128  for (int sensor_index = 0; sensor_index < _accel.subscription_count; sensor_index++) {
1129 
1130  // check that the sensor we are checking against is not the same as the primary
1131  if ((_accel.priority[sensor_index] > 0) && (sensor_index != _accel.last_best_vote)) {
1132 
1133  float accel_diff_sum_sq = 0.0f; // sum of differences squared for a single sensor comparison agains the primary
1134 
1135  // calculate accel_diff_sum_sq for the specified sensor against the primary
1136  for (unsigned axis_index = 0; axis_index < 3; axis_index++) {
1137  _accel_diff[axis_index][check_index] = 0.95f * _accel_diff[axis_index][check_index] + 0.05f *
1138  (_last_sensor_data[_accel.last_best_vote].accelerometer_m_s2[axis_index] -
1139  _last_sensor_data[sensor_index].accelerometer_m_s2[axis_index]);
1140  accel_diff_sum_sq += _accel_diff[axis_index][check_index] * _accel_diff[axis_index][check_index];
1141 
1142  }
