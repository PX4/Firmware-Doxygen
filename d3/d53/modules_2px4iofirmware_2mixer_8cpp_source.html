99 int mixer_mix_threadsafe(float *outputs, volatile uint16_t *limits)
100 {
101  /* poor mans mutex */
102  if ((r_status_flags & PX4IO_P_STATUS_FLAGS_MIXER_OK) == 0) {
103  return 0;
104  }
105 
106  in_mixer = true;
107  int mixcount = mixer_group.mix(&outputs[0], PX4IO_SERVO_COUNT);

114 void
115 mixer_tick(void)
116 {
117  /* check if the mixer got modified */
118  mixer_handle_text_create_mixer();
119 
120  /* check that we are receiving fresh data from the FMU */
121  if ((system_state.fmu_data_received_time == 0) ||
122  hrt_elapsed_time(&system_state.fmu_data_received_time) > FMU_INPUT_DROP_LIMIT_US) {
123 
124  /* too long without FMU input, time to go to failsafe */
125  if (r_status_flags & PX4IO_P_STATUS_FLAGS_FMU_OK) {
126  isr_debug(1, "AP RX timeout");
127  }
128 
129  PX4_ATOMIC_MODIFY_CLEAR(r_status_flags, (PX4IO_P_STATUS_FLAGS_FMU_OK));
130  PX4_ATOMIC_MODIFY_OR(r_status_alarms, PX4IO_P_STATUS_ALARMS_FMU_LOST);
131 
132  } else {
133  PX4_ATOMIC_MODIFY_OR(r_status_flags, PX4IO_P_STATUS_FLAGS_FMU_OK);
134 
135  /* this flag is never cleared once OK */
136  PX4_ATOMIC_MODIFY_OR(r_status_flags, PX4IO_P_STATUS_FLAGS_FMU_INITIALIZED);
137 
138  if (system_state.fmu_data_received_time > last_fmu_update) {
139  new_fmu_data = true;
140  last_fmu_update = system_state.fmu_data_received_time;
141  }
142  }
143 
144  /* default to failsafe mixing - it will be forced below if flag is set */
145  source = MIX_FAILSAFE;
146 
147  /*
148  * Decide which set of controls we're using.
149  */
150 
151  /* Do not mix if we have raw PWM and FMU is ok. */
152  if ((r_status_flags & PX4IO_P_STATUS_FLAGS_RAW_PWM) &&
153  (r_status_flags & PX4IO_P_STATUS_FLAGS_FMU_OK)) {
154 
155  if ((r_status_flags & PX4IO_P_STATUS_FLAGS_OVERRIDE) > 0) {
156  /* a channel based override has been
157  * triggered, with FMU active */
158  source = MIX_OVERRIDE_FMU_OK;
159 
160  } else {
161  /* don't actually mix anything - copy values from r_page_direct_pwm */
162  source = MIX_NONE;
163  memcpy(r_page_servos, r_page_direct_pwm, sizeof(uint16_t)*PX4IO_SERVO_COUNT);
164  }
165 
166  } else {
167 
168  if (!(r_status_flags & PX4IO_P_STATUS_FLAGS_OVERRIDE) &&
169  (r_status_flags & PX4IO_P_STATUS_FLAGS_FMU_OK) &&
170  (r_status_flags & PX4IO_P_STATUS_FLAGS_MIXER_OK)) {
171 
172  /* mix from FMU controls */
173  source = MIX_FMU;
174  }
175 
176  else if (r_status_flags & PX4IO_P_STATUS_FLAGS_OVERRIDE) {
177 
178  if (r_status_flags & PX4IO_P_STATUS_FLAGS_FMU_OK) {
179 
180  /* if allowed, mix from RC inputs directly up to available rc channels */
181  source = MIX_OVERRIDE_FMU_OK;
182 
183  } else {
184  /* if allowed, mix from RC inputs directly */
185  source = MIX_OVERRIDE;
186  }
187  }
188  }
189 
190  /*
191  * Decide whether the servos should be armed right now.
192  *
193  * We must be armed, and we must have a PWM source; either raw from
194  * FMU or from the mixer.
195  *
196  */
197  should_arm = (
198  /* IO initialised without error */ (r_status_flags & PX4IO_P_STATUS_FLAGS_INIT_OK)
199  /* and IO is armed */ && (r_status_flags & PX4IO_P_STATUS_FLAGS_SAFETY_OFF)
200  /* and FMU is armed */ && (
201  ((r_setup_arming & PX4IO_P_SETUP_ARMING_FMU_ARMED)
202  /* and there is valid input via or mixer */ && (r_status_flags & PX4IO_P_STATUS_FLAGS_MIXER_OK))
203  /* or direct PWM is set */ || (r_status_flags & PX4IO_P_STATUS_FLAGS_RAW_PWM)
204  /* or failsafe was set manually */ || ((r_setup_arming & PX4IO_P_SETUP_ARMING_FAILSAFE_CUSTOM)
205  && !(r_status_flags & PX4IO_P_STATUS_FLAGS_FMU_OK))
206  )
207  );
208 
209  should_arm_nothrottle = (
210  /* IO initialised without error */ (r_status_flags & PX4IO_P_STATUS_FLAGS_INIT_OK)
211  /* and IO is armed */ && (r_status_flags & PX4IO_P_STATUS_FLAGS_SAFETY_OFF)
212  /* and there is valid input via or mixer */ && (r_status_flags & PX4IO_P_STATUS_FLAGS_MIXER_OK));
213 
214  should_always_enable_pwm = (r_setup_arming & PX4IO_P_SETUP_ARMING_ALWAYS_PWM_ENABLE)
215  && (r_status_flags & PX4IO_P_STATUS_FLAGS_INIT_OK)
216  && (r_status_flags & PX4IO_P_STATUS_FLAGS_FMU_OK);
217 
218  /*
219  * Check if failsafe termination is set - if yes,
220  * set the force failsafe flag once entering the first
221  * failsafe condition.
222  */
223  if ( /* if we have requested flight termination style failsafe (noreturn) */
224  (r_setup_arming & PX4IO_P_SETUP_ARMING_TERMINATION_FAILSAFE) &&
225  /* and we ended up in a failsafe condition */
226  (source == MIX_FAILSAFE) &&
227  /* and we should be armed, so we intended to provide outputs */
228  should_arm &&
229  /* and FMU is initialized */
230  (r_status_flags & PX4IO_P_STATUS_FLAGS_FMU_INITIALIZED)) {
231  PX4_ATOMIC_MODIFY_OR(r_setup_arming, PX4IO_P_SETUP_ARMING_FORCE_FAILSAFE);
232  }
233 
234  /*
235  * Check if we should force failsafe - and do it if we have to
236  */
237  if (r_setup_arming & PX4IO_P_SETUP_ARMING_FORCE_FAILSAFE) {
238  source = MIX_FAILSAFE;
239  }
240 
241  /*
242  * Set failsafe status flag depending on mixing source
243  */
244  if (source == MIX_FAILSAFE) {
245  PX4_ATOMIC_MODIFY_OR(r_status_flags, PX4IO_P_STATUS_FLAGS_FAILSAFE);
246 
247  } else {
248  PX4_ATOMIC_MODIFY_CLEAR(r_status_flags, (PX4IO_P_STATUS_FLAGS_FAILSAFE));
249  }
250 
251  /*
252  * Set simple mixer trim values. If the OK flag is set the mixer is fully loaded.
253  */
254  if (update_trims && r_status_flags & PX4IO_P_STATUS_FLAGS_MIXER_OK) {
255  update_trims = false;
256  mixer_group.set_trims(r_page_servo_control_trim, PX4IO_SERVO_COUNT);
257  }
258 
259  /*
260  * Update air-mode parameter
261  */
262  mixer_group.set_airmode(REG_TO_BOOL(r_setup_airmode));
263 
264 
265  /*
266  * Run the mixers.
267  */
268  if (source == MIX_FAILSAFE) {
269 
270  /* copy failsafe values to the servo outputs */
271  for (unsigned i = 0; i < PX4IO_SERVO_COUNT; i++) {
272  r_page_servos[i] = r_page_servo_failsafe[i];
273 
274  /* safe actuators for FMU feedback */
275  r_page_actuators[i] = FLOAT_TO_REG((r_page_servos[i] - 1500) / 600.0f);
276  }
277 
278 
279  } else if (source != MIX_NONE && (r_status_flags & PX4IO_P_STATUS_FLAGS_MIXER_OK)
280  && !(r_setup_arming & PX4IO_P_SETUP_ARMING_LOCKDOWN)) {
281 
282  float outputs[PX4IO_SERVO_COUNT];
283  unsigned mixed;
284 
285  if (REG_TO_FLOAT(r_setup_slew_max) > FLT_EPSILON) {
286  /* maximum value the outputs of the multirotor mixer are allowed to change in this cycle
287  * factor 2 is needed because actuator outputs are in the range [-1,1]
288  */
289  float delta_out_max = 2.0f * 1000.0f * dt / (r_page_servo_control_max[0] - r_page_servo_control_min[0]) / REG_TO_FLOAT(
290  r_setup_slew_max);
291  mixer_group.set_max_delta_out_once(delta_out_max);
292  }
293 
294  /* update parameter for mc thrust model if it updated */
295  if (update_mc_thrust_param) {
296  mixer_group.set_thrust_factor(REG_TO_FLOAT(r_setup_thr_fac));
297  update_mc_thrust_param = false;
298  }
299 
300  /* mix */
301  mixed = mixer_mix_threadsafe(&outputs[0], &r_mixer_limits);
302 
303  /* the pwm limit call takes care of out of band errors */
304  pwm_limit_calc(should_arm, should_arm_nothrottle, mixed, r_setup_pwm_reverse, r_page_servo_disarmed,
305  r_page_servo_control_min, r_page_servo_control_max, outputs, r_page_servos, &pwm_limit);
306 
307  /* clamp unused outputs to zero */
308  for (unsigned i = mixed; i < PX4IO_SERVO_COUNT; i++) {
309  r_page_servos[i] = 0;
310  outputs[i] = 0.0f;
311  }
312 
313  /* store normalized outputs */
314  for (unsigned i = 0; i < PX4IO_SERVO_COUNT; i++) {
315  r_page_actuators[i] = FLOAT_TO_REG(outputs[i]);
316  }
317 
318 
319  if (mixed && new_fmu_data) {
320  new_fmu_data = false;
321 
322  /* Trigger all timer's channels in Oneshot mode to fire
323  * the oneshots with updated values.
324  */
325 
326  up_pwm_update();
327  }
328  }
329 
330  /* set arming */
331  bool needs_to_arm = (should_arm || should_arm_nothrottle || should_always_enable_pwm);
332 
333  /* lockdown means to send a valid pulse which disables the outputs */
334  if (r_setup_arming & PX4IO_P_SETUP_ARMING_LOCKDOWN) {
335  needs_to_arm = true;
336  }
337 
338  if (needs_to_arm && !mixer_servos_armed) {
339  /* need to arm, but not armed */
340  up_pwm_servo_arm(true);
341  mixer_servos_armed = true;
342  PX4_ATOMIC_MODIFY_OR(r_status_flags, PX4IO_P_STATUS_FLAGS_OUTPUTS_ARMED);
343  isr_debug(5, "> PWM enabled");
344 
345  } else if (!needs_to_arm && mixer_servos_armed) {
346  /* armed but need to disarm */
347  up_pwm_servo_arm(false);
348  mixer_servos_armed = false;
349  PX4_ATOMIC_MODIFY_CLEAR(r_status_flags, (PX4IO_P_STATUS_FLAGS_OUTPUTS_ARMED));
350  isr_debug(5, "> PWM disabled");
351  }
352 
353  if (mixer_servos_armed && (should_arm || should_arm_nothrottle)
354  && !(r_setup_arming & PX4IO_P_SETUP_ARMING_LOCKDOWN)) {
355  /* update the servo outputs. */
356  for (unsigned i = 0; i < PX4IO_SERVO_COUNT; i++) {
357  up_pwm_servo_set(i, r_page_servos[i]);
358  }
359 
360  /* set S.BUS1 or S.BUS2 outputs */
361 
362  if (r_setup_features & PX4IO_P_SETUP_FEATURES_SBUS2_OUT) {
363  sbus2_output(_sbus_fd, r_page_servos, PX4IO_SERVO_COUNT);
364 
365  } else if (r_setup_features & PX4IO_P_SETUP_FEATURES_SBUS1_OUT) {
366  sbus1_output(_sbus_fd, r_page_servos, PX4IO_SERVO_COUNT);
367  }
368 
369  } else if (mixer_servos_armed && (should_always_enable_pwm
370  || (r_setup_arming & PX4IO_P_SETUP_ARMING_LOCKDOWN))) {
371  /* set the disarmed servo outputs. */
372  for (unsigned i = 0; i < PX4IO_SERVO_COUNT; i++) {
373  up_pwm_servo_set(i, r_page_servo_disarmed[i]);
374  /* copy values into reporting register */
375  r_page_servos[i] = r_page_servo_disarmed[i];
376  }
377 
378  /* set S.BUS1 or S.BUS2 outputs */
379  if (r_setup_features & PX4IO_P_SETUP_FEATURES_SBUS1_OUT) {
380  sbus1_output(_sbus_fd, r_page_servo_disarmed, PX4IO_SERVO_COUNT);
381  }
382 
383  if (r_setup_features & PX4IO_P_SETUP_FEATURES_SBUS2_OUT) {
384  sbus2_output(_sbus_fd, r_page_servo_disarmed, PX4IO_SERVO_COUNT);
385  }
386  }
387 }

389 static int
390 mixer_callback(uintptr_t handle,
391  uint8_t control_group,
392  uint8_t control_index,
393  float &control)
394 {
395  control = 0.0f;
396 
397  if (control_group >= PX4IO_CONTROL_GROUPS) {
398  return -1;
399  }
400 
401  switch (source) {
402  case MIX_FMU:
403  if (control_index < PX4IO_CONTROL_CHANNELS && control_group < PX4IO_CONTROL_GROUPS) {
404  control = REG_TO_FLOAT(r_page_controls[CONTROL_PAGE_INDEX(control_group, control_index)]);
405  break;
406  }
407 
408  return -1;
409 
410  case MIX_OVERRIDE:
411  if (r_page_rc_input[PX4IO_P_RC_VALID] & (1 << CONTROL_PAGE_INDEX(control_group, control_index))) {
412  control = REG_TO_FLOAT(r_page_rc_input[PX4IO_P_RC_BASE + control_index]);
413  break;
414  }
415 
416  return -1;
417 
418  case MIX_OVERRIDE_FMU_OK:
419 
420  /* FMU is ok but we are in override mode, use direct rc control for the available rc channels. The remaining channels are still controlled by the fmu */
421  if (r_page_rc_input[PX4IO_P_RC_VALID] & (1 << CONTROL_PAGE_INDEX(control_group, control_index))) {
422  control = REG_TO_FLOAT(r_page_rc_input[PX4IO_P_RC_BASE + control_index]);
423  break;
424 
425  } else if (control_index < PX4IO_CONTROL_CHANNELS && control_group < PX4IO_CONTROL_GROUPS) {
426  control = REG_TO_FLOAT(r_page_controls[CONTROL_PAGE_INDEX(control_group, control_index)]);
427  break;
428  }
429 
430  return -1;
431 
432  case MIX_FAILSAFE:
433  case MIX_NONE:
434  control = 0.0f;
435  return -1;
436  }
437 
438  /* apply trim offsets for override channels */
439  if (source == MIX_OVERRIDE || source == MIX_OVERRIDE_FMU_OK) {
440  if (control_group == actuator_controls_s::GROUP_INDEX_ATTITUDE &&
441  control_index == actuator_controls_s::INDEX_ROLL) {
442  control *= REG_TO_FLOAT(r_setup_scale_roll);
443  control += REG_TO_FLOAT(r_setup_trim_roll);
444 
445  } else if (control_group == actuator_controls_s::GROUP_INDEX_ATTITUDE &&
446  control_index == actuator_controls_s::INDEX_PITCH) {
447  control *= REG_TO_FLOAT(r_setup_scale_pitch);
448  control += REG_TO_FLOAT(r_setup_trim_pitch);
449 
450  } else if (control_group == actuator_controls_s::GROUP_INDEX_ATTITUDE &&
451  control_index == actuator_controls_s::INDEX_YAW) {
452  control *= REG_TO_FLOAT(r_setup_scale_yaw);
453  control += REG_TO_FLOAT(r_setup_trim_yaw);
454  }
455  }
456 
457  /* limit output */
458  if (control > 1.0f) {
459  control = 1.0f;
460 
461  } else if (control < -1.0f) {
462  control = -1.0f;
463  }
464 
465  /* motor spinup phase - lock throttle to zero */
466  if ((pwm_limit.state == PWM_LIMIT_STATE_RAMP) || (should_arm_nothrottle && !should_arm)) {
467  if ((control_group == actuator_controls_s::GROUP_INDEX_ATTITUDE ||
468  control_group == actuator_controls_s::GROUP_INDEX_ATTITUDE_ALTERNATE) &&
469  control_index == actuator_controls_s::INDEX_THROTTLE) {
470  /* limit the throttle output to zero during motor spinup,
471  * as the motors cannot follow any demand yet
472  */
473  control = 0.0f;
474  }
475  }
476 
477  /* only safety off, but not armed - set throttle as invalid */
478  if (should_arm_nothrottle && !should_arm) {
479  if ((control_group == actuator_controls_s::GROUP_INDEX_ATTITUDE ||
480  control_group == actuator_controls_s::GROUP_INDEX_ATTITUDE_ALTERNATE) &&
481  control_index == actuator_controls_s::INDEX_THROTTLE) {
482  /* mark the throttle as invalid */
483  control = NAN;
484  }
485  }
486 
487  return 0;
488 }

500 int
501 mixer_handle_text_create_mixer()
502 {
503  /* only run on update */
504  if (!mixer_update_pending) {
505  return 0;
506  }
507 
508  /* do not allow a mixer change while safety off and FMU armed */
509  if ((r_status_flags & PX4IO_P_STATUS_FLAGS_SAFETY_OFF) &&
510  (r_setup_arming & PX4IO_P_SETUP_ARMING_FMU_ARMED)) {
511  return 1;
512  }
513 
514  /* abort if we're in the mixer - it will be tried again in the next iteration */
515  if (in_mixer) {
516  return 1;
517  }
518 
519  /* process the text buffer, adding new mixers as their descriptions can be parsed */
520  unsigned resid = mixer_text_length;
521  mixer_group.load_from_buf(&mixer_text[0], resid);
522 
523  /* if anything was parsed */
524  if (resid != mixer_text_length) {
525 
526  isr_debug(2, "used %u", mixer_text_length - resid);
527 
528  /* copy any leftover text to the base of the buffer for re-use */
529  if (resid > 0) {
530  memmove(&mixer_text[0], &mixer_text[mixer_text_length - resid], resid);
531  /* enforce null termination */
532  mixer_text[resid] = '\0';
533  }
534 
535  mixer_text_length = resid;
536  }
537 
538  mixer_update_pending = false;
539 
540  return 0;
541 }

543 int
544 mixer_handle_text(const void *buffer, size_t length)
545 {
546  /* do not allow a mixer change while safety off and FMU armed */
547  if ((r_status_flags & PX4IO_P_STATUS_FLAGS_SAFETY_OFF) &&
548  (r_setup_arming & PX4IO_P_SETUP_ARMING_FMU_ARMED)) {
549  return 1;
550  }
551 
552  /* disable mixing, will be enabled once load is complete */
553  PX4_ATOMIC_MODIFY_CLEAR(r_status_flags, PX4IO_P_STATUS_FLAGS_MIXER_OK);
554 
555  /* set the update flags to dirty so we reload those values after a mixer change */
556  update_trims = true;
557  update_mc_thrust_param = true;
558 
559  /* abort if we're in the mixer - the caller is expected to retry */
560  if (in_mixer) {
561  return 1;
562  }
563 
564  px4io_mixdata *msg = (px4io_mixdata *)buffer;
565 
566  isr_debug(2, "mix txt %u", length);
567 
568  if (length < sizeof(px4io_mixdata)) {
569  return 0;
570  }
571 
572  unsigned text_length = length - sizeof(px4io_mixdata);
573 
574  switch (msg->action) {
575  case F2I_MIXER_ACTION_RESET:
576  isr_debug(2, "reset");
577 
578  /* THEN actually delete it */
579  mixer_group.reset();
580  mixer_text_length = 0;
581 
582  /* FALLTHROUGH */
583  case F2I_MIXER_ACTION_APPEND:
584  isr_debug(2, "append %d", length);
585 
586  /* check for overflow - this would be really fatal */
587  if ((mixer_text_length + text_length + 1) > sizeof(mixer_text)) {
588  PX4_ATOMIC_MODIFY_CLEAR(r_status_flags, PX4IO_P_STATUS_FLAGS_MIXER_OK);
589  return 0;
590  }
591 
592  /* check if the last item has been processed - bail out if not */
593  if (mixer_update_pending) {
594  return 1;
595  }
596 
597  /* append mixer text and nul-terminate, guard against overflow */
598  memcpy(&mixer_text[mixer_text_length], msg->text, text_length);
599  mixer_text_length += text_length;
600  mixer_text[mixer_text_length] = '\0';
601  isr_debug(2, "buflen %u", mixer_text_length);
602 
603  /* flag the buffer as ready */
604  mixer_update_pending = true;
605 
606  break;
607  }
608 
609  return 0;

613 mixer_set_failsafe()
614 {
615  /*
616  * Check if a custom failsafe value has been written,
617  * or if the mixer is not ok and bail out.
618  */
619 
620  if ((r_setup_arming & PX4IO_P_SETUP_ARMING_FAILSAFE_CUSTOM) ||
621  !(r_status_flags & PX4IO_P_STATUS_FLAGS_MIXER_OK)) {
622  return;
623  }
624 
625  /* set failsafe defaults to the values for all inputs = 0 */
626  float outputs[PX4IO_SERVO_COUNT];
627  unsigned mixed;
628 
629  if (REG_TO_FLOAT(r_setup_slew_max) > FLT_EPSILON) {
630  /* maximum value the outputs of the multirotor mixer are allowed to change in this cycle
631  * factor 2 is needed because actuator outputs are in the range [-1,1]
632  */
633  float delta_out_max = 2.0f * 1000.0f * dt / (r_page_servo_control_max[0] - r_page_servo_control_min[0]) / REG_TO_FLOAT(
634  r_setup_slew_max);
635  mixer_group.set_max_delta_out_once(delta_out_max);
636  }
637 
638  /* update parameter for mc thrust model if it updated */
639  if (update_mc_thrust_param) {
640  mixer_group.set_thrust_factor(REG_TO_FLOAT(r_setup_thr_fac));
641  update_mc_thrust_param = false;
642  }
643 
644  /* mix */
645  mixed = mixer_mix_threadsafe(&outputs[0], &r_mixer_limits);
646 
647  /* scale to PWM and update the servo outputs as required */
648  for (unsigned i = 0; i < mixed; i++) {
649 
650  /* scale to servo output */
651  r_page_servo_failsafe[i] = (outputs[i] * 600.0f) + 1500;
652 
653  }
654 
655  /* disable the rest of the outputs */
656  for (unsigned i = mixed; i < PX4IO_SERVO_COUNT; i++) {
657  r_page_servo_failsafe[i] = 0;
658  }
659 
660 }
