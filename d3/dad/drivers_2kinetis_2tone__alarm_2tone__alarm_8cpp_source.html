34 /**
35  * Driver for the PX4 audio alarm port, /dev/tone_alarm.
36  *
37  * The tone_alarm driver supports a set of predefined "alarm"
38  * tunes and one user-supplied tune.
39  *
40  * The TONE_SET_ALARM ioctl can be used to select a predefined
41  * alarm tune, from 1 - <TBD>. Selecting tune zero silences
42  * the alarm.
43  *
44  * Tunes follow the syntax of the Microsoft GWBasic/QBasic PLAY
45  * statement, with some exceptions and extensions.
46  *
47  * From Wikibooks:
48  *
49  * PLAY "[string expression]"
50  *
51  * Used to play notes and a score ... The tones are indicated by letters A through G.
52  * Accidentals are indicated with a "+" or "#" (for sharp) or "-" (for flat)
53  * immediately after the note letter. See this example:
54  *
55  * PLAY "C C# C C#"
56  *
57  * Whitespaces are ignored inside the string expression. There are also codes that
58  * set the duration, octave and tempo. They are all case-insensitive. PLAY executes
59  * the commands or notes the order in which they appear in the string. Any indicators
60  * that change the properties are effective for the notes following that indicator.
61  *
62  * Ln Sets the duration (length) of the notes. The variable n does not indicate an actual duration
63  * amount but rather a note type; L1 - whole note, L2 - half note, L4 - quarter note, etc.
64  * (L8, L16, L32, L64, ...). By default, n = 4.
65  * For triplets and quintets, use L3, L6, L12, ... and L5, L10, L20, ... series respectively.
66  * The shorthand notation of length is also provided for a note. For example, "L4 CDE L8 FG L4 AB"
67  * can be shortened to "L4 CDE F8G8 AB". F and G play as eighth notes while others play as quarter notes.
68  * On Sets the current octave. Valid values for n are 0 through 6. An octave begins with C and ends with B.
69  * Remember that C- is equivalent to B.
70  * < > Changes the current octave respectively down or up one level.
71  * Nn Plays a specified note in the seven-octave range. Valid values are from 0 to 84. (0 is a pause.)
72  * Cannot use with sharp and flat. Cannot use with the shorthand notation neither.
73  * MN Stand for Music Normal. Note duration is 7/8ths of the length indicated by Ln. It is the default mode.
74  * ML Stand for Music Legato. Note duration is full length of that indicated by Ln.
75  * MS Stand for Music Staccato. Note duration is 3/4ths of the length indicated by Ln.
76  * Pn Causes a silence (pause) for the length of note indicated (same as Ln).
77  * Tn Sets the number of "L4"s per minute (tempo). Valid values are from 32 to 255. The default value is T120.
78  * . When placed after a note, it causes the duration of the note to be 3/2 of the set duration.
79  * This is how to get "dotted" notes. "L4 C#." would play C sharp as a dotted quarter note.
80  * It can be used for a pause as well.
81  *
82  * Extensions/variations:
83  *
84  * MB MF The MF command causes the tune to play once and then stop. The MB command causes the
85  * tune to repeat when it ends.
86  *
87  */

364  /* Select a the clock source to the TPM */
365 
366  uint32_t regval = _REG(KINETIS_SIM_SOPT2);
367  regval &= ~(SIM_SOPT2_TPMSRC_MASK);
368  regval |= BOARD_TPM_CLKSRC;
369  _REG(KINETIS_SIM_SOPT2) = regval;

373

376  _REG(KINETIS_SIM_SCGC2) = regval;
377 
378  /* disable and configure the timer */
379 
380  rSC = TPM_SC_TOF;
381 
382  rCNT = 0;
383  rMOD = TONE_ALARM_COUNTER_PERIOD - 1;
384 
385  rSTATUS = (TPM_STATUS_TOF | STATUS);
386 
387  /* Configure for output compare to toggle on over flow */
388 
389  rCNSC = (TPM_CnSC_CHF | TPM_CnSC_MSA | TPM_CnSC_ELSA);
390 
391  rCOMBINE = 0;
392  rPOL = 0;
393  rFILTER = 0;
394  rQDCTRL = 0;
395  rCONF = TPM_CONF_DBGMODE_CONT;
396 
397  /* toggle the CC output each time the count passes 0 */
398 
399  rCNV = 0;
400 
401  /* enable the timer */
402 
403  rSC |= (TPM_SC_CMOD_LPTPM_CLK | TONE_ALARM_TIMER_PRESCALE);
404 
405 
406  /* default the timer to a modulo value of 1; playing notes will change this */

408 
409  DEVICE_DEBUG("ready");

413 unsigned
414 ToneAlarm::note_to_divisor(unsigned note)
415 {
416  // compute the frequency first (Hz)
417  float freq = 880.0f * expf(logf(2.0f) * ((int)note - 46) / 12.0f);
418 
419 
420  /* Since we use a toggle, there is an inherent divide by 2
421  * So we need 2 X frequency or
422  * div = fTimer / (2 * f)
423  * div = (fTimer / 2) / f
424  */
425 
426  return (TONE_ALARM_TIMER_FREQ >> 1) / freq;
427 }

429 unsigned
430 ToneAlarm::note_duration(unsigned &silence, unsigned note_length, unsigned dots)
431 {
432  unsigned whole_note_period = (60 * 1000000 * 4) / _tempo;
433 
434  if (note_length == 0) {
435  note_length = 1;
436  }
437 
438  unsigned note_period = whole_note_period / note_length;
439 
440  switch (_note_mode) {
441  case MODE_NORMAL:
442  silence = note_period / 8;
443  break;
444 
445  case MODE_STACCATO:
446  silence = note_period / 4;
447  break;
448 
449  default:
450  case MODE_LEGATO:
451  silence = 0;
452  break;
453  }
454 
455  note_period -= silence;
456 
457  unsigned dot_extension = note_period / 2;
458 
459  while (dots--) {
460  note_period += dot_extension;
461  dot_extension /= 2;
462  }
463 
464  return note_period;
465 }

489 ToneAlarm::start_note(unsigned note)
490 {
491  // check if circuit breaker is enabled
492  if (_cbrk == CBRK_UNINIT) {
493  _cbrk = circuit_breaker_enabled("CBRK_BUZZER", CBRK_BUZZER_KEY);

501  rCNT = 0;

503 
504  rSC |= (TPM_SC_CMOD_LPTPM_CLK);
505 
506  // configure the GPIO to enable timer output
507  px4_arch_configgpio(GPIO_TONE_ALARM);

544 void
545 ToneAlarm::next_note()
546 {
547  // do we have an inter-note gap to wait for?
548  if (_silence_length > 0) {
549  stop_note();
550  hrt_call_after(&_note_call, (hrt_abstime)_silence_length, (hrt_callout)next_trampoline, this);
551  _silence_length = 0;
552  return;
553  }
554 
555  // make sure we still have a tune - may be removed by the write / ioctl handler
556  if ((_next == nullptr) || (_tune == nullptr)) {
557  stop_note();
558  return;
559  }
560 
561  // parse characters out of the string until we have resolved a note
562  unsigned note = 0;
563  unsigned note_length = _note_length;
564  unsigned duration;
565 
566  while (note == 0) {
567  // we always need at least one character from the string
568  int c = next_char();
569 
570  if (c == 0) {
571  goto tune_end;
572  }
573 
574  _next++;
575 
576  switch (c) {
577  case 'L': // select note length
578  _note_length = next_number();
579 
580  if (_note_length < 1) {
581  goto tune_error;
582  }
583 
584  break;
585 
586  case 'O': // select octave
587  _octave = next_number();
588 
589  if (_octave > 6) {
590  _octave = 6;
591  }
592 
593  break;
594 
595  case '<': // decrease octave
596  if (_octave > 0) {
597  _octave--;
598  }
599 
600  break;
601 
602  case '>': // increase octave
603  if (_octave < 6) {
604  _octave++;
605  }
606 
607  break;
608 
609  case 'M': // select inter-note gap
610  c = next_char();
611 
612  if (c == 0) {
613  goto tune_error;
614  }
615 
616  _next++;
617 
618  switch (c) {
619  case 'N':
620  _note_mode = MODE_NORMAL;
621  break;
622 
623  case 'L':
624  _note_mode = MODE_LEGATO;
625  break;
626 
627  case 'S':
628  _note_mode = MODE_STACCATO;
629  break;
630 
631  case 'F':
632  _repeat = false;
633  break;
634 
635  case 'B':
636  _repeat = true;
637  break;
638 
639  default:
640  goto tune_error;
641  }
642 
643  break;
644 
645  case 'P': // pause for a note length
646  stop_note();
647  hrt_call_after(&_note_call,
648  (hrt_abstime)rest_duration(next_number(), next_dots()),
649  (hrt_callout)next_trampoline,
650  this);
651  return;
652 
653  case 'T': { // change tempo
654  unsigned nt = next_number();
655 
656  if ((nt >= 32) && (nt <= 255)) {
657  _tempo = nt;
658 
659  } else {
660  goto tune_error;
661  }
662 
663  break;
664  }
665 
666  case 'N': // play an arbitrary note
667  note = next_number();
668 
669  if (note > 84) {
670  goto tune_error;
671  }
672 
673  if (note == 0) {
674  // this is a rest - pause for the current note length
675  hrt_call_after(&_note_call,
676  (hrt_abstime)rest_duration(_note_length, next_dots()),
677  (hrt_callout)next_trampoline,
678  this);
679  return;
680  }
681 
682  break;
683 
684  case 'A'...'G': // play a note in the current octave
685  note = _note_tab[c - 'A'] + (_octave * 12) + 1;
686  c = next_char();
687 
688  switch (c) {
689  case '#': // up a semitone
690  case '+':
691  if (note < 84) {
692  note++;
693  }
694 
695  _next++;
696  break;
697 
698  case '-': // down a semitone
699  if (note > 1) {
700  note--;
701  }
702 
703  _next++;
704  break;
705 
706  default:
707  // 0 / no next char here is OK
708  break;
709  }
710 
711  // shorthand length notation
712  note_length = next_number();
713 
714  if (note_length == 0) {
715  note_length = _note_length;
716  }
717 
718  break;
719 
720  default:
721  goto tune_error;
722  }
723  }
724 
725  // compute the duration of the note and the following silence (if any)
726  duration = note_duration(_silence_length, note_length, next_dots());
727 
728  // start playing the note
729  start_note(note);
730 
731  // and arrange a callback when the note should stop
732  hrt_call_after(&_note_call, (hrt_abstime)duration, (hrt_callout)next_trampoline, this);
733  return;
734 
735  // tune looks bad (unexpected EOF, bad character, etc.)
736 tune_error:
737  syslog(LOG_ERR, "tune error\n");
738  _repeat = false; // don't loop on error
739 
740  // stop (and potentially restart) the tune
741 tune_end:
742  stop_note();
743 
744  if (_repeat) {
745  start_tune(_tune);
746 
747  } else {
748  _tune = nullptr;
749  _default_tune_number = 0;
750  }
751 
752  return;
753 }
