495 L3GD20::probe()
496 {
497  /* read dummy value to void to clear SPI statemachine on sensor */
498  (void)read_reg(ADDR_WHO_AM_I);
499 
500  bool success = false;
501  uint8_t v = 0;
502 
503  /* verify that the device is attached and functioning, accept
504  * L3GD20, L3GD20H and L3G4200D */
505  if ((v = read_reg(ADDR_WHO_AM_I)) == WHO_I_AM) {
506  _orientation = SENSOR_BOARD_ROTATION_DEFAULT;
507  success = true;
508 
509  } else if ((v = read_reg(ADDR_WHO_AM_I)) == WHO_I_AM_H) {
510  _orientation = SENSOR_BOARD_ROTATION_180_DEG;
511  success = true;
512 
513  } else if ((v = read_reg(ADDR_WHO_AM_I)) == WHO_I_AM_L3G4200D) {
514  /* Detect the L3G4200D used on AeroCore */
515  _is_l3g4200d = true;
516  _orientation = SENSOR_BOARD_ROTATION_DEFAULT;
517  success = true;
518  }
519 
520  if (success) {
521  _checked_values[0] = v;
522  return OK;
523  }
524 
525  return -EIO;

528 ssize_t
529 L3GD20::read(struct file *filp, char *buffer, size_t buflen)
530 {
531  unsigned count = buflen / sizeof(struct gyro_report);
532  struct gyro_report *gbuf = reinterpret_cast<struct gyro_report *>(buffer);
533  int ret = 0;
534 
535  /* buffer must be large enough */
536  if (count < 1) {
537  return -ENOSPC;
538  }
539 
540  /* if automatic measurement is enabled */
541  if (_call_interval > 0) {
542 
543  /*
544  * While there is space in the caller's buffer, and reports, copy them.
545  * Note that we may be pre-empted by the measurement code while we are doing this;
546  * we are careful to avoid racing with it.
547  */
548  while (count--) {
549  if (_reports->get(gbuf)) {
550  ret += sizeof(*gbuf);
551  gbuf++;
552  }
553  }
554 
555  /* if there was no data, warn the caller */
556  return ret ? ret : -EAGAIN;
557  }
558 
559  /* manual measurement */
560  _reports->flush();
561  measure();
562 
563  /* measurement will have generated a report, copy it out */
564  if (_reports->get(gbuf)) {
565  ret = sizeof(*gbuf);
566  }
567 
568  return ret;
569 }

571 int
572 L3GD20::ioctl(struct file *filp, int cmd, unsigned long arg)
573 {
574  switch (cmd) {
575 
576  case SENSORIOCSPOLLRATE: {
577  switch (arg) {
578 
579  /* switching to manual polling */
580  case SENSOR_POLLRATE_MANUAL:
581  stop();
582  _call_interval = 0;
583  return OK;
584 
585  /* external signalling not supported */
586  case SENSOR_POLLRATE_EXTERNAL:
587 
588  /* zero would be bad */
589  case 0:
590  return -EINVAL;
591 
592  /* set default/max polling rate */
593  case SENSOR_POLLRATE_MAX:
594  case SENSOR_POLLRATE_DEFAULT:
595  if (_is_l3g4200d) {
596  return ioctl(filp, SENSORIOCSPOLLRATE, L3G4200D_DEFAULT_RATE);
597  }
598 
599  return ioctl(filp, SENSORIOCSPOLLRATE, L3GD20_DEFAULT_RATE);
600 
601  /* adjust to a legal polling interval in Hz */
602  default: {
603  /* do we need to start internal polling? */
604  bool want_start = (_call_interval == 0);
605 
606  /* convert hz to hrt interval via microseconds */
607  unsigned ticks = 1000000 / arg;
608 
609  /* check against maximum sane rate */
610  if (ticks < 1000) {
611  return -EINVAL;
612  }
613 
614  /* update interval for next measurement */
615  /* XXX this is a bit shady, but no other way to adjust... */
616  _call_interval = ticks;
617 
618  _call.period = _call_interval - L3GD20_TIMER_REDUCTION;
619 
620  /* adjust filters */
621  float cutoff_freq_hz = _gyro_filter_x.get_cutoff_freq();
622  float sample_rate = 1.0e6f / ticks;
623  set_driver_lowpass_filter(sample_rate, cutoff_freq_hz);
624 
625  /* if we need to start the poll state machine, do it */
626  if (want_start) {
627  start();
628  }
629 
630  return OK;
631  }
632  }
633  }
634 
635  case SENSORIOCGPOLLRATE:
636  if (_call_interval == 0) {
637  return SENSOR_POLLRATE_MANUAL;
638  }
639 
640  return 1000000 / _call_interval;
641 
642  case SENSORIOCSQUEUEDEPTH: {
643  /* lower bound is mandatory, upper bound is a sanity check */
644  if ((arg < 1) || (arg > 100)) {
645  return -EINVAL;
646  }
647 
648  irqstate_t flags = px4_enter_critical_section();
649 
650  if (!_reports->resize(arg)) {
651  px4_leave_critical_section(flags);
652  return -ENOMEM;
653  }
654 
655  px4_leave_critical_section(flags);
656 
657  return OK;
658  }
659 
660  case SENSORIOCRESET:
661  reset();
662  return OK;
663 
664  case GYROIOCSSAMPLERATE:
665  return set_samplerate(arg);
666 
667  case GYROIOCGSAMPLERATE:
668  return _current_rate;

906 void
907 L3GD20::check_registers(void)
908 {
909  uint8_t v;
910 
911  if ((v = read_reg(_checked_registers[_checked_next])) != _checked_values[_checked_next]) {
912  /*
913  if we get the wrong value then we know the SPI bus
914  or sensor is very sick. We set _register_wait to 20
915  and wait until we have seen 20 good values in a row
916  before we consider the sensor to be OK again.
917  */
918  perf_count(_bad_registers);
919 
920  /*
921  try to fix the bad register value. We only try to
922  fix one per loop to prevent a bad sensor hogging the
923  bus. We skip zero as that is the WHO_AM_I, which
924  is not writeable
925  */
926  if (_checked_next != 0) {
927  write_reg(_checked_registers[_checked_next], _checked_values[_checked_next]);
928  }
929 
930  _register_wait = 20;
931  }
932 
933  _checked_next = (_checked_next + 1) % L3GD20_NUM_CHECKED_REGISTERS;
934 }

936 void
937 L3GD20::measure()
938 {
939  /* status register and data as read back from the device */
940 #pragma pack(push, 1)
941  struct {
942  uint8_t cmd;
943  int8_t temp;
944  uint8_t status;
945  int16_t x;
946  int16_t y;
947  int16_t z;
948  } raw_report;
949 #pragma pack(pop)
950 
951  gyro_report report;
952 
953  /* start the performance counter */
954  perf_begin(_sample_perf);
955 
956  check_registers();
957 
958  /* fetch data from the sensor */
959  memset(&raw_report, 0, sizeof(raw_report));
960  raw_report.cmd = ADDR_OUT_TEMP | DIR_READ | ADDR_INCREMENT;
961  transfer((uint8_t *)&raw_report, (uint8_t *)&raw_report, sizeof(raw_report));
962 
963  if (!(raw_report.status & STATUS_ZYXDA)) {
964  perf_end(_sample_perf);
965  perf_count(_duplicates);
966  return;
967  }
968 
969  /*
970  * 1) Scale raw value to SI units using scaling from datasheet.
971  * 2) Subtract static offset (in SI units)
972  * 3) Scale the statically calibrated values with a linear
973  * dynamically obtained factor
974  *
975  * Note: the static sensor offset is the number the sensor outputs
976  * at a nominally 'zero' input. Therefore the offset has to
977  * be subtracted.
978  *
979  * Example: A gyro outputs a value of 74 at zero angular rate
980  * the offset is 74 from the origin and subtracting
981  * 74 from all measurements centers them around zero.
982  */
983  report.timestamp = hrt_absolute_time();
984  report.error_count = perf_event_count(_bad_registers);
985 
986  switch (_orientation) {
987 
988  case SENSOR_BOARD_ROTATION_000_DEG:
989  /* keep axes in place */
990  report.x_raw = raw_report.x;
991  report.y_raw = raw_report.y;
992  break;
993 
994  case SENSOR_BOARD_ROTATION_090_DEG:
995  /* swap x and y */
996  report.x_raw = raw_report.y;
997  report.y_raw = raw_report.x;
998  break;
999 
1000  case SENSOR_BOARD_ROTATION_180_DEG:
1001  /* swap x and y and negate both */
1002  report.x_raw = ((raw_report.x == -32768) ? 32767 : -raw_report.x);
1003  report.y_raw = ((raw_report.y == -32768) ? 32767 : -raw_report.y);
1004  break;
1005 
1006  case SENSOR_BOARD_ROTATION_270_DEG:
1007  /* swap x and y and negate y */
1008  report.x_raw = raw_report.y;
1009  report.y_raw = ((raw_report.x == -32768) ? 32767 : -raw_report.x);
1010  break;
1011  }
1012 
1013  report.z_raw = raw_report.z;
1014 
1015  float xraw_f = report.x_raw;
1016  float yraw_f = report.y_raw;
1017  float zraw_f = report.z_raw;
1018 
1019  // apply user specified rotation
1020  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
1021 
1022  float xin = ((xraw_f * _gyro_range_scale) - _gyro_scale.x_offset) * _gyro_scale.x_scale;
1023  float yin = ((yraw_f * _gyro_range_scale) - _gyro_scale.y_offset) * _gyro_scale.y_scale;
1024  float zin = ((zraw_f * _gyro_range_scale) - _gyro_scale.z_offset) * _gyro_scale.z_scale;
1025 
1026  report.x = _gyro_filter_x.apply(xin);
1027  report.y = _gyro_filter_y.apply(yin);
1028  report.z = _gyro_filter_z.apply(zin);
1029 
1030  matrix::Vector3f gval(xin, yin, zin);
1031  matrix::Vector3f gval_integrated;
1032 
1033  bool gyro_notify = _gyro_int.put(report.timestamp, gval, gval_integrated, report.integral_dt);
1034  report.x_integral = gval_integrated(0);
1035  report.y_integral = gval_integrated(1);
1036  report.z_integral = gval_integrated(2);
1037 
1038  report.temperature = L3GD20_TEMP_OFFSET_CELSIUS - raw_report.temp;
1039 
1040  report.scaling = _gyro_range_scale;
1041 
1042  /* return device ID */
1043  report.device_id = _device_id.devid;
1044 
1045  _reports->force(&report);
1046 
1047  if (gyro_notify) {
1048  /* notify anyone waiting for data */
1049  poll_notify(POLLIN);
1050 
1051  /* publish for subscribers */
1052  if (!(_pub_blocked)) {
1053  /* publish it */
1054  orb_publish(ORB_ID(sensor_gyro), _gyro_topic, &report);
1055  }
1056  }
1057 
1058  _read++;
1059 
1060  /* stop the perf counter */
1061  perf_end(_sample_perf);
1062 }
