64 int syslink_parse_char(syslink_parse_state *state, char c, syslink_message_t *msg)
65 {
66 
67  switch (state->state) {
68  case SYSLINK_STATE_START:
69  if (c == syslink_magic[state->index]) {
70  state->index++;
71 
72  } else {
73  state->index = 0;
74  }
75 
76  if (syslink_magic[state->index] == '\x00') {
77  state->state = SYSLINK_STATE_TYPE;
78  }
79 
80  break;
81 
82  case SYSLINK_STATE_TYPE:
83  msg->type = c;
84  state->state = SYSLINK_STATE_LENGTH;
85  break;
86 
87  case SYSLINK_STATE_LENGTH:
88  msg->length = c;
89 
90  if (c > SYSLINK_MAX_DATA_LEN) { // Too long
91  state->state = SYSLINK_STATE_START;
92 
93  } else {
94  state->state = c > 0 ? SYSLINK_STATE_DATA : SYSLINK_STATE_CKSUM;
95  }
96 
97  state->index = 0;
98  break;
99 
100  case SYSLINK_STATE_DATA:
101  msg->data[state->index++] = c;
102 
103  if (state->index >= msg->length) {
104  state->state = SYSLINK_STATE_CKSUM;
105  state->index = 0;
106  syslink_compute_cksum(msg);
107  }
108 
109  break;
110 
111  case SYSLINK_STATE_CKSUM:
112  if (c != msg->cksum[state->index]) {
113  PX4_INFO("Bad checksum");
114  state->state = SYSLINK_STATE_START;
115  state->index = 0;
116  break;
117  }
118 
119  state->index++;
120 
121  if (state->index >= (int)sizeof(msg->cksum)) {
122  state->state = SYSLINK_STATE_START;
123  state->index = 0;
124  return 1;
125  }
126 
127 
128  break;
129  }
130 
131  return 0;

133 }

135 /*
136  Computes Fletcher 8bit checksum per RFC1146
137  A := A + D[i]
138  B := B + A
139 */
140 void syslink_compute_cksum(syslink_message_t *msg)
141 {
142  uint8_t a = 0, b = 0;
143  uint8_t *Di = (uint8_t *)msg, *end = Di + (2 + msg->length) * sizeof(uint8_t);
144 
145  while (Di < end) {
146  a = a + *Di;
147  b = b + a;
148  ++Di;
149  }
150 
151  msg->cksum[0] = a;
152  msg->cksum[1] = b;
153 }
