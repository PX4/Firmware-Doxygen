105 int motor_test_main(int argc, char *argv[])
106 {
107  int channel = -1; //default to all channels
108  unsigned long lval;
109  float value = 0.0f;
110  int ch;
111 
112  int myoptind = 1;
113  const char *myoptarg = NULL;
114 
115  while ((ch = px4_getopt(argc, argv, "m:p:", &myoptind, &myoptarg)) != EOF) {
116  switch (ch) {
117 
118  case 'm':
119  /* Read in motor number */
120  channel = (int)strtol(myoptarg, NULL, 0);
121  break;
122 
123  case 'p':
124  /* Read in power value */
125  lval = strtoul(myoptarg, NULL, 0);
126 
127  if (lval > 100) {
128  usage("value invalid");
129  return 1;
130  }
131 
132  value = ((float)lval) / 100.f;
133  break;
134 
135  default:
136  usage(NULL);
137  return 1;
138  }
139  }
140 
141  bool run_test = true;
142 
143  if (myoptind >= 0 && myoptind < argc) {
144  if (strcmp("stop", argv[myoptind]) == 0) {
145  channel = -1;
146  value = 0.f;
147 
148  } else if (strcmp("iterate", argv[myoptind]) == 0) {
149  value = 0.3f;
150 
151  for (int i = 0; i < 8; ++i) {
152  motor_test(i, value);
153  usleep(500000);
154  motor_test(i, 0.f);
155  usleep(10000);
156  }
157 
158  run_test = false;
159 
160  } else if (strcmp("test", argv[myoptind]) == 0) {
161  // nothing to do
162  } else {
163  usage(NULL);
164  return 0;
165  }
166 
167  } else {
168  usage(NULL);
169  return 0;
170  }
171 
172  if (run_test) {
173  if (channel < 0) {
174  for (int i = 0; i < 8; ++i) {
175  motor_test(i, value);
176  usleep(10000);
177  }
178 
179  } else {
180  motor_test(channel, value);
181  }
182  }
183 
184  return 0;
185 }
