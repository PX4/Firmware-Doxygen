78 void Tailsitter::update_vtol_state()
79 {
80 
81  /* simple logic using a two way switch to perform transitions.
82  * after flipping the switch the vehicle will start tilting in MC control mode, picking up
83  * forward speed. After the vehicle has picked up enough and sufficient pitch angle the uav will go into FW mode.
84  * For the backtransition the pitch is controlled in MC mode again and switches to full MC control reaching the sufficient pitch angle.
85  */
86 
87  matrix::Eulerf euler = matrix::Quatf(_v_att->q);
88  float pitch = euler.theta();
89 
90  if (!_attc->is_fixed_wing_requested()) {
91 
92  switch (_vtol_schedule.flight_mode) { // user switchig to MC mode
93  case MC_MODE:
94  break;
95 
96  case FW_MODE:
97  _vtol_schedule.flight_mode = TRANSITION_BACK;
98  _vtol_schedule.transition_start = hrt_absolute_time();
99  break;
100 
101  case TRANSITION_FRONT_P1:
102  // failsafe into multicopter mode
103  _vtol_schedule.flight_mode = MC_MODE;
104  break;
105 
106  case TRANSITION_BACK:
107 
108  // check if we have reached pitch angle to switch to MC mode
109  if (pitch >= PITCH_TRANSITION_BACK) {
110  _vtol_schedule.flight_mode = MC_MODE;
111  }
112 
113  break;
114  }
115 
116  } else { // user switchig to FW mode
117 
118  switch (_vtol_schedule.flight_mode) {
119  case MC_MODE:
120  // initialise a front transition
121  _vtol_schedule.flight_mode = TRANSITION_FRONT_P1;
122  _vtol_schedule.transition_start = hrt_absolute_time();
123  break;
124 
125  case FW_MODE:
126  break;
127 
128  case TRANSITION_FRONT_P1: {
129 
130  bool airspeed_condition_satisfied = _airspeed->indicated_airspeed_m_s >= _params->transition_airspeed;
131  airspeed_condition_satisfied |= _params->airspeed_disabled;
132 
133  // check if we have reached airspeed and pitch angle to switch to TRANSITION P2 mode
134  if ((airspeed_condition_satisfied && pitch <= PITCH_TRANSITION_FRONT_P1) || can_transition_on_ground()) {
135  _vtol_schedule.flight_mode = FW_MODE;
136  }
137 
138  break;
139  }
140 
141  case TRANSITION_BACK:
142  // failsafe into fixed wing mode
143  _vtol_schedule.flight_mode = FW_MODE;
144  break;
145  }
146  }
147 
148  // map tailsitter specific control phases to simple control modes
149  switch (_vtol_schedule.flight_mode) {
150  case MC_MODE:
151  _vtol_mode = ROTARY_WING;
152  _vtol_vehicle_status->vtol_in_trans_mode = false;
153  _flag_was_in_trans_mode = false;
154  break;
155 
156  case FW_MODE:
157  _vtol_mode = FIXED_WING;
158  _vtol_vehicle_status->vtol_in_trans_mode = false;
159  _flag_was_in_trans_mode = false;
160  break;
161 
162  case TRANSITION_FRONT_P1:
163  _vtol_mode = TRANSITION_TO_FW;
164  _vtol_vehicle_status->vtol_in_trans_mode = true;
165  break;
166 
167  case TRANSITION_BACK:
168  _vtol_mode = TRANSITION_TO_MC;
169  _vtol_vehicle_status->vtol_in_trans_mode = true;
170  break;
171  }
172 }

174 void Tailsitter::update_transition_state()
175 {
176  float time_since_trans_start = (float)(hrt_absolute_time() - _vtol_schedule.transition_start) * 1e-6f;
177 
178  if (!_flag_was_in_trans_mode) {
179  // save desired heading for transition and last thrust value
180  _yaw_transition = _v_att_sp->yaw_body;
181  //transition should start from current attitude instead of current setpoint
182  matrix::Eulerf euler = matrix::Quatf(_v_att->q);
183  _pitch_transition_start = euler.theta();
184  _thrust_transition_start = _v_att_sp->thrust;
185  _flag_was_in_trans_mode = true;
186  }
187 
188  if (_vtol_schedule.flight_mode == TRANSITION_FRONT_P1) {
189 
190  // create time dependant pitch angle set point + 0.2 rad overlap over the switch value
191  _v_att_sp->pitch_body = _pitch_transition_start - fabsf(PITCH_TRANSITION_FRONT_P1 - _pitch_transition_start) *
192  time_since_trans_start / _params->front_trans_duration;
193  _v_att_sp->pitch_body = math::constrain(_v_att_sp->pitch_body, PITCH_TRANSITION_FRONT_P1 - 0.2f,
194  _pitch_transition_start);
195 
196  // disable mc yaw control once the plane has picked up speed
197  if (_airspeed->indicated_airspeed_m_s > ARSP_YAW_CTRL_DISABLE) {
198  _mc_yaw_weight = 0.0f;
199 
200  } else {
201  _mc_yaw_weight = 1.0f;
202  }
203 
204  _mc_roll_weight = 1.0f;
205  _mc_pitch_weight = 1.0f;
206 
207  } else if (_vtol_schedule.flight_mode == TRANSITION_BACK) {
208 
209  if (!flag_idle_mc) {
210  flag_idle_mc = set_idle_mc();
211  }
212 
213  // create time dependant pitch angle set point stating at -pi/2 + 0.2 rad overlap over the switch value
214  _v_att_sp->pitch_body = M_PI_2_F + _pitch_transition_start + fabsf(PITCH_TRANSITION_BACK + 1.57f) *
215  time_since_trans_start / _params->back_trans_duration;
216  _v_att_sp->pitch_body = math::constrain(_v_att_sp->pitch_body, -2.0f, PITCH_TRANSITION_BACK + 0.2f);
217 
218  // keep yaw disabled
219  _mc_yaw_weight = 0.0f;
220 
221  // smoothly move control weight to MC
222  _mc_roll_weight = _mc_pitch_weight = time_since_trans_start / _params->back_trans_duration;
223 
224  }
225 
226  if (_v_control_mode->flag_control_climb_rate_enabled) {
227  _v_att_sp->thrust = _params->front_trans_throttle;
228 
229  } else {
230  _v_att_sp->thrust = _mc_virtual_att_sp->thrust;
231  }
232 
233  _mc_roll_weight = math::constrain(_mc_roll_weight, 0.0f, 1.0f);
234  _mc_yaw_weight = math::constrain(_mc_yaw_weight, 0.0f, 1.0f);
235  _mc_pitch_weight = math::constrain(_mc_pitch_weight, 0.0f, 1.0f);
236 
237  // compute desired attitude and thrust setpoint for the transition
238 
239  _v_att_sp->timestamp = hrt_absolute_time();
240  _v_att_sp->roll_body = 0.0f;
241  _v_att_sp->yaw_body = _yaw_transition;
242 
243  matrix::Quatf q_sp = matrix::Eulerf(_v_att_sp->roll_body, _v_att_sp->pitch_body, _v_att_sp->yaw_body);
244  q_sp.copyTo(_v_att_sp->q_d);
245  _v_att_sp->q_d_valid = true;
246 }

267 void Tailsitter::fill_actuator_outputs()
268 {
269  _actuators_out_0->timestamp = hrt_absolute_time();
270  _actuators_out_0->timestamp_sample = _actuators_mc_in->timestamp_sample;
271 
272  _actuators_out_1->timestamp = hrt_absolute_time();
273  _actuators_out_1->timestamp_sample = _actuators_fw_in->timestamp_sample;
274 
275  switch (_vtol_mode) {
276  case ROTARY_WING:
277  _actuators_out_0->control[actuator_controls_s::INDEX_ROLL] = _actuators_mc_in->control[actuator_controls_s::INDEX_ROLL];
278  _actuators_out_0->control[actuator_controls_s::INDEX_PITCH] =
279  _actuators_mc_in->control[actuator_controls_s::INDEX_PITCH];
280  _actuators_out_0->control[actuator_controls_s::INDEX_YAW] = _actuators_mc_in->control[actuator_controls_s::INDEX_YAW];
281  _actuators_out_0->control[actuator_controls_s::INDEX_THROTTLE] =
282  _actuators_mc_in->control[actuator_controls_s::INDEX_THROTTLE];
283 
284  if (_params->elevons_mc_lock) {
285  _actuators_out_1->control[0] = 0;
286  _actuators_out_1->control[1] = 0;
287 
288  } else {
289  // NOTE: There is no mistake in the line below, multicopter yaw axis is controlled by elevon roll actuation!
290  _actuators_out_1->control[actuator_controls_s::INDEX_ROLL] =
291  _actuators_mc_in->control[actuator_controls_s::INDEX_YAW]; //roll elevon
292  _actuators_out_1->control[actuator_controls_s::INDEX_PITCH] =
293  _actuators_mc_in->control[actuator_controls_s::INDEX_PITCH]; //pitch elevon
294  }
295 
296  break;
297 
298  case FIXED_WING:
299  // in fixed wing mode we use engines only for providing thrust, no moments are generated
300  _actuators_out_0->control[actuator_controls_s::INDEX_ROLL] = 0;
301  _actuators_out_0->control[actuator_controls_s::INDEX_PITCH] = 0;
302  _actuators_out_0->control[actuator_controls_s::INDEX_YAW] = 0;
303  _actuators_out_0->control[actuator_controls_s::INDEX_THROTTLE] =
304  _actuators_fw_in->control[actuator_controls_s::INDEX_THROTTLE];
305 
306  _actuators_out_1->control[actuator_controls_s::INDEX_ROLL] =
307  -_actuators_fw_in->control[actuator_controls_s::INDEX_ROLL]; // roll elevon
308  _actuators_out_1->control[actuator_controls_s::INDEX_PITCH] =
309  _actuators_fw_in->control[actuator_controls_s::INDEX_PITCH]; // pitch elevon
310  _actuators_out_1->control[actuator_controls_s::INDEX_YAW] =
311  _actuators_fw_in->control[actuator_controls_s::INDEX_YAW]; // yaw
312  _actuators_out_1->control[actuator_controls_s::INDEX_THROTTLE] =
313  _actuators_fw_in->control[actuator_controls_s::INDEX_THROTTLE]; // throttle
314  break;
315 
316  case TRANSITION_TO_FW:
317  case TRANSITION_TO_MC:
318  // in transition engines are mixed by weight (BACK TRANSITION ONLY)
319  _actuators_out_0->control[actuator_controls_s::INDEX_ROLL] = _actuators_mc_in->control[actuator_controls_s::INDEX_ROLL]
320  * _mc_roll_weight;
321  _actuators_out_0->control[actuator_controls_s::INDEX_PITCH] =
322  _actuators_mc_in->control[actuator_controls_s::INDEX_PITCH] * _mc_pitch_weight;
323  _actuators_out_0->control[actuator_controls_s::INDEX_YAW] = _actuators_mc_in->control[actuator_controls_s::INDEX_YAW] *
324  _mc_yaw_weight;
325  _actuators_out_0->control[actuator_controls_s::INDEX_THROTTLE] =
326  _actuators_mc_in->control[actuator_controls_s::INDEX_THROTTLE];
327 
328  // NOTE: There is no mistake in the line below, multicopter yaw axis is controlled by elevon roll actuation!
329  _actuators_out_1->control[actuator_controls_s::INDEX_ROLL] = -_actuators_fw_in->control[actuator_controls_s::INDEX_ROLL]
330  * (1 - _mc_yaw_weight);
331  _actuators_out_1->control[actuator_controls_s::INDEX_PITCH] =
332  _actuators_mc_in->control[actuator_controls_s::INDEX_PITCH] * _mc_pitch_weight;
333  // **LATER** + (_actuators_fw_in->control[actuator_controls_s::INDEX_PITCH] + _params->fw_pitch_trim) *(1 - _mc_pitch_weight);
334  _actuators_out_1->control[actuator_controls_s::INDEX_THROTTLE] =
335  _actuators_fw_in->control[actuator_controls_s::INDEX_THROTTLE];
336  break;
337  }
338 }
