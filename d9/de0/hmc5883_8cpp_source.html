434 int HMC5883::set_range(unsigned range)
435 {
436  if (range < 0.88f) {
437  _range_bits = 0x00;
438  _range_scale = 1.0f / 1370.0f;
439  _range_ga = 0.88f;
440 
441  } else if (range <= 1.3f) {
442  _range_bits = 0x01;
443  _range_scale = 1.0f / 1090.0f;
444  _range_ga = 1.3f;
445 
446  } else if (range <= 2) {
447  _range_bits = 0x02;
448  _range_scale = 1.0f / 820.0f;
449  _range_ga = 1.9f;
450 
451  } else if (range <= 3) {
452  _range_bits = 0x03;
453  _range_scale = 1.0f / 660.0f;
454  _range_ga = 2.5f;
455 
456  } else if (range <= 4) {
457  _range_bits = 0x04;
458  _range_scale = 1.0f / 440.0f;
459  _range_ga = 4.0f;
460 
461  } else if (range <= 4.7f) {
462  _range_bits = 0x05;
463  _range_scale = 1.0f / 390.0f;
464  _range_ga = 4.7f;
465 
466  } else if (range <= 5.6f) {
467  _range_bits = 0x06;
468  _range_scale = 1.0f / 330.0f;
469  _range_ga = 5.6f;
470 
471  } else {
472  _range_bits = 0x07;
473  _range_scale = 1.0f / 230.0f;
474  _range_ga = 8.1f;
475  }
476 
477  int ret;
478 
479  /*
480  * Send the command to set the range
481  */
482  ret = write_reg(ADDR_CONF_B, (_range_bits << 5));
483 
484  if (OK != ret) {
485  perf_count(_comms_errors);
486  }
487 
488  uint8_t range_bits_in = 0;
489  ret = read_reg(ADDR_CONF_B, range_bits_in);
490 
491  if (OK != ret) {
492  perf_count(_comms_errors);
493  }
494 
495  return !(range_bits_in == (_range_bits << 5));
496 }

552 ssize_t
553 HMC5883::read(struct file *filp, char *buffer, size_t buflen)
554 {
555  unsigned count = buflen / sizeof(struct mag_report);
556  struct mag_report *mag_buf = reinterpret_cast<struct mag_report *>(buffer);
557  int ret = 0;
558 
559  /* buffer must be large enough */
560  if (count < 1) {
561  return -ENOSPC;
562  }
563 
564  /* if automatic measurement is enabled */
565  if (_measure_ticks > 0) {
566  /*
567  * While there is space in the caller's buffer, and reports, copy them.
568  * Note that we may be pre-empted by the workq thread while we are doing this;
569  * we are careful to avoid racing with them.
570  */
571  while (count--) {
572  if (_reports->get(mag_buf)) {
573  ret += sizeof(struct mag_report);
574  mag_buf++;
575  }
576  }
577 
578  /* if there was no data, warn the caller */
579  return ret ? ret : -EAGAIN;
580  }
581 
582  /* manual measurement - run one conversion */
583  /* XXX really it'd be nice to lock against other readers here */
584  do {
585  _reports->flush();
586 
587  /* trigger a measurement */
588  if (OK != measure()) {
589  ret = -EIO;
590  break;
591  }
592 
593  /* wait for it to complete */
594  usleep(HMC5883_CONVERSION_INTERVAL);
595 
596  /* run the collection phase */
597  if (OK != collect()) {
598  ret = -EIO;
599  break;
600  }
601 
602  if (_reports->get(mag_buf)) {
603  ret = sizeof(struct mag_report);
604  }
605  } while (0);
606 
607  return ret;
608 }

610 int
611 HMC5883::ioctl(struct file *filp, int cmd, unsigned long arg)
612 {
613  unsigned dummy = arg;
614 
615  switch (cmd) {
616  case SENSORIOCSPOLLRATE: {
617  switch (arg) {
618 
619  /* switching to manual polling */
620  case SENSOR_POLLRATE_MANUAL:
621  stop();
622  _measure_ticks = 0;
623  return OK;
624 
625  /* external signalling (DRDY) not supported */
626  case SENSOR_POLLRATE_EXTERNAL:
627 
628  /* zero would be bad */
629  case 0:
630  return -EINVAL;
631 
632  /* set default/max polling rate */
633  case SENSOR_POLLRATE_MAX:
634  case SENSOR_POLLRATE_DEFAULT: {
635  /* do we need to start internal polling? */
636  bool want_start = (_measure_ticks == 0);
637 
638  /* set interval for next measurement to minimum legal value */
639  _measure_ticks = USEC2TICK(HMC5883_CONVERSION_INTERVAL);
640 
641  /* if we need to start the poll state machine, do it */
642  if (want_start) {
643  start();
644  }
645 
646  return OK;
647  }
648 
649  /* adjust to a legal polling interval in Hz */
650  default: {
651  /* do we need to start internal polling? */
652  bool want_start = (_measure_ticks == 0);
653 
654  /* convert hz to tick interval via microseconds */
655  unsigned ticks = USEC2TICK(1000000 / arg);
656 
657  /* check against maximum rate */
658  if (ticks < USEC2TICK(HMC5883_CONVERSION_INTERVAL)) {
659  return -EINVAL;
660  }
661 
662  /* update interval for next measurement */
663  _measure_ticks = ticks;
664 
665  /* if we need to start the poll state machine, do it */
666  if (want_start) {
667  start();
668  }
669 
670  return OK;
671  }
672  }
673  }
674 
675  case SENSORIOCGPOLLRATE:
676  if (_measure_ticks == 0) {
677  return SENSOR_POLLRATE_MANUAL;
678  }
679 
680  return 1000000 / TICK2USEC(_measure_ticks);
681 
682  case SENSORIOCSQUEUEDEPTH: {
683  /* lower bound is mandatory, upper bound is a sanity check */
684  if ((arg < 1) || (arg > 100)) {
685  return -EINVAL;
686  }
687 
688  irqstate_t flags = px4_enter_critical_section();
689 
690  if (!_reports->resize(arg)) {
691  px4_leave_critical_section(flags);
692  return -ENOMEM;
693  }
694 
695  px4_leave_critical_section(flags);
696 
697  return OK;
698  }
699 
700  case SENSORIOCRESET:
701  return reset();
702 
703  case MAGIOCSSAMPLERATE:
704  /* same as pollrate because device is in single measurement mode*/
705  return ioctl(filp, SENSORIOCSPOLLRATE, arg);
706 
707  case MAGIOCGSAMPLERATE:
708  /* same as pollrate because device is in single measurement mode*/
709  return 1000000 / TICK2USEC(_measure_ticks);
710 
711  case MAGIOCSRANGE:
712  return set_range(arg);
713 
714  case MAGIOCGRANGE:
715  return _range_ga;
716 
717  case MAGIOCSSCALE:
718  /* set new scale factors */
719  memcpy(&_scale, (struct mag_calibration_s *)arg, sizeof(_scale));
720  return 0;
721 
722  case MAGIOCGSCALE:
723  /* copy out scale factors */
724  memcpy((struct mag_calibration_s *)arg, &_scale, sizeof(_scale));
725  return 0;
726 
727  case MAGIOCCALIBRATE:
728  return calibrate(filp, arg);
729 
730  case MAGIOCEXSTRAP:
731  return set_excitement(arg);
732 
733  case MAGIOCGEXTERNAL:
734  DEVICE_DEBUG("MAGIOCGEXTERNAL in main driver");
735  return _interface->ioctl(cmd, dummy);
736 
737  case MAGIOCSTEMPCOMP:
738  return set_temperature_compensation(arg);
739 
740  default:
741  /* give it to the superclass */
742  return CDev::ioctl(filp, cmd, arg);
743  }

782 void
783 HMC5883::cycle()
784 {
785  if (_measure_ticks == 0) {
786  return;
787  }
788 
789  /* collection phase? */
790  if (_collect_phase) {
791 
792  /* perform collection */
793  if (OK != collect()) {
794  DEVICE_DEBUG("collection error");
795  /* restart the measurement state machine */
796  start();
797  return;
798  }
799 
800  /* next phase is measurement */
801  _collect_phase = false;
802 
803  /*
804  * Is there a collect->measure gap?
805  */
806  if (_measure_ticks > USEC2TICK(HMC5883_CONVERSION_INTERVAL)) {
807 
808  /* schedule a fresh cycle call when we are ready to measure again */
809  work_queue(HPWORK,
810  &_work,
811  (worker_t)&HMC5883::cycle_trampoline,
812  this,
813  _measure_ticks - USEC2TICK(HMC5883_CONVERSION_INTERVAL));
814 
815  return;
816  }
817  }
818 
819  /* measurement phase */
820  if (OK != measure()) {
821  DEVICE_DEBUG("measure error");
822  }
823 
824  /* next phase is collection */
825  _collect_phase = true;
826 
827  if (_measure_ticks > 0) {
828  /* schedule a fresh cycle call when the measurement is done */
829  work_queue(HPWORK,
830  &_work,
831  (worker_t)&HMC5883::cycle_trampoline,
832  this,
833  USEC2TICK(HMC5883_CONVERSION_INTERVAL));
834  }
835 }

854 int
855 HMC5883::collect()
856 {
857 #pragma pack(push, 1)
858  struct { /* status register and data as read back from the device */
859  uint8_t x[2];
860  uint8_t z[2];
861  uint8_t y[2];
862  } hmc_report;
863 #pragma pack(pop)
864  struct {
865  int16_t x, y, z;
866  } report;
867 
868  int ret;
869  uint8_t check_counter;
870 
871  perf_begin(_sample_perf);
872  struct mag_report new_report;
873  bool sensor_is_onboard = false;
874 
875  float xraw_f;
876  float yraw_f;
877  float zraw_f;
878 
879  /* this should be fairly close to the end of the measurement, so the best approximation of the time */
880  new_report.timestamp = hrt_absolute_time();
881  new_report.error_count = perf_event_count(_comms_errors);
882  new_report.scaling = _range_scale;
883  new_report.device_id = _device_id.devid;
884 
885  /*
886  * @note We could read the status register here, which could tell us that
887  * we were too early and that the output registers are still being
888  * written. In the common case that would just slow us down, and
889  * we're better off just never being early.
890  */
891 
892  /* get measurements from the device */
893  ret = _interface->read(ADDR_DATA_OUT_X_MSB, (uint8_t *)&hmc_report, sizeof(hmc_report));
894 
895  if (ret != OK) {
896  perf_count(_comms_errors);
897  DEVICE_DEBUG("data/status read error");
898  goto out;
899  }
900 
901  /* swap the data we just received */
902  report.x = (((int16_t)hmc_report.x[0]) << 8) + hmc_report.x[1];
903  report.y = (((int16_t)hmc_report.y[0]) << 8) + hmc_report.y[1];
904  report.z = (((int16_t)hmc_report.z[0]) << 8) + hmc_report.z[1];
905 
906  /*
907  * If any of the values are -4096, there was an internal math error in the sensor.
908  * Generalise this to a simple range check that will also catch some bit errors.
909  */
910  if ((abs(report.x) > 2048) ||
911  (abs(report.y) > 2048) ||
912  (abs(report.z) > 2048)) {
913  perf_count(_comms_errors);
914  goto out;
915  }
916 
917  /* get measurements from the device */
918  new_report.temperature = 0;
919 
920  if (_conf_reg & HMC5983_TEMP_SENSOR_ENABLE) {
921  /*
922  if temperature compensation is enabled read the
923  temperature too.
924 
925  We read the temperature every 10 samples to avoid
926  excessive I2C traffic
927  */
928  if (_temperature_counter++ == 10) {
929  uint8_t raw_temperature[2];
930 
931  _temperature_counter = 0;
932 
933  ret = _interface->read(ADDR_TEMP_OUT_MSB,
934  raw_temperature, sizeof(raw_temperature));
935 
936  if (ret == OK) {
937  int16_t temp16 = (((int16_t)raw_temperature[0]) << 8) +
938  raw_temperature[1];
939  new_report.temperature = 25 + (temp16 / (16 * 8.0f));
940  _temperature_error_count = 0;
941 
942  } else {
943  _temperature_error_count++;
944 
945  if (_temperature_error_count == 10) {
946  /*
947  it probably really is an old HMC5883,
948  and can't do temperature. Disable it
949  */
950  _temperature_error_count = 0;
951  DEVICE_DEBUG("disabling temperature compensation");
952  set_temperature_compensation(0);
953  }
954  }
955 
956  } else {
957  new_report.temperature = _last_report.temperature;
958  }
959  }
960 
961  /*
962  * RAW outputs
963  *
964  * to align the sensor axes with the board, x and y need to be flipped
965  * and y needs to be negated
966  */
967  new_report.x_raw = -report.y;
968  new_report.y_raw = report.x;
969  /* z remains z */
970  new_report.z_raw = report.z;
971 
972  /* scale values for output */
973 
974  // XXX revisit for SPI part, might require a bus type IOCTL
975  unsigned dummy;
976  sensor_is_onboard = !_interface->ioctl(MAGIOCGEXTERNAL, dummy);
977  new_report.is_external = !sensor_is_onboard;
978 
979  if (sensor_is_onboard) {
980  // convert onboard so it matches offboard for the
981  // scaling below
982  report.y = -report.y;
983  report.x = -report.x;
984  }
985 
986  /* the standard external mag by 3DR has x pointing to the
987  * right, y pointing backwards, and z down, therefore switch x
988  * and y and invert y */
989  xraw_f = -report.y;
990  yraw_f = report.x;
991  zraw_f = report.z;
992 
993  // apply user specified rotation
994  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
995 
996  new_report.x = ((xraw_f * _range_scale) - _scale.x_offset) * _scale.x_scale;
997  /* flip axes and negate value for y */
998  new_report.y = ((yraw_f * _range_scale) - _scale.y_offset) * _scale.y_scale;
999  /* z remains z */
1000  new_report.z = ((zraw_f * _range_scale) - _scale.z_offset) * _scale.z_scale;
1001 
1002  if (!(_pub_blocked)) {
1003 
1004  if (_mag_topic != nullptr) {
1005  /* publish it */
1006  orb_publish(ORB_ID(sensor_mag), _mag_topic, &new_report);
1007 
1008  } else {
1009  _mag_topic = orb_advertise_multi(ORB_ID(sensor_mag), &new_report,
1010  &_orb_class_instance, (sensor_is_onboard) ? ORB_PRIO_HIGH : ORB_PRIO_MAX);
1011 
1012  if (_mag_topic == nullptr) {
1013  DEVICE_DEBUG("ADVERT FAIL");
1014  }
1015  }
1016  }
1017 
1018  _last_report = new_report;
1019 
1020  /* post a report to the ring */
1021  _reports->force(&new_report);
1022 
1023  /* notify anyone waiting for data */
1024  poll_notify(POLLIN);
1025 
1026  /*
1027  periodically check the range register and configuration
1028  registers. With a bad I2C cable it is possible for the
1029  registers to become corrupt, leading to bad readings. It
1030  doesn't happen often, but given the poor cables some
1031  vehicles have it is worth checking for.
1032  */
1033  check_counter = perf_event_count(_sample_perf) % 256;
1034 
1035  if (check_counter == 0) {
1036  check_range();
1037  }
1038 
1039  if (check_counter == 128) {
1040  check_conf();
1041  }
1042 
1043  ret = OK;
1044 
1045 out:
1046  perf_end(_sample_perf);
1047  return ret;
1048 }

1103  goto out;

1109  goto out;
1110  }
1111

1114  ret = 1;
1115  goto out;
1116  }

1119  for (uint8_t i = 0; i < 10; i++) {
1120  struct pollfd fds;
1121

1160  warn("ERROR: READ 2");
1161  ret = -EIO;
1162  goto out;
1163  }
1164

1166  fabsf(expected_cal[1] / report.y),
1167  fabsf(expected_cal[2] / report.z)
1168  };
1169 
1170  if (cal[0] > 0.3f && cal[0] < 1.7f &&
1171  cal[1] > 0.3f && cal[1] < 1.7f &&
1172  cal[2] > 0.3f && cal[2] < 1.7f) {
1173  good_count++;
1174  sum_excited[0] += cal[0];
1175  sum_excited[1] += cal[1];
1176  sum_excited[2] += cal[2];
1177  }
1178  }
1179 
1180  if (good_count < 5) {
1181  ret = -EIO;
1182  goto out;
1183  }
1184 
1185  float scaling[3];
1186 
1187  scaling[0] = sum_excited[0] / good_count;
1188  scaling[1] = sum_excited[1] / good_count;
1189  scaling[2] = sum_excited[2] / good_count;
1190 
1191  /* set scaling in device */
1192  mscale_previous.x_scale = 1.0f / scaling[0];
1193  mscale_previous.y_scale = 1.0f / scaling[1];
1194  mscale_previous.z_scale = 1.0f / scaling[2];
1195 
1196  ret = OK;
1197 
1198 out:
1199 
1200  if (OK != ioctl(filp, MAGIOCSSCALE, (long unsigned int)&mscale_previous)) {
1201  warn("FAILED: MAGIOCSSCALE 2");
1202  }
1203

1205  /* Set to 1.9 Gauss */
1206  if (OK != ::ioctl(fd, MAGIOCSRANGE, 2)) {
1207  warnx("FAILED: MAGIOCSRANGE 1.9 Ga");
1208  }
1209

1211  warnx("FAILED: MAGIOCEXSTRAP 0");
1212  }

1217  warnx("FAILED: SCALE");
1218  ret = PX4_ERROR;
1219  }
1220 
1221  }
1222 
1223  return ret;

1616 /**
1617  * Automatic scale calibration.
1618  *
1619  * Basic idea:
1620  *
1621  * output = (ext field +- 1.1 Ga self-test) * scale factor
1622  *
1623  * and consequently:
1624  *
1625  * 1.1 Ga = (excited - normal) * scale factor
1626  * scale factor = (excited - normal) / 1.1 Ga
1627  *
1628  * sxy = (excited - normal) / 766 | for conf reg. B set to 0x60 / Gain = 3
1629  * sz = (excited - normal) / 713 | for conf reg. B set to 0x60 / Gain = 3
1630  *
1631  * By subtracting the non-excited measurement the pure 1.1 Ga reading
1632  * can be extracted and the sensitivity of all axes can be matched.
1633  *
1634  * SELF TEST OPERATION
1635  * To check the HMC5883L for proper operation, a self test feature in incorporated
1636  * in which the sensor offset straps are excited to create a nominal field strength
1637  * (bias field) to be measured. To implement self test, the least significant bits
1638  * (MS1 and MS0) of configuration register A are changed from 00 to 01 (positive bias)
1639  * or 10 (negetive bias), e.g. 0x11 or 0x12.
1640  * Then, by placing the mode register into single-measurement mode (0x01),
1641  * two data acquisition cycles will be made on each magnetic vector.
1642  * The first acquisition will be a set pulse followed shortly by measurement
1643  * data of the external field. The second acquisition will have the offset strap
1644  * excited (about 10 mA) in the positive bias mode for X, Y, and Z axes to create
1645  * about a ±1.1 gauss self test field plus the external field. The first acquisition
1646  * values will be subtracted from the second acquisition, and the net measurement
1647  * will be placed into the data output registers.
1648  * Since self test adds ~1.1 Gauss additional field to the existing field strength,
1649  * using a reduced gain setting prevents sensor from being saturated and data registers
1650  * overflowed. For example, if the configuration register B is set to 0x60 (Gain=3),
1651  * values around +766 LSB (1.16 Ga * 660 LSB/Ga) will be placed in the X and Y data
1652  * output registers and around +713 (1.08 Ga * 660 LSB/Ga) will be placed in Z data
1653  * output register. To leave the self test mode, change MS1 and MS0 bit of the
1654  * configuration register A back to 00 (Normal Measurement Mode), e.g. 0x10.
1655  * Using the self test method described above, the user can scale sensor
1656  */
1657 int calibrate(enum HMC5883_BUS busid)
1658 {
1659  int ret;
1660  struct hmc5883_bus_option &bus = find_bus(busid);

1662 
1663  int fd = open(path, O_RDONLY);
1664

1667  }

1670  warnx("failed to enable sensor calibration mode");
