115 int RoboClaw::readEncoder(e_motor motor)
116 {
117  uint16_t sum = 0;
118 
119  if (motor == MOTOR_1) {
120  _sendCommand(CMD_READ_ENCODER_1, nullptr, 0, sum);
121 
122  } else if (motor == MOTOR_2) {
123  _sendCommand(CMD_READ_ENCODER_2, nullptr, 0, sum);
124  }
125 
126  uint8_t rbuf[50];
127  usleep(5000);
128  int nread = read(_uart, rbuf, 50);
129 
130  if (nread < 6) {
131  printf("failed to read\n");
132  return -1;
133  }
134 
135  //printf("received: \n");
136  //for (int i=0;i<nread;i++) {
137  //printf("%d\t", rbuf[i]);
138  //}
139  //printf("\n");
140  uint32_t count = 0;
141  uint8_t *countBytes = (uint8_t *)(&count);
142  countBytes[3] = rbuf[0];
143  countBytes[2] = rbuf[1];
144  countBytes[1] = rbuf[2];
145  countBytes[0] = rbuf[3];
146  uint8_t status = rbuf[4];
147  uint8_t checksum = rbuf[5];
148  uint8_t checksum_computed = (sum + _sumBytes(rbuf, 5)) &
149  checksum_mask;
150 
151  // check if checksum is valid
152  if (checksum != checksum_computed) {
153  printf("checksum failed: expected %d got %d\n",
154  checksum, checksum_computed);
155  return -1;
156  }
157 
158  int overFlow = 0;
159 
160  if (status & STATUS_REVERSE) {
161  //printf("roboclaw: reverse\n");
162  }
163 
164  if (status & STATUS_UNDERFLOW) {
165  //printf("roboclaw: underflow\n");
166  overFlow = -1;
167 
168  } else if (status & STATUS_OVERFLOW) {
169  //printf("roboclaw: overflow\n");
170  overFlow = +1;
171  }
172 
173  static int64_t overflowAmount = 0x100000000LL;
174 
175  if (motor == MOTOR_1) {
176  _motor1Overflow += overFlow;
177  _motor1Position = float(int64_t(count) +
178  _motor1Overflow * overflowAmount) / _pulsesPerRev;
179 
180  } else if (motor == MOTOR_2) {
181  _motor2Overflow += overFlow;
182  _motor2Position = float(int64_t(count) +
183  _motor2Overflow * overflowAmount) / _pulsesPerRev;
184  }
185 
186  return 0;
187 }

227 int RoboClaw::setMotorSpeed(e_motor motor, float value)
228 {
229  uint16_t sum = 0;
230 
231  // bound
232  if (value > 1) { value = 1; }
233 
234  if (value < -1) { value = -1; }
235 
236  uint8_t speed = fabs(value) * 127;
237 
238  // send command
239  if (motor == MOTOR_1) {
240  if (value > 0) {
241  return _sendCommand(CMD_DRIVE_FWD_1, &speed, 1, sum);
242 
243  } else {
244  return _sendCommand(CMD_DRIVE_REV_1, &speed, 1, sum);
245  }
246 
247  } else if (motor == MOTOR_2) {
248  if (value > 0) {
249  return _sendCommand(CMD_DRIVE_FWD_2, &speed, 1, sum);
250 
251  } else {
252  return _sendCommand(CMD_DRIVE_REV_2, &speed, 1, sum);
253  }
254  }
255 
256  return -1;

290 int RoboClaw::update()
291 {

297 
298  // if new data, send to motors

300  _actuators.update();
301  setMotorDutyCycle(MOTOR_1, _actuators.get().control[CH_VOLTAGE_LEFT]);

303  }
304

322 int RoboClaw::_sendCommand(e_command cmd, uint8_t *data,
323  size_t n_data, uint16_t &prev_sum)
324 {
325  tcflush(_uart, TCIOFLUSH); // flush buffers
326  uint8_t buf[n_data + 3];
327  buf[0] = _address;
328  buf[1] = cmd;
329 
330  for (size_t i = 0; i < n_data; i++) {
331  buf[i + 2] = data[n_data - i - 1]; // MSB
332  }
333 
334  uint16_t sum = _sumBytes(buf, n_data + 2);
335  prev_sum += sum;
336  buf[n_data + 2] = sum & checksum_mask;
337  //printf("\nmessage:\n");
338  //for (size_t i=0;i<n_data+3;i++) {
339  //printf("%d\t", buf[i]);
340  //}
341  //printf("\n");
342  return write(_uart, buf, n_data + 3);
343 }
