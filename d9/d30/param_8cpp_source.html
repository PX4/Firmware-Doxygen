95 ### Description
96 Command to access and manipulate parameters via shell or script.
97 
98 This is used for example in the startup script to set airframe-specific parameters.
99 
100 Parameters are automatically saved when changed, eg. with `param set`. They are typically stored to FRAM

484 static void
485 do_show_print(void *arg, param_t param)
486 {
487  int32_t i;
488  float f;
489  const char *search_string = (const char *)arg;
490  const char *p_name = (const char *)param_name(param);
491 
492  /* print nothing if search string is invalid and not matching */
493  if (!(arg == nullptr)) {
494 
495  /* start search */
496  const char *ss = search_string;
497  const char *pp = p_name;
498 
499  /* XXX this comparison is only ok for trailing wildcards */
500  while (*ss != '\0' && *pp != '\0') {
501 
502  // case insensitive comparison (param_name is always upper case)
503  if (toupper(*ss) == *pp) {
504  ss++;
505  pp++;
506 
507  } else if (*ss == '*') {
508  if (*(ss + 1) != '\0') {
509  PX4_WARN("* symbol only allowed at end of search string");
510  // FIXME - should exit
511  return;
512  }
513 
514  pp++;
515 
516  } else {
517  /* param not found */
518  return;
519  }
520  }
521 
522  /* the search string must have been consumed */
523  if (!(*ss == '\0' || *ss == '*') || *pp != '\0') {
524  return;
525  }
526  }
527 
528  PARAM_PRINT("%c %c %s [%d,%d] : ", (param_used(param) ? 'x' : ' '),
529  param_value_unsaved(param) ? '*' : (param_value_is_default(param) ? ' ' : '+'),
530  param_name(param), param_get_used_index(param), param_get_index(param));
531 
532  /*
533  * This case can be expanded to handle printing common structure types.
534  */
535 
536  switch (param_type(param)) {
537  case PARAM_TYPE_INT32:
538  if (!param_get(param, &i)) {
539  PARAM_PRINT("%ld\n", (long)i);
540  return;
541  }
542 
543  break;
544 
545  case PARAM_TYPE_FLOAT:
546  if (!param_get(param, &f)) {
547  PARAM_PRINT("%4.4f\n", (double)f);
548  return;
549  }
550 
551  break;
552 
553  case PARAM_TYPE_STRUCT ... PARAM_TYPE_STRUCT_MAX:
554  PARAM_PRINT("<struct type %d size %zu>\n", 0 + param_type(param), param_size(param));
555  return;
556 
557  default:
558  PARAM_PRINT("<unknown type %d>\n", 0 + param_type(param));
559  return;
560  }
561 
562  PARAM_PRINT("<error fetching parameter %lu>\n", (unsigned long)param);
563 }

565 static int
566 do_set(const char *name, const char *val, bool fail_on_not_found)
567 {
568  int32_t i;
569  float f;
570  param_t param = param_find(name);
571 
572  /* set nothing if parameter cannot be found */
573  if (param == PARAM_INVALID) {
574  /* param not found - fail silenty in scripts as it prevents booting */
575  PX4_ERR("Parameter %s not found.", name);
576  return (fail_on_not_found) ? 1 : 0;
577  }
578 
579  /*
580  * Set parameter if type is known and conversion from string to value turns out fine
581  */
582 
583  switch (param_type(param)) {
584  case PARAM_TYPE_INT32:
585  if (!param_get(param, &i)) {
586 
587  /* convert string */
588  char *end;
589  int32_t newval = strtol(val, &end, 10);
590 
591  if (i != newval) {
592  PARAM_PRINT("%c %s: ",
593  param_value_unsaved(param) ? '*' : (param_value_is_default(param) ? ' ' : '+'),
594  param_name(param));
595  PARAM_PRINT("curr: %ld", (long)i);
596  param_set(param, &newval);
597  PARAM_PRINT(" -> new: %ld\n", (long)newval);
598  }
599  }
600 
601  break;
602 
603  case PARAM_TYPE_FLOAT:
604  if (!param_get(param, &f)) {
605 
606  /* convert string */
607  char *end;
608  float newval = strtod(val, &end);
609 #pragma GCC diagnostic push
610 #pragma GCC diagnostic ignored "-Wfloat-equal"
611 
612  if (f != newval) {
613 #pragma GCC diagnostic pop
614  PARAM_PRINT("%c %s: ",
615  param_value_unsaved(param) ? '*' : (param_value_is_default(param) ? ' ' : '+'),
616  param_name(param));
617  PARAM_PRINT("curr: %4.4f", (double)f);
618  param_set(param, &newval);
619  PARAM_PRINT(" -> new: %4.4f\n", (double)newval);
620  }
621 
622  }
623 
624  break;
625 
626  default:
627  PX4_ERR("<unknown / unsupported type %d>\n", 0 + param_type(param));
628  return 1;
629  }
630 
631  return 0;
632 }

634 static int
635 do_compare(const char *name, char *vals[], unsigned comparisons, enum COMPARE_OPERATOR cmp_op)
636 {
637  int32_t i;
638  float f;
639  param_t param = param_find(name);
640 
641  /* set nothing if parameter cannot be found */
642  if (param == PARAM_INVALID) {
643  /* param not found */
644  PX4_ERR("Parameter %s not found", name);
645  return 1;
646  }
647 
648  /*
649  * Set parameter if type is known and conversion from string to value turns out fine
650  */
651 
652  int ret = 1;
653 
654  switch (param_type(param)) {
655  case PARAM_TYPE_INT32:
656  if (!param_get(param, &i)) {
657 
658  /* convert string */
659  char *end;
660 
661  for (unsigned k = 0; k < comparisons; k++) {
662 
663  int j = strtol(vals[k], &end, 10);
664 
665  if (((cmp_op == COMPARE_OPERATOR_EQUAL) && (i == j)) ||
666  ((cmp_op == COMPARE_OPERATOR_GREATER) && (i > j))) {
667  PX4_DEBUG(" %ld: ", (long)i);
668  ret = 0;
669  }
670  }
671  }
672 
673  break;
674 
675  case PARAM_TYPE_FLOAT:
676  if (!param_get(param, &f)) {
677 
678  /* convert string */
679  char *end;
680 
681  for (unsigned k = 0; k < comparisons; k++) {
682 
683  float g = strtod(vals[k], &end);
684 
685  if (((cmp_op == COMPARE_OPERATOR_EQUAL) && (fabsf(f - g) < 1e-7f)) ||
686  ((cmp_op == COMPARE_OPERATOR_GREATER) && (f > g))) {
687  PX4_DEBUG(" %4.4f: ", (double)f);
688  ret = 0;
689  }
690  }
691  }
692 
693  break;
694 
695  default:
696  PX4_ERR("<unknown / unsupported type %d>", 0 + param_type(param));
697  return 1;
698  }
699 
700  if (ret == 0) {
701  PX4_DEBUG("%c %s: match\n",
702  param_value_unsaved(param) ? '*' : (param_value_is_default(param) ? ' ' : '+'),
703  param_name(param));
704  }
705 
706  return ret;
707 }
