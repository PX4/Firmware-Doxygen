101 void BBBlueADC::_measure()
102 {
103 #ifdef __DF_BBBLUE
104  px4_adc_msg_t tmp_samples[BBBLUE_MAX_ADC_CHANNELS];
105 
106  for (int i = 0; i < BBBLUE_MAX_ADC_USER_CHANNELS; ++i) {
107  tmp_samples[i].am_channel = i;
108  tmp_samples[i].am_data = rc_adc_read_raw(i);
109  }
110 
111  tmp_samples[BBBLUE_MAX_ADC_USER_CHANNELS].am_channel = BBBLUE_ADC_LIPO_BATTERY_VOLTAGE_CHANNEL;
112  tmp_samples[BBBLUE_MAX_ADC_USER_CHANNELS].am_data = rc_adc_read_raw(BBBLUE_ADC_LIPO_BATTERY_VOLTAGE_CHANNEL);
113 
114  tmp_samples[BBBLUE_ADC_DC_JACK_VOLTAGE_CHANNEL].am_channel = BBBLUE_ADC_DC_JACK_VOLTAGE_CHANNEL;
115  tmp_samples[BBBLUE_ADC_DC_JACK_VOLTAGE_CHANNEL].am_data = rc_adc_read_raw(BBBLUE_ADC_DC_JACK_VOLTAGE_CHANNEL);
116 
117  pthread_mutex_lock(&_samples_lock);
118  memcpy(&_samples, &tmp_samples, sizeof(tmp_samples));
119  pthread_mutex_unlock(&_samples_lock);
120 #endif
121 }

144 ssize_t BBBlueADC::devRead(void *buf, size_t count)
145 {
146  const size_t maxsize = sizeof(_samples);
147  int ret;
148 
149  if (count > maxsize) {
150  count = maxsize;
151  }
152 
153  ret = pthread_mutex_trylock(&_samples_lock);
154 
155  if (ret != 0) {
156  return 0;
157  }
158 
159  memcpy(buf, &_samples, count);
160  pthread_mutex_unlock(&_samples_lock);
161 
162  return count;
163 }

167 int bbblue_adc_main(int argc, char *argv[])
168 {
169  int ret;
170 
171  if (argc < 2) {
172  PX4_WARN("usage: {start|stop|test}");
173  return PX4_ERROR;
174  }
175 
176  if (!strcmp(argv[1], "start")) {
177  if (instance) {
178  PX4_WARN("already started");
179  return PX4_OK;
180  }
181 
182  instance = new BBBlueADC;
183 
184  if (!instance) {
185  PX4_WARN("not enough memory");
186  return PX4_ERROR;
187  }
188 
189  if (instance->init() != PX4_OK) {
190  delete instance;
191  instance = nullptr;
192  PX4_WARN("init failed");
193  return PX4_ERROR;
194  }
195 
196  PX4_INFO("BBBlueADC started");
197  return PX4_OK;
198 
199  } else if (!strcmp(argv[1], "stop")) {
200  if (!instance) {
201  PX4_WARN("already stopped");
202  return PX4_OK;
203  }
204 
205  delete instance;
206  instance = nullptr;
207  return PX4_OK;
208 
209  } else if (!strcmp(argv[1], "test")) {
210  if (!instance) {
211  PX4_ERR("start first");
212  return PX4_ERROR;
213  }
214 
215  px4_adc_msg_t adc_msgs[BBBLUE_MAX_ADC_CHANNELS];
216 
217  ret = instance->devRead((char *)&adc_msgs, sizeof(adc_msgs));
218 
219  if (ret < 0) {
220  PX4_ERR("ret: %s (%d)\n", strerror(ret), ret);
221  return ret;
222 
223  } else if (ret != sizeof(adc_msgs)) {
224  PX4_ERR("incomplete read: %d expected %d", ret, sizeof(adc_msgs));
225  return ret;
226  }
227 
228  for (int i = 0; i < BBBLUE_MAX_ADC_USER_CHANNELS; ++i) {
229  PX4_INFO("ADC channel: %d; value: %d", (int)adc_msgs[i].am_channel,
230  adc_msgs[i].am_data);
231  }
232 
233  for (int i = BBBLUE_MAX_ADC_USER_CHANNELS; i < BBBLUE_MAX_ADC_CHANNELS; ++i) {
234  PX4_INFO("ADC channel: %d; value: %d, voltage: %6.2f V", (int)adc_msgs[i].am_channel,
235  adc_msgs[i].am_data, adc_msgs[i].am_data * 1.8 / 4095.0 * 11.0);
236 
237  }
238 
239  return PX4_OK;
240 
241  } else {
242  PX4_WARN("action (%s) not supported", argv[1]);
243 
244  return PX4_ERROR;
245  }
246 
247  return PX4_OK;
248 
249 }
