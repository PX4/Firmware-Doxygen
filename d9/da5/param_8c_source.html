176 /** lock the parameter store for read access */
177 static void
178 param_lock_reader(void)
179 {
180  do {} while (px4_sem_wait(&reader_lock_holders_lock) != 0);
181 
182  ++reader_lock_holders;
183 
184  if (reader_lock_holders == 1) {
185  // the first reader takes the lock, the next ones are allowed to just continue
186  do {} while (px4_sem_wait(&param_sem) != 0);
187  }
188 
189  px4_sem_post(&reader_lock_holders_lock);
190 }

329 param_t
330 param_find_internal(const char *name, bool notification)
331 {
332  perf_begin(param_find_perf);
333 
334  param_t middle;
335  param_t front = 0;
336  param_t last = get_param_info_count();
337 
338  /* perform a binary search of the known parameters */
339 
340  while (front <= last) {
341  middle = front + (last - front) / 2;
342  int ret = strcmp(name, param_info_base[middle].name);
343 
344  if (ret == 0) {
345  if (notification) {
346  param_set_used_internal(middle);
347  }
348 
349  perf_end(param_find_perf);
350  return middle;
351 
352  } else if (middle == front) {
353  /* An end point has been hit, but there has been no match */
354  break;
355 
356  } else if (ret < 0) {
357  last = middle;
358 
359  } else {
360  front = middle;
361  }
362  }
363 
364  perf_end(param_find_perf);
365 
366  /* not found */
367  return PARAM_INVALID;
368 }

420 param_t
421 param_for_used_index(unsigned index)
422 {
423  int count = get_param_info_count();
424 
425  if (count && (int)index < count) {
426  /* walk all params and count used params */
427  unsigned used_count = 0;
428 
429  for (int i = 0; i < size_param_changed_storage_bytes; i++) {
430  for (int j = 0; j < bits_per_allocation_unit; j++) {
431  if (param_changed_storage[i] & (1 << j)) {
432 
433  /* we found the right used count,
434  * return the param value
435  */
436  if (index == used_count) {
437  return (param_t)(i * bits_per_allocation_unit + j);
438  }
439 
440  used_count++;
441  }
442  }
443  }
444  }
445 
446  return PARAM_INVALID;

640 /**
641  * Automatically save the parameters after a timeout and limited rate.
642  *
643  * This needs to be called with the writer lock held (it's not necessary that it's the writer lock, but it
644  * needs to be the same lock as autosave_worker() and param_control_autosave() use).
645  */
646 static void
647 param_autosave(void)
648 {
649 #ifndef PARAM_NO_AUTOSAVE
650 
651  if (autosave_scheduled || autosave_disabled) {
652  return;
653  }
654 
655  // wait at least 300ms before saving, because:
656  // - tasks often call param_set() for multiple params, so this avoids unnecessary save calls
657  // - the logger stores changed params. He gets notified on a param change via uORB and then
658  // looks at all unsaved params.
659  hrt_abstime delay = 300 * 1000;
660 
661  const hrt_abstime rate_limit = 2000 * 1000; // rate-limit saving to 2 seconds
662  hrt_abstime last_save_elapsed = hrt_elapsed_time(&last_autosave_timestamp);
663 
664  if (last_save_elapsed < rate_limit && rate_limit > last_save_elapsed + delay) {
665  delay = rate_limit - last_save_elapsed;
666  }
667 
668  autosave_scheduled = true;
669  work_queue(LPWORK, &autosave_work, (worker_t)&autosave_worker, NULL, USEC2TICK(delay));
670 #endif /* PARAM_NO_AUTOSAVE */
671 }

689 static int
690 param_set_internal(param_t param, const void *val, bool mark_saved, bool notify_changes)
691 {
692  int result = -1;
693  bool params_changed = false;
694 
695  param_lock_writer();
696  perf_begin(param_set_perf);
697 
698  if (param_values == NULL) {
699  utarray_new(param_values, &param_icd);
700  }
701 
702  if (param_values == NULL) {
703  PX4_ERR("failed to allocate modified values array");
704  goto out;
705  }
706 
707  if (handle_in_range(param)) {
708 
709  struct param_wbuf_s *s = param_find_changed(param);
710 
711  if (s == NULL) {
712 
713  /* construct a new parameter */
714  struct param_wbuf_s buf = {
715  .param = param,
716  .val.p = NULL,
717  .unsaved = false
718  };
719  params_changed = true;
720 
721  /* add it to the array and sort */
722  utarray_push_back(param_values, &buf);
723  utarray_sort(param_values, param_compare_values);
724 
725  /* find it after sorting */
726  s = param_find_changed(param);
727  }
728 
729  /* update the changed value */
730  switch (param_type(param)) {
731 
732  case PARAM_TYPE_INT32:
733  params_changed = params_changed || s->val.i != *(int32_t *)val;
734  s->val.i = *(int32_t *)val;
735  break;
736 
737  case PARAM_TYPE_FLOAT:
738  params_changed = params_changed || fabsf(s->val.f - * (float *)val) > FLT_EPSILON;
739  s->val.f = *(float *)val;
740  break;
741 
742  case PARAM_TYPE_STRUCT ... PARAM_TYPE_STRUCT_MAX:
743  if (s->val.p == NULL) {
744  size_t psize = param_size(param);
745 
746  if (psize > 0) {
747  s->val.p = malloc(psize);
748 
749  } else {
750  s->val.p = NULL;
751  }
752 
753  if (s->val.p == NULL) {
754  PX4_ERR("failed to allocate parameter storage");
755  goto out;
756  }
757  }
758 
759  memcpy(s->val.p, val, param_size(param));
760  params_changed = true;
761  break;
762 
763  default:
764  goto out;
765  }
766 
767  s->unsaved = !mark_saved;
768  result = 0;
769 
770  if (!mark_saved) { // this is false when importing parameters
771  param_autosave();
772  }
773  }
774 
775 out:
776  perf_end(param_set_perf);
777  param_unlock_writer();
778 
779  /*
780  * If we set something, now that we have unlocked, go ahead and advertise that
781  * a thing has been set.

1032 int
1033 param_export(int fd, bool only_unsaved)
1034 {
1035  perf_begin(param_export_perf);
1036 
1037  struct param_wbuf_s *s = NULL;
1038  int result = -1;
1039 
1040  struct bson_encoder_s encoder;
1041 
1042  int shutdown_lock_ret = px4_shutdown_lock();
1043 
1044  if (shutdown_lock_ret) {
1045  PX4_ERR("px4_shutdown_lock() failed (%i)", shutdown_lock_ret);
1046  }
1047 
1048  // take the file lock
1049  do {} while (px4_sem_wait(&param_sem_save) != 0);
1050 
1051  param_lock_reader();
1052 
1053  uint8_t bson_buffer[256];
1054  bson_encoder_init_buf_file(&encoder, fd, &bson_buffer, sizeof(bson_buffer));
1055 
1056  /* no modified parameters -> we are done */
1057  if (param_values == NULL) {
1058  result = 0;
1059  goto out;
1060  }
1061 
1062  while ((s = (struct param_wbuf_s *)utarray_next(param_values, s)) != NULL) {
1063  /*
1064  * If we are only saving values changed since last save, and this
1065  * one hasn't, then skip it
1066  */
1067  if (only_unsaved && !s->unsaved) {
1068  continue;
1069  }
1070 
1071  s->unsaved = false;
1072 
1073  const char *name = param_name(s->param);
1074  const size_t size = param_size(s->param);
1075 
1076  /* append the appropriate BSON type object */
1077  switch (param_type(s->param)) {
1078 
1079  case PARAM_TYPE_INT32: {
1080  const int32_t i = s->val.i;
1081 
1082  debug("exporting: %s (%d) size: %d val: %d", name, s->param, size, i);
1083 
1084  if (bson_encoder_append_int(&encoder, name, i)) {
1085  PX4_ERR("BSON append failed for '%s'", name);
1086  goto out;
1087  }
1088  }
1089  break;
1090 
1091  case PARAM_TYPE_FLOAT: {
1092  const double f = (double)s->val.f;
1093 
1094  debug("exporting: %s (%d) size: %d val: %.3f", name, s->param, size, (double)f);
1095 
1096  if (bson_encoder_append_double(&encoder, name, f)) {
1097  PX4_ERR("BSON append failed for '%s'", name);
1098  goto out;
1099  }
1100  }
1101  break;
1102 
1103  case PARAM_TYPE_STRUCT ... PARAM_TYPE_STRUCT_MAX: {
1104  const void *value_ptr = param_get_value_ptr(s->param);
1105 
1106  /* lock as short as possible */
1107  if (bson_encoder_append_binary(&encoder,
1108  name,
1109  BSON_BIN_BINARY,
1110  size,
1111  value_ptr)) {
1112 
1113  PX4_ERR("BSON append failed for '%s'", name);
1114  goto out;
1115  }
1116  }
1117  break;
1118 
1119  default:
1120  PX4_ERR("unrecognized parameter type");
1121  goto out;
1122  }
1123  }
1124 
1125  result = 0;
1126 
1127 out:
1128 
1129  if (result == 0) {
1130  if (bson_encoder_fini(&encoder) != PX4_OK) {
1131  PX4_ERR("bson encoder finish failed");
1132  }
1133  }
1134 
1135  param_unlock_reader();
1136 
1137  px4_sem_post(&param_sem_save);
1138 
1139  if (shutdown_lock_ret == 0) {
1140  px4_shutdown_unlock();
1141  }
1142 
1143  perf_end(param_export_perf);
1144 
1145  return result;
1146 }

1152 static int
1153 param_import_callback(bson_decoder_t decoder, void *private, bson_node_t node)
1154 {
1155  float f;
1156  int32_t i;
1157  void *v, *tmp = NULL;
1158  int result = -1;
1159  struct param_import_state *state = (struct param_import_state *)private;
1160 
1161  /*
1162  * EOO means the end of the parameter object. (Currently not supporting
1163  * nested BSON objects).
1164  */
1165  if (node->type == BSON_EOO) {
1166  debug("end of parameters");
1167  return 0;
1168  }
1169 
1170  /*
1171  * Find the parameter this node represents. If we don't know it,
1172  * ignore the node.
1173  */
1174  param_t param = param_find_no_notification(node->name);
1175 
1176  if (param == PARAM_INVALID) {
1177  debug("ignoring unrecognised parameter '%s'", node->name);
1178  return 1;
1179  }
1180 
1181  /*
1182  * Handle setting the parameter from the node
1183  */
1184 
1185  switch (node->type) {
1186  case BSON_INT32:
1187  if (param_type(param) != PARAM_TYPE_INT32) {
1188  PX4_WARN("unexpected type for %s", node->name);
1189  result = 1; // just skip this entry
1190  goto out;
1191  }
1192 
1193  i = node->i;
1194  v = &i;
1195 
1196  debug("importing: %s (%d) = %d", node->name, param, i);
1197  break;
1198 
1199  case BSON_DOUBLE:
1200  if (param_type(param) != PARAM_TYPE_FLOAT) {
1201  PX4_WARN("unexpected type for %s", node->name);
1202  result = 1; // just skip this entry
1203  goto out;
1204  }
1205 
1206  f = node->d;
1207  v = &f;
1208 
1209  debug("importing: %s (%d) = %.3f", node->name, param, (double)f);
1210  break;
1211 
1212  case BSON_BINDATA:
1213  if (node->subtype != BSON_BIN_BINARY) {
1214  PX4_WARN("unexpected subtype for %s", node->name);
1215  result = 1; // just skip this entry
1216  goto out;
1217  }
1218 
1219  if (bson_decoder_data_pending(decoder) != param_size(param)) {
1220  PX4_WARN("bad size for '%s'", node->name);
1221  result = 1; // just skip this entry
1222  goto out;
1223  }
1224 
1225  /* XXX check actual file data size? */
1226  size_t psize = param_size(param);
1227 
1228  if (psize > 0) {
1229  tmp = malloc(psize);
1230 
1231  } else {
1232  tmp = NULL;
1233  }
1234 
1235  if (tmp == NULL) {
1236  PX4_ERR("failed allocating for '%s'", node->name);
1237  goto out;
1238  }
1239 
1240  if (bson_decoder_copy_data(decoder, tmp)) {
1241  PX4_ERR("failed copying data for '%s'", node->name);
1242  goto out;
1243  }
1244 
1245  v = tmp;
1246  break;
1247 
1248  default:
1249  debug("unrecognised node type");
1250  goto out;
1251  }
1252 
1253  if (param_set_internal(param, v, state->mark_saved, true)) {
1254  debug("error setting value for '%s'", node->name);
1255  goto out;
1256  }
1257 
1258  if (tmp != NULL) {
1259  free(tmp);
1260  tmp = NULL;
1261  }
1262 
1263  /* don't return zero, that means EOF */
1264  result = 1;
1265 
1266 out:
1267 
1268  if (tmp != NULL) {
1269  free(tmp);
1270  }
1271 
1272  return result;
1273 }
