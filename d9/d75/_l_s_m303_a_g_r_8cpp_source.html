139 bool
140 LSM303AGR::self_test()
141 {
142  // Magnetometer self-test procedure (LSM303AGR DocID027765 Rev 5 page 25/68)
143 
144  uint8_t status_m = 0;
145 
146  write_reg(CFG_REG_A_M, 0x0C);
147  write_reg(CFG_REG_B_M, 0x02);
148  write_reg(CFG_REG_C_M, 0x10);
149 
150  // sleep 20ms
151  usleep(20000);
152 
153  uint16_t OUTX_NOST = 0;
154  uint16_t OUTY_NOST = 0;
155  uint16_t OUTZ_NOST = 0;
156 
157  // Check Zyxda 50 times and discard
158  // average x, y, z
159  for (int i = 0; i < 50; i++) {
160 
161  status_m = read_reg(STATUS_REG_M);
162 
163  OUTX_NOST += read_reg(OUTX_L_REG_M) + (read_reg(OUTX_H_REG_M) << 8);
164  OUTY_NOST += read_reg(OUTY_L_REG_M) + (read_reg(OUTY_H_REG_M) << 8);
165  OUTZ_NOST += read_reg(OUTZ_L_REG_M) + (read_reg(OUTZ_H_REG_M) << 8);
166  }
167 
168  // enable self-test
169  write_reg(CFG_REG_C_M, 0x12);
170 
171  // wait for 60ms
172  usleep(60000);
173 
174  // Check Zyxda
175  status_m = read_reg(STATUS_REG_M);
176 
177  // Read mag x, y, z to clear Zyxda bit
178  read_reg(OUTX_L_REG_M);
179  read_reg(OUTX_H_REG_M);
180  read_reg(OUTY_L_REG_M);
181  read_reg(OUTY_H_REG_M);
182  read_reg(OUTZ_L_REG_M);
183  read_reg(OUTZ_H_REG_M);
184 
185  uint16_t OUTX_ST = 0;
186  uint16_t OUTY_ST = 0;
187  uint16_t OUTZ_ST = 0;
188 
189  // Read the output registers after checking the Zyxda bit 50 times
190  // average x, y, z
191  for (int i = 0; i < 50; i++) {
192 
193  status_m = read_reg(STATUS_REG_M);
194 
195  OUTX_NOST += read_reg(OUTX_L_REG_M) + (read_reg(OUTX_H_REG_M) << 8);
196  OUTY_NOST += read_reg(OUTY_L_REG_M) + (read_reg(OUTY_H_REG_M) << 8);
197  OUTZ_NOST += read_reg(OUTZ_L_REG_M) + (read_reg(OUTZ_H_REG_M) << 8);
198  }
199 
200  const uint16_t abs_x = abs(OUTX_ST - OUTX_NOST);
201  const uint16_t abs_y = abs(OUTY_ST - OUTY_NOST);
202  const uint16_t abs_z = abs(OUTZ_ST - OUTZ_NOST);
203 
204  // TODO: proper ranges?
205  const bool x_valid = (abs_x > 0 && abs_x < UINT16_MAX);
206  const bool y_valid = (abs_y > 0 && abs_y < UINT16_MAX);
207  const bool z_valid = (abs_z > 0 && abs_z < UINT16_MAX);
208 
209  if (!x_valid || !y_valid || !z_valid) {
210 
211  PX4_ERR("self-test failed");
212 
213  PX4_INFO("STATUS_M: %X", status_m);
214  PX4_INFO("ABS(OUTX_ST - OUTX_NOST) = %d", abs_x);
215  PX4_INFO("ABS(OUTY_ST - OUTY_NOST) = %d", abs_y);
216  PX4_INFO("ABS(OUTZ_ST - OUTZ_NOST) = %d", abs_z);
217  }
218 
219  // disable self test
220  write_reg(CFG_REG_C_M, 0x10);
221 
222  // Idle mode
223  write_reg(CFG_REG_A_M, 0x03);
224 
225  return true;
226 }

241 int
242 LSM303AGR::ioctl(struct file *filp, int cmd, unsigned long arg)
243 {
244  switch (cmd) {
245  case SENSORIOCSPOLLRATE: {
246  switch (arg) {
247 
248  /* switching to manual polling */
249  case SENSOR_POLLRATE_MANUAL:
250  stop();
251  _measure_ticks = 0;
252  return OK;
253 
254  /* external signalling (DRDY) not supported */
255  case SENSOR_POLLRATE_EXTERNAL:
256 
257  /* zero would be bad */
258  case 0:
259  return -EINVAL;
260 
261  /* set default/max polling rate */
262  case SENSOR_POLLRATE_MAX:
263  case SENSOR_POLLRATE_DEFAULT: {
264  /* do we need to start internal polling? */
265  bool want_start = (_measure_ticks == 0);
266 
267  /* set interval for next measurement to minimum legal value */
268  _measure_ticks = USEC2TICK(CONVERSION_INTERVAL);
269 
270  /* if we need to start the poll state machine, do it */
271  if (want_start) {
272  start();
273  }
274 
275  return OK;
276  }
277 
278  /* adjust to a legal polling interval in Hz */
279  default: {
280  /* do we need to start internal polling? */
281  bool want_start = (_measure_ticks == 0);
282 
283  /* convert hz to tick interval via microseconds */
284  unsigned ticks = USEC2TICK(1000000 / arg);
285 
286  /* check against maximum rate */
287  if (ticks < USEC2TICK(CONVERSION_INTERVAL)) {
288  return -EINVAL;
289  }
290 
291  /* update interval for next measurement */
292  _measure_ticks = ticks;
293 
294  /* if we need to start the poll state machine, do it */
295  if (want_start) {
296  start();
297  }
298 
299  return OK;
300  }
301  }
302  }
303 
304  case SENSORIOCGPOLLRATE:
305  if (_measure_ticks == 0) {
306  return SENSOR_POLLRATE_MANUAL;
307  }
308 
309  return 1000000 / TICK2USEC(_measure_ticks);
310 
311  case SENSORIOCRESET:
312  return reset();
313 
314  case MAGIOCSSAMPLERATE:
315  /* same as pollrate because device is in single measurement mode*/
316  return ioctl(filp, SENSORIOCSPOLLRATE, arg);
317 
318  case MAGIOCGSAMPLERATE:
319  /* same as pollrate because device is in single measurement mode*/
320  return 1000000 / TICK2USEC(_measure_ticks);
321 
322  case MAGIOCSSCALE:
323  /* set new scale factors */
324  memcpy(&_mag_scale, (struct mag_calibration_s *)arg, sizeof(_mag_scale));
325 
326  return 0;
327 
328  case MAGIOCGSCALE:
329  /* copy out scale factors */
330  memcpy((struct mag_calibration_s *)arg, &_mag_scale, sizeof(_mag_scale));
331  return 0;
332 
333  case MAGIOCGEXTERNAL:
334  return external();
335 
336  default:
337  /* give it to the superclass */
338  return CDev::ioctl(filp, cmd, arg);
339  }
340 }

394 void
395 LSM303AGR::cycle()
396 {
397  if (_measure_ticks == 0) {
398  return;
399  }
400 
401  /* collection phase? */
402  if (_collect_phase) {
403 
404  /* perform collection */
405  if (OK != collect()) {
406  DEVICE_DEBUG("collection error");
407  /* restart the measurement state machine */
408  start();
409  return;
410  }
411 
412  /* next phase is measurement */
413  _collect_phase = false;
414 
415  /*
416  * Is there a collect->measure gap?
417  */
418  if (_measure_ticks > USEC2TICK(CONVERSION_INTERVAL)) {
419 
420  /* schedule a fresh cycle call when we are ready to measure again */
421  work_queue(HPWORK,
422  &_work,
423  (worker_t)&LSM303AGR::cycle_trampoline,
424  this,
425  _measure_ticks - USEC2TICK(CONVERSION_INTERVAL));
426 
427  return;
428  }
429  }
430 
431  /* measurement phase */
432  measure();
433 
434  /* next phase is collection */
435  _collect_phase = true;
436 
437  if (_measure_ticks > 0) {
438  /* schedule a fresh cycle call when the measurement is done */
439  work_queue(HPWORK, &_work, (worker_t)&LSM303AGR::cycle_trampoline, this, USEC2TICK(CONVERSION_INTERVAL));
440  }
441 }

452 int
453 LSM303AGR::collect()
454 {
455  const uint8_t status = read_reg(STATUS_REG_M);
456 
457  // only publish new data
458  if (status & STATUS_REG_M_Zyxda) {
459  /* start the performance counter */
460  perf_begin(_mag_sample_perf);
461 
462  mag_report mag_report = {};
463  mag_report.timestamp = hrt_absolute_time();
464 
465  // switch to right hand coordinate system in place
466  mag_report.x_raw = read_reg(OUTX_L_REG_M) + (read_reg(OUTX_H_REG_M) << 8);
467  mag_report.y_raw = read_reg(OUTY_L_REG_M) + (read_reg(OUTY_H_REG_M) << 8);
468  mag_report.z_raw = -(read_reg(OUTZ_L_REG_M) + (read_reg(OUTZ_H_REG_M) << 8));
469 
470  float xraw_f = mag_report.x_raw;
471  float yraw_f = mag_report.y_raw;
472  float zraw_f = mag_report.z_raw;
473 
474  /* apply user specified rotation */
475  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
476 
477  mag_report.x = ((xraw_f * _mag_range_scale) - _mag_scale.x_offset) * _mag_scale.x_scale;
478  mag_report.y = ((yraw_f * _mag_range_scale) - _mag_scale.y_offset) * _mag_scale.y_scale;
479  mag_report.z = ((zraw_f * _mag_range_scale) - _mag_scale.z_offset) * _mag_scale.z_scale;
480  mag_report.scaling = _mag_range_scale;
481  mag_report.error_count = perf_event_count(_bad_registers) + perf_event_count(_bad_values);
482 
483  /* remember the temperature. The datasheet isn't clear, but it
484  * seems to be a signed offset from 25 degrees C in units of 0.125C
485  */
486  //mag_report.temperature = 25 + (raw_mag_report.temperature * 0.125f);;
487  mag_report.device_id = _device_id.devid;
488  mag_report.is_external = external();
489 
490  if (!(_pub_blocked)) {
491 
492  if (_mag_topic != nullptr) {
493  /* publish it */
494  orb_publish(ORB_ID(sensor_mag), _mag_topic, &mag_report);
495 
496  } else {
497  _mag_topic = orb_advertise_multi(ORB_ID(sensor_mag), &mag_report, &_mag_orb_class_instance,
498  (mag_report.is_external) ? ORB_PRIO_HIGH : ORB_PRIO_MAX);
499 
500  if (_mag_topic == nullptr) {
501  DEVICE_DEBUG("ADVERT FAIL");
502  }
503  }
504  }
505 
506 
507  /* stop the perf counter */
508  perf_end(_mag_sample_perf);
509  }
510 
511  return OK;
512 }
