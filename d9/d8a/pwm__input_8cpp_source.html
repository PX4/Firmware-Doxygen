316 /*
317  * Initialise the timer we are going to use.
318  */
319 void PWMIN::_timer_init(void)
320 {
321  /* run with interrupts disabled in case the timer is already
322  * setup. We don't want it firing while we are doing the setup */
323  irqstate_t flags = px4_enter_critical_section();
324 
325  /* configure input pin */
326  px4_arch_configgpio(GPIO_PWM_IN);
327 
328  // XXX refactor this out of this driver
329  /* configure reset pin */
330  px4_arch_configgpio(GPIO_VDD_RANGEFINDER_EN);
331 
332  /* claim our interrupt vector */
333  irq_attach(PWMIN_TIMER_VECTOR, pwmin_tim_isr, NULL);
334 
335  /* Clear no bits, set timer enable bit.*/
336  modifyreg32(PWMIN_TIMER_POWER_REG, 0, PWMIN_TIMER_POWER_BIT);
337 
338  /* disable and configure the timer */
339  rCR1 = 0;
340  rCR2 = 0;
341  rSMCR = 0;
342  rDIER = DIER_PWMIN_A;
343  rCCER = 0; /* unlock CCMR* registers */
344  rCCMR1 = CCMR1_PWMIN;
345  rCCMR2 = CCMR2_PWMIN;
346  rSMCR = SMCR_PWMIN_1; /* Set up mode */
347  rSMCR = SMCR_PWMIN_2; /* Enable slave mode controller */
348  rCCER = CCER_PWMIN;
349  rDCR = 0;
350 
351  /* for simplicity scale by the clock in MHz. This gives us
352  * readings in microseconds which is typically what is needed
353  * for a PWM input driver */
354  uint32_t prescaler = PWMIN_TIMER_CLOCK / 1000000UL;
355 
356  /*
357  * define the clock speed. We want the highest possible clock
358  * speed that avoids overflows.
359  */
360  rPSC = prescaler - 1;
361 
362  /* run the full span of the counter. All timers can handle
363  * uint16 */
364  rARR = UINT16_MAX;
365 
366  /* generate an update event; reloads the counter, all registers */
367  rEGR = GTIM_EGR_UG;
368 
369  /* enable the timer */
370  rCR1 = GTIM_CR1_CEN;
371 
372  /* enable interrupts */
373  up_enable_irq(PWMIN_TIMER_VECTOR);
374 
375  px4_leave_critical_section(flags);
376 
377  _timer_started = true;
378 }

433 /*
434  * handle ioctl requests
435  */
436 int
437 PWMIN::ioctl(struct file *filp, int cmd, unsigned long arg)
438 {
439  switch (cmd) {
440  case SENSORIOCSQUEUEDEPTH: {
441  /* lower bound is mandatory, upper bound is a sanity check */
442  if ((arg < 1) || (arg > 500)) {
443  return -EINVAL;
444  }
445 
446  irqstate_t flags = px4_enter_critical_section();
447 
448  if (!_reports->resize(arg)) {
449  px4_leave_critical_section(flags);
450  return -ENOMEM;
451  }
452 
453  px4_leave_critical_section(flags);
454 
455  return OK;
456  }
457 
458  case SENSORIOCRESET:
459  /* user has asked for the timer to be reset. This may
460  * be needed if the pin was used for a different
461  * purpose (such as PWM output) */
462  _timer_init();
463  /* also reset the sensor */
464  hard_reset();
465  return OK;
466 
467  default:
468  /* give it to the superclass */
469  return CDev::ioctl(filp, cmd, arg);
470  }
471 }

474 /*
475  * read some samples from the device
476  */
477 ssize_t
478 PWMIN::read(struct file *filp, char *buffer, size_t buflen)
479 {
480  _last_read_time = hrt_absolute_time();
481 
482  unsigned count = buflen / sizeof(struct pwm_input_s);
483  struct pwm_input_s *buf = reinterpret_cast<struct pwm_input_s *>(buffer);
484  int ret = 0;
485 
486  /* buffer must be large enough */
487  if (count < 1) {
488  return -ENOSPC;
489  }
490 
491  while (count--) {
492  if (_reports->get(buf)) {
493  ret += sizeof(struct pwm_input_s);
494  buf++;
495  }
496  }
497 
498  /* if there was no data, warn the caller */
499  return ret ? ret : -EAGAIN;
500 }

502 /*
503  * publish some data from the ISR in the ring buffer
504  */
505 void PWMIN::publish(uint16_t status, uint32_t period, uint32_t pulse_width)
506 {
507  /* if we missed an edge, we have to give up */
508  if (status & SR_OVF_PWMIN) {
509  _error_count++;
510  return;
511  }
512 
513  _last_poll_time = hrt_absolute_time();
514 
515  struct pwm_input_s pwmin_report;
516  pwmin_report.timestamp = _last_poll_time;
517  pwmin_report.error_count = _error_count;
518  pwmin_report.period = period;
519  pwmin_report.pulse_width = pulse_width;
520 
521  _reports->force(&pwmin_report);
522 }

544 static int pwmin_tim_isr(int irq, void *context, void *arg)
545 {
546  uint16_t status = rSR;
547  uint32_t period = rCCR_PWMIN_A;
548  uint32_t pulse_width = rCCR_PWMIN_B;
549 
550  /* ack the interrupts we just read */
551  rSR = 0;
552 
553  if (g_dev != nullptr) {
554  g_dev->publish(status, period, pulse_width);
555  }
556 
557  return OK;
558 }
