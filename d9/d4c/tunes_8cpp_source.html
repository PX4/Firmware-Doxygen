88 int Tunes::set_control(const tune_control_s &tune_control)
89 {
90  // Sanity check
91  if (tune_control.tune_id >= _default_tunes_size) {
92  return -EINVAL;
93  }
94 
95  // Accept new tune ?
96  if (_repeat || // Repeated tunes can always be interrupted
97  _tune == nullptr || // No tune is currently being played
98  tune_control.tune_override || // Override interrupts everything
99  tune_control.tune_id == static_cast<int>(TuneID::STARTUP) ||
100  tune_control.tune_id == static_cast<int>(TuneID::ERROR_TUNE) ||
101  tune_control.tune_id == static_cast<int>(TuneID::NOTIFY_POSITIVE) ||
102  tune_control.tune_id == static_cast<int>(TuneID::NOTIFY_NEUTRAL) ||
103  tune_control.tune_id == static_cast<int>(TuneID::NOTIFY_NEGATIVE)) {
104 
105  // Reset repeat flag. Can jump to true again while tune is being parsed later
106  _repeat = false;
107 
108  // Reset octave, tempo etc.
109  reset(_repeat);
110 
111  // Strength will remain valid for the entire tune, unless interrupted.
112  if ((unsigned)tune_control.strength <= TUNE_MAX_STRENGTH) {
113  _strength = (unsigned)tune_control.strength;
114 
115  } else {
116  _strength = TUNE_MAX_STRENGTH;
117  }
118 
119  // Special treatment for custom tunes
120  if (tune_control.tune_id == static_cast<int>(TuneID::CUSTOM)) {
121  _using_custom_msg = true;
122  _frequency = (unsigned)tune_control.frequency;
123  _duration = (unsigned)tune_control.duration;
124  _silence = (unsigned)tune_control.silence;
125 
126  } else {
127  _using_custom_msg = false;
128  _tune = _default_tunes[tune_control.tune_id];
129  _tune_start_ptr = _default_tunes[tune_control.tune_id];
130  _next = _tune;
131  }
132  }
133 
134  return OK;
135 }

171 int Tunes::get_next_tune(unsigned &frequency, unsigned &duration,
172  unsigned &silence)
173 {
174  // Return the vaules for frequency and duration if the custom msg was received
175  if (_using_custom_msg) {
176  _using_custom_msg = false;
177  frequency = _frequency;
178  duration = _duration;
179  silence = _silence;
180  return TUNE_STOP;
181  }
182 
183  // make sure we still have a tune
184  if ((_next == nullptr) || (_tune == nullptr)) {
185  return TUNE_ERROR;
186  }
187 
188  // parse characters out of the string until we have resolved a note
189  unsigned note = 0;
190  unsigned note_length = _note_length;
191 
192  while (note == 0) {
193  // we always need at least one character from the string
194  int c = next_char();
195 
196  if (c == 0) {
197  silence = 0;
198  goto tune_end;
199  }
200 
201  _next++;
202 
203  switch (c) {
204  case 'L': // select note length
205  _note_length = next_number();
206 
207  if (_note_length < 1) {
208  goto tune_error;
209  }
210 
211  break;
212 
213  case 'O': // select octave
214  _octave = next_number();
215 
216  if (_octave > 6) {
217  _octave = 6;
218  }
219 
220  break;
221 
222  case '<': // decrease octave
223  if (_octave > 0) {
224  _octave--;
225  }
226 
227  break;
228 
229  case '>': // increase octave
230  if (_octave < 6) {
231  _octave++;
232  }
233 
234  break;
235 
236  case 'M': // select inter-note gap
237  c = next_char();
238 
239  if (c == 0) {
240  goto tune_error;
241  }
242 
243  _next++;
244 
245  switch (c) {
246  case 'N':
247  _note_mode = NoteMode::NORMAL;
248  break;
249 
250  case 'L':
251  _note_mode = NoteMode::LEGATO;
252  break;
253 
254  case 'S':
255  _note_mode = NoteMode::STACCATO;
256  break;
257 
258  case 'F':
259  _repeat = false;
260  break;
261 
262  case 'B':
263  _repeat = true;
264  break;
265 
266  default:
267  goto tune_error;
268  }
269 
270  break;
271 
272  case 'P': // pause for a note length
273  frequency = 0;
274  duration = 0;
275  silence = rest_duration(next_number(), next_dots());
276  return TUNE_CONTINUE;
277 
278  case 'T': { // change tempo
279  unsigned nt = next_number();
280 
281  if ((nt >= 32) && (nt <= 255)) {
282  _tempo = nt;
283 
284  } else {
285  goto tune_error;
286  }
287 
288  break;
289  }
290 
291  case 'N': // play an arbitrary note
292  note = next_number();
293 
294  if (note > 84) {
295  goto tune_error;
296  }
297 
298  if (note == 0) {
299  // this is a rest - pause for the current note length
300  silence = rest_duration(_note_length, next_dots());
301  return TUNE_CONTINUE;
302  }
303 
304  break;
305 
306  case 'A'...'G': // play a note in the current octave
307  note = _note_tab[c - 'A'] + (_octave * 12) + 1;
308  c = next_char();
309 
310  switch (c) {
311  case '#': // up a semitone
312  case '+':
313  if (note < 84) {
314  note++;
315  }
316 
317  _next++;
318  break;
319 
320  case '-': // down a semitone
321  if (note > 1) {
322  note--;
323  }
324 
325  _next++;
326  break;
327 
328  default:
329  // 0 / no next char here is OK
330  break;
331  }
332 
333  // shorthand length notation
334  note_length = next_number();
335 
336  if (note_length == 0) {
337  note_length = _note_length;
338  }
339 
340  break;
341 
342  default:
343  goto tune_error;
344  }
345  }
346 
347  // compute the duration of the note and the following silence (if any)
348  duration = note_duration(silence, note_length, next_dots());
349 
350  // compute the note frequency
351  frequency = note_to_frequency(note);
352  return TUNE_CONTINUE;
353 
354  // tune looks bad (unexpected EOF, bad character, etc.)
355 tune_error:
356  // syslog(LOG_ERR, "tune error\n");
357  _repeat = false; // don't loop on error
358  reset(_repeat);
359  return TUNE_ERROR;
360  // stop (and potentially restart) the tune
361 tune_end:
362  // restore intial parameter
363  reset(_repeat);
364 
365  if (_repeat) {
366  return TUNE_CONTINUE;
367 
368  } else {
369  return TUNE_STOP;
370  }
371 }

379 unsigned Tunes::note_duration(unsigned &silence, unsigned note_length, unsigned dots) const
380 {
381  unsigned whole_note_period = BEAT_TIME_CONVERSION / _tempo;
382 
383  if (note_length == 0) {
384  note_length = 1;
385  }
386 
387  unsigned note_period = whole_note_period / note_length;
388 
389  switch (_note_mode) {
390  case NoteMode::NORMAL:
391  silence = note_period / 8;
392  break;
393 
394  case NoteMode::STACCATO:
395  silence = note_period / 4;
396  break;
397 
398  case NoteMode::LEGATO:
399  default:
400  silence = 0;
401  break;
402  }
403 
404  note_period -= silence;
405 
406  unsigned dot_extension = note_period / 2;
407 
408  while (dots--) {
409  note_period += dot_extension;
410  dot_extension /= 2;
411  }
412 
413  return note_period;
414 }
