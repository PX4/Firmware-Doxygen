329 void
330 CameraTrigger::update_intervalometer()
331 {
332 
333  // the actual intervalometer runs in interrupt context, so we only need to call
334  // control_intervalometer once on enabling/disabling trigger to schedule the calls.
335 
336  if (_trigger_enabled && !_trigger_paused) {
337  // schedule trigger on and off calls
338  hrt_call_every(&_engagecall, 0, (_interval * 1000),
339  (hrt_callout)&CameraTrigger::engage, this);
340 
341  // schedule trigger on and off calls
342  hrt_call_every(&_disengagecall, 0 + (_activation_time * 1000), (_interval * 1000),
343  (hrt_callout)&CameraTrigger::disengage, this);
344 
345  }

349 void
350 CameraTrigger::update_distance()
351 {
352 
353  if (_lpos_sub < 0) {
354  _lpos_sub = orb_subscribe(ORB_ID(vehicle_local_position));
355  }
356 
357  if (_turning_on) {
358  return;
359  }
360 
361  if (_trigger_enabled) {
362 
363  struct vehicle_local_position_s local = {};
364  orb_copy(ORB_ID(vehicle_local_position), _lpos_sub, &local);
365 
366  if (local.xy_valid) {
367 
368  // Initialize position if not done yet
369  matrix::Vector2f current_position(local.x, local.y);
370 
371  if (!_valid_position) {
372  // First time valid position, take first shot
373  _last_shoot_position = current_position;
374  _valid_position = local.xy_valid;
375  shoot_once();
376  }
377 
378  // Check that distance threshold is exceeded
379  if (matrix::Vector2f(_last_shoot_position - current_position).length() >= _distance) {
380  shoot_once();
381  _last_shoot_position = current_position;
382 
383  }
384  }
385  }

434 void
435 CameraTrigger::start()
436 {
437 
438  if ((_trigger_mode == TRIGGER_MODE_INTERVAL_ALWAYS_ON ||
439  _trigger_mode == TRIGGER_MODE_DISTANCE_ALWAYS_ON) &&
440  _camera_interface->has_power_control() &&
441  !_camera_interface->is_powered_on()) {
442 
443  // If in always-on mode and the interface supports it, enable power to the camera
444  toggle_power();
445  enable_keep_alive(true);
446 
447  } else {
448  enable_keep_alive(false);
449  }
450 
451  // enable immediately if configured that way
452  if (_trigger_mode == TRIGGER_MODE_INTERVAL_ALWAYS_ON) {
453  // enable and start triggering
454  _trigger_enabled = true;
455  update_intervalometer();
456 
457  } else if (_trigger_mode == TRIGGER_MODE_DISTANCE_ALWAYS_ON) {
458  // just enable, but do not fire. actual trigger is based on distance covered
459  _trigger_enabled = true;
460  }
461 
462  // start to monitor at high rate for trigger enable command
463  work_queue(LPWORK, &_work, (worker_t)&CameraTrigger::cycle_trampoline, this, USEC2TICK(1));
464

494 void
495 CameraTrigger::cycle_trampoline(void *arg)
496 {
497 
498  CameraTrigger *trig = reinterpret_cast<CameraTrigger *>(arg);
499 
500  // default loop polling interval
501  int poll_interval_usec = 5000;
502 
503  if (trig->_command_sub < 0) {
504  trig->_command_sub = orb_subscribe(ORB_ID(vehicle_command));
505  }
506 
507  bool updated = false;
508  orb_check(trig->_command_sub, &updated);
509 
510  struct vehicle_command_s cmd;
511  unsigned cmd_result = vehicle_command_s::VEHICLE_CMD_RESULT_TEMPORARILY_REJECTED;
512  bool need_ack = false;
513 
514  // this flag is set when the polling loop is slowed down to allow the camera to power on
515  trig->_turning_on = false;
516 
517  // these flags are used to detect state changes in the command loop
518  bool main_state = trig->_trigger_enabled;
519  bool pause_state = trig->_trigger_paused;
520 
521  // Command handling
522  if (updated) {
523 
524  orb_copy(ORB_ID(vehicle_command), trig->_command_sub, &cmd);
525 
526  if (cmd.command == vehicle_command_s::VEHICLE_CMD_DO_DIGICAM_CONTROL) {
527 
528  need_ack = true;
529 
530  if (commandParamToInt(cmd.param7) == 1) {
531  // test shots are not logged or forwarded to GCS for geotagging
532  trig->_test_shot = true;
533 
534  }
535 
536  if (commandParamToInt((float)cmd.param5) == 1) {
537  // Schedule shot
538  trig->_one_shot = true;
539 
540  }
541 
542  cmd_result = vehicle_command_s::VEHICLE_CMD_RESULT_ACCEPTED;
543 
544  } else if (cmd.command == vehicle_command_s::VEHICLE_CMD_DO_TRIGGER_CONTROL) {
545 
546  need_ack = true;
547 
548  if (commandParamToInt(cmd.param3) == 1) {
549  // pause triggger
550  trig->_trigger_paused = true;
551 
552  } else if (commandParamToInt(cmd.param3) == 0) {
553  trig->_trigger_paused = false;
554 
555  }
556 
557  if (commandParamToInt(cmd.param2) == 1) {
558  // reset trigger sequence
559  trig->_trigger_seq = 0;
560 
561  }
562 
563  if (commandParamToInt(cmd.param1) == 1) {
564  trig->_trigger_enabled = true;
565 
566  } else if (commandParamToInt(cmd.param1) == 0) {
567  trig->_trigger_enabled = false;
568 
569  }
570 
571  cmd_result = vehicle_command_s::VEHICLE_CMD_RESULT_ACCEPTED;
572 
573  } else if (cmd.command == vehicle_command_s::VEHICLE_CMD_DO_SET_CAM_TRIGG_DIST) {
574 
575  need_ack = true;
576 
577  /*
578  * TRANSITIONAL SUPPORT ADDED AS OF 11th MAY 2017 (v1.6 RELEASE)
579  */
580 
581  if (cmd.param1 > 0.0f) {
582  trig->_distance = cmd.param1;
583  param_set_no_notification(trig->_p_distance, &(trig->_distance));
584 
585  trig->_trigger_enabled = true;
586  trig->_trigger_paused = false;
587 
588  } else if (commandParamToInt(cmd.param1) == 0) {
589  trig->_trigger_paused = true;
590 
591  } else if (commandParamToInt(cmd.param1) == -1) {
592  trig->_trigger_enabled = false;
593  }
594 
595  // We can only control the shutter integration time of the camera in GPIO mode (for now)
596  if (cmd.param2 > 0.0f) {
597  if (trig->_camera_interface_mode == CAMERA_INTERFACE_MODE_GPIO) {
598  trig->_activation_time = cmd.param2;
599  param_set_no_notification(trig->_p_activation_time, &(trig->_activation_time));
600  }
601  }
602 
603  // Trigger once immediately if param is set
604  if (cmd.param3 > 0.0f) {
605  // Schedule shot
606  trig->_one_shot = true;
607  }
608 
609  cmd_result = vehicle_command_s::VEHICLE_CMD_RESULT_ACCEPTED;
610 
611  } else if (cmd.command == vehicle_command_s::VEHICLE_CMD_DO_SET_CAM_TRIGG_INTERVAL) {
612 
613  need_ack = true;
614 
615  if (cmd.param1 > 0.0f) {
616  trig->_interval = cmd.param1;
617  param_set_no_notification(trig->_p_interval, &(trig->_interval));
618  }
619 
620  // We can only control the shutter integration time of the camera in GPIO mode
621  if (cmd.param2 > 0.0f) {
622  if (trig->_camera_interface_mode == CAMERA_INTERFACE_MODE_GPIO) {
623  trig->_activation_time = cmd.param2;
624  param_set_no_notification(trig->_p_activation_time, &(trig->_activation_time));
625  }
626  }
627 
628  cmd_result = vehicle_command_s::VEHICLE_CMD_RESULT_ACCEPTED;
629 
630  }
631 
632  }
633 
634  // State change handling
635  if ((main_state != trig->_trigger_enabled) ||
636  (pause_state != trig->_trigger_paused) ||
637  trig->_one_shot) {
638 
639  if (trig->_trigger_enabled || trig->_one_shot) { // Just got enabled via a command
640 
641  // If camera isn't already powered on, we enable power to it
642  if (!trig->_camera_interface->is_powered_on() &&
643  trig->_camera_interface->has_power_control()) {
644 
645  trig->toggle_power();
646  trig->enable_keep_alive(true);
647 
648  // Give the camera time to turn on before starting to send trigger signals
649  poll_interval_usec = 3000000;
650  trig->_turning_on = true;
651  }
652 
653  } else if (!trig->_trigger_enabled || trig->_trigger_paused) { // Just got disabled/paused via a command
654 
655  // Power off the camera if we are disabled
656  if (trig->_camera_interface->is_powered_on() &&
657  trig->_camera_interface->has_power_control() &&
658  !trig->_trigger_enabled) {
659 
660  trig->enable_keep_alive(false);
661  trig->toggle_power();
662  }
663 
664  // cancel all calls for both disabled and paused
665  hrt_cancel(&trig->_engagecall);
666  hrt_cancel(&trig->_disengagecall);
667 
668  // ensure that the pin is off
669  hrt_call_after(&trig->_disengagecall, 0,
670  (hrt_callout)&CameraTrigger::disengage, trig);
671 
672  // reset distance counter if needed
673  if (trig->_trigger_mode == TRIGGER_MODE_DISTANCE_ON_CMD ||
674  trig->_trigger_mode == TRIGGER_MODE_DISTANCE_ALWAYS_ON) {
675 
676  // this will force distance counter reinit on getting enabled/unpaused
677  trig->_valid_position = false;
678 
679  }
680 
681  }
682 
683  // only run on state changes, not every loop iteration
684  if (trig->_trigger_mode == TRIGGER_MODE_INTERVAL_ON_CMD) {
685 
686  // update intervalometer state and reset calls
687  trig->update_intervalometer();
688 
689  }
690 
691  }
692 
693  // run every loop iteration and trigger if needed
694  if (trig->_trigger_mode == TRIGGER_MODE_DISTANCE_ON_CMD ||
695  trig->_trigger_mode == TRIGGER_MODE_DISTANCE_ALWAYS_ON) {
696 
697  // update distance counter and trigger
698  trig->update_distance();
699 
700  }
701 
702  // One shot command-based capture handling
703  if (trig->_one_shot && !trig->_turning_on) {
704 
705  // One-shot trigger
706  trig->shoot_once();
707  trig->_one_shot = false;
708 
709  if (trig->_test_shot) {
710  trig->_test_shot = false;
711  }
712 
713  }
714 
715  // Command ACK handling
716  if (updated && need_ack) {
717  vehicle_command_ack_s command_ack = {};
718  command_ack.timestamp = hrt_absolute_time();
719  command_ack.command = cmd.command;
720  command_ack.result = (uint8_t)cmd_result;
721  command_ack.target_system = cmd.source_system;
722  command_ack.target_component = cmd.source_component;
723

745 
746  if (trig->_test_shot) {
