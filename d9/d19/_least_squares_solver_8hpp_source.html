44  for (size_t i = j; i < M; i++) {
45  normx += _A(i,j) * _A(i,j);
46  }
47  normx = sqrt(normx);
48  Type s = _A(j,j) > 0 ? Type(-1) : Type(1);
49  Type u1 = _A(j,j) - s*normx;
50  // prevent divide by zero
51  // also covers u1. normx is never negative
