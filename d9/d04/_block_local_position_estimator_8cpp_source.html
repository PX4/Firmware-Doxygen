737 void BlockLocalPositionEstimator::updateSSParams()
738 {
739  // input noise covariance matrix
740  _R.setZero();
741  _R(U_ax, U_ax) = _accel_xy_stddev.get() * _accel_xy_stddev.get();
742  _R(U_ay, U_ay) = _accel_xy_stddev.get() * _accel_xy_stddev.get();
743  _R(U_az, U_az) = _accel_z_stddev.get() * _accel_z_stddev.get();
744 
745  // process noise power matrix
746  _Q.setZero();
747  float pn_p_sq = _pn_p_noise_density.get() * _pn_p_noise_density.get();
748  float pn_v_sq = _pn_v_noise_density.get() * _pn_v_noise_density.get();
749  _Q(X_x, X_x) = pn_p_sq;
750  _Q(X_y, X_y) = pn_p_sq;
751  _Q(X_z, X_z) = pn_p_sq;
752  _Q(X_vx, X_vx) = pn_v_sq;
753  _Q(X_vy, X_vy) = pn_v_sq;
754  _Q(X_vz, X_vz) = pn_v_sq;
755 
756  // technically, the noise is in the body frame,
757  // but the components are all the same, so
758  // ignoring for now
759  float pn_b_sq = _pn_b_noise_density.get() * _pn_b_noise_density.get();
760  _Q(X_bx, X_bx) = pn_b_sq;
761  _Q(X_by, X_by) = pn_b_sq;
762  _Q(X_bz, X_bz) = pn_b_sq;
763 
764  // terrain random walk noise ((m/s)/sqrt(hz)), scales with velocity
765  float pn_t_noise_density =
766  _pn_t_noise_density.get() +
767  (_t_max_grade.get() / 100.0f) * sqrtf(_x(X_vx) * _x(X_vx) + _x(X_vy) * _x(X_vy));
768  _Q(X_tz, X_tz) = pn_t_noise_density * pn_t_noise_density;
769 }

771 void BlockLocalPositionEstimator::predict()
772 {
773  // get acceleration
774  matrix::Quatf q(&_sub_att.get().q[0]);
775  _eul = matrix::Euler<float>(q);
776  _R_att = matrix::Dcm<float>(q);
777  Vector3f a(_sub_sensor.get().accelerometer_m_s2);
778  // note, bias is removed in dynamics function
779  _u = _R_att * a;
780  _u(U_az) += CONSTANTS_ONE_G; // add g
781 
782  // update state space based on new states
783  updateSSStates();
784 
785  // continuous time kalman filter prediction
786  // integrate runge kutta 4th order
787  // TODO move rk4 algorithm to matrixlib
788  // https://en.wikipedia.org/wiki/Runge%E2%80%93Kutta_methods
789  float h = getDt();
790  Vector<float, n_x> k1, k2, k3, k4;
791  k1 = dynamics(0, _x, _u);
792  k2 = dynamics(h / 2, _x + k1 * h / 2, _u);
793  k3 = dynamics(h / 2, _x + k2 * h / 2, _u);
794  k4 = dynamics(h, _x + k3 * h, _u);
795  Vector<float, n_x> dx = (k1 + k2 * 2 + k3 * 2 + k4) * (h / 6);
796 
797  // don't integrate position if no valid xy data
798  if (!(_estimatorInitialized & EST_XY)) {
799  dx(X_x) = 0;
800  dx(X_vx) = 0;
801  dx(X_y) = 0;
802  dx(X_vy) = 0;
803  }
804 
805  // don't integrate z if no valid z data
806  if (!(_estimatorInitialized & EST_Z)) {
807  dx(X_z) = 0;
808  }
809 
810  // don't integrate tz if no valid tz data
811  if (!(_estimatorInitialized & EST_TZ)) {
812  dx(X_tz) = 0;
813  }
814 
815  // saturate bias
816  float bx = dx(X_bx) + _x(X_bx);
817  float by = dx(X_by) + _x(X_by);
818  float bz = dx(X_bz) + _x(X_bz);
819 
820  if (std::abs(bx) > BIAS_MAX) {
821  bx = BIAS_MAX * bx / std::abs(bx);
822  dx(X_bx) = bx - _x(X_bx);
823  }
824 
825  if (std::abs(by) > BIAS_MAX) {
826  by = BIAS_MAX * by / std::abs(by);
827  dx(X_by) = by - _x(X_by);
828  }
829 
830  if (std::abs(bz) > BIAS_MAX) {
831  bz = BIAS_MAX * bz / std::abs(bz);
832  dx(X_bz) = bz - _x(X_bz);
833  }
834 
835  // propagate
836  _x += dx;
837  Matrix<float, n_x, n_x> dP = (_A * _P + _P * _A.transpose() +
838  _B * _R * _B.transpose() + _Q) * getDt();
839 
840  // covariance propagation logic
841  for (int i = 0; i < n_x; i++) {
842  if (_P(i, i) > P_MAX) {
843  // if diagonal element greater than max, stop propagating
844  dP(i, i) = 0;
845 
846  for (int j = 0; j < n_x; j++) {
847  dP(i, j) = 0;
848  dP(j, i) = 0;
849  }
850  }
851  }
852 
853  _P += dP;
854  _xLowPass.update(_x);
855  _aglLowPass.update(agl());
856 }
