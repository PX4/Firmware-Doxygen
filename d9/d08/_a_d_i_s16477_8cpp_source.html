301 int
302 ADIS16477::ioctl(struct file *filp, int cmd, unsigned long arg)
303 {
304  switch (cmd) {
305  case SENSORIOCRESET:
306  return reset();
307 
308  case SENSORIOCSPOLLRATE: {
309  switch (arg) {
310 
311  /* switching to manual polling */
312  case SENSOR_POLLRATE_MANUAL:
313  stop();
314  _call_interval = 0;
315  return OK;
316 
317  /* external signalling not supported */
318  case SENSOR_POLLRATE_EXTERNAL:
319 
320  /* zero would be bad */
321  case 0:
322  return -EINVAL;
323 
324  /* set default/max polling rate */
325  case SENSOR_POLLRATE_MAX:
326  return ioctl(filp, SENSORIOCSPOLLRATE, 1000);
327 
328  case SENSOR_POLLRATE_DEFAULT:
329  return ioctl(filp, SENSORIOCSPOLLRATE, ADIS16477_ACCEL_DEFAULT_RATE);
330 
331  /* adjust to a legal polling interval in Hz */
332  default: {
333  /* do we need to start internal polling? */
334  bool want_start = (_call_interval == 0);
335 
336  /* convert hz to hrt interval via microseconds */
337  unsigned ticks = 1000000 / arg;
338 
339  /* check against maximum sane rate */
340  if (ticks < 1000) {
341  return -EINVAL;
342  }
343 
344  // adjust filters
345  float cutoff_freq_hz = _accel_filter_x.get_cutoff_freq();
346  float sample_rate = 1.0e6f / ticks;
347  _accel_filter_x.set_cutoff_frequency(sample_rate, cutoff_freq_hz);
348  _accel_filter_y.set_cutoff_frequency(sample_rate, cutoff_freq_hz);
349  _accel_filter_z.set_cutoff_frequency(sample_rate, cutoff_freq_hz);
350 
351  float cutoff_freq_hz_gyro = _gyro_filter_x.get_cutoff_freq();
352  _gyro_filter_x.set_cutoff_frequency(sample_rate, cutoff_freq_hz_gyro);
353  _gyro_filter_y.set_cutoff_frequency(sample_rate, cutoff_freq_hz_gyro);
354  _gyro_filter_z.set_cutoff_frequency(sample_rate, cutoff_freq_hz_gyro);
355 
356  /* update interval for next measurement */
357  /* XXX this is a bit shady, but no other way to adjust... */
358  _call.period = _call_interval = ticks;
359 
360  /* if we need to start the poll state machine, do it */
361  if (want_start) {
362  start();
363  }
364 
365  return OK;
366  }
367  }
368  }
369 
370  case SENSORIOCGPOLLRATE:
371  if (_call_interval == 0) {
372  return SENSOR_POLLRATE_MANUAL;
373  }
374 
375  return 1000000 / _call_interval;
376 
377  case ACCELIOCGSAMPLERATE:
378  return _sample_rate;
379 
380  case ACCELIOCSSAMPLERATE:
381  _set_sample_rate(arg);
382  return OK;
383 
384  case ACCELIOCSSCALE: {
385  /* copy scale, but only if off by a few percent */
386  struct accel_calibration_s *s = (struct accel_calibration_s *) arg;
387  float sum = s->x_scale + s->y_scale + s->z_scale;
388 
389  if (sum > 2.0f && sum < 4.0f) {
390  memcpy(&_accel_scale, s, sizeof(_accel_scale));
391  return OK;
392 
393  } else {
394  return -EINVAL;
395  }
396  }
397 
398  case ACCELIOCGSCALE:
399  /* copy scale out */
400  memcpy((struct accel_calibration_s *) arg, &_accel_scale, sizeof(_accel_scale));
401  return OK;
402 
403  case ACCELIOCSRANGE:
404  return -EINVAL;
405 
406  case ACCELIOCGRANGE:
407  return (unsigned long)((_accel_range_m_s2) / CONSTANTS_ONE_G + 0.5f);
408 
409  default:
410  /* give it to the superclass */
411  return SPI::ioctl(filp, cmd, arg);
412  }

520 int
521 ADIS16477::measure()
522 {
523  perf_begin(_sample_perf);
524 
525  // Fetch the full set of measurements from the ADIS16477 in one pass (burst read).
526  ADISReport adis_report;
527  adis_report.cmd = ((GLOB_CMD | DIR_READ) << 8) & 0xff00;
528 
529  // ADIS16477 burst report should be 176 bits
530  static_assert(sizeof(adis_report) == (176 / 8), "ADIS16477 report not 176 bits");
531 
532  if (OK != transferhword((uint16_t *)&adis_report, ((uint16_t *)&adis_report), sizeof(adis_report) / sizeof(uint16_t))) {
533  perf_end(_sample_perf);
534  return -EIO;
535  }
536 
537  // Calculate checksum and compare
538 
539  // Checksum = DIAG_STAT, Bits[15:8] + DIAG_STAT, Bits[7:0] +
540  // X_GYRO_OUT, Bits[15:8] + X_GYRO_OUT, Bits[7:0] +
541  // Y_GYRO_OUT, Bits[15:8] + Y_GYRO_OUT, Bits[7:0] +
542  // Z_GYRO_OUT, Bits[15:8] + Z_GYRO_OUT, Bits[7:0] +
543  // X_ACCL_OUT, Bits[15:8] + X_ACCL_OUT, Bits[7:0] +
544  // Y_ACCL_OUT, Bits[15:8] + Y_ACCL_OUT, Bits[7:0] +
545  // Z_ACCL_OUT, Bits[15:8] + Z_ACCL_OUT, Bits[7:0] +
546  // TEMP_OUT, Bits[15:8] + TEMP_OUT, Bits[7:0] +
547  // DATA_CNTR, Bits[15:8] + DATA_CNTR, Bits[7:0]
548  uint8_t *checksum_helper = (uint8_t *)&adis_report.diag_stat;
549 
550  uint8_t checksum = 0;
551 
552  for (int i = 0; i < 18; i++) {
553  checksum += checksum_helper[i];
554  }
555 
556  if (adis_report.checksum != checksum) {
557  DEVICE_DEBUG("adis_report.checksum: %X vs calculated: %X", adis_report.checksum, checksum);
558  perf_event_count(_bad_transfers);
559  perf_end(_sample_perf);
560  return -EIO;
561  }
562 
563  // Check all Status/Error Flag Indicators (DIAG_STAT)
564  if (adis_report.diag_stat != 0) {
565  perf_event_count(_bad_transfers);
566  perf_end(_sample_perf);
567  return -EIO;
568  }
569 
570  publish_gyro(adis_report);
571  publish_accel(adis_report);
572 
573  /* stop measuring */
574  perf_end(_sample_perf);
575  return OK;
576 }

578 bool
579 ADIS16477::publish_accel(const ADISReport &report)
580 {
581  accel_report arb = {};
582  arb.timestamp = hrt_absolute_time();
583  arb.device_id = _device_id.devid;
584  arb.error_count = perf_event_count(_bad_transfers);
585 
586  // raw sensor readings
587  arb.x_raw = report.accel_x;
588  arb.y_raw = report.accel_y;
589  arb.z_raw = report.accel_z;
590  arb.scaling = _accel_range_scale;
591 
592  float xraw_f = report.accel_x * _accel_range_scale;
593  float yraw_f = report.accel_y * _accel_range_scale;
594  float zraw_f = report.accel_z * _accel_range_scale;
595 
596  // apply user specified rotation
597  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
598 
599  float x_in_new = (xraw_f - _accel_scale.x_offset) * _accel_scale.x_scale;
600  float y_in_new = (yraw_f - _accel_scale.y_offset) * _accel_scale.y_scale;
601  float z_in_new = (zraw_f - _accel_scale.z_offset) * _accel_scale.z_scale;
602 
603  arb.x = _accel_filter_x.apply(x_in_new);
604  arb.y = _accel_filter_y.apply(y_in_new);
605  arb.z = _accel_filter_z.apply(z_in_new);
606 
607  matrix::Vector3f aval(x_in_new, y_in_new, z_in_new);
608  matrix::Vector3f aval_integrated;
609 
610  bool accel_notify = _accel_int.put(arb.timestamp, aval, aval_integrated, arb.integral_dt);
611  arb.x_integral = aval_integrated(0);
612  arb.y_integral = aval_integrated(1);
613  arb.z_integral = aval_integrated(2);
614 
615  /* Temperature report: */
616  // temperature 1 LSB = 0.1°C
617  arb.temperature = report.temp * 0.1;
618 
619  if (accel_notify) {
620  orb_publish(ORB_ID(sensor_accel), _accel_topic, &arb);
621  }
622 
623  return accel_notify;
624 }

653  grb.y = _gyro_filter_y.apply(y_gyro_in_new);
654  grb.z = _gyro_filter_z.apply(z_gyro_in_new);

670  }
671
