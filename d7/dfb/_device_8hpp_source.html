167  /*
168  broken out device elements. The bitfields are used to keep
169  the overall value small enough to fit in a float accurately,
170  which makes it possible to transport over the MAVLink
171  parameter protocol without loss of information.
172  */
173  struct DeviceStructure {
174  enum DeviceBusType bus_type : 3;
175  uint8_t bus: 5; // which instance of the bus type
176  uint8_t address; // address on the bus (eg. I2C address)
177  uint8_t devtype; // device class specific device type
178  };
