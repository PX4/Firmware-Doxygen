146 static void led_pwm_timer_set_rate(unsigned timer, unsigned rate)
147 {
148 
149  irqstate_t flags = px4_enter_critical_section();
150 
151  uint32_t save = rSC(timer);
152  rSC(timer) = save & ~(FTM_SC_CLKS_MASK);
153 
154  /* configure the timer to update at the desired rate */
155  rMOD(timer) = (LED_PWM_FREQ / rate) - 1;
156  rSC(timer) = save;
157 
158  px4_leave_critical_section(flags);
159 }

175 static void
176 led_pwm_timer_init(unsigned timer)
177 {
178  /* valid Timer */
179 
180  if (led_pwm_timers[timer].base != 0) {
181 
182  /* enable the timer clock before we try to talk to it */
183 
184  uint32_t regval = _REG(led_pwm_timers[timer].clock_register);
185  regval |= led_pwm_timers[timer].clock_bit;
186  _REG(led_pwm_timers[timer].clock_register) = regval;
187 
188  /* disable and configure the timer */
189 
190  rSC(timer) = FTM_SC_CLKS_NONE;
191  rCNT(timer) = 0;
192 
193  rMODE(timer) = 0;
194  rSYNCONF(timer) = (FTM_SYNCONF_SYNCMODE | FTM_SYNCONF_SWWRBUF | FTM_SYNCONF_SWRSTCNT);
195 
196  /* Set to run in debug mode */
197 
198  rCONF(timer) |= FTM_CONF_BDMMODE_MASK;
199 
200  /* enable the timer */
201 
202  led_pwm_timer_set_PWM_mode(timer);
203 
204  /*
205  * Note we do the Standard PWM Out init here
206  * default to updating at 50Hz
207  */
208 
209  led_pwm_timer_set_rate(timer, 50);
210  }

227 led_pwm_channel_init(unsigned channel)
228 {
229  /* Only initialize used channels */
230 
231  if (led_pwm_channels[channel].timer_channel) {
232  unsigned timer = led_pwm_channels[channel].timer_index;
233 
234  irqstate_t flags = px4_enter_critical_section();
235 
236  /* configure the GPIO first */
237 
238  px4_arch_configgpio(led_pwm_channels[channel].gpio_out);
239 
240  /* configure the channel */
241 
242  uint32_t chan = led_pwm_channels[channel].timer_channel - 1;
243 
244  uint16_t rvalue = REG(timer, KINETIS_FTM_CSC_OFFSET(chan));
245  rvalue &= ~CnSC_RESET;
246  rvalue |= CnSC_PWMOUT_INIT;
247  REG(timer, KINETIS_FTM_CSC_OFFSET(chan)) = rvalue;
248  REG(timer, KINETIS_FTM_CV_OFFSET(0)) = 0;
249  px4_leave_critical_section(flags);

253 int
254 led_pwm_servo_set(unsigned channel, uint8_t cvalue)
255 {
256  if (channel >= arraySize(led_pwm_channels)) {
257  return -1;
258  }
259 
260  unsigned timer = led_pwm_channels[channel].timer_index;
261 
262  /* test timer for validity */
263  if ((led_pwm_timers[timer].base == 0) ||
264  (led_pwm_channels[channel].gpio_out == 0)) {
265  return -1;
266  }
267 
268  unsigned period = led_pwm_timer_get_period(timer);
269 
270  unsigned value = (unsigned)cvalue * period / 255;
271 
272  /* configure the channel */
273  if (value > 0) {
274  value--;
275  }
276 
277  irqstate_t flags = px4_enter_critical_section();
278  uint32_t save = rSC(timer);
279  rSC(timer) = save & ~(FTM_SC_CLKS_MASK);
280  REG(timer, KINETIS_FTM_CV_OFFSET(led_pwm_channels[channel].timer_channel - 1)) = value;
281  rSC(timer) = save;
282  px4_leave_critical_section(flags);
283 
284  return 0;
285 }
286 unsigned
287 led_pwm_servo_get(unsigned channel)
288 {
289  if (channel >= 3) {
290  return 0;
291  }
292 
293  unsigned timer = led_pwm_channels[channel].timer_index;
294  servo_position_t value = 0;
295 
296  /* test timer for validity */
297  if ((led_pwm_timers[timer].base == 0) ||
298  (led_pwm_channels[channel].timer_channel == 0)) {
299  return value;
300  }
301 
302  value = REG(timer, KINETIS_FTM_CV_OFFSET(led_pwm_channels[channel].timer_channel - 1));
303  unsigned period = led_pwm_timer_get_period(timer);
304  return ((value + 1) * 255 / period);
