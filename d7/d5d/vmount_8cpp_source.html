183 static int vmount_thread_main(int argc, char *argv[])
184 {
185  ParameterHandles param_handles;
186  Parameters params;
187  OutputConfig output_config;
188  ThreadData thread_data;
189  memset(&params, 0, sizeof(params));
190 
191 
192  InputTest *test_input = nullptr;
193 
194 #ifdef __PX4_NUTTX
195  /* the NuttX optarg handler does not
196  * ignore argv[0] like the POSIX handler
197  * does, nor does it deal with non-flag
198  * verbs well. So we Remove the application
199  * name and the verb.
200  */
201  argc -= 1;
202  argv += 1;
203 #endif
204 
205  if (argc > 0 && !strcmp(argv[0], "test")) {
206  PX4_INFO("Starting in test mode");
207 
208  const char *axis_names[3] = {"roll", "pitch", "yaw"};
209  float angles[3] = { 0.f, 0.f, 0.f };
210 
211  if (argc == 3) {
212  bool found_axis = false;
213 
214  for (int i = 0 ; i < 3; ++i) {
215  if (!strcmp(argv[1], axis_names[i])) {
216  long angle_deg = strtol(argv[2], nullptr, 0);
217  angles[i] = (float)angle_deg;
218  found_axis = true;
219  }
220  }
221 
222  if (!found_axis) {
223  usage();
224  return -1;
225  }
226 
227  test_input = new InputTest(angles[0], angles[1], angles[2]);
228 
229  if (!test_input) {
230  PX4_ERR("memory allocation failed");
231  return -1;
232  }
233 
234  } else {
235  usage();
236  return -1;
237  }
238  }
239 
240  if (!get_params(param_handles, params)) {
241  PX4_ERR("could not get mount parameters!");
242  delete test_input;
243  return -1;
244  }
245 
246  int parameter_update_sub = orb_subscribe(ORB_ID(parameter_update));
247  thread_running = true;
248  ControlData *control_data = nullptr;
249  g_thread_data = &thread_data;
250 
251  int last_active = 0;
252 
253  while (!thread_should_exit) {
254 
255  if (!thread_data.input_objs[0] && (params.mnt_mode_in >= 0 || test_input)) { //need to initialize
256 
257  output_config.gimbal_normal_mode_value = params.mnt_ob_norm_mode;
258  output_config.gimbal_retracted_mode_value = params.mnt_ob_lock_mode;
259  output_config.pitch_scale = 1.0f / ((params.mnt_range_pitch / 2.0f) * M_DEG_TO_RAD_F);
260  output_config.roll_scale = 1.0f / ((params.mnt_range_roll / 2.0f) * M_DEG_TO_RAD_F);
261  output_config.yaw_scale = 1.0f / ((params.mnt_range_yaw / 2.0f) * M_DEG_TO_RAD_F);
262  output_config.pitch_offset = params.mnt_off_pitch * M_DEG_TO_RAD_F;
263  output_config.roll_offset = params.mnt_off_roll * M_DEG_TO_RAD_F;
264  output_config.yaw_offset = params.mnt_off_yaw * M_DEG_TO_RAD_F;
265  output_config.mavlink_sys_id = params.mnt_mav_sysid;
266  output_config.mavlink_comp_id = params.mnt_mav_compid;
267 
268  bool alloc_failed = false;
269  thread_data.input_objs_len = 1;
270 
271  if (test_input) {
272  thread_data.input_objs[0] = test_input;
273 
274  } else {
275  switch (params.mnt_mode_in) {
276  case 0:
277 
278  // Automatic
279  thread_data.input_objs[0] = new InputMavlinkCmdMount(params.mnt_do_stab);
280  thread_data.input_objs[1] = new InputMavlinkROI();
281 
282  // RC is on purpose last here so that if there are any mavlink
283  // messages, they will take precedence over RC.
284  // This logic is done further below while update() is called.
285  thread_data.input_objs[2] = new InputRC(params.mnt_do_stab, params.mnt_man_roll, params.mnt_man_pitch, params.mnt_man_yaw);
286  thread_data.input_objs_len = 3;
287 
288  break;
289 
290  case 1: //RC
291  thread_data.input_objs[0] = new InputRC(params.mnt_do_stab, params.mnt_man_roll, params.mnt_man_pitch, params.mnt_man_yaw);
292  break;
293 
294  case 2: //MAVLINK_ROI
295  thread_data.input_objs[0] = new InputMavlinkROI();
296  break;
297 
298  case 3: //MAVLINK_DO_MOUNT
299  thread_data.input_objs[0] = new InputMavlinkCmdMount(params.mnt_do_stab);
300  break;
301 
302  default:
303  PX4_ERR("invalid input mode %i", params.mnt_mode_in);
304  break;
305  }
306  }
307 
308  for (int i = 0; i < thread_data.input_objs_len; ++i) {
309  if (!thread_data.input_objs[i]) {
310  alloc_failed = true;
311  }
312  }
313 
314  switch (params.mnt_mode_out) {
315  case 0: //AUX
316  thread_data.output_obj = new OutputRC(output_config);
317 
318  if (!thread_data.output_obj) { alloc_failed = true; }
319 
320  break;
321 
322  case 1: //MAVLINK
323  thread_data.output_obj = new OutputMavlink(output_config);
324 
325  if (!thread_data.output_obj) { alloc_failed = true; }
326 
327  break;
328 
329  default:
330  PX4_ERR("invalid output mode %i", params.mnt_mode_out);
331  thread_should_exit = true;
332  break;
333  }
334 
335  if (alloc_failed) {
336  thread_data.input_objs_len = 0;
337  PX4_ERR("memory allocation failed");
338  thread_should_exit = true;
339  }
340 
341  if (thread_should_exit) {
342  break;
343  }
344 
345  int ret = thread_data.output_obj->initialize();
346 
347  if (ret) {
348  PX4_ERR("failed to initialize output mode (%i)", ret);
349  thread_should_exit = true;
350  break;
351  }
352  }
353 
354  if (thread_data.input_objs_len > 0) {
355 
356  //get input: we cannot make the timeout too large, because the output needs to update
357  //periodically for stabilization and angle updates.
358 
359  for (int i = 0; i < thread_data.input_objs_len; ++i) {
360 
361  bool already_active = (last_active == i);
362 
363  ControlData *control_data_to_check = nullptr;
364  unsigned int poll_timeout = already_active ? 50 : 0; // poll only on active input to reduce latency
365  int ret = thread_data.input_objs[i]->update(poll_timeout, &control_data_to_check, already_active);
366 
367  if (ret) {
368  PX4_ERR("failed to read input %i (ret: %i)", i, ret);
369  continue;
370  }
371 
372  if (control_data_to_check != nullptr || already_active) {
373  control_data = control_data_to_check;
374  last_active = i;
375  }
376  }
377 
378  //update output
379  int ret = thread_data.output_obj->update(control_data);
380 
381  if (ret) {
382  PX4_ERR("failed to write output (%i)", ret);
383  break;
384  }
385 
386  thread_data.output_obj->publish();
387 
388  } else {
389  //wait for parameter changes. We still need to wake up regularily to check for thread exit requests
390  usleep(1e6);
391  }
392 
393  if (test_input && test_input->finished()) {
394  thread_should_exit = true;
395  break;
396  }
397 
398 
399  //check for parameter changes
400  bool updated;
401 
402  if (orb_check(parameter_update_sub, &updated) == 0 && updated) {
403  parameter_update_s param_update;
404  orb_copy(ORB_ID(parameter_update), parameter_update_sub, &param_update);
405  update_params(param_handles, params, updated);
406 
407  if (updated) {
408  //re-init objects
409  for (int i = 0; i < input_objs_len_max; ++i) {
410  if (thread_data.input_objs[i]) {
411  delete (thread_data.input_objs[i]);
412  thread_data.input_objs[i] = nullptr;
413  }
414  }
415 
416  thread_data.input_objs_len = 0;
417 
418  last_active = 0;
419 
420  if (thread_data.output_obj) {
421  delete (thread_data.output_obj);
422  thread_data.output_obj = nullptr;
423  }
424  }
425  }
426  }
427 
428  g_thread_data = nullptr;
429 
430  orb_unsubscribe(parameter_update_sub);
431 
432  for (int i = 0; i < input_objs_len_max; ++i) {
433  if (thread_data.input_objs[i]) {
434  delete (thread_data.input_objs[i]);
435  thread_data.input_objs[i] = nullptr;
436  }
437  }
438 
439  thread_data.input_objs_len = 0;
440 
441  if (thread_data.output_obj) {
442  delete (thread_data.output_obj);
443  thread_data.output_obj = nullptr;
444  }
445 
446  thread_running = false;
447  return 0;
448 }

462  if (!strcmp(argv[1], "start") || !strcmp(argv[1], "test")) {
463 
464  /* this is not an error */

466  PX4_WARN("mount driver already running");
467  return 0;
468  }
469 
470  thread_should_exit = false;

472  SCHED_DEFAULT,
473  SCHED_PRIORITY_DEFAULT,
474  1900,
475  vmount_thread_main,
476  (char *const *)argv + 1);
477 
478  int counter = 0;
479 
480  while (!thread_running && vmount_task >= 0) {
481  usleep(5000);
482 
483  if (++counter >= 100) {
484  break;
485  }
486  }
487 
488  if (vmount_task < 0) {
489  PX4_ERR("failed to start");
490  return -1;
491  }
492 
493  return counter < 100 || thread_should_exit ? 0 : -1;
494  }
495 
496  if (!strcmp(argv[1], "stop")) {
497 
498  /* this is not an error */
499  if (!thread_running) {
500  PX4_WARN("mount driver not running");
501  return 0;
502  }
503 
504  thread_should_exit = true;
505 
506  while (thread_running) {
507  usleep(100000);
508  }
509 
510  return 0;
511  }
512 
513  if (!strcmp(argv[1], "status")) {
514  if (thread_running && g_thread_data) {
515 
516  for (int i = 0; i < g_thread_data->input_objs_len; ++i) {
517  g_thread_data->input_objs[i]->print_status();
518  }
519 
520  if (g_thread_data->input_objs_len == 0) {
521  PX4_INFO("Input: None");
522  }
523 
524  if (g_thread_data->output_obj) {
525  g_thread_data->output_obj->print_status();
526 
527  } else {
528  PX4_INFO("Output: None");
529  }
530 
531  } else {
532  PX4_INFO("not running");
533  }
534 
535  return 0;
536  }

538  PX4_ERR("unrecognized command");
539  usage();
