51 /**
52 * Attempt to start driver on all available I2C busses.
53 *
54 * This function will return as soon as the first sensor
55 * is detected on one of the available busses or if no
56 * sensors are detected.
57 *
58 */
59 int
60 start()
61 {
62  for (unsigned i = 0; i < NUM_I2C_BUS_OPTIONS; i++) {
63  if (start_bus(i2c_bus_options[i]) == PX4_OK) {
64  return PX4_OK;
65  }
66  }
67 
68  return PX4_ERROR;
69 }

71 /**
72  * Start the driver on a specific bus.
73  *
74  * This function call only returns once the driver is up and running
75  * or failed to detect the sensor.
76  */
77 int
78 start_bus(uint8_t i2c_bus)
79 {
80  int fd = -1;
81 
82  if (g_dev != nullptr) {
83  PX4_ERR("already started");
84  goto fail;
85  }
86 
87  g_dev = new MS5525(i2c_bus, I2C_ADDRESS_1_MS5525DSO, PATH_MS5525);
88 
89  /* check if the MS4525DO was instantiated */
90  if (g_dev == nullptr) {
91  goto fail;
92  }
93 
94  /* try to initialize */
95  if (g_dev->init() != PX4_OK) {
96  goto fail;
97  }
98 
99  /* set the poll rate to default, starts automatic data collection */
100  fd = px4_open(PATH_MS5525, O_RDONLY);
101 
102  if (fd < 0) {
103  goto fail;
104  }
105 
106  if (px4_ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
107  goto fail;
108  }
109 
110  return PX4_OK;
111 
112 fail:
113 
114  if (g_dev != nullptr) {
115  delete g_dev;
116  g_dev = nullptr;
117  }
118 
119  return PX4_ERROR;
120 }

122 // stop the driver
123 int stop()
124 {
125  if (g_dev != nullptr) {

127  g_dev = nullptr;
128 
129  } else {

133 
134  return PX4_OK;
135 }

137 // reset the driver
138 int reset()
139 {
140  int fd = px4_open(PATH_MS5525, O_RDONLY);
141 
142  if (fd < 0) {
143  PX4_ERR("failed ");
144  return PX4_ERROR;
145  }
146 
147  if (px4_ioctl(fd, SENSORIOCRESET, 0) < 0) {
148  PX4_ERR("driver reset failed");
149  return PX4_ERROR;
150  }
151 
152  if (px4_ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
153  PX4_ERR("driver poll restart failed");
154  return PX4_ERROR;
155  }
156 
157  return PX4_OK;
158 }

174 int
175 ms5525_airspeed_main(int argc, char *argv[])
176 {
177  uint8_t i2c_bus = PX4_I2C_BUS_DEFAULT;
178 
179  int myoptind = 1;
180  int ch;
181  const char *myoptarg = nullptr;
182  bool start_all = false;
183 
184  while ((ch = px4_getopt(argc, argv, "ab:", &myoptind, &myoptarg)) != EOF) {
185  switch (ch) {
186  case 'b':
187  i2c_bus = atoi(myoptarg);
188  break;
189 
190  case 'a':
191  start_all = true;
192  break;
193 
194  default:
195  ms5525_airspeed_usage();
196  return 0;
197  }
198  }
199 
200  if (myoptind >= argc) {
201  ms5525_airspeed_usage();
202  return -1;
203  }
204 
205  /*
206  * Start/load the driver.
207  */
208  if (!strcmp(argv[myoptind], "start")) {
209  if (start_all) {
210  return ms5525_airspeed::start();
211 
212  } else {
213  return ms5525_airspeed::start_bus(i2c_bus);
214  }
215  }
216 
217  /*
218  * Stop the driver
219  */
220  if (!strcmp(argv[myoptind], "stop")) {
221  return ms5525_airspeed::stop();
222  }
223 
224  /*
225  * Reset the driver.
226  */
227  if (!strcmp(argv[myoptind], "reset")) {
228  return ms5525_airspeed::reset();
229  }
230 
231  ms5525_airspeed_usage();
232  return 0;
233 }
