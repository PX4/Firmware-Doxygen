207  /* Panic Logging in Battery Backed Up Files */
208 
209  /*
210  * In an ideal world, if a fault happens in flight the
211  * system save it to BBSRAM will then reboot. Upon
212  * rebooting, the system will log the fault to disk, recover
213  * the flight state and continue to fly. But if there is
214  * a fault on the bench or in the air that prohibit the recovery
215  * or committing the log to disk, the things are too broken to
216  * fly. So the question is:
217  *
218  * Did we have a hard fault and not make it far enough
219  * through the boot sequence to commit the fault data to
220  * the SD card?
221  */

235  */

238

243 
244  if (reboots > 2 || bytesWaiting != 0) {
245

247  * to the console.
248  */
249 
250  hardfault_write("boot", fileno(stdout), HARDFAULT_DISPLAY_FORMAT, false);
251

255

282

294  break;
295 
296  case 'B':

299  break;
300 
301  default:
302  break;
303  } // Inner Switch
304

306  "Enter C - Clear the fault log\n" \
307  "Enter D - Dump fault log\n\n?>");
308  fflush(stdout);
309 
310  if (!dbgContinue) {
311  c = getchar();
312  }
313 
314  break;
315 
316  } // outer switch
317  } // for
318 
319  } // inner if
320  } // outer if
321
