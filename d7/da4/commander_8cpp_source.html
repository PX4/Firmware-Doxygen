66 void Commander::ManualControlInputCallback(const px4::manual_control_setpointConstPtr &msg)
67 {
68  _got_manual_control = true;
69 
70  /* fill vehicle control mode based on (faked) stick positions*/
71  EvalSwitches(msg, _msg_vehicle_status, _msg_commander_state, _msg_vehicle_control_mode);
72  _msg_vehicle_control_mode.timestamp = px4::get_time_micros();
73 
74  /* fill actuator armed */
75  float arm_th = 0.95;
76  _msg_actuator_armed.timestamp = px4::get_time_micros();
77 
78  if (_msg_actuator_armed.armed) {
79  /* Check for disarm */
80  if (msg->r < -arm_th && msg->z < (1 - arm_th)) {
81  _msg_actuator_armed.armed = false;
82  _msg_vehicle_control_mode.flag_armed = false;
83  _msg_vehicle_status.arming_state = _msg_vehicle_status.ARMING_STATE_STANDBY;
84  }
85 
86  } else {
87  /* Check for arm */
88  if (msg->r > arm_th && msg->z < (1 - arm_th)) {
89  _msg_actuator_armed.armed = true;
90  _msg_vehicle_control_mode.flag_armed = true;
91  _msg_vehicle_status.arming_state = _msg_vehicle_status.ARMING_STATE_ARMED;
92  }
93  }
94 
95  /* fill vehicle status */
96  _msg_vehicle_status.timestamp = px4::get_time_micros();
97  _msg_vehicle_status.hil_state = _msg_vehicle_status.HIL_STATE_OFF;
98  _msg_vehicle_status.is_rotary_wing = true;
99 
100  _vehicle_control_mode_pub.publish(_msg_vehicle_control_mode);
101  _actuator_armed_pub.publish(_msg_actuator_armed);
102  _vehicle_status_pub.publish(_msg_vehicle_status);
103  _commander_state_pub.publish(_msg_commander_state);
104 
105  /* Fill parameter update */
106  if (px4::get_time_micros() - _msg_parameter_update.timestamp > 1e6) {
107  _msg_parameter_update.timestamp = px4::get_time_micros();
108  _parameter_update_pub.publish(_msg_parameter_update);
109  }
110 }

143 void Commander::EvalSwitches(const px4::manual_control_setpointConstPtr &msg,
144  px4::vehicle_status &msg_vehicle_status,
145  px4::commander_state &msg_commander_state,
146  px4::vehicle_control_mode &msg_vehicle_control_mode)
147 {
148  // XXX this is a minimal implementation. If more advanced functionalities are
149  // needed consider a full port of the commander
150 
151 
152  if (msg->offboard_switch == px4::manual_control_setpoint::SWITCH_POS_ON) {
153  SetOffboardControl(_msg_offboard_control_mode, msg_vehicle_control_mode);
154  return;
155  }
156 
157  msg_vehicle_control_mode.flag_control_offboard_enabled = false;
158 
159  switch (msg->mode_switch) {
160  case px4::manual_control_setpoint::SWITCH_POS_NONE:
161  ROS_WARN("Joystick button mapping error, main mode not set");
162  break;
163 
164  case px4::manual_control_setpoint::SWITCH_POS_OFF: // MANUAL
165  msg_commander_state.main_state = msg_commander_state.MAIN_STATE_MANUAL;
166  msg_vehicle_status.nav_state = msg_vehicle_status.NAVIGATION_STATE_MANUAL;
167  msg_vehicle_control_mode.flag_control_manual_enabled = true;
168  msg_vehicle_control_mode.flag_control_rates_enabled = true;
169  msg_vehicle_control_mode.flag_control_attitude_enabled = true;
170  msg_vehicle_control_mode.flag_control_altitude_enabled = false;
171  msg_vehicle_control_mode.flag_control_climb_rate_enabled = false;
172  msg_vehicle_control_mode.flag_control_position_enabled = false;
173  msg_vehicle_control_mode.flag_control_velocity_enabled = false;
174 
175  break;
176 
177  case px4::manual_control_setpoint::SWITCH_POS_MIDDLE: // ASSIST
178  if (msg->posctl_switch == px4::manual_control_setpoint::SWITCH_POS_ON) {
179  msg_commander_state.main_state = msg_commander_state.MAIN_STATE_POSCTL;
180  msg_vehicle_status.nav_state = msg_vehicle_status.NAVIGATION_STATE_POSCTL;
181  msg_vehicle_control_mode.flag_control_manual_enabled = true;
182  msg_vehicle_control_mode.flag_control_rates_enabled = true;
183  msg_vehicle_control_mode.flag_control_attitude_enabled = true;
184  msg_vehicle_control_mode.flag_control_altitude_enabled = true;
185  msg_vehicle_control_mode.flag_control_climb_rate_enabled = true;
186  msg_vehicle_control_mode.flag_control_position_enabled = true;
187  msg_vehicle_control_mode.flag_control_velocity_enabled = true;
188 
189  } else {
190  msg_commander_state.main_state = msg_commander_state.MAIN_STATE_ALTCTL;
191  msg_vehicle_status.nav_state = msg_vehicle_status.NAVIGATION_STATE_ALTCTL;
192  msg_vehicle_control_mode.flag_control_manual_enabled = true;
193  msg_vehicle_control_mode.flag_control_rates_enabled = true;
194  msg_vehicle_control_mode.flag_control_attitude_enabled = true;
195  msg_vehicle_control_mode.flag_control_altitude_enabled = true;
196  msg_vehicle_control_mode.flag_control_climb_rate_enabled = true;
197  msg_vehicle_control_mode.flag_control_position_enabled = false;
198  msg_vehicle_control_mode.flag_control_velocity_enabled = false;
199  }
200 
201  break;
202  }

207 {

209

218

224  _msg_vehicle_control_mode.flag_armed = true;
225
