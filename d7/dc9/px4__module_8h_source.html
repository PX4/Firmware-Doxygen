60 /**
61  ** class ModuleBase
62  *
63  * Base class for modules, implementing common functionality, such as 'start',
64  * 'stop' and 'status' commands.
65  * Currently does not support modules which allow multiple instances, such as
66  * mavlink.
67  *
68  * The class is implemented as curiously recurring template pattern (CRTP). It
69  * allows to have a static object in the base class that is different for
70  * each module type, and call static methods from the base class.
71  *
72  * Required methods for a derived class:
73  *
74  * When running in its own thread:
75  static int task_spawn(int argc, char *argv[]) {
76  // call px4_task_spawn_cmd() with &run_trampoline as startup method
77  // optional: wait until _object is not null, which means the task got initialized (use a timeout)
78  // set _task_id and return 0
79  // on error return != 0 (and _task_id must be -1)
80  }
81  static T *instantiate(int argc, char *argv[]) {
82  // this is called from within the new thread, from run_trampoline()
83  // parse the arguments
84  // create a new object T & return it
85  // or return nullptr on error
86  }
87  static int custom_command(int argc, char *argv[]) {
88  // support for custom commands
89  // it none are supported, just do:
90  return print_usage("unrecognized command");
91  }
92  static int print_usage(const char *reason = nullptr) {
93  // use the PRINT_MODULE_* methods...
94  }
95  *
96  * When running on the work queue:
97  * - custom_command & print_usage as above
98  static int task_spawn(int argc, char *argv[]) {
99  // parse the arguments
100  // set _object (here or from the work_queue() callback)
101  // call work_queue() (with a custom cycle trampoline)
102  // optional: wait until _object is not null, which means the task got initialized (use a timeout)
103  // set _task_id to task_id_is_work_queue and return 0
104  // on error return != 0 (and _task_id must be -1)
105  }
106  */
107 template<class T>
108 class ModuleBase
109 {
110 public:
111  ModuleBase() = default;
112  virtual ~ModuleBase() {}
113 
114  /**
115  * main entry point. Should be called directly from the module's main method.
116  */
117  static int main(int argc, char *argv[])
118  {
119  if (argc <= 1 || strcmp(argv[1], "help") == 0 || strcmp(argv[1], "-h") == 0) {
120  return T::print_usage();
121  }
122 
123  if (strcmp(argv[1], "start") == 0) {
124  // we pass the 'start' argument too, because later on px4_getopt() will ignore the first argument
125  return start_command_base(argc - 1, argv + 1);
126  }
127 
128  if (strcmp(argv[1], "stop") == 0) {
129  return stop_command();
130  }
131 
132  if (strcmp(argv[1], "status") == 0) {
133  return status_command();
134  }
135 
136  lock_module(); // we better lock here, as the method could access _object
137  int ret = T::custom_command(argc - 1, argv + 1);
138  unlock_module();
139 
140  return ret;
141  }
142 
143  /**
144  * Entry point for px4_task_spawn_cmd() if the module runs in its own thread.
145  * It does:
146  * - instantiate the object
147  * - call run() on it to execute the main loop
148  * - cleanup: delete the object
149  * @param argc start argument(s)
150  * @param argv start argument(s)
151  */
152  static int run_trampoline(int argc, char *argv[])
153  {
154  int ret = 0;
155 
156 #ifdef __PX4_NUTTX
157  // on NuttX task_create() adds the task name as first argument
158  argc -= 1;
159  argv += 1;
160 #endif
161 
162  _object = T::instantiate(argc, argv);
163 
164  if (_object) {
165  T *object = (T *)_object;
166  object->run();
167 
168  } else {
169  PX4_ERR("failed to instantiate object");
170  ret = -1;
171  }
172 
173  exit_and_cleanup();
174 
175  return ret;
176  }
177 
178  /**
179  * handle 'command start': check if already running and call T::task_spawn() if it's not
180  */
181  static int start_command_base(int argc, char *argv[])
182  {
183  int ret = 0;
184  lock_module();
185 
186  if (is_running()) {
187  ret = -1;
188  PX4_ERR("Task already running");
189 
190  } else {
191  ret = T::task_spawn(argc, argv);
192 
193  if (ret < 0) {
194  PX4_ERR("Task start failed (%i)", ret);
195  }
196  }
197 
198  unlock_module();
199  return ret;
200  }
201 
202  /**
203  * handle 'command stop': check if already running and if it is, request the module to stop
204  * and wait for it.
205  */
206  static int stop_command()
207  {
208  int ret = 0;
209  lock_module();
210 
211  if (is_running()) {
212  if (_object) {
213  T *object = (T *)_object;
214  object->request_stop();
215 
216  unsigned int i = 0;
217 
218  do {
219  unlock_module();
220  usleep(20000); // 20 ms
221  lock_module();
222 
223  if (++i > 100 && _task_id != -1) { // wait at most 2 sec
224  if (_task_id != task_id_is_work_queue) {
225  px4_task_delete(_task_id);
226  }
227 
228  _task_id = -1;
229 
230  if (_object) {
231  delete _object;
232  _object = nullptr;
233  }
234 
235  ret = -1;
236  break;
237  }
238  } while (_task_id != -1);
239 
240  } else {
241  // very unlikely event and can only happen on work queues:
242  // the starting thread does not wait for the work queue to initialize,
243  // and inside the work queue, the allocation of _object fails
244  // and exit_and_cleanup() is not called.
245  _task_id = -1;
246  }
247  }
248 
249  unlock_module();
250  return ret;
251  }
252 
253  /**
254  *
255  * handle 'command status': check if running and call print_status() if it is
256  * @return 0 on success, -1 if not running
257  */
258  static int status_command()
259  {
260  int ret = -1;
261  lock_module();
262 
263  if (is_running() && _object) {
264  T *object = (T *)_object;
265  ret = object->print_status();
266 
267  } else {
268  PX4_INFO("not running");
269  }
270 
271  unlock_module();
272  return ret;
273  }

299  /** Tell the module to stop (used from outside or inside the module thread) */
300  virtual void request_stop() { _task_should_exit = true; }
301 
302  /** check if the module should stop (used within the module thread) */
303  bool should_exit() const { return _task_should_exit; }

305  /**
306  * Exit the module and delete the object. Called from within the module's thread.
307  * For work queue modules, this needs to be called from the derived class in the
308  * cycle method, when should_exit() returns true.
309  */
310  static void exit_and_cleanup()
311  {
312  // we need to take the lock here:
313  // - if startup fails and we're faster than the parent thread, it will set
314  // _task_id and subsequently it will look like the task is running
315  // - deleting the object must take place inside the lock
316  lock_module();
317 
318  if (_object) {
319  delete _object;
320  _object = nullptr;
321  }
322 
323  _task_id = -1; // signal a potentially waiting thread for the module to exit that it can continue
324  unlock_module();
325  }
326 
327  /**
328  * Wait until _object got initialized (from the new thread). This can be called from task_spawn().
329  * @return 0 on success, -1 on timeout.
330  */
331  static int wait_until_running()
332  {
333  int i = 0;
334 
335  do {
336  /* wait up to 1s */
337  usleep(2500);
338 
339  } while (!_object && ++i < 400);
340 
341  if (i == 400) {
342  PX4_ERR("Timed out while waiting for thread to start");
343  return -1;
344  }
345 
346  return 0;
347  }
348 
349  /** get the module's object instance (this is null if it's not running) */
350  static T *get_instance()
351  {
352  return (T *)_object;
353  }

356  static volatile T *_object; ///< instance if the module is running
357  static int _task_id; ///< task handle: -1 = invalid, otherwise task is assumed to be running

359  static constexpr const int task_id_is_work_queue = -2; ///< special value if task runs on the work queue
