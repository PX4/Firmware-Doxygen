211 static uint8_t crc8(uint8_t *p, uint8_t len)
212 {
213  uint16_t i;
214  uint16_t crc = 0x0;
215 
216  while (len--) {
217  i = (crc ^ *p++) & 0xFF;
218  crc = (crc_table[i] ^ (crc << 8)) & 0xFF;
219  }
220 
221  return crc & 0xFF;
222 }

274 int
275 TERARANGER::init()
276 {
277  int ret = PX4_ERROR;
278  int hw_model;
279  param_get(param_find("SENS_EN_TRANGER"), &hw_model);
280 
281  switch (hw_model) {
282  case 0: /* Disabled */
283  DEVICE_LOG("Disabled");
284  return ret;
285 
286  case 1: /* Autodetect */
287  /* Assume TROne */
288  set_device_address(TRONE_BASEADDR);
289 
290  if (I2C::init() != OK) {
291  set_device_address(TREVO_BASEADDR);
292 
293  if (I2C::init() != OK) {
294  goto out;
295 
296  } else {
297  _min_distance = TREVO_60M_MIN_DISTANCE;
298  _max_distance = TREVO_60M_MAX_DISTANCE;
299  }
300 
301  } else {
302  _min_distance = TRONE_MIN_DISTANCE;
303  _max_distance = TRONE_MAX_DISTANCE;
304  }
305 
306  break;
307 
308  case 2: /* TROne */
309  set_device_address(TRONE_BASEADDR);
310 
311  if (I2C::init() != OK) {
312  goto out;
313  }
314 
315  _min_distance = TRONE_MIN_DISTANCE;
316  _max_distance = TRONE_MAX_DISTANCE;
317  break;
318 
319  case 3: /* TREvo60m */
320  set_device_address(TREVO_BASEADDR);
321 
322  /* do I2C init (and probe) first */
323  if (I2C::init() != OK) {
324  goto out;
325  }
326 
327  _min_distance = TREVO_60M_MIN_DISTANCE;
328  _max_distance = TREVO_60M_MAX_DISTANCE;
329  break;
330 
331  case 4: /* TREvo600Hz */
332  set_device_address(TREVO_BASEADDR);
333 
334  /* do I2C init (and probe) first */
335  if (I2C::init() != OK) {
336  goto out;
337  }
338 
339  _min_distance = TREVO_600HZ_MIN_DISTANCE;
340  _max_distance = TREVO_600HZ_MAX_DISTANCE;
341  break;
342 
343  default:
344  DEVICE_LOG("invalid HW model %d.", hw_model);
345  return ret;
346  }
347 
348  /* allocate basic report buffers */
349  _reports = new ringbuffer::RingBuffer(2, sizeof(distance_sensor_s));
350 
351  if (_reports == nullptr) {
352  goto out;
353  }
354 
355  _class_instance = register_class_devname(RANGE_FINDER_BASE_DEVICE_PATH);
356 
357  if (_class_instance == CLASS_DEVICE_PRIMARY) {
358  /* get a publish handle on the range finder topic */
359  struct distance_sensor_s ds_report;
360  measure();
361  _reports->get(&ds_report);
362 
363  _distance_sensor_topic = orb_advertise_multi(ORB_ID(distance_sensor), &ds_report,
364  &_orb_class_instance, ORB_PRIO_LOW);
365 
366  if (_distance_sensor_topic == nullptr) {
367  DEVICE_LOG("failed to create distance_sensor object. Did you start uOrb?");
368  }
369  }
370 
371  ret = OK;
372  /* sensor is ok, but we don't really know if it is within range */
373  _sensor_ok = true;
374 out:
375  return ret;
376 }

424 int
425 TERARANGER::ioctl(device::file_t *filp, int cmd, unsigned long arg)
426 {
427  switch (cmd) {
428 
429  case SENSORIOCSPOLLRATE: {
430  switch (arg) {
431 
432  /* switching to manual polling */
433  case SENSOR_POLLRATE_MANUAL:
434  stop();
435  _measure_ticks = 0;
436  return OK;
437 
438  /* external signalling (DRDY) not supported */
439  case SENSOR_POLLRATE_EXTERNAL:
440 
441  /* zero would be bad */
442  case 0:
443  return -EINVAL;
444 
445  /* set default/max polling rate */
446  case SENSOR_POLLRATE_MAX:
447  case SENSOR_POLLRATE_DEFAULT: {
448  /* do we need to start internal polling? */
449  bool want_start = (_measure_ticks == 0);
450 
451  /* set interval for next measurement to minimum legal value */
452  _measure_ticks = USEC2TICK(TERARANGER_CONVERSION_INTERVAL);
453 
454  /* if we need to start the poll state machine, do it */
455  if (want_start) {
456  start();
457  }
458 
459  return OK;
460  }
461 
462  /* adjust to a legal polling interval in Hz */
463  default: {
464  /* do we need to start internal polling? */
465  bool want_start = (_measure_ticks == 0);
466 
467  /* convert hz to tick interval via microseconds */
468  int ticks = USEC2TICK(1000000 / arg);
469 
470  /* check against maximum rate */
471  if (ticks < USEC2TICK(TERARANGER_CONVERSION_INTERVAL)) {
472  return -EINVAL;
473  }
474 
475  /* update interval for next measurement */
476  _measure_ticks = ticks;
477 
478  /* if we need to start the poll state machine, do it */
479  if (want_start) {
480  start();
481  }
482 
483  return OK;
484  }
485  }
486  }
487 
488  case SENSORIOCGPOLLRATE:
489  if (_measure_ticks == 0) {
490  return SENSOR_POLLRATE_MANUAL;
491  }
492 
493  return (1000 / _measure_ticks);
494 
495  case SENSORIOCSQUEUEDEPTH: {
496  /* lower bound is mandatory, upper bound is a sanity check */
497  if ((arg < 1) || (arg > 100)) {
498  return -EINVAL;
499  }
500 
501  ATOMIC_ENTER;
502 
503  if (!_reports->resize(arg)) {
504  ATOMIC_LEAVE;
505  return -ENOMEM;
506  }
507 
508  ATOMIC_LEAVE;
509 
510  return OK;
511  }
512 
513  case SENSORIOCRESET:
514  /* XXX implement this */
515  return -EINVAL;
516 
517  default:
518  /* give it to the superclass */
519  return I2C::ioctl(filp, cmd, arg);
520  }
521 }

523 ssize_t
524 TERARANGER::read(device::file_t *filp, char *buffer, size_t buflen)
525 {
526  unsigned count = buflen / sizeof(struct distance_sensor_s);
527  struct distance_sensor_s *rbuf = reinterpret_cast<struct distance_sensor_s *>(buffer);
528  int ret = 0;
529 
530  /* buffer must be large enough */
531  if (count < 1) {
532  return -ENOSPC;
533  }
534 
535  /* if automatic measurement is enabled */
536  if (_measure_ticks > 0) {
537 
538  /*
539  * While there is space in the caller's buffer, and reports, copy them.
540  * Note that we may be pre-empted by the workq thread while we are doing this;
541  * we are careful to avoid racing with them.
542  */
543  while (count--) {
544  if (_reports->get(rbuf)) {
545  ret += sizeof(*rbuf);
546  rbuf++;
547  }
548  }
549 
550  /* if there was no data, warn the caller */
551  return ret ? ret : -EAGAIN;
552  }
553 
554  /* manual measurement - run one conversion */
555  do {
556  _reports->flush();
557 
558  /* trigger a measurement */
559  if (OK != measure()) {
560  ret = -EIO;
561  break;
562  }
563 
564  /* wait for it to complete */
565  usleep(TERARANGER_CONVERSION_INTERVAL);
566 
567  /* run the collection phase */
568  if (OK != collect()) {
569  ret = -EIO;
570  break;
571  }
572 
573  /* state machine will have generated a report, copy it out */
574  if (_reports->get(rbuf)) {
575  ret = sizeof(*rbuf);
576  }
577 
578  } while (0);
579 
580  return ret;
581 }

605 int
606 TERARANGER::collect()
607 {
608  int ret = -EIO;
609 
610  /* read from the sensor */
611  uint8_t val[3] = {0, 0, 0};
612 
613  perf_begin(_sample_perf);
614 
615  ret = transfer(nullptr, 0, &val[0], 3);
616 
617  if (ret < 0) {
618  DEVICE_LOG("error reading from sensor: %d", ret);
619  perf_count(_comms_errors);
620  perf_end(_sample_perf);
621  return ret;
622  }
623 
624  uint16_t distance_mm = (val[0] << 8) | val[1];
625  float distance_m = float(distance_mm) * 1e-3f;
626  struct distance_sensor_s report;
627 
628  report.timestamp = hrt_absolute_time();
629  /* there is no enum item for a combined LASER and ULTRASOUND which it should be */
630  report.type = distance_sensor_s::MAV_DISTANCE_SENSOR_LASER;
631  report.orientation = _rotation;
632  report.current_distance = distance_m;
633  report.min_distance = get_minimum_distance();
634  report.max_distance = get_maximum_distance();
635  report.covariance = 0.0f;
636  report.signal_quality = -1;
637  /* TODO: set proper ID */
638  report.id = 0;
639 
640  // This validation check can be used later
641  _valid = crc8(val, 2) == val[2] && (float)report.current_distance > report.min_distance
642  && (float)report.current_distance < report.max_distance ? 1 : 0;
643 
644  /* publish it, if we are the primary */
645  if (_distance_sensor_topic != nullptr) {
646  orb_publish(ORB_ID(distance_sensor), _distance_sensor_topic, &report);
647  }
648 
649  _reports->force(&report);
650 
651  /* notify anyone waiting for data */
652  poll_notify(POLLIN);
653 
654  ret = OK;
655 
656  perf_end(_sample_perf);
657  return ret;
658 }

686 TERARANGER::cycle()
687 {
688  /* collection phase? */
689  if (_collect_phase) {
690 
691  /* perform collection */
692  if (OK != collect()) {
693  DEVICE_LOG("collection error");
694  /* restart the measurement state machine */
695  start();
696  return;
697  }
698 
699  /* next phase is measurement */
700  _collect_phase = false;
701 
702  /*
703  * Is there a collect->measure gap?
704  */
705  if (_measure_ticks > USEC2TICK(TERARANGER_CONVERSION_INTERVAL)) {
706  /* schedule a fresh cycle call when we are ready to measure again */
707  work_queue(HPWORK,
708  &_work,
709  (worker_t)&TERARANGER::cycle_trampoline,
710  this,
711  _measure_ticks - USEC2TICK(TERARANGER_CONVERSION_INTERVAL));
712 
713  return;
714  }
715  }
716 
717  /* measurement phase */
718  if (OK != measure()) {
719  DEVICE_LOG("measure error");
720  }
721 
722  /* next phase is collection */
723  _collect_phase = true;
724 
725  /* schedule a fresh cycle call when the measurement is done */
726  work_queue(HPWORK,
727  &_work,
728  (worker_t)&TERARANGER::cycle_trampoline,
729  this,
730  USEC2TICK(TERARANGER_CONVERSION_INTERVAL));
731 }

857 /**
858  * Perform some basic functional tests on the driver;
859  * make sure we can collect data from the sensor in polled
860  * and automatic modes.
861  */
862 int
863 test()
864 {
865  struct distance_sensor_s report;
866  ssize_t sz;
867  int ret;
868 
869  int fd = px4_open(TERARANGER_DEVICE_PATH, O_RDONLY);
870 
871  if (fd < 0) {
872  PX4_ERR("%s open failed (try 'teraranger start' if the driver is not running)", TERARANGER_DEVICE_PATH);
873  return PX4_ERROR;
874  }
875 
876  /* do a simple demand read */
877  sz = read(fd, &report, sizeof(report));
878 
879  if (sz != sizeof(report)) {
880  PX4_ERR("immediate read failed");
881  return PX4_ERROR;
882  }
883 
884  print_message(report);
885 
886  /* start the sensor polling at 2Hz */
887  if (OK != ioctl(fd, SENSORIOCSPOLLRATE, 2)) {
888  PX4_ERR("failed to set 2Hz poll rate");
889  return PX4_ERROR;
890  }
891 
892  /* read the sensor 50x and report each value */
893  for (unsigned i = 0; i < 50; i++) {
894  struct pollfd fds;
895 
896  /* wait for data to be ready */
897  fds.fd = fd;
898  fds.events = POLLIN;
899  ret = poll(&fds, 1, 2000);
900 
901  if (ret != 1) {
902  PX4_ERR("timed out waiting for sensor data");
903  return PX4_ERROR;
904  }
905 
906  /* now go get it */
907  sz = read(fd, &report, sizeof(report));
908 
909  if (sz != sizeof(report)) {
910  PX4_ERR("periodic read failed");
911  return PX4_ERROR;
912  }
913 
914  print_message(report);
915  }
916 
917  /* reset the sensor polling to default rate */
918  if (OK != ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT)) {
919  PX4_ERR("failed to set default poll rate");
920  return PX4_ERROR;
921  }
922 
923  px4_close(fd);
924  PX4_INFO("PASS");
925  return PX4_OK;
926 
927 }
