535 void
536 FXAS21002C::reset()
537 {
538  /* write 0 0 0 000 00 = 0x00 to CTRL_REG1 to place FXOS21002 in Standby
539  * [6]: RST=0
540  * [5]: ST=0 self test disabled
541  * [4-2]: DR[2-0]=000 for 200Hz ODR
542  * [1-0]: Active=0, Ready=0 for Standby mode
543  */
544 
545  write_reg(FXAS21002C_CTRL_REG1, 0);
546 
547  /* write 0000 0000 = 0x00 to CTRL_REG0 to configure range and filters
548  * [7-6]: BW[1-0]=00, LPF disabled
549  * [5]: SPIW=0 4 wire SPI
550  * [4-3]: SEL[1-0]=00 for 10Hz HPF at 200Hz ODR
551  * [2]: HPF_EN=0 disable HPF
552  * [1-0]: FS[1-0]=00 for 1600dps (TBD CHANGE TO 2000dps when final trimmed parts available)
553  */
554 
555  write_checked_reg(FXAS21002C_CTRL_REG0, 0);
556 
557  /* write CTRL_REG1 to configure 800Hz ODR and enter Active mode */
558 
559  write_checked_reg(FXAS21002C_CTRL_REG1, CTRL_REG_DR_800HZ | CTRL_REG1_ACTIVE);
560 
561  /* Set the default */
562 
563  set_samplerate(0);
564  set_range(FXAS21002C_DEFAULT_RANGE_DPS);
565  set_driver_lowpass_filter(FXAS21002C_DEFAULT_RATE, FXAS21002C_DEFAULT_FILTER_FREQ);
566  _read = 0;
567 }

583 ssize_t
584 FXAS21002C::read(struct file *filp, char *buffer, size_t buflen)
585 {
586  unsigned count = buflen / sizeof(struct gyro_report);
587  struct gyro_report *gbuf = reinterpret_cast<struct gyro_report *>(buffer);
588  int ret = 0;
589 
590  /* buffer must be large enough */
591  if (count < 1) {
592  return -ENOSPC;
593  }
594 
595  /* if automatic measurement is enabled */
596  if (_call_interval > 0) {
597 
598  /*
599  * While there is space in the caller's buffer, and reports, copy them.
600  * Note that we may be pre-empted by the measurement code while we are doing this;
601  * we are careful to avoid racing with it.
602  */
603  while (count--) {
604  if (_reports->get(gbuf)) {
605  ret += sizeof(*gbuf);
606  gbuf++;
607  }
608  }
609 
610  /* if there was no data, warn the caller */
611  return ret ? ret : -EAGAIN;
612  }
613 
614  /* manual measurement */
615  _reports->flush();
616  measure();
617 
618  /* measurement will have generated a report, copy it out */
619  if (_reports->get(gbuf)) {
620  ret = sizeof(*gbuf);
621  }
622 
623  return ret;
624 }

643  /* zero would be bad */
644  case 0:
645  return -EINVAL;
646 
647  /* set default/max polling rate */
648  case SENSOR_POLLRATE_MAX:
649  case SENSOR_POLLRATE_DEFAULT:
650  return ioctl(filp, SENSORIOCSPOLLRATE, FXAS21002C_DEFAULT_RATE);
651 
652  /* adjust to a legal polling interval in Hz */
653  default: {
654  /* do we need to start internal polling? */
655  bool want_start = (_call_interval == 0);
656 
657  /* convert hz to hrt interval via microseconds */
658  unsigned ticks = 1000000 / arg;
659 
660  /* check against maximum sane rate */
661  if (ticks < 1000) {
662  return -EINVAL;
663  }
664 
665  /* update interval for next measurement */
666  /* XXX this is a bit shady, but no other way to adjust... */
667  _call_interval = ticks;
668 
669  _gyro_call.period = _call_interval - FXAS21002C_TIMER_REDUCTION;
670 
671  /* adjust filters */
672  float cutoff_freq_hz = _gyro_filter_x.get_cutoff_freq();
673  float sample_rate = 1.0e6f / ticks;
674  set_driver_lowpass_filter(sample_rate, cutoff_freq_hz);
675 
676  /* if we need to start the poll state machine, do it */
677  if (want_start) {
678  start();
679  }
680 
681  return OK;
682  }
683  }
684  }
685 
686  case SENSORIOCGPOLLRATE:
687  if (_call_interval == 0) {
688  return SENSOR_POLLRATE_MANUAL;
689  }
690 
691  return 1000000 / _call_interval;
692 
693  case SENSORIOCSQUEUEDEPTH: {
694  /* lower bound is mandatory, upper bound is a sanity check */
695  if ((arg < 1) || (arg > 100)) {
696  return -EINVAL;
697  }
698 
699  irqstate_t flags = px4_enter_critical_section();
700 
701  if (!_reports->resize(arg)) {
702  px4_leave_critical_section(flags);
703  return -ENOMEM;
704  }
705 
706  px4_leave_critical_section(flags);
707 
708  return OK;
709  }
710 
711  case SENSORIOCRESET:
712  reset();
713  return OK;
714 
715  case GYROIOCSSAMPLERATE:
716  return set_samplerate(arg);
717 
718  case GYROIOCGSAMPLERATE:
719  return _current_rate;
720 
721  case GYROIOCSSCALE:
722  /* copy scale in */
723  memcpy(&_gyro_scale, (struct gyro_calibration_s *) arg, sizeof(_gyro_scale));
724  return OK;
725 
726  case GYROIOCGSCALE:
727  /* copy scale out */
728  memcpy((struct gyro_calibration_s *) arg, &_gyro_scale, sizeof(_gyro_scale));
729  return OK;
730 
731  case GYROIOCSRANGE:
732  /* arg should be in dps */
733  return set_range(arg);
734 
735  case GYROIOCGRANGE:
736  /* convert to dps and round */
737  return (unsigned long)(_gyro_range_rad_s * 180.0f / M_PI_F + 0.5f);
738 
739  default:
740  /* give it to the superclass */
741  return SPI::ioctl(filp, cmd, arg);
742  }

934 void
935 FXAS21002C::check_registers(void)
936 {
937 
938  uint8_t v;
939 
940  if ((v = read_reg(_checked_registers[_checked_next])) != _checked_values[_checked_next]) {
941  /*
942  if we get the wrong value then we know the SPI bus
943  or sensor is very sick. We set _register_wait to 20
944  and wait until we have seen 20 good values in a row
945  before we consider the sensor to be OK again.
946  */
947  perf_count(_bad_registers);
948 
949  /*
950  try to fix the bad register value. We only try to
951  fix one per loop to prevent a bad sensor hogging the
952  bus. We skip zero as that is the WHO_AM_I, which
953  is not writeable
954  */
955  if (_checked_next != 0) {
956  write_reg(_checked_registers[_checked_next], _checked_values[_checked_next]);
957  }
958 
959  _register_wait = 20;
960  }
961 
962  _checked_next = (_checked_next + 1) % FXAS21002C_NUM_CHECKED_REGISTERS;
963 }

965 void
966 FXAS21002C::measure()
967 {
968  /* status register and data as read back from the device */
969 
970 #pragma pack(push, 1)
971  struct {
972  uint8_t cmd;
973  uint8_t status;
974  int16_t x;
975  int16_t y;
976  int16_t z;
977  } raw_gyro_report;
978 #pragma pack(pop)
979 
980  struct gyro_report gyro_report;
981 
982  /* start the performance counter */
983  perf_begin(_sample_perf);
984 
985  check_registers();
986 
987  if (_register_wait != 0) {
988  // we are waiting for some good transfers before using
989  // the sensor again.
990  _register_wait--;
991  perf_end(_sample_perf);
992  return;
993  }
994 
995  /* fetch data from the sensor */
996  memset(&raw_gyro_report, 0, sizeof(raw_gyro_report));
997  raw_gyro_report.cmd = DIR_READ(FXAS21002C_STATUS);
998  transfer((uint8_t *)&raw_gyro_report, (uint8_t *)&raw_gyro_report, sizeof(raw_gyro_report));
999 
1000  if (!(raw_gyro_report.status & DR_STATUS_ZYXDR)) {
1001  perf_end(_sample_perf);
1002  perf_count(_duplicates);
1003  return;
1004  }
1005 
1006  /*
1007  * The TEMP register contains an 8-bit 2's complement temperature value with a range
1008  * of –128 °C to +127 °C and a scaling of 1 °C/LSB. The temperature data is only
1009  * compensated (factory trim values applied) when the device is operating in the Active
1010  * mode and actively measuring the angular rate.
1011  */
1012 
1013  if ((_read % _current_rate) == 0) {
1014  _last_temperature = read_reg(FXAS21002C_TEMP) * 1.0f;
1015  gyro_report.temperature = _last_temperature;
1016  }
1017 
1018  /*
1019  * 1) Scale raw value to SI units using scaling from datasheet.
1020  * 2) Subtract static offset (in SI units)
1021  * 3) Scale the statically calibrated values with a linear
1022  * dynamically obtained factor
1023  *
1024  * Note: the static sensor offset is the number the sensor outputs
1025  * at a nominally 'zero' input. Therefore the offset has to
1026  * be subtracted.
1027  *
1028  * Example: A gyro outputs a value of 74 at zero angular rate
1029  * the offset is 74 from the origin and subtracting
1030  * 74 from all measurements centers them around zero.
1031  */
1032 
1033  gyro_report.timestamp = hrt_absolute_time();
1034 
1035  // report the error count as the number of bad
1036  // register reads. This allows the higher level
1037  // code to decide if it should use this sensor based on
1038  // whether it has had failures
1039  gyro_report.error_count = perf_event_count(_bad_registers);
1040 
1041  gyro_report.x_raw = swap16(raw_gyro_report.x);
1042  gyro_report.y_raw = swap16(raw_gyro_report.y);
1043  gyro_report.z_raw = swap16(raw_gyro_report.z);
1044 
1045  float xraw_f = gyro_report.x_raw;
1046  float yraw_f = gyro_report.y_raw;
1047  float zraw_f = gyro_report.z_raw;
1048 
1049  // apply user specified rotation
1050  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
1051 
1052  float x_in_new = ((xraw_f * _gyro_range_scale) - _gyro_scale.x_offset) * _gyro_scale.x_scale;
1053  float y_in_new = ((yraw_f * _gyro_range_scale) - _gyro_scale.y_offset) * _gyro_scale.y_scale;
1054  float z_in_new = ((zraw_f * _gyro_range_scale) - _gyro_scale.z_offset) * _gyro_scale.z_scale;
1055 
1056  gyro_report.x = _gyro_filter_x.apply(x_in_new);
1057  gyro_report.y = _gyro_filter_y.apply(y_in_new);
1058  gyro_report.z = _gyro_filter_z.apply(z_in_new);
1059 
1060  matrix::Vector3f gval(x_in_new, y_in_new, z_in_new);
1061  matrix::Vector3f gval_integrated;
1062 
1063  bool gyro_notify = _gyro_int.put(gyro_report.timestamp, gval, gval_integrated, gyro_report.integral_dt);
1064  gyro_report.x_integral = gval_integrated(0);
1065  gyro_report.y_integral = gval_integrated(1);
1066  gyro_report.z_integral = gval_integrated(2);
1067 
1068  gyro_report.scaling = _gyro_range_scale;
1069 
1070  /* return device ID */
1071  gyro_report.device_id = _device_id.devid;
1072 
1073 
1074  _reports->force(&gyro_report);
1075 
1076  /* notify anyone waiting for data */
1077  if (gyro_notify) {
1078  poll_notify(POLLIN);
1079 
1080  if (!(_pub_blocked)) {
1081  /* publish it */
1082  orb_publish(ORB_ID(sensor_gyro), _gyro_topic, &gyro_report);
1083  }
1084  }
1085 
1086  _read++;
1087 
1088  /* stop the perf counter */
1089  perf_end(_sample_perf);
1090 }
