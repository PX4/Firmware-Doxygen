328 bool Replay::readAndAddSubscription(std::ifstream &file, uint16_t msg_size)
329 {
330  _read_buffer.reserve(msg_size + 1);
331  char *message = (char *)_read_buffer.data();
332  streampos this_message_pos = file.tellg() - (streamoff)ULOG_MSG_HEADER_LEN;
333  file.read(message, msg_size);
334  message[msg_size] = 0;
335 
336  if (!file) {
337  return false;
338  }
339 
340  if (file.tellg() <= _subscription_file_pos) { //already read this subscription
341  return true;
342  }
343 
344  _subscription_file_pos = file.tellg();
345 
346  uint8_t multi_id = *(uint8_t *)message;
347  uint16_t msg_id = ((uint16_t) message[1]) | (((uint16_t) message[2]) << 8);
348  string topic_name(message + 3);
349  const orb_metadata *orb_meta = findTopic(topic_name);
350 
351  if (!orb_meta) {
352  PX4_WARN("Topic %s not found internally. Will ignore it", topic_name.c_str());
353  return true;
354  }
355 
356  CompatBase *compat = nullptr;
357 
358  //check the format: the field definitions must match
359  //FIXME: this should check recursively, all used nested types
360  string file_format = _file_formats[topic_name];
361 
362  if (file_format != orb_meta->o_fields) {
363  // check if we have a compatibility conversion available
364  if (topic_name == "sensor_combined") {
365  if (string(orb_meta->o_fields) == "uint64_t timestamp;float[3] gyro_rad;uint32_t gyro_integral_dt;"
366  "int32_t accelerometer_timestamp_relative;float[3] accelerometer_m_s2;"
367  "uint32_t accelerometer_integral_dt" &&
368  file_format == "uint64_t timestamp;float[3] gyro_rad;float gyro_integral_dt;"
369  "int32_t accelerometer_timestamp_relative;float[3] accelerometer_m_s2;"
370  "float accelerometer_integral_dt;") {
371  int gyro_integral_dt_offset_log;
372  int gyro_integral_dt_offset_intern;
373  int accelerometer_integral_dt_offset_log;
374  int accelerometer_integral_dt_offset_intern;
375  int unused;
376 
377  if (findFieldOffset(file_format, "gyro_integral_dt", gyro_integral_dt_offset_log, unused) &&
378  findFieldOffset(orb_meta->o_fields, "gyro_integral_dt", gyro_integral_dt_offset_intern, unused) &&
379  findFieldOffset(file_format, "accelerometer_integral_dt", accelerometer_integral_dt_offset_log, unused) &&
380  findFieldOffset(orb_meta->o_fields, "accelerometer_integral_dt", accelerometer_integral_dt_offset_intern, unused)) {
381  compat = new CompatSensorCombinedDtType(gyro_integral_dt_offset_log, gyro_integral_dt_offset_intern,
382  accelerometer_integral_dt_offset_log, accelerometer_integral_dt_offset_intern);
383  }
384  }
385  }
386 
387  if (!compat) {
388  PX4_WARN("Formats for %s don't match. Will ignore it.", topic_name.c_str());
389  PX4_WARN(" Internal format: %s", orb_meta->o_fields);
390  PX4_WARN(" File format : %s", file_format.c_str());
391  return true; // not a fatal error
392  }
393  }
394 
395  Subscription *subscription = new Subscription();
396  subscription->orb_meta = orb_meta;
397  subscription->multi_id = multi_id;
398  subscription->compat = compat;
399 
400 
401  //find the timestamp offset
402  int field_size;
403  bool timestamp_found = findFieldOffset(orb_meta->o_fields, "timestamp", subscription->timestamp_offset, field_size);
404 
405  if (!timestamp_found) {
406  return true;
407  }
408 
409  if (field_size != 8) {
410  PX4_ERR("Unsupported timestamp with size %i, ignoring the topic %s", field_size, orb_meta->o_name);
411  return true;
412  }
413 
414  //find first data message (and the timestamp)
415  streampos cur_pos = file.tellg();
416  subscription->next_read_pos = this_message_pos; //this will be skipped
417 
418  if (!nextDataMessage(file, *subscription, msg_id)) {
419  return false;
420  }
421 
422  file.seekg(cur_pos);
423 
424  if (!subscription->orb_meta) {
425  //no message found. This is not a fatal error
426  return true;
427  }
428 
429  PX4_DEBUG("adding subscription for %s (msg_id %i)", subscription->orb_meta->o_name, msg_id);
430 
431  //add subscription
432  if (_subscriptions.size() <= msg_id) {
433  _subscriptions.resize(msg_id + 1);
434  }
435 
436  _subscriptions[msg_id] = subscription;
437 
438  onSubscriptionAdded(*_subscriptions[msg_id], msg_id);
439 
440  return true;
441 }

556 bool Replay::nextDataMessage(std::ifstream &file, Subscription &subscription, int msg_id)
557 {
558  ulog_message_header_s message_header;
559  file.seekg(subscription.next_read_pos);
560  //ignore the first message (it's data we already read)
561  file.read((char *)&message_header, ULOG_MSG_HEADER_LEN);
562 
563  if (file) {
564  file.seekg(message_header.msg_size, ios::cur);
565  }
566 
567  uint16_t file_msg_id;
568  bool done = false;
569 
570  while (file && !done) {
571  streampos cur_pos = file.tellg();
572  file.read((char *)&message_header, ULOG_MSG_HEADER_LEN);
573 
574  if (!file) {
575  break;
576  }
577 
578  if (((streamoff)cur_pos) + ULOG_MSG_HEADER_LEN + message_header.msg_size > _read_until_file_position) {
579  file.setstate(std::ios::eofbit);
580  break;
581  }
582 
583  switch (message_header.msg_type) {
584  case (int)ULogMessageType::ADD_LOGGED_MSG:
585  readAndAddSubscription(file, message_header.msg_size);
586  break;
587 
588  case (int)ULogMessageType::DATA:
589  file.read((char *)&file_msg_id, sizeof(file_msg_id));
590 
591  if (file) {
592  if (msg_id == file_msg_id) {
593  if (message_header.msg_size == subscription.orb_meta->o_size_no_padding + 2) {
594  subscription.next_read_pos = cur_pos;
595  file.seekg(subscription.timestamp_offset, ios::cur);
596  file.read((char *)&subscription.next_timestamp, sizeof(subscription.next_timestamp));
597  done = true;
598 
599  } else { //sanity check failed!
600  PX4_ERR("data message %s has wrong size %i (expected %i). Skipping",
601  subscription.orb_meta->o_name, message_header.msg_size,
602  subscription.orb_meta->o_size_no_padding + 2);
603  file.seekg(message_header.msg_size - sizeof(file_msg_id), ios::cur);
604  }
605 
606  } else { //not the one we are looking for
607  file.seekg(message_header.msg_size - sizeof(file_msg_id), ios::cur);
608  }
609  }
610 
611  break;
612 
613  case (int)ULogMessageType::REMOVE_LOGGED_MSG: //skip these
614  case (int)ULogMessageType::PARAMETER:
615  case (int)ULogMessageType::DROPOUT:
616  case (int)ULogMessageType::INFO:
617  case (int)ULogMessageType::INFO_MULTIPLE:
618  case (int)ULogMessageType::SYNC:
619  case (int)ULogMessageType::LOGGING:
620  file.seekg(message_header.msg_size, ios::cur);
621  break;
622 
623  default:
624  //this really should not happen
625  PX4_ERR("unknown log message type %i, size %i (offset %i)",
626  (int)message_header.msg_type, (int)message_header.msg_size, (int)file.tellg());
627  file.seekg(message_header.msg_size, ios::cur);
628  break;
629  }
630  }
631 
632  if (file.eof()) { //no more data messages for this subscription
633  subscription.orb_meta = nullptr;
634  file.clear();
635  }
636 
637  return file.good();
638 }

801  }
