93 MultirotorMixer::from_text(Mixer::ControlCallback control_cb, uintptr_t cb_handle, const char *buf, unsigned &buflen)
94 {
95  MultirotorGeometry geometry = MultirotorGeometry::MAX_GEOMETRY;
96  char geomname[8];
97  int s[4];
98  int used;
99 
100  /* enforce that the mixer ends with a new line */
101  if (!string_well_formed(buf, buflen)) {
102  return nullptr;
103  }
104 
105  if (sscanf(buf, "R: %7s %d %d %d %d%n", geomname, &s[0], &s[1], &s[2], &s[3], &used) != 5) {
106  debug("multirotor parse failed on '%s'", buf);
107  return nullptr;
108  }
109 
110  if (used > (int)buflen) {
111  debug("OVERFLOW: multirotor spec used %d of %u", used, buflen);
112  return nullptr;
113  }
114 
115  buf = skipline(buf, buflen);
116 
117  if (buf == nullptr) {
118  debug("no line ending, line is incomplete");
119  return nullptr;
120  }
121 
122  debug("remaining in buf: %d, first char: %c", buflen, buf[0]);
123 
124  for (MultirotorGeometryUnderlyingType i = 0; i < (MultirotorGeometryUnderlyingType)MultirotorGeometry::MAX_GEOMETRY;
125  i++) {
126  if (!strcmp(geomname, _config_key[i])) {
127  geometry = (MultirotorGeometry)i;
128  break;
129  }
130  }
131 
132  if (geometry == MultirotorGeometry::MAX_GEOMETRY) {
133  debug("unrecognised geometry '%s'", geomname);
134  return nullptr;
135  }
136 
137  debug("adding multirotor mixer '%s'", geomname);
138 
139  return new MultirotorMixer(
140  control_cb,
141  cb_handle,
142  geometry,
143  s[0] / 10000.0f,
144  s[1] / 10000.0f,

149 unsigned
150 MultirotorMixer::mix(float *outputs, unsigned space)
151 {
152  /* Summary of mixing strategy:
153  1) mix roll, pitch and thrust without yaw.
154  2) if some outputs violate range [0,1] then try to shift all outputs to minimize violation ->
155  increase or decrease total thrust (boost). The total increase or decrease of thrust is limited
156  (max_thrust_diff). If after the shift some outputs still violate the bounds then scale roll & pitch.
157  In case there is violation at the lower and upper bound then try to shift such that violation is equal
158  on both sides.
159  3) mix in yaw and scale if it leads to limit violation.
160  4) scale all outputs to range [idle_speed,1]
161  */
162 
163  float roll = math::constrain(get_control(0, 0) * _roll_scale, -1.0f, 1.0f);
164  float pitch = math::constrain(get_control(0, 1) * _pitch_scale, -1.0f, 1.0f);
165  float yaw = math::constrain(get_control(0, 2) * _yaw_scale, -1.0f, 1.0f);
166  float thrust = math::constrain(get_control(0, 3), 0.0f, 1.0f);
167  float min_out = 1.0f;
168  float max_out = 0.0f;
169 
170  // clean out class variable used to capture saturation
171  _saturation_status.value = 0;
172 
173  /* perform initial mix pass yielding unbounded outputs, ignore yaw */
174  for (unsigned i = 0; i < _rotor_count; i++) {
175  float out = roll * _rotors[i].roll_scale +
176  pitch * _rotors[i].pitch_scale +
177  thrust * _rotors[i].thrust_scale;
178 
179  /* calculate min and max output values */
180  if (out < min_out) {
181  min_out = out;
182  }
183 
184  if (out > max_out) {
185  max_out = out;
186  }
187 
188  outputs[i] = out;
189  }
190 
191  float boost = 0.0f; // value added to demanded thrust (can also be negative)
192  float roll_pitch_scale = 1.0f; // scale for demanded roll and pitch
193  float delta_out_max = max_out - min_out; // distance between the two extrema
194 
195  // If the difference between the to extrema is smaller than 1.0, the boost can safely unsaturate a motor if needed
196  // without saturating another one.
197  // Otherwise, a scaler is computed to make the distance between the two extrema exacly 1.0 and the boost
198  // value is computed to maximize the roll-pitch control.
199  //
200  // Note: thrust boost is computed assuming thrust_scale==1 for all motors.
201  // On asymmetric platforms, some motors have thrust_scale<1,
202  // which may result in motor saturation after thrust boost is applied
203  // TODO: revise the saturation/boosting strategy
204  if (delta_out_max <= 1.0f) {
205  if (min_out < 0.0f) {
206  boost = -min_out;
207 
208  } else if (max_out > 1.0f) {
209  boost = -(max_out - 1.0f);
210  }
211 
212  } else {
213  roll_pitch_scale = 1.0f / (delta_out_max);
214  boost = 1.0f - ((max_out - thrust) * roll_pitch_scale + thrust);
215  }
216 
217  if (!_airmode) {
218  // disable positive boosting if not in air-mode
219  // boosting can only be positive when min_out < 0.0
220  // roll_pitch_scale is reduced accordingly
221  if (boost > 0.0f) {
222  roll_pitch_scale = thrust / (thrust - min_out);
223  boost = 0.0f;
224  }
225  }
226 
227  // capture saturation
228  if (min_out < 0.0f) {
229  _saturation_status.flags.motor_neg = true;
230  }
231 
232  if (max_out > 1.0f) {
233  _saturation_status.flags.motor_pos = true;
234  }
235 
236  // Thrust reduction is used to reduce the collective thrust if we hit
237  // the upper throttle limit
238  float thrust_reduction = 0.0f;
239 
240  // mix again but now with thrust boost, scale roll/pitch and also add yaw
241  for (unsigned i = 0; i < _rotor_count; i++) {
242  float out = (roll * _rotors[i].roll_scale +
243  pitch * _rotors[i].pitch_scale) * roll_pitch_scale +
244  yaw * _rotors[i].yaw_scale +
245  (thrust + boost) * _rotors[i].thrust_scale;
246 
247  // scale yaw if it violates limits. inform about yaw limit reached
248  if (out < 0.0f) {
249  if (fabsf(_rotors[i].yaw_scale) <= FLT_EPSILON) {
250  yaw = 0.0f;
251 
252  } else {
253  yaw = -((roll * _rotors[i].roll_scale + pitch * _rotors[i].pitch_scale) *
254  roll_pitch_scale + thrust + boost) / _rotors[i].yaw_scale;
255  }
256 
257  } else if (out > 1.0f) {
258  // allow to reduce thrust to get some yaw response
259  float prop_reduction = fminf(0.15f, out - 1.0f);
260  // keep the maximum requested reduction
261  thrust_reduction = fmaxf(thrust_reduction, prop_reduction);
262 
263  if (fabsf(_rotors[i].yaw_scale) <= FLT_EPSILON) {
264  yaw = 0.0f;
265 
266  } else {
267  yaw = (1.0f - ((roll * _rotors[i].roll_scale + pitch * _rotors[i].pitch_scale) *
268  roll_pitch_scale + (thrust - thrust_reduction) + boost)) / _rotors[i].yaw_scale;
269  }
270  }
271  }
272 
273  // Apply collective thrust reduction, the maximum for one prop
274  thrust -= thrust_reduction;
275 
276  // add yaw and scale outputs to range idle_speed...1
277  for (unsigned i = 0; i < _rotor_count; i++) {
278  outputs[i] = (roll * _rotors[i].roll_scale +
279  pitch * _rotors[i].pitch_scale) * roll_pitch_scale +
280  yaw * _rotors[i].yaw_scale +
281  (thrust + boost) * _rotors[i].thrust_scale;
282 
283  /*
284  implement simple model for static relationship between applied motor pwm and motor thrust
285  model: thrust = (1 - _thrust_factor) * PWM + _thrust_factor * PWM^2
286  this model assumes normalized input / output in the range [0,1] so this is the right place
287  to do it as at this stage the outputs are in that range.
288  */
289  if (_thrust_factor > 0.0f) {
290  outputs[i] = -(1.0f - _thrust_factor) / (2.0f * _thrust_factor) + sqrtf((1.0f - _thrust_factor) *
291  (1.0f - _thrust_factor) / (4.0f * _thrust_factor * _thrust_factor) + (outputs[i] < 0.0f ? 0.0f : outputs[i] /
292  _thrust_factor));
293  }
294 
295  outputs[i] = math::constrain(_idle_speed + (outputs[i] * (1.0f - _idle_speed)), _idle_speed, 1.0f);
296 
297  }
298 
299  /* slew rate limiting and saturation checking */
300  for (unsigned i = 0; i < _rotor_count; i++) {
301  bool clipping_high = false;
302  bool clipping_low = false;
303 
304  // check for saturation against static limits
305  if (outputs[i] > 0.99f) {
306  clipping_high = true;
307 
308  } else if (outputs[i] < _idle_speed + 0.01f) {
309  clipping_low = true;
310 
311  }
312 
313  // check for saturation against slew rate limits
314  if (_delta_out_max > 0.0f) {
315  float delta_out = outputs[i] - _outputs_prev[i];
316 
317  if (delta_out > _delta_out_max) {
318  outputs[i] = _outputs_prev[i] + _delta_out_max;
319  clipping_high = true;
320 
321  } else if (delta_out < -_delta_out_max) {
322  outputs[i] = _outputs_prev[i] - _delta_out_max;
323  clipping_low = true;
324 
325  }
326  }
327 
328  _outputs_prev[i] = outputs[i];
329 
330  // update the saturation status report
331  update_saturation_status(i, clipping_high, clipping_low);
332  }
333 
334  // this will force the caller of the mixer to always supply new slew rate values, otherwise no slew rate limiting will happen
335  _delta_out_max = 0.0f;
336 
337  return _rotor_count;
338 }

340 /*
341  * This function update the control saturation status report using the following inputs:
342  *
343  * index: 0 based index identifying the motor that is saturating
344  * clipping_high: true if the motor demand is being limited in the positive direction
345  * clipping_low: true if the motor demand is being limited in the negative direction
346 */
347 void
348 MultirotorMixer::update_saturation_status(unsigned index, bool clipping_high, bool clipping_low)
349 {
350  // The motor is saturated at the upper limit
351  // check which control axes and which directions are contributing
352  if (clipping_high) {
353  if (_rotors[index].roll_scale > 0.0f) {
354  // A positive change in roll will increase saturation
355  _saturation_status.flags.roll_pos = true;
356 
357  } else if (_rotors[index].roll_scale < 0.0f) {
358  // A negative change in roll will increase saturation
359  _saturation_status.flags.roll_neg = true;
360  }
361 
362  // check if the pitch input is saturating
363  if (_rotors[index].pitch_scale > 0.0f) {
364  // A positive change in pitch will increase saturation
365  _saturation_status.flags.pitch_pos = true;
366 
367  } else if (_rotors[index].pitch_scale < 0.0f) {
368  // A negative change in pitch will increase saturation
369  _saturation_status.flags.pitch_neg = true;
370  }
371 
372  // check if the yaw input is saturating
373  if (_rotors[index].yaw_scale > 0.0f) {
374  // A positive change in yaw will increase saturation
375  _saturation_status.flags.yaw_pos = true;
376 
377  } else if (_rotors[index].yaw_scale < 0.0f) {
378  // A negative change in yaw will increase saturation
379  _saturation_status.flags.yaw_neg = true;
380  }
381 
382  // A positive change in thrust will increase saturation
383  _saturation_status.flags.thrust_pos = true;
384 
385  }
386 
387  // The motor is saturated at the lower limit
388  // check which control axes and which directions are contributing
389  if (clipping_low) {
390  // check if the roll input is saturating
391  if (_rotors[index].roll_scale > 0.0f) {
392  // A negative change in roll will increase saturation
393  _saturation_status.flags.roll_neg = true;
394 
395  } else if (_rotors[index].roll_scale < 0.0f) {
396  // A positive change in roll will increase saturation
397  _saturation_status.flags.roll_pos = true;
398  }
399 
400  // check if the pitch input is saturating
401  if (_rotors[index].pitch_scale > 0.0f) {
402  // A negative change in pitch will increase saturation
403  _saturation_status.flags.pitch_neg = true;
404 
405  } else if (_rotors[index].pitch_scale < 0.0f) {
406  // A positive change in pitch will increase saturation
407  _saturation_status.flags.pitch_pos = true;
408  }
409 
410  // check if the yaw input is saturating
411  if (_rotors[index].yaw_scale > 0.0f) {
412  // A negative change in yaw will increase saturation
413  _saturation_status.flags.yaw_neg = true;
414 
415  } else if (_rotors[index].yaw_scale < 0.0f) {
416  // A positive change in yaw will increase saturation
417  _saturation_status.flags.yaw_pos = true;
418  }
419 
420  // A negative change in thrust will increase saturation
421  _saturation_status.flags.thrust_neg = true;
422  }
423 
424  _saturation_status.flags.valid = true;
425 }
