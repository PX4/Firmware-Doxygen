161 int BMI055_gyro::reset()
162 {
163  write_reg(BMI055_GYR_SOFTRESET, BMI055_SOFT_RESET);//Soft-reset
164  usleep(5000);
165  write_checked_reg(BMI055_GYR_BW, 0); // Write Gyro Bandwidth
166  write_checked_reg(BMI055_GYR_RANGE, 0);// Write Gyro range
167  write_checked_reg(BMI055_GYR_INT_EN_0, BMI055_GYR_DRDY_INT_EN); //Enable DRDY interrupt
168  write_checked_reg(BMI055_GYR_INT_MAP_1, BMI055_GYR_DRDY_INT1); //Map DRDY interrupt on pin INT1
169 
170  set_gyro_range(BMI055_GYRO_DEFAULT_RANGE_DPS);// set Gyro range
171  gyro_set_sample_rate(BMI055_GYRO_DEFAULT_RATE);// set Gyro ODR
172 
173  //Enable Gyroscope in normal mode
174  write_reg(BMI055_GYR_LPM1, BMI055_GYRO_NORMAL);
175  up_udelay(1000);
176 
177  uint8_t retries = 10;
178 
179  while (retries--) {
180  bool all_ok = true;
181 
182  for (uint8_t i = 0; i < BMI055_GYRO_NUM_CHECKED_REGISTERS; i++) {
183  if (read_reg(_checked_registers[i]) != _checked_values[i]) {
184  write_reg(_checked_registers[i], _checked_values[i]);
185  all_ok = false;
186  }
187  }
188 
189  if (all_ok) {
190  break;
191  }
192  }
193 
194  return OK;
195 }

270 ssize_t
271 BMI055_gyro::read(struct file *filp, char *buffer, size_t buflen)
272 {
273  unsigned count = buflen / sizeof(gyro_report);
274 
275  /* buffer must be large enough */
276  if (count < 1) {
277  return -ENOSPC;
278  }
279 
280  /* if automatic measurement is not enabled, get a fresh measurement into the buffer */
281  if (_call_interval == 0) {
282  _gyro_reports->flush();
283  measure();
284  }
285 
286  /* if no data, error (we could block here) */
287  if (_gyro_reports->empty()) {
288  return -EAGAIN;
289  }
290 
291  /* copy reports out of our buffer to the caller */
292  gyro_report *grp = reinterpret_cast<gyro_report *>(buffer);
293  int transferred = 0;
294 
295  while (count--) {
296  if (!_gyro_reports->get(grp)) {
297  break;
298  }
299 
300  transferred++;
301  grp++;
302  }
303 
304  /* return the number of bytes transferred */
305  return (transferred * sizeof(gyro_report));
306 }

308 int
309 BMI055_gyro::ioctl(struct file *filp, int cmd, unsigned long arg)
310 {
311  switch (cmd) {
312 
313  case SENSORIOCRESET:
314  return reset();
315 
316  case SENSORIOCSPOLLRATE: {
317  switch (arg) {
318 
319  /* switching to manual polling */
320  case SENSOR_POLLRATE_MANUAL:
321  stop();
322  _call_interval = 0;
323  return OK;
324 
325  /* external signalling not supported */
326  case SENSOR_POLLRATE_EXTERNAL:
327 
328  /* zero would be bad */
329  case 0:
330  return -EINVAL;
331 
332  /* set default/max polling rate */
333  case SENSOR_POLLRATE_MAX:
334  return ioctl(filp, SENSORIOCSPOLLRATE, BMI055_GYRO_MAX_RATE);
335 
336  case SENSOR_POLLRATE_DEFAULT:
337  return ioctl(filp, SENSORIOCSPOLLRATE, BMI055_GYRO_DEFAULT_RATE);
338 
339  /* adjust to a legal polling interval in Hz */
340  default: {
341  /* do we need to start internal polling? */
342  bool want_start = (_call_interval == 0);
343 
344  /* convert hz to hrt interval via microseconds */
345  unsigned ticks = 1000000 / arg;
346 
347  /* check against maximum rate */
348  if (ticks < 1000) {
349  return -EINVAL;
350  }
351 
352  // adjust filters
353  float cutoff_freq_hz_gyro = _gyro_filter_x.get_cutoff_freq();
354  float sample_rate = 1.0e6f / ticks;
355  _gyro_filter_x.set_cutoff_frequency(sample_rate, cutoff_freq_hz_gyro);
356  _gyro_filter_y.set_cutoff_frequency(sample_rate, cutoff_freq_hz_gyro);
357  _gyro_filter_z.set_cutoff_frequency(sample_rate, cutoff_freq_hz_gyro);
358 
359  /* update interval for next measurement */
360  _call_interval = ticks;
361 
362  /*
363  set call interval faster than the sample time. We
364  then detect when we have duplicate samples and reject
365  them. This prevents aliasing due to a beat between the
366  stm32 clock and the bmi055 clock
367  */
368  _call.period = _call_interval - BMI055_TIMER_REDUCTION;
369 
370  /* if we need to start the poll state machine, do it */
371  if (want_start) {
372  start();
373  }
374 
375  return OK;
376  }
377  }
378  }
379 
380  case SENSORIOCGPOLLRATE:
381  if (_call_interval == 0) {
382  return SENSOR_POLLRATE_MANUAL;
383  }
384 
385  return 1000000 / _call_interval;
386 
387  case SENSORIOCSQUEUEDEPTH: {
388  /* lower bound is mandatory, upper bound is a sanity check */
389  if ((arg < 1) || (arg > 100)) {
390  return -EINVAL;
391  }
392 
393  irqstate_t flags = px4_enter_critical_section();
394 
395  if (!_gyro_reports->resize(arg)) {
396  px4_leave_critical_section(flags);
397  return -ENOMEM;
398  }
399 
400  px4_leave_critical_section(flags);
401 
402  return OK;
403  }
404 
405  case GYROIOCGSAMPLERATE:
406  return _gyro_sample_rate;
407 
408  case GYROIOCSSAMPLERATE:
409  return gyro_set_sample_rate(arg);
410 
411  case GYROIOCSSCALE:
412  /* copy scale in */
413  memcpy(&_gyro_scale, (struct gyro_calibration_s *) arg, sizeof(_gyro_scale));
414  return OK;
415 
416  case GYROIOCGSCALE:
417  /* copy scale out */
418  memcpy((struct gyro_calibration_s *) arg, &_gyro_scale, sizeof(_gyro_scale));
419  return OK;
420 
421  case GYROIOCSRANGE:
422  return set_gyro_range(arg);
423 
424  case GYROIOCGRANGE:
425  return (unsigned long)(_gyro_range_rad_s * 180.0f / M_PI_F + 0.5f);
426 
427  default:
428  /* give it to the superclass */
429  return SPI::ioctl(filp, cmd, arg);
430  }
431 }

538 void
539 BMI055_gyro::check_registers(void)
540 {
541  uint8_t v;
542 
543  if ((v = read_reg(_checked_registers[_checked_next])) !=
544  _checked_values[_checked_next]) {
545  _checked_bad[_checked_next] = v;
546 
547  /*
548  if we get the wrong value then we know the SPI bus
549  or sensor is very sick. We set _register_wait to 20
550  and wait until we have seen 20 good values in a row
551  before we consider the sensor to be OK again.
552  */
553  perf_count(_bad_registers);
554 
555  /*
556  try to fix the bad register value. We only try to
557  fix one per loop to prevent a bad sensor hogging the
558  bus.
559  */
560  if (_register_wait == 0 || _checked_next == 0) {
561  // if the product_id is wrong then reset the
562  // sensor completely
563  write_reg(BMI055_GYR_SOFTRESET, BMI055_SOFT_RESET);
564  _reset_wait = hrt_absolute_time() + 10000;
565  _checked_next = 0;
566 
567  } else {
568  write_reg(_checked_registers[_checked_next], _checked_values[_checked_next]);
569  // waiting 3ms between register writes seems
570  // to raise the chance of the sensor
571  // recovering considerably
572  _reset_wait = hrt_absolute_time() + 3000;
573  }
574 
575  _register_wait = 20;
576  }
577 
578  _checked_next = (_checked_next + 1) % BMI055_GYRO_NUM_CHECKED_REGISTERS;
579 }

581 void
582 BMI055_gyro::measure()
583 {
584  perf_count(_measure_interval);
585 
586  if (hrt_absolute_time() < _reset_wait) {
587  // we're waiting for a reset to complete
588  return;
589  }

591  struct BMI_GyroReport bmi_gyroreport;
592 
593  struct Report {
594  int16_t temp;
595  int16_t gyro_x;
596  int16_t gyro_y;
597  int16_t gyro_z;
598  } report;
599 
600  /* start measuring */
601  perf_begin(_sample_perf);
602 
603  /*
604  * Fetch the full set of measurements from the BMI055 gyro in one pass.
605  */
606  bmi_gyroreport.cmd = BMI055_GYR_X_L | DIR_READ;
607 
608 
609  if (OK != transfer((uint8_t *)&bmi_gyroreport, ((uint8_t *)&bmi_gyroreport), sizeof(bmi_gyroreport))) {
610  return;
611  }
612 
613  check_registers();
614 
615  uint8_t temp = read_reg(BMI055_ACC_TEMP);
616 
617  report.temp = temp;
618 
619  report.gyro_x = bmi_gyroreport.gyro_x;
620  report.gyro_y = bmi_gyroreport.gyro_y;
621  report.gyro_z = bmi_gyroreport.gyro_z;
622 
623  if (report.temp == 0 &&
624  report.gyro_x == 0 &&
625  report.gyro_y == 0 &&
626  report.gyro_z == 0) {
627  // all zero data - probably a SPI bus error
628  perf_count(_bad_transfers);
629  perf_end(_sample_perf);
630  // note that we don't call reset() here as a reset()
631  // costs 20ms with interrupts disabled. That means if
632  // the bmi055 does go bad it would cause a FMU failure,
633  // regardless of whether another sensor is available,
634  return;
635  }
636 
637  if (_register_wait != 0) {
638  // we are waiting for some good transfers before using
639  // the sensor again, but don't return any data yet
640  _register_wait--;
641  return;
642  }
643 
644  /*
645  * Report buffers.
646  */
647  gyro_report grb;
648 
649  grb.timestamp = hrt_absolute_time();
650 
651  // report the error count as the sum of the number of bad
652  // transfers and bad register reads. This allows the higher
653  // level code to decide if it should use this sensor based on
654  // whether it has had failures
655  grb.error_count = perf_event_count(_bad_transfers) + perf_event_count(_bad_registers);
656 
657  /*
658  * 1) Scale raw value to SI units using scaling from datasheet.
659  * 2) Subtract static offset (in SI units)
660  * 3) Scale the statically calibrated values with a linear
661  * dynamically obtained factor
662  *
663  * Note: the static sensor offset is the number the sensor outputs
664  * at a nominally 'zero' input. Therefore the offset has to
665  * be subtracted.
666  *
667  * Example: A gyro outputs a value of 74 at zero angular rate
668  * the offset is 74 from the origin and subtracting
669  * 74 from all measurements centers them around zero.
670  */
671 
672  grb.x_raw = report.gyro_x;
673  grb.y_raw = report.gyro_y;
674  grb.z_raw = report.gyro_z;
675 
676  float xraw_f = report.gyro_x;
677  float yraw_f = report.gyro_y;
678  float zraw_f = report.gyro_z;
679 
680  // apply user specified rotation
681  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
682 
683  float x_gyro_in_new = ((xraw_f * _gyro_range_scale) - _gyro_scale.x_offset) * _gyro_scale.x_scale;
684  float y_gyro_in_new = ((yraw_f * _gyro_range_scale) - _gyro_scale.y_offset) * _gyro_scale.y_scale;
685  float z_gyro_in_new = ((zraw_f * _gyro_range_scale) - _gyro_scale.z_offset) * _gyro_scale.z_scale;
686 
687  grb.x = _gyro_filter_x.apply(x_gyro_in_new);
