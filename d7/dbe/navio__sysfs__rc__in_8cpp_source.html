107 int RcInput::navio_rc_init()
108 {
109  int i;
110  char buf[64];
111 
112  for (i = 0; i < _channels; ++i) {
113  ::snprintf(buf, sizeof(buf), "%s/ch%d", RCINPUT_DEVICE_PATH_BASE, i);
114  int fd = ::open(buf, O_RDONLY);
115 
116  if (fd < 0) {
117  PX4_WARN("error: open %d failed", i);
118  break;
119  }
120 
121  _ch_fd[i] = fd;
122  }
123 
124  for (; i < input_rc_s::RC_INPUT_MAX_CHANNELS; ++i) {
125  _data.values[i] = UINT16_MAX;
126  }
127 
128  _rcinput_pub = orb_advertise(ORB_ID(input_rc), &_data);
129 
130  if (_rcinput_pub == nullptr) {
131  PX4_WARN("error: advertise failed");
132  return -1;
133  }
134 
135  return 0;
136 }

180 void RcInput::_measure(void)
181 {
182  uint64_t ts;
183  char buf[12];
184 
185  for (int i = 0; i < _channels; ++i) {
186  int res;
187 
188  if ((res = ::pread(_ch_fd[i], buf, sizeof(buf) - 1, 0)) < 0) {
189  _data.values[i] = UINT16_MAX;
190  continue;
191  }
192 
193  buf[sizeof(buf) - 1] = '\0';
194 
195  _data.values[i] = atoi(buf);
196  }
197 
198  ts = hrt_absolute_time();
199  _data.timestamp = ts;
200  _data.timestamp_last_signal = ts;
201  _data.channel_count = _channels;
202  _data.rssi = 100;
203  _data.rc_lost_frame_count = 0;
204  _data.rc_total_frame_count = 1;
205  _data.rc_ppm_frame_length = 100;
206  _data.rc_failsafe = false;
207  _data.rc_lost = false;
208  _data.input_source = input_rc_s::RC_INPUT_SOURCE_PX4IO_PPM;
209 
210  orb_publish(ORB_ID(input_rc), _rcinput_pub, &_data);
211 }

284  return 0;

286 
287  if (!strcmp(argv[1], "status")) {
288  if (rc_input != nullptr && rc_input->isRunning()) {
289  PX4_INFO("running");
