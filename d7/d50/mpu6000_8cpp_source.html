37  * Driver for the Invensense MPU6000, MPU6050, ICM20608, and ICM20602 connected via

39  *
40  * When the device is on the SPI bus the hrt is used to provide thread of

49  *
50  * The command line option -T 6000|20608|20602 (default 6000) selects between

742 int MPU6000::reset()
743 {
744  // if the mpu6000 is initialized after the l3gd20 and lsm303d
745  // then if we don't do an irqsave/irqrestore here the mpu6000
746  // frequently comes up in a bad state where all transfers
747  // come as zero
748  uint8_t tries = 5;
749  irqstate_t state;
750 
751  while (--tries != 0) {
752  state = px4_enter_critical_section();
753 
754  // Hold off sampling for 60 ms
755  _reset_wait = hrt_absolute_time() + 60000;
756 
757  write_reg(MPUREG_PWR_MGMT_1, BIT_H_RESET);
758  up_udelay(10000);
759 
760  // Wake up device and select GyroZ clock. Note that the
761  // MPU6000 starts up in sleep mode, and it can take some time
762  // for it to come out of sleep
763  write_checked_reg(MPUREG_PWR_MGMT_1, MPU_CLK_SEL_PLLGYROZ);
764  up_udelay(1000);
765 
766  // Enable I2C bus or Disable I2C bus (recommended on data sheet)
767  write_checked_reg(MPUREG_USER_CTRL, is_i2c() ? 0 : BIT_I2C_IF_DIS);
768 
769  px4_leave_critical_section(state);
770 
771  if (read_reg(MPUREG_PWR_MGMT_1) == MPU_CLK_SEL_PLLGYROZ) {
772  break;
773  }
774 
775  perf_count(_reset_retries);
776  usleep(2000);
777  }
778 
779  // Hold off sampling for 30 ms
780 
781  state = px4_enter_critical_section();
782  _reset_wait = hrt_absolute_time() + 30000;
783  px4_leave_critical_section(state);
784 
785  if (read_reg(MPUREG_PWR_MGMT_1) != MPU_CLK_SEL_PLLGYROZ) {
786  return -EIO;
787  }
788 
789  usleep(1000);
790 
791  // SAMPLE RATE
792  _set_sample_rate(_sample_rate);
793  usleep(1000);
794 
795  _set_dlpf_filter(MPU6000_DEFAULT_ONCHIP_FILTER_FREQ);
796 
797  if (is_icm_device()) {
798  _set_icm_acc_dlpf_filter(MPU6000_DEFAULT_ONCHIP_FILTER_FREQ);
799  }
800 
801  usleep(1000);
802  // Gyro scale 2000 deg/s ()
803  write_checked_reg(MPUREG_GYRO_CONFIG, BITS_FS_2000DPS);
804  usleep(1000);
805 
806  // correct gyro scale factors
807  // scale to rad/s in SI units
808  // 2000 deg/s = (2000/180)*PI = 34.906585 rad/s
809  // scaling factor:
810  // 1/(2^15)*(2000/180)*PI
811  _gyro_range_scale = (0.0174532 / 16.4);//1.0f / (32768.0f * (2000.0f / 180.0f) * M_PI_F);
812  _gyro_range_rad_s = (2000.0f / 180.0f) * M_PI_F;
813 
814  set_accel_range(MPU6000_ACCEL_DEFAULT_RANGE_G);
815 
816  usleep(1000);
817 
818  // INT CFG => Interrupt on Data Ready
819  write_checked_reg(MPUREG_INT_ENABLE, BIT_RAW_RDY_EN); // INT: Raw data ready
820  usleep(1000);
821  write_checked_reg(MPUREG_INT_PIN_CFG, BIT_INT_ANYRD_2CLEAR); // INT: Clear on any read
822  usleep(1000);
823 
824  if (is_icm_device()) {
825  write_checked_reg(MPUREG_ICM_UNDOC1, MPUREG_ICM_UNDOC1_VALUE);
826  }
827 
828  // Oscillator set
829  // write_reg(MPUREG_PWR_MGMT_1,MPU_CLK_SEL_PLLGYROZ);
830  usleep(1000);
831  return OK;

1080  float gyro_baseline[3];
1081  float accel[3];
1082  float gyro[3];
1083  float accel_ftrim[3];
1084  float gyro_ftrim[3];

1087

1089  memset(gyro_baseline, 0, sizeof(gyro_baseline));

1099  accel_baseline[1] += int16_t_from_bytes(mpu_report.accel_y);

1101  gyro_baseline[0] += int16_t_from_bytes(mpu_report.gyro_x);
1102  gyro_baseline[1] += int16_t_from_bytes(mpu_report.gyro_y);
1103  gyro_baseline[2] += int16_t_from_bytes(mpu_report.gyro_z);
1104  }

1106 #if 1
1107  write_reg(MPUREG_GYRO_CONFIG,

1109  BITS_GYRO_ST_X |
1110  BITS_GYRO_ST_Y |
1111  BITS_GYRO_ST_Z);

1172  (int)(1000 * accel[i]),
1173  (int)(1000 * diff),
1174  (int)(1000 * accel_ftrim[i]),
1175  (int)err);
1176

1178  ::printf("FAIL\n");

1180  }
1181  }
1182

1187  (unsigned)i,
1188  (int)(1000 * gyro_baseline[i]),
1189  (int)(1000 * gyro[i]),
1190  (int)(1000 * (gyro[i] - gyro_baseline[i])),
1191  (int)(1000 * gyro_ftrim[i]),
1192  (int)err);
1193 
1194  if (fabsf(err) > 14) {
1195  ::printf("FAIL\n");
1196  ret = -EIO;
1197  }
1198  }
1199 
1200  write_reg(MPUREG_GYRO_CONFIG, saved_gyro_config);
1201  write_reg(MPUREG_ACCEL_CONFIG, saved_accel_config);
1202 
1203  _in_factory_test = false;
1204 
1205  if (ret == OK) {
1206  ::printf("PASSED\n");
1207  }
1208 
1209  return ret;
1210 }

1673 void
1674 MPU6000::check_registers(void)
1675 {
1676  /*
1677  we read the register at full speed, even though it isn't
1678  listed as a high speed register. The low speed requirement
1679  for some registers seems to be a propgation delay
1680  requirement for changing sensor configuration, which should
1681  not apply to reading a single register. It is also a better
1682  test of SPI bus health to read at the same speed as we read
1683  the data registers.
1684  */
1685  uint8_t v;
1686 
1687  // the MPUREG_ICM_UNDOC1 is specific to the ICM20608 (and undocumented)
1688  if ((_checked_registers[_checked_next] == MPUREG_ICM_UNDOC1 && !is_icm_device())) {
1689  _checked_next = (_checked_next + 1) % MPU6000_NUM_CHECKED_REGISTERS;
1690  }
1691 
1692  if ((v = read_reg(_checked_registers[_checked_next], MPU6000_HIGH_BUS_SPEED)) !=
1693  _checked_values[_checked_next]) {
1694  /*
1695  if we get the wrong value then we know the SPI bus
1696  or sensor is very sick. We set _register_wait to 20
1697  and wait until we have seen 20 good values in a row
1698  before we consider the sensor to be OK again.
1699  */
1700  perf_count(_bad_registers);
1701 
1702  /*
1703  try to fix the bad register value. We only try to
1704  fix one per loop to prevent a bad sensor hogging the
1705  bus.
1706  */
1707  if (_register_wait == 0 || _checked_next == MPU6000_CHECKED_PRODUCT_ID_INDEX) {
1708  // if the product_id is wrong then reset the
1709  // sensor completely
1710  write_reg(MPUREG_PWR_MGMT_1, BIT_H_RESET);
1711  // after doing a reset we need to wait a long
1712  // time before we do any other register writes
1713  // or we will end up with the mpu6000 in a
1714  // bizarre state where it has all correct
1715  // register values but large offsets on the
1716  // accel axes
1717  _reset_wait = hrt_absolute_time() + 10000;
1718  _checked_next = 0;
1719 
1720  } else {
1721  write_reg(_checked_registers[_checked_next], _checked_values[_checked_next]);
1722  // waiting 3ms between register writes seems
1723  // to raise the chance of the sensor
1724  // recovering considerably
1725  _reset_wait = hrt_absolute_time() + 3000;
1726  }
1727 
1728  _register_wait = 20;
1729  }
1730 
1731  _checked_next = (_checked_next + 1) % MPU6000_NUM_CHECKED_REGISTERS;
1732 }

1738

1740  // don't publish any data while in factory test mode
1741  return OK;

1749  struct MPUReport mpu_report;
1750 
1751  struct Report {
1752  int16_t accel_x;
1753  int16_t accel_y;
1754  int16_t accel_z;
1755  int16_t temp;
1756  int16_t gyro_x;
1757  int16_t gyro_y;
1758  int16_t gyro_z;
1759  } report;
1760 
1761  /* start measuring */
1762  perf_begin(_sample_perf);
1763 
1764  /*
1765  * Fetch the full set of measurements from the MPU6000 in one pass.
1766  */
1767 
1768  // sensor transfer at high clock speed
1769 
1770  if (sizeof(mpu_report) != _interface->read(MPU6000_SET_SPEED(MPUREG_INT_STATUS, MPU6000_HIGH_BUS_SPEED),
1771  (uint8_t *)&mpu_report,
1772  sizeof(mpu_report))) {
1773  return -EIO;
1774  }
1775 
1776  check_registers();
1777 
1778  /*
1779  see if this is duplicate accelerometer data. Note that we
1780  can't use the data ready interrupt status bit in the status
1781  register as that also goes high on new gyro data, and when
1782  we run with BITS_DLPF_CFG_256HZ_NOLPF2 the gyro is being
1783  sampled at 8kHz, so we would incorrectly think we have new
1784  data when we are in fact getting duplicate accelerometer data.
1785  */
1786  if (!_got_duplicate && memcmp(&mpu_report.accel_x[0], &_last_accel[0], 6) == 0) {
1787  // it isn't new data - wait for next timer
1788  perf_end(_sample_perf);
1789  perf_count(_duplicates);
1790  _got_duplicate = true;
1791  return OK;
1792  }
1793 
1794  memcpy(&_last_accel[0], &mpu_report.accel_x[0], 6);
1795  _got_duplicate = false;
1796 
1797  /*
1798  * Convert from big to little endian
1799  */
1800 
1801  report.accel_x = int16_t_from_bytes(mpu_report.accel_x);
1802  report.accel_y = int16_t_from_bytes(mpu_report.accel_y);
1803  report.accel_z = int16_t_from_bytes(mpu_report.accel_z);
1804 
1805  report.temp = int16_t_from_bytes(mpu_report.temp);
1806 
1807  report.gyro_x = int16_t_from_bytes(mpu_report.gyro_x);
1808  report.gyro_y = int16_t_from_bytes(mpu_report.gyro_y);
1809  report.gyro_z = int16_t_from_bytes(mpu_report.gyro_z);
1810 
1811  if (report.accel_x == 0 &&
1812  report.accel_y == 0 &&
1813  report.accel_z == 0 &&
1814  report.temp == 0 &&
1815  report.gyro_x == 0 &&
1816  report.gyro_y == 0 &&
1817  report.gyro_z == 0) {
1818  // all zero data - probably a SPI bus error
1819  perf_count(_bad_transfers);
1820  perf_end(_sample_perf);
1821  // note that we don't call reset() here as a reset()
1822  // costs 20ms with interrupts disabled. That means if
1823  // the mpu6k does go bad it would cause a FMU failure,
1824  // regardless of whether another sensor is available,
1825  return -EIO;
1826  }
1827 
1828  if (_register_wait != 0) {
1829  // we are waiting for some good transfers before using
1830  // the sensor again, don't return any data yet
1831  _register_wait--;
1832  return OK;
1833  }
1834 
1835 
1836  /*
1837  * Swap axes and negate y
1838  */
1839  int16_t accel_xt = report.accel_y;
1840  int16_t accel_yt = ((report.accel_x == -32768) ? 32767 : -report.accel_x);
1841 
1842  int16_t gyro_xt = report.gyro_y;
1843  int16_t gyro_yt = ((report.gyro_x == -32768) ? 32767 : -report.gyro_x);
1844

1846  * Apply the swap
1847  */

1849  report.accel_y = accel_yt;
1850  report.gyro_x = gyro_xt;
1851  report.gyro_y = gyro_yt;

1854  * Report buffers.
1855  */

1857  gyro_report grb;
1858

1860  * Adjust and scale results to m/s^2.
1861  */

1865  // transfers and bad register reads. This allows the higher
1866  // level code to decide if it should use this sensor based on
1867  // whether it has had failures
1868  grb.error_count = arb.error_count = perf_event_count(_bad_transfers) + perf_event_count(_bad_registers);
1869

1871  * 1) Scale raw value to SI units using scaling from datasheet.
1872  * 2) Subtract static offset (in SI units)
1873  * 3) Scale the statically calibrated values with a linear
1874  * dynamically obtained factor
1875  *
1876  * Note: the static sensor offset is the number the sensor outputs
1877  * at a nominally 'zero' input. Therefore the offset has to
1878  * be subtracted.
1879  *
1880  * Example: A gyro outputs a value of 74 at zero angular rate
1881  * the offset is 74 from the origin and subtracting
1882  * 74 from all measurements centers them around zero.
1883  */

1887

1889  arb.x_raw = report.accel_x;
1890  arb.y_raw = report.accel_y;
1891  arb.z_raw = report.accel_z;

1894  float yraw_f = report.accel_y;
1895  float zraw_f = report.accel_z;
1896

1899

1918

1929  grb.y_raw = report.gyro_y;
1930  grb.z_raw = report.gyro_z;

1933  yraw_f = report.gyro_y;
1934  zraw_f = report.gyro_z;
