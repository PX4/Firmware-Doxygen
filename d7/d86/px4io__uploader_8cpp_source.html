73 int
74 PX4IO_Uploader::upload(const char *filenames[])
75 {
76  int ret;
77  const char *filename = NULL;
78  size_t fw_size;
79 
80 #ifndef PX4IO_SERIAL_DEVICE
81 #error Must define PX4IO_SERIAL_DEVICE in board configuration to support firmware upload
82 #endif
83 
84  /* allow an early abort and look for file first */
85  for (unsigned i = 0; filenames[i] != nullptr; i++) {
86  _fw_fd = open(filenames[i], O_RDONLY);
87 
88  if (_fw_fd < 0) {
89  log("failed to open %s", filenames[i]);
90  continue;
91  }
92 
93  log("using firmware from %s", filenames[i]);
94  filename = filenames[i];
95  break;
96  }
97 
98  if (filename == NULL) {
99  log("no firmware found");
100  close(_io_fd);
101  _io_fd = -1;
102  return -ENOENT;
103  }
104 
105  _io_fd = open(PX4IO_SERIAL_DEVICE, O_RDWR);
106 
107  if (_io_fd < 0) {
108  log("could not open interface");
109  return -errno;
110  }
111 
112  /* save initial uart configuration to reset after the update */
113  struct termios t_original;
114  tcgetattr(_io_fd, &t_original);
115 
116  /* adjust line speed to match bootloader */
117  struct termios t;
118  tcgetattr(_io_fd, &t);
119  cfsetspeed(&t, 115200);
120  tcsetattr(_io_fd, TCSANOW, &t);
121 
122  /* look for the bootloader for 150 ms */
123  for (int i = 0; i < 15; i++) {
124  ret = sync();
125 
126  if (ret == OK) {
127  break;
128 
129  } else {
130  usleep(10000);
131  }
132  }
133 
134  if (ret != OK) {
135  /* this is immediately fatal */
136  log("bootloader not responding");
137  tcsetattr(_io_fd, TCSANOW, &t_original);
138  close(_io_fd);
139  _io_fd = -1;
140  return -EIO;
141  }
142 
143  struct stat st;
144 
145  if (stat(filename, &st) != 0) {
146  log("Failed to stat %s - %d\n", filename, (int)errno);
147  tcsetattr(_io_fd, TCSANOW, &t_original);
148  close(_io_fd);
149  _io_fd = -1;
150  return -errno;
151  }
152 
153  fw_size = st.st_size;
154 
155  if (_fw_fd == -1) {
156  tcsetattr(_io_fd, TCSANOW, &t_original);
157  close(_io_fd);
158  _io_fd = -1;
159  return -ENOENT;
160  }
161 
162  /* do the usual program thing - allow for failure */
163  for (unsigned retries = 0; retries < 1; retries++) {
164  if (retries > 0) {
165  log("retrying update...");
166  ret = sync();
167 
168  if (ret != OK) {
169  /* this is immediately fatal */
170  log("bootloader not responding");
171  tcsetattr(_io_fd, TCSANOW, &t_original);
172  close(_io_fd);
173  _io_fd = -1;
174  return -EIO;
175  }
176  }
177 
178  ret = get_info(INFO_BL_REV, bl_rev);
179 
180  if (ret == OK) {
181  if (bl_rev <= BL_REV) {
182  log("found bootloader revision: %d", bl_rev);
183 
184  } else {
185  log("found unsupported bootloader revision %d, exiting", bl_rev);
186  tcsetattr(_io_fd, TCSANOW, &t_original);
187  close(_io_fd);
188  _io_fd = -1;
189  return OK;
190  }
191  }
192 
193  ret = erase();
194 
195  if (ret != OK) {
196  log("erase failed");
197  continue;
198  }
199 
200  ret = program(fw_size);
201 
202  if (ret != OK) {
203  log("program failed");
204  continue;
205  }
206 
207  if (bl_rev <= 2) {
208  ret = verify_rev2(fw_size);
209 
210  } else {
211  /* verify rev 3 and higher. Every version *needs* to be verified. */
212  ret = verify_rev3(fw_size);
213  }
214 
215  if (ret != OK) {
216  log("verify failed");
217  continue;
218  }
219 
220  ret = reboot();
221 
222  if (ret != OK) {
223  log("reboot failed");
224  tcsetattr(_io_fd, TCSANOW, &t_original);
225  close(_io_fd);
226  _io_fd = -1;
227  return ret;
228  }
229 
230  log("update complete");
231 
232  ret = OK;
233  break;
234  }
235 
236  /* reset uart to previous/default baudrate */
237  tcsetattr(_io_fd, TCSANOW, &t_original);
238 
239  close(_fw_fd);
240  close(_io_fd);
241  _io_fd = -1;
242 
243  // sleep for enough time for the IO chip to boot. This makes
244  // forceupdate more reliably startup IO again after update
245  up_udelay(100 * 1000);
246 
247  return ret;
248 }

431 PX4IO_Uploader::program(size_t fw_size)
432 {
433  uint8_t *file_buf;
434  ssize_t count;
435  int ret;
436  size_t sent = 0;
437 
438  file_buf = new uint8_t[PROG_MULTI_MAX];
439 
440  if (!file_buf) {
441  log("Can't allocate program buffer");
442  return -ENOMEM;
443  }
444 
445  ASSERT((fw_size & 3) == 0);
446  ASSERT((PROG_MULTI_MAX & 3) == 0);
447 
448  log("programming %u bytes...", (unsigned)fw_size);
449 
450  ret = lseek(_fw_fd, 0, SEEK_SET);
451 
452  while (sent < fw_size) {
453  /* get more bytes to program */
454  size_t n = fw_size - sent;
455 
456  if (n > PROG_MULTI_MAX) {
457  n = PROG_MULTI_MAX;
458  }
459 
460  count = read_with_retry(_fw_fd, file_buf, n);
461 
462  if (count != (ssize_t)n) {
463  log("firmware read of %u bytes at %u failed -> %d errno %d",
464  (unsigned)n,
465  (unsigned)sent,
466  (int)count,
467  (int)errno);
468  ret = -errno;
469  break;
470  }
471 
472  sent += count;
473 
474  send(PROTO_PROG_MULTI);
475  send(count);
476  send(file_buf, count);
477  send(PROTO_EOC);
478 
479  ret = get_sync(1000);
480 
481  if (ret != OK) {
482  break;
483  }
484  }
485 
486  delete [] file_buf;
487  return ret;
488 }

491 PX4IO_Uploader::verify_rev2(size_t fw_size)
492 {
493  uint8_t file_buf[4];
494  ssize_t count;
495  int ret;
496  size_t sent = 0;
497 
498  log("verify...");
499  lseek(_fw_fd, 0, SEEK_SET);
500 
501  send(PROTO_CHIP_VERIFY);
502  send(PROTO_EOC);
503  ret = get_sync();
504 
505  if (ret != OK) {
506  return ret;
507  }
508 
509  while (sent < fw_size) {
510  /* get more bytes to verify */
511  size_t n = fw_size - sent;
512 
513  if (n > sizeof(file_buf)) {
514  n = sizeof(file_buf);
515  }
516 
517  count = read_with_retry(_fw_fd, file_buf, n);
518 
519  if (count != (ssize_t)n) {
520  log("firmware read of %u bytes at %u failed -> %d errno %d",
521  (unsigned)n,
522  (unsigned)sent,
523  (int)count,
524  (int)errno);
525  }
526 
527  if (count == 0) {
528  break;
529  }
530 
531  sent += count;
532 
533  if (count < 0) {
534  return -errno;
535  }
536 
537  ASSERT((count % 4) == 0);
538 
539  send(PROTO_READ_MULTI);
540  send(count);
541  send(PROTO_EOC);
542 
543  for (ssize_t i = 0; i < count; i++) {
544  uint8_t c;
545 
546  ret = recv_byte_with_timeout(&c, 5000);
547 
548  if (ret != OK) {
549  log("%d: got %d waiting for bytes", sent + i, ret);
550  return ret;
551  }
552 
553  if (c != file_buf[i]) {
554  log("%d: got 0x%02x expected 0x%02x", sent + i, c, file_buf[i]);
555  return -EINVAL;
556  }
557  }
558 
559  ret = get_sync();
560 
561  if (ret != OK) {
562  log("timeout waiting for post-verify sync");
563  return ret;
564  }
565  }
566 
567  return OK;
568 }

570 int
571 PX4IO_Uploader::verify_rev3(size_t fw_size_local)
572 {
573  int ret;
574  uint8_t file_buf[4];
575  ssize_t count;
576  uint32_t sum = 0;
577  uint32_t bytes_read = 0;
578  uint32_t crc = 0;
579  uint32_t fw_size_remote;
580  uint8_t fill_blank = 0xff;
581 
582  log("verify...");
583  lseek(_fw_fd, 0, SEEK_SET);
584 
585  ret = get_info(INFO_FLASH_SIZE, fw_size_remote);
586  send(PROTO_EOC);
587 
588  if (ret != OK) {
589  log("could not read firmware size");
590  return ret;
591  }
592 
593  /* read through the firmware file again and calculate the checksum*/
594  while (bytes_read < fw_size_local) {
595  size_t n = fw_size_local - bytes_read;
596 
597  if (n > sizeof(file_buf)) {
598  n = sizeof(file_buf);
599  }
600 
601  count = read_with_retry(_fw_fd, file_buf, n);
602 
603  if (count != (ssize_t)n) {
604  log("firmware read of %u bytes at %u failed -> %d errno %d",
605  (unsigned)n,
606  (unsigned)bytes_read,
607  (int)count,
608  (int)errno);
609  }
610 
611  /* set the rest to ff */
612  if (count == 0) {
613  break;
614  }
615 
616  /* stop if the file cannot be read */
617  if (count < 0) {
618  return -errno;
619  }
620 
621  /* calculate crc32 sum */
622  sum = crc32part((uint8_t *)&file_buf, sizeof(file_buf), sum);
623 
624  bytes_read += count;
625  }
626 
627  /* fill the rest with 0xff */
628  while (bytes_read < fw_size_remote) {
629  sum = crc32part(&fill_blank, sizeof(fill_blank), sum);
630  bytes_read += sizeof(fill_blank);
631  }
632 
633  /* request CRC from IO */
634  send(PROTO_GET_CRC);
635  send(PROTO_EOC);
636 
637  ret = recv_bytes((uint8_t *)(&crc), sizeof(crc));
638 
639  if (ret != OK) {
640  log("did not receive CRC checksum");
641  return ret;
642  }
643 
644  ret = get_sync();
645 
646  if (ret != OK) {
647  log("did not receive CRC checksum");
648  return ret;
649  }
650 
651  /* compare the CRC sum from the IO with the one calculated */
652  if (sum != crc) {
653  log("CRC wrong: received: %d, expected: %d", crc, sum);
654  return -EINVAL;
655  }
656 
657  return OK;
658 }
