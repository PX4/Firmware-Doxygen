102 static int parse_options(int argc, char *argv[])
103 {
104  int ch;
105  int myoptind = 1;
106  const char *myoptarg = nullptr;
107 
108  while ((ch = px4_getopt(argc, argv, "t:d:u:l:w:b:p:r:s:", &myoptind, &myoptarg)) != EOF) {
109  switch (ch) {
110  case 't': _options.transport = strcmp(myoptarg, "UDP") == 0 ?
111  options::eTransports::UDP
112  : options::eTransports::UART; break;
113 
114  case 'd': if (nullptr != myoptarg) strcpy(_options.device, myoptarg); break;
115 
116  case 'u': _options.update_time_ms = strtol(myoptarg, nullptr, 10); break;
117 
118  case 'l': _options.loops = strtol(myoptarg, nullptr, 10); break;
119 
120  case 'w': _options.sleep_ms = strtol(myoptarg, nullptr, 10); break;
121 
122  case 'b': _options.baudrate = getbaudrate(myoptarg); break;
123 
124  case 'p': _options.poll_ms = strtol(myoptarg, nullptr, 10); break;
125 
126  case 'r': _options.recv_port = strtoul(myoptarg, nullptr, 10); break;
127 
128  case 's': _options.send_port = strtoul(myoptarg, nullptr, 10); break;
129 
130  default:
131  usage(argv[1]);
132  return -1;
133  }
134  }
135 
136  if (_options.sleep_ms < 1) {
137  _options.sleep_ms = 1;
138  PX4_ERR("sleep time too low, using 1 ms");
139  }
140 
141  if (_options.poll_ms < 1) {
142  _options.poll_ms = 1;
143  PX4_ERR("poll timeout too low, using 1 ms");
144  }
145 
146  return 0;
147 }

149 static int micrortps_start(int argc, char *argv[])
150 {
151  if (0 > parse_options(argc, argv)) {
152  printf("EXITING...\n");
153  _rtps_task = -1;
154  return -1;
155  }
156 
157  switch (_options.transport) {
158  case options::eTransports::UART: {
159  transport_node = new UART_node(_options.device, _options.baudrate.code, _options.poll_ms);
160  printf("\nUART transport: device: %s; baudrate: %d; sleep: %dms; poll: %dms\n\n",
161  _options.device, _options.baudrate.val, _options.sleep_ms, _options.poll_ms);
162  }
163  break;
164 
165  case options::eTransports::UDP: {
166  transport_node = new UDP_node(_options.recv_port, _options.send_port);
167  printf("\nUDP transport: recv port: %u; send port: %u; sleep: %dms\n\n",
168  _options.recv_port, _options.send_port, _options.sleep_ms);
169  }
170  break;
171 
172  default:
173  _rtps_task = -1;
174  printf("EXITING...\n");
175  return -1;
176  }
177 
178  if (0 > transport_node->init()) {
179  printf("EXITING...\n");
180  _rtps_task = -1;
181  return -1;
182  }
183 
184 
185  struct timespec begin;
186 
187  int total_read = 0, loop = 0;
188 
189  uint32_t received = 0;
190 
191  micrortps_start_topics(begin, total_read, received, loop);
192 
193  struct timespec end;
194 
195  px4_clock_gettime(CLOCK_REALTIME, &end);
196 
197  double elapsed_secs = double(end.tv_sec - begin.tv_sec) + double(end.tv_nsec - begin.tv_nsec) / double(1000000000);
198 
199  printf("RECEIVED: %lu messages in %d LOOPS, %d bytes in %.03f seconds - %.02fKB/s\n\n",
200  (unsigned long)received, loop, total_read, elapsed_secs, (double)total_read / (1000 * elapsed_secs));
201 
202  delete transport_node;
203 
204  transport_node = nullptr;
205 
206  PX4_INFO("exiting");
207 
208  fflush(stdout);
209 
210  _rtps_task = -1;
211 
212  return 0;
213 }
