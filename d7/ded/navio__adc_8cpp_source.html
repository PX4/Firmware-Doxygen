120 void NavioADC::_measure()
121 {
122  px4_adc_msg_t tmp_samples[ADC_MAX_CHAN];
123 
124  for (int i = 0; i < ADC_MAX_CHAN; ++i) {
125  int ret;
126 
127  /* Currently we only use these channels */
128  if (i != NAVIO_ADC_BATTERY_VOLTAGE_CHANNEL &&
129  i != NAVIO_ADC_BATTERY_CURRENT_CHANNEL) {
130  tmp_samples[i].am_channel = i;
131  tmp_samples[i].am_data = 0;
132 
133  continue;
134  }
135 
136  ret = read_channel(&tmp_samples[i], i);
137 
138  if (ret < 0) {
139  PX4_ERR("read_channel(%d): %d", i, ret);
140  tmp_samples[i].am_channel = i;
141  tmp_samples[i].am_data = 0;
142  }
143  }
144 
145  tmp_samples[NAVIO_ADC_BATTERY_VOLTAGE_CHANNEL].am_channel = ADC_BATTERY_VOLTAGE_CHANNEL;
146  tmp_samples[NAVIO_ADC_BATTERY_CURRENT_CHANNEL].am_channel = ADC_BATTERY_CURRENT_CHANNEL;
147 
148  pthread_mutex_lock(&_samples_lock);
149  memcpy(&_samples, &tmp_samples, sizeof(tmp_samples));
150  pthread_mutex_unlock(&_samples_lock);
151 }

153 int NavioADC::init()
154 {
155  int ret;
156 
157  ret = DriverFramework::VirtDevObj::init();
158 
159  if (ret != PX4_OK) {
160  PX4_ERR("init failed");
161  return ret;
162  }
163 
164  for (int i = 0; i < ADC_MAX_CHAN; i++) {
165  char channel_path[sizeof(ADC_SYSFS_PATH)];
166  strncpy(channel_path, ADC_SYSFS_PATH, sizeof(ADC_SYSFS_PATH));
167  channel_path[sizeof(ADC_SYSFS_PATH) - 2] += i;
168 
169  _fd[i] = ::open(channel_path, O_RDONLY);
170 
171  if (_fd[i] == -1) {
172  int err = errno;
173  ret = -1;
174  PX4_ERR("init: open: %s (%d)", strerror(err), err);
175  goto cleanup;
176  }
177  }
178 
179  return PX4_OK;
180 
181 cleanup:
182 
183  for (int i = 0; i < ADC_MAX_CHAN; i++) {
184  if (_fd[i] != -1) {
185  close(_fd[i]);
186  }
187  }
188 
189  return ret;
190 }

197 ssize_t NavioADC::devRead(void *buf, size_t count)
198 {
199  const size_t maxsize = sizeof(_samples);
200  int ret;
201 
202  if (count > maxsize) {

207 
208  if (ret != 0) {
209  return 0;

211 
212  memcpy(buf, &_samples, count);
213  pthread_mutex_unlock(&_samples_lock);

218 int NavioADC::read_channel(px4_adc_msg_t *adc_msg, int channel)
219 {
220  char buffer[11]; /* 32bit max INT has maximum 10 chars */
221  int ret;
222 
223  if (channel < 0 || channel >= ADC_MAX_CHAN) {
224  return -EINVAL;
225  }
226 
227  ret = lseek(_fd[channel], 0, SEEK_SET);
228 
229  if (ret == -1) {
230  ret = errno;
231  PX4_ERR("read_channel %d: lseek: %s (%d)", channel, strerror(ret), ret);
232  return ret;
233  }
234 
235  ret = ::read(_fd[channel], buffer, sizeof(buffer) - 1);
236 
237  if (ret == -1) {
238  ret = errno;
239  PX4_ERR("read_channel %d: read: %s (%d)", channel, strerror(ret), ret);
240  return ret;
241 
242  } else if (ret == 0) {
243  PX4_ERR("read_channel %d: read empty", channel);
244  ret = -EINVAL;
245  return ret;
246  }
247 
248  buffer[ret] = 0;
249 
250  adc_msg->am_channel = channel;
251  adc_msg->am_data = strtol(buffer, NULL, 10);
252  ret = 0;
253 
254  return ret;
255 }
