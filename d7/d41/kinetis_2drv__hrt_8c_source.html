284 /**
285  * Initialize the timer we are going to use.
286  */
287 static void hrt_tim_init(void)
288 {
289 
290  /* Select a the clock source to the TPM */
291 
292  uint32_t regval = _REG(KINETIS_SIM_SOPT2);
293  regval &= ~(SIM_SOPT2_TPMSRC_MASK);
294  regval |= BOARD_TPM_CLKSRC;
295  _REG(KINETIS_SIM_SOPT2) = regval;
296 
297 
298  /* Enabled System Clock Gating Control for TPM */
299 
300  regval = _REG(KINETIS_SIM_SCGC2);
301  regval |= HRT_SIM_SCGC2_TPM;
302  _REG(KINETIS_SIM_SCGC2) = regval;
303 
304 
305  /* claim our interrupt vector */
306 
307  irq_attach(HRT_TIMER_VECTOR, hrt_tim_isr, NULL);
308 
309  /* disable and configure the timer */
310 
311  rSC = TPM_SC_TOF;
312 
313  rCNT = 0;
314  rMOD = HRT_COUNTER_PERIOD - 1;
315 
316  rSTATUS = (TPM_STATUS_TOF | STATUS_HRT | STATUS_PPM);
317  rCNSC_HRT = (TPM_CnSC_CHF | TPM_CnSC_CHIE | TPM_CnSC_MSA);
318  rCNSC_PPM = (TPM_CnSC_CHF | CNSC_PPM);
319  rCOMBINE = 0;
320  rPOL = 0;
321  rFILTER = 0;
322  rQDCTRL = 0;
323  rCONF = TPM_CONF_DBGMODE_CONT;
324 
325  /* set an initial capture a little ways off */
326 
327  rCNV_PPM = 0;
328  rCNV_HRT = 1000;
329 
330  /* enable the timer */
331 
332  rSC |= (TPM_SC_TOIE | TPM_SC_CMOD_LPTPM_CLK | TPM_SC_PS_DIV16);
333 
334  /* enable interrupts */
335  up_enable_irq(HRT_TIMER_VECTOR);
336 }

338 #ifdef HRT_PPM_CHANNEL
339 /**
340  * Handle the PPM decoder state machine.
341  */
342 static void hrt_ppm_decode(uint32_t status)
343 {
344  uint16_t count = rCNV_PPM;
345  uint16_t width;
346  uint16_t interval;
347  unsigned i;
348 
349  /* how long since the last edge? - this handles counter wrapping implicitly. */
350  width = count - ppm.last_edge;
351 
352 #if PPM_DEBUG
353  ppm_edge_history[ppm_edge_next++] = width;
354 
355  if (ppm_edge_next >= EDGE_BUFFER_COUNT) {
356  ppm_edge_next = 0;
357  }
358 
359 #endif
360 
361  /*
362  * if this looks like a start pulse, then push the last set of values
363  * and reset the state machine
364  */
365  if (width >= PPM_MIN_START) {
366 
367  /*
368  * If the number of channels changes unexpectedly, we don't want
369  * to just immediately jump on the new count as it may be a result
370  * of noise or dropped edges. Instead, take a few frames to settle.
371  */
372  if (ppm.next_channel != ppm_decoded_channels) {
373  static unsigned new_channel_count;
374  static unsigned new_channel_holdoff;
375 
376  if (new_channel_count != ppm.next_channel) {
377  /* start the lock counter for the new channel count */
378  new_channel_count = ppm.next_channel;
379  new_channel_holdoff = PPM_CHANNEL_LOCK;
380 
381  } else if (new_channel_holdoff > 0) {
382  /* this frame matched the last one, decrement the lock counter */
383  new_channel_holdoff--;
384 
385  } else {
386  /* we have seen PPM_CHANNEL_LOCK frames with the new count, accept it */
387  ppm_decoded_channels = new_channel_count;
388  new_channel_count = 0;
389  }
390 
391  } else {
392  /* frame channel count matches expected, let's use it */
393  if (ppm.next_channel >= PPM_MIN_CHANNELS) {
394  for (i = 0; i < ppm.next_channel; i++) {
395  ppm_buffer[i] = ppm_temp_buffer[i];
396  }
397 
398  ppm_last_valid_decode = hrt_absolute_time();
399 
400  }
401  }
402 
403  /* reset for the next frame */
404  ppm.next_channel = 0;
405 
406  /* next edge is the reference for the first channel */
407  ppm.phase = ARM;
408 
409  ppm.last_edge = count;
410  return;
411  }
412 
413  switch (ppm.phase) {
414  case UNSYNCH:
415  /* we are waiting for a start pulse - nothing useful to do here */
416  break;
417 
418  case ARM:
419 
420  /* we expect a pulse giving us the first mark */
421  if (width < PPM_MIN_PULSE_WIDTH || width > PPM_MAX_PULSE_WIDTH) {
422  goto error; /* pulse was too short or too long */
423  }
424 
425  /* record the mark timing, expect an inactive edge */
426  ppm.last_mark = ppm.last_edge;
427 
428  /* frame length is everything including the start gap */
429  ppm_frame_length = (uint16_t)(ppm.last_edge - ppm.frame_start);
430  ppm.frame_start = ppm.last_edge;
431  ppm.phase = ACTIVE;
432  break;
433 
434  case INACTIVE:
435 
436  /* we expect a short pulse */
437  if (width < PPM_MIN_PULSE_WIDTH || width > PPM_MAX_PULSE_WIDTH) {
438  goto error; /* pulse was too short or too long */
439  }
440 
441  /* this edge is not interesting, but now we are ready for the next mark */
442  ppm.phase = ACTIVE;
443  break;
444 
445  case ACTIVE:
446  /* determine the interval from the last mark */
447  interval = count - ppm.last_mark;
448  ppm.last_mark = count;
449 
450 #if PPM_DEBUG
451  ppm_pulse_history[ppm_pulse_next++] = interval;
452 
453  if (ppm_pulse_next >= EDGE_BUFFER_COUNT) {
454  ppm_pulse_next = 0;
455  }
456 
457 #endif
458 
459  /* if the mark-mark timing is out of bounds, abandon the frame */
460  if ((interval < PPM_MIN_CHANNEL_VALUE) || (interval > PPM_MAX_CHANNEL_VALUE)) {
461  goto error;
462  }
463 
464  /* if we have room to store the value, do so */
465  if (ppm.next_channel < PPM_MAX_CHANNELS) {
466  ppm_temp_buffer[ppm.next_channel++] = interval;
467  }
468 
469  ppm.phase = INACTIVE;
470  break;
471 
472  }
473 
474  ppm.last_edge = count;
475  return;
476 
477  /* the state machine is corrupted; reset it */
478 
479 error:
480  /* we don't like the state of the decoder, reset it and try again */
481  ppm.phase = UNSYNCH;
482  ppm_decoded_channels = 0;
483 
484 }
485 #endif /* HRT_PPM_CHANNEL */

487 /**
488  * Handle the compare interrupt by calling the callout dispatcher
489  * and then re-scheduling the next deadline.
490  */
491 static int
492 hrt_tim_isr(int irq, void *context, void *arg)
493 {
494  /* grab the timer for latency tracking purposes */
495 
496  latency_actual = rCNT;
497 
498  /* copy interrupt status */
499  uint32_t status = rSTATUS;
500 
501  /* ack the interrupts we just read */
502 
503  rSTATUS = status;
504 
505 #ifdef HRT_PPM_CHANNEL
506 
507  /* was this a PPM edge? */
508  if (status & (STATUS_PPM)) {
509  hrt_ppm_decode(status);
510  }
511 
512 #endif
513 
514  /* was this a timer tick? */
515  if (status & STATUS_HRT) {
516 
517  /* do latency calculations */
518  hrt_latency_update();
519 
520  /* run any callouts that have met their deadline */
521  hrt_call_invoke();
522 
523  /* and schedule the next interrupt */
524  hrt_call_reschedule();
525  }
526 
527  return OK;
528 }

530 /**
531  * Fetch a never-wrapping absolute time value in microseconds from
532  * some arbitrary epoch shortly after system start.
533  */
534 hrt_abstime
535 hrt_absolute_time(void)
536 {
537  hrt_abstime abstime;
538  uint32_t count;
539  irqstate_t flags;
540 
541  /*
542  * Counter state. Marked volatile as they may change
543  * inside this routine but outside the irqsave/restore
544  * pair. Discourage the compiler from moving loads/stores
545  * to these outside of the protected range.
546  */
547  static volatile hrt_abstime base_time;
548  static volatile uint32_t last_count;
549 
550  /* prevent re-entry */
551  flags = px4_enter_critical_section();
552 
553  /* get the current counter value */
554  count = rCNT;
555 
556  /*
557  * Determine whether the counter has wrapped since the
558  * last time we're called.
559  *
560  * This simple test is sufficient due to the guarantee that
561  * we are always called at least once per counter period.
562  */
563  if (count < last_count) {
564  base_time += HRT_COUNTER_PERIOD;
565  }
566 
567  /* save the count for next time */
568  last_count = count;
569 
570  /* compute the current time */
571  abstime = HRT_COUNTER_SCALE(base_time + count);
572 
573  px4_leave_critical_section(flags);
574 
575  return abstime;
576 }

683 static void
684 hrt_call_internal(struct hrt_call *entry, hrt_abstime deadline, hrt_abstime interval, hrt_callout callout, void *arg)
685 {
686  irqstate_t flags = px4_enter_critical_section();
687 
688  /* if the entry is currently queued, remove it */
689  /* note that we are using a potentially uninitialized
690  entry->link here, but it is safe as sq_rem() doesn't
691  dereference the passed node unless it is found in the
692  list. So we potentially waste a bit of time searching the
693  queue for the uninitialized entry->link but we don't do
694  anything actually unsafe.

741 hrt_call_enter(struct hrt_call *entry)
742 {

769 hrt_call_invoke(void)
770 {
771  struct hrt_call *call;
772  hrt_abstime deadline;
773 
774  while (true) {
775  /* get the current time */
776  hrt_abstime now = hrt_absolute_time();
777 
778  call = (struct hrt_call *)sq_peek(&callout_queue);
779 
780  if (call == NULL) {
781  break;
782  }
783 
784  if (call->deadline > now) {
785  break;
786  }
787 
788  sq_rem(&call->link, &callout_queue);
789  hrtinfo("call pop\n");
790 
791  /* save the intended deadline for periodic calls */
792  deadline = call->deadline;
793 
794  /* zero the deadline, as the call has occurred */
795  call->deadline = 0;
796 
797  /* invoke the callout (if there is one) */
798  if (call->callout) {
799  hrtinfo("call %p: %p(%p)\n", call, call->callout, call->arg);
800  call->callout(call->arg);
801  }
802 
803  /* if the callout has a non-zero period, it has to be re-entered */
804  if (call->period != 0) {
805  // re-check call->deadline to allow for
806  // callouts to re-schedule themselves
807  // using hrt_call_delay()
808  if (call->deadline <= now) {
809  call->deadline = deadline + call->period;
810  }
811 
812  hrt_call_enter(call);
813  }
814  }
815 }

817 /**
818  * Reschedule the next timer interrupt.
819  *
820  * This routine must be called with interrupts disabled.
821  */
822 static void
823 hrt_call_reschedule()
824 {
825  hrt_abstime now = hrt_absolute_time();
826  struct hrt_call *next = (struct hrt_call *)sq_peek(&callout_queue);
827  hrt_abstime deadline = now + HRT_INTERVAL_MAX;
828 
829  /*
830  * Determine what the next deadline will be.
831  *
832  * Note that we ensure that this will be within the counter
833  * period, so that when we truncate all but the low 16 bits
834  * the next time the compare matches it will be the deadline
835  * we want.
836  *
837  * It is important for accurate timekeeping that the compare
838  * interrupt fires sufficiently often that the base_time update in
839  * hrt_absolute_time runs at least once per timer period.
840  */
841  if (next != NULL) {
842  hrtinfo("entry in queue\n");
843 
844  if (next->deadline <= (now + HRT_INTERVAL_MIN)) {
845  hrtinfo("pre-expired\n");
846  /* set a minimal deadline so that we call ASAP */
847  deadline = now + HRT_INTERVAL_MIN;
848 
849  } else if (next->deadline < deadline) {
850  hrtinfo("due soon\n");
851  deadline = next->deadline;
852  }
853  }
854 
855  hrtinfo("schedule for %u at %u\n", (unsigned)(deadline & 0xffffffff), (unsigned)(now & 0xffffffff));
856 
857  /* set the new compare value and remember it for latency tracking */
858  rCNV_HRT = latency_baseline = deadline & 0xffff;
859 }
