158 int gpio_led_main(int argc, char *argv[])
159 {
160  if (argc < 2) {
161  print_usage(NULL);
162  exit(1);
163  } else {
164 
165  if (!strcmp(argv[1], "start")) {
166  if (gpio_led_state != Stopped) {
167  PX4_WARN("already running");
168  exit(1);
169  }
170 
171  /* by default GPIO_SERVO_1 on FMUv2 */
172  int pin = 1;
173 
174  /* pin name to display */
175  char pin_name[sizeof(PIN_NAME) + 2] = PIN_NAME;
176 
177  if (argc > 2) {
178  if (!strcmp(argv[2], "-p")) {
179 
180  unsigned int n = strtoul(argv[3], NULL, 10);
181 
182  if (n >= GPIO_MIN_SERVO_PIN && n <= GPIO_MAX_SERVO_PIN) {
183  pin = 1 << (n - 1);
184  snprintf(pin_name, sizeof(pin_name), "AUX OUT %d", n);
185  } else {
186  PX4_ERR("unsupported pin: %s (valid values are %d-%d)", argv[3], GPIO_MIN_SERVO_PIN, GPIO_MAX_SERVO_PIN);
187  exit(1);
188  }
189 
190  }
191  }
192 
193  gpio_led_data = malloc(sizeof(struct gpio_led_s));
194  if (gpio_led_data == NULL) {
195  PX4_ERR("failed to allocate memory!");
196  exit(1);
197  } else {
198  memset(gpio_led_data, 0, sizeof(struct gpio_led_s));
199  gpio_led_data->pin = pin;
200  int ret = work_queue(LPWORK, &(gpio_led_data->work), gpio_led_start, gpio_led_data, 0);
201 
202  if (ret != 0) {
203  PX4_ERR("failed to queue work: %d", ret);
204  goto out;
205  } else {
206  usleep(1000000/CYCLE_RATE_HZ);
207  if (gpio_led_state != Running) {
208  gpio_led_state = Stopped;
209  goto out;
210  }
211  PX4_INFO("start, using pin: %s", pin_name);
212  exit(0);
213  }
214  }
215 
216  } else if (!strcmp(argv[1], "stop")) {
217  if (gpio_led_state == Running) {
218  gpio_led_state = Stopping;
219  while(gpio_led_state != Stopped) {
220  usleep(1000000/CYCLE_RATE_HZ);
221  }
222  PX4_INFO("stopped");
223  free (gpio_led_data);
224  gpio_led_data = NULL;
225  exit(0);
226  } else {
227  PX4_WARN("not running");
228  exit(1);
229  }
230 
231  } else {
232  print_usage("unrecognized command");
233  exit(1);
234  }
235  }
236 out:
237  free (gpio_led_data);
238  gpio_led_data = NULL;
239  exit(1);
240 }

242 void gpio_led_start(FAR void *arg)
243 {
244  FAR struct gpio_led_s *priv = (FAR struct gpio_led_s *)arg;
245 
246  char *gpio_dev = PX4FMU_DEVICE_PATH;
247 
248  /* open GPIO device */
249  priv->gpio_fd = open(gpio_dev, 0);
250 
251  if (priv->gpio_fd < 0) {
252  PX4_ERR("gpio_led: GPIO device \"%s\" open fail\n", gpio_dev);
253  gpio_led_state = Falied;
254  return;
255  }
256 
257  /* configure GPIO pin */
258 
259  ioctl(priv->gpio_fd, GPIO_SET_OUTPUT, priv->pin);
260 
261  /* initialize vehicle status structure */
262  memset(&priv->vehicle_status, 0, sizeof(priv->vehicle_status));
263 
264  /* initialize battery status structure */
265  memset(&priv->battery_status, 0, sizeof(priv->battery_status));
266 
267  /* subscribe to vehicle status topic */
268  priv->vehicle_status_sub = orb_subscribe(ORB_ID(vehicle_status));
269 
270  /* subscribe to battery status topic */
271  priv->battery_status_sub = orb_subscribe(ORB_ID(battery_status));
272 
273  /* add worker to queue */
274  int ret = work_queue(LPWORK, &priv->work, gpio_led_cycle, priv, 0);
275 
276  if (ret != 0) {
277  PX4_ERR("gpio_led: failed to queue work: %d\n", ret);
278  close(priv->gpio_fd);
279  gpio_led_state = Falied;
280  return;
281  }
282  gpio_led_state = Running;
283 }

285 void gpio_led_cycle(FAR void *arg)
286 {
287  FAR struct gpio_led_s *priv = (FAR struct gpio_led_s *)arg;
288 
289  /* check for vehicle status updates*/
290  bool updated;
291  orb_check(priv->vehicle_status_sub, &updated);
292 
293  if (updated) {
294  orb_copy(ORB_ID(vehicle_status), priv->vehicle_status_sub, &priv->vehicle_status);
295  }
296 
297  orb_check(priv->battery_status_sub, &updated);
298 
299  if (updated) {
300  orb_copy(ORB_ID(battery_status), priv->battery_status_sub, &priv->battery_status);
301  }
302 
303  /* select pattern for current vehiclestatus */
304  int pattern = 0;
305 
306  if (priv->vehicle_status.arming_state == VEHICLE_STATUS_ARMING_STATE_ARMED) {
307  if (priv->battery_status.warning == BATTERY_STATUS_BATTERY_WARNING_NONE
308  && !priv->vehicle_status.failsafe) {
309  pattern = 0x3f; // ****** solid (armed)
310 
311  } else {
312  pattern = 0x3e; // *****_ slow blink (armed, battery low or failsafe)
313  }
314 
315  } else if (priv->vehicle_status.arming_state == VEHICLE_STATUS_ARMING_STATE_STANDBY) {
316  pattern = 0x38; // ***___ slow blink (disarmed, ready)
317 
318  } else if (priv->vehicle_status.arming_state == VEHICLE_STATUS_ARMING_STATE_STANDBY_ERROR) {
319  pattern = 0x28; // *_*___ slow double blink (disarmed, error)
320 
321  }
322 
323  /* blink pattern */
324  bool led_state_new = (pattern & (1 << priv->counter)) != 0;
325 
326  if (led_state_new != priv->led_state) {
327  priv->led_state = led_state_new;
328 
329  if (led_state_new) {
330  ioctl(priv->gpio_fd, GPIO_SET, priv->pin);
331 
332  } else {
333  ioctl(priv->gpio_fd, GPIO_CLEAR, priv->pin);
334  }
335  }
336 
337  priv->counter++;
338 
339  if (priv->counter > 5) {
340  priv->counter = 0;
341  }
342 
343  /* repeat cycle at 5 Hz */
344  if (gpio_led_state == Running) {
345  work_queue(LPWORK, &priv->work, gpio_led_cycle, priv, USEC2TICK(1000000/CYCLE_RATE_HZ));
346 
347  } else {
348  /* switch off LED on stop */
349  ioctl(priv->gpio_fd, GPIO_CLEAR, priv->pin);
350  orb_unsubscribe(priv->vehicle_status_sub);
351  orb_unsubscribe(priv->battery_status_sub);
352  close(priv->gpio_fd);
353  gpio_led_state = Stopped;
354  }
