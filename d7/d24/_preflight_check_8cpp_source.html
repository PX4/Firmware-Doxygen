75 static int check_calibration(DevHandle &h, const char *param_template, int &devid)
76 {
77  bool calibration_found = false;
78 
79  devid = h.ioctl(DEVIOCGDEVICEID, 0);
80 
81  char s[20];
82  int instance = 0;
83 
84  /* old style transition: check param values */
85  while (!calibration_found) {
86  sprintf(s, param_template, instance);
87  param_t parm = param_find(s);
88 
89  /* if the calibration param is not present, abort */
90  if (parm == PARAM_INVALID) {
91  break;
92  }
93 
94  /* if param get succeeds */
95  int32_t calibration_devid;
96 
97  if (!param_get(parm, &(calibration_devid))) {
98 
99  /* if the devid matches, exit early */
100  if (devid == calibration_devid) {
101  calibration_found = true;
102  break;
103  }
104  }
105 
106  instance++;
107  }
108 
109  return !calibration_found;
110 }

153 static bool imuConsistencyCheck(orb_advert_t *mavlink_log_pub, vehicle_status_s &status, bool report_status)
154 {
155  bool success = true; // start with a pass and change to a fail if any test fails
156  float test_limit = 1.0f; // pass limit re-used for each test
157 
158  // Get sensor_preflight data if available and exit with a fail recorded if not
159  int sensors_sub = orb_subscribe(ORB_ID(sensor_preflight));
160  sensor_preflight_s sensors = {};
161 
162  if (orb_copy(ORB_ID(sensor_preflight), sensors_sub, &sensors) != PX4_OK) {
163  goto out;
164  }
165 
166  // Use the difference between IMU's to detect a bad calibration.
167  // If a single IMU is fitted, the value being checked will be zero so this check will always pass.
168  param_get(param_find("COM_ARM_IMU_ACC"), &test_limit);
169 
170  if (sensors.accel_inconsistency_m_s_s > test_limit) {
171  if (report_status) {
172  mavlink_log_critical(mavlink_log_pub, "PREFLIGHT FAIL: ACCELS INCONSISTENT - CHECK CAL");
173  set_health_flags_healthy(subsystem_info_s::SUBSYSTEM_TYPE_ACC, false, status);
174  set_health_flags_healthy(subsystem_info_s::SUBSYSTEM_TYPE_ACC2, false, status);
175  }
176  success = false;
177  goto out;
178 
179  } else if (sensors.accel_inconsistency_m_s_s > test_limit * 0.8f) {
180  if (report_status) {
181  mavlink_log_info(mavlink_log_pub, "PREFLIGHT ADVICE: ACCELS INCONSISTENT - CHECK CAL");
182  }
183  }
184 
185  // Fail if gyro difference greater than 5 deg/sec and notify if greater than 2.5 deg/sec
186  param_get(param_find("COM_ARM_IMU_GYR"), &test_limit);
187 
188  if (sensors.gyro_inconsistency_rad_s > test_limit) {
189  if (report_status) {
190  mavlink_log_critical(mavlink_log_pub, "PREFLIGHT FAIL: GYROS INCONSISTENT - CHECK CAL");
191  set_health_flags_healthy(subsystem_info_s::SUBSYSTEM_TYPE_GYRO, false, status);
192  set_health_flags_healthy(subsystem_info_s::SUBSYSTEM_TYPE_GYRO2, false, status);
193  }
194 
195  success = false;
196  goto out;
197 
198  } else if (sensors.gyro_inconsistency_rad_s > test_limit * 0.5f) {
199  if (report_status) {
200  mavlink_log_info(mavlink_log_pub, "PREFLIGHT ADVICE: GYROS INCONSISTENT - CHECK CAL");
201  }
202  }
203 
204 out:

392 static bool airspeedCheck(orb_advert_t *mavlink_log_pub, vehicle_status_s &status, bool optional, bool report_fail, bool prearm)
393 {
394  bool present = true;
395  bool success = true;
396 
397  int fd_airspeed = orb_subscribe(ORB_ID(airspeed));
398  airspeed_s airspeed = {};
399 
400  int fd_diffpres = orb_subscribe(ORB_ID(differential_pressure));
401  differential_pressure_s differential_pressure = {};
402 
403  if ((orb_copy(ORB_ID(differential_pressure), fd_diffpres, &differential_pressure) != PX4_OK) ||
404  (hrt_elapsed_time(&differential_pressure.timestamp) > 1000000)) {
405  if (report_fail && !optional) {
406  mavlink_log_critical(mavlink_log_pub, "PREFLIGHT FAIL: AIRSPEED SENSOR MISSING");
407  }
408  present = false;
409  success = false;
410  goto out;
411  }
412 
413  if ((orb_copy(ORB_ID(airspeed), fd_airspeed, &airspeed) != PX4_OK) ||
414  (hrt_elapsed_time(&airspeed.timestamp) > 1000000)) {
415  if (report_fail && !optional) {
416  mavlink_log_critical(mavlink_log_pub, "PREFLIGHT FAIL: AIRSPEED SENSOR MISSING");
417  }
418  present = false;
419  success = false;
420  goto out;
421  }
422 
423  /*
424  * Check if voter thinks the confidence is low. High-end sensors might have virtually zero noise
425  * on the bench and trigger false positives of the voter. Therefore only fail this
426  * for a pre-arm check, as then the cover is off and the natural airflow in the field
427  * will ensure there is not zero noise.
428  */
429  if (prearm && fabsf(airspeed.confidence) < 0.95f) {
430  if (report_fail) {
431  mavlink_log_critical(mavlink_log_pub, "PREFLIGHT FAIL: AIRSPEED SENSOR STUCK");
432  }
433  present = true;
434  success = false;
435  goto out;
436  }
437 
438  /**
439  * Check if differential pressure is off by more than 15Pa which equals ~5m/s when measuring no airspeed.
440  * Negative and positive offsets are considered. Do not check anymore while arming because pitot cover
441  * might have been removed.
442  */
443  if (fabsf(differential_pressure.differential_pressure_filtered_pa) > 15.0f && !prearm) {
444  if (report_fail) {
445  mavlink_log_critical(mavlink_log_pub, "PREFLIGHT FAIL: CHECK AIRSPEED CAL OR PITOT");
446  }
447  present = true;
448  success = false;
449  goto out;
450  }
451 
452 out:
453  set_health_flags(subsystem_info_s::SUBSYSTEM_TYPE_DIFFPRESSURE, present, !optional, success, status);
454  orb_unsubscribe(fd_airspeed);
455  orb_unsubscribe(fd_diffpres);
456  return success;
457 }

603  }
604

607  }

630  } else {
631  // The EKF is using GPS so check for bad quality on key performance indicators
632  bool gps_quality_fail = ((status.gps_check_fail_flags & ((1 << estimator_status_s::GPS_CHECK_FAIL_MIN_SAT_COUNT)
633  + (1 << estimator_status_s::GPS_CHECK_FAIL_MIN_GDOP)
634  + (1 << estimator_status_s::GPS_CHECK_FAIL_MAX_HORZ_ERR)
635  + (1 << estimator_status_s::GPS_CHECK_FAIL_MAX_VERT_ERR)
636  + (1 << estimator_status_s::GPS_CHECK_FAIL_MAX_SPD_ERR))) > 0);
637 
638  if (gps_quality_fail) {
639  if (report_fail) mavlink_log_critical(mavlink_log_pub, "PREFLIGHT FAIL: GPS QUALITY POOR");
640  gps_success = false;
641  success = false;
642  goto out;
643  }
644  }
645  }
646 
647 out:
648  set_health_flags(subsystem_info_s::SUBSYSTEM_TYPE_AHRS, present, !optional, success && present, vehicle_status);
649  set_health_flags(subsystem_info_s::SUBSYSTEM_TYPE_GPS, gps_present, enforce_gps_required, gps_success, vehicle_status);
650  orb_unsubscribe(sub);
651  return success;

661  }

667  const bool checkDynamic = !hil_enabled;

671 
672  /* Perform airspeed check only if circuit breaker is not
673  * engaged and it's not a rotary wing */
674  if (!status_flags.circuit_breaker_engaged_airspd_check && (!status.is_rotary_wing || status.is_vtol)) {
675  checkAirspeed = true;
676  }
677 
678  reportFailures = (reportFailures && status_flags.condition_system_hotplug_timeout && !status_flags.condition_calibration_enabled);
679 
680 #ifdef __PX4_QURT
681  // WARNING: Preflight checks are important and should be added back when
682  // all the sensors are supported
683  PX4_WARN("Preflight checks always pass on Snapdragon.");
684  checkSensors = false;
685 #elif defined(__PX4_POSIX_RPI)
686  PX4_WARN("Preflight checks for mag, acc, gyro always pass on RPI");
687  checkSensors = false;
688 #elif defined(__PX4_POSIX_BBBLUE)
689  PX4_WARN("Preflight checks for mag, acc, gyro always pass on BBBLUE");
690  checkSensors = false;
691 #elif defined(__PX4_POSIX_BEBOP)
692  PX4_WARN("Preflight checks always pass on Bebop.");
693  checkSensors = false;
694 #elif defined(__PX4_POSIX_OCPOC)
695  PX4_WARN("Preflight checks always pass on OcPoC.");
696  checkSensors = false;
697 #endif
698 
699  bool failed = false;
700 
701  /* ---- MAG ---- */
702  if (checkSensors) {
703  bool prime_found = false;
704  int32_t prime_id = 0;
705  param_get(param_find("CAL_MAG_PRIME"), &prime_id);
706  int32_t sys_has_mag = 1;
707  param_get(param_find("SYS_HAS_MAG"), &sys_has_mag);
708 
709  bool mag_fail_reported = false;
710 
711  /* check all sensors individually, but fail only for mandatory ones */
712  for (unsigned i = 0; i < max_optional_mag_count; i++) {
713  bool required = (i < max_mandatory_mag_count) && sys_has_mag == 1;
714  int device_id = -1;
715 
716  if (!magnometerCheck(mavlink_log_pub, status, i, !required, device_id, (reportFailures && !failed && !mag_fail_reported)) && required) {
717  failed = true;
718  mag_fail_reported = true;
719  }
720 
721  if (device_id == prime_id) {
722  prime_found = true;
723  }
724  }
725 
726  /* check if the primary device is present */
727  if (!prime_found && prime_id != 0) {
728  if ((reportFailures && !failed)) {
729  mavlink_log_critical(mavlink_log_pub, "Primary compass not found");
730  }
731  set_health_flags(subsystem_info_s::SUBSYSTEM_TYPE_MAG, false, true, false, status);
732  failed = true;
733  }
734 
735  /* mag consistency checks (need to be performed after the individual checks) */
736  if (!magConsistencyCheck(mavlink_log_pub, status, (reportFailures && !failed))) {
737  failed = true;
738  }
739  }
740 
741  /* ---- ACCEL ---- */
742  if (checkSensors) {
743  bool prime_found = false;
744  int32_t prime_id = 0;
745  param_get(param_find("CAL_ACC_PRIME"), &prime_id);
746 
747  bool accel_fail_reported = false;
748 
749  /* check all sensors individually, but fail only for mandatory ones */
750  for (unsigned i = 0; i < max_optional_accel_count; i++) {
751  bool required = (i < max_mandatory_accel_count);
752  int device_id = -1;
753 
754  if (!accelerometerCheck(mavlink_log_pub, status, i, !required, checkDynamic, device_id, (reportFailures && !failed && !accel_fail_reported)) && required) {
755  failed = true;
756  accel_fail_reported = true;
757  }
758 
759  if (device_id == prime_id) {
760  prime_found = true;
761  }
762  }
763

765  if (!prime_found && prime_id != 0) {
766  if ((reportFailures && !failed)) {
767  mavlink_log_critical(mavlink_log_pub, "Primary accelerometer not found");
768  }
769  set_health_flags(subsystem_info_s::SUBSYSTEM_TYPE_ACC, false, true, false, status);
770  failed = true;
771  }

801  }

803  failed = true;
804  }

808  if (checkSensors) {
809  bool prime_found = false;
810  int32_t prime_id = 0;
811  param_get(param_find("CAL_BARO_PRIME"), &prime_id);
812  int32_t sys_has_baro = 1;
813  param_get(param_find("SYS_HAS_BARO"), &sys_has_baro);

828  prime_found = true;
829  }

833  // // check if the primary device is present
834  if (!prime_found && prime_id != 0) {
835  if (reportFailures && !failed) {
836  mavlink_log_critical(mavlink_log_pub, "Primary barometer not operational");
837  }
838  set_health_flags(subsystem_info_s::SUBSYSTEM_TYPE_ABSPRESSURE, false, true, false, status);
839  failed = true;
840  }
841  }
842 
843  /* ---- IMU CONSISTENCY ---- */
844  // To be performed after the individual sensor checks have completed
845  if (checkSensors) {
846  if (!imuConsistencyCheck(mavlink_log_pub, status, (reportFailures && !failed))) {
847  failed = true;
848  }
849  }
850 
851  /* ---- AIRSPEED ---- */
852  if (checkAirspeed) {
853  int32_t optional = 0;
854  param_get(param_find("FW_ARSP_MODE"), &optional);
855 
856  if (!airspeedCheck(mavlink_log_pub, status, (bool)optional, reportFailures && !failed, prearm) && !(bool)optional) {
857  failed = true;
858  }
859  }
860 
861  /* ---- RC CALIBRATION ---- */
862  if (checkRC) {
863  if (rc_calibration_check(mavlink_log_pub, reportFailures && !failed, status.is_vtol) != OK) {
864  if (reportFailures) {
865  mavlink_log_critical(mavlink_log_pub, "RC calibration check failed");
866  }
867 
868  failed = true;
869 
870  set_health_flags(subsystem_info_s::SUBSYSTEM_TYPE_RCRECEIVER, status_flags.rc_signal_found_once, true, false, status);
871  status_flags.rc_calibration_valid = false;
872  } else {
873  // The calibration is fine, but only set the overall health state to true if the signal is not currently lost
874  status_flags.rc_calibration_valid = true;
875  set_health_flags(subsystem_info_s::SUBSYSTEM_TYPE_RCRECEIVER, status_flags.rc_signal_found_once, true, !status.rc_signal_lost, status);
876  }
877  }
878 
879  /* ---- SYSTEM POWER ---- */
880  if (checkPower) {
881  if (!powerCheck(mavlink_log_pub, status, (reportFailures && !failed), prearm)) {
882  failed = true;
883  }
884  }
885 
886  /* ---- Navigation EKF ---- */
887  // only check EKF2 data if EKF2 is selected as the estimator and GNSS checking is enabled
888  int32_t estimator_type;
889  param_get(param_find("SYS_MC_EST_GROUP"), &estimator_type);
890 
891  if (estimator_type == 2) {
892  // don't report ekf failures for the first 10 seconds to allow time for the filter to start
893  bool report_ekf_fail = (time_since_boot > 10 * 1000000);
894  if (!ekf2Check(mavlink_log_pub, status, false, reportFailures && report_ekf_fail && !failed, checkGNSS)) {
895  failed = true;
896  }
897  }
898 
899  /* Report status */
900  return !failed;
901 }
