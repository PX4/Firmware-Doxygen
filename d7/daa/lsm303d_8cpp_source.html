811 int
812 LSM303D::ioctl(struct file *filp, int cmd, unsigned long arg)
813 {
814  switch (cmd) {
815 
816  case SENSORIOCSPOLLRATE: {
817  switch (arg) {
818 
819  /* switching to manual polling */
820  case SENSOR_POLLRATE_MANUAL:
821  stop();
822  _call_accel_interval = 0;
823  return OK;
824 
825  /* external signalling not supported */
826  case SENSOR_POLLRATE_EXTERNAL:
827 
828  /* zero would be bad */
829  case 0:
830  return -EINVAL;
831 
832  /* set default/max polling rate */
833  case SENSOR_POLLRATE_MAX:
834  return ioctl(filp, SENSORIOCSPOLLRATE, 1600);
835 
836  case SENSOR_POLLRATE_DEFAULT:
837  return ioctl(filp, SENSORIOCSPOLLRATE, LSM303D_ACCEL_DEFAULT_RATE);
838 
839  /* adjust to a legal polling interval in Hz */
840  default: {
841  /* do we need to start internal polling? */
842  bool want_start = (_call_accel_interval == 0);
843 
844  /* convert hz to hrt interval via microseconds */
845  unsigned ticks = 1000000 / arg;
846 
847  /* check against maximum sane rate */
848  if (ticks < 500) {
849  return -EINVAL;
850  }
851 
852  /* adjust filters */
853  accel_set_driver_lowpass_filter((float)arg, _accel_filter_x.get_cutoff_freq());
854 
855  /* update interval for next measurement */
856  /* XXX this is a bit shady, but no other way to adjust... */
857  _call_accel_interval = ticks;
858 
859  _accel_call.period = _call_accel_interval - LSM303D_TIMER_REDUCTION;
860 
861  /* if we need to start the poll state machine, do it */
862  if (want_start) {
863  start();
864  }
865 
866  return OK;
867  }
868  }
869  }
870 
871  case SENSORIOCGPOLLRATE:
872  if (_call_accel_interval == 0) {
873  return SENSOR_POLLRATE_MANUAL;
874  }
875 
876  return 1000000 / _call_accel_interval;
877 
878  case SENSORIOCSQUEUEDEPTH: {
879  /* lower bound is mandatory, upper bound is a sanity check */
880  if ((arg < 1) || (arg > 100)) {
881  return -EINVAL;
882  }
883 
884  irqstate_t flags = px4_enter_critical_section();
885 
886  if (!_accel_reports->resize(arg)) {
887  px4_leave_critical_section(flags);
888  return -ENOMEM;
889  }
890 
891  px4_leave_critical_section(flags);
892 
893  return OK;
894  }
895 
896  case SENSORIOCRESET:
897  reset();
898  return OK;
899 
900  case ACCELIOCSSAMPLERATE:
901  return accel_set_samplerate(arg);
902 
903  case ACCELIOCGSAMPLERATE:
904  return _accel_samplerate;
905 
906  case ACCELIOCSSCALE: {
907  /* copy scale, but only if off by a few percent */
908  struct accel_calibration_s *s = (struct accel_calibration_s *) arg;
909  float sum = s->x_scale + s->y_scale + s->z_scale;
910 
911  if (sum > 2.0f && sum < 4.0f) {
912  memcpy(&_accel_scale, s, sizeof(_accel_scale));
913  return OK;
914 
915  } else {
916  return -EINVAL;
917  }
918  }
919 
920  case ACCELIOCSRANGE:
921  /* arg needs to be in G */
922  return accel_set_range(arg);
923 
924  case ACCELIOCGRANGE:
925  /* convert to m/s^2 and return rounded in G */
926  return (unsigned long)((_accel_range_m_s2) / CONSTANTS_ONE_G + 0.5f);
927 
928  case ACCELIOCGSCALE:
929  /* copy scale out */
930  memcpy((struct accel_calibration_s *) arg, &_accel_scale, sizeof(_accel_scale));
931  return OK;
932 
933  default:
934  /* give it to the superclass */
935  return SPI::ioctl(filp, cmd, arg);
936  }

1353 void
1354 LSM303D::check_registers(void)
1355 {
1356  uint8_t v;
1357 
1358  if ((v = read_reg(_checked_registers[_checked_next])) != _checked_values[_checked_next]) {
1359  /*
1360  if we get the wrong value then we know the SPI bus
1361  or sensor is very sick. We set _register_wait to 20
1362  and wait until we have seen 20 good values in a row
1363  before we consider the sensor to be OK again.
1364  */
1365  perf_count(_bad_registers);
1366 
1367  /*
1368  try to fix the bad register value. We only try to
1369  fix one per loop to prevent a bad sensor hogging the
1370  bus. We skip zero as that is the WHO_AM_I, which
1371  is not writeable
1372  */
1373  if (_checked_next != 0) {
1374  write_reg(_checked_registers[_checked_next], _checked_values[_checked_next]);
1375  }
1376 
1377  _register_wait = 20;
1378  }
1379 
1380  _checked_next = (_checked_next + 1) % LSM303D_NUM_CHECKED_REGISTERS;
1381 }

1383 void
1384 LSM303D::measure()
1385 {
1386  /* status register and data as read back from the device */
1387 
1388 #pragma pack(push, 1)
1389  struct {
1390  uint8_t cmd;
1391  uint8_t status;
1392  int16_t x;
1393  int16_t y;
1394  int16_t z;
1395  } raw_accel_report;
1396 #pragma pack(pop)
1397 
1398  accel_report accel_report;
1399 
1400  /* start the performance counter */
1401  perf_begin(_accel_sample_perf);
1402 
1403  check_registers();
1404 
1405  if (_register_wait != 0) {
1406  // we are waiting for some good transfers before using
1407  // the sensor again.
1408  _register_wait--;
1409  perf_end(_accel_sample_perf);
1410  return;
1411  }
1412 
1413  /* fetch data from the sensor */
1414  memset(&raw_accel_report, 0, sizeof(raw_accel_report));
1415  raw_accel_report.cmd = ADDR_STATUS_A | DIR_READ | ADDR_INCREMENT;
1416  transfer((uint8_t *)&raw_accel_report, (uint8_t *)&raw_accel_report, sizeof(raw_accel_report));
1417 
1418  if (!(raw_accel_report.status & REG_STATUS_A_NEW_ZYXADA)) {
1419  perf_end(_accel_sample_perf);
1420  perf_count(_accel_duplicates);
1421  return;
1422  }
1423 
1424  /*
1425  * 1) Scale raw value to SI units using scaling from datasheet.
1426  * 2) Subtract static offset (in SI units)
1427  * 3) Scale the statically calibrated values with a linear
1428  * dynamically obtained factor
1429  *
1430  * Note: the static sensor offset is the number the sensor outputs
1431  * at a nominally 'zero' input. Therefore the offset has to
1432  * be subtracted.
1433  *
1434  * Example: A gyro outputs a value of 74 at zero angular rate
1435  * the offset is 74 from the origin and subtracting
1436  * 74 from all measurements centers them around zero.
1437  */
1438 
1439  accel_report.timestamp = hrt_absolute_time();
1440 
1441  // use the temperature from the last mag reading
1442  accel_report.temperature = _last_temperature;
1443 
1444  // report the error count as the sum of the number of bad
1445  // register reads and bad values. This allows the higher level
1446  // code to decide if it should use this sensor based on
1447  // whether it has had failures
1448  accel_report.error_count = perf_event_count(_bad_registers) + perf_event_count(_bad_values);
1449 
1450  accel_report.x_raw = raw_accel_report.x;
1451  accel_report.y_raw = raw_accel_report.y;
1452  accel_report.z_raw = raw_accel_report.z;
1453 
1454  float xraw_f = raw_accel_report.x;
1455  float yraw_f = raw_accel_report.y;
1456  float zraw_f = raw_accel_report.z;
1457 
1458  // apply user specified rotation
1459  rotate_3f(_rotation, xraw_f, yraw_f, zraw_f);
1460 
1461  float x_in_new = ((xraw_f * _accel_range_scale) - _accel_scale.x_offset) * _accel_scale.x_scale;
1462  float y_in_new = ((yraw_f * _accel_range_scale) - _accel_scale.y_offset) * _accel_scale.y_scale;
1463  float z_in_new = ((zraw_f * _accel_range_scale) - _accel_scale.z_offset) * _accel_scale.z_scale;
1464 
1465  /*
1466  we have logs where the accelerometers get stuck at a fixed
1467  large value. We want to detect this and mark the sensor as
1468  being faulty
1469  */
1470  if (fabsf(_last_accel[0] - x_in_new) < 0.001f &&
1471  fabsf(_last_accel[1] - y_in_new) < 0.001f &&
1472  fabsf(_last_accel[2] - z_in_new) < 0.001f &&
1473  fabsf(x_in_new) > 20 &&
1474  fabsf(y_in_new) > 20 &&
1475  fabsf(z_in_new) > 20) {
1476  _constant_accel_count += 1;
1477 
1478  } else {
1479  _constant_accel_count = 0;
1480  }
1481 
1482  if (_constant_accel_count > 100) {
1483  // we've had 100 constant accel readings with large
1484  // values. The sensor is almost certainly dead. We
1485  // will raise the error_count so that the top level
1486  // flight code will know to avoid this sensor, but
1487  // we'll still give the data so that it can be logged
1488  // and viewed
1489  perf_count(_bad_values);
1490  _constant_accel_count = 0;
1491  }
1492 
1493  _last_accel[0] = x_in_new;
1494  _last_accel[1] = y_in_new;
1495  _last_accel[2] = z_in_new;
1496 
1497  accel_report.x = _accel_filter_x.apply(x_in_new);
1498  accel_report.y = _accel_filter_y.apply(y_in_new);
1499  accel_report.z = _accel_filter_z.apply(z_in_new);
1500 
1501  matrix::Vector3f aval(x_in_new, y_in_new, z_in_new);
1502  matrix::Vector3f aval_integrated;
1503 
1504  bool accel_notify = _accel_int.put(accel_report.timestamp, aval, aval_integrated, accel_report.integral_dt);
1505  accel_report.x_integral = aval_integrated(0);
1506  accel_report.y_integral = aval_integrated(1);
1507  accel_report.z_integral = aval_integrated(2);
1508 
1509  accel_report.scaling = _accel_range_scale;
1510 
1511  /* return device ID */
1512  accel_report.device_id = _device_id.devid;
1513 
1514  _accel_reports->force(&accel_report);
1515 
1516  /* notify anyone waiting for data */
1517  if (accel_notify) {
1518  poll_notify(POLLIN);
1519 
1520  if (!(_pub_blocked)) {
1521  /* publish it */
1522  orb_publish(ORB_ID(sensor_accel), _accel_topic, &accel_report);
1523  }
1524  }
1525 
1526  _accel_read++;
1527 
1528  /* stop the perf counter */
1529  perf_end(_accel_sample_perf);
1530 }
