78 void
79 GpsFailure::on_active()
80 {
81  switch (_gpsf_state) {
82  case GPSF_STATE_LOITER: {
83  /* Position controller does not run in this mode:
84  * navigator has to publish an attitude setpoint */
85  vehicle_attitude_setpoint_s att_sp = {};
86  att_sp.timestamp = hrt_absolute_time();
87  att_sp.roll_body = math::radians(_param_openlooploiter_roll.get());
88  att_sp.pitch_body = math::radians(_param_openlooploiter_pitch.get());
89  att_sp.thrust = _param_openlooploiter_thrust.get();
90 
91  Quatf q(Eulerf(att_sp.roll_body, att_sp.pitch_body, 0.0f));
92  q.copyTo(att_sp.q_d);
93  att_sp.q_d_valid = true;
94 
95  if (_att_sp_pub != nullptr) {
96  /* publish att sp*/
97  orb_publish(ORB_ID(vehicle_attitude_setpoint), _att_sp_pub, &att_sp);
98 
99  } else {
100  /* advertise and publish */
101  _att_sp_pub = orb_advertise(ORB_ID(vehicle_attitude_setpoint), &att_sp);
102  }
103 
104  /* Measure time */
105  if ((_param_loitertime.get() > FLT_EPSILON) &&
106  (hrt_elapsed_time(&_timestamp_activation) > _param_loitertime.get() * 1e6f)) {
107  /* no recovery, advance the state machine */
108  PX4_WARN("GPS not recovered, switching to next failure state");
109  advance_gpsf();
110  }
111 
112  break;
113  }
114 
115  case GPSF_STATE_TERMINATE:
116  set_gpsf_item();
117  advance_gpsf();
118  break;
119 
120  default:
121  break;
122  }
123 }

125 void
126 GpsFailure::set_gpsf_item()
127 {
128  struct position_setpoint_triplet_s *pos_sp_triplet = _navigator->get_position_setpoint_triplet();
129 
130  /* Set pos sp triplet to invalid to stop pos controller */
131  pos_sp_triplet->previous.valid = false;
132  pos_sp_triplet->current.valid = false;
133  pos_sp_triplet->next.valid = false;
134 
135  switch (_gpsf_state) {
136  case GPSF_STATE_TERMINATE: {
137  /* Request flight termination from commander */
138  _navigator->get_mission_result()->flight_termination = true;
139  _navigator->set_mission_result_updated();
140  PX4_WARN("GPS failure: request flight termination");
141  }
142  break;
143 
144  default:
145  break;
146  }
147 
148  _navigator->set_position_setpoint_triplet_updated();
149 }

151 void
152 GpsFailure::advance_gpsf()
153 {
154  switch (_gpsf_state) {
155  case GPSF_STATE_NONE:
156  _gpsf_state = GPSF_STATE_LOITER;
157  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "Global position failure: fixed bank loiter");
158  break;
159 
160  case GPSF_STATE_LOITER:
161  _gpsf_state = GPSF_STATE_TERMINATE;
162  mavlink_log_emergency(_navigator->get_mavlink_log_pub(), "no GPS recovery, terminating flight");
163  break;
164 
165  case GPSF_STATE_TERMINATE:
166  PX4_WARN("terminate");
167  _gpsf_state = GPSF_STATE_END;
168  break;
169 
170  default:
171  break;
172  }
173 }
