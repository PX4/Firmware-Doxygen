338 int UavcanNode::set_param(int remote_node_id, const char *name, char *value)
339 {
340  uavcan::protocol::param::GetSet::Request req;
341  uavcan::protocol::param::GetSet::Response resp;
342  set_setget_response(&resp);
343  int rv = get_set_param(remote_node_id, name, req);
344 
345  if (rv < 0 || resp.name.empty()) {
346  std::printf("Failed to retrieve param: %s\n", name);
347  rv = -1;
348 
349  } else {
350 
351  rv = 0;
352  req = {};
353 
354  if (resp.value.is(uavcan::protocol::param::Value::Tag::integer_value)) {
355  int64_t i = std::strtoull(value, NULL, 10);
356  int64_t min = resp.min_value.to<uavcan::protocol::param::NumericValue::Tag::integer_value>();
357  int64_t max = resp.max_value.to<uavcan::protocol::param::NumericValue::Tag::integer_value>();
358 
359  if (i >= min && i <= max) {
360  req.value.to<uavcan::protocol::param::Value::Tag::integer_value>() = i;
361 
362  } else {
363  std::printf("Invalid value for: %s must be between %lld and %lld\n", name, min, max);
364  rv = -1;
365  }
366 
367  } else if (resp.value.is(uavcan::protocol::param::Value::Tag::real_value)) {
368  float f = static_cast<float>(std::atof(value));
369  float min = resp.min_value.to<uavcan::protocol::param::NumericValue::Tag::real_value>();
370  float max = resp.max_value.to<uavcan::protocol::param::NumericValue::Tag::real_value>();
371 
372  if (f >= min && f <= max) {
373  req.value.to<uavcan::protocol::param::Value::Tag::real_value>() = f;
374 
375  } else {
376  std::printf("Invalid value for: %s must be between %.4f and %.4f\n", name, static_cast<double>(min),
377  static_cast<double>(max));
378  rv = -1;
379  }
380 
381  } else if (resp.value.is(uavcan::protocol::param::Value::Tag::boolean_value)) {
382  int8_t i = (value[0] == '1' || value[0] == 't') ? 1 : 0;
383  req.value.to<uavcan::protocol::param::Value::Tag::boolean_value>() = i;
384 
385  } else if (resp.value.is(uavcan::protocol::param::Value::Tag::string_value)) {
386  req.value.to<uavcan::protocol::param::Value::Tag::string_value>() = value;
387  }
388 
389  if (rv == 0) {
390  rv = get_set_param(remote_node_id, name, req);
391 
392  if (rv < 0 || resp.name.empty()) {
393  std::printf("Failed to set param: %s\n", name);
394  return -1;
395  }
396 
397  return 0;
398  }
399  }
400 
401  free_setget_response();
402  return rv;

709 void UavcanNode::handle_time_sync(const uavcan::TimerEvent &)
710 {
711 
712  /*
713  * Check whether there are higher priority masters in the network.
714  * If there are, we need to activate the local slave in order to sync with them.
715  */
716  if (_time_sync_slave.isActive()) { // "Active" means that the slave tracks at least one remote master in the network
717  if (_node.getNodeID() < _time_sync_slave.getMasterNodeID()) {
718  /*
719  * We're the highest priority master in the network.
720  * We need to suppress the slave now to prevent it from picking up unwanted sync messages from
721  * lower priority masters.
722  */
723  _time_sync_slave.suppress(true); // SUPPRESS
724 
725  } else {
726  /*
727  * There is at least one higher priority master in the network.
728  * We need to allow the slave to adjust our local clock in order to be in sync.
729  */
730  _time_sync_slave.suppress(false); // UNSUPPRESS
731  }
732 
733  } else {
734  /*
735  * There are no other time sync masters in the network, so we're the only time source.
736  * The slave must be suppressed anyway to prevent it from disrupting the local clock if a new
737  * lower priority master suddenly appears in the network.
738  */
739  _time_sync_slave.suppress(true);
740  }
741 
742  /*
743  * Publish the sync message now, even if we're not a higher priority master.
744  * Other nodes will be able to pick the right master anyway.
745  */
746  _time_sync_master.publish();
747 }

771  PX4_ERR("Failed to start time_sync_slave");
772  _task_should_exit = true;
773  }
774

776  * System RTC setting to the GPS) we would call uavcan_stm32::clock::setUtc() when that
777  * happens, but for now we use adjustUtc with a correction of the hrt so that the
778  * time bases are the same
779  */

782  _master_timer.startPeriodic(uavcan::MonotonicDuration::fromMSec(1000));

787 
788  if (busevent_fd < 0) {
789  PX4_ERR("Failed to open %s", uavcan_stm32::BusEvent::DevName);
790  _task_should_exit = true;
791  }
792

794  * XXX Mixing logic/subscriptions shall be moved into UavcanEscController::update();
795  * IO multiplexing shall be done here.
796  */
797 
798  _node.setModeOperational();
799

801  * This event is needed to wake up the thread on CAN bus activity (RX/TX/Error).
802  * Please note that with such multiplexing it is no longer possible to rely only on
803  * the value returned from poll() to detect whether actuator control has timed out or not.
804  * Instead, all ORB events need to be checked individually (see below).
805  */
806  add_poll_fd(busevent_fd);
807

809  * setup poll to look for actuator direct input if we are
810  * subscribed to the topic
811  */
812  if (_actuator_direct_sub != -1) {
813  _actuator_direct_poll_fd_num = add_poll_fd(_actuator_direct_sub);
814  }

821

823  bool param_updated = false;
824  orb_check(params_sub, &param_updated);
825 
826  if (param_updated) {
827  struct parameter_update_s update;
828  orb_copy(ORB_ID(parameter_update), params_sub, &update);
829  update_params();
830  }
831 
832  switch (_fw_server_action) {
833  case Start:
834  _fw_server_status = start_fw_server();
835  break;
836 
837  case Stop:
838  _fw_server_status = stop_fw_server();
839  break;
840 
841  case CheckFW:
842  _fw_server_status = request_fw_check();
843  break;
844 
845  case None:
846  default:
847  break;
848  }
849 
850  // update actuator controls subscriptions if needed
851  if (_groups_subscribed != _groups_required) {
852  subscribe();
853  _groups_subscribed = _groups_required;
854  }
855 
856  // Mutex is unlocked while the thread is blocked on IO multiplexing
857  (void)pthread_mutex_unlock(&_node_mutex);
858 
859  const int poll_ret = ::poll(_poll_fds, _poll_fds_num, PollTimeoutMs);
860 
861  (void)pthread_mutex_lock(&_node_mutex);
862 
863  node_spin_once(); // Non-blocking
864 
865  bool new_output = false;
866 
867  // this would be bad...
868  if (poll_ret < 0) {
869  DEVICE_LOG("poll error %d", errno);
870  continue;
871 
872  } else {
873  // get controls for required topics
874  bool controls_updated = false;
875 
876  for (unsigned i = 0; i < actuator_controls_s::NUM_ACTUATOR_CONTROL_GROUPS; i++) {
877  if (_control_subs[i] >= 0) {
878  if (_poll_fds[_poll_ids[i]].revents & POLLIN) {
879  controls_updated = true;
880  orb_copy(_control_topics[i], _control_subs[i], &_controls[i]);
881  }
882  }
883  }
884 
885  /*
886  see if we have any direct actuator updates
887  */
888  if (_actuator_direct_sub != -1 &&
889  (_poll_fds[_actuator_direct_poll_fd_num].revents & POLLIN) &&
890  orb_copy(ORB_ID(actuator_direct), _actuator_direct_sub, &_actuator_direct) == OK &&
891  !_test_in_progress) {
892  if (_actuator_direct.nvalues > actuator_outputs_s::NUM_ACTUATOR_OUTPUTS) {
893  _actuator_direct.nvalues = actuator_outputs_s::NUM_ACTUATOR_OUTPUTS;
894  }
895 
896  memcpy(&_outputs.output[0], &_actuator_direct.values[0],
897  _actuator_direct.nvalues * sizeof(float));
898  _outputs.noutputs = _actuator_direct.nvalues;
899  new_output = true;
900  }
901 
902  // can we mix?
903  if (_test_in_progress) {
904  memset(&_outputs, 0, sizeof(_outputs));
905 
906  if (_test_motor.motor_number < actuator_outputs_s::NUM_ACTUATOR_OUTPUTS) {
907  _outputs.output[_test_motor.motor_number] = _test_motor.value * 2.0f - 1.0f;
908  _outputs.noutputs = _test_motor.motor_number + 1;
909  }
910 
911  new_output = true;
912 
913  } else if (controls_updated && (_mixers != nullptr)) {
914 
915  // XXX one output group has 8 outputs max,
916  // but this driver could well serve multiple groups.
917  unsigned num_outputs_max = 8;
918 
919  _mixers->set_airmode(_airmode);
920 
921  // Do mixing
922  _outputs.noutputs = _mixers->mix(&_outputs.output[0], num_outputs_max);
923 
924  new_output = true;
925  }
926  }
927 
928  if (new_output) {
929  // iterate actuators, checking for valid values
930  for (uint8_t i = 0; i < _outputs.noutputs; i++) {
931  // last resort: catch NaN, INF and out-of-band errors
932  if (!isfinite(_outputs.output[i])) {
933  /*
934  * Value is NaN, INF or out of band - set to the minimum value.
935  * This will be clearly visible on the servo status and will limit the risk of accidentally
936  * spinning motors. It would be deadly in flight.
937  */
938  _outputs.output[i] = -1.0f;
939  }
940 
941  // never go below min
942  if (_outputs.output[i] < -1.0f) {
943  _outputs.output[i] = -1.0f;
944  }
945 
946  // never go above max
947  if (_outputs.output[i] > 1.0f) {
948  _outputs.output[i] = 1.0f;
949  }
950  }
951 
952  // Output to the bus
953  _esc_controller.update_outputs(_outputs.output, _outputs.noutputs);
954  _outputs.timestamp = hrt_absolute_time();
955 
956  // use first valid timestamp_sample for latency tracking
957  for (int i = 0; i < actuator_controls_s::NUM_ACTUATOR_CONTROL_GROUPS; i++) {
958  const bool required = _groups_required & (1 << i);
959  const hrt_abstime &timestamp_sample = _controls[i].timestamp_sample;
960 
961  if (required && (timestamp_sample > 0)) {
962  perf_set_elapsed(_perf_control_latency, _outputs.timestamp - timestamp_sample);
963  break;
964  }
965  }
966  }
967 
968 
969  // Check motor test state
970  bool updated = false;
971  orb_check(_test_motor_sub, &updated);
972 
973  if (updated) {
974  orb_copy(ORB_ID(test_motor), _test_motor_sub, &_test_motor);
975 
976  // Update the test status and check that we're not locked down
977  _test_in_progress = (_test_motor.value > 0);
978  _esc_controller.arm_single_esc(_test_motor.motor_number, _test_in_progress);
979  }
980 
981  // Check arming state
982  orb_check(_armed_sub, &updated);
983 
984  if (updated) {
985  orb_copy(ORB_ID(actuator_armed), _armed_sub, &_armed);
986 
987  // Update the armed status and check that we're not locked down and motor
988  // test is not running
989  bool set_armed = _armed.armed && !_armed.lockdown && !_armed.manual_lockdown && !_test_in_progress;
990 
991  arm_actuators(set_armed);
992 
993  if (_armed.soft_stop) {
994  _esc_controller.enable_idle_throttle_when_armed(false);
995 
996  } else {
997  _esc_controller.enable_idle_throttle_when_armed(_idle_throttle_when_armed > 0);
998  }
999  }
1000  }
1001 
1002  orb_unsubscribe(params_sub);
1003 
1004  (void)::close(busevent_fd);
1005 
1006  teardown();
1007 
1008  exit(0);
1009 }
