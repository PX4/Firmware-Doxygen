54 float ECL_PitchController::control_attitude(const struct ECL_ControlData &ctl_data)
55 {
56 
57  /* Do not calculate control signal with bad inputs */
58  if (!(ISFINITE(ctl_data.pitch_setpoint) &&
59  ISFINITE(ctl_data.roll) &&
60  ISFINITE(ctl_data.pitch) &&
61  ISFINITE(ctl_data.airspeed))) {
62  ECL_WARN("not controlling pitch");
63  return _rate_setpoint;
64  }
65

68 
69  /* Apply P controller: rate setpoint from current error and time constant */
70  _rate_setpoint = pitch_error / _tc;

75 float ECL_PitchController::control_bodyrate(const struct ECL_ControlData &ctl_data)
76 {
77  /* Do not calculate control signal with bad inputs */
78  if (!(ISFINITE(ctl_data.roll) &&
79  ISFINITE(ctl_data.pitch) &&
80  ISFINITE(ctl_data.body_y_rate) &&
81  ISFINITE(ctl_data.body_z_rate) &&
82  ISFINITE(ctl_data.yaw_rate_setpoint) &&
83  ISFINITE(ctl_data.airspeed_min) &&
84  ISFINITE(ctl_data.airspeed_max) &&
85  ISFINITE(ctl_data.scaler))) {
86  return math::constrain(_last_output, -1.0f, 1.0f);
87  }
88 
89  /* get the usual dt estimate */
90  uint64_t dt_micros = ecl_elapsed_time(&_last_run);
91  _last_run = ecl_absolute_time();
92  float dt = (float)dt_micros * 1e-6f;
93 
94  /* lock integral for long intervals */
95  bool lock_integrator = ctl_data.lock_integrator;
96 
97  if (dt_micros > 500000) {
98  lock_integrator = true;
99  }
100 
101  _rate_error = _bodyrate_setpoint - ctl_data.body_y_rate;
102 
103  if (!lock_integrator && _k_i > 0.0f) {
104 
105  float id = _rate_error * dt * ctl_data.scaler;
106 
107  /*
108  * anti-windup: do not allow integrator to increase if actuator is at limit
109  */
110  if (_last_output < -1.0f) {
111  /* only allow motion to center: increase value */
112  id = math::max(id, 0.0f);
113 
114  } else if (_last_output > 1.0f) {
115  /* only allow motion to center: decrease value */
116  id = math::min(id, 0.0f);
117  }
118 
119  /* add and constrain */
120  _integrator = math::constrain(_integrator + id * _k_i, -_integrator_max, _integrator_max);
121  }
122 
123  /* Apply PI rate controller and store non-limited output */
124  _last_output = _bodyrate_setpoint * _k_ff * ctl_data.scaler +
125  _rate_error * _k_p * ctl_data.scaler * ctl_data.scaler
126  + _integrator; //scaler is proportional to 1/airspeed
127 
128  return math::constrain(_last_output, -1.0f, 1.0f);
129 }
