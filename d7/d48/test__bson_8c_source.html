100 static int
101 decode_callback(bson_decoder_t decoder, void *private, bson_node_t node)
102 {
103  unsigned len;
104 
105  if (!strcmp(node->name, "bool1")) {
106  if (node->type != BSON_BOOL) {
107  PX4_ERR("FAIL: decoder: bool1 type %d, expected %d", node->type, BSON_BOOL);
108  return 1;
109  }
110 
111  if (node->b != sample_bool) {
112  PX4_ERR("FAIL: decoder: bool1 value %s, expected %s",
113  (node->b ? "true" : "false"),
114  (sample_bool ? "true" : "false"));
115  return 1;
116  }
117 
118  PX4_INFO("PASS: decoder: bool1");
119  return 1;
120  }
121 
122  if (!strcmp(node->name, "int1")) {
123  if (node->type != BSON_INT32) {
124  PX4_ERR("FAIL: decoder: int1 type %d, expected %d", node->type, BSON_INT32);
125  return 1;
126  }
127 
128  if (node->i != sample_small_int) {
129  PX4_ERR("FAIL: decoder: int1 value %" PRIu64 ", expected %d", node->i, sample_small_int);
130  return 1;
131  }
132 
133  warnx("PASS: decoder: int1");
134  return 1;
135  }
136 
137  if (!strcmp(node->name, "int2")) {
138  if (node->type != BSON_INT64) {
139  PX4_ERR("FAIL: decoder: int2 type %d, expected %d", node->type, BSON_INT64);
140  return 1;
141  }
142 
143  if (node->i != sample_big_int) {
144  PX4_ERR("FAIL: decoder: int2 value %" PRIu64 ", expected %" PRIu64, node->i, sample_big_int);
145  return 1;
146  }
147 
148  warnx("PASS: decoder: int2");
149  return 1;
150  }
151 
152  if (!strcmp(node->name, "double1")) {
153  if (node->type != BSON_DOUBLE) {
154  PX4_ERR("FAIL: decoder: double1 type %d, expected %d", node->type, BSON_DOUBLE);
155  return 1;
156  }
157 
158  if (fabs(node->d - sample_double) > 1e-12) {
159  PX4_ERR("FAIL: decoder: double1 value %f, expected %f", node->d, sample_double);
160  return 1;
161  }
162 
163  warnx("PASS: decoder: double1");
164  return 1;
165  }
166 
167  if (!strcmp(node->name, "string1")) {
168  if (node->type != BSON_STRING) {
169  PX4_ERR("FAIL: decoder: string1 type %d, expected %d", node->type, BSON_STRING);
170  return 1;
171  }
172 
173  len = bson_decoder_data_pending(decoder);
174 
175  if (len != strlen(sample_string) + 1) {
176  PX4_ERR("FAIL: decoder: string1 length %d wrong, expected %zd", len, strlen(sample_string) + 1);
177  return 1;
178  }
179 
180  char sbuf[len];
181 
182  if (bson_decoder_copy_data(decoder, sbuf)) {
183  PX4_ERR("FAIL: decoder: string1 copy failed");
184  return 1;
185  }
186 
187  if (bson_decoder_data_pending(decoder) != 0) {
188  PX4_ERR("FAIL: decoder: string1 copy did not exhaust all data");
189  return 1;
190  }
191 
192  if (sbuf[len - 1] != '\0') {
193  PX4_ERR("FAIL: decoder: string1 not 0-terminated");
194  return 1;
195  }
196 
197  if (strcmp(sbuf, sample_string)) {
198  PX4_ERR("FAIL: decoder: string1 value '%s', expected '%s'", sbuf, sample_string);
199  return 1;
200  }
201 
202  warnx("PASS: decoder: string1");
203  return 1;
204  }
205 
206  if (!strcmp(node->name, "data1")) {
207  if (node->type != BSON_BINDATA) {
208  PX4_ERR("FAIL: decoder: data1 type %d, expected %d", node->type, BSON_BINDATA);
209  return 1;
210  }
211 
212  len = bson_decoder_data_pending(decoder);
213 
214  if (len != sizeof(sample_data)) {
215  PX4_ERR("FAIL: decoder: data1 length %d, expected %zu", len, sizeof(sample_data));
216  return 1;
217  }
218 
219  if (node->subtype != BSON_BIN_BINARY) {
220  PX4_ERR("FAIL: decoder: data1 subtype %d, expected %d", node->subtype, BSON_BIN_BINARY);
221  return 1;
222  }
223 
224  uint8_t dbuf[len];
225 
226  if (bson_decoder_copy_data(decoder, dbuf)) {
227  PX4_ERR("FAIL: decoder: data1 copy failed");
228  return 1;
229  }
230 
231  if (bson_decoder_data_pending(decoder) != 0) {
232  PX4_ERR("FAIL: decoder: data1 copy did not exhaust all data");
233  return 1;
234  }
235 
236  if (memcmp(sample_data, dbuf, len) != 0) {
237  PX4_ERR("FAIL: decoder: data1 compare fail");
238  return 1;
239  }
240 
241  PX4_INFO("PASS: decoder: data1");
242  return 1;
243  }
244 
245  if (node->type != BSON_EOO) {
246  PX4_ERR("FAIL: decoder: unexpected node name '%s'", node->name);
247  }
248 
249  return 1;
250 }

262 int
263 test_bson(int argc, char *argv[])
264 {
265  struct bson_encoder_s encoder;
266  struct bson_decoder_s decoder;
267  void *buf;
268  int len;
269 
270  /* encode data to a memory buffer */
271  if (bson_encoder_init_buf(&encoder, NULL, 0)) {
272  PX4_ERR("FAIL: bson_encoder_init_buf");
273  return 1;
274  }
275 
276  encode(&encoder);
277  len = bson_encoder_buf_size(&encoder);
278 
279  if (len <= 0) {
280  PX4_ERR("FAIL: bson_encoder_buf_len");
281  return 1;
282  }
283 
284  buf = bson_encoder_buf_data(&encoder);
285 
286  if (buf == NULL) {
287  PX4_ERR("FAIL: bson_encoder_buf_data");
288  return 1;
289  }
290 
291  /* now test-decode it */
292  if (bson_decoder_init_buf(&decoder, buf, len, decode_callback, NULL)) {
293  PX4_ERR("FAIL: bson_decoder_init_buf");
294  return 1;
295  }
296 
297  decode(&decoder);
298  free(buf);
299 
300  return PX4_OK;
301 }
