118 int
119 MS5611_I2C::read(unsigned offset, void *data, unsigned count)
120 {
121  union _cvt {
122  uint8_t b[4];
123  uint32_t w;
124  } *cvt = (_cvt *)data;
125  uint8_t buf[3];
126 
127  /* read the most recent measurement */
128  uint8_t cmd = 0;
129  int ret = transfer(&cmd, 1, &buf[0], 3);
130 
131  if (ret == PX4_OK) {
132  /* fetch the raw value */
133  cvt->b[0] = buf[2];
134  cvt->b[1] = buf[1];
135  cvt->b[2] = buf[0];
136  cvt->b[3] = 0;
137  }
138 
139  return ret;
140 }

228 int
229 MS5611_I2C::_read_prom()
230 {
231  uint8_t prom_buf[2];
232  union {
233  uint8_t b[2];
234  uint16_t w;
235  } cvt;
236 
237  /*
238  * Wait for PROM contents to be in the device (2.8 ms) in the case we are
239  * called immediately after reset.
240  */
241  usleep(3000);
242 
243  uint8_t last_val = 0;
244  bool bits_stuck = true;
245 
246  /* read and convert PROM words */
247  for (int i = 0; i < 8; i++) {
248  uint8_t cmd = ADDR_PROM_SETUP + (i * 2);
249 
250  if (PX4_OK != transfer(&cmd, 1, &prom_buf[0], 2)) {
251  break;
252  }
253 
254  /* check if all bytes are zero */
255  if (i == 0) {
256  /* initialize to first byte read */
257  last_val = prom_buf[0];
258  }
259 
260  if (prom_buf[0] != last_val || prom_buf[1] != last_val) {
261  bits_stuck = false;
262  }
263 
264  /* assemble 16 bit value and convert from big endian (sensor) to little endian (MCU) */
265  cvt.b[0] = prom_buf[1];
266  cvt.b[1] = prom_buf[0];
267  _prom.c[i] = cvt.w;
268  }
269 
270  /* calculate CRC and return success/failure accordingly */
271  return (ms5611::crc4(&_prom.c[0]) && !bits_stuck) ? PX4_OK : -EIO;
272 }
