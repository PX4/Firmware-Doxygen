149 int
150 PCA8574::ioctl(struct file *filp, int cmd, unsigned long arg)
151 {
152  int ret = ENOTTY;
153 
154  switch (cmd) {
155  case IOX_SET_VALUE ...(IOX_SET_VALUE + 8): {
156  // set the specified on / off state
157  uint8_t position = (1 << (cmd - IOX_SET_VALUE));
158  uint8_t prev = _values_out;
159 
160  if (arg) {
161  _values_out |= position;
162 
163  } else {
164  _values_out &= ~(position);
165  }
166 
167  if (_values_out != prev) {
168  if (_values_out) {
169  _mode = IOX_MODE_ON;
170  }
171 
172  send_led_values();
173  }
174 
175  return OK;
176  }
177 
178  case IOX_SET_MASK:
179  send_led_enable(arg);
180  return OK;
181 
182  case IOX_GET_MASK: {
183  uint8_t val;
184  ret = get(val);
185 
186  if (ret == OK) {
187  return val;
188 
189  } else {
190  return -1;
191  }
192  }
193 
194  case IOX_SET_MODE:
195 
196  if (_mode != (IOX_MODE)arg) {
197 
198  switch ((IOX_MODE)arg) {
199  case IOX_MODE_OFF:
200  _values_out = 0xFF;
201  break;
202 
203  case IOX_MODE_ON:
204  _values_out = 0;
205  break;
206 
207  case IOX_MODE_TEST_OUT:
208  break;
209 
210  default:
211  return -1;
212  }
213 
214  _mode = (IOX_MODE)arg;
215  send_led_values();
216  }
217 
218  return OK;
219 
220  default:
221  // see if the parent class can make any use of it
222  ret = CDev::ioctl(filp, cmd, arg);
223  break;
224  }
225 
226  return ret;
227 }

238 /**
239  * Main loop function
240  */
241 void
242 PCA8574::led()
243 {
244  if (_mode == IOX_MODE_TEST_OUT) {
245 
246  // we count only seven states
247  _counter &= 0xF;
248  _counter++;
249 
250  for (int i = 0; i < 8; i++) {
251  if (i < _counter) {
252  _values_out |= (1 << i);
253 
254  } else {
255  _values_out &= ~(1 << i);
256  }
257  }
258 
259  _update_out = true;
260  _should_run = true;
261 
262  } else if (_mode == IOX_MODE_OFF) {
263  _update_out = true;
264  _should_run = false;
265 
266  } else {
267 
268  // Any of the normal modes
269  if (_blinking > 0) {
270  /* we need to be running to blink */
271  _should_run = true;
272 
273  } else {
274  _should_run = false;
275  }
276  }
277 
278  if (_update_out) {
279  uint8_t msg;
280 
281  if (_blinking) {
282  msg = (_values_out & _blinking & _blink_phase);
283 
284  // wipe out all positions that are marked as blinking
285  msg &= ~(_blinking);
286 
287  // fill blink positions
288  msg |= ((_blink_phase) ? _blinking : 0);
289 
290  _blink_phase = !_blink_phase;
291 
292  } else {
293  msg = _values_out;
294  }
295 
296  int ret = transfer(&msg, sizeof(msg), nullptr, 0);
297 
298  if (!ret) {
299  _update_out = false;
300  }
301  }
302 
303  // check if any activity remains, else stp
304  if (!_should_run) {
305  _running = false;
306  return;
307  }
308 
309  // re-queue ourselves to run again later
310  _running = true;
311  work_queue(LPWORK, &_work, (worker_t)&PCA8574::led_trampoline, this, _led_interval);
312 }
