151 void SendEvent::process_commands()
152 {
153  if (!_subscriber_handler.vehicle_command_updated()) {
154  return;
155  }
156 
157  struct vehicle_command_s cmd;
158 
159  orb_copy(ORB_ID(vehicle_command), _subscriber_handler.get_vehicle_command_sub(), &cmd);
160 
161  bool got_temperature_calibration_command = false, accel = false, baro = false, gyro = false;
162 
163  switch (cmd.command) {
164  case vehicle_command_s::VEHICLE_CMD_PREFLIGHT_CALIBRATION:
165  if ((int)(cmd.param1) == vehicle_command_s::PREFLIGHT_CALIBRATION_TEMPERATURE_CALIBRATION) {
166  gyro = true;
167  got_temperature_calibration_command = true;
168  }
169 
170  if ((int)(cmd.param5) == vehicle_command_s::PREFLIGHT_CALIBRATION_TEMPERATURE_CALIBRATION) {
171  accel = true;
172  got_temperature_calibration_command = true;
173  }
174 
175  if ((int)(cmd.param7) == vehicle_command_s::PREFLIGHT_CALIBRATION_TEMPERATURE_CALIBRATION) {
176  baro = true;
177  got_temperature_calibration_command = true;
178  }
179 
180  if (got_temperature_calibration_command) {
181  if (run_temperature_calibration(accel, baro, gyro) == 0) {
182  answer_command(cmd, vehicle_command_s::VEHICLE_CMD_RESULT_ACCEPTED);
183 
184  } else {
185  answer_command(cmd, vehicle_command_s::VEHICLE_CMD_RESULT_FAILED);
186  }
187  }
188 
189  break;
190  }
191

194 void SendEvent::answer_command(const vehicle_command_s &cmd, unsigned result)
195 {
196  /* publish ACK */
197  vehicle_command_ack_s command_ack = {};
198  command_ack.timestamp = hrt_absolute_time();
199  command_ack.command = cmd.command;
200  command_ack.result = (uint8_t)result;
201  command_ack.target_system = cmd.source_system;
202  command_ack.target_component = cmd.source_component;
203 
204  if (_command_ack_pub != nullptr) {
205  orb_publish(ORB_ID(vehicle_command_ack), _command_ack_pub, &command_ack);
206 
207  } else {
208  _command_ack_pub = orb_advertise_queue(ORB_ID(vehicle_command_ack), &command_ack,
209  vehicle_command_ack_s::ORB_QUEUE_LENGTH);
210  }
211 }

246 int SendEvent::custom_command(int argc, char *argv[])
247 {
248  if (!strcmp(argv[0], "temperature_calibration")) {
249 
250  if (!is_running()) {
251  PX4_ERR("background task not running");
252  return -1;
253  }
254 
255  bool gyro_calib = false, accel_calib = false, baro_calib = false;
256  bool calib_all = true;
257  int myoptind = 1;
258  int ch;
259  const char *myoptarg = nullptr;
260 
261  while ((ch = px4_getopt(argc, argv, "abg", &myoptind, &myoptarg)) != EOF) {
262  switch (ch) {
263  case 'a':
264  accel_calib = true;
265  calib_all = false;
266  break;
267 
268  case 'b':
269  baro_calib = true;
270  calib_all = false;
271  break;
272 
273  case 'g':
274  gyro_calib = true;
275  calib_all = false;
276  break;
277 
278  default:
279  print_usage("unrecognized flag");
280  return 1;
281  }
282  }
283 
284  vehicle_command_s vcmd = {};
285  vcmd.timestamp = hrt_absolute_time();
286  vcmd.param1 = (float)((gyro_calib || calib_all) ? vehicle_command_s::PREFLIGHT_CALIBRATION_TEMPERATURE_CALIBRATION : NAN);
287  vcmd.param2 = NAN;
288  vcmd.param3 = NAN;
289  vcmd.param4 = NAN;
290  vcmd.param5 = ((accel_calib || calib_all) ? vehicle_command_s::PREFLIGHT_CALIBRATION_TEMPERATURE_CALIBRATION : (double)NAN);
291  vcmd.param6 = (double)NAN;
292  vcmd.param7 = (float)((baro_calib || calib_all) ? vehicle_command_s::PREFLIGHT_CALIBRATION_TEMPERATURE_CALIBRATION : NAN);
293  vcmd.command = vehicle_command_s::VEHICLE_CMD_PREFLIGHT_CALIBRATION;
294 
295  orb_advertise_queue(ORB_ID(vehicle_command), &vcmd, vehicle_command_s::ORB_QUEUE_LENGTH);
296 
297  } else {
298  print_usage("unrecognized command");
299  }
300 
301  return 0;
302 }
