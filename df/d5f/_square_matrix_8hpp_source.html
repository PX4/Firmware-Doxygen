301  * cholesky decomposition
302  *
303  * Note: A must be positive definite
304  */
305 template<typename Type, size_t M>
306 SquareMatrix <Type, M> cholesky(const SquareMatrix<Type, M> & A)
307 {
308  SquareMatrix<Type, M> L;
309  for (size_t j = 0; j < M; j++) {
310  for (size_t i = j; i < M; i++) {
311  if (i==j) {
312  float sum = 0;
313  for (size_t k = 0; k < j; k++) {
314  sum += L(j, k)*L(j, k);
315  }
316  Type res = A(j, j) - sum;
317  if (res <= 0) {
318  L(j, j) = 0;
319  } else {
320  L(j, j) = sqrt(res);
321  }
322  } else {
323  float sum = 0;
324  for (size_t k = 0; k < j; k++) {
325  sum += L(i, k)*L(j, k);
326  }
327  if (L(j, j) <= 0) {
328  L(i, j) = 0;
329  } else {
330  L(i, j) = (A(i, j) - sum)/L(j, j);
331  }
332  }
333  }
334  }
335  return L;
336 }
