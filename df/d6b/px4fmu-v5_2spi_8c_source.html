132 __EXPORT int stm32_spi_bus_initialize(void)
133 {
134  /* Configure SPI-based devices */
135 
136  /* Get the SPI port for the Sensors */
137 
138  spi_sensors = stm32_spibus_initialize(PX4_SPI_BUS_SENSORS);
139 
140  if (!spi_sensors) {
141  PX4_ERR("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_SENSORS);
142  return -ENODEV;
143  }
144 
145  /* Default PX4_SPI_BUS_SENSORS to 1MHz and de-assert the known chip selects. */
146 
147  SPI_SETFREQUENCY(spi_sensors, 10000000);
148  SPI_SETBITS(spi_sensors, 8);
149  SPI_SETMODE(spi_sensors, SPIDEV_MODE3);
150 
151  for (int cs = PX4_SENSORS_BUS_FIRST_CS; cs <= PX4_SENSORS_BUS_LAST_CS; cs++) {
152  SPI_SELECT(spi_sensors, cs, false);
153  }
154 
155  /* Get the SPI port for the Memory */
156 
157  spi_memory = stm32_spibus_initialize(PX4_SPI_BUS_MEMORY);
158 
159  if (!spi_memory) {
160  PX4_ERR("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_MEMORY);
161  return -ENODEV;
162  }
163 
164  /* Default PX4_SPI_BUS_MEMORY to 12MHz and de-assert the known chip selects.
165  */
166 
167  SPI_SETFREQUENCY(spi_memory, 12 * 1000 * 1000);
168  SPI_SETBITS(spi_memory, 8);
169  SPI_SETMODE(spi_memory, SPIDEV_MODE3);
170 
171  for (int cs = PX4_MEMORY_BUS_FIRST_CS; cs <= PX4_MEMORY_BUS_LAST_CS; cs++) {
172  SPI_SELECT(spi_memory, cs, false);
173  }
174 
175  /* Get the SPI port for the BARO */
176 
177  spi_baro = stm32_spibus_initialize(PX4_SPI_BUS_BARO);
178 
179  if (!spi_baro) {
180  PX4_ERR("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_BARO);
181  return -ENODEV;
182  }
183 
184  /* MS5611 has max SPI clock speed of 20MHz
185  */
186 
187  SPI_SETFREQUENCY(spi_baro, 20 * 1000 * 1000);
188  SPI_SETBITS(spi_baro, 8);
189  SPI_SETMODE(spi_baro, SPIDEV_MODE3);
190 
191  for (int cs = PX4_BARO_BUS_FIRST_CS; cs <= PX4_BARO_BUS_LAST_CS; cs++) {
192  SPI_SELECT(spi_baro, cs, false);
193  }
194 
195  /* Get the SPI port for the PX4_SPI_EXTERNAL1 */
196 
197  spi_ext = stm32_spibus_initialize(PX4_SPI_BUS_EXTERNAL1);
198 
199  if (!spi_ext) {
200  PX4_ERR("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_EXTERNAL1);
201  return -ENODEV;
202  }
203 
204  SPI_SETFREQUENCY(spi_ext, 8 * 1000 * 1000);
205  SPI_SETBITS(spi_ext, 8);
206  SPI_SETMODE(spi_ext, SPIDEV_MODE3);
207 
208  for (int cs = PX4_EXTERNAL1_BUS_FIRST_CS; cs <= PX4_EXTERNAL1_BUS_LAST_CS; cs++) {
209  SPI_SELECT(spi_ext, cs, false);
210  }
211 
212  /* Get the SPI port for the PX4_SPI_EXTERNAL2 */
213 
214  spi_ext = stm32_spibus_initialize(PX4_SPI_BUS_EXTERNAL2);
215 
216  if (!spi_ext) {
217  PX4_ERR("[boot] FAILED to initialize SPI port %d\n", PX4_SPI_BUS_EXTERNAL2);
218  return -ENODEV;
219  }
220

222  SPI_SETBITS(spi_ext, 8);
223  SPI_SETMODE(spi_ext, SPIDEV_MODE3);
224 
225  for (int cs = PX4_EXTERNAL2_BUS_FIRST_CS; cs <= PX4_EXTERNAL2_BUS_LAST_CS; cs++) {
226  SPI_SELECT(spi_ext, cs, false);
227  }
228 
229  return OK;

241 __EXPORT void stm32_spi1select(FAR struct spi_dev_s *dev, uint32_t devid, bool selected)
242 {
243  /* SPI select is active low, so write !selected to select the device */
244 
245  int sel = (int) devid;
246  ASSERT(PX4_SPI_BUS_ID(sel) == PX4_SPI_BUS_SENSORS);
247 
248  /* Making sure the other peripherals are not selected */
249 
250  for (size_t cs = 0; arraySize(spi1selects_gpio) > 1 && cs < arraySize(spi1selects_gpio); cs++) {
251  if (spi1selects_gpio[cs] != 0) {
252  stm32_gpiowrite(spi1selects_gpio[cs], 1);
253  }
254  }
255 
256  uint32_t gpio = spi1selects_gpio[PX4_SPI_DEV_ID(sel)];
257 
258  if (gpio) {
259  stm32_gpiowrite(gpio, !selected);
260  }
261 }

452 __EXPORT void board_spi_reset(int ms)
453 {
454  /* disable SPI bus */
455  for (size_t cs = 0; arraySize(spi1selects_gpio) > 1 && cs < arraySize(spi1selects_gpio); cs++) {
456  if (spi1selects_gpio[cs] != 0) {
457  stm32_configgpio(_PIN_OFF(spi1selects_gpio[cs]));
458  }
459  }
460 
461  stm32_configgpio(GPIO_SPI1_SCK_OFF);
462  stm32_configgpio(GPIO_SPI1_MISO_OFF);
463  stm32_configgpio(GPIO_SPI1_MOSI_OFF);
464 
465 
466 #if BOARD_USE_DRDY
467  stm32_configgpio(GPIO_DRDY_OFF_SPI1_DRDY1_ICM20689);
468  stm32_configgpio(GPIO_DRDY_OFF_SPI1_DRDY2_BMI055_GYRO);
469  stm32_configgpio(GPIO_DRDY_OFF_SPI1_DRDY3_BMI055_ACC);
470  stm32_configgpio(GPIO_DRDY_OFF_SPI1_DRDY4_ICM20602);
471  stm32_configgpio(GPIO_DRDY_OFF_SPI1_DRDY5_BMI055_GYRO);
472  stm32_configgpio(GPIO_DRDY_OFF_SPI1_DRDY6_BMI055_ACC);
473 #endif
474  /* set the sensor rail off */
475  stm32_gpiowrite(GPIO_VDD_3V3_SENSORS_EN, 0);
476 
477  /* wait for the sensor rail to reach GND */
478  usleep(ms * 1000);
479  warnx("reset done, %d ms", ms);
480 
481  /* re-enable power */
482 
483  /* switch the sensor rail back on */
484  stm32_gpiowrite(GPIO_VDD_3V3_SENSORS_EN, 1);
485 
486  /* wait a bit before starting SPI, different times didn't influence results */
487  usleep(100);
488 
489  /* reconfigure the SPI pins */
490  for (size_t cs = 0; arraySize(spi1selects_gpio) > 1 && cs < arraySize(spi1selects_gpio); cs++) {
491  if (spi1selects_gpio[cs] != 0) {
492  stm32_configgpio(spi1selects_gpio[cs]);
493  }
494  }
495 
496  stm32_configgpio(GPIO_SPI1_SCK);
497  stm32_configgpio(GPIO_SPI1_MISO);
498  stm32_configgpio(GPIO_SPI1_MOSI);
499 
500 #if BOARD_USE_DRDY
501  stm32_configgpio(GPIO_SPI1_DRDY1_ICM20689);
502  stm32_configgpio(GPIO_SPI1_DRDY2_BMI055_GYRO);
503  stm32_configgpio(GPIO_SPI1_DRDY3_BMI055_ACC);
504  stm32_configgpio(GPIO_SPI1_DRDY4_ICM20602);
505  stm32_configgpio(GPIO_SPI1_DRDY5_BMI055_GYRO);
506  stm32_configgpio(GPIO_SPI1_DRDY6_BMI055_ACC);
507 #endif
