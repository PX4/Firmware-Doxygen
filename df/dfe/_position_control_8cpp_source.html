76 void PositionControl::generateThrustYawSetpoint(const float &dt)
77 {
78  if (_skip_controller) {
79  // Already received a valid thrust set-point.
80  // Limit the thrust vector.
81  float thr_mag = _thr_sp.length();
82 
83  if (thr_mag > MPC_THR_MAX.get()) {
84  _thr_sp = _thr_sp.normalized() * MPC_THR_MAX.get();
85 
86  } else if (thr_mag < MPC_MANTHR_MIN.get() && thr_mag > FLT_EPSILON) {
87  _thr_sp = _thr_sp.normalized() * MPC_MANTHR_MIN.get();
88  }
89 
90  // Just set the set-points equal to the current vehicle state.
91  _pos_sp = _pos;
92  _vel_sp = _vel;
93  _acc_sp = _acc;
94 
95  } else {
96  _positionController();
97  _velocityController(dt);
98  }

101 void PositionControl::_interfaceMapping()
102 {
103  // if noting is valid, then apply failsafe landing
104  bool failsafe = false;
105 
106  // Respects FlightTask interface, where NAN-set-points are of no interest
107  // and do not require control. A valide position and velocity setpoint will
108  // be mapped to a desired position setpoint with a feed-forward term.
109  for (int i = 0; i <= 2; i++) {
110 
111  if (PX4_ISFINITE(_pos_sp(i))) {
112  // Position control is required
113 
114  if (!PX4_ISFINITE(_vel_sp(i))) {
115  // Velocity is not used as feedforward term.
116  _vel_sp(i) = 0.0f;
117  }
118 
119  // thrust setpoint is not supported in position control
120  _thr_sp(i) = 0.0f;
121 
122  // to run position control, we require valid position and velocity
123  if (!PX4_ISFINITE(_pos(i)) || !PX4_ISFINITE(_vel(i))) {
124  failsafe = true;
125  }
126 
127  } else if (PX4_ISFINITE(_vel_sp(i))) {
128 
129  // Velocity controller is active without position control.
130  // Set the desired position set-point equal to current position
131  // such that error is zero.
132  if (PX4_ISFINITE(_pos(i))) {
133  _pos_sp(i) = _pos(i);
134 
135  } else {
136  _pos_sp(i) = _pos(i) = 0.0f;
137  }
138 
139  // thrust setpoint is not supported in position control
140  _thr_sp(i) = 0.0f;
141 
142  // to run velocity control, we require valid velocity
143  if (!PX4_ISFINITE(_vel(i))) {
144  failsafe = true;
145  }
146 
147  } else if (PX4_ISFINITE(_thr_sp(i))) {
148 
149  // Thrust setpoint was generated from sticks directly.
150  // Set all other set-points equal MC states.
151  if (PX4_ISFINITE(_pos(i))) {
152  _pos_sp(i) = _pos(i);
153 
154  } else {
155  _pos_sp(i) = _pos(i) = 0.0f;
156  }
157 
158  if (PX4_ISFINITE(_vel(i))) {
159  _vel_sp(i) = _vel(i);
160 
161  } else {
162  _vel_sp(i) = _vel(i) = 0.0f;
163  }
164 
165  // Reset the Integral term.
166  _thr_int(i) = 0.0f;
167  // Don't require velocity derivative.
168  _vel_dot(i) = 0.0f;
169 
170  } else {
171  // nothing is valid. do failsafe
172  failsafe = true;
173  }
174  }
175 
176  // ensure that vel_dot is finite, otherwise set to 0
177  if (!PX4_ISFINITE(_vel_dot(0)) || !PX4_ISFINITE(_vel_dot(1))) {
178  _vel_dot(0) = _vel_dot(1) = 0.0f;
179  }
180 
181  if (!PX4_ISFINITE(_vel_dot(2))) {
182  _vel_dot(2) = 0.0f;
183  }
184 
185  if (!PX4_ISFINITE(_yawspeed_sp)) {
186  // Set the yawspeed to 0 since not used.
187  _yawspeed_sp = 0.0f;
188  }
189 
190  if (!PX4_ISFINITE(_yaw_sp)) {
191  // Set the yaw-sp equal the current yaw.
192  // That is the best we can do and it also
193  // agrees with FlightTask-interface definition.
194  if (PX4_ISFINITE(_yaw)) {
195  _yaw_sp = _yaw;
196 
197  } else {
198  failsafe = true;
199  }
200  }
201 
202  // check failsafe
203  if (failsafe) {
204  _skip_controller = true;
205 
206  // point the thrust upwards
207  _thr_sp(0) = _thr_sp(1) = 0.0f;
208  // throttle down such that vehicle goes down with
209  // 70% of throttle range between min and hover
210  _thr_sp(2) = -(MPC_THR_MIN.get() + (MPC_THR_HOVER.get() - MPC_THR_MIN.get()) * 0.7f);
211  }
212 }

214 void PositionControl::_positionController()
215 {
216  // P-position controller
217  Vector3f vel_sp_position = (_pos_sp - _pos).emult(Vector3f(MPC_XY_P.get(), MPC_XY_P.get(), MPC_Z_P.get()));
218  _vel_sp = vel_sp_position + _vel_sp;
219 
220  // Constrain horizontal velocity by prioritizing the velocity component along the
221  // the desired position setpoint over the feed-forward term.
222  Vector2f vel_sp_xy = ControlMath::constrainXY(Vector2f(&vel_sp_position(0)),
223  Vector2f(&(_vel_sp - vel_sp_position)(0)), _constraints.speed_xy);
224  _vel_sp(0) = vel_sp_xy(0);
225  _vel_sp(1) = vel_sp_xy(1);
226  // Constrain velocity in z-direction.
227  _vel_sp(2) = math::constrain(_vel_sp(2), -_constraints.speed_up, _constraints.speed_down);
228 }

230 void PositionControl::_velocityController(const float &dt)
231 {
232 
233  // Generate desired thrust setpoint.
234  // PID
235  // u_des = P(vel_err) + D(vel_err_dot) + I(vel_integral)
236  // Umin <= u_des <= Umax
237  //
238  // Anti-Windup:
239  // u_des = _thr_sp; r = _vel_sp; y = _vel
240  // u_des >= Umax and r - y >= 0 => Saturation = true
241  // u_des >= Umax and r - y <= 0 => Saturation = false
242  // u_des <= Umin and r - y <= 0 => Saturation = true
243  // u_des <= Umin and r - y >= 0 => Saturation = false
244  //
245  // Notes:
246  // - PID implementation is in NED-frame
247  // - control output in D-direction has priority over NE-direction
248  // - the equilibrium point for the PID is at hover-thrust
249  // - the maximum tilt cannot exceed 90 degrees. This means that it is
250  // not possible to have a desired thrust direction pointing in the positive
251  // D-direction (= downward)
252  // - the desired thrust in D-direction is limited by the thrust limits
253  // - the desired thrust in NE-direction is limited by the thrust excess after
254  // consideration of the desired thrust in D-direction. In addition, the thrust in
255  // NE-direction is also limited by the maximum tilt.
256 
257  Vector3f vel_err = _vel_sp - _vel;
258 
259  // Consider thrust in D-direction.
260  float thrust_desired_D = MPC_Z_VEL_P.get() * vel_err(2) + MPC_Z_VEL_D.get() * _vel_dot(2) + _thr_int(
261  2) - MPC_THR_HOVER.get();
262 
263  // The Thrust limits are negated and swapped due to NED-frame.
264  float uMax = -MPC_THR_MIN.get();
265  float uMin = -MPC_THR_MAX.get();
266 
267  // Apply Anti-Windup in D-direction.
268  bool stop_integral_D = (thrust_desired_D >= uMax && vel_err(2) >= 0.0f) ||
269  (thrust_desired_D <= uMin && vel_err(2) <= 0.0f);
270 
271  if (!stop_integral_D) {
272  _thr_int(2) += vel_err(2) * MPC_Z_VEL_I.get() * dt;
273 
274  // limit thrust integral
275  _thr_int(2) = math::min(fabsf(_thr_int(2)), MPC_THR_MAX.get()) * math::sign(_thr_int(2));
276  }
277 
278  // Saturate thrust setpoint in D-direction.
279  _thr_sp(2) = math::constrain(thrust_desired_D, uMin, uMax);
280 
281  if (fabsf(_thr_sp(0)) + fabsf(_thr_sp(1)) > FLT_EPSILON) {
282  // Thrust set-point in NE-direction is already provided. Only
283  // scaling by the maximum tilt is required.
284  float thr_xy_max = fabsf(_thr_sp(2)) * tanf(_constraints.tilt);
285  _thr_sp(0) *= thr_xy_max;
286  _thr_sp(1) *= thr_xy_max;
287 
288  } else {
289  // PID-velocity controller for NE-direction.
290  Vector2f thrust_desired_NE;
291  thrust_desired_NE(0) = MPC_XY_VEL_P.get() * vel_err(0) + MPC_XY_VEL_D.get() * _vel_dot(0) + _thr_int(0);
292  thrust_desired_NE(1) = MPC_XY_VEL_P.get() * vel_err(1) + MPC_XY_VEL_D.get() * _vel_dot(1) + _thr_int(1);
293 
294  // Get maximum allowed thrust in NE based on tilt and excess thrust.
295  float thrust_max_NE_tilt = fabsf(_thr_sp(2)) * tanf(_constraints.tilt);
296  float thrust_max_NE = sqrtf(MPC_THR_MAX.get() * MPC_THR_MAX.get() - _thr_sp(2) * _thr_sp(2));
297  thrust_max_NE = math::min(thrust_max_NE_tilt, thrust_max_NE);
298 
299  // Saturate thrust in NE-direction.
300  _thr_sp(0) = thrust_desired_NE(0);
301  _thr_sp(1) = thrust_desired_NE(1);
302 
303  if (thrust_desired_NE * thrust_desired_NE > thrust_max_NE * thrust_max_NE) {
304  float mag = thrust_desired_NE.length();
305  _thr_sp(0) = thrust_desired_NE(0) / mag * thrust_max_NE;
306  _thr_sp(1) = thrust_desired_NE(1) / mag * thrust_max_NE;
307  }
308 
309  // Get the direction of (r-y) in NE-direction.
310  float direction_NE = Vector2f(&vel_err(0)) * Vector2f(&_vel_sp(0));
311 
312  // Apply Anti-Windup in NE-direction.
313  bool stop_integral_NE = (thrust_desired_NE * thrust_desired_NE >= thrust_max_NE * thrust_max_NE &&
314  direction_NE >= 0.0f);
315 
316  if (!stop_integral_NE) {
317  _thr_int(0) += vel_err(0) * MPC_XY_VEL_I.get() * dt;
318  _thr_int(1) += vel_err(1) * MPC_XY_VEL_I.get() * dt;
319 
320  // magnitude of thrust integral can never exceed maximum throttle in NE
321  float integral_mag_NE = Vector2f(&_thr_int(0)).length();
322 
323  if (integral_mag_NE > 0.0f && integral_mag_NE > thrust_max_NE) {
324  _thr_int(0) = _thr_int(0) / integral_mag_NE * thrust_max_NE;
325  _thr_int(1) = _thr_int(1) / integral_mag_NE * thrust_max_NE;
326  }
327  }
328  }
