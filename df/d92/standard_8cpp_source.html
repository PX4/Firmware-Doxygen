111 void Standard::update_vtol_state()
112 {
113  /* After flipping the switch the vehicle will start the pusher (or tractor) motor, picking up
114  * forward speed. After the vehicle has picked up enough speed the rotors shutdown.
115  * For the back transition the pusher motor is immediately stopped and rotors reactivated.
116  */
117 
118  float mc_weight = _mc_roll_weight;
119  float time_since_trans_start = (float)(hrt_absolute_time() - _vtol_schedule.transition_start) * 1e-6f;
120 
121  if (!_attc->is_fixed_wing_requested()) {
122 
123  // the transition to fw mode switch is off
124  if (_vtol_schedule.flight_mode == MC_MODE) {
125  // in mc mode
126  _vtol_schedule.flight_mode = MC_MODE;
127  mc_weight = 1.0f;
128  _pusher_throttle = 0.0f;
129  _reverse_output = 0.0f;
130 
131  } else if (_vtol_schedule.flight_mode == FW_MODE) {
132  // transition to mc mode
133  if (_vtol_vehicle_status->vtol_transition_failsafe == true) {
134  // Failsafe event, engage mc motors immediately
135  _vtol_schedule.flight_mode = MC_MODE;
136  _pusher_throttle = 0.0f;
137  _reverse_output = 0.0f;
138 
139 
140  } else {
141  // Regular backtransition
142  _vtol_schedule.flight_mode = TRANSITION_TO_MC;
143  _vtol_schedule.transition_start = hrt_absolute_time();
144  _reverse_output = _params_standard.reverse_output;
145 
146  }
147 
148  } else if (_vtol_schedule.flight_mode == TRANSITION_TO_FW) {
149  // failsafe back to mc mode
150  _vtol_schedule.flight_mode = MC_MODE;
151  mc_weight = 1.0f;
152  _pusher_throttle = 0.0f;
153  _reverse_output = 0.0f;
154 
155 
156  } else if (_vtol_schedule.flight_mode == TRANSITION_TO_MC) {
157  // transition to MC mode if transition time has passed or forward velocity drops below MPC cruise speed
158 
159  const matrix::Dcmf R_to_body(matrix::Quatf(_v_att->q).inversed());
160  const matrix::Vector3f vel = R_to_body * matrix::Vector3f(_local_pos->vx, _local_pos->vy, _local_pos->vz);
161 
162  float x_vel = vel(0);
163 
164  if (time_since_trans_start > _params->back_trans_duration ||
165  (_local_pos->v_xy_valid && x_vel <= _params->mpc_xy_cruise)) {
166  _vtol_schedule.flight_mode = MC_MODE;
167  }
168 
169  }
170 
171  } else {
172  // the transition to fw mode switch is on
173  if (_vtol_schedule.flight_mode == MC_MODE || _vtol_schedule.flight_mode == TRANSITION_TO_MC) {
174  // start transition to fw mode
175  /* NOTE: The failsafe transition to fixed-wing was removed because it can result in an
176  * unsafe flying state. */
177  _vtol_schedule.flight_mode = TRANSITION_TO_FW;
178  _vtol_schedule.transition_start = hrt_absolute_time();
179 
180  } else if (_vtol_schedule.flight_mode == FW_MODE) {
181  // in fw mode
182  _vtol_schedule.flight_mode = FW_MODE;
183  mc_weight = 0.0f;
184 
185  } else if (_vtol_schedule.flight_mode == TRANSITION_TO_FW) {
186  // continue the transition to fw mode while monitoring airspeed for a final switch to fw mode
187  if (((_params->airspeed_disabled ||
188  _airspeed->indicated_airspeed_m_s >= _params->transition_airspeed) &&
189  time_since_trans_start > _params->front_trans_time_min) ||
190  can_transition_on_ground()) {
191 
192  _vtol_schedule.flight_mode = FW_MODE;
193 
194  // don't set pusher throttle here as it's being ramped up elsewhere
195  _trans_finished_ts = hrt_absolute_time();
196  }
197 
198  }
199  }
200 
201  _mc_roll_weight = mc_weight;
202  _mc_pitch_weight = mc_weight;
203  _mc_yaw_weight = mc_weight;
204  _mc_throttle_weight = mc_weight;
205 
206  // map specific control phases to simple control modes
207  switch (_vtol_schedule.flight_mode) {
208  case MC_MODE:
209  _vtol_mode = mode::ROTARY_WING;
210  break;
211 
212  case FW_MODE:
213  _vtol_mode = mode::FIXED_WING;
214  break;
215 
216  case TRANSITION_TO_FW:
217  _vtol_mode = mode::TRANSITION_TO_FW;
218  break;
219 
220  case TRANSITION_TO_MC:
221  _vtol_mode = mode::TRANSITION_TO_MC;
222  break;
223  }
224 }

226 void Standard::update_transition_state()
227 {
228  float mc_weight = 1.0f;
229  float time_since_trans_start = (float)(hrt_absolute_time() - _vtol_schedule.transition_start) * 1e-6f;
230 
231  VtolType::update_transition_state();
232 
233  // copy virtual attitude setpoint to real attitude setpoint
234  memcpy(_v_att_sp, _mc_virtual_att_sp, sizeof(vehicle_attitude_setpoint_s));
235 
236  if (_vtol_schedule.flight_mode == TRANSITION_TO_FW) {
237  if (_params_standard.pusher_ramp_dt <= 0.0f) {
238  // just set the final target throttle value
239  _pusher_throttle = _params->front_trans_throttle;
240 
241  } else if (_pusher_throttle <= _params->front_trans_throttle) {
242  // ramp up throttle to the target throttle value
243  _pusher_throttle = _params->front_trans_throttle * time_since_trans_start / _params_standard.pusher_ramp_dt;
244  }
245 
246  // do blending of mc and fw controls if a blending airspeed has been provided and the minimum transition time has passed
247  if (_airspeed_trans_blend_margin > 0.0f &&
248  _airspeed->indicated_airspeed_m_s > 0.0f &&
249  _airspeed->indicated_airspeed_m_s >= _params->airspeed_blend &&
250  time_since_trans_start > _params->front_trans_time_min) {
251 
252  mc_weight = 1.0f - fabsf(_airspeed->indicated_airspeed_m_s - _params->airspeed_blend) /
253  _airspeed_trans_blend_margin;
254  // time based blending when no airspeed sensor is set
255 
256  } else if (_params->airspeed_disabled) {
257  mc_weight = 1.0f - time_since_trans_start / _params->front_trans_time_min;
258  mc_weight = math::constrain(2.0f * mc_weight, 0.0f, 1.0f);
259 
260  }
261 
262  // ramp up FW_PSP_OFF
263  _v_att_sp->pitch_body = _params_standard.pitch_setpoint_offset * (1.0f - mc_weight);
264  matrix::Quatf q_sp(matrix::Eulerf(_v_att_sp->roll_body, _v_att_sp->pitch_body, _v_att_sp->yaw_body));
265  q_sp.copyTo(_v_att_sp->q_d);
266  _v_att_sp->q_d_valid = true;
267 
268  // check front transition timeout
269  if (_params->front_trans_timeout > FLT_EPSILON) {
270  if (time_since_trans_start > _params->front_trans_timeout) {
271  // transition timeout occured, abort transition
272  _attc->abort_front_transition("Transition timeout");
273  }
274  }
275 
276  } else if (_vtol_schedule.flight_mode == TRANSITION_TO_MC) {
277 
278  // maintain FW_PSP_OFF
279  _v_att_sp->pitch_body = _params_standard.pitch_setpoint_offset;
280  matrix::Quatf q_sp(matrix::Eulerf(_v_att_sp->roll_body, _v_att_sp->pitch_body, _v_att_sp->yaw_body));
281  q_sp.copyTo(_v_att_sp->q_d);
282  _v_att_sp->q_d_valid = true;
283 
284  _pusher_throttle = 0.0f;
285 
286  if (time_since_trans_start >= _params_standard.reverse_delay) {
287  // Handle throttle reversal for active breaking
288  float thrscale = (time_since_trans_start - _params_standard.reverse_delay) / (_params_standard.pusher_ramp_dt);
289  thrscale = math::constrain(thrscale, 0.0f, 1.0f);
290  _pusher_throttle = thrscale * _params->back_trans_throttle;
291  }
292 
293  // continually increase mc attitude control as we transition back to mc mode
294  if (_params_standard.back_trans_ramp > FLT_EPSILON) {
295  mc_weight = time_since_trans_start / _params_standard.back_trans_ramp;
296 
297  }
298 
299  // in back transition we need to start the MC motors again
300  if (_motor_state != ENABLED) {
301  _motor_state = set_motor_state(_motor_state, ENABLED);
302  }
303  }
304 
305  mc_weight = math::constrain(mc_weight, 0.0f, 1.0f);
306 
307  _mc_roll_weight = mc_weight;
308  _mc_pitch_weight = mc_weight;
309  _mc_yaw_weight = mc_weight;
310  _mc_throttle_weight = mc_weight;
311 }

313 void Standard::update_mc_state()
314 {
315  VtolType::update_mc_state();
316 
317  // if the thrust scale param is zero or the drone is on manual mode,
318  // then the pusher-for-pitch strategy is disabled and we can return
319  if (_params_standard.forward_thrust_scale < FLT_EPSILON ||
320  !_v_control_mode->flag_control_position_enabled) {
321  return;
322  }
323 
324  // Do not engage pusher assist during a failsafe event
325  // There could be a problem with the fixed wing drive
326  if (_attc->get_vtol_vehicle_status()->vtol_transition_failsafe) {
327  return;
328  }
329 
330  // disable pusher assist during landing
331  if (_attc->get_pos_sp_triplet()->current.valid
332  && _attc->get_pos_sp_triplet()->current.type == position_setpoint_s::SETPOINT_TYPE_LAND) {
333  return;
334  }
335 
336  matrix::Dcmf R(matrix::Quatf(_v_att->q));
337  matrix::Dcmf R_sp(matrix::Quatf(_v_att_sp->q_d));
338  matrix::Eulerf euler(R);
339  matrix::Eulerf euler_sp(R_sp);
340  _pusher_throttle = 0.0f;
341 
342  // direction of desired body z axis represented in earth frame
343  matrix::Vector3f body_z_sp(R_sp(0, 2), R_sp(1, 2), R_sp(2, 2));
344 
345  // rotate desired body z axis into new frame which is rotated in z by the current
346  // heading of the vehicle. we refer to this as the heading frame.
347  matrix::Dcmf R_yaw = matrix::Eulerf(0.0f, 0.0f, -euler(2));
348  body_z_sp = R_yaw * body_z_sp;
349  body_z_sp.normalize();
350 
351  // calculate the desired pitch seen in the heading frame
352  // this value corresponds to the amount the vehicle would try to pitch forward
353  float pitch_forward = atan2f(body_z_sp(0), body_z_sp(2));
354 
355  // only allow pitching forward up to threshold, the rest of the desired
356  // forward acceleration will be compensated by the pusher
357  if (pitch_forward < -_params_standard.down_pitch_max) {
358  // desired roll angle in heading frame stays the same
359  float roll_new = -asinf(body_z_sp(1));
360 
361  _pusher_throttle = (sinf(-pitch_forward) - sinf(_params_standard.down_pitch_max))
362  * _params_standard.forward_thrust_scale;
363 
364  // return the vehicle to level position
365  float pitch_new = 0.0f;
366 
367  // create corrected desired body z axis in heading frame
368  matrix::Dcmf R_tmp = matrix::Eulerf(roll_new, pitch_new, 0.0f);
369  matrix::Vector3f tilt_new(R_tmp(0, 2), R_tmp(1, 2), R_tmp(2, 2));
370 
371  // rotate the vector into a new frame which is rotated in z by the desired heading
372  // with respect to the earh frame.
373  const float yaw_error = wrap_pi(euler_sp(2) - euler(2));
374  matrix::Dcmf R_yaw_correction = matrix::Eulerf(0.0f, 0.0f, -yaw_error);
375  tilt_new = R_yaw_correction * tilt_new;
376 
377  // now extract roll and pitch setpoints
378  _v_att_sp->pitch_body = atan2f(tilt_new(0), tilt_new(2));
379  _v_att_sp->roll_body = -asinf(tilt_new(1));
380  R_sp = matrix::Eulerf(_v_att_sp->roll_body, _v_att_sp->pitch_body, euler_sp(2));
381  matrix::Quatf q_sp(R_sp);
382  q_sp.copyTo(_v_att_sp->q_d);
383  }
384 
385  _pusher_throttle = _pusher_throttle < 0.0f ? 0.0f : _pusher_throttle;
386 
387 }

398 void Standard::fill_actuator_outputs()
399 {
400  // multirotor controls
401  _actuators_out_0->timestamp = hrt_absolute_time();
402  _actuators_out_0->timestamp_sample = _actuators_mc_in->timestamp_sample;
403 
404  // roll
405  _actuators_out_0->control[actuator_controls_s::INDEX_ROLL] =
406  _actuators_mc_in->control[actuator_controls_s::INDEX_ROLL] * _mc_roll_weight;
407  // pitch
408  _actuators_out_0->control[actuator_controls_s::INDEX_PITCH] =
409  _actuators_mc_in->control[actuator_controls_s::INDEX_PITCH] * _mc_pitch_weight;
410  // yaw
411  _actuators_out_0->control[actuator_controls_s::INDEX_YAW] =
412  _actuators_mc_in->control[actuator_controls_s::INDEX_YAW] * _mc_yaw_weight;
413  // throttle
414  _actuators_out_0->control[actuator_controls_s::INDEX_THROTTLE] =
415  _actuators_mc_in->control[actuator_controls_s::INDEX_THROTTLE] * _mc_throttle_weight;
416 
417 
418  // fixed wing controls
419  _actuators_out_1->timestamp = hrt_absolute_time();
420  _actuators_out_1->timestamp_sample = _actuators_fw_in->timestamp_sample;
421 
422  if (_vtol_schedule.flight_mode != MC_MODE) {
423  // roll
424  _actuators_out_1->control[actuator_controls_s::INDEX_ROLL] =
425  -_actuators_fw_in->control[actuator_controls_s::INDEX_ROLL];
426 
427  // pitch
428  _actuators_out_1->control[actuator_controls_s::INDEX_PITCH] =
429  _actuators_fw_in->control[actuator_controls_s::INDEX_PITCH];
430  // yaw
431  _actuators_out_1->control[actuator_controls_s::INDEX_YAW] =
432  _actuators_fw_in->control[actuator_controls_s::INDEX_YAW];
433 
434  _actuators_out_1->control[actuator_controls_s::INDEX_AIRBRAKES] = _reverse_output;
435 
436  } else {
437 
438  if (_params->elevons_mc_lock) {
439  // zero outputs when inactive
440  _actuators_out_1->control[actuator_controls_s::INDEX_ROLL] = 0.0f;
441  _actuators_out_1->control[actuator_controls_s::INDEX_PITCH] = 0.0f;
442  _actuators_out_1->control[actuator_controls_s::INDEX_YAW] = 0.0f;
443  _actuators_out_1->control[actuator_controls_s::INDEX_AIRBRAKES] = 0.0f;
444 
445  } else {
446  // roll
447  _actuators_out_1->control[actuator_controls_s::INDEX_ROLL] =
448  -_actuators_fw_in->control[actuator_controls_s::INDEX_ROLL];
449 
450  // pitch
451  _actuators_out_1->control[actuator_controls_s::INDEX_PITCH] =
452  _actuators_fw_in->control[actuator_controls_s::INDEX_PITCH];
453 
454  _actuators_out_1->control[actuator_controls_s::INDEX_YAW] = 0.0f;
455  _actuators_out_1->control[actuator_controls_s::INDEX_AIRBRAKES] = 0.0f;
456  }
457  }
458 
459  // set the fixed wing throttle control
460  if (_vtol_schedule.flight_mode == FW_MODE) {
461 
462  // take the throttle value commanded by the fw controller
463  _actuators_out_1->control[actuator_controls_s::INDEX_THROTTLE] =
464  _actuators_fw_in->control[actuator_controls_s::INDEX_THROTTLE];
465 
466  } else {
467  // otherwise we may be ramping up the throttle during the transition to fw mode
468  _actuators_out_1->control[actuator_controls_s::INDEX_THROTTLE] = _pusher_throttle;
469  }
