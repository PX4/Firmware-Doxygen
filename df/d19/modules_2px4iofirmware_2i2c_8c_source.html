98 void
99 interface_init(void)
100 {
101  debug("i2c init");
102 
103  /* allocate DMA handles and initialise DMA */
104  rx_dma = stm32_dmachannel(DMACHAN_I2C1_RX);
105  i2c_rx_setup();
106  tx_dma = stm32_dmachannel(DMACHAN_I2C1_TX);
107  i2c_tx_setup();
108 
109  /* enable the i2c block clock and reset it */
110  modifyreg32(STM32_RCC_APB1ENR, 0, RCC_APB1ENR_I2C1EN);
111  modifyreg32(STM32_RCC_APB1RSTR, 0, RCC_APB1RSTR_I2C1RST);
112  modifyreg32(STM32_RCC_APB1RSTR, RCC_APB1RSTR_I2C1RST, 0);
113 
114  /* configure the i2c GPIOs */
115  px4_arch_configgpio(GPIO_I2C1_SCL);
116  px4_arch_configgpio(GPIO_I2C1_SDA);
117 
118  /* soft-reset the block */
119  rCR1 |= I2C_CR1_SWRST;
120  rCR1 = 0;
121 
122  /* set for DMA operation */
123  rCR2 |= I2C_CR2_ITEVFEN | I2C_CR2_ITERREN | I2C_CR2_DMAEN;
124 
125  /* set the frequency value in CR2 */
126  rCR2 &= ~I2C_CR2_FREQ_MASK;
127  rCR2 |= STM32_PCLK1_FREQUENCY / 1000000;
128 
129  /* set divisor and risetime for fast mode */
130  uint16_t result = STM32_PCLK1_FREQUENCY / (400000 * 25);
131 
132  if (result < 1) {
133  result = 1;
134  }
135 
136  result = 3;
137  rCCR &= ~I2C_CCR_CCR_MASK;
138  rCCR |= I2C_CCR_DUTY | I2C_CCR_FS | result;
139  rTRISE = (uint16_t)((((STM32_PCLK1_FREQUENCY / 1000000) * 300) / 1000) + 1);
140 
141  /* set our device address */
142  rOAR1 = 0x1a << 1;
143 
144  /* enable event interrupts */
145  irq_attach(STM32_IRQ_I2C1EV, i2c_interrupt, NULL);
146  irq_attach(STM32_IRQ_I2C1ER, i2c_interrupt, NULL);
147  up_enable_irq(STM32_IRQ_I2C1EV);
148  up_enable_irq(STM32_IRQ_I2C1ER);
149 
150  /* and enable the I2C port */
151  rCR1 |= I2C_CR1_ACK | I2C_CR1_PE;
152 
153 #ifdef DEBUG
154  i2c_dump();
155 #endif
156 }

194 static int
195 i2c_interrupt(int irq, FAR void *context, FAR void *args)
196 {
197  uint16_t sr1 = rSR1;
198 
199  if (sr1 & (I2C_SR1_STOPF | I2C_SR1_AF | I2C_SR1_ADDR)) {
200 
201  if (sr1 & I2C_SR1_STOPF) {
202  /* write to CR1 to clear STOPF */
203  (void)rSR1; /* as recommended, re-read SR1 */
204  rCR1 |= I2C_CR1_PE;
205  }
206 
207  /* DMA never stops, so we should do that now */
208  switch (direction) {
209  case DIR_TX:
210  i2c_tx_complete();
211  break;
212 
213  case DIR_RX:
214  i2c_rx_complete();
215  break;
216 
217  default:
218  /* not currently transferring - must be a new txn */
219  break;
220  }
221 
222  direction = DIR_NONE;
223  }
224 
225  if (sr1 & I2C_SR1_ADDR) {
226 
227  /* clear ADDR to ack our selection and get direction */
228  (void)rSR1; /* as recommended, re-read SR1 */
229  uint16_t sr2 = rSR2;
230 
231  if (sr2 & I2C_SR2_TRA) {
232  /* we are the transmitter */
233 
234  direction = DIR_TX;
235 
236  } else {
237  /* we are the receiver */
238 
239  direction = DIR_RX;
240  }
241  }
242 
243  /* clear any errors that might need it (this handles AF as well */
244  if (sr1 & I2C_SR1_ERRORMASK) {
245  rSR1 = 0;
246  }
247 
248  return 0;
249 }

251 static void
252 i2c_rx_setup(void)
253 {
254  /*
255  * Note that we configure DMA in circular mode; this means that a too-long
256  * transfer will overwrite the buffer, but that avoids us having to deal with
257  * bailing out of a transaction while the master is still babbling at us.
258  */
259  rx_len = 0;
260  stm32_dmasetup(rx_dma, (uintptr_t)&rDR, (uintptr_t)&rx_buf[0], sizeof(rx_buf),
261  DMA_CCR_CIRC |
262  DMA_CCR_MINC |
263  DMA_CCR_PSIZE_32BITS |
264  DMA_CCR_MSIZE_8BITS |
265  DMA_CCR_PRIMED);
266 
267  stm32_dmastart(rx_dma, NULL, NULL, false);
268 }

270 static void
271 i2c_rx_complete(void)
272 {
273  rx_len = sizeof(rx_buf) - stm32_dmaresidual(rx_dma);
274  stm32_dmastop(rx_dma);
275 
276  if (rx_len >= 2) {
277  selected_page = rx_buf[0];
278  selected_offset = rx_buf[1];
279 
280  /* work out how many registers are being written */
281  unsigned count = (rx_len - 2) / 2;
282 
283  if (count > 0) {
284  registers_set(selected_page, selected_offset, (const uint16_t *)&rx_buf[2], count);
285 
286  } else {
287  /* no registers written, must be an address cycle */
288  uint16_t *regs;
289  unsigned reg_count;
290 
291  /* work out which registers are being addressed */
292  int ret = registers_get(selected_page, selected_offset, &regs, &reg_count);
293 
294  if (ret == 0) {
295  tx_buf = (uint8_t *)regs;
296  tx_len = reg_count * 2;
297 
298  } else {
299  tx_buf = junk_buf;
300  tx_len = sizeof(junk_buf);
301  }
302 
303  /* disable interrupts while reconfiguring DMA for the selected registers */
304  irqstate_t flags = px4_enter_critical_section();
305 
306  stm32_dmastop(tx_dma);
307  i2c_tx_setup();
308 
309  px4_leave_critical_section(flags);
310  }
311  }
312 
313  /* prepare for the next transaction */
314  i2c_rx_setup();
315 }

317 static void
318 i2c_tx_setup(void)
319 {
320  /*
321  * Note that we configure DMA in circular mode; this means that a too-long
322  * transfer will copy the buffer more than once, but that avoids us having
323  * to deal with bailing out of a transaction while the master is still
324  * babbling at us.
325  */
326  stm32_dmasetup(tx_dma, (uintptr_t)&rDR, (uintptr_t)&tx_buf[0], tx_len,
327  DMA_CCR_DIR |
328  DMA_CCR_CIRC |
329  DMA_CCR_MINC |
330  DMA_CCR_PSIZE_8BITS |
331  DMA_CCR_MSIZE_8BITS |
332  DMA_CCR_PRIMED);
333 
334  stm32_dmastart(tx_dma, NULL, NULL, false);
335 }
