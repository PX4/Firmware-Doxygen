108 /**
109  * @brief Starts the driver.
110  */
111 void start(enum LL40LS_BUS busid, uint8_t rotation)
112 {
113  int fd, ret;
114 
115  if (instance) {
116  PX4_INFO("driver already started");
117  }
118 
119  if (busid == LL40LS_BUS_PWM) {
120  instance = new LidarLitePWM(LL40LS_DEVICE_PATH_PWM, rotation);
121 
122  if (!instance) {
123  PX4_ERR("Failed to instantiate LidarLitePWM");
124  return;
125  }
126 
127  if (instance->init() != PX4_OK) {
128  PX4_ERR("failed to initialize LidarLitePWM");
129  stop();
130  return;
131  }
132 
133  } else {
134  for (uint8_t i = 0; i < (sizeof(bus_options) / sizeof(bus_options[0])); i++) {
135  if (busid != LL40LS_BUS_I2C_ALL && busid != bus_options[i].busid) {
136  continue;
137  }
138 
139  instance = new LidarLiteI2C(bus_options[i].busnum, bus_options[i].devname, rotation);
140 
141  if (!instance) {
142  PX4_ERR("Failed to instantiate LidarLiteI2C");
143  return;
144  }
145 
146  if (instance->init() == PX4_OK) {
147  break;
148  }
149 
150  PX4_ERR("failed to initialize LidarLiteI2C on busnum=%u", bus_options[i].busnum);
151  stop();
152  return;
153  }
154  }
155 
156  if (!instance) {
157  PX4_WARN("No LidarLite found");
158  return;
159  }
160 
161  fd = px4_open(instance->get_dev_name(), O_RDONLY);
162 
163  if (fd == -1) {
164  PX4_ERR("Error opening fd");
165  stop();
166  return;
167  }
168 
169  ret = px4_ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT);
170  px4_close(fd);
171 
172  if (ret < 0) {
173  PX4_ERR("pollrate fail");
174  stop();
175  return;
176  }
177 }

188 /**
189  * @brief Performs some basic functional tests on the driver;
190  * make sure we can collect data from the sensor in polled
191  * and automatic modes.
192  */
193 void
194 test()
195 {
196  struct distance_sensor_s report;
197  ssize_t sz;
198  int ret;
199 
200  if (!instance) {
201  PX4_ERR("No ll40ls driver running");
202  return;
203  }
204 
205  int fd = px4_open(instance->get_dev_name(), O_RDONLY);
206 
207  if (fd < 0) {
208  PX4_ERR("Error opening fd");
209  return;
210  }
211 
212  /* Do a simple demand read. */
213  sz = px4_read(fd, &report, sizeof(report));
214 
215  if (sz != sizeof(report)) {
216  PX4_ERR("immediate read failed");
217  return;
218  }
219 
220  print_message(report);
221 
222  /* Start the sensor polling at 2Hz. */
223  if (PX4_OK != px4_ioctl(fd, SENSORIOCSPOLLRATE, 2)) {
224  PX4_ERR("failed to set 2Hz poll rate");
225  return;
226  }
227 
228  /* Read the sensor 5 times and report each value. */
229  for (unsigned i = 0; i < 5; i++) {
230  px4_pollfd_struct_t fds;
231 
232  /* Wait for data to be ready. */
233  fds.fd = fd;
234  fds.events = POLLIN;
235  ret = px4_poll(&fds, 1, 2000);
236 
237  if (ret != 1) {
238  PX4_WARN("timed out waiting for sensor data");
239  return;
240  }
241 
242  /* Now go get it. */
243  sz = px4_read(fd, &report, sizeof(report));
244 
245  if (sz != sizeof(report)) {
246  PX4_WARN("periodic read failed");
247  return;
248  }
249 
250  print_message(report);
251  }
252 
253  /* Reset the sensor polling to default rate. */
254  if (PX4_OK != px4_ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT)) {
255  PX4_WARN("failed to set default poll rate");
256  }
257 
258  px4_close(fd);
259 }

265 reset()
266 {
267  if (!instance) {
268  PX4_WARN("No ll40ls driver running");
269  return;
270  }
271 
272  int fd = px4_open(instance->get_dev_name(), O_RDONLY);
273 
274  if (fd < 0) {
275  PX4_ERR("Error opening fd");
276  return;
277  }
278 
279  if (px4_ioctl(fd, SENSORIOCRESET, 0) < 0) {
280  PX4_ERR("driver reset failed");
281  px4_close(fd);
282  return;
283  }
284 
285  if (px4_ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
286  PX4_ERR("driver poll restart failed");
287  px4_close(fd);
288  return;
289  }

339 int
340 ll40ls_main(int argc, char *argv[])
341 {
342  int ch;
343  int myoptind = 1;
344  const char *myoptarg = nullptr;
345  enum LL40LS_BUS busid = LL40LS_BUS_I2C_ALL;
346  uint8_t rotation = distance_sensor_s::ROTATION_DOWNWARD_FACING;
347 
348  while ((ch = px4_getopt(argc, argv, "IXR:", &myoptind, &myoptarg)) != EOF) {
349  switch (ch) {
350 #ifdef PX4_I2C_BUS_ONBOARD
351 
352  case 'I':
353  busid = LL40LS_BUS_I2C_INTERNAL;
354  break;
355 #endif
356 
357  case 'X':
358  busid = LL40LS_BUS_I2C_EXTERNAL;
359  break;
360 
361  case 'R':
362  rotation = (uint8_t)atoi(myoptarg);
363  PX4_INFO("Setting Lidar orientation to %d", (int)rotation);
364  break;
365 
366  default:
367  ll40ls::usage();
368  return 0;
369  }
370  }
371 
372  /* Determine protocol first because it's needed next. */
373  if (argc > myoptind + 1) {
374  const char *protocol = argv[myoptind + 1];
375 
376  if (!strcmp(protocol, "pwm")) {
377  busid = LL40LS_BUS_PWM;;
378 
379  } else if (!strcmp(protocol, "i2c")) {
380  // Do nothing
381 
382  } else {
383  warnx("unknown protocol, choose pwm or i2c");
384  ll40ls::usage();
385  return 0;
386  }
387  }
388 
389  /* Now determine action. */
390  if (argc > myoptind) {
391  const char *verb = argv[myoptind];
392 
393  if (!strcmp(verb, "start")) {
394  ll40ls::start(busid, rotation);
395 
396  } else if (!strcmp(verb, "stop")) {
397  ll40ls::stop();
398 
399  } else if (!strcmp(verb, "test")) {
400  ll40ls::test();
401 
402  } else if (!strcmp(verb, "reset")) {
403  ll40ls::reset();
404 
405  } else if (!strcmp(verb, "regdump")) {
406  ll40ls::regdump();
407 
408  } else if (!strcmp(verb, "info") || !strcmp(verb, "status")) {
409  ll40ls::info();
410 
411  } else {
412  ll40ls::usage();
413  }
414 
415  return 0;
416  }
417 
418  warnx("unrecognized command, try 'start', 'test', 'reset', 'info' or 'regdump'");
419  ll40ls::usage();
420  return 0;
421 }
