165 static int io_timer_handler(uint16_t timer_index)
166 {
167  /* Read the count at the time of the interrupt */
168 
169  uint16_t count = rCNT(timer_index);
170 
171  /* Read the HRT at the time of the interrupt */
172 
173  hrt_abstime now = hrt_absolute_time();
174 
175  const io_timers_t *tmr = &io_timers[timer_index];
176 
177  /* What is pending */
178 
179  uint32_t statusr = rSTATUS(timer_index);
180 
181  /* Acknowledge all that are pending */
182 
183  rSTATUS(timer_index) = 0;
184 
185  /* Iterate over the timer_io_channels table */
186 
187  for (unsigned chan_index = tmr->first_channel_index; chan_index <= tmr->last_channel_index; chan_index++) {
188 
189  uint16_t chan = 1 << chan_index;
190 
191  if (statusr & chan) {
192 
193  io_timer_channel_stats[chan_index].isr_cout++;
194 
195  /* Call the client to read the rCnV etc and clear the CHnF */
196 
197  if (channel_handlers[chan_index].callback) {
198  channel_handlers[chan_index].callback(channel_handlers[chan_index].context, tmr,
199  chan_index, &timer_io_channels[chan_index],
200  now, count, _REG32(tmr->base, KINETIS_FTM_CV_OFFSET(chan_index)));
201  }
202  }
203 
204  /* Did it set again during call out ?*/
205 
206  if (rSTATUS(timer_index) & chan) {
207 
208  /* Error we has a second edge before we serviced the fist */
209 
210  io_timer_channel_stats[chan_index].overflows++;
211  }
212  }
213 
214  return 0;
215 }

506 void io_timer_trigger(void)
507 {
508  int oneshots = io_timer_get_mode_channels(IOTimerChanMode_OneShot);
509  uint32_t action_cache[MAX_IO_TIMERS] = {0};
510  int actions = 0;
511 
512  /* Pre-calculate the list of timers to Trigger */
513 
514  for (int timer = 0; timer < MAX_IO_TIMERS; timer++) {
515  if (validate_timer_index(timer) == 0) {
516  int channels = get_timer_channels(timer);
517 
518  if (oneshots & channels) {
519  action_cache[actions++] = io_timers[timer].base;
520  }
521  }
522  }
523 
524  /* Now do them all wit the shortest delay in between */
525 
526  irqstate_t flags = px4_enter_critical_section();
527 
528  for (actions = 0; action_cache[actions] != 0 && actions < MAX_IO_TIMERS; actions++) {
529  _REG32(action_cache[actions], KINETIS_FTM_SYNC_OFFSET) |= FTM_SYNC;
530  }
531 
532  px4_leave_critical_section(flags);
533 }

593 {

595

597 
598  uint32_t channels = get_timer_channels(timer);

601

603  channel_allocations[IOTimerChanMode_OneShot] |
604  channel_allocations[IOTimerChanMode_NotUsed])) ==
605  channels) {
606 
607  /* Change only a timer that is owned by pwm or one shot */
608 
609  /* Request to use OneShot ?*/
610 
611  if (rate == 0) {
612 
613  /* Request to use OneShot
614  *
615  * We are here because ALL these channels were either PWM or Oneshot
616  * Now they need to be Oneshot
617  */
618 
619  /* Did the allocation change */
620  if (reallocate_channel_resources(channels, IOTimerChanMode_PWMOut, IOTimerChanMode_OneShot)) {
621  io_timer_set_oneshot_mode(timer);
622  }
623 
624  } else {
625 
626  /* Request to use PWM
627  *
628  * We are here because ALL these channels were either PWM or Oneshot
629  * Now they need to be PWM
630  */
631

633  io_timer_set_PWM_mode(timer);
634  }
635 
636  timer_set_rate(timer, rate);
637  }
638 
639  rv = OK;
640  }
641

715 int io_timer_set_enable(bool state, io_timer_channel_mode_t mode, io_timer_channel_allocation_t masks)
716 {
717  typedef struct action_cache_rp_t {
718  uint32_t cnsc_offset;
719  uint32_t cnsc_value;
720  uint32_t gpio;
721  } action_cache_rp_t;
722  struct action_cache_t {
723  uint32_t base;
724  uint32_t index;
725  action_cache_rp_t cnsc[MAX_CHANNELS_PER_TIMER];
726  } action_cache[MAX_IO_TIMERS];
727 
728  memset(action_cache, 0, sizeof(action_cache));
729 
730  uint32_t bits = state ? CnSC_PWMOUT_INIT : 0;
731 
732  switch (mode) {
733  case IOTimerChanMode_NotUsed:
734  case IOTimerChanMode_PWMOut:
735  case IOTimerChanMode_OneShot:
736  case IOTimerChanMode_Trigger:
737  break;
738 
739  case IOTimerChanMode_PWMIn:
740  case IOTimerChanMode_Capture:
741  if (state) {
742  bits |= FTM_CSC_CHIE;
743  }
744 
745  break;
746 
747  default:
748  return -EINVAL;
749  }
750 
751  /* Was the request for all channels in this mode ?*/
752 
753  if (masks == IO_TIMER_ALL_MODES_CHANNELS) {
754 
755  /* Yes - we provide them */
756 
757  masks = channel_allocations[mode];
758 
759  } else {
760 
761  /* No - caller provided mask */
762 
763  /* Only allow the channels in that mode to be affected */
764 
765  masks &= channel_allocations[mode];
766 
767  }
768 
769  /* Pre calculate all the changes */
770 
771  for (int chan_index = 0; masks != 0 && chan_index < MAX_TIMER_IO_CHANNELS; chan_index++) {
772  if (masks & (1 << chan_index)) {
773  masks &= ~(1 << chan_index);
774 
775  if (io_timer_validate_channel_index(chan_index) == 0) {
776  uint32_t chan = timer_io_channels[chan_index].timer_channel - 1;
777  uint32_t timer = channels_timer(chan_index);
778  action_cache[timer].base = io_timers[timer].base;
779  action_cache[timer].cnsc[action_cache[timer].index].cnsc_offset = io_timers[timer].base + KINETIS_FTM_CSC_OFFSET(chan);
780  action_cache[timer].cnsc[action_cache[timer].index].cnsc_value = bits;
781 
782  if ((state &&
783  (mode == IOTimerChanMode_PWMOut ||
784  mode == IOTimerChanMode_OneShot ||
785  mode == IOTimerChanMode_Trigger))) {
786  action_cache[timer].cnsc[action_cache[timer].index].gpio = timer_io_channels[chan_index].gpio_out;
787  }
788 
789  action_cache[timer].index++;
790  }
791  }
792  }
793 
794  irqstate_t flags = px4_enter_critical_section();
795 
796 
797  for (unsigned actions = 0; actions < arraySize(action_cache); actions++) {
798  uint32_t any_on = 0;
799 
800  if (action_cache[actions].base != 0) {
801  for (unsigned int index = 0; index < action_cache[actions].index; index++) {
802 
803  if (action_cache[actions].cnsc[index].gpio) {
804  px4_arch_configgpio(action_cache[actions].cnsc[index].gpio);
805  }
806 
807  _REG(action_cache[actions].cnsc[index].cnsc_offset) = action_cache[actions].cnsc[index].cnsc_value;
808  any_on |= action_cache[actions].cnsc[index].cnsc_value;
809  }
810 
811  /* Any On ?*/
812 
813  /* Assume not */
814  uint32_t regval = _REG32(action_cache[actions].base, KINETIS_FTM_SC_OFFSET);
815  regval &= ~(FTM_SC_CLKS_MASK);
816 
817  if (any_on != 0) {
818 
819  /* force an update to preload all registers */
820  _REG32(action_cache[actions].base, KINETIS_FTM_SYNC_OFFSET) |= FTM_SYNC;
821 
822  /* arm requires the timer be enabled */
823  regval |= (FTM_SC_CLKS_EXTCLK);
824  }
825 
826  _REG32(action_cache[actions].base, KINETIS_FTM_SC_OFFSET) = regval;
827  }
828  }
829 
830  px4_leave_critical_section(flags);
831 
832  return 0;
833 }
