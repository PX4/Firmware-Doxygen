74 void px4_backtrace()

77  void *buffer[10];

79  int bt_size;
80  int idx;
81 
82  bt_size = backtrace(buffer, 10);
83  callstack = backtrace_symbols(buffer, bt_size);
84 
85  PX4_INFO("Backtrace: %d", bt_size);
86 
87  for (idx = 0; idx < bt_size; idx++) {
88  PX4_INFO("%s", callstack[idx]);
89  }
90 
91  free(callstack);
