418 int DfMPU6050Wrapper::_publish(struct imu_sensor_data &data)
419 {
420  /* Check if calibration values are still up-to-date. */
421  bool updated;
422  orb_check(_param_update_sub, &updated);
423 
424  if (updated) {
425  parameter_update_s parameter_update;
426  orb_copy(ORB_ID(parameter_update), _param_update_sub, &parameter_update);
427 
428  _update_accel_calibration();
429  _update_gyro_calibration();
430  }
431 
432  uint64_t now = hrt_absolute_time();
433 
434  matrix::Vector3f vec_integrated_unused;
435  uint32_t integral_dt_unused;
436 
437  matrix::Vector3f accel_val(data.accel_m_s2_x, data.accel_m_s2_y, data.accel_m_s2_z);
438 
439  // apply sensor rotation on the accel measurement
440  accel_val = _rotation_matrix * accel_val;
441 
442  accel_val(0) = (accel_val(0) - _accel_calibration.x_offset) * _accel_calibration.x_scale;
443  accel_val(1) = (accel_val(1) - _accel_calibration.y_offset) * _accel_calibration.y_scale;
444  accel_val(2) = (accel_val(2) - _accel_calibration.z_offset) * _accel_calibration.z_scale;
445 
446  _accel_int.put(now,
447  accel_val,
448  vec_integrated_unused,
449  integral_dt_unused);
450 
451  matrix::Vector3f gyro_val(data.gyro_rad_s_x, data.gyro_rad_s_y, data.gyro_rad_s_z);
452 
453  // apply sensor rotation on the gyro measurement
454  gyro_val = _rotation_matrix * gyro_val;
455 
456  gyro_val(0) = (gyro_val(0) - _gyro_calibration.x_offset) * _gyro_calibration.x_scale;
457  gyro_val(1) = (gyro_val(1) - _gyro_calibration.y_offset) * _gyro_calibration.y_scale;
458  gyro_val(2) = (gyro_val(2) - _gyro_calibration.z_offset) * _gyro_calibration.z_scale;
459 
460  _gyro_int.put(now,
461  gyro_val,
462  vec_integrated_unused,
463  integral_dt_unused);
464 
465  // If we are not receiving the last sample from the FIFO buffer yet, let's stop here
466  // and wait for more packets.
467  if (!data.is_last_fifo_sample) {
468  return 0;
469  }
470 
471  // The driver empties the FIFO buffer at 1kHz, however we only need to publish at 250Hz.
472  // Therefore, only publish every forth time.
473  ++_publish_count;
474 
475  if (_publish_count < 4) {
476  return 0;
477  }
478 
479  _publish_count = 0;
480 
481  // Update all the counters.
482  perf_set_count(_read_counter, data.read_counter);
483  perf_set_count(_error_counter, data.error_counter);
484  perf_set_count(_fifo_overflow_counter, data.fifo_overflow_counter);
485  perf_set_count(_fifo_corruption_counter, data.fifo_overflow_counter);
486  perf_set_count(_gyro_range_hit_counter, data.gyro_range_hit_counter);
487  perf_set_count(_accel_range_hit_counter, data.accel_range_hit_counter);
488 
489  perf_begin(_publish_perf);
490 
491  accel_report accel_report = {};
492  gyro_report gyro_report = {};
493 
494  accel_report.timestamp = gyro_report.timestamp = hrt_absolute_time();
495 
496  // TODO: get these right
497  gyro_report.scaling = -1.0f;
498  gyro_report.device_id = m_id.dev_id;
499 
500  accel_report.scaling = -1.0f;
501  accel_report.device_id = m_id.dev_id;
502 
503  // TODO: remove these (or get the values)
504  gyro_report.x_raw = 0;
505  gyro_report.y_raw = 0;
506  gyro_report.z_raw = 0;
507 
508  accel_report.x_raw = 0;
509  accel_report.y_raw = 0;
510  accel_report.z_raw = 0;
511 
512  matrix::Vector3f gyro_val_filt;
513  matrix::Vector3f accel_val_filt;
514 
515  // Read and reset.
516  matrix::Vector3f gyro_val_integ = _gyro_int.get_and_filtered(true, gyro_report.integral_dt, gyro_val_filt);
517  matrix::Vector3f accel_val_integ = _accel_int.get_and_filtered(true, accel_report.integral_dt, accel_val_filt);
518 
519  // Use the filtered (by integration) values to get smoother / less noisy data.
520  gyro_report.x = gyro_val_filt(0);
521  gyro_report.y = gyro_val_filt(1);
522  gyro_report.z = gyro_val_filt(2);
523 
524  accel_report.x = accel_val_filt(0);
525  accel_report.y = accel_val_filt(1);
526  accel_report.z = accel_val_filt(2);
527 
528  gyro_report.x_integral = gyro_val_integ(0);
529  gyro_report.y_integral = gyro_val_integ(1);
530  gyro_report.z_integral = gyro_val_integ(2);
531 
532  accel_report.x_integral = accel_val_integ(0);
533  accel_report.y_integral = accel_val_integ(1);
534  accel_report.z_integral = accel_val_integ(2);
535 
536  // TODO: when is this ever blocked?
537  if (!(m_pub_blocked)) {
538 
539  if (_gyro_topic == nullptr) {
540  _gyro_topic = orb_advertise_multi(ORB_ID(sensor_gyro), &gyro_report,
541  &_gyro_orb_class_instance, ORB_PRIO_DEFAULT);
542 
543  } else {
544  orb_publish(ORB_ID(sensor_gyro), _gyro_topic, &gyro_report);
545  }
546 
547  if (_accel_topic == nullptr) {
548  _accel_topic = orb_advertise_multi(ORB_ID(sensor_accel), &accel_report,
549  &_accel_orb_class_instance, ORB_PRIO_DEFAULT);
550 
551  } else {
552  orb_publish(ORB_ID(sensor_accel), _accel_topic, &accel_report);
553  }
554 
555  // Report if there are high vibrations, every 10 times it happens.
556  const bool threshold_reached = (data.accel_range_hit_counter - _last_accel_range_hit_count > 10);
557 
558  // Report every 5s.
559  const bool due_to_report = (hrt_elapsed_time(&_last_accel_range_hit_time) > 5000000);
560 
561  if (threshold_reached && due_to_report) {
562  mavlink_log_critical(&_mavlink_log_pub,
563  "High accelerations, range exceeded %llu times",
564  data.accel_range_hit_counter);
565 
566  _last_accel_range_hit_time = hrt_absolute_time();
567  _last_accel_range_hit_count = data.accel_range_hit_counter;
568  }
569  }
570 
571  perf_end(_publish_perf);
572 
573  // TODO: check the return codes of this function
574  return 0;

591 
592  if (g_dev == nullptr) {
593  PX4_ERR("failed instantiating DfMPU6050Wrapper object");
594  return -1;
595  }
596 
597  int ret = g_dev->start();
598 
599  if (ret != 0) {
600  PX4_ERR("DfMPU6050Wrapper start failed");
601  return ret;
602  }
603 
604  // Open the IMU sensor
605  DevHandle h;
606  DevMgr::getHandle(IMU_DEVICE_PATH, h);
607 
608  if (!h.isValid()) {
609  DF_LOG_INFO("Error: unable to obtain a valid handle for the receiver at: %s (%d)",
610  IMU_DEVICE_PATH, h.getError());
611  return -1;
612  }
613 
614  DevMgr::releaseHandle(h);
615 
616  return 0;
617 }
