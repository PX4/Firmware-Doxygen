80 LaunchDetectionResult LaunchDetector::getLaunchDetected()
81 {
82  if (launchDetectionEnabled()) {
83  if (_activeLaunchDetectionMethodIndex < 0) {
84  /* None of the active _launchmethods has detected a launch, check all _launchmethods */
85  for (unsigned i = 0; i < (sizeof(_launchMethods) / sizeof(_launchMethods[0])); i++) {
86  if (_launchMethods[i]->getLaunchDetected() != LAUNCHDETECTION_RES_NONE) {
87  PX4_INFO("selecting launchmethod %d", i);
88  _activeLaunchDetectionMethodIndex = i; // from now on only check this method
89  return _launchMethods[i]->getLaunchDetected();
90  }
91  }
92 
93  } else {
94  return _launchMethods[_activeLaunchDetectionMethodIndex]->getLaunchDetected();
95  }
96  }
97 
98  return LAUNCHDETECTION_RES_NONE;
99 }

101 float LaunchDetector::getPitchMax(float pitchMaxDefault)
