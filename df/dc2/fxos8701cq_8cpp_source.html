560

563

565  goto out;

567 
568  _mag_reports = new ringbuffer::RingBuffer(2, sizeof(mag_report));
569 
570  if (_mag_reports == nullptr) {
571  goto out;
572  }
573 
574  reset();
575 
576  /* do CDev init for the mag device node */
577  ret = _mag->init();
578 
579  if (ret != OK) {
580  PX4_ERR("MAG init failed");
581  goto out;
582  }
583 
584  /* fill report structures */
585  measure();
586 
587  /* advertise sensor topic, measure manually to initialize valid report */
588  struct mag_report mrp;
589  _mag_reports->get(&mrp);
590 
591  /* measurement will have generated a report, publish */
592  _mag->_mag_topic = orb_advertise_multi(ORB_ID(sensor_mag), &mrp,
593  &_mag->_mag_orb_class_instance, ORB_PRIO_LOW);
594 
595  if (_mag->_mag_topic == nullptr) {
596  PX4_ERR("ADVERT ERR");
597  }
598

612  }

619 void
620 FXOS8701CQ::reset()
621 {
622 
623  /* enable accel set it To Standby */
624 
625  write_checked_reg(FXOS8701CQ_CTRL_REG1, 0);
626  write_checked_reg(FXOS8701CQ_XYZ_DATA_CFG, 0);
627 
628  /* Use hybird mode to read Accel and Mag */
629 
630  write_checked_reg(FXOS8701CQ_M_CTRL_REG1, M_CTRL_REG1_HMS_AM | M_CTRL_REG1_OS(7));
631 
632  /* Use the hybird auto increment mode to read all the data at the same time */
633 
634  write_checked_reg(FXOS8701CQ_M_CTRL_REG2, CTRL_REG2_AUTO_INC);
635 
636  accel_set_range(FXOS8701C_ACCEL_DEFAULT_RANGE_G);
637  accel_set_samplerate(FXOS8701C_ACCEL_DEFAULT_RATE);
638  accel_set_driver_lowpass_filter((float)FXOS8701C_ACCEL_DEFAULT_RATE, (float)FXOS8701C_ACCEL_DEFAULT_DRIVER_FILTER_FREQ);

1255 void
1256 FXOS8701CQ::measure()
1257 {
1258  /* status register and data as read back from the device */
1259 
1260 #pragma pack(push, 1)
1261  struct {
1262  uint8_t cmd[2];
1263  uint8_t status;
1264  int16_t x;
1265  int16_t y;
1266  int16_t z;
1267  int16_t mx;
1268  int16_t my;
1269  int16_t mz;
1270  } raw_accel_mag_report;
1271 #pragma pack(pop)
1272 
1273  accel_report accel_report;
1274 
1275  /* start the performance counter */
1276  perf_begin(_accel_sample_perf);
1277 
1278  check_registers();
1279 
1280  if (_register_wait != 0) {
1281  // we are waiting for some good transfers before using
1282  // the sensor again.
1283  _register_wait--;
1284  perf_end(_accel_sample_perf);
1285  return;
1286  }
1287 
1288  /* fetch data from the sensor */
1289  memset(&raw_accel_mag_report, 0, sizeof(raw_accel_mag_report));
1290  raw_accel_mag_report.cmd[0] = DIR_READ(FXOS8701CQ_DR_STATUS);
1291  raw_accel_mag_report.cmd[1] = ADDR_7(FXOS8701CQ_DR_STATUS);
1292  transfer((uint8_t *)&raw_accel_mag_report, (uint8_t *)&raw_accel_mag_report, sizeof(raw_accel_mag_report));
1293 
1294  if (!(raw_accel_mag_report.status & DR_STATUS_ZYXDR)) {
1295  perf_end(_accel_sample_perf);
1296  perf_count(_accel_duplicates);
1297  return;
1298  }
1299 
1300  /*
1301  * 1) Scale raw value to SI units using scaling from datasheet.
1302  * 2) Subtract static offset (in SI units)
1303  * 3) Scale the statically calibrated values with a linear
1304  * dynamically obtained factor
1305  *
1306  * Note: the static sensor offset is the number the sensor outputs
1307  * at a nominally 'zero' input. Therefore the offset has to
1308  * be subtracted.
1309  *
1310  * Example: A gyro outputs a value of 74 at zero angular rate
1311  * the offset is 74 from the origin and subtracting
1312  * 74 from all measurements centers them around zero.
1313  */
1314 
1315  accel_report.timestamp = hrt_absolute_time();
1316 
1317  /*
1318  * Eight-bit 2s complement sensor temperature value with 0.96 C/LSB sensitivity.
1319  * Temperature data is only valid between 40 C and 125 C. The temperature sensor
1320  * output is only valid when M_CTRL_REG1[m_hms] > 0b00. Please note that the
1321  * temperature sensor is uncalibrated and its output for a given temperature will vary from
1322  * one device to the next
1323  */
1324 
1325  _last_temperature = (read_reg(FXOS8701CQ_TEMP)) * 0.96f;
1326 
1327  accel_report.temperature = _last_temperature;
1328 
1329  // report the error count as the sum of the number of bad
1330  // register reads and bad values. This allows the higher level
1331  // code to decide if it should use this sensor based on
1332  // whether it has had failures
1333  accel_report.error_count = perf_event_count(_bad_registers) + perf_event_count(_bad_values);
1334 
1335  accel_report.x_raw = swap16RightJustify14(raw_accel_mag_report.x);
1336  accel_report.y_raw = swap16RightJustify14(raw_accel_mag_report.y);
1337  accel_report.z_raw = swap16RightJustify14(raw_accel_mag_report.z);
1338 
1339  /* Save off the Mag readings todo: revist integrating theses */
1340 
1341  _last_raw_mag_x = swap16(raw_accel_mag_report.mx);
1342  _last_raw_mag_y = swap16(raw_accel_mag_report.my);
1343  _last_raw_mag_z = swap16(raw_accel_mag_report.mz);
1344 
1345  float xraw_f = accel_report.x_raw;
1346  float yraw_f = accel_report.y_raw;
1347  float zraw_f = accel_report.z_raw;
1348 
1349  // apply user specified rotation
