95 transition_result_t arming_state_transition(vehicle_status_s *status, const battery_status_s &battery,
96  const safety_s &safety, const arming_state_t new_arming_state, actuator_armed_s *armed, const bool fRunPreArmChecks,
97  orb_advert_t *mavlink_log_pub, vehicle_status_flags_s *status_flags, const uint8_t arm_requirements,
98  const hrt_abstime &time_since_boot)
99 {
100  // Double check that our static arrays are still valid
101  static_assert(vehicle_status_s::ARMING_STATE_INIT == 0, "ARMING_STATE_INIT == 0");
102  static_assert(vehicle_status_s::ARMING_STATE_IN_AIR_RESTORE == vehicle_status_s::ARMING_STATE_MAX - 1,
103  "ARMING_STATE_IN_AIR_RESTORE == ARMING_STATE_MAX - 1");
104 
105  transition_result_t ret = TRANSITION_DENIED;
106  arming_state_t current_arming_state = status->arming_state;
107  bool feedback_provided = false;
108 
109  const bool hil_enabled = (status->hil_state == vehicle_status_s::HIL_STATE_ON);
110 
111  /* only check transition if the new state is actually different from the current one */
112  if (new_arming_state == current_arming_state) {
113  ret = TRANSITION_NOT_CHANGED;
114 
115  } else {
116 
117  /*
118  * Get sensing state if necessary
119  */
120  bool preflight_check_ret = true;
121  bool prearm_check_ret = true;
122 
123  const bool checkGNSS = (arm_requirements & ARM_REQ_GPS_BIT);
124 
125  /* only perform the pre-arm check if we have to */
126  if (fRunPreArmChecks && (new_arming_state == vehicle_status_s::ARMING_STATE_ARMED)
127  && !hil_enabled) {
128 
129  preflight_check_ret = Preflight::preflightCheck(mavlink_log_pub, *status, *status_flags, checkGNSS, true, true, time_since_boot);
130 
131  if (preflight_check_ret) {
132  status_flags->condition_system_sensors_initialized = true;
133  }
134 
135  feedback_provided = true;
136  }
137 
138  /* re-run the pre-flight check as long as sensors are failing */
139  if (!status_flags->condition_system_sensors_initialized
140  && fRunPreArmChecks
141  && ((new_arming_state == vehicle_status_s::ARMING_STATE_ARMED) || (new_arming_state == vehicle_status_s::ARMING_STATE_STANDBY))
142  && !hil_enabled) {
143 
144  if ((last_preflight_check == 0) || (hrt_elapsed_time(&last_preflight_check) > 1000 * 1000)) {
145 
146  status_flags->condition_system_sensors_initialized = Preflight::preflightCheck(mavlink_log_pub, *status, *status_flags, checkGNSS, false, false, time_since_boot);
147 
148  last_preflight_check = hrt_absolute_time();
149  }
150  }
151 
152  // Check that we have a valid state transition
153  bool valid_transition = arming_transitions[new_arming_state][status->arming_state];
154 
155  if (valid_transition) {
156  // We have a good transition. Now perform any secondary validation.
157  if (new_arming_state == vehicle_status_s::ARMING_STATE_ARMED) {
158 
159  // Do not perform pre-arm checks if coming from in air restore
160  // Allow if vehicle_status_s::HIL_STATE_ON
161  if (status->arming_state != vehicle_status_s::ARMING_STATE_IN_AIR_RESTORE &&
162  !hil_enabled) {
163 
164  if (preflight_check_ret) {
165  // only bother running prearm if preflight was successful
166  prearm_check_ret = prearm_check(mavlink_log_pub, *status_flags, battery, safety, arm_requirements, time_since_boot);
167  }
168 
169  if (!(preflight_check_ret && prearm_check_ret)) {
170  // the prearm and preflight checks already print the rejection reason
171  feedback_provided = true;
172  valid_transition = false;
173  }
174  }
175  }
176  }
177 
178  if (hil_enabled) {
179  /* enforce lockdown in HIL */
180  armed->lockdown = true;
181  status_flags->condition_system_sensors_initialized = true;
182 
183  /* recover from a prearm fail */
184  if (status->arming_state == vehicle_status_s::ARMING_STATE_STANDBY_ERROR) {
185  status->arming_state = vehicle_status_s::ARMING_STATE_STANDBY;
186  }
187 
188  // HIL can always go to standby
189  if (new_arming_state == vehicle_status_s::ARMING_STATE_STANDBY) {
190  valid_transition = true;
191  }
192  }
193 
194  if (!hil_enabled &&
195  (new_arming_state == vehicle_status_s::ARMING_STATE_STANDBY) &&
196  (status->arming_state != vehicle_status_s::ARMING_STATE_STANDBY_ERROR)) {
197 
198  // Sensors need to be initialized for STANDBY state, except for HIL
199  if (!status_flags->condition_system_sensors_initialized) {
200  feedback_provided = true;
201  valid_transition = false;
202  }
203  }
204 
205  // Finish up the state transition
206  if (valid_transition) {
207  armed->armed = (new_arming_state == vehicle_status_s::ARMING_STATE_ARMED);
208  armed->ready_to_arm = (new_arming_state == vehicle_status_s::ARMING_STATE_ARMED) || (new_arming_state == vehicle_status_s::ARMING_STATE_STANDBY);
209  ret = TRANSITION_CHANGED;
210  status->arming_state = new_arming_state;
211 
212  if (new_arming_state == vehicle_status_s::ARMING_STATE_ARMED) {
213  armed->armed_time_ms = hrt_absolute_time() / 1000;
214 
215  } else {
216  armed->armed_time_ms = 0;
217  }
218  }
219  }
220 
221  if (ret == TRANSITION_DENIED) {
222  /* print to MAVLink and console if we didn't provide any feedback yet */
223  if (!feedback_provided) {
224  mavlink_log_critical(mavlink_log_pub, "TRANSITION_DENIED: %s to %s",
225  arming_state_names[status->arming_state], arming_state_names[new_arming_state]);
226  }
227  }
228 
229  return ret;
230 }

232 bool is_safe(const safety_s &safety, const actuator_armed_s &armed)
233 {
234  // System is safe if:
235  // 1) Not armed
236  // 2) Armed, but in software lockdown (HIL)
237  // 3) Safety switch is present AND engaged -> actuators locked
238  const bool lockdown = (armed.lockdown || armed.manual_lockdown);
239 
240  return !armed.armed || (armed.armed && lockdown) || (safety.safety_switch_available && !safety.safety_off);
241 }

243 transition_result_t
244 main_state_transition(const vehicle_status_s &status, const main_state_t new_main_state,
245  const vehicle_status_flags_s &status_flags, commander_state_s *internal_state)
246 {
247  // IMPORTANT: The assumption of callers of this function is that the execution of
248  // this check if essentially "free". Therefore any runtime checking in here has to be
249  // kept super lightweight. No complex logic or calls on external function should be
250  // implemented here.
251 
252  transition_result_t ret = TRANSITION_DENIED;
253 
254  /* transition may be denied even if the same state is requested because conditions may have changed */
255  switch (new_main_state) {
256  case commander_state_s::MAIN_STATE_MANUAL:
257  case commander_state_s::MAIN_STATE_STAB:
258  case commander_state_s::MAIN_STATE_ACRO:
259  case commander_state_s::MAIN_STATE_RATTITUDE:
260  ret = TRANSITION_CHANGED;
261  break;
262 
263  case commander_state_s::MAIN_STATE_ALTCTL:
264 
265  /* need at minimum altitude estimate */
266  if (status_flags.condition_local_altitude_valid ||
267  status_flags.condition_global_position_valid) {
268  ret = TRANSITION_CHANGED;
269  }
270 
271  break;
272 
273  case commander_state_s::MAIN_STATE_POSCTL:
274 
275  /* need at minimum local position estimate */
276  if (status_flags.condition_local_position_valid ||
277  status_flags.condition_global_position_valid) {
278  ret = TRANSITION_CHANGED;
279  }
280 
281  break;
282 
283  case commander_state_s::MAIN_STATE_AUTO_LOITER:
284 
285  /* need global position estimate */
286  if (status_flags.condition_global_position_valid) {
287  ret = TRANSITION_CHANGED;
288  }
289 
290  break;
291 
292  case commander_state_s::MAIN_STATE_AUTO_FOLLOW_TARGET:
293 
294  /* FOLLOW only implemented in MC */
295  if (status.is_rotary_wing) {
296  ret = TRANSITION_CHANGED;
297  }
298 
299  break;
300 
301  case commander_state_s::MAIN_STATE_AUTO_MISSION:
302 
303  /* need global position, home position, and a valid mission */
304  if (status_flags.condition_global_position_valid &&
305  status_flags.condition_auto_mission_available) {
306 
307  ret = TRANSITION_CHANGED;
308  }
309 
310  break;
311 
312  case commander_state_s::MAIN_STATE_AUTO_RTL:
313 
314  /* need global position and home position */
315  if (status_flags.condition_global_position_valid && status_flags.condition_home_position_valid) {
316  ret = TRANSITION_CHANGED;
317  }
318 
319  break;
320 
321  case commander_state_s::MAIN_STATE_AUTO_TAKEOFF:
322  case commander_state_s::MAIN_STATE_AUTO_LAND:
323 
324  /* need local position */
325  if (status_flags.condition_local_position_valid) {
326  ret = TRANSITION_CHANGED;
327  }
328 
329  break;
330 
331  case commander_state_s::MAIN_STATE_AUTO_PRECLAND:
332 
333  /* need local and global position, and precision land only implemented for multicopters */
334  if (status_flags.condition_local_position_valid
335  && status_flags.condition_global_position_valid
336  && status.is_rotary_wing) {
337 
338  ret = TRANSITION_CHANGED;
339  }
340 
341  break;
342 
343  case commander_state_s::MAIN_STATE_OFFBOARD:
344 
345  /* need offboard signal */
346  if (!status_flags.offboard_control_signal_lost) {
347 
348  ret = TRANSITION_CHANGED;
349  }
350 
351  break;
352 
353  case commander_state_s::MAIN_STATE_MAX:
354  default:
355  break;
356  }
357 
358  if (ret == TRANSITION_CHANGED) {
359  if (internal_state->main_state != new_main_state) {
360  internal_state->main_state = new_main_state;
361  internal_state->timestamp = hrt_absolute_time();
362 
363  } else {
364  ret = TRANSITION_NOT_CHANGED;
365  }
366  }
367 
368  return ret;
369 }

371 /**
372  * Transition from one hil state to another
373  */
374 transition_result_t hil_state_transition(hil_state_t new_state, orb_advert_t status_pub,
375  vehicle_status_s *current_status, orb_advert_t *mavlink_log_pub)
376 {
377  transition_result_t ret = TRANSITION_DENIED;
378 
379  if (current_status->hil_state == new_state) {
380  ret = TRANSITION_NOT_CHANGED;
381 
382  } else {
383  switch (new_state) {
384  case vehicle_status_s::HIL_STATE_OFF:
385  /* The system is in HITL mode and unexpected things can happen if we disable HITL without rebooting. */
386  mavlink_log_critical(mavlink_log_pub, "Set SYS_HITL to 0 and reboot to disable HITL.");
387  ret = TRANSITION_DENIED;
388  break;
389 
390  case vehicle_status_s::HIL_STATE_ON:
391  if (current_status->arming_state == vehicle_status_s::ARMING_STATE_INIT
392  || current_status->arming_state == vehicle_status_s::ARMING_STATE_STANDBY
393  || current_status->arming_state == vehicle_status_s::ARMING_STATE_STANDBY_ERROR) {
394 
395  ret = TRANSITION_CHANGED;
396 
397  int32_t hitl_on = 0;
398  param_get(param_find("SYS_HITL"), &hitl_on);
399 
400  if (hitl_on) {
401  mavlink_log_info(mavlink_log_pub, "Enabled Hardware-in-the-loop simulation (HITL).");
402 
403  } else {
404  mavlink_log_critical(mavlink_log_pub, "Set SYS_HITL to 1 and reboot to enable HITL.");
405  ret = TRANSITION_DENIED;
406  }
407 
408  } else {
409  mavlink_log_critical(mavlink_log_pub, "Not switching to HITL when armed.");
410  ret = TRANSITION_DENIED;
411  }
412 
413  break;
414 
415  default:
416  PX4_WARN("Unknown HITL state");
417  break;
418  }
419  }
420 
421  if (ret == TRANSITION_CHANGED) {
422  current_status->hil_state = new_state;
423  current_status->timestamp = hrt_absolute_time();
424  orb_publish(ORB_ID(vehicle_status), status_pub, current_status);
425  }
426 
427  return ret;
428 }

443  * Check failsafe and main status and set navigation status for navigator accordingly

445 bool set_nav_state(vehicle_status_s *status, actuator_armed_s *armed, commander_state_s *internal_state,
446  orb_advert_t *mavlink_log_pub, const link_loss_actions_t data_link_loss_act, const bool mission_finished,
447  const bool stay_in_failsafe, const vehicle_status_flags_s &status_flags, bool landed,
448  const link_loss_actions_t rc_loss_act, const int offb_loss_act, const int offb_loss_rc_act,
449  const int posctl_nav_loss_act)
450 {
451  navigation_state_t nav_state_old = status->nav_state;
452 
453  const bool data_link_loss_act_configured = data_link_loss_act > link_loss_actions_t::DISABLED;
454  const bool rc_loss_act_configured = rc_loss_act > link_loss_actions_t::DISABLED;
455  const bool rc_lost = rc_loss_act_configured && (status->rc_signal_lost);
456 
457  bool is_armed = (status->arming_state == vehicle_status_s::ARMING_STATE_ARMED);
458  bool old_failsafe = status->failsafe;
459  status->failsafe = false;
460 
461  // Safe to do reset flags here, as if loss state persists flags will be restored in the code below
462  reset_link_loss_globals(armed, old_failsafe, rc_loss_act);
463  reset_link_loss_globals(armed, old_failsafe, data_link_loss_act);
464 
465  /* evaluate main state to decide in normal (non-failsafe) mode */
466  switch (internal_state->main_state) {
467  case commander_state_s::MAIN_STATE_ACRO:
468  case commander_state_s::MAIN_STATE_MANUAL:
469  case commander_state_s::MAIN_STATE_RATTITUDE:
470  case commander_state_s::MAIN_STATE_STAB:
471  case commander_state_s::MAIN_STATE_ALTCTL:
472 
473  /* require RC for all manual modes */
474  if (rc_lost && is_armed) {
475  enable_failsafe(status, old_failsafe, mavlink_log_pub, reason_no_rc);
476 
477  set_link_loss_nav_state(status, armed, status_flags, internal_state, rc_loss_act,
478  vehicle_status_s::NAVIGATION_STATE_AUTO_RCRECOVER);
479 
480  } else {
481  switch (internal_state->main_state) {
482  case commander_state_s::MAIN_STATE_ACRO:
483  status->nav_state = vehicle_status_s::NAVIGATION_STATE_ACRO;
484  break;
485 
486  case commander_state_s::MAIN_STATE_MANUAL:
487  status->nav_state = vehicle_status_s::NAVIGATION_STATE_MANUAL;
488  break;
489 
490  case commander_state_s::MAIN_STATE_RATTITUDE:
491  status->nav_state = vehicle_status_s::NAVIGATION_STATE_RATTITUDE;
492  break;
493 
494  case commander_state_s::MAIN_STATE_STAB:
495  status->nav_state = vehicle_status_s::NAVIGATION_STATE_STAB;
496  break;
497 
498  case commander_state_s::MAIN_STATE_ALTCTL:
499  status->nav_state = vehicle_status_s::NAVIGATION_STATE_ALTCTL;
500  break;
501 
502  default:
503  status->nav_state = vehicle_status_s::NAVIGATION_STATE_MANUAL;
504  break;
505  }
506  }
507 
508  break;
509 
510  case commander_state_s::MAIN_STATE_POSCTL: {
511 
512  if (rc_lost && is_armed) {
513  enable_failsafe(status, old_failsafe, mavlink_log_pub, reason_no_rc);
514 
515  set_link_loss_nav_state(status, armed, status_flags, internal_state, rc_loss_act,
516  vehicle_status_s::NAVIGATION_STATE_AUTO_RCRECOVER);
517 
518  /* As long as there is RC, we can fallback to ALTCTL, or STAB. */
519  /* A local position estimate is enough for POSCTL for multirotors,
520  * this enables POSCTL using e.g. flow.
521  * For fixedwing, a global position is needed. */
522 
523  } else if (is_armed
524  && check_invalid_pos_nav_state(status, old_failsafe, mavlink_log_pub, status_flags, !(posctl_nav_loss_act == 1),
525  !status->is_rotary_wing)) {
526  // nothing to do - everything done in check_invalid_pos_nav_state
527 
528  } else {
529  status->nav_state = vehicle_status_s::NAVIGATION_STATE_POSCTL;
530  }
531  }
532  break;
533 
534  case commander_state_s::MAIN_STATE_AUTO_MISSION:
535 
536  /* go into failsafe
537  * - if commanded to do so
538  * - if we have an engine failure
539  * - if we have vtol transition failure
540  * - depending on datalink, RC and if the mission is finished */
541 
542  if (is_armed && check_invalid_pos_nav_state(status, old_failsafe, mavlink_log_pub, status_flags, false, true)) {
543  // nothing to do - everything done in check_invalid_pos_nav_state
544  } else if (status->engine_failure) {
545  status->nav_state = vehicle_status_s::NAVIGATION_STATE_AUTO_LANDENGFAIL;
546 
547  } else if (status_flags.vtol_transition_failure) {
548  status->nav_state = vehicle_status_s::NAVIGATION_STATE_AUTO_RTL;
549 
550  } else if (status->mission_failure) {
551  status->nav_state = vehicle_status_s::NAVIGATION_STATE_AUTO_RTL;
552 
553  } else if (data_link_loss_act_configured && status->data_link_lost && is_armed) {
554  /* datalink loss enabled:
555  * check for datalink lost: this should always trigger RTGS */
556  enable_failsafe(status, old_failsafe, mavlink_log_pub, reason_no_datalink);
557 
558  set_link_loss_nav_state(status, armed, status_flags, internal_state, data_link_loss_act,
559  vehicle_status_s::NAVIGATION_STATE_AUTO_RTGS);
560 
561  } else if (!data_link_loss_act_configured && status->rc_signal_lost && status->data_link_lost && !landed
562  && mission_finished && is_armed) {
563  /* datalink loss DISABLED:
564  * check if both, RC and datalink are lost during the mission
565  * or all links are lost after the mission finishes in air: this should always trigger RCRECOVER */
566  enable_failsafe(status, old_failsafe, mavlink_log_pub, reason_no_datalink);
567 
568  set_link_loss_nav_state(status, armed, status_flags, internal_state, rc_loss_act,
569  vehicle_status_s::NAVIGATION_STATE_AUTO_RCRECOVER);
570 
571  } else if (!stay_in_failsafe) {
572  /* stay where you are if you should stay in failsafe, otherwise everything is perfect */
