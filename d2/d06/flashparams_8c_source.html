75 static int
76 param_export_internal(bool only_unsaved)
77 {
78  struct param_wbuf_s *s = NULL;
79  struct bson_encoder_s encoder;
80  int result = -1;
81 
82  /* Use realloc */
83 
84  bson_encoder_init_buf(&encoder, NULL, 0);
85 
86  /* no modified parameters -> we are done */
87  if (param_values == NULL) {
88  result = 0;
89  goto out;
90  }
91 
92  while ((s = (struct param_wbuf_s *)utarray_next(param_values, s)) != NULL) {
93 
94  int32_t i;
95  float f;
96 
97  /*
98  * If we are only saving values changed since last save, and this
99  * one hasn't, then skip it
100  */
101  if (only_unsaved && !s->unsaved) {
102  continue;
103  }
104 
105  s->unsaved = false;
106 
107  /* append the appropriate BSON type object */
108 
109  switch (param_type(s->param)) {
110 
111  case PARAM_TYPE_INT32:
112  i = s->val.i;
113 
114  if (bson_encoder_append_int(&encoder, param_name(s->param), i)) {
115  debug("BSON append failed for '%s'", param_name(s->param));
116  goto out;
117  }
118 
119  break;
120 
121  case PARAM_TYPE_FLOAT:
122  f = s->val.f;
123 
124  if (bson_encoder_append_double(&encoder, param_name(s->param), f)) {
125  debug("BSON append failed for '%s'", param_name(s->param));
126  goto out;
127  }
128 
129  break;
130 
131  case PARAM_TYPE_STRUCT ... PARAM_TYPE_STRUCT_MAX:
132  if (bson_encoder_append_binary(&encoder,
133  param_name(s->param),
134  BSON_BIN_BINARY,
135  param_size(s->param),
136  param_get_value_ptr_external(s->param))) {
137  debug("BSON append failed for '%s'", param_name(s->param));
138  goto out;
139  }
140 
141  break;
142 
143  default:
144  debug("unrecognized parameter type");
145  goto out;
146  }
147  }
148 
149  result = 0;
150 
151 out:
152 
153  if (result == 0) {
154 
155  /* Finalize the bison encoding*/
156 
157  bson_encoder_fini(&encoder);
158 
159  /* Get requiered space */
160 
161  size_t buf_size = bson_encoder_buf_size(&encoder);
162 
163  int shutdown_lock_ret = px4_shutdown_lock();
164 
165  if (shutdown_lock_ret) {
166  PX4_ERR("px4_shutdown_lock() failed (%i)", shutdown_lock_ret);
167  }
168 
169  /* Get a buffer from the flash driver with enough space */
170 
171  uint8_t *buffer;
172  result = parameter_flashfs_alloc(parameters_token, &buffer, &buf_size);
173 
174  if (result == OK) {
175 
176  /* Check for a write that has no changes */
177 
178  uint8_t *was_buffer;
179  size_t was_buf_size;
180  int was_result = parameter_flashfs_read(parameters_token, &was_buffer, &was_buf_size);
181 
182  void *enc_buff = bson_encoder_buf_data(&encoder);
183 
184  bool commit = was_result < OK || was_buf_size != buf_size || 0 != memcmp(was_buffer, enc_buff, was_buf_size);
185 
186  if (commit) {
187 
188  memcpy(buffer, enc_buff, buf_size);
189  result = parameter_flashfs_write(parameters_token, buffer, buf_size);
190  result = result == buf_size ? OK : -EFBIG;
191 
192  }
193 
194  free(enc_buff);
195  parameter_flashfs_free();
196  }
197 
198  if (shutdown_lock_ret == 0) {
199  px4_shutdown_unlock();
200  }
201 
202  }
203 
204  return result;
205 }

211 static int
212 param_import_callback(bson_decoder_t decoder, void *private, bson_node_t node)
213 {
214  float f;
215  int32_t i;
216  void *v, *tmp = NULL;
217  int result = -1;
218  struct param_import_state *state = (struct param_import_state *)private;
219 
220  /*
221  * EOO means the end of the parameter object. (Currently not supporting
222  * nested BSON objects).
223  */
224  if (node->type == BSON_EOO) {
225  debug("end of parameters");
226  return 0;
227  }
228 
229  /*
230  * Find the parameter this node represents. If we don't know it,
231  * ignore the node.
232  */
233  param_t param = param_find_no_notification(node->name);
234 
235  if (param == PARAM_INVALID) {
236  debug("ignoring unrecognised parameter '%s'", node->name);
237  return 1;
238  }
239 
240  /*
241  * Handle setting the parameter from the node
242  */
243 
244  switch (node->type) {
245  case BSON_INT32:
246  if (param_type(param) != PARAM_TYPE_INT32) {
247  PX4_WARN("unexpected type for %s", node->name);
248  result = 1; // just skip this entry
249  goto out;
250  }
251 
252  i = node->i;
253  v = &i;
254  break;
255 
256  case BSON_DOUBLE:
257  if (param_type(param) != PARAM_TYPE_FLOAT) {
258  PX4_WARN("unexpected type for %s", node->name);
259  result = 1; // just skip this entry
260  goto out;
261  }
262 
263  f = node->d;
264  v = &f;
265  break;
266 
267  case BSON_BINDATA:
268  if (node->subtype != BSON_BIN_BINARY) {
269  PX4_WARN("unexpected type for %s", node->name);
270  result = 1; // just skip this entry
271  goto out;
272  }
273 
274  if (bson_decoder_data_pending(decoder) != param_size(param)) {
275  PX4_WARN("bad size for '%s'", node->name);
276  result = 1; // just skip this entry
277  goto out;
278  }
279 
280  /* XXX check actual file data size? */
281  tmp = malloc(param_size(param));
282 
283  if (tmp == NULL) {
284  debug("failed allocating for '%s'", node->name);
285  goto out;
286  }
287 
288  if (bson_decoder_copy_data(decoder, tmp)) {
289  debug("failed copying data for '%s'", node->name);
290  goto out;
291  }
292 
293  v = tmp;
294  break;
295 
296  default:
297  debug("unrecognised node type");
298  goto out;
299  }
300 
301  if (param_set_external(param, v, state->mark_saved, true)) {
302 
303  debug("error setting value for '%s'", node->name);
304  goto out;
305  }
306 
307  if (tmp != NULL) {
308  free(tmp);
309  tmp = NULL;
310  }
311 
312  /* don't return zero, that means EOF */
313  result = 1;
314 
315 out:
316 
317  if (tmp != NULL) {
318  free(tmp);
319  }
320 
321  return result;
322 }

324 static int
325 param_import_internal(bool mark_saved)
326 {
327  struct bson_decoder_s decoder;
328  int result = -1;
329  struct param_import_state state;
330 
331  uint8_t *buffer = 0;
332  size_t buf_size;
333  parameter_flashfs_read(parameters_token, &buffer, &buf_size);
334 
335  if (bson_decoder_init_buf(&decoder, buffer, buf_size, param_import_callback, &state)) {
336  debug("decoder init failed");
337  goto out;
338  }
339 
340  state.mark_saved = mark_saved;
341 
342  do {
343  result = bson_decoder_next(&decoder);
344 
345  } while (result > 0);
346 
347 out:
348 
349  if (result < 0) {
350  debug("BSON error decoding parameters");
351  }
352 
353  return result;
354 }
