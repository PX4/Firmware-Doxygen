125 unsigned
126 MixerGroup::set_trims(int16_t *values, unsigned n)
127 {
128  Mixer *mixer = _first;
129  unsigned index = 0;
130 
131  while ((mixer != nullptr) && (index < n)) {
132  /* convert from integer to float */
133  float offset = (float)values[index] / 10000;
134 
135  /* to be safe, clamp offset to range of [-500, 500] usec */
136  if (offset < -1.0f) { offset = -1.0f; }
137 
138  if (offset > 1.0f) { offset = 1.0f; }
139 
140  debug("set trim: %d, offset: %5.3f", values[index], (double)offset);
141  index += mixer->set_trim(offset);
142  mixer = mixer->_next;
143  }
144 
145  return index;
146 }

148 /*
149  * get_trims() has no effect except for the SimpleMixer implementation for which get_trim()
150  * always returns the value one and sets the trim value.
151  * The only other existing implementation is MultirotorMixer, which ignores the trim value
152  * and returns _rotor_count.
153  */
154 unsigned
155 MixerGroup::get_trims(int16_t *values)
156 {
157  Mixer *mixer = _first;
158  unsigned index_mixer = 0;
159  unsigned index = 0;
160  float trim;
161 
162  while (mixer != nullptr) {
163  trim = 0;
164  index_mixer += mixer->get_trim(&trim);
165 
166  // MultirotorMixer returns the number of motors so we
167  // loop through index_mixer and set the same trim value for all motors
168  while (index < index_mixer) {
169  values[index] = trim * 10000;
170  index++;
171  }
172 
173  mixer = mixer->_next;
174  }
175 
176  return index;
177 }

240 int
241 MixerGroup::load_from_buf(const char *buf, unsigned &buflen)
242 {
243  int ret = -1;
244  const char *end = buf + buflen;
245 
246  /*
247  * Loop until either we have emptied the buffer, or we have failed to
248  * allocate something when we expected to.
249  */
250  while (buflen > 0) {
251  Mixer *m = nullptr;
252  const char *p = end - buflen;
253  unsigned resid = buflen;
254 
255  /*
256  * Use the next character as a hint to decide which mixer class to construct.
257  */
258  switch (*p) {
259  case 'Z':
260  m = NullMixer::from_text(p, resid);
261  break;
262 
263  case 'M':
264  m = SimpleMixer::from_text(_control_cb, _cb_handle, p, resid);
265  break;
266 
267  case 'R':
268  m = MultirotorMixer::from_text(_control_cb, _cb_handle, p, resid);
269  break;
270 
271  case 'H':
272  m = HelicopterMixer::from_text(_control_cb, _cb_handle, p, resid);
273  break;
274 
275  default:
276  /* it's probably junk or whitespace, skip a byte and retry */
277  buflen--;
278  continue;
279  }
280 
281  /*
282  * If we constructed something, add it to the group.
283  */
284  if (m != nullptr) {
285  add_mixer(m);
286 
287  /* we constructed something */
288  ret = 0;
289 
290  /* only adjust buflen if parsing was successful */
291  buflen = resid;
292  debug("SUCCESS - buflen: %d", buflen);
293 
294  } else {
295 
296  /*
297  * There is data in the buffer that we expected to parse, but it didn't,
298  * so give up for now.
299  */
300  break;
301  }
302  }
303 
304  /* nothing more in the buffer for us now */
305  return ret;
306 }
