55 /**
56  * Start the driver.
57  *
58  * This function call only returns once the driver is
59  * up and running or failed to detect the sensor.
60  */
61 void
62 start(enum Rotation rotation)
63 {
64  int fd_mag = -1;
65 
66  if (g_dev != nullptr) {
67  errx(0, "already started");
68  }
69 
70  /* create the driver */
71 #if defined(PX4_SPIDEV_LSM303A_M) && defined(PX4_SPIDEV_LSM303A_X)
72  g_dev = new LSM303AGR(PX4_SPI_BUS_SENSOR5, LSM303AGR_DEVICE_PATH_MAG, PX4_SPIDEV_LSM303A_M, rotation);
73 #else
74  errx(0, "External SPI not available");
75 #endif
76 
77  if (g_dev == nullptr) {
78  PX4_ERR("alloc failed");
79  goto fail;
80  }
81 
82  if (OK != g_dev->init()) {
83  goto fail;
84  }
85 
86  fd_mag = px4_open(LSM303AGR_DEVICE_PATH_MAG, O_RDONLY);
87 
88  /* don't fail if open cannot be opened */
89  if (0 <= fd_mag) {
90  if (px4_ioctl(fd_mag, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT) < 0) {
91  goto fail;
92  }
93  }
94 
95  px4_close(fd_mag);
96 
97  exit(0);
98 fail:
99 
100  if (g_dev != nullptr) {
101  delete g_dev;
102  g_dev = nullptr;
103  }
104 
105  errx(1, "driver start failed");
106 }

135 int
136 lsm303agr_main(int argc, char *argv[])
137 {
138  int ch;
139  enum Rotation rotation = ROTATION_NONE;
140 
141  /* jump over start/off/etc and look at options first */
142  while ((ch = getopt(argc, argv, "XR:a:")) != EOF) {
143  switch (ch) {
144  case 'R':
145  rotation = (enum Rotation)atoi(optarg);
146  break;
147 
148  default:
149  lsm303agr::usage();
150  exit(0);
151  }
152  }
153 
154  const char *verb = argv[optind];
155 
156  /*
157  * Start/load the driver.
158 
159  */
160  if (!strcmp(verb, "start")) {
161  lsm303agr::start(rotation);
162  }
163 
164  /*
165  * Print driver information.
166  */
167  if (!strcmp(verb, "info")) {
168  lsm303agr::info();
169  }
170 
171  errx(1, "unrecognized command, try 'start', 'info'");
172 }
