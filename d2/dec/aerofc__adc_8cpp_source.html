139 int aerofc_adc_main(int argc, char *argv[])
140 {
141  int ch;
142  enum AEROFC_ADC_BUS busid = AEROFC_ADC_BUS_ALL;
143 
144  while ((ch = getopt(argc, argv, "XI")) != EOF) {
145  switch (ch) {
146  case 'X':
147  busid = AEROFC_ADC_BUS_I2C_EXTERNAL;
148  break;
149 
150  case 'I':
151  busid = AEROFC_ADC_BUS_I2C_INTERNAL;
152  break;
153 
154  default:
155  help();
156  return -1;
157  }
158  }
159 
160  if (optind >= argc) {
161  help();
162  return PX4_ERROR;
163  }
164 
165  const char *verb = argv[optind];
166 
167  if (!strcmp(verb, "start")) {
168  if (instance) {
169  warn("AEROFC_ADC was already started");
170  return PX4_OK;
171  }
172 
173  for (uint8_t i = 0; i < (sizeof(bus_options) / sizeof(bus_options[0])); i++) {
174  if (busid != AEROFC_ADC_BUS_ALL && busid != bus_options[i].busid) {
175  continue;
176  }
177 
178  instance = new AEROFC_ADC(bus_options[i].busnum);
179 
180  if (!instance) {
181  warn("No memory to instance AEROFC_ADC");
182  return PX4_ERROR;
183  }
184 
185  if (instance->init() == PX4_OK) {
186  break;
187  }
188 
189  warn("AEROFC_ADC not found on busnum=%u", bus_options[i].busnum);
190  delete instance;
191  instance = nullptr;
192  }
193 
194  if (!instance) {
195  warn("AEROFC_ADC not found");
196  return PX4_ERROR;
197  }
198 
199  } else if (!strcmp(verb, "test")) {
200  test();
201 
202  } else {
203  warn("Action not supported");
204  help();
205  return PX4_ERROR;
206  }
207 
208  return PX4_OK;
209 }

238 int AEROFC_ADC::probe()
239 {
240  uint8_t buffer[2];
241  int ret;
242 
243  _retries = 3;
244 
245  /* Enable ADC */
246  buffer[0] = ADC_ENABLE_REG;
247  buffer[1] = 0x01;
248  ret = transfer(buffer, 2, NULL, 0);
249 
250  if (ret != PX4_OK) {
251  goto error;
252  }
253 
254  usleep(10000);
255 
256  /* Read ADC value */
257  buffer[0] = ADC_CHANNEL_REG;
258  ret = transfer(buffer, 1, buffer, 2);
259 
260  if (ret != PX4_OK) {
261  goto error;
262  }
263 
264  return PX4_OK;
265
