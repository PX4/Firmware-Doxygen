124 int uORBTest::UnitTest::latency_test(orb_id_t T, bool print)
125 {
126  test_note("---------------- LATENCY TEST ------------------");
127  S t;
128  t.val = 308;
129  t.time = hrt_absolute_time();
130 
131  orb_advert_t pfd0 = orb_advertise(T, &t);
132 
133  if (pfd0 == nullptr) {
134  return test_fail("orb_advertise failed (%i)", errno);
135  }
136 
137  char *const args[1] = { nullptr };
138 
139  pubsubtest_print = print;
140  pubsubtest_passed = false;
141 
142  /* test pub / sub latency */
143 
144  // Can't pass a pointer in args, must be a null terminated
145  // array of strings because the strings are copied to
146  // prevent access if the caller data goes out of scope
147  int pubsub_task = px4_task_spawn_cmd("uorb_latency",
148  SCHED_DEFAULT,
149  SCHED_PRIORITY_MAX,
150  1700,
151  (px4_main_t)&uORBTest::UnitTest::pubsubtest_threadEntry,
152  args);
153 
154  /* give the test task some data */
155  while (!pubsubtest_passed) {
156  ++t.val;
157  t.time = hrt_absolute_time();
158 
159  if (PX4_OK != orb_publish(T, pfd0, &t)) {
160  return test_fail("mult. pub0 timing fail");
161  }
162 
163  /* simulate >800 Hz system operation */
164  usleep(1000);
165  }
166 
167  if (pubsub_task < 0) {
168  return test_fail("failed launching task");
169  }
170 
171  orb_unadvertise(pfd0);
172 
173  return pubsubtest_res;
174 }
