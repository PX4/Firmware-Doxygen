59 void pwm_limit_calc(const bool armed, const bool pre_armed, const unsigned num_channels, const uint16_t reverse_mask,
60  const uint16_t *disarmed_pwm, const uint16_t *min_pwm, const uint16_t *max_pwm,
61  const float *output, uint16_t *effective_pwm, pwm_limit_t *limit)
62 {
63 
64  /* first evaluate state changes */
65  switch (limit->state) {
66  case PWM_LIMIT_STATE_INIT:
67 
68  if (armed) {
69 
70  /* set arming time for the first call */
71  if (limit->time_armed == 0) {
72  limit->time_armed = hrt_absolute_time();
73  }
74 
75  if (hrt_elapsed_time(&limit->time_armed) >= INIT_TIME_US) {
76  limit->state = PWM_LIMIT_STATE_OFF;
77  }
78  }
79 
80  break;
81 
82  case PWM_LIMIT_STATE_OFF:
83  if (armed) {
84  limit->state = PWM_LIMIT_STATE_RAMP;
85 
86  /* reset arming time, used for ramp timing */
87  limit->time_armed = hrt_absolute_time();
88  }
89 
90  break;
91 
92  case PWM_LIMIT_STATE_RAMP:
93  if (!armed) {
94  limit->state = PWM_LIMIT_STATE_OFF;
95 
96  } else if (hrt_elapsed_time(&limit->time_armed) >= RAMP_TIME_US) {
97  limit->state = PWM_LIMIT_STATE_ON;
98  }
99 
100  break;
101 
102  case PWM_LIMIT_STATE_ON:
103  if (!armed) {
104  limit->state = PWM_LIMIT_STATE_OFF;
105  }
106 
107  break;
108 
109  default:
110  break;
111  }
112 
113  /* if the system is pre-armed, the limit state is temporarily on,
114  * as some outputs are valid and the non-valid outputs have been
115  * set to NaN. This is not stored in the state machine though,
116  * as the throttle channels need to go through the ramp at
117  * regular arming time.
118  */
119 
120  unsigned local_limit_state = limit->state;
121 
122  if (pre_armed) {
123  local_limit_state = PWM_LIMIT_STATE_ON;
124  }
125 
126  unsigned progress;
127 
128  /* then set effective_pwm based on state */
129  switch (local_limit_state) {
130  case PWM_LIMIT_STATE_OFF:
131  case PWM_LIMIT_STATE_INIT:
132  for (unsigned i = 0; i < num_channels; i++) {
133  effective_pwm[i] = disarmed_pwm[i];
134  }
135 
136  break;
137 
138  case PWM_LIMIT_STATE_RAMP: {
139  hrt_abstime diff = hrt_elapsed_time(&limit->time_armed);
140 
141  progress = diff * PROGRESS_INT_SCALING / RAMP_TIME_US;
142 
143  if (progress > PROGRESS_INT_SCALING) {
144  progress = PROGRESS_INT_SCALING;
145  }
146 
147  for (unsigned i = 0; i < num_channels; i++) {
148 
149  float control_value = output[i];
150 
151  /* check for invalid / disabled channels */
152  if (!PX4_ISFINITE(control_value)) {
153  effective_pwm[i] = disarmed_pwm[i];
154  continue;
155  }
156 
157  uint16_t ramp_min_pwm;
158 
159  /* if a disarmed pwm value was set, blend between disarmed and min */
160  if (disarmed_pwm[i] > 0) {
161 
162  /* safeguard against overflows */
163  unsigned disarmed = disarmed_pwm[i];
164 
165  if (disarmed > min_pwm[i]) {
166  disarmed = min_pwm[i];
167  }
168 
169  unsigned disarmed_min_diff = min_pwm[i] - disarmed;
170  ramp_min_pwm = disarmed + (disarmed_min_diff * progress) / PROGRESS_INT_SCALING;
171 
172  } else {
173 
174  /* no disarmed pwm value set, choose min pwm */
175  ramp_min_pwm = min_pwm[i];
176  }
177 
178  if (reverse_mask & (1 << i)) {
179  control_value = -1.0f * control_value;
180  }
181 
182  effective_pwm[i] = control_value * (max_pwm[i] - ramp_min_pwm) / 2 + (max_pwm[i] + ramp_min_pwm) / 2;
183 
184  /* last line of defense against invalid inputs */
185  if (effective_pwm[i] < ramp_min_pwm) {
186  effective_pwm[i] = ramp_min_pwm;
187 
188  } else if (effective_pwm[i] > max_pwm[i]) {
189  effective_pwm[i] = max_pwm[i];
190  }
191  }
192  }
193  break;
194 
195  case PWM_LIMIT_STATE_ON:
196 
197  for (unsigned i = 0; i < num_channels; i++) {
198 
199  float control_value = output[i];
200 
201  /* check for invalid / disabled channels */
202  if (!PX4_ISFINITE(control_value)) {
203  effective_pwm[i] = disarmed_pwm[i];
204  continue;
205  }
206 
207  if (reverse_mask & (1 << i)) {
208  control_value = -1.0f * control_value;
209  }
210 
211  effective_pwm[i] = control_value * (max_pwm[i] - min_pwm[i]) / 2 + (max_pwm[i] + min_pwm[i]) / 2;
212 
213  /* last line of defense against invalid inputs */
214  if (effective_pwm[i] < min_pwm[i]) {
215  effective_pwm[i] = min_pwm[i];
216 
217  } else if (effective_pwm[i] > max_pwm[i]) {
218  effective_pwm[i] = max_pwm[i];
219  }
220 
221  }
222 
223  break;
224 
225  default:
226  break;
227  }
