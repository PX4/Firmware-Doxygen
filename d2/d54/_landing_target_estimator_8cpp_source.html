81 void LandingTargetEstimator::update()
82 {
83  _check_params(false);
84 
85  _update_topics();
86 
87  /* predict */
88  if (_estimator_initialized) {
89  if (hrt_absolute_time() - _last_update > landing_target_estimator_TIMEOUT_US) {
90  PX4_WARN("Timeout");
91  _estimator_initialized = false;
92 
93  } else {
94  float dt = (hrt_absolute_time() - _last_predict) / SEC2USEC;
95 
96  // predict target position with the help of accel data
97  matrix::Vector3f a;
98 
99  if (_vehicleAttitude_valid && _sensorBias_valid) {
100  matrix::Quaternion<float> q_att(&_vehicleAttitude.q[0]);
101  _R_att = matrix::Dcm<float>(q_att);
102  a(0) = _sensorBias.accel_x;
103  a(1) = _sensorBias.accel_y;
104  a(2) = _sensorBias.accel_z;
105  a = _R_att * a;
106 
107  } else {
108  a.zero();
109  }
110 
111  _kalman_filter_x.predict(dt, -a(0), _params.acc_unc);
112  _kalman_filter_y.predict(dt, -a(1), _params.acc_unc);
113 
114  _last_predict = hrt_absolute_time();
115  }
116  }
117 
118  if (!_new_irlockReport) {
119  // nothing to do
120  return;
121  }
122 
123  // mark this sensor measurement as consumed
124  _new_irlockReport = false;
125 
126  if (!_vehicleAttitude_valid || !_vehicleLocalPosition_valid || !_vehicleLocalPosition.dist_bottom_valid) {
127  // don't have the data needed for an update
128  return;
129  }
130 
131  if (!PX4_ISFINITE(_irlockReport.pos_y) || !PX4_ISFINITE(_irlockReport.pos_x)) {
132  return;
133  }
134 
135  // TODO account for sensor orientation as set by parameter
136  // default orientation has camera x pointing in body y, camera y in body -x
137 
138  matrix::Vector<float, 3> sensor_ray; // ray pointing towards target in body frame
139  sensor_ray(0) = -_irlockReport.pos_y * _params.scale_y; // forward
140  sensor_ray(1) = _irlockReport.pos_x * _params.scale_x; // right
141  sensor_ray(2) = 1.0f;
142 
143  // rotate the unit ray into the navigation frame, assume sensor frame = body frame
144  matrix::Quaternion<float> q_att(&_vehicleAttitude.q[0]);
145  _R_att = matrix::Dcm<float>(q_att);
146  sensor_ray = _R_att * sensor_ray;
147 
148  if (fabsf(sensor_ray(2)) < 1e-6f) {
149  // z component of measurement unsafe, don't use this measurement
150  return;
151  }
152 
153  float dist = _vehicleLocalPosition.dist_bottom;
154 
155  // scale the ray s.t. the z component has length of dist
156  _rel_pos(0) = sensor_ray(0) / sensor_ray(2) * dist;
157  _rel_pos(1) = sensor_ray(1) / sensor_ray(2) * dist;
158 
159  if (!_estimator_initialized) {
160  PX4_INFO("Init");
161  float vx_init = _vehicleLocalPosition.v_xy_valid ? -_vehicleLocalPosition.vx : 0.f;
162  float vy_init = _vehicleLocalPosition.v_xy_valid ? -_vehicleLocalPosition.vy : 0.f;
163  _kalman_filter_x.init(_rel_pos(0), vx_init, _params.pos_unc_init, _params.vel_unc_init);
164  _kalman_filter_y.init(_rel_pos(1), vy_init, _params.pos_unc_init, _params.vel_unc_init);
165 
166  _estimator_initialized = true;
167  _last_update = hrt_absolute_time();
168  _last_predict = _last_update;
169 
170  } else {
171  // update
172  bool update_x = _kalman_filter_x.update(_rel_pos(0), _params.meas_unc * dist * dist);
173  bool update_y = _kalman_filter_y.update(_rel_pos(1), _params.meas_unc * dist * dist);
174 
175  if (!update_x || !update_y) {
176  if (!_faulty) {
177  _faulty = true;
178  PX4_WARN("Landing target measurement rejected:%s%s", update_x ? "" : " x", update_y ? "" : " y");
179  }
180 
181  } else {
182  _faulty = false;
183  }
184 
185  if (!_faulty) {
186  // only publish if both measurements were good
187 
188  _target_pose.timestamp = _irlockReport.timestamp;
189 
190  float x, xvel, y, yvel, covx, covx_v, covy, covy_v;
191  _kalman_filter_x.getState(x, xvel);
192  _kalman_filter_x.getCovariance(covx, covx_v);
193 
194  _kalman_filter_y.getState(y, yvel);
195  _kalman_filter_y.getCovariance(covy, covy_v);
196 
197  _target_pose.is_static = (_params.mode == TargetMode::Stationary);
198 
199  _target_pose.rel_pos_valid = true;
200  _target_pose.rel_vel_valid = true;
201  _target_pose.x_rel = x;
202  _target_pose.y_rel = y;
203  _target_pose.z_rel = dist;
204  _target_pose.vx_rel = xvel;
205  _target_pose.vy_rel = yvel;
206 
207  _target_pose.cov_x_rel = covx;
208  _target_pose.cov_y_rel = covy;
209 
210  _target_pose.cov_vx_rel = covx_v;
211  _target_pose.cov_vy_rel = covy_v;
212 
213  if (_vehicleLocalPosition_valid && _vehicleLocalPosition.xy_valid) {
214  _target_pose.x_abs = x + _vehicleLocalPosition.x;
215  _target_pose.y_abs = y + _vehicleLocalPosition.y;
216  _target_pose.z_abs = dist + _vehicleLocalPosition.z;
217  _target_pose.abs_pos_valid = true;
218 
219  } else {
220  _target_pose.abs_pos_valid = false;
221  }
222 
223  if (_targetPosePub == nullptr) {
224  _targetPosePub = orb_advertise(ORB_ID(landing_target_pose), &_target_pose);
225 
226  } else {
227  orb_publish(ORB_ID(landing_target_pose), _targetPosePub, &_target_pose);
228  }
229 
230  _last_update = hrt_absolute_time();
231  _last_predict = _last_update;
232  }
233 
234  float innov_x, innov_cov_x, innov_y, innov_cov_y;
235  _kalman_filter_x.getInnovations(innov_x, innov_cov_x);
236  _kalman_filter_y.getInnovations(innov_y, innov_cov_y);
237 
238  _target_innovations.timestamp = _irlockReport.timestamp;
239  _target_innovations.innov_x = innov_x;
240  _target_innovations.innov_cov_x = innov_cov_x;
241  _target_innovations.innov_y = innov_y;
242  _target_innovations.innov_cov_y = innov_cov_y;
243 
244  if (_targetInnovationsPub == nullptr) {
245  _targetInnovationsPub = orb_advertise(ORB_ID(landing_target_innovations), &_target_innovations);
246 
247  } else {
248  orb_publish(ORB_ID(landing_target_innovations), _targetInnovationsPub, &_target_innovations);
249  }
250  }

290 bool LandingTargetEstimator::_orb_update(const struct orb_metadata *meta, int handle, void *buffer)
291 {
292  bool newData = false;
293 
294  // check if there is new data to grab
295  if (orb_check(handle, &newData) != OK) {
296  return false;
297  }
298
