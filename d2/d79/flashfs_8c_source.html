169 /****************************************************************************
170  * Name: blank_check
171  *
172  * Description:
173  * Given a pointer to a flash entry header and a new size
174  *
175  * Input Parameters:
176  * pf - A pointer to the current flash entry header
177  * new_size - The total number of bytes to be written
178  *
179  * Returned value:
180  * true if space is blank, If it is not blank it returns false
181  *
182  ****************************************************************************/
183 
184 static bool blank_check(flash_entry_header_t *pf,
185  size_t new_size)
186 {
187  bool rv = true;
188  uint32_t *pm = (uint32_t *) pf;
189  new_size /= sizeof(uint32_t);
190 
191  while (new_size-- && rv) {
192  if (!blank_flash(pm++)) {
193  rv = false;
194  }
195  }
196 
197  return rv;
198 }

425 /****************************************************************************
426  * Name: find_entry
427  *
428  * Description:
429  * This helper function locates an "file" from the the file token
430  *
431  * Input Parameters:
432  * token - A flash file token, the pseudo file name
433  *
434  * Returned value:
435  * On Success a pointer to flash entry header or NULL on failure
436  *
437  *
438  ****************************************************************************/
439 
440 static flash_entry_header_t *find_entry(flash_file_token_t token)
441 {
442  for (int s = 0; sector_map[s].address; s++) {
443 
444  h_magic_t *pmagic = (h_magic_t *) sector_map[s].address;
445  h_magic_t *pe = pmagic + (sector_map[s].size / sizeof(h_magic_t)) - 1;
446 
447  /* Hunt for Magic Signature */
448 cont:
449 
450  while (pmagic != pe && !valid_magic(pmagic)) {
451  pmagic++;
452  }
453 
454  /* Did we reach the end
455  * if so try the next sector */
456 
457  if (pmagic == pe) { continue; }
458 
459  /* Found a magic So assume it is a file header */
460 
461  flash_entry_header_t *pf = (flash_entry_header_t *) pmagic;
462 
463  /* Test the CRC */
464 
465  if (pf->crc == crc32(entry_crc_start(pf), entry_crc_length(pf))) {
466 
467  /* Good CRC is it the one we are looking for ?*/
468 
469  if (valid_entry(pf) && pf->file_token.t == token.t) {
470 
471  return pf;
472 
473  } else {
474 
475  /* Not the one we wanted but we can trust the size */
476 
477  pf = next_entry(pf);
478  pmagic = (h_magic_t *) pf;
479 
480  /* If the next one is erased */
481 
482  if (blank_entry(pf)) {
483  continue;
484  }
485  }
486 
487  goto cont;
488 
489  } else {
490 
491  /* in valid CRC so keep looking */
492 
493  pmagic++;
494  }
495  }
496 
497  return NULL;
498 }

500 /****************************************************************************
501  * Name: find_free
502  *
503  * Description:
504  * This helper function locates free space for the number of bytes required
505  *
506  * Input Parameters:
507  * required - Number of bytes required
508  *
509  * Returned value:
510  * On Success a pointer to flash entry header or NULL on failure
511  *
512  *
513  ****************************************************************************/
514 
515 static flash_entry_header_t *find_free(data_size_t required)
516 {
517  for (int s = 0; sector_map[s].address; s++) {
518 
519  h_magic_t *pmagic = (h_magic_t *) sector_map[s].address;
520  h_magic_t *pe = pmagic + (sector_map[s].size / sizeof(h_magic_t)) - 1;
521 
522  /* Hunt for Magic Signature */
523 
524  do {
525 
526  if (valid_magic(pmagic)) {
527 
528  flash_entry_header_t *pf = (flash_entry_header_t *) pmagic;
529 
530  /* Test the CRC */
531 
532  if (pf->crc == crc32(entry_crc_start(pf), entry_crc_length(pf))) {
533 
534  /* Valid Magic and CRC look for the next record*/
535 
536  pmagic = ((uint32_t *) next_entry(pf));
537 
538  } else {
539 
540  pmagic++;
541  }
542  }
543 
544  if (blank_magic(pmagic)) {
545 
546  flash_entry_header_t *pf = (flash_entry_header_t *) pmagic;
547 
548  if (blank_entry(pf) && blank_check(pf, required)) {
549  return pf;
550  }
551 
552  }
553  } while (++pmagic != pe);
554  }
555 
556  return NULL;
557 }

688  *
689  * Description:
690  * Given a pointer to a flash entry header and a new size

692  * Input Parameters:

702 static sector_descriptor_t *check_free_space_in_sector(flash_entry_header_t

704 {

706  uint8_t *psector_first = (uint8_t *) sm->address;

709 
710  if (pnext_end >= psector_first && pnext_end <= psector_last) {
711  sm = 0;
712  }

766  *
767  * Input Parameters:

779 int

785

787

789

791  size_t alignment = sizeof(h_magic_t) - 1;

793  total_size += size_adjust;
794

796

798

804

808  return -ENOSPC;
809  }
810 
811  } else {
812 
813  /* Do we have space after the entry in the sector for the update */
814 
815  sector_descriptor_t *current_sector = check_free_space_in_sector(pf,
816  total_size);
817 
818 
819  if (current_sector == 0) {
820 
821  /* Mark the last entry erased */
822 
823  /* todo:consider a 2 stage erase or write before erase and do a fs check
824  * at start up
825  */
826
