395 bool
396 MavlinkReceiver::evaluate_target_ok(int command, int target_system, int target_component)
397 {
398  /* evaluate if this system should accept this command */
399  bool target_ok = false;
400 
401  switch (command) {
402 
403  case MAV_CMD_REQUEST_AUTOPILOT_CAPABILITIES:
404  case MAV_CMD_REQUEST_PROTOCOL_VERSION:
405  /* broadcast and ignore component */
406  target_ok = (target_system == 0) || (target_system == mavlink_system.sysid);
407  break;
408 
409  default:
410  target_ok = (target_system == mavlink_system.sysid) && ((target_component == mavlink_system.compid)
411  || (target_component == MAV_COMP_ID_ALL));
412  break;
413  }
414 
415  return target_ok;
416 }

442 void
443 MavlinkReceiver::send_storage_information(int storage_id)
444 {
445  mavlink_storage_information_t storage_info{};
446  const char *microsd_dir = PX4_STORAGEDIR;
447 
448  if (storage_id == 0 || storage_id == 1) { // request is for all or the first storage
449  storage_info.storage_id = 1;
450 
451  struct statfs statfs_buf;
452  uint64_t total_bytes = 0;
453  uint64_t avail_bytes = 0;
454 
455  if (statfs(microsd_dir, &statfs_buf) == 0) {
456  total_bytes = (uint64_t)statfs_buf.f_blocks * statfs_buf.f_bsize;
457  avail_bytes = (uint64_t)statfs_buf.f_bavail * statfs_buf.f_bsize;
458  }
459 
460  if (total_bytes == 0) { // on NuttX we get 0 total bytes if no SD card is inserted
461  storage_info.storage_count = 0;
462  storage_info.status = 0; // not available
463 
464  } else {
465  storage_info.storage_count = 1;
466  storage_info.status = 2; // available & formatted
467  storage_info.total_capacity = total_bytes / 1024. / 1024.;
468  storage_info.available_capacity = avail_bytes / 1024. / 1024.;
469  storage_info.used_capacity = (total_bytes - avail_bytes) / 1024. / 1024.;
470  }
471 
472  } else {
473  // only one storage supported
474  storage_info.storage_id = storage_id;
475  storage_info.storage_count = 1;
476  }
477 
478  storage_info.time_boot_ms = hrt_absolute_time() / 1000;
479  mavlink_msg_storage_information_send_struct(_mavlink->get_channel(), &storage_info);
480 }

540 template <class T>
541 void MavlinkReceiver::handle_message_command_both(mavlink_message_t *msg, const T &cmd_mavlink,
542  const vehicle_command_s &vehicle_command)
543 {
544  bool target_ok = evaluate_target_ok(cmd_mavlink.command, cmd_mavlink.target_system, cmd_mavlink.target_component);
545 
546  bool send_ack = true;
547  uint8_t result = vehicle_command_ack_s::VEHICLE_RESULT_ACCEPTED;
548 
549  if (!target_ok) {
550  acknowledge(msg->sysid, msg->compid, cmd_mavlink.command, vehicle_command_ack_s::VEHICLE_RESULT_FAILED);
551  return;
552  }
553 
554  if (cmd_mavlink.command == MAV_CMD_REQUEST_AUTOPILOT_CAPABILITIES) {
555  /* send autopilot version message */
556  _mavlink->send_autopilot_capabilites();
557 
558  } else if (cmd_mavlink.command == MAV_CMD_REQUEST_PROTOCOL_VERSION) {
559  /* send protocol version message */
560  _mavlink->send_protocol_version();
561 
562  } else if (cmd_mavlink.command == MAV_CMD_GET_HOME_POSITION) {
563  _mavlink->configure_stream_threadsafe("HOME_POSITION", 0.5f);
564 
565  } else if (cmd_mavlink.command == MAV_CMD_SET_MESSAGE_INTERVAL) {
566  if (set_message_interval((int)(cmd_mavlink.param1 + 0.5f), cmd_mavlink.param2, cmd_mavlink.param3)) {
567  result = vehicle_command_ack_s::VEHICLE_RESULT_FAILED;
568  }
569 
570  } else if (cmd_mavlink.command == MAV_CMD_GET_MESSAGE_INTERVAL) {
571  get_message_interval((int)cmd_mavlink.param1);
572 
573  } else if (cmd_mavlink.command == MAV_CMD_REQUEST_FLIGHT_INFORMATION) {
574  send_flight_information();
575 
576  } else if (cmd_mavlink.command == MAV_CMD_REQUEST_STORAGE_INFORMATION) {
577  if ((int)(cmd_mavlink.param2 + 0.5f) == 1) {
578  send_storage_information(cmd_mavlink.param1 + 0.5f);
579  }
580 
581  } else {
582 
583  send_ack = false;
584 
585  if (msg->sysid == mavlink_system.sysid && msg->compid == mavlink_system.compid) {
586  PX4_WARN("ignoring CMD with same SYS/COMP (%d/%d) ID", mavlink_system.sysid, mavlink_system.compid);
587  return;
588  }
589 
590  if (cmd_mavlink.command == MAV_CMD_LOGGING_START) {
591  // check that we have enough bandwidth available: this is given by the configured logger topics
592  // and rates. The 5000 is somewhat arbitrary, but makes sure that we cannot enable log streaming
593  // on a radio link
594  if (_mavlink->get_data_rate() < 5000) {
595  send_ack = true;
596  result = vehicle_command_ack_s::VEHICLE_RESULT_DENIED;
597  _mavlink->send_statustext_critical("Not enough bandwidth to enable log streaming");
598 
599  } else {
600  // we already instanciate the streaming object, because at this point we know on which
601  // mavlink channel streaming was requested. But in fact it's possible that the logger is
602  // not even running. The main mavlink thread takes care of this by waiting for an ack
603  // from the logger.
604  _mavlink->try_start_ulog_streaming(msg->sysid, msg->compid);
605  }
606 
607  } else if (cmd_mavlink.command == MAV_CMD_LOGGING_STOP) {
608  _mavlink->request_stop_ulog_streaming();
609  }
610 
611  if (!send_ack) {
612  if (_cmd_pub == nullptr) {
613  _cmd_pub = orb_advertise_queue(ORB_ID(vehicle_command), &vehicle_command, vehicle_command_s::ORB_QUEUE_LENGTH);

1748  */
1749  return (_mom_switch_pos[sw] * 1000) / (state_count - 1) + 1000;
