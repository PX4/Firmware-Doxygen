107 void
108 PrecLand::on_active()
109 {
110  // get new target measurement
111  orb_check(_target_pose_sub, &_target_pose_updated);
112 
113  if (_target_pose_updated) {
114  orb_copy(ORB_ID(landing_target_pose), _target_pose_sub, &_target_pose);
115  _target_pose_valid = true;
116  }
117 
118  if ((hrt_elapsed_time(&_target_pose.timestamp) / 1e6f) > _param_timeout.get()) {
119  _target_pose_valid = false;
120  }
121 
122  // stop if we are landed
123  if (_navigator->get_land_detected()->landed) {
124  switch_to_state_done();
125  }
126 
127  switch (_state) {
128  case PrecLandState::Start:
129  run_state_start();
130  break;
131 
132  case PrecLandState::HorizontalApproach:
133  run_state_horizontal_approach();
134  break;
135 
136  case PrecLandState::DescendAboveTarget:
137  run_state_descend_above_target();
138  break;
139 
140  case PrecLandState::FinalApproach:
141  run_state_final_approach();
142  break;
143 
144  case PrecLandState::Search:
145  run_state_search();
146  break;
147 
148  case PrecLandState::Fallback:
149  run_state_fallback();
150  break;
151 
152  case PrecLandState::Done:
153  // nothing to do
154  break;
155 
156  default:
157  // unknown state
158  break;
159  }

163 void
164 PrecLand::run_state_start()
165 {
166  // check if target visible and go to horizontal approach
167  if (switch_to_state_horizontal_approach()) {
168  return;
169  }
170 
171  if (_mode == PrecLandMode::Opportunistic) {
172  // could not see the target immediately, so just fall back to normal landing
173  if (!switch_to_state_fallback()) {
174  PX4_ERR("Can't switch to search or fallback landing");
175  }
176  }
177 
178  position_setpoint_triplet_s *pos_sp_triplet = _navigator->get_position_setpoint_triplet();
179  float dist = get_distance_to_next_waypoint(pos_sp_triplet->current.lat, pos_sp_triplet->current.lon,
180  _navigator->get_global_position()->lat, _navigator->get_global_position()->lon);
181 
182  // check if we've reached the start point
183  if (dist < _navigator->get_acceptance_radius()) {
184  if (!_point_reached_time) {
185  _point_reached_time = hrt_absolute_time();
186  }
187 
188  // if we don't see the target after 1 second, search for it
189  if (_param_search_timeout.get() > 0) {
190 
191  if (hrt_absolute_time() - _point_reached_time > 2000000) {
192  if (!switch_to_state_search()) {
193  if (!switch_to_state_fallback()) {
194  PX4_ERR("Can't switch to search or fallback landing");
195  }
196  }
197  }
198 
199  } else {
200  if (!switch_to_state_fallback()) {
201  PX4_ERR("Can't switch to search or fallback landing");
202  }
203  }
204  }

207 void
208 PrecLand::run_state_horizontal_approach()
209 {
210  position_setpoint_triplet_s *pos_sp_triplet = _navigator->get_position_setpoint_triplet();
211 
212  // check if target visible, if not go to start
213  if (!check_state_conditions(PrecLandState::HorizontalApproach)) {
214  PX4_WARN("Lost landing target while landing (horizontal approach).");
215 
216  // Stay at current position for searching for the landing target
217  pos_sp_triplet->current.lat = _navigator->get_global_position()->lat;
218  pos_sp_triplet->current.lon = _navigator->get_global_position()->lon;
219  pos_sp_triplet->current.alt = _navigator->get_global_position()->alt;
220 
221  if (!switch_to_state_start()) {
222  if (!switch_to_state_fallback()) {
223  PX4_ERR("Can't switch to fallback landing");
224  }
225  }
226 
227  return;
228  }
229 
230  if (check_state_conditions(PrecLandState::DescendAboveTarget)) {
231  if (!_point_reached_time) {
232  _point_reached_time = hrt_absolute_time();
233  }
234 
235  if (hrt_absolute_time() - _point_reached_time > 2000000) {
236  // if close enough for descent above target go to descend above target
237  if (switch_to_state_descend_above_target()) {
238  return;
239  }
240  }
241 
242  }
243 
244  if (hrt_absolute_time() - _state_start_time > STATE_TIMEOUT) {
245  PX4_ERR("Precision landing took too long during horizontal approach phase.");
246 
247  if (switch_to_state_fallback()) {
248  return;
249  }
250 
251  PX4_ERR("Can't switch to fallback landing");
252  }
253 
254  float x = _target_pose.x_abs;
255  float y = _target_pose.y_abs;
256 
257  slewrate(x, y);
258 
259  // XXX need to transform to GPS coords because mc_pos_control only looks at that
260  double lat, lon;
261  map_projection_reproject(&_map_ref, x, y, &lat, &lon);
262 
263  pos_sp_triplet->current.lat = lat;
264  pos_sp_triplet->current.lon = lon;
265  pos_sp_triplet->current.alt = _approach_alt;
266  pos_sp_triplet->current.type = position_setpoint_s::SETPOINT_TYPE_POSITION;
267 
268  _navigator->set_position_setpoint_triplet_updated();
269 }

314 void
315 PrecLand::run_state_search()
316 {
317  // check if we can see the target
318  if (check_state_conditions(PrecLandState::HorizontalApproach)) {
319  if (!_target_acquired_time) {
320  // target just became visible. Stop climbing, but give it some margin so we don't stop too apruptly
321  _target_acquired_time = hrt_absolute_time();
322  position_setpoint_triplet_s *pos_sp_triplet = _navigator->get_position_setpoint_triplet();
323  float new_alt = _navigator->get_global_position()->alt + 1.0f;
324  pos_sp_triplet->current.alt = new_alt < pos_sp_triplet->current.alt ? new_alt : pos_sp_triplet->current.alt;
325  _navigator->set_position_setpoint_triplet_updated();
326  }
327 
328  }
329 
330  // stay at that height for a second to allow the vehicle to settle
331  if (_target_acquired_time && (hrt_absolute_time() - _target_acquired_time) > 1000000) {
332  // try to switch to horizontal approach
333  if (switch_to_state_horizontal_approach()) {
334  return;
335  }
336  }
337 
338  // check if search timed out and go to fallback
339  if (hrt_absolute_time() - _state_start_time > _param_search_timeout.get()*SEC2USEC) {
340  PX4_WARN("Search timed out");
341 
342  if (!switch_to_state_fallback()) {
343  PX4_ERR("Can't switch to fallback landing");
344  }
345  }

455 bool PrecLand::check_state_conditions(PrecLandState state)
456 {
457  vehicle_local_position_s *vehicle_local_position = _navigator->get_local_position();
458 
459  switch (state) {
460  case PrecLandState::Start:
461  return _search_cnt <= _param_max_searches.get();
462 
463  case PrecLandState::HorizontalApproach:
464 
465  // if we're already in this state, only want to make it invalid if we reached the target but can't see it anymore
466  if (_state == PrecLandState::HorizontalApproach) {
467  if (fabsf(_target_pose.x_abs - vehicle_local_position->x) < _param_hacc_rad.get()
468  && fabsf(_target_pose.y_abs - vehicle_local_position->y) < _param_hacc_rad.get()) {
469  // we've reached the position where we last saw the target. If we don't see it now, we need to do something
470  return _target_pose_valid && _target_pose.abs_pos_valid;
471 
472  } else {
473  // We've seen the target sometime during horizontal approach.
474  // Even if we don't see it as we're moving towards it, continue approaching last known location
475  return true;
476  }
477  }
478 
479  // If we're trying to switch to this state, the target needs to be visible
480  return _target_pose_updated && _target_pose_valid && _target_pose.abs_pos_valid;
481 
482  case PrecLandState::DescendAboveTarget:
483 
484  // if we're already in this state, only leave it if target becomes unusable, don't care about horizontall offset to target
485  if (_state == PrecLandState::DescendAboveTarget) {
486  // if we're close to the ground, we're more critical of target timeouts so we quickly go into descend
487  if (check_state_conditions(PrecLandState::FinalApproach)) {
488  return hrt_absolute_time() - _target_pose.timestamp < 500000; // 0.5s
489 
490  } else {
491  return _target_pose_valid && _target_pose.abs_pos_valid;
492  }
493 
494  } else {
495  // if not already in this state, need to be above target to enter it
496  return _target_pose_updated && _target_pose.abs_pos_valid
497  && fabsf(_target_pose.x_abs - vehicle_local_position->x) < _param_hacc_rad.get()
498  && fabsf(_target_pose.y_abs - vehicle_local_position->y) < _param_hacc_rad.get();
499  }
500 
501  case PrecLandState::FinalApproach:
502  return _target_pose_valid && _target_pose.abs_pos_valid
503  && (_target_pose.z_abs - vehicle_local_position->z) < _param_final_approach_alt.get();
504 
505  case PrecLandState::Search:
506  return true;
507 
508  case PrecLandState::Fallback:
509  return true;
510 
511  default:
512  return false;
513  }

516 void PrecLand::slewrate(float &sp_x, float &sp_y)
517 {
518  matrix::Vector2f sp_curr(sp_x, sp_y);
519  uint64_t now = hrt_absolute_time();
520 
521  float dt = (now - _last_slewrate_time);
522 
523  if (dt < 1) {
524  // bad dt, can't divide by it
525  return;
526  }
527 
528  dt /= SEC2USEC;
529 
530  if (!_last_slewrate_time) {
531  // running the first time since switching to precland
532 
533  // assume dt will be about 50000us
534  dt = 50000 / SEC2USEC;
535 
536  // set a best guess for previous setpoints for smooth transition
537  map_projection_project(&_map_ref, _navigator->get_position_setpoint_triplet()->current.lat,
538  _navigator->get_position_setpoint_triplet()->current.lon, &_sp_pev(0), &_sp_pev(1));
539  _sp_pev_prev(0) = _sp_pev(0) - _navigator->get_local_position()->vx * dt;
540  _sp_pev_prev(1) = _sp_pev(1) - _navigator->get_local_position()->vy * dt;
541  }
542 
543  _last_slewrate_time = now;
544 
545  // limit the setpoint speed to the maximum cruise speed
546  matrix::Vector2f sp_vel = (sp_curr - _sp_pev) / dt; // velocity of the setpoints
547 
548  if (sp_vel.length() > _param_xy_vel_cruise.get()) {
549  sp_vel = sp_vel.normalized() * _param_xy_vel_cruise.get();
550  sp_curr = _sp_pev + sp_vel * dt;
551  }
552 
553  // limit the setpoint acceleration to the maximum acceleration
554  matrix::Vector2f sp_acc = (sp_curr - _sp_pev * 2 + _sp_pev_prev) / (dt * dt); // acceleration of the setpoints
555 
556  if (sp_acc.length() > _param_acceleration_hor.get()) {
557  sp_acc = sp_acc.normalized() * _param_acceleration_hor.get();
558  sp_curr = _sp_pev * 2 - _sp_pev_prev + sp_acc * (dt * dt);
559  }
560 
561  // limit the setpoint speed such that we can stop at the setpoint given the maximum acceleration/deceleration
562  float max_spd = sqrtf(_param_acceleration_hor.get() * ((matrix::Vector2f)(_sp_pev - matrix::Vector2f(sp_x,
563  sp_y))).length());
564  sp_vel = (sp_curr - _sp_pev) / dt; // velocity of the setpoints
565 
566  if (sp_vel.length() > max_spd) {
567  sp_vel = sp_vel.normalized() * max_spd;
568  sp_curr = _sp_pev + sp_vel * dt;
569  }
570 
571  _sp_pev_prev = _sp_pev;
572  _sp_pev = sp_curr;
573 
574  sp_x = sp_curr(0);
575  sp_y = sp_curr(1);
576 }
