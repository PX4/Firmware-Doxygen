63 int RGBLED::start()
64 {
65  int res = DevObj::init();
66 
67  if (res != 0) {
68  DF_LOG_ERR("could not init DevObj (%i)", res);
69  return res;
70  }
71 
72  res = _gpioR.exportPin();
73 
74  if (res != 0) {
75  PX4_ERR("red led: failed to export");
76  goto cleanup;
77  }
78 
79  res = _gpioR.setDirection(LinuxGPIO::Direction::OUT);
80 
81  if (res != 0) {
82  PX4_ERR("red led: failed to set direction");
83  goto cleanup;
84  }
85 
86  res = _gpioG.exportPin();
87 
88  if (res != 0) {
89  PX4_ERR("green led: failed to export");
90  goto cleanup;
91  }
92 
93  res = _gpioG.setDirection(LinuxGPIO::Direction::OUT);
94 
95  if (res != 0) {
96  PX4_ERR("green led: failed to set direction");
97  goto cleanup;
98  }
99 
100  res = _gpioB.exportPin();
101 
102  if (res != 0) {
103  PX4_ERR("blue led: failed to export");
104  goto cleanup;
105  }
106 
107  res = _gpioB.setDirection(LinuxGPIO::Direction::OUT);
108 
109  if (res != 0) {
110  PX4_ERR("blue led: failed to set direction");
111  goto cleanup;
112  }
113 
114  // update at fixed interval
115  DevObj::setSampleInterval(_led_controller.maximum_update_interval());
116 
117  res = DevObj::start();
118 
119  if (res != 0) {
120  DF_LOG_ERR("could not start DevObj (%i)", res);
121  return res;
122  }
123 
124  return res;
125 
126 cleanup:
127  _gpioR.unexportPin();
128  _gpioG.unexportPin();
129  _gpioB.unexportPin();
130 
131  return res;
132 }

153 void RGBLED::_measure()
154 {
155  if (!_led_controller.is_init()) {
156  int led_control_sub = orb_subscribe(ORB_ID(led_control));
157  _led_controller.init(led_control_sub);
158  }
159 
160  LedControlData led_control_data;
161 
162  if (_led_controller.update(led_control_data) == 1) {
163  switch (led_control_data.leds[0].color) {
164  case led_control_s::COLOR_RED:
165  _gpioR.writeValue(LinuxGPIO::Value::LOW);
166  _gpioG.writeValue(LinuxGPIO::Value::HIGH);
167  _gpioB.writeValue(LinuxGPIO::Value::HIGH);
168  break;
169 
170  case led_control_s::COLOR_GREEN:
171  _gpioR.writeValue(LinuxGPIO::Value::HIGH);
172  _gpioG.writeValue(LinuxGPIO::Value::LOW);
173  _gpioB.writeValue(LinuxGPIO::Value::HIGH);
174  break;
175 
176  case led_control_s::COLOR_BLUE:
177  _gpioR.writeValue(LinuxGPIO::Value::HIGH);
178  _gpioG.writeValue(LinuxGPIO::Value::HIGH);
179  _gpioB.writeValue(LinuxGPIO::Value::LOW);
180  break;
181 
182  case led_control_s::COLOR_AMBER: //make it the same as yellow
183  case led_control_s::COLOR_YELLOW:
184  _gpioR.writeValue(LinuxGPIO::Value::LOW);
185  _gpioG.writeValue(LinuxGPIO::Value::LOW);
186  _gpioB.writeValue(LinuxGPIO::Value::HIGH);
187  break;
188 
189  case led_control_s::COLOR_PURPLE:
190  _gpioR.writeValue(LinuxGPIO::Value::LOW);
191  _gpioG.writeValue(LinuxGPIO::Value::HIGH);
192  _gpioB.writeValue(LinuxGPIO::Value::LOW);
193  break;
194 
195  case led_control_s::COLOR_CYAN:
196  _gpioR.writeValue(LinuxGPIO::Value::HIGH);
197  _gpioG.writeValue(LinuxGPIO::Value::LOW);
198  _gpioB.writeValue(LinuxGPIO::Value::LOW);
199  break;
200 
201  case led_control_s::COLOR_WHITE:
202  _gpioR.writeValue(LinuxGPIO::Value::LOW);
203  _gpioG.writeValue(LinuxGPIO::Value::LOW);
204  _gpioB.writeValue(LinuxGPIO::Value::LOW);
205  break;
206 
207  default: // led_control_s::COLOR_OFF
208  _gpioR.writeValue(LinuxGPIO::Value::HIGH);
209  _gpioG.writeValue(LinuxGPIO::Value::HIGH);
210  _gpioB.writeValue(LinuxGPIO::Value::HIGH);
211  break;
212  }
213  }
