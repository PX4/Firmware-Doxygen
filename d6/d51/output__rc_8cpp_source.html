61 int OutputRC::update(const ControlData *control_data)
62 {
63  if (control_data) {
64  //got new command
65  _retract_gimbal = control_data->gimbal_shutter_retract;
66  _set_angle_setpoints(control_data);
67  }
68 
69  _handle_position_update();
70 
71  hrt_abstime t = hrt_absolute_time();
72  _calculate_output_angles(t);
73 
74  actuator_controls_s actuator_controls;
75  actuator_controls.timestamp = hrt_absolute_time();
76  // _angle_outputs are in radians, actuator_controls are in [-1, 1]
77  actuator_controls.control[0] = (_angle_outputs[0] + _config.roll_offset) * _config.roll_scale;
78  actuator_controls.control[1] = (_angle_outputs[1] + _config.pitch_offset) * _config.pitch_scale;
79  actuator_controls.control[2] = (_angle_outputs[2] + _config.yaw_offset) * _config.yaw_scale;
80  actuator_controls.control[3] = _retract_gimbal ? _config.gimbal_retracted_mode_value : _config.gimbal_normal_mode_value;
81 
82  int instance;
83  orb_publish_auto(ORB_ID(actuator_controls_2), &_actuator_controls_pub, &actuator_controls,
84  &instance, ORB_PRIO_DEFAULT);
85 
86  _last_update = t;
87 
88  return 0;
89 }
