53 bool
54 Integrator::put(uint64_t timestamp, matrix::Vector3f &val, matrix::Vector3f &integral, uint32_t &integral_dt)
55 {
56  if (_last_integration_time == 0) {
57  /* this is the first item in the integrator */
58  _last_integration_time = timestamp;
59  _last_reset_time = timestamp;
60  _last_val = val;
61 
62  return false;
63  }
64 
65  double dt = 0.0;
66 
67  // Integrate:
68  // Leave dt at 0 if the integration time does not make sense.
69  // Without this check the integral is likely to explode.
70  if (timestamp >= _last_integration_time) {
71  dt = (double)(timestamp - _last_integration_time) / 1000000.0;
72  }
73 
74  // Use trapezoidal integration to calculate the delta integral
75  matrix::Vector3f delta_alpha = (val + _last_val) * dt * 0.5f;
76  _last_val = val;
77 
78  // Calculate coning corrections if required
79  if (_coning_comp_on) {
80  // Coning compensation derived by Paul Riseborough and Jonathan Challinger,
81  // following:
82  // Tian et al (2010) Three-loop Integration of GPS and Strapdown INS with Coning and Sculling Compensation
83  // Sourced: http://www.sage.unsw.edu.au/snap/publications/tian_etal2010b.pdf
84  // Simulated: https://github.com/priseborough/InertialNav/blob/master/models/imu_error_modelling.m
85  _beta += ((_last_alpha + _last_delta_alpha * (1.0f / 6.0f)) % delta_alpha) * 0.5f;
86  _last_delta_alpha = delta_alpha;
87  _last_alpha = _alpha;
88  }
89 
90  // accumulate delta integrals
91  _alpha += delta_alpha;
92 
93  _last_integration_time = timestamp;
94 
95  // Only do auto reset if auto reset interval is not 0.
96  if (_auto_reset_interval > 0 && (timestamp - _last_reset_time) >= _auto_reset_interval) {
97 
98  // apply coning corrections if required
99  if (_coning_comp_on) {
100  integral = _alpha + _beta;
101 
102  } else {
103  integral = _alpha;
104  }
105 
106  // reset the integrals and coning corrections
107  _reset(integral_dt);
108 
109  return true;
110 
111  } else {
112  return false;
113  }
114 }
