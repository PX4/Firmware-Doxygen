91 __EXPORT void stm32_spi1select(FAR struct spi_dev_s *dev, uint32_t devid, bool selected)
92 {
93  /* SPI select is active low, so write !selected to select the device */
94 
95  switch (devid) {
96 
97  /* intended fallthrough */
98  case PX4_SPIDEV_ICM_20602:
99 
100  /* intended fallthrough */
101  case PX4_SPIDEV_ICM_20608:
102  /* Making sure the other peripherals are not selected */
103  px4_arch_gpiowrite(GPIO_SPI_CS_ICM_2060X, !selected);
104  px4_arch_gpiowrite(GPIO_SPI_CS_BARO, 1);
105  px4_arch_gpiowrite(GPIO_SPI_CS_MPU, 1);
106  break;
107 
108  case PX4_SPIDEV_BARO:
109  /* Making sure the other peripherals are not selected */
110  px4_arch_gpiowrite(GPIO_SPI_CS_ICM_2060X, 1);
111  px4_arch_gpiowrite(GPIO_SPI_CS_BARO, !selected);
112  px4_arch_gpiowrite(GPIO_SPI_CS_MPU, 1);
113  break;
114 
115  case PX4_SPIDEV_MPU:
116  /* Making sure the other peripherals are not selected */
117  px4_arch_gpiowrite(GPIO_SPI_CS_ICM_2060X, 1);
118  px4_arch_gpiowrite(GPIO_SPI_CS_BARO, 1);
119  px4_arch_gpiowrite(GPIO_SPI_CS_MPU, !selected);
120  break;
121 
122  default:
123  break;
124  }
125 }

148 __EXPORT void board_spi_reset(int ms)
149 {
150  /* disable SPI bus */
151  px4_arch_configgpio(GPIO_SPI_CS_ICM_2060X_OFF);
152  px4_arch_configgpio(GPIO_SPI_CS_BARO_OFF);
153  px4_arch_configgpio(GPIO_SPI_CS_MPU_OFF);
154 
155  px4_arch_gpiowrite(GPIO_SPI_CS_ICM_2060X_OFF, 0);
156  px4_arch_gpiowrite(GPIO_SPI_CS_BARO_OFF, 0);
157  px4_arch_gpiowrite(GPIO_SPI_CS_MPU_OFF, 0);
158 
159  px4_arch_configgpio(GPIO_SPI1_SCK_OFF);
160  px4_arch_configgpio(GPIO_SPI1_MISO_OFF);
161  px4_arch_configgpio(GPIO_SPI1_MOSI_OFF);
162 
163  px4_arch_gpiowrite(GPIO_SPI1_SCK_OFF, 0);
164  px4_arch_gpiowrite(GPIO_SPI1_MISO_OFF, 0);
165  px4_arch_gpiowrite(GPIO_SPI1_MOSI_OFF, 0);
166 
167  px4_arch_configgpio(GPIO_EXTI_ICM_2060X_DRDY_OFF);
168  px4_arch_configgpio(GPIO_EXTI_MPU_DRDY_OFF);
169 
170  px4_arch_gpiowrite(GPIO_EXTI_ICM_2060X_DRDY_OFF, 0);
171  px4_arch_gpiowrite(GPIO_EXTI_MPU_DRDY_OFF, 0);
172 
173  /* set the sensor rail off */
174  px4_arch_configgpio(GPIO_VDD_3V3_SENSORS_EN);
175  px4_arch_gpiowrite(GPIO_VDD_3V3_SENSORS_EN, 0);
176 
177  /* wait for the sensor rail to reach GND */
178  usleep(ms * 1000);
179  warnx("reset done, %d ms", ms);
180 
181  /* re-enable power */
182 
183  /* switch the sensor rail back on */
184  px4_arch_gpiowrite(GPIO_VDD_3V3_SENSORS_EN, 1);
185 
186  /* wait a bit before starting SPI, different times didn't influence results */
187  usleep(100);
188 
189  /* reconfigure the SPI pins */
190 #ifdef CONFIG_STM32_SPI1
191  px4_arch_configgpio(GPIO_SPI_CS_ICM_2060X);
192  px4_arch_configgpio(GPIO_SPI_CS_BARO);
193  px4_arch_configgpio(GPIO_SPI_CS_MPU);
194 
195  px4_arch_configgpio(GPIO_SPI1_SCK);
196  px4_arch_configgpio(GPIO_SPI1_MISO);
197  px4_arch_configgpio(GPIO_SPI1_MOSI);
198 
199  // // XXX bring up the EXTI pins again
200  // px4_arch_configgpio(GPIO_EXTI_MPU_DRDY);
201  // px4_arch_configgpio(GPIO_EXTI_ICM_2060X_DRDY);
202 
203 #endif
204 }
