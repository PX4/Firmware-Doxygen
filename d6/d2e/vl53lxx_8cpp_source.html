246 int
247 VL53LXX::sensorInit()
248 {
249  uint8_t val = 0;
250  int ret = OK;
251  float rate_limit;
252  uint8_t rate_limit_split[2];
253 
254  // I2C at 2.8V on sensor side of level shifter
255  ret |= readRegister(VHV_CONFIG_PAD_SCL_SDA_EXTSUP_HW_REG, val);
256 
257  if (ret != OK) {
258  ret = PX4_ERROR;
259  return ret;
260  }
261 
262  ret |= writeRegister(VHV_CONFIG_PAD_SCL_SDA_EXTSUP_HW_REG, val | 0x01);
263 
264  // set I2C to standard mode
265  ret |= writeRegister(0x88, 0x00);
266 
267  ret |= writeRegister(0x80, 0x01);
268  ret |= writeRegister(0xFF, 0x01);
269  ret |= writeRegister(0x00, 0x00);
270  ret |= readRegister(0x91, val);
271  ret |= writeRegister(0x00, 0x01);
272  ret |= writeRegister(0xFF, 0x00);
273  ret |= writeRegister(0x80, 0x00);
274 
275  if (ret != OK) {
276  ret = PX4_ERROR;
277  return ret;
278  }
279 
280  _stop_variable = val;
281 
282  // disable SIGNAL_RATE_MSRC (bit 1) and SIGNAL_RATE_PRE_RANGE (bit 4) limit checks
283  readRegister(MSRC_CONFIG_CONTROL_REG, val);
284  writeRegister(MSRC_CONFIG_CONTROL_REG, val | 0x12);
285 
286  // Set signal rate limit to 0.1
287  rate_limit = 0.1 * 65536;
288  rate_limit_split[0] = (((uint16_t)rate_limit) >> 8);
289  rate_limit_split[1] = (uint16_t)rate_limit;
290 
291  writeRegisterMulti(FINAL_RANGE_CONFIG_MIN_COUNT_RATE_RTN_LIMIT_REG, &rate_limit_split[0], 2);
292  writeRegister(SYSTEM_SEQUENCE_CONFIG_REG, 0xFF);
293 
294  spadCalculations();
295 
296  return ret;

341 int
342 VL53LXX::ioctl(device::file_t *filp, int cmd, unsigned long arg)
343 {
344  switch (cmd) {
345 
346  case SENSORIOCSPOLLRATE: {
347  switch (arg) {
348 
349  case 0:
350  return -EINVAL;
351 
352  case SENSOR_POLLRATE_DEFAULT: {
353  /* do we need to start internal polling? */
354  bool want_start = (_measure_ticks == 0);
355 
356  /* set interval for next measurement to minimum legal value */
357  _measure_ticks = USEC2TICK(VL53LXX_SAMPLE_RATE);
358 
359  /* if we need to start the poll state machine, do it */
360  if (want_start) {
361  start();
362  }
363 
364  start();
365 
366  return OK;
367  }
368 
369  case SENSOR_POLLRATE_MANUAL: {
370 
371  stop();
372  _measure_ticks = 0;
373  return OK;
374  }
375 
376  /* adjust to a legal polling interval in Hz */
377  default: {
378  /* do we need to start internal polling? */
379  bool want_start = (_measure_ticks == 0);
380 
381  /* convert hz to tick interval via microseconds */
382  unsigned ticks = USEC2TICK(1000000 / arg);
383 
384  /* update interval for next measurement */
385  _measure_ticks = ticks;
386 
387  /* if we need to start the poll state machine, do it */
388  if (want_start) {
389  start();
390  }
391 
392  return OK;
393  }
394  }
395  }
396 
397  default:
398  /* give it to the superclass */
399  return I2C::ioctl(filp, cmd, arg);
400  }

404 ssize_t
405 VL53LXX::read(device::file_t *filp, char *buffer, size_t buflen)
406 {
407  unsigned count = buflen / sizeof(struct distance_sensor_s);
408  struct distance_sensor_s *rbuf = reinterpret_cast<struct distance_sensor_s *>(buffer);
409  int ret = 0;
410 
411  /* buffer must be large enough */
412  if (count < 1) {
413  return -ENOSPC;
414  }
415 
416  /* if automatic measurement is enabled */
417  if (_measure_ticks > 0) {
418 
419  /*
420  * While there is space in the caller's buffer, and reports, copy them.
421  * Note that we may be pre-empted by the workq thread while we are doing this;
422  * we are careful to avoid racing with them.
423  */
424  while (count--) {
425  if (_reports->get(rbuf)) {
426  ret += sizeof(*rbuf);
427  rbuf++;
428  }
429  }
430 
431  /* if there was no data, warn the caller */
432  return ret ? ret : -EAGAIN;
433  }
434 
435  /* manual measurement - run one conversion */
436  do {
437  _reports->flush();
438 
439  /* trigger a measurement */
440  if (OK != measure()) {
441  ret = -EIO;
442  break;
443  }
444 
445  while (!_collect_phase);
446 
447  /* run the collection phase */
448  if (OK != collect()) {
449  ret = -EIO;
450  break;
451  }
452 
453  /* state machine will have generated a report, copy it out */
454  if (_reports->get(rbuf)) {
455  ret = sizeof(*rbuf);
456  }
457 
458  } while (0);
459 
460  return ret;
461 }

566 int
567 VL53LXX::measure()
568 {
569  int ret = OK;
570  uint8_t wait_for_measurement;
571  uint8_t system_start;
572 
573  /*
574  * Send the command to begin a measurement.
575  */
576  const uint8_t cmd = RESULT_RANGE_STATUS_REG + 10;
577 
578  if (_new_measurement) {
579 
580  _new_measurement = false;
581 
582  writeRegister(0x80, 0x01);
583  writeRegister(0xFF, 0x01);
584  writeRegister(0x00, 0x00);
585  writeRegister(0x91, _stop_variable);
586  writeRegister(0x00, 0x01);
587  writeRegister(0xFF, 0x00);
588  writeRegister(0x80, 0x00);
589 
590  writeRegister(SYSRANGE_START_REG, 0x01);
591 
592  readRegister(SYSRANGE_START_REG, system_start);
593 
594  if ((system_start & 0x01) == 1) {
595  work_queue(LPWORK, &_work, (worker_t)&VL53LXX::cycle_trampoline, this,
596  USEC2TICK(VL53LXX_US)); // reschedule every 1 ms until measurement is ready
597  ret = OK;
598  return ret;
599 
600  } else {
601  _measurement_started = true;
602  }
603 
604  }
605 
606  if (!_collect_phase && !_measurement_started) {
607 
608  readRegister(SYSRANGE_START_REG, system_start);
609 
610  if ((system_start & 0x01) == 1) {
611  work_queue(LPWORK, &_work, (worker_t)&VL53LXX::cycle_trampoline, this,
612  USEC2TICK(VL53LXX_US)); // reschedule every 1 ms until measurement is ready
613  ret = OK;
614  return ret;
615 
616  } else {
617  _measurement_started = true;
618  }
619  }
620 
621  readRegister(RESULT_INTERRUPT_STATUS_REG, wait_for_measurement);
622 
623  if ((wait_for_measurement & 0x07) == 0) {
624  work_queue(LPWORK, &_work, (worker_t)&VL53LXX::cycle_trampoline, this,
625  USEC2TICK(VL53LXX_US)); // reschedule every 1 ms until measurement is ready
626  ret = OK;
627  return ret;
628  }
629 
630  _collect_phase = true;
631 
632  ret = transfer(&cmd, sizeof(cmd), nullptr, 0);
633 
634  if (OK != ret) {
635  perf_count(_comms_errors);
636  DEVICE_LOG("i2c::transfer returned %d", ret);
637  return ret;
638  }
639 
640  return ret;
641 }

644 int
645 VL53LXX::collect()
646 {
647  int ret = -EIO;
648 
649  /* read from the sensor */
650  uint8_t val[2] = {0, 0};
651 
652  perf_begin(_sample_perf);
653 
654  _collect_phase = false;
655 
656  ret = transfer(nullptr, 0, &val[0], 2);
657 
658  if (ret < 0) {
659  DEVICE_LOG("error reading from sensor: %d", ret);
660  perf_count(_comms_errors);
661  perf_end(_sample_perf);
662  return ret;
663  }
664 
665  uint16_t distance_mm = (val[0] << 8) | val[1];
666  float distance_m = float(distance_mm) * 1e-3f;
667  struct distance_sensor_s report;
668 
669  report.timestamp = hrt_absolute_time();
670  report.type = distance_sensor_s::MAV_DISTANCE_SENSOR_LASER;
671  report.orientation = _rotation;
672 
673  report.current_distance = distance_m;
674 
675  report.min_distance = VL53LXX_MIN_RANGING_DISTANCE;
676  report.max_distance = VL53LXX_MAX_RANGING_DISTANCE;
677  report.covariance = 0.0f;
678  report.signal_quality = -1;
679 
680  /* TODO: set proper ID */
681  report.id = 0;
682 
683  /* publish it, if we are the primary */
684  if (_class_instance == CLASS_DEVICE_PRIMARY) {
685  orb_publish_auto(ORB_ID(distance_sensor), &_distance_sensor_topic, &report, &_orb_class_instance,
686  ORB_PRIO_DEFAULT);
687  }
688 
689  _reports->force(&report);
690 
691  /* notify anyone waiting for data */
692  poll_notify(POLLIN);
693 
694  ret = OK;
695 
696  perf_end(_sample_perf);
697 
698  return ret;
699 }
