263 void
264 FixedwingAttitudeControl::vehicle_manual_poll()
265 {
266  // only update manual if in a manual mode
267  if (_vcontrol_mode.flag_control_manual_enabled) {
268 
269  // Always copy the new manual setpoint, even if it wasn't updated, to fill the _actuators with valid values
270  if (orb_copy(ORB_ID(manual_control_setpoint), _manual_sub, &_manual) == PX4_OK) {
271 
272  // Check if we are in rattitude mode and the pilot is above the threshold on pitch
273  if (_vcontrol_mode.flag_control_rattitude_enabled) {
274  if (fabsf(_manual.y) > _parameters.rattitude_thres || fabsf(_manual.x) > _parameters.rattitude_thres) {
275  _vcontrol_mode.flag_control_attitude_enabled = false;
276  }
277  }
278 
279  if (!_vcontrol_mode.flag_control_climb_rate_enabled &&
280  !_vcontrol_mode.flag_control_offboard_enabled) {
281 
282  if (_vcontrol_mode.flag_control_attitude_enabled) {
283  // STABILIZED mode generate the attitude setpoint from manual user inputs
284  _att_sp.timestamp = hrt_absolute_time();
285  _att_sp.roll_body = _manual.y * _parameters.man_roll_max + _parameters.rollsp_offset_rad;
286  _att_sp.roll_body = math::constrain(_att_sp.roll_body, -_parameters.man_roll_max, _parameters.man_roll_max);
287  _att_sp.pitch_body = -_manual.x * _parameters.man_pitch_max + _parameters.pitchsp_offset_rad;
288  _att_sp.pitch_body = math::constrain(_att_sp.pitch_body, -_parameters.man_pitch_max, _parameters.man_pitch_max);
289  _att_sp.yaw_body = 0.0f;
290  _att_sp.thrust = _manual.z;
291 
292  Quatf q(Eulerf(_att_sp.roll_body, _att_sp.pitch_body, _att_sp.yaw_body));
293  q.copyTo(_att_sp.q_d);
294  _att_sp.q_d_valid = true;
295 
296  if (_attitude_sp_pub != nullptr) {
297  /* publish the attitude rates setpoint */
298  orb_publish(_attitude_setpoint_id, _attitude_sp_pub, &_att_sp);
299 
300  } else if (_attitude_setpoint_id) {
301  /* advertise the attitude rates setpoint */
302  _attitude_sp_pub = orb_advertise(_attitude_setpoint_id, &_att_sp);
303  }
304 
305  } else if (_vcontrol_mode.flag_control_rates_enabled &&
306  !_vcontrol_mode.flag_control_attitude_enabled) {
307 
308  // RATE mode we need to generate the rate setpoint from manual user inputs
309  _rates_sp.timestamp = hrt_absolute_time();
310  _rates_sp.roll = _manual.y * _parameters.acro_max_x_rate_rad;
311  _rates_sp.pitch = -_manual.x * _parameters.acro_max_y_rate_rad;
312  _rates_sp.yaw = _manual.r * _parameters.acro_max_z_rate_rad;
313  _rates_sp.thrust = _manual.z;
314 
315  if (_rate_sp_pub != nullptr) {
316  /* publish the attitude rates setpoint */
317  orb_publish(_rates_sp_id, _rate_sp_pub, &_rates_sp);
318 
319  } else if (_rates_sp_id) {
320  /* advertise the attitude rates setpoint */
321  _rate_sp_pub = orb_advertise(_rates_sp_id, &_rates_sp);
322  }
323 
324  } else {
325  /* manual/direct control */
326  _actuators.control[actuator_controls_s::INDEX_ROLL] = _manual.y * _parameters.man_roll_scale + _parameters.trim_roll;
327  _actuators.control[actuator_controls_s::INDEX_PITCH] = -_manual.x * _parameters.man_pitch_scale +
328  _parameters.trim_pitch;
329  _actuators.control[actuator_controls_s::INDEX_YAW] = _manual.r * _parameters.man_yaw_scale + _parameters.trim_yaw;
330  _actuators.control[actuator_controls_s::INDEX_THROTTLE] = _manual.z;
331  }
332  }
333  }
334  }

407 void FixedwingAttitudeControl::run()
408 {
409  /*
410  * do subscriptions
411  */
412  _att_sub = orb_subscribe(ORB_ID(vehicle_attitude));
413  _att_sp_sub = orb_subscribe(ORB_ID(vehicle_attitude_setpoint));
414  _vcontrol_mode_sub = orb_subscribe(ORB_ID(vehicle_control_mode));
415  _params_sub = orb_subscribe(ORB_ID(parameter_update));
416  _manual_sub = orb_subscribe(ORB_ID(manual_control_setpoint));
417  _global_pos_sub = orb_subscribe(ORB_ID(vehicle_global_position));
418  _vehicle_status_sub = orb_subscribe(ORB_ID(vehicle_status));
419  _vehicle_land_detected_sub = orb_subscribe(ORB_ID(vehicle_land_detected));
420  _battery_status_sub = orb_subscribe(ORB_ID(battery_status));
421 
422  parameters_update();
423 
424  /* get an initial update for all sensor and status data */
425  vehicle_setpoint_poll();
426  vehicle_control_mode_poll();
427  vehicle_manual_poll();
428  vehicle_status_poll();
429  vehicle_land_detected_poll();
430 
431  /* wakeup source */
432  px4_pollfd_struct_t fds[1];
433 
434  /* Setup of loop */
435  fds[0].fd = _att_sub;
436  fds[0].events = POLLIN;
437 
438  while (!should_exit()) {
439 
440  /* only update parameters if they changed */
441  bool params_updated = false;
442  orb_check(_params_sub, &params_updated);
443 
444  if (params_updated) {
445  /* read from param to clear updated flag */
446  parameter_update_s update;
447  orb_copy(ORB_ID(parameter_update), _params_sub, &update);
448 
449  /* update parameters from storage */
450  parameters_update();
451  }
452 
453  /* wait for up to 500ms for data */
454  int pret = px4_poll(&fds[0], (sizeof(fds) / sizeof(fds[0])), 100);
455 
456  /* timed out - periodic check for _task_should_exit, etc. */
457  if (pret == 0) {
458  continue;
459  }
460 
461  /* this is undesirable but not much we can do - might want to flag unhappy status */
462  if (pret < 0) {
463  PX4_WARN("poll error %d, %d", pret, errno);
464  continue;
465  }
466 
467  perf_begin(_loop_perf);
468 
469  /* only run controller if attitude changed */
470  if (fds[0].revents & POLLIN) {
471  static uint64_t last_run = 0;
472  float deltaT = (hrt_absolute_time() - last_run) / 1000000.0f;
473  last_run = hrt_absolute_time();
474 
475  /* guard against too large deltaT's */
476  if (deltaT > 1.0f) {
477  deltaT = 0.01f;
478  }

480  /* load local copies */

869 void FixedwingAttitudeControl::control_flaps(const float dt)
870 {
871  /* default flaps to center */
872  float flap_control = 0.0f;
873 
874  /* map flaps by default to manual if valid */
875  if (PX4_ISFINITE(_manual.flaps) && _vcontrol_mode.flag_control_manual_enabled
876  && fabsf(_parameters.flaps_scale) > 0.01f) {
877  flap_control = 0.5f * (_manual.flaps + 1.0f) * _parameters.flaps_scale;
878 
879  } else if (_vcontrol_mode.flag_control_auto_enabled
880  && fabsf(_parameters.flaps_scale) > 0.01f) {
881  switch (_att_sp.apply_flaps) {
882  case vehicle_attitude_setpoint_s::FLAPS_OFF : flap_control = 0.0f; // no flaps
883  break;
884 
885  case vehicle_attitude_setpoint_s::FLAPS_LAND : flap_control = 1.0f * _parameters.flaps_scale; // landing flaps
886  break;
887 
888  case vehicle_attitude_setpoint_s::FLAPS_TAKEOFF : flap_control = 1.0f * _parameters.flaps_scale *
889  _parameters.flaps_takeoff_scale; // take-off flaps
890  break;
891  }
892  }
893 
894  // move the actual control value continuous with time, full flap travel in 1sec
895  if (fabsf(_flaps_applied - flap_control) > 0.01f) {
896  _flaps_applied += (_flaps_applied - flap_control) < 0 ? dt : -dt;
897 
898  } else {
899  _flaps_applied = flap_control;
900  }
901 
902  /* default flaperon to center */
903  float flaperon_control = 0.0f;
904 
905  /* map flaperons by default to manual if valid */
906  if (PX4_ISFINITE(_manual.aux2) && _vcontrol_mode.flag_control_manual_enabled
907  && fabsf(_parameters.flaperon_scale) > 0.01f) {
908  flaperon_control = 0.5f * (_manual.aux2 + 1.0f) * _parameters.flaperon_scale;
909 
910  } else if (_vcontrol_mode.flag_control_auto_enabled
911  && fabsf(_parameters.flaperon_scale) > 0.01f) {
912  flaperon_control = (_att_sp.apply_flaps == vehicle_attitude_setpoint_s::FLAPS_LAND) ? 1.0f *
913  _parameters.flaperon_scale : 0.0f;
914  }
915 
916  // move the actual control value continuous with time, full flap travel in 1sec
917  if (fabsf(_flaperons_applied - flaperon_control) > 0.01f) {
918  _flaperons_applied += (_flaperons_applied - flaperon_control) < 0 ? dt : -dt;
919 
920  } else {
921  _flaperons_applied = flaperon_control;
922  }
923 }
