95 bool FixedwingLandDetector::_get_landed_state()
96 {
97  // only trigger flight conditions if we are armed
98  if (!_arming.armed) {
99  return true;
100  }
101 
102  bool landDetected = false;
103 
104  if (hrt_elapsed_time(&_local_pos.timestamp) < 500 * 1000) {
105 
106  // horizontal velocity
107  float val = 0.97f * _velocity_xy_filtered + 0.03f * sqrtf(_local_pos.vx * _local_pos.vx + _local_pos.vy *
108  _local_pos.vy);
109 
110  if (PX4_ISFINITE(val)) {
111  _velocity_xy_filtered = val;
112  }
113 
114  // vertical velocity
115  val = 0.99f * _velocity_z_filtered + 0.01f * fabsf(_local_pos.vz);
116 
117  if (PX4_ISFINITE(val)) {
118  _velocity_z_filtered = val;
119  }
120 
121  _airspeed_filtered = 0.95f * _airspeed_filtered + 0.05f * _airspeed.true_airspeed_m_s;
122 
123  // a leaking lowpass prevents biases from building up, but
124  // gives a mostly correct response for short impulses
125  const float acc_hor = sqrtf(_sensors.accel_x * _sensors.accel_x +
126  _sensors.accel_y * _sensors.accel_y);
127  _accel_horz_lp = _accel_horz_lp * 0.8f + acc_hor * 0.18f;
128 
129  // crude land detector for fixedwing
130  landDetected = _velocity_xy_filtered < _params.maxVelocity
131  && _velocity_z_filtered < _params.maxClimbRate
132  && _airspeed_filtered < _params.maxAirSpeed
133  && _accel_horz_lp < _params.maxXYAccel;
134 
135  } else {
136  // Control state topic has timed out and we need to assume we're landed.
137  landDetected = true;
138  }
139 
140  return landDetected;
141 }
