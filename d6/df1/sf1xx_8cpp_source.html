247  break;

249  case 3: /* SF10/c (100m 16Hz) */

343 int
344 SF1XX::ioctl(device::file_t *filp, int cmd, unsigned long arg)
345 {
346  switch (cmd) {
347 
348  case SENSORIOCSPOLLRATE: {
349  switch (arg) {
350 
351  /* switching to manual polling */
352  case SENSOR_POLLRATE_MANUAL:
353  stop();
354  _measure_ticks = 0;
355  return OK;
356 
357  /* external signalling (DRDY) not supported */
358  case SENSOR_POLLRATE_EXTERNAL:
359 
360  /* zero would be bad */
361  case 0:
362  return -EINVAL;
363 
364  /* set default/max polling rate */
365  case SENSOR_POLLRATE_MAX:
366  case SENSOR_POLLRATE_DEFAULT: {
367  /* do we need to start internal polling? */
368  bool want_start = (_measure_ticks == 0);
369 
370  /* set interval for next measurement to minimum legal value */
371  _measure_ticks = USEC2TICK(_conversion_interval);
372 
373  /* if we need to start the poll state machine, do it */
374  if (want_start) {
375  start();
376 
377  }
378 
379  return OK;
380  }
381 
382  /* adjust to a legal polling interval in Hz */
383  default: {
384  /* do we need to start internal polling? */
385  bool want_start = (_measure_ticks == 0);
386 
387  /* convert hz to tick interval via microseconds */
388  int ticks = USEC2TICK(1000000 / arg);
389 
390  /* check against maximum rate */
391  if (ticks < USEC2TICK(_conversion_interval)) {
392  return -EINVAL;
393  }
394 
395  /* update interval for next measurement */
396  _measure_ticks = ticks;
397 
398  /* if we need to start the poll state machine, do it */
399  if (want_start) {
400  start();
401  }
402 
403  return OK;
404  }
405  }
406  }
407 
408  case SENSORIOCGPOLLRATE:
409  if (_measure_ticks == 0) {
410  return SENSOR_POLLRATE_MANUAL;
411  }
412 
413  return (1000 / _measure_ticks);
414 
415  case SENSORIOCSQUEUEDEPTH: {
416  /* lower bound is mandatory, upper bound is a sanity check */
417  if ((arg < 1) || (arg > 100)) {
418  return -EINVAL;
419  }
420 
421  ATOMIC_ENTER;
422 
423  if (!_reports->resize(arg)) {
424  ATOMIC_LEAVE;
425  return -ENOMEM;
426  }
427 
428  ATOMIC_LEAVE;
429 
430  return OK;
431  }
432 
433  case SENSORIOCRESET:
434  /* XXX implement this */
435  return -EINVAL;
436 
437  default:
438  /* give it to the superclass */
439  return I2C::ioctl(filp, cmd, arg);
440  }
441 }

443 ssize_t
444 SF1XX::read(device::file_t *filp, char *buffer, size_t buflen)
445 {
446  unsigned count = buflen / sizeof(struct distance_sensor_s);
447  struct distance_sensor_s *rbuf = reinterpret_cast<struct distance_sensor_s *>(buffer);
448  int ret = 0;
449 
450  /* buffer must be large enough */
451  if (count < 1) {
452  return -ENOSPC;
453  }
454 
455  /* if automatic measurement is enabled */
456  if (_measure_ticks > 0) {
457 
458  /*
459  * While there is space in the caller's buffer, and reports, copy them.
460  * Note that we may be pre-empted by the workq thread while we are doing this;
461  * we are careful to avoid racing with them.
462  */
463  while (count--) {
464  if (_reports->get(rbuf)) {
465  ret += sizeof(*rbuf);
466  rbuf++;
467  }
468  }
469 
470  /* if there was no data, warn the caller */
471  return ret ? ret : -EAGAIN;
472  }
473 
474  /* manual measurement - run one conversion */
475  do {
476  _reports->flush();
477 
478  /* trigger a measurement */
479  if (OK != measure()) {
480  ret = -EIO;
481  break;
482  }
483 
484  /* wait for it to complete */
485  usleep(_conversion_interval);
486 
487  /* run the collection phase */
488  if (OK != collect()) {
489  ret = -EIO;
490  break;
491  }
492 
493  /* state machine will have generated a report, copy it out */
494  if (_reports->get(rbuf)) {
495  ret = sizeof(*rbuf);
496  }
497 
498  } while (0);
499 
500  return ret;
501 }

526 int
527 SF1XX::collect()
528 {
529  int ret = -EIO;
530 
531  /* read from the sensor */
532  uint8_t val[2] = {0, 0};
533  perf_begin(_sample_perf);
534 
535  ret = transfer(nullptr, 0, &val[0], 2);
536 
537  if (ret < 0) {
538  DEVICE_DEBUG("error reading from sensor: %d", ret);
539  perf_count(_comms_errors);
540  perf_end(_sample_perf);
541  return ret;
542  }
543 
544  uint16_t distance_cm = val[0] << 8 | val[1];
545  float distance_m = float(distance_cm) * 1e-2f;
546 
547  struct distance_sensor_s report;
548  report.timestamp = hrt_absolute_time();
549  report.type = distance_sensor_s::MAV_DISTANCE_SENSOR_LASER;
550  report.orientation = _rotation;
551  report.current_distance = distance_m;
552  report.min_distance = get_minimum_distance();
553  report.max_distance = get_maximum_distance();
554  report.covariance = 0.0f;
555  report.signal_quality = -1;
556  /* TODO: set proper ID */
557  report.id = 0;
558 
559  /* publish it, if we are the primary */
560  if (_distance_sensor_topic != nullptr) {
561  orb_publish(ORB_ID(distance_sensor), _distance_sensor_topic, &report);
562  }
563 
564  _reports->force(&report);
565 
566  /* notify anyone waiting for data */
567  poll_notify(POLLIN);
568 
569  ret = OK;
570 
571  perf_end(_sample_perf);
572  return ret;
573 }

742 /**
743  * Perform some basic functional tests on the driver;
744  * make sure we can collect data from the sensor in polled
745  * and automatic modes.
746  */
747 int
748 test()
749 {
750  struct distance_sensor_s report;
751  ssize_t sz;
752  int ret;
753 
754  int fd = px4_open(SF1XX_DEVICE_PATH, O_RDONLY);
755 
756  if (fd < 0) {
757  PX4_ERR("%s open failed (try 'sf1xx start' if the driver is not running)", SF1XX_DEVICE_PATH);
758  return PX4_ERROR;
759  }
760 
761  /* do a simple demand read */
762  sz = read(fd, &report, sizeof(report));
763 
764  if (sz != sizeof(report)) {
765  PX4_ERR("immediate read failed");
766  return PX4_ERROR;
767  }
768 
769  print_message(report);
770 
771  /* start the sensor polling at 2Hz */
772  if (OK != ioctl(fd, SENSORIOCSPOLLRATE, 2)) {
773  PX4_ERR("failed to set 2Hz poll rate");
774  return PX4_ERROR;
775  }
776 
777  /* read the sensor 5x and report each value */
778  for (unsigned i = 0; i < 5; i++) {
779  struct pollfd fds;
780 
781  /* wait for data to be ready */
782  fds.fd = fd;
783  fds.events = POLLIN;
784  ret = poll(&fds, 1, 2000);
785 
786  if (ret != 1) {
787  PX4_ERR("timed out waiting for sensor data");
788  return PX4_ERROR;
789  }
790 
791  /* now go get it */
792  sz = read(fd, &report, sizeof(report));
793 
794  if (sz != sizeof(report)) {
795  PX4_ERR("periodic read failed");
796  return PX4_ERROR;
797  }
798 
799  print_message(report);
800  }
801 
802  /* reset the sensor polling to default rate */
803  if (OK != ioctl(fd, SENSORIOCSPOLLRATE, SENSOR_POLLRATE_DEFAULT)) {
804  PX4_ERR("failed to set default poll rate");
805  return PX4_ERROR;
806  }
807 
808  px4_close(fd);
809 
810  PX4_INFO("PASS");
811  return PX4_OK;
812 }
