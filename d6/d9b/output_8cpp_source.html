108 float OutputBase::_calculate_pitch(double lon, double lat, float altitude,
109  const vehicle_global_position_s &global_position)
110 {
111  if (!map_projection_initialized(&_projection_reference)) {
112  map_projection_init(&_projection_reference, global_position.lat, global_position.lon);
113  }
114 
115  float x1, y1, x2, y2;
116  map_projection_project(&_projection_reference, lat, lon, &x1, &y1);
117  map_projection_project(&_projection_reference, global_position.lat, global_position.lon, &x2, &y2);
118  float dx = x1 - x2, dy = y1 - y2;
119  float target_distance = sqrtf(dx * dx + dy * dy);
120  float z = altitude - global_position.alt;
121 
122  return atan2f(z, target_distance);
123 }

125 void OutputBase::_set_angle_setpoints(const ControlData *control_data)
126 {
127  _cur_control_data = control_data;
128 
129  for (int i = 0; i < 3; ++i) {
130  _stabilize[i] = control_data->stabilize_axis[i];
131  _angle_speeds[i] = 0.f;

133 
134  switch (control_data->type) {

136  for (int i = 0; i < 3; ++i) {

141  _angle_setpoints[i] = control_data->type_data.angle.angles[i];
142  }
143  }
144 
145  break;
146 
147  case ControlData::Type::LonLat:
148  _handle_position_update(true);
149  break;
150 
151  case ControlData::Type::Neutral:
152  _angle_setpoints[0] = 0.f;
153  _angle_setpoints[1] = 0.f;
154  _angle_setpoints[2] = 0.f;

159 void OutputBase::_handle_position_update(bool force_update)
160 {
161  bool need_update = force_update;
162 
163  if (!_cur_control_data || _cur_control_data->type != ControlData::Type::LonLat) {
164  return;
165  }
166 
167  if (!force_update) {
168  orb_check(_vehicle_global_position_sub, &need_update);
169  }
170 
171  if (!need_update) {
172  return;
173  }
174 
175  vehicle_global_position_s vehicle_global_position;
176  orb_copy(ORB_ID(vehicle_global_position), _vehicle_global_position_sub, &vehicle_global_position);
177 
178  float pitch;
179  const double &lon = _cur_control_data->type_data.lonlat.lon;
180  const double &lat = _cur_control_data->type_data.lonlat.lat;
181  const float &alt = _cur_control_data->type_data.lonlat.altitude;
182 
183  if (_cur_control_data->type_data.lonlat.pitch_fixed_angle >= -M_PI_F) {
184  pitch = _cur_control_data->type_data.lonlat.pitch_fixed_angle;
185 
186  } else {
187  pitch = _calculate_pitch(lon, lat, alt, vehicle_global_position);
188  }
189 
190  float roll = _cur_control_data->type_data.lonlat.roll_angle;
191  float yaw = get_bearing_to_next_waypoint(vehicle_global_position.lat, vehicle_global_position.lon, lat, lon)
192  - vehicle_global_position.yaw;
193 
194  _angle_setpoints[0] = roll;
195  _angle_setpoints[1] = pitch + _cur_control_data->type_data.lonlat.pitch_angle_offset;
196  _angle_setpoints[2] = yaw + _cur_control_data->type_data.lonlat.yaw_angle_offset;
197 }

199 void OutputBase::_calculate_output_angles(const hrt_abstime &t)
200 {
201  //take speed into account
202  float dt = (t - _last_update) / 1.e6f;
203 
204  for (int i = 0; i < 3; ++i) {
205  _angle_setpoints[i] += dt * _angle_speeds[i];
206  }
207 
208  //get the output angles and stabilize if necessary
209  vehicle_attitude_s vehicle_attitude;
210 
211  if (_stabilize[0] || _stabilize[1] || _stabilize[2]) {
212  orb_copy(ORB_ID(vehicle_attitude), _vehicle_attitude_sub, &vehicle_attitude);
213  }
214 
215  matrix::Eulerf euler = matrix::Quatf(vehicle_attitude.q);
216 
217  for (int i = 0; i < 3; ++i) {
218  if (_stabilize[i]) {
219  _angle_outputs[i] = _angle_setpoints[i] - euler(i);
220 
221  } else {
222  _angle_outputs[i] = _angle_setpoints[i];
223  }
224 
225  //bring angles into proper range [-pi, pi]
226  while (_angle_outputs[i] > M_PI_F) { _angle_outputs[i] -= 2.f * M_PI_F; }
227 
228  while (_angle_outputs[i] < -M_PI_F) { _angle_outputs[i] += 2.f * M_PI_F; }
229  }
230 }
