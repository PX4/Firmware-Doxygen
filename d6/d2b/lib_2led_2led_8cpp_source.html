48 int LedController::update(LedControlData &control_data)
49 {
50  bool updated = false;
51 
52  orb_check(_led_control_sub, &updated);
53 
54  while (updated || _force_update) {
55  // handle new state
56  led_control_s led_control;
57 
58  if (orb_copy(ORB_ID(led_control), _led_control_sub, &led_control) == 0) {
59 
60  // don't apply the new state just yet to avoid interrupting an ongoing blinking state
61  for (int i = 0; i < BOARD_MAX_LEDS; ++i) {
62  if (led_control.led_mask & (1 << i)) {
63  // if next state has already a higher priority state than
64  // led_control, set lower prio state directly, so that this
65  // information is not lost
66  if (_states[i].next_state.is_valid() && led_control.priority < _states[i].next_state.priority) {
67  _states[i].set(led_control);
68 
69  } else {
70  // if a lower prio event is already in next state and a
71  // higher prio event is coming in
72  if (_states[i].next_state.is_valid() && led_control.priority > _states[i].next_state.priority) {
73  _states[i].apply_next_state();
74  }
75 
76  _states[i].next_state.set(led_control);
77  }
78  }
79  }
80  }
81 
82  _force_update = false;
83 
84  orb_check(_led_control_sub, &updated);
85  }
86 
87  bool had_changes = false; // did one of the outputs change?
88 
89  // handle state updates
90  hrt_abstime now = hrt_absolute_time();
91  uint16_t blink_delta_t = (uint16_t)((now - _last_update_call) / 100); // Note: this is in 0.1ms
92  constexpr uint16_t breathe_duration = BREATHE_INTERVAL * BREATHE_STEPS / 100;
93 
94  int num_blinking_leds = 0;
95  int num_blinking_do_not_change_state = 0;
96  int current_priorities[BOARD_MAX_LEDS];
97 
98  for (int i = 0; i < BOARD_MAX_LEDS; ++i) {
99  int priority = led_control_s::MAX_PRIORITY;
100 
101  for (; priority >= 0; --priority) {
102 
103  PerPriorityData &cur_data = _states[i].priority[priority];
104 
105  if (cur_data.mode == led_control_s::MODE_DISABLED) {
106  continue; // handle next priority
107  }
108 
109  // handle state updates
110  uint16_t current_blink_duration = 0;
111 
112  switch (cur_data.mode) {
113  case led_control_s::MODE_FLASH:
114  case led_control_s::MODE_BLINK_FAST:
115  current_blink_duration = BLINK_FAST_DURATION / 100;
116  break;
117 
118  case led_control_s::MODE_BLINK_NORMAL:
119  current_blink_duration = BLINK_NORMAL_DURATION / 100;
120  break;
121 
122  case led_control_s::MODE_BLINK_SLOW:
123  current_blink_duration = BLINK_SLOW_DURATION / 100;
124  break;
125 
126  case led_control_s::MODE_BREATHE:
127  _states[i].current_blinking_time += blink_delta_t;
128 
129  while (_states[i].current_blinking_time > breathe_duration) {
130  _states[i].current_blinking_time -= breathe_duration;
131  }
132 
133  had_changes = true;
134  break;
135  }
136 
137  if (current_blink_duration > 0) {
138  ++num_blinking_leds;
139 
140  if ((_states[i].current_blinking_time += blink_delta_t) > current_blink_duration) {
141  _states[i].current_blinking_time -= current_blink_duration;
142 
143  if (cur_data.blink_times_left == 246) {
144  // handle toggling for infinite case: decrease between 255 and 246
145  // In order to handle the flash mode infinite case it needs a
146  // total of 10 steps.
147  cur_data.blink_times_left = 255;
148  ++num_blinking_do_not_change_state;
149 
150  } else if (--cur_data.blink_times_left == 0) {
151  cur_data.mode = led_control_s::MODE_DISABLED;
152  _states[i].current_blinking_time = 0;
153 
154  } else if (cur_data.blink_times_left % 2 == 1) {
155  ++num_blinking_do_not_change_state;
156  }
157 
158  had_changes = true;
159 
160  } else {
161  ++num_blinking_do_not_change_state;
162  }
163  }
164 
165  break; // handle next led
166  }
167 
168  current_priorities[i] = priority;
169 
170  }
171 
172  // handle next state:
173  // only allow a state change if no led blinks or at least one of the blinking leds signals that it's ok to switch.
174  // This makes sure all leds are kept in sync, but does not allow interrupting at arbitrary points.
175  if (num_blinking_leds == 0 || num_blinking_leds > num_blinking_do_not_change_state) {
176  for (int i = 0; i < BOARD_MAX_LEDS; ++i) {
177  if (_states[i].next_state.is_valid()) {
178  int next_priority = (int)_states[i].next_state.priority;
179 
180  if (next_priority >= current_priorities[i]) {
181  _states[i].current_blinking_time = 0;
182  had_changes = true;
183  }
184 
185  _states[i].apply_next_state();
186  _states[i].next_state.reset();
187  }
188 
189  }
190  }
191 
192  _last_update_call = now;
193 
194  if (!had_changes) {
195  return 0;
196  }
197 
198  // create output
199  get_control_data(control_data);
200 
201  return 1;
202 }

204 void LedController::get_control_data(LedControlData &control_data)
205 {
206  _breathe_enabled = false;
207 
208  for (int i = 0; i < BOARD_MAX_LEDS; ++i) {
209  control_data.leds[i].color = led_control_s::COLOR_OFF; // set output to a defined state
210  control_data.leds[i].brightness = 255;
211 
212  for (int priority = led_control_s::MAX_PRIORITY; priority >= 0; --priority) {
213  bool flash_output_active = true;
214  const PerPriorityData &cur_data = _states[i].priority[priority];
215 
216  if (cur_data.mode == led_control_s::MODE_DISABLED) {
217  continue; // handle next priority
218  }
219 
220  switch (cur_data.mode) {
221  case led_control_s::MODE_ON:
222  control_data.leds[i].color = cur_data.color;
223  break;
224 
225  case led_control_s::MODE_BREATHE: {
226  // fade on and off
227  int counter = _states[i].current_blinking_time / (BREATHE_INTERVAL / 100);
228  int n = counter >= (BREATHE_STEPS / 2) ? BREATHE_STEPS - counter : counter;
229  control_data.leds[i].brightness = (n * n) * 255 / (BREATHE_STEPS * BREATHE_STEPS / 4); // (n/(steps/2))^2
230  control_data.leds[i].color = cur_data.color;
231  _breathe_enabled = true;
232  break;
233  }
234 
235  case led_control_s::MODE_FLASH:
236  if (cur_data.blink_times_left % 10 < 6) { // 2 blinks, then turn off for the rest of the cycle
237  flash_output_active = false;
238  }
239 
240  /* FALLTHROUGH */
241  case led_control_s::MODE_BLINK_FAST:
242  case led_control_s::MODE_BLINK_NORMAL:
243  case led_control_s::MODE_BLINK_SLOW:
244  if (cur_data.blink_times_left % 2 == 0 && flash_output_active) {
245  control_data.leds[i].color = cur_data.color;
246  }
247 
248  break;
249  // MODE_OFF does not need to be handled, it's already set above
250  }
251 
252  break; // handle next led
253  }
254  }
255 }
