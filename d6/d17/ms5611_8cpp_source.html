283 int
284 MS5611::init()
285 {
286  int ret;
287  bool autodetect = false;
288 
289  ret = CDev::init();
290 
291  if (ret != OK) {
292  DEVICE_DEBUG("CDev init failed");
293  goto out;
294  }
295 
296  /* allocate basic report buffers */
297  _reports = new ringbuffer::RingBuffer(2, sizeof(sensor_baro_s));
298 
299  if (_reports == nullptr) {
300  DEVICE_DEBUG("can't get memory for reports");
301  ret = -ENOMEM;
302  goto out;
303  }
304 
305  /* register alternate interfaces if we have to */
306  _class_instance = register_class_devname(BARO_BASE_DEVICE_PATH);
307 
308  struct baro_report brp;
309  /* do a first measurement cycle to populate reports with valid data */
310  _measure_phase = 0;
311  _reports->flush();
312 
313  if (_device_type == MS56XX_DEVICE) {
314  autodetect = true;
315  /* try first with MS5611 and fallback to MS5607 */
316  _device_type = MS5611_DEVICE;
317  }
318 
319  while (true) {
320  /* do temperature first */
321  if (OK != measure()) {
322  ret = -EIO;
323  break;
324  }
325 
326  usleep(MS5611_CONVERSION_INTERVAL);
327 
328  if (OK != collect()) {
329  ret = -EIO;
330  break;
331  }
332 
333  /* now do a pressure measurement */
334  if (OK != measure()) {
335  ret = -EIO;
336  break;
337  }
338 
339  usleep(MS5611_CONVERSION_INTERVAL);
340 
341  if (OK != collect()) {
342  ret = -EIO;
343  break;
344  }
345 
346  /* state machine will have generated a report, copy it out */
347  _reports->get(&brp);
348 
349  if (autodetect) {
350  if (_device_type == MS5611_DEVICE) {
351  if (brp.pressure < 520.0f) {
352  /* This is likely not this device, try again */
353  _device_type = MS5607_DEVICE;
354  _measure_phase = 0;
355 
356  continue;
357  }
358 
359  } else if (_device_type == MS5607_DEVICE) {
360  if (brp.pressure < 520.0f) {
361  /* Both devices returned a very low pressure;
362  * have fun on Everest using MS5611 */
363  _device_type = MS5611_DEVICE;
364  }
365  }
366  }
367 
368  switch (_device_type) {
369  default:
370 
371  /* fall through */
372  case MS5611_DEVICE:
373  _device_id.devid_s.devtype = DRV_BARO_DEVTYPE_MS5611;
374  break;
375 
376  case MS5607_DEVICE:
377  _device_id.devid_s.devtype = DRV_BARO_DEVTYPE_MS5607;
378  break;
379  }
380 
381  /* ensure correct devid */
382  brp.device_id = _device_id.devid;
383 
384  ret = OK;
385 
386  _baro_topic = orb_advertise_multi(ORB_ID(sensor_baro), &brp,
387  &_orb_class_instance, _interface->external() ? ORB_PRIO_HIGH : ORB_PRIO_DEFAULT);
388 
389  if (_baro_topic == nullptr) {
390  warnx("failed to create sensor_baro publication");
391  }
392 
393  break;
394  }
395 
396 out:
397  return ret;
398 }

400 ssize_t
401 MS5611::read(struct file *filp, char *buffer, size_t buflen)
402 {
403  unsigned count = buflen / sizeof(struct baro_report);
404  struct baro_report *brp = reinterpret_cast<struct baro_report *>(buffer);
405  int ret = 0;
406 
407  /* buffer must be large enough */
408  if (count < 1) {
409  return -ENOSPC;
410  }
411 
412  /* if automatic measurement is enabled */
413  if (_measure_ticks > 0) {
414 
415  /*
416  * While there is space in the caller's buffer, and reports, copy them.
417  * Note that we may be pre-empted by the workq thread while we are doing this;
418  * we are careful to avoid racing with them.
419  */
420  while (count--) {
421  if (_reports->get(brp)) {
422  ret += sizeof(*brp);
423  brp++;
424  }
425  }
426 
427  /* if there was no data, warn the caller */
428  return ret ? ret : -EAGAIN;
429  }
430 
431  /* manual measurement - run one conversion */
432  do {
433  _measure_phase = 0;
434  _reports->flush();
435 
436  /* do temperature first */
437  if (OK != measure()) {
438  ret = -EIO;
439  break;
440  }
441 
442  usleep(MS5611_CONVERSION_INTERVAL);
443 
444  if (OK != collect()) {
445  ret = -EIO;
446  break;
447  }
448 
449  /* now do a pressure measurement */
450  if (OK != measure()) {
451  ret = -EIO;
452  break;
453  }
454 
455  usleep(MS5611_CONVERSION_INTERVAL);
456 
457  if (OK != collect()) {
458  ret = -EIO;
459  break;
460  }
461 
462  /* state machine will have generated a report, copy it out */
463  if (_reports->get(brp)) {
464  ret = sizeof(*brp);
465  }
466 
467  } while (0);
468 
469  return ret;
470 }

472 int
473 MS5611::ioctl(struct file *filp, int cmd, unsigned long arg)
474 {
475  switch (cmd) {
476 
477  case SENSORIOCSPOLLRATE: {
478  switch (arg) {
479 
480  /* switching to manual polling */
481  case SENSOR_POLLRATE_MANUAL:
482  stop_cycle();
483  _measure_ticks = 0;
484  return OK;
485 
486  /* external signalling not supported */
487  case SENSOR_POLLRATE_EXTERNAL:
488 
489  /* zero would be bad */
490  case 0:
491  return -EINVAL;
492 
493  /* set default/max polling rate */
494  case SENSOR_POLLRATE_MAX:
495  case SENSOR_POLLRATE_DEFAULT: {
496  /* do we need to start internal polling? */
497  bool want_start = (_measure_ticks == 0);
498 
499  /* set interval for next measurement to minimum legal value */
500  _measure_ticks = USEC2TICK(MS5611_CONVERSION_INTERVAL);
501 
502  /* if we need to start the poll state machine, do it */
503  if (want_start) {
504  start_cycle();
505  }
506 
507  return OK;
508  }
509 
510  /* adjust to a legal polling interval in Hz */
511  default: {
512  /* do we need to start internal polling? */
513  bool want_start = (_measure_ticks == 0);
514 
515  /* convert hz to tick interval via microseconds */
516  unsigned ticks = USEC2TICK(1000000 / arg);
517 
518  /* check against maximum rate */
519  if (ticks < USEC2TICK(MS5611_CONVERSION_INTERVAL)) {
520  return -EINVAL;
521  }
522 
523  /* update interval for next measurement */
524  _measure_ticks = ticks;
525 
526  /* if we need to start the poll state machine, do it */
527  if (want_start) {
528  start_cycle();
529  }
530 
531  return OK;
532  }
533  }
534  }
535 
536  case SENSORIOCGPOLLRATE:
537  if (_measure_ticks == 0) {
538  return SENSOR_POLLRATE_MANUAL;
539  }
540 
541  return (1000 / _measure_ticks);
542 
543  case SENSORIOCSQUEUEDEPTH: {
544  /* lower bound is mandatory, upper bound is a sanity check */
545  if ((arg < 1) || (arg > 100)) {
546  return -EINVAL;
547  }
548 
549  irqstate_t flags = px4_enter_critical_section();
550 
551  if (!_reports->resize(arg)) {
552  px4_leave_critical_section(flags);
553  return -ENOMEM;
554  }
555 
556  px4_leave_critical_section(flags);
557  return OK;
558  }
559 
560  case SENSORIOCRESET:
561  /*
562  * Since we are initialized, we do not need to do anything, since the
563  * PROM is correctly read and the part does not need to be configured.
564  */
565  return OK;
566 
567  default:
568  break;
569  }
570 
571  /* give it to the bus-specific superclass */
572  // return bus_ioctl(filp, cmd, arg);
573  return CDev::ioctl(filp, cmd, arg);
574 }

603 void
604 MS5611::cycle()
605 {
606  int ret;
607  unsigned dummy;
608 
609  /* collection phase? */
610  if (_collect_phase) {
611 
612  /* perform collection */
613  ret = collect();
614 
615  if (ret != OK) {
616  if (ret == -6) {
617  /*
618  * The ms5611 seems to regularly fail to respond to
619  * its address; this happens often enough that we'd rather not
620  * spam the console with a message for this.
621  */
622  } else {
623  //DEVICE_LOG("collection error %d", ret);
624  }
625 
626  /* issue a reset command to the sensor */
627  _interface->ioctl(IOCTL_RESET, dummy);
628  /* reset the collection state machine and try again - we need
629  * to wait 2.8 ms after issuing the sensor reset command
630  * according to the MS5611 datasheet
631  */
632  start_cycle(USEC2TICK(2800));
633  return;
634  }
635 
636  /* next phase is measurement */
637  _collect_phase = false;
638 
639  /*
640  * Is there a collect->measure gap?
641  * Don't inject one after temperature measurements, so we can keep
642  * doing pressure measurements at something close to the desired rate.
643  */
644  if ((_measure_phase != 0) &&
645  (_measure_ticks > USEC2TICK(MS5611_CONVERSION_INTERVAL))) {
646 
647  /* schedule a fresh cycle call when we are ready to measure again */
648  work_queue(HPWORK,
649  &_work,
650  (worker_t)&MS5611::cycle_trampoline,
651  this,
652  _measure_ticks - USEC2TICK(MS5611_CONVERSION_INTERVAL));
653 
654  return;
655  }
656  }
657 
658  /* measurement phase */
659  ret = measure();
660 
661  if (ret != OK) {
662  /* issue a reset command to the sensor */
663  _interface->ioctl(IOCTL_RESET, dummy);
664  /* reset the collection state machine and try again */
665  start_cycle();
666  return;
667  }
668 
669  /* next phase is collection */
670  _collect_phase = true;
671 
672  /* schedule a fresh cycle call when the measurement is done */
673  work_queue(HPWORK,
674  &_work,
675  (worker_t)&MS5611::cycle_trampoline,
676  this,
677  USEC2TICK(MS5611_CONVERSION_INTERVAL));
678 }

706 int
707 MS5611::collect()
708 {
709  int ret;
710  uint32_t raw;
711 
712  perf_begin(_sample_perf);
713 
714  struct baro_report report;
715  /* this should be fairly close to the end of the conversion, so the best approximation of the time */
716  report.timestamp = hrt_absolute_time();
717  report.error_count = perf_event_count(_comms_errors);
718 
719  /* read the most recent measurement - read offset/size are hardcoded in the interface */
720  ret = _interface->read(0, (void *)&raw, 0);
721 
722  if (ret < 0) {
723  perf_count(_comms_errors);
724  perf_end(_sample_perf);
725  return ret;
726  }
727 
728  /* handle a measurement */
729  if (_measure_phase == 0) {
730 
731  /* temperature offset (in ADC units) */
732  int32_t dT = (int32_t)raw - ((int32_t)_prom.c5_reference_temp << 8);
733 
734  /* absolute temperature in centidegrees - note intermediate value is outside 32-bit range */
735  _TEMP = 2000 + (int32_t)(((int64_t)dT * _prom.c6_temp_coeff_temp) >> 23);
736 
737  /* base sensor scale/offset values */
738  if (_device_type == MS5611_DEVICE) {
739 
740  /* Perform MS5611 Caculation */
741 
742  _OFF = ((int64_t)_prom.c2_pressure_offset << 16) + (((int64_t)_prom.c4_temp_coeff_pres_offset * dT) >> 7);
743  _SENS = ((int64_t)_prom.c1_pressure_sens << 15) + (((int64_t)_prom.c3_temp_coeff_pres_sens * dT) >> 8);
744 
745  /* MS5611 temperature compensation */
746 
747  if (_TEMP < 2000) {
748 
749  int32_t T2 = POW2(dT) >> 31;
750 
751  int64_t f = POW2((int64_t)_TEMP - 2000);
752  int64_t OFF2 = 5 * f >> 1;
753  int64_t SENS2 = 5 * f >> 2;
754 
755  if (_TEMP < -1500) {
756 
757  int64_t f2 = POW2(_TEMP + 1500);
758  OFF2 += 7 * f2;
759  SENS2 += 11 * f2 >> 1;
760  }
761 
762  _TEMP -= T2;
763  _OFF -= OFF2;
764  _SENS -= SENS2;
765  }
766 
767  } else if (_device_type == MS5607_DEVICE) {
768 
769  /* Perform MS5607 Caculation */
770 
771  _OFF = ((int64_t)_prom.c2_pressure_offset << 17) + (((int64_t)_prom.c4_temp_coeff_pres_offset * dT) >> 6);
772  _SENS = ((int64_t)_prom.c1_pressure_sens << 16) + (((int64_t)_prom.c3_temp_coeff_pres_sens * dT) >> 7);
773 
774  /* MS5607 temperature compensation */
775 
776  if (_TEMP < 2000) {
777 
778  int32_t T2 = POW2(dT) >> 31;
779 
780  int64_t f = POW2((int64_t)_TEMP - 2000);
781  int64_t OFF2 = 61 * f >> 4;
782  int64_t SENS2 = 2 * f;
783 
784  if (_TEMP < -1500) {
785  int64_t f2 = POW2(_TEMP + 1500);
786  OFF2 += 15 * f2;
787  SENS2 += 8 * f2;
788  }
789 
790  _TEMP -= T2;
791  _OFF -= OFF2;
792  _SENS -= SENS2;
793  }
794  }
795 
796  } else {
797 
798  /* pressure calculation, result in Pa */
799  int32_t P = (((raw * _SENS) >> 21) - _OFF) >> 15;
800  _P = P * 0.01f;
801  _T = _TEMP * 0.01f;
802 
803  /* generate a new report */
804  report.temperature = _TEMP / 100.0f;
805  report.pressure = P / 100.0f; /* convert to millibar */
806 
807  /* return device ID */
808  report.device_id = _device_id.devid;
809 
810  /* publish it */
811  if (!(_pub_blocked) && _baro_topic != nullptr) {
812  /* publish it */
813  orb_publish(ORB_ID(sensor_baro), _baro_topic, &report);
814  }
815 
816  _reports->force(&report);
817 
818  /* notify anyone waiting for data */
819  poll_notify(POLLIN);
820  }
821 
822  /* update the measurement state machine */
823  INCREMENT(_measure_phase, MS5611_MEASUREMENT_RATIO + 1);
824 
825  perf_end(_sample_perf);
826 
827  return OK;
828 }

889 /**
890  * MS5611 crc4 cribbed from the datasheet
891  */
892 bool
893 crc4(uint16_t *n_prom)
894 {
895  int16_t cnt;
896  uint16_t n_rem;
897  uint16_t crc_read;
898  uint8_t n_bit;
899 
900  n_rem = 0x00;
901 
902  /* save the read crc */
903  crc_read = n_prom[7];
904 
905  /* remove CRC byte */
906  n_prom[7] = (0xFF00 & (n_prom[7]));
907 
908  for (cnt = 0; cnt < 16; cnt++) {
909  /* uneven bytes */
910  if (cnt & 1) {
911  n_rem ^= (uint8_t)((n_prom[cnt >> 1]) & 0x00FF);
912 
913  } else {
914  n_rem ^= (uint8_t)(n_prom[cnt >> 1] >> 8);
915  }
916 
917  for (n_bit = 8; n_bit > 0; n_bit--) {
918  if (n_rem & 0x8000) {
919  n_rem = (n_rem << 1) ^ 0x3000;
920 
921  } else {
922  n_rem = (n_rem << 1);
923  }
924  }
925  }
926 
927  /* final 4 bit remainder is CRC value */
928  n_rem = (0x000F & (n_rem >> 12));
929  n_prom[7] = crc_read;
930 
931  /* return true if CRCs match */
932  return (0x000F & crc_read) == (n_rem ^ 0x00);
933 }

1088  }
