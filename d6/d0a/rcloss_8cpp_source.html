89 void
90 RCLoss::set_rcl_item()
91 {
92  struct position_setpoint_triplet_s *pos_sp_triplet = _navigator->get_position_setpoint_triplet();
93 
94  pos_sp_triplet->previous = pos_sp_triplet->current;
95  _navigator->set_can_loiter_at_sp(false);
96 
97  switch (_rcl_state) {
98  case RCL_STATE_LOITER: {
99  _mission_item.lat = _navigator->get_global_position()->lat;
100  _mission_item.lon = _navigator->get_global_position()->lon;
101  _mission_item.altitude = _navigator->get_global_position()->alt;
102  _mission_item.altitude_is_relative = false;
103  _mission_item.yaw = NAN;
104  _mission_item.loiter_radius = _navigator->get_loiter_radius();
105  _mission_item.nav_cmd = NAV_CMD_LOITER_TIME_LIMIT;
106  _mission_item.acceptance_radius = _navigator->get_acceptance_radius();
107  _mission_item.time_inside = _param_loitertime.get() < 0.0f ? 0.0f : _param_loitertime.get();
108  _mission_item.autocontinue = true;
109  _mission_item.origin = ORIGIN_ONBOARD;
110 
111  _navigator->set_can_loiter_at_sp(true);
112  break;
113  }
114 
115  case RCL_STATE_TERMINATE: {
116  /* Request flight termination from the commander */
117  _navigator->get_mission_result()->flight_termination = true;
118  _navigator->set_mission_result_updated();
119  warnx("rc not recovered: request flight termination");
120  pos_sp_triplet->previous.valid = false;
121  pos_sp_triplet->current.valid = false;
122  pos_sp_triplet->next.valid = false;
123  break;
124  }
125 
126  default:
127  break;
128  }
129 
130  reset_mission_item_reached();
131 
132  /* convert mission item to current position setpoint and make it valid */
133  mission_apply_limitation(_mission_item);
134  mission_item_to_position_setpoint(_mission_item, &pos_sp_triplet->current);
135  pos_sp_triplet->next.valid = false;
136 
137  _navigator->set_position_setpoint_triplet_updated();
138 }

140 void
141 RCLoss::advance_rcl()
142 {
143  switch (_rcl_state) {
144  case RCL_STATE_NONE:
145  if (_param_loitertime.get() > 0.0f) {
146  warnx("RC loss, OBC mode, loiter");
147  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "rc loss, loitering");
148  _rcl_state = RCL_STATE_LOITER;
149 
150  } else {
151  warnx("RC loss, OBC mode, slip loiter, terminate");
152  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "rc loss, terminating");
153  _rcl_state = RCL_STATE_TERMINATE;
154  _navigator->get_mission_result()->stay_in_failsafe = true;
155  _navigator->set_mission_result_updated();
156  reset_mission_item_reached();
157  }
158 
159  break;
160 
161  case RCL_STATE_LOITER:
162  _rcl_state = RCL_STATE_TERMINATE;
163  warnx("time is up, no RC regain, terminating");
164  mavlink_log_critical(_navigator->get_mavlink_log_pub(), "RC not regained, terminating");
165  _navigator->get_mission_result()->stay_in_failsafe = true;
166  _navigator->set_mission_result_updated();
167  reset_mission_item_reached();
168  break;
169 
170  case RCL_STATE_TERMINATE:
171  warnx("rcl end");
172  _rcl_state = RCL_STATE_END;
173  break;
174 
175  default:
176  break;
177  }
178 }
