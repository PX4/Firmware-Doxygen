52 /* SPI bus defining tools
53  *
54  * For new boards we use a board_config.h to define all the SPI functionality
55  * A board provides SPI bus definitions and a set of buses that should be
56  * enumerated as well as chip selects that will be iterateable
57  *
58  * We will use these in macros place of the uint32_t enumeration to select a
59  * specific SPI device on given SPI1 bus.
60  *
61  * These macros will define BUS:DEV For clarity and indexing
62  *
63  * The board config then defines:
64  * 1) PX4_SPI_BUS_xxx Ids -the buses as a 1 based PX4_SPI_BUS_xxx as n+1 aping to SPI n
65  * where n is {1-highest SPI supported by SoC}
66  * 2) PX4_SPIDEV_yyyy handles - PX4_SPIDEV_xxxxx handles using the macros below.
67  * 3) PX4_xxxx_BUS_CS_GPIO - a set of chip selects that are indexed by the handles. and of set to 0 are
68  * ignored.
69  * 4) PX4_xxxx_BUS_FIRST_CS and PX4_xxxxx_BUS_LAST_CS as PX4_SPIDEV_lll for the first CS and
70  * PX4_SPIDEV_hhhh as the last CS
71  *
72  * Example:
73  *
74  * The PX4_SPI_BUS_xxx
75  * #define PX4_SPI_BUS_SENSORS 1
76  * #define PX4_SPI_BUS_MEMORY 2
77  *
78  * the PX4_SPIDEV_yyyy
79  * #define PX4_SPIDEV_ICM_20689 PX4_MK_SPI_SEL(PX4_SPI_BUS_SENSORS,0)
80  * #define PX4_SPIDEV_ICM_20602 PX4_MK_SPI_SEL(PX4_SPI_BUS_SENSORS,1)
81  * #define PX4_SPIDEV_BMI055_GYRO PX4_MK_SPI_SEL(PX4_SPI_BUS_SENSORS,2)
82  *
83  * The PX4_xxxx_BUS_CS_GPIO
84  * #define PX4_SENSOR_BUS_CS_GPIO {GPIO_SPI_CS_ICM20689, GPIO_SPI_CS_ICM20602, GPIO_SPI_CS_BMI055_GYR,...
85  *
86  * The PX4_xxxx_BUS_FIRST_CS and PX4_xxxxx_BUS_LAST_CS
87  * #define PX4_SENSORS_BUS_FIRST_CS PX4_SPIDEV_ICM_20689
88  * #define PX4_SENSORS_BUS_LAST_CS PX4_SPIDEV_BMI055_ACCEL
89  *
90  *
91  */
92 #define PX4_SPIDEV_ID(type, index) ((((type) & 0xffff) << 16) | ((index) & 0xffff))

325 /* UUID
326  *
327  * Define the types used for board UUID, MFG UID and PX4 GUID
328  *
329  * A type suitable for holding the byte format of the UUID
330  *
331  * The original PX4 stm32 (legacy) based implementation **displayed** the
332  * UUID as: ABCD EFGH IJKL
333  * Where:
334  * A was bit 31 and D was bit 0
335  * E was bit 63 and H was bit 32
336  * I was bit 95 and L was bit 64
337  *
338  * Since the string was used by some manufactures to identify the units
339  * it must be preserved. DEPRECATED - This will be removed in PX4 Release
340  * 1.7.0.
341  *
342  * For new targets moving forward we will use an 18 byte globally unique
343  * PX4 GUID in the form:
344  *
345  * <ARCH MSD><ARCH LSD>[MNOP] IJKL EFGH ABCD
346  *
347  * Where <ARCH MSD><ARCH LSD> are a monotonic ordinal number assigned by
348  * PX4 to a chip architecture (PX4_SOC_ARCH_ID). The 2 bytes are used to
349  * create a globally unique ID when prepended to a padded CPU UUID.
350  *
351  * In the case where the MFG UUID is shorter than 16 bytes it will be
352  * padded with 0's starting at offset [2] until
353  * PX4_GUID_BYTE_LENGTH-PX4_CPU_UUID_BYTE_LENGTH -1
354  *
355  * I.E. For the STM32
356  * offset:0 1 2 3 4 5 6 - 17
357  * <ARCH MSD><ARCH LSD>[0][0][0][0]<MSD CPU UUID>...<LSD CPU UUID>
358  *
359  * I.E. For the Kinetis
360  * offset:0 1 2 - 17
361  * <ARCH MSD><ARCH LSD><MSD CPU UUID>...<LSD CPU UUID>
362  */
363 
364 /* Define the PX4 Globally unique ID (GUID) length and format size */
365 #define PX4_GUID_BYTE_LENGTH 18
366 #define PX4_GUID_FORMAT_SIZE ((2*PX4_GUID_BYTE_LENGTH)+1)
367
