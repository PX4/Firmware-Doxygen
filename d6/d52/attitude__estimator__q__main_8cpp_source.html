255 void AttitudeEstimatorQ::task_main()
256 {
257 
258 #ifdef __PX4_POSIX
259  perf_counter_t _perf_accel(perf_alloc_once(PC_ELAPSED, "sim_accel_delay"));
260  perf_counter_t _perf_mpu(perf_alloc_once(PC_ELAPSED, "sim_mpu_delay"));
261  perf_counter_t _perf_mag(perf_alloc_once(PC_ELAPSED, "sim_mag_delay"));
262 #endif
263 
264  _sensors_sub = orb_subscribe(ORB_ID(sensor_combined));
265  _vision_sub = orb_subscribe(ORB_ID(vehicle_vision_attitude));
266  _mocap_sub = orb_subscribe(ORB_ID(att_pos_mocap));
267  _params_sub = orb_subscribe(ORB_ID(parameter_update));
268  _global_pos_sub = orb_subscribe(ORB_ID(vehicle_global_position));
269  _magnetometer_sub = orb_subscribe(ORB_ID(vehicle_magnetometer));
270 
271  update_parameters(true);
272 
273  hrt_abstime last_time = 0;
274 
275  px4_pollfd_struct_t fds[1] = {};
276  fds[0].fd = _sensors_sub;
277  fds[0].events = POLLIN;
278 
279  while (!_task_should_exit) {
280  int ret = px4_poll(fds, 1, 1000);
281 
282  if (ret < 0) {
283  // Poll error, sleep and try again
284  usleep(10000);
285  PX4_WARN("POLL ERROR");
286  continue;
287 
288  } else if (ret == 0) {
289  // Poll timeout, do nothing
290  PX4_WARN("POLL TIMEOUT");
291  continue;
292  }
293 
294  update_parameters(false);
295 
296  // Update sensors
297  sensor_combined_s sensors;
298 
299  if (orb_copy(ORB_ID(sensor_combined), _sensors_sub, &sensors) == PX4_OK) {
300  // Feed validator with recent sensor data
301 
302  if (sensors.timestamp > 0) {
303  _gyro(0) = sensors.gyro_rad[0];
304  _gyro(1) = sensors.gyro_rad[1];
305  _gyro(2) = sensors.gyro_rad[2];
306  }
307 
308  if (sensors.accelerometer_timestamp_relative != sensor_combined_s::RELATIVE_TIMESTAMP_INVALID) {
309  _accel(0) = sensors.accelerometer_m_s2[0];
310  _accel(1) = sensors.accelerometer_m_s2[1];
311  _accel(2) = sensors.accelerometer_m_s2[2];
312 
313  if (_accel.length() < 0.01f) {
314  PX4_ERR("degenerate accel!");
315  continue;
316  }
317  }
318 
319  _data_good = true;
320  }
321 
322  // Update magnetometer
323  bool magnetometer_updated = false;
324  orb_check(_magnetometer_sub, &magnetometer_updated);
325 
326  if (magnetometer_updated) {
327  vehicle_magnetometer_s magnetometer;
328 
329  if (orb_copy(ORB_ID(vehicle_magnetometer), _magnetometer_sub, &magnetometer) == PX4_OK) {
330  _mag(0) = magnetometer.magnetometer_ga[0];
331  _mag(1) = magnetometer.magnetometer_ga[1];
332  _mag(2) = magnetometer.magnetometer_ga[2];
333 
334  if (_mag.length() < 0.01f) {
335  PX4_ERR("degenerate mag!");
336  continue;
337  }
338  }
339 
340  }
341 
342  // Update vision and motion capture heading
343  bool vision_updated = false;
344  orb_check(_vision_sub, &vision_updated);
345 
346  if (vision_updated) {
347  vehicle_attitude_s vision;
348 
349  if (orb_copy(ORB_ID(vehicle_vision_attitude), _vision_sub, &vision) == PX4_OK) {
350  Quatf q(vision.q);
351 
352  Dcmf Rvis = Quatf(vision.q);
353  Vector3f v(1.0f, 0.0f, 0.4f);
354 
355  // Rvis is Rwr (robot respect to world) while v is respect to world.
356  // Hence Rvis must be transposed having (Rwr)' * Vw
357  // Rrw * Vw = vn. This way we have consistency
358  _vision_hdg = Rvis.transpose() * v;
359 
360  // vision external heading usage (ATT_EXT_HDG_M 1)
361  if (_ext_hdg_mode == 1) {
362  // Check for timeouts on data
363  _ext_hdg_good = vision.timestamp > 0 && (hrt_elapsed_time(&vision.timestamp) < 500000);
364  }
365  }
366  }
367 
368  bool mocap_updated = false;
369  orb_check(_mocap_sub, &mocap_updated);
370 
371  if (mocap_updated) {
372  att_pos_mocap_s mocap;
373 
374  if (orb_copy(ORB_ID(att_pos_mocap), _mocap_sub, &mocap) == PX4_OK) {
375  Dcmf Rmoc = Quatf(mocap.q);
376  Vector3f v(1.0f, 0.0f, 0.4f);
377 
378  // Rmoc is Rwr (robot respect to world) while v is respect to world.
379  // Hence Rmoc must be transposed having (Rwr)' * Vw
380  // Rrw * Vw = vn. This way we have consistency
381  _mocap_hdg = Rmoc.transpose() * v;
382 
383  // Motion Capture external heading usage (ATT_EXT_HDG_M 2)
384  if (_ext_hdg_mode == 2) {
385  // Check for timeouts on data
386  _ext_hdg_good = mocap.timestamp > 0 && (hrt_elapsed_time(&mocap.timestamp) < 500000);
387  }
388  }
389  }
390 
391  bool gpos_updated = false;
392  orb_check(_global_pos_sub, &gpos_updated);
393 
394  if (gpos_updated) {
395  vehicle_global_position_s gpos;
396 
397  if (orb_copy(ORB_ID(vehicle_global_position), _global_pos_sub, &gpos) == PX4_OK) {
398  if (_mag_decl_auto && gpos.eph < 20.0f && hrt_elapsed_time(&gpos.timestamp) < 1000000) {
399  /* set magnetic declination automatically */
400  update_mag_declination(math::radians(get_mag_declination(gpos.lat, gpos.lon)));
401  }
402 
403  if (_acc_comp && gpos.timestamp != 0 && hrt_absolute_time() < gpos.timestamp + 20000 && gpos.eph < 5.0f && _inited) {
404  /* position data is actual */
405  Vector3f vel(gpos.vel_n, gpos.vel_e, gpos.vel_d);
406 
407  /* velocity updated */
408  if (_vel_prev_t != 0 && gpos.timestamp != _vel_prev_t) {
409  float vel_dt = (gpos.timestamp - _vel_prev_t) / 1e6f;
410  /* calculate acceleration in body frame */
411  _pos_acc = _q.conjugate_inversed((vel - _vel_prev) / vel_dt);
412  }
413 
414  _vel_prev_t = gpos.timestamp;
415  _vel_prev = vel;
416 
417  } else {
418  /* position data is outdated, reset acceleration */
419  _pos_acc.zero();
420  _vel_prev.zero();
421  _vel_prev_t = 0;
422  }
423  }
424  }
425 
426  /* time from previous iteration */
427  hrt_abstime now = hrt_absolute_time();
428  const float dt = math::constrain((now - last_time) / 1e6f, _dt_min, _dt_max);
429  last_time = now;
430 
431  if (update(dt)) {
432  vehicle_attitude_s att = {};
433  att.timestamp = sensors.timestamp;
434  att.rollspeed = _rates(0);
435  att.pitchspeed = _rates(1);
436  att.yawspeed = _rates(2);
437  _q.copyTo(att.q);
438 
439  /* the instance count is not used here */
440  int att_inst;
441  orb_publish_auto(ORB_ID(vehicle_attitude), &_att_pub, &att, &att_inst, ORB_PRIO_HIGH);
442  }
443  }
444 
445 #ifdef __PX4_POSIX
446  perf_end(_perf_accel);
447  perf_end(_perf_mpu);
448  perf_end(_perf_mag);
449 #endif
450 
451  orb_unsubscribe(_params_sub);
452  orb_unsubscribe(_sensors_sub);
453  orb_unsubscribe(_global_pos_sub);
454  orb_unsubscribe(_vision_sub);
455  orb_unsubscribe(_mocap_sub);
456  orb_unsubscribe(_magnetometer_sub);
457 }

509 bool AttitudeEstimatorQ::init()
510 {
511  // Rotation matrix can be easily constructed from acceleration and mag field vectors
512  // 'k' is Earth Z axis (Down) unit vector in body frame
513  Vector3f k = -_accel;
514  k.normalize();
515 
516  // 'i' is Earth X axis (North) unit vector in body frame, orthogonal with 'k'
517  Vector3f i = (_mag - k * (_mag * k));
518  i.normalize();
519 
520  // 'j' is Earth Y axis (East) unit vector in body frame, orthogonal with 'k' and 'i'
521  Vector3f j = k % i;
522 
523  // Fill rotation matrix
524  Dcmf R;
525  R.setRow(0, i);
526  R.setRow(1, j);
527  R.setRow(2, k);
528 
529  // Convert to quaternion
530  _q = R;
531 
532  // Compensate for magnetic declination
533  Quatf decl_rotation = Eulerf(0.0f, 0.0f, _mag_decl);
534  _q = _q * decl_rotation;
535 
536  _q.normalize();
537 
538  if (PX4_ISFINITE(_q(0)) && PX4_ISFINITE(_q(1)) &&
539  PX4_ISFINITE(_q(2)) && PX4_ISFINITE(_q(3)) &&
540  _q.length() > 0.95f && _q.length() < 1.05f) {
541  _inited = true;
542 
543  } else {
544  _inited = false;
545  }
546 
547  return _inited;
548 }

550 bool AttitudeEstimatorQ::update(float dt)
551 {
552  if (!_inited) {
553 
554  if (!_data_good) {
555  return false;
556  }
557 
558  return init();
559  }
560 
561  Quatf q_last = _q;
562 
563  // Angular rate of correction
564  Vector3f corr;
565  float spinRate = _gyro.length();
566 
567  if (_ext_hdg_mode > 0 && _ext_hdg_good) {
568  if (_ext_hdg_mode == 1) {
569  // Vision heading correction
570  // Project heading to global frame and extract XY component
571  Vector3f vision_hdg_earth = _q.conjugate(_vision_hdg);
572  float vision_hdg_err = wrap_pi(atan2f(vision_hdg_earth(1), vision_hdg_earth(0)));
573  // Project correction to body frame
574  corr += _q.conjugate_inversed(Vector3f(0.0f, 0.0f, -vision_hdg_err)) * _w_ext_hdg;
575  }
576 
577  if (_ext_hdg_mode == 2) {
578  // Mocap heading correction
579  // Project heading to global frame and extract XY component
580  Vector3f mocap_hdg_earth = _q.conjugate(_mocap_hdg);
581  float mocap_hdg_err = wrap_pi(atan2f(mocap_hdg_earth(1), mocap_hdg_earth(0)));
582  // Project correction to body frame
583  corr += _q.conjugate_inversed(Vector3f(0.0f, 0.0f, -mocap_hdg_err)) * _w_ext_hdg;
584  }
585  }
586 
587  if (_ext_hdg_mode == 0 || !_ext_hdg_good) {
588  // Magnetometer correction
589  // Project mag field vector to global frame and extract XY component
590  Vector3f mag_earth = _q.conjugate(_mag);
591  float mag_err = wrap_pi(atan2f(mag_earth(1), mag_earth(0)) - _mag_decl);
592  float gainMult = 1.0f;
593  const float fifty_dps = 0.873f;
594 
595  if (spinRate > fifty_dps) {
596  gainMult = math::min(spinRate / fifty_dps, 10.0f);
597  }
598 
599  // Project magnetometer correction to body frame
600  corr += _q.conjugate_inversed(Vector3f(0.0f, 0.0f, -mag_err)) * _w_mag * gainMult;
601  }
602 
603  _q.normalize();
604 
605 
606  // Accelerometer correction
607  // Project 'k' unit vector of earth frame to body frame
608  // Vector3f k = _q.conjugate_inversed(Vector3f(0.0f, 0.0f, 1.0f));
609  // Optimized version with dropped zeros
610  Vector3f k(
611  2.0f * (_q(1) * _q(3) - _q(0) * _q(2)),
612  2.0f * (_q(2) * _q(3) + _q(0) * _q(1)),
613  (_q(0) * _q(0) - _q(1) * _q(1) - _q(2) * _q(2) + _q(3) * _q(3))
614  );
615 
616  // If we are not using acceleration compensation based on GPS velocity,
617  // fuse accel data only if its norm is close to 1 g (reduces drift).
618  const float accel_norm_sq = _accel.norm_squared();
619  const float upper_accel_limit = CONSTANTS_ONE_G * 1.1f;
620  const float lower_accel_limit = CONSTANTS_ONE_G * 0.9f;
621 
622  if (_acc_comp || (accel_norm_sq > lower_accel_limit * lower_accel_limit &&
623  accel_norm_sq < upper_accel_limit * upper_accel_limit)) {
624  corr += (k % (_accel - _pos_acc).normalized()) * _w_accel;
625  }
626 
627  // Gyro bias estimation
628  if (spinRate < 0.175f) {
629  _gyro_bias += corr * (_w_gyro_bias * dt);
630 
631  for (int i = 0; i < 3; i++) {
632  _gyro_bias(i) = math::constrain(_gyro_bias(i), -_bias_max, _bias_max);
633  }
634 
635  }
636

652  _rates.zero();
653  _gyro_bias.zero();
654  return false;
655  }
