1 /**
2  * @file Quaternion.hpp
3  *
4  * All rotations and axis systems follow the right-hand rule.
5  * The Hamilton quaternion convention including its product definition is used.
6  *
7  * In order to rotate a vector in frame b (v_b) to frame n by a righthand
8  * rotation defined by the quaternion q_nb (from frame b to n)
9  * one can use the following operation:
10  * v_n = q_nb * [0;v_b] * q_nb^(-1)
11  *
12  * Just like DCM's: v_n = C_nb * v_b (vector rotation)
13  * M_n = C_nb * M_b * C_nb^(-1) (matrix rotation)
14  *
15  * or similarly the reverse operation
16  * v_b = q_nb^(-1) * [0;v_n] * q_nb
17  *
18  * where q_nb^(-1) represents the inverse of the quaternion q_nb^(-1) = q_bn
19  *
20  * The product z of two quaternions z = q2 * q1 represents an intrinsic rotation
21  * in the order of first q1 followed by q2.
22  * The first element of the quaternion
23  * represents the real part, thus, a quaternion representing a zero-rotation
24  * is defined as (1,0,0,0).
25  *
26  * @author James Goppert <james.goppert@gmail.com>
27  */

139  *
140  * This sets the instance to a quaternion representing coordinate transformation from
141  * frame 2 to frame 1 where the rotation from frame 1 to frame 2 is described

143  *

146  Quaternion(const Euler<Type> &euler) :
147  Vector<Type, 4>()
148  {
149  Quaternion &q = *this;
150  Type cosPhi_2 = Type(cos(euler.phi() / Type(2.0)));
151  Type cosTheta_2 = Type(cos(euler.theta() / Type(2.0)));
152  Type cosPsi_2 = Type(cos(euler.psi() / Type(2.0)));
153  Type sinPhi_2 = Type(sin(euler.phi() / Type(2.0)));

156  q(0) = cosPhi_2 * cosTheta_2 * cosPsi_2 +
157  sinPhi_2 * sinTheta_2 * sinPsi_2;
158  q(1) = sinPhi_2 * cosTheta_2 * cosPsi_2 -
159  cosPhi_2 * sinTheta_2 * sinPsi_2;
160  q(2) = cosPhi_2 * sinTheta_2 * cosPsi_2 +
161  sinPhi_2 * cosTheta_2 * sinPsi_2;
162  q(3) = cosPhi_2 * cosTheta_2 * sinPsi_2 -
163  sinPhi_2 * sinTheta_2 * cosPsi_2;

189  /**
190  * Quaternion from two vectors
191  * Generates shortest rotation from source to destination vector
192  *
193  * @param dst destination vector (no need to normalize)
194  * @param src source vector (no need to normalize)
195  * @param eps epsilon threshold which decides if a value is considered zero
196  */
197  Quaternion(const Vector3<Type> &src, const Vector3<Type> &dst, const Type eps = Type(1e-5)) :
198  Vector<Type, 4>()
199  {
200  Quaternion &q = *this;
201  Vector3<Type> cr = src.cross(dst);
202  float dt = src.dot(dst);
203  /* If the two vectors are parallel, cross product is zero
204  * If they point opposite, the dot product is negative */
205  if (cr.norm() < eps && dt < 0) {
206  cr = src.abs();
207  if (cr(0) < cr(1)) {
208  if (cr(0) < cr(2)) {
209  cr = Vector3<Type>(1, 0, 0);
210  } else {
211  cr = Vector3<Type>(0, 0, 1);
212  }
213  } else {
214  if (cr(1) < cr(2)) {
215  cr = Vector3<Type>(0, 1, 0);
216  } else {
217  cr = Vector3<Type>(0, 0, 1);
218  }
219  }
220  q(0) = Type(0);
221  cr = src.cross(cr);
222  } else {
223  /* Half-Way Quaternion Solution */
224  q(0) = src.dot(dst) + sqrt(src.norm_squared() * dst.norm_squared());
225  }
226  q(1) = cr(0);
227  q(2) = cr(1);
228  q(3) = cr(2);
229  q.normalize();
230  }
